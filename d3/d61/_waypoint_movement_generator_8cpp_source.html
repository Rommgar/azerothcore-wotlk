96 bool WaypointMovementGenerator<Creature>::StartMove(Creature* creature)
97 {
98  if (!i_path || i_path->empty())
99  return false;
100 
101  // Xinef: Dont allow dead creatures to move
102  if (!creature->IsAlive())
103  return false;
104 
105  if (Stopped())
106  return true;
107 
108  bool transportPath = creature->HasUnitMovementFlag(MOVEMENTFLAG_ONTRANSPORT) && creature->GetTransGUID();
109 
110  if (m_isArrivalDone)
111  {
112  // Xinef: not true... update this at every waypoint!
113  //if ((i_currentNode == i_path->size() - 1) && !repeating) // If that's our last waypoint
114  {
115  float x = i_path->at(i_currentNode)->x;
116  float y = i_path->at(i_currentNode)->y;
117  float z = i_path->at(i_currentNode)->z;
118  float o = creature->GetOrientation();
119 
120  if (!transportPath)
121  creature->SetHomePosition(x, y, z, o);
122  else
123  {
124  if (Transport* trans = (creature->GetTransport() ? creature->GetTransport()->ToMotionTransport() : nullptr))
125  {
126  o -= trans->GetOrientation();
127  creature->SetTransportHomePosition(x, y, z, o);
128  trans->CalculatePassengerPosition(x, y, z, &o);
129  creature->SetHomePosition(x, y, z, o);
130  }
131  else
132  transportPath = false;
133  // else if (vehicle) - this should never happen, vehicle offsets are const
134  }
135  }
136 
137  // Xinef: moved the upper IF here
138  if ((i_currentNode == i_path->size() - 1) && !repeating) // If that's our last waypoint
139  {
140  creature->GetMotionMaster()->Initialize();
141  return false;
142  }
143 
144  i_currentNode = (i_currentNode + 1) % i_path->size();
145  }
146 
147  // xinef: do not initialize motion if we got stunned in movementinform
148  if (creature->HasUnitState(UNIT_STATE_NOT_MOVE) || creature->IsMovementPreventedByCasting())
149  {
150  return true;
151  }
152 
153  WaypointData const* node = i_path->at(i_currentNode);
154 
155  m_isArrivalDone = false;
156 
157  creature->AddUnitState(UNIT_STATE_ROAMING_MOVE);
158 
159  Movement::Location formationDest(node->x, node->y, node->z, 0.0f);
160  Movement::MoveSplineInit init(creature);
161 
163  if (transportPath)
164  {
165  init.DisableTransportPathTransformations();
166  if (TransportBase* trans = creature->GetDirectTransport())
167  trans->CalculatePassengerPosition(formationDest.x, formationDest.y, formationDest.z, &formationDest.orientation);
168  }
169 
170  float z = node->z;
171  creature->UpdateAllowedPositionZ(node->x, node->y, z);
174  init.MoveTo(node->x, node->y, z, true, true);
175 
177  if (node->orientation && node->delay)
178  init.SetFacing(node->orientation);
179 
180  switch (node->move_type)
181  {
182  case WAYPOINT_MOVE_TYPE_LAND:
183  init.SetAnimation(Movement::ToGround);
184  break;
185  case WAYPOINT_MOVE_TYPE_TAKEOFF:
186  init.SetAnimation(Movement::ToFly);
187  break;
188  case WAYPOINT_MOVE_TYPE_RUN:
189  init.SetWalk(false);
190  break;
191  case WAYPOINT_MOVE_TYPE_WALK:
192  init.SetWalk(true);
193  break;
194  }
195 
196  init.Launch();
197 
198  //Call for creature group update
199  if (creature->GetFormation() && creature->GetFormation()->getLeader() == creature)
200  creature->GetFormation()->LeaderMoveTo(formationDest.x, formationDest.y, formationDest.z, node->move_type == WAYPOINT_MOVE_TYPE_RUN);
201 
202  return true;
203 }

206 {
207  // Waypoint movement can be switched on/off
208  // This is quite handy for escort quests and other stuff
209  if (creature->HasUnitState(UNIT_STATE_NOT_MOVE) || creature->IsMovementPreventedByCasting())
210  {
211  creature->StopMoving();
212  Stop(1000);
213  return true;
214  }
215 
216  // prevent a crash at empty waypoint path.
217  if (!i_path || i_path->empty())
218  return false;
219 
220  // Xinef: Dont allow dead creatures to move

222  return false;
223 
224  if (Stopped())
225  {
226  if (CanMove(diff))
227  return StartMove(creature);
228  }

234  {

236  // xinef: code to detect pre-empetively if we should start movement to next waypoint
237  // xinef: do not start pre-empetive movement if current node has delay or we are ending waypoint movement
238  //if (!finished && !i_path->at(i_currentNode)->delay && ((i_currentNode != i_path->size() - 1) || repeating))
239  // finished = (creature->movespline->_Spline().length(creature->movespline->_currentSplineIdx() + 1) - creature->movespline->timePassed()) < 200;
240 
241  if (finished)
242  {
243  OnArrived(creature);
244  return StartMove(creature);

343 bool FlightPathMovementGenerator::DoUpdate(Player* player, uint32 /*diff*/)
344 {
345  if (!player)
346  return false;
347 
348  // xinef: map was switched
349  if (_mapSwitch)
350  {
351  DoInitialize(player);
352  _mapSwitch = false;
353  return true;
354  }
355 
356  uint32 pointId = (uint32)player->movespline->currentPathIdx();
357  if (pointId > i_currentNode)
358  {
359  bool departureEvent = true;
360  do
361  {
362  if (i_currentNode >= i_path.size())
363  {
364  LOG_INFO("misc", "TAXI NODE WAS GREATER THAN PATH SIZE, %s, POINTID: %u, NODESIZE: %lu, CURRENT: %u",
365  player->GetGUID().ToString().c_str(), pointId, i_path.size(), i_currentNode);
366  player->CleanupAfterTaxiFlight();
367  return false;
368  }
369 
370  if (i_path[i_currentNode]->mapid != player->GetMapId())
371  {
372  LOG_INFO("misc", "Player on different map, curmap: %u, pointmap: %u, nodesize: %lu, currentnode: %u", player->GetMapId(), i_path[i_currentNode]->mapid, i_path.size(), i_currentNode);
373  player->CleanupAfterTaxiFlight();
374  return false;
375  }
376 
377  DoEventIfAny(player, i_path[i_currentNode], departureEvent);
378 
379  // xinef: erase any previous points
380  uint32 curSize = _pointsForPathSwitch.size();
381  while (!_pointsForPathSwitch.empty() && _pointsForPathSwitch.front() <= i_currentNode)
382  _pointsForPathSwitch.pop_front();
383 
384  // xinef: switch destination only once
385  if (curSize != _pointsForPathSwitch.size())
386  player->m_taxi.NextTaxiDestination();
387 
388  if (pointId == i_currentNode)
389  break;
390 
391  if (i_currentNode == _preloadTargetNode && player->GetMapId() == _endMapId)
392  PreloadEndGrid();
393  i_currentNode += (uint32)departureEvent;
394  departureEvent = !departureEvent;
395 
396  // xinef: map should be switched, do not rely on client packets QQ
397  if (i_currentNode + 1 < i_path.size() && i_path[i_currentNode + 1]->mapid != player->GetMapId())
398  {
399  ++i_currentNode;
400  _mapSwitch = true;
401  player->TeleportTo(i_path[i_currentNode]->mapid, i_path[i_currentNode]->x, i_path[i_currentNode]->y, i_path[i_currentNode]->z, player->GetOrientation(), TELE_TO_NOT_LEAVE_TAXI);
402  return true;
403  }
404 
405  // xinef: reached the end
406  if (i_currentNode >= i_path.size() - 1)
407  {
408  player->CleanupAfterTaxiFlight();
409  player->SetFallInformation(time(nullptr), player->GetPositionZ());
410  if (player->pvpInfo.IsHostile)
411  player->CastSpell(player, 2479, true);
412 
413  return false;
414  }
415  } while (true);
416  }
417 
418  return i_currentNode < (i_path.size() - 1);
419 }
