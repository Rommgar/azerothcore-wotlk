174  void UpdateAI(uint32 diff) override
175  {
176  if (!IsInRoom(me))
177  return;
178 
179  if (!UpdateVictim())
180  return;
181 
182  events.Update(diff);
183 
184  switch (events.ExecuteEvent())
185  {
186  case EVENT_DISRUPTION:
187  me->CastSpell(me, SPELL_SPELL_DISRUPTION, false);
188  events.RepeatEvent(10000);
189  break;
190  case EVENT_DECEPIT_FEVER:
191  me->CastSpell(me, RAID_MODE(SPELL_DECREPIT_FEVER_10, SPELL_DECREPIT_FEVER_25), false);
192  events.RepeatEvent(urand(22000, 25000));
193  break;
194  case EVENT_PLAGUE_CLOUD:
195  me->CastSpell(me, SPELL_PLAGUE_CLOUD, false);
196  break;
197  case EVENT_SWITCH_PHASE:
198  if (currentPhase == PHASE_SLOW_DANCE)
199  {
200  StartFightPhase(PHASE_FAST_DANCE);
201  }
202  else
203  {
204  StartFightPhase(PHASE_SLOW_DANCE);
205  Talk(EMOTE_DANCE_END); // avoid play the emote on aggro
206  }
207  break;
208  case EVENT_ERUPT_SECTION:
209  if (pInstance)
210  {
211  pInstance->SetData(DATA_HEIGAN_ERUPTION, currentSection);
212  if (currentSection == 3)
213  {
214  moveRight = false;
215  }
216  else if (currentSection == 0)
217  {
218  moveRight = true;
219  }
220  moveRight ? currentSection++ : currentSection--;
221  }
222  if (currentPhase == PHASE_SLOW_DANCE)
223  {
224  Talk(SAY_TAUNT);
225  }
226  events.RepeatEvent(currentPhase == PHASE_SLOW_DANCE ? 10000 : 4000);
227  break;
228  case EVENT_SAFETY_DANCE:
229  {
230  Map::PlayerList const& pList = me->GetMap()->GetPlayers();
231  for (const auto& itr : pList)
232  {
233  if (IsInRoom(itr.GetSource()) && !itr.GetSource()->IsAlive())
234  {
235  pInstance->SetData(DATA_DANCE_FAIL, 0);
236  pInstance->SetData(DATA_IMMORTAL_FAIL, 0);
237  return;
238  }
239  }
240  events.RepeatEvent(5000);
241  return;
242  }
243  }
244  DoMeleeAttackIfReady();
245  }
