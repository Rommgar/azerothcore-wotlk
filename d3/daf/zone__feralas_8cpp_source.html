248  _events.ScheduleEvent(EVENT_WANDERING_RANDOM, 8 * IN_MILLISECONDS);
249  break;

254  case EVENT_WANDERING_RANDOM:
255  me->SetHomePosition(me->GetPosition());
256  me->GetMotionMaster()->MoveRandom(15.f);
257  break;
258  case EVENT_FINAL_TALK:
259  if (Creature* robckbiter = ObjectAccessor::GetCreature(*me, _rockbiterGUID))
260  {
261  robckbiter->AI()->Talk(TALK_0, me);

263  me->DespawnOrUnsummon(10 * IN_MILLISECONDS);
264  break;
