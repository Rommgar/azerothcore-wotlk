109  void DoAction(int32 actionId) override
110  {
111  if (me->IsAlive())
112  {
113  Talk(SAY_OGRE_DEATH);
114  if (actionId == MAX_ADD_NUMBER)
115  me->SetLootMode(1);
116  }
117  else if (actionId == MAX_ADD_NUMBER)
118  {
119  me->loot.clear();
120  me->loot.FillLoot(me->GetCreatureTemplate()->lootid, LootTemplates_Creature, me->GetLootRecipient(), false, false, 1, me);
121  me->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
122  _JustDied();
123  }
124  }
125 
126  void EnterCombat(Unit* /*who*/) override
127  {
128  _EnterCombat();
129  Talk(SAY_AGGRO);
130 
131  events.ScheduleEvent(EVENT_ARCING_SMASH, 6000);
132  events.ScheduleEvent(EVENT_MIGHTY_BLOW, 20000);
133  events.ScheduleEvent(EVENT_WHIRLWIND, 30000);
134  events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
135  }
136 
137  void UpdateAI(uint32 diff) override
138  {
139  if (!UpdateVictim())
140  return;
141 
142  events.Update(diff);
143  if (me->HasUnitState(UNIT_STATE_CASTING))
144  return;
145 
146  switch (events.ExecuteEvent())
147  {
148  case EVENT_ARCING_SMASH:
149  me->CastSpell(me->GetVictim(), SPELL_ARCING_SMASH, false);
150  events.ScheduleEvent(EVENT_ARCING_SMASH, 10000);
151  break;
152  case EVENT_MIGHTY_BLOW:
153  me->CastSpell(me->GetVictim(), SPELL_MIGHTY_BLOW, false);
154  events.ScheduleEvent(EVENT_MIGHTY_BLOW, 16000);
155  break;
156  case EVENT_WHIRLWIND:
157  events.DelayEvents(15000);
158  me->CastSpell(me, SPELL_WHIRLWIND, false);
159  events.ScheduleEvent(EVENT_WHIRLWIND, 54000);
160  break;
161  case EVENT_CHECK_HEALTH:
162  if (me->HealthBelowPct(50))
163  {
164  Talk(SAY_ENRAGE);
165  me->CastSpell(me, SPELL_FLURRY, true);
166  events.ScheduleEvent(EVENT_CHARGING, 0);
167  events.ScheduleEvent(EVENT_ROAR, 3000);
168  break;
169  }
170  events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
171  break;
172  case EVENT_ROAR:
173  me->CastSpell(me, SPELL_ROAR, false);
174  events.ScheduleEvent(EVENT_ROAR, 40000);
175  break;
176  case EVENT_CHARGING:
177  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1))
178  me->CastSpell(target, SPELL_BERSERKER_C, false);
179  events.ScheduleEvent(EVENT_CHARGING, 35000);
180  break;
181  }
182 
183  DoMeleeAttackIfReady();
184  }
185  };
