35 void IpLocationStore::Load()
36 {
37  _ipLocationStore.clear();
38  LOG_INFO("server.loading", "Loading IP Location Database...");
39 
40  std::string databaseFilePath = sConfigMgr->GetOption<std::string>("IPLocationFile", "");
41  if (databaseFilePath.empty())
42  {
43  LOG_INFO("server.loading", " ");
44  return;
45  }
46 
47  // Check if file exists
48  std::ifstream databaseFile(databaseFilePath);
49  if (!databaseFile)
50  {
51  LOG_ERROR("server.loading", "IPLocation: No ip database file exists (%s).", databaseFilePath.c_str());
52  return;
53  }
54 
55  if (!databaseFile.is_open())
56  {
57  LOG_ERROR("server.loading", "IPLocation: Ip database file (%s) can not be opened.", databaseFilePath.c_str());
58  return;
59  }
60 
61  std::string ipFrom;
62  std::string ipTo;
63  std::string countryCode;
64  std::string countryName;
65 
66  while (databaseFile.good())
67  {
68  // Read lines
69  if (!std::getline(databaseFile, ipFrom, ','))
70  break;
71  if (!std::getline(databaseFile, ipTo, ','))
72  break;
73  if (!std::getline(databaseFile, countryCode, ','))
74  break;
75  if (!std::getline(databaseFile, countryName, '\n'))
76  break;
77 
78  // Remove new lines and return
79  countryName.erase(std::remove(countryName.begin(), countryName.end(), '\r'), countryName.end());
80  countryName.erase(std::remove(countryName.begin(), countryName.end(), '\n'), countryName.end());
81 
82  // Remove quotation marks
83  ipFrom.erase(std::remove(ipFrom.begin(), ipFrom.end(), '"'), ipFrom.end());
84  ipTo.erase(std::remove(ipTo.begin(), ipTo.end(), '"'), ipTo.end());
85  countryCode.erase(std::remove(countryCode.begin(), countryCode.end(), '"'), countryCode.end());
86  countryName.erase(std::remove(countryName.begin(), countryName.end(), '"'), countryName.end());
87 
88  // Convert country code to lowercase
89  std::transform(countryCode.begin(), countryCode.end(), countryCode.begin(), ::tolower);
90 
91  auto IpFrom = Acore::StringTo<uint32>(ipFrom);
92  auto IpTo = Acore::StringTo<uint32>(ipTo);
93 
94  if (!IpFrom || !IpTo)
95  continue;
96 
97  _ipLocationStore.emplace_back(*IpFrom, *IpTo, std::move(countryCode), std::move(countryName));
98  }
99 
100  std::sort(_ipLocationStore.begin(), _ipLocationStore.end(), [](IpLocationRecord const& a, IpLocationRecord const& b) { return a.IpFrom < b.IpFrom; });
101  ASSERT(std::is_sorted(_ipLocationStore.begin(), _ipLocationStore.end(), [](IpLocationRecord const& a, IpLocationRecord const& b) { return a.IpFrom < b.IpTo; }),
102  "Overlapping IP ranges detected in database file");
103 
104  databaseFile.close();
105 
106  LOG_INFO("server.loading", ">> Loaded %u ip location entries.", static_cast<uint32>(_ipLocationStore.size()));
107  LOG_INFO("server.loading", " ");
108 }

110 IpLocationRecord const* IpLocationStore::GetLocationRecord(std::string const& ipAddress) const
111 {
112  uint32 ip = Acore::Net::address_to_uint(Acore::Net::make_address_v4(ipAddress));
113  auto itr = std::upper_bound(_ipLocationStore.begin(), _ipLocationStore.end(), ip, [](uint32 ip, IpLocationRecord const& loc) { return ip < loc.IpTo; });
114  if (itr == _ipLocationStore.end())
115  {
116  return nullptr;
117  }
118 
119  if (ip < itr->IpFrom)
120  {
121  return nullptr;
122  }
123 
124  return &(*itr);
125 }
