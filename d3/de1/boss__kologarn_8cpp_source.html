390  void UpdateAI(uint32 diff) override
391  {
392  if (!UpdateVictim())
393  {
394  EnterEvadeMode();
395  return;
396  }
397 
398  events.Update(diff);
399  if (me->HasUnitState(UNIT_STATE_CASTING))
400  return;
401 
402  switch (events.ExecuteEvent())
403  {
404  case EVENT_PREPARE_BREATH:
405  breathReady = true;
406  break;
407  case EVENT_STONE_SHOUT:
408  if (_left || _right)
409  {
410  return;
411  }
412 
413  me->CastSpell(me->GetVictim(), SPELL_STONE_SHOUT, false);
414  events.ScheduleEvent(EVENT_STONE_SHOUT, 2000);
415  break;
416  case EVENT_SMASH:
417  if (_left && _right)
418  me->CastSpell(me->GetVictim(), SPELL_OVERHEAD_SMASH, false);
419  else if (_left || _right)
420  me->CastSpell(me->GetVictim(), SPELL_ONEARMED_OVERHEAD_SMASH, false);
421 
422  events.DelayEvents(1000);
423  events.ScheduleEvent(EVENT_SMASH, 14000);
424  return;
425  case EVENT_SWEEP:
426  if (_left)
427  {
428  if (Creature* cr = me->FindNearestCreature(NPC_SWEEP_TRIGGER, 300))
429  cr->CastSpell(cr, SPELL_ARM_SWEEP, false);
430 
431  if (urand(0, 1))
432  Talk(SAY_SHOCKWAVE);
433  }
434 
435  events.DelayEvents(1000);
436  events.ScheduleEvent(EVENT_SWEEP, 17000);
437  return;
438  case EVENT_GRIP:
439  events.ScheduleEvent(EVENT_GRIP, 25000);
440  if (!_right)
441  break;
442 
443  me->CastSpell(me, SPELL_STONE_GRIP, false);
444  Talk(SAY_GRAB_PLAYER);
445  Talk(EMOTE_STONE_GRIP);
446  return;
447  case EVENT_FOCUSED_EYEBEAM:
448  {
449  events.ScheduleEvent(EVENT_FOCUSED_EYEBEAM, 20000);
450 
451  if ((eyebeamTarget = SelectTarget(SELECT_TARGET_FARTHEST, 0, 0, true)))
452  {
453  me->CastSpell(eyebeamTarget, SPELL_FOCUSED_EYEBEAM_SUMMON, false);
454  }
455 
456  Talk(EMOTE_EYES);
457  return;
458  }
459  case EVENT_RESTORE_ARM_LEFT:
460  // shouldn't happen
461  AttachLeftArm();
462  return;
463  case EVENT_RESTORE_ARM_RIGHT:
464  // shouldn't happen
465  AttachRightArm();
466  return;
467  }
468 
469  //Make sure our attack is ready and we aren't currently casting before checking distance
470  if (me->isAttackReady() && me->GetVictim()) // victim could die by a spell (IMPORTANT!!!) and kologarn entered evade mode
471  {
472  //If we are within range melee the target
473  if (me->IsWithinMeleeRange(me->GetVictim()))
474  {
475  me->AttackerStateUpdate(me->GetVictim());
476  me->resetAttackTimer();
477  return;
478  }
479  else if (Unit* tgt = me->SelectNearbyTarget())
480  {
481  me->AttackerStateUpdate(tgt);
482  me->resetAttackTimer();
483  return;
484  }
485 
486  if (breathReady)
487  me->CastSpell(me->GetVictim(), SPELL_PETRIFYING_BREATH, false);
488  me->resetAttackTimer();
489  }
490  }
