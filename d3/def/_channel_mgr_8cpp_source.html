47 void ChannelMgr::LoadChannels()
48 {
49  uint32 oldMSTime = getMSTime();
50  uint32 count = 0;
51 
52  // 0 1 2 3 4 5
53  QueryResult result = CharacterDatabase.PQuery("SELECT channelId, name, team, announce, ownership, password FROM channels ORDER BY channelId ASC");
54  if (!result)
55  {
56  LOG_INFO("server.loading", ">> Loaded 0 channels. DB table `channels` is empty.");
57  return;
58  }
59 
60  std::vector<std::pair<std::string, uint32>> toDelete;
61  do
62  {
63  Field* fields = result->Fetch();
64 
65  uint32 channelDBId = fields[0].GetUInt32();
66  std::string channelName = fields[1].GetString();
67  TeamId team = TeamId(fields[2].GetUInt32());
68  std::string password = fields[5].GetString();
69 
70  std::wstring channelWName;
71  if (!Utf8toWStr(channelName, channelWName))
72  {
73  LOG_ERROR("server.loading", "Failed to load channel '%s' from database - invalid utf8 sequence? Deleted.", channelName.c_str());
74  toDelete.push_back({ channelName, team });
75  continue;
76  }
77 
78  ChannelMgr* mgr = forTeam(team);
79  if (!mgr)
80  {
81  LOG_ERROR("server.loading", "Failed to load custom chat channel '%s' from database - invalid team %u. Deleted.", channelName.c_str(), team);
82  toDelete.push_back({ channelName, team });
83  continue;
84  }
85 
86  Channel* newChannel = new Channel(channelName, 0, channelDBId, team, fields[3].GetUInt8(), fields[4].GetUInt8());
87  newChannel->SetPassword(password);
88  mgr->channels[channelWName] = newChannel;
89 
90  if (QueryResult banResult = CharacterDatabase.PQuery("SELECT playerGUID, banTime FROM channels_bans WHERE channelId = %u", channelDBId))
91  {
92  do
93  {
94  Field* banFields = banResult->Fetch();
95  if (!banFields)
96  break;
97  newChannel->AddBan(ObjectGuid::Create<HighGuid::Player>(banFields[0].GetUInt32()), banFields[1].GetUInt32());
98  } while (banResult->NextRow());
99  }
100 
101  if (channelDBId > ChannelMgr::_channelIdMax)
102  ChannelMgr::_channelIdMax = channelDBId;
103  ++count;
104  } while (result->NextRow());
105 
106  for (auto pair : toDelete)
107  {
108  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CHANNEL);
109  stmt->setString(0, pair.first);
110  stmt->setUInt32(1, pair.second);
111  CharacterDatabase.Execute(stmt);
112  }
113 
114  LOG_INFO("server.loading", ">> Loaded %u channels in %ums", count, GetMSTimeDiffToNow(oldMSTime));
115  LOG_INFO("server.loading", " ");
116 }

163 void ChannelMgr::LoadChannelRights()
164 {
165  uint32 oldMSTime = getMSTime();
166  channels_rights.clear();
167 
168  QueryResult result = CharacterDatabase.Query("SELECT name, flags, speakdelay, joinmessage, delaymessage, moderators FROM channels_rights");
169  if (!result)
170  {
171  LOG_INFO("server.loading", ">> Loaded 0 Channel Rights!");
172  LOG_INFO("server.loading", " ");
173  return;
174  }
175 
176  uint32 count = 0;
177  do
178  {
179  Field* fields = result->Fetch();
180  std::set<uint32> moderators;
181  const char* moderatorList = fields[5].GetCString();
182  if (moderatorList)
183  {
184  Tokenizer tokens(moderatorList, ' ');
185  for (Tokenizer::const_iterator i = tokens.begin(); i != tokens.end(); ++i)
186  {
187  uint64 moderator_acc = atol(*i);
188  if (moderator_acc && ((uint32)moderator_acc) == moderator_acc)
189  moderators.insert((uint32)moderator_acc);
190  }
191  }
192 
193  SetChannelRightsFor(fields[0].GetString(), fields[1].GetUInt32(), fields[2].GetUInt32(), fields[3].GetString(), fields[4].GetString(), moderators);
194 
195  ++count;
196  } while (result->NextRow());
197 
198  LOG_INFO("server.loading", ">> Loaded %d Channel Rights in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
199  LOG_INFO("server.loading", " ");
200 }
