82 bool WorldSocket::Update()
83 {
84  EncryptablePacket* queued;
85  MessageBuffer buffer(_sendBufferSize);
86  while (_bufferQueue.Dequeue(queued))
87  {
88  ServerPktHeader header(queued->size() + 2, queued->GetOpcode());
89  if (queued->NeedsEncryption())
90  _authCrypt.EncryptSend(header.header, header.getHeaderLength());
91 
92  if (buffer.GetRemainingSpace() < queued->size() + header.getHeaderLength())
93  {
94  QueuePacket(std::move(buffer));
95  buffer.Resize(_sendBufferSize);
96  }
97 
98  if (buffer.GetRemainingSpace() >= queued->size() + header.getHeaderLength())
99  {
100  buffer.Write(header.header, header.getHeaderLength());
101  if (!queued->empty())
102  buffer.Write(queued->contents(), queued->size());
103  }
104  else // single packet larger than 4096 bytes
105  {
106  MessageBuffer packetBuffer(queued->size() + header.getHeaderLength());
107  packetBuffer.Write(header.header, header.getHeaderLength());
108  if (!queued->empty())
109  packetBuffer.Write(queued->contents(), queued->size());
110 
111  QueuePacket(std::move(packetBuffer));
112  }
113 
114  delete queued;
115  }
116 
117  if (buffer.GetActiveSize() > 0)
118  QueuePacket(std::move(buffer));
119 
120  if (!BaseSocket::Update())
121  return false;
122 
123  _queryProcessor.ProcessReadyCallbacks();
124 
125  return true;
126 }

147 void WorldSocket::ReadHandler()
148 {
149  if (!IsOpen())
150  return;
151 
152  MessageBuffer& packet = GetReadBuffer();
153  while (packet.GetActiveSize() > 0)
154  {
155  if (_headerBuffer.GetRemainingSpace() > 0)
156  {
157  // need to receive the header
158  std::size_t readHeaderSize = std::min(packet.GetActiveSize(), _headerBuffer.GetRemainingSpace());
159  _headerBuffer.Write(packet.GetReadPointer(), readHeaderSize);
160  packet.ReadCompleted(readHeaderSize);
161 
162  if (_headerBuffer.GetRemainingSpace() > 0)
163  {
164  // Couldn't receive the whole header this time.
165  ASSERT(packet.GetActiveSize() == 0);
166  break;
167  }
168 
169  // We just received nice new header
170  if (!ReadHeaderHandler())
171  {
172  CloseSocket();
173  return;
174  }
175  }
176 
177  // We have full read header, now check the data payload
178  if (_packetBuffer.GetRemainingSpace() > 0)
179  {
180  // need more data in the payload
181  std::size_t readDataSize = std::min(packet.GetActiveSize(), _packetBuffer.GetRemainingSpace());
182  _packetBuffer.Write(packet.GetReadPointer(), readDataSize);
183  packet.ReadCompleted(readDataSize);
184 
185  if (_packetBuffer.GetRemainingSpace() > 0)
186  {
187  // Couldn't receive the whole data this time.
188  ASSERT(packet.GetActiveSize() == 0);
189  break;
190  }
191  }
192 
193  // just received fresh new payload
194  ReadDataHandlerResult result = ReadDataHandler();
195  _headerBuffer.Reset();
196 
197  if (result != ReadDataHandlerResult::Ok)
198  {
199  if (result != ReadDataHandlerResult::WaitingForQuery)
200  {
201  CloseSocket();
202  }
203 
204  return;
205  }
206  }
207 
208  AsyncRead();
209 }

211 bool WorldSocket::ReadHeaderHandler()
212 {
213  ASSERT(_headerBuffer.GetActiveSize() == sizeof(ClientPktHeader));
214 
215  if (_authCrypt.IsInitialized())
216  {
217  _authCrypt.DecryptRecv(_headerBuffer.GetReadPointer(), sizeof(ClientPktHeader));
218  }
219 
220  ClientPktHeader* header = reinterpret_cast<ClientPktHeader*>(_headerBuffer.GetReadPointer());
221  EndianConvertReverse(header->size);
222  EndianConvert(header->cmd);
223 
224  if (!header->IsValidSize() || !header->IsValidOpcode())
225  {
226  LOG_ERROR("network", "WorldSocket::ReadHeaderHandler(): client %s sent malformed packet (size: %hu, cmd: %u)",
227  GetRemoteIpAddress().to_string().c_str(), header->size, header->cmd);
228 
229  return false;
230  }
231 
232  header->size -= sizeof(header->cmd);
233  _packetBuffer.Resize(header->size);
234 
235  return true;
236 }

305 WorldSocket::ReadDataHandlerResult WorldSocket::ReadDataHandler()
306 {
307  ClientPktHeader* header = reinterpret_cast<ClientPktHeader*>(_headerBuffer.GetReadPointer());
308  OpcodeClient opcode = static_cast<OpcodeClient>(header->cmd);
309 
310  WorldPacket packet(opcode, std::move(_packetBuffer));
311  WorldPacket* packetToQueue;
312 
313  if (sPacketLog->CanLogPacket())
314  sPacketLog->LogPacket(packet, CLIENT_TO_SERVER, GetRemoteIpAddress(), GetRemotePort());
315 
316  std::unique_lock<std::mutex> sessionGuard(_worldSessionLock, std::defer_lock);
317 
318  switch (opcode)
319  {
320  case CMSG_PING:
321  {
322  LogOpcodeText(opcode, sessionGuard);
323  try
324  {
325  return HandlePing(packet) ? ReadDataHandlerResult::Ok : ReadDataHandlerResult::Error;
326  }
327  catch (ByteBufferException const&)
328  {
329  }
330  LOG_ERROR("network", "WorldSocket::ReadDataHandler(): client %s sent malformed CMSG_PING", GetRemoteIpAddress().to_string().c_str());
331  return ReadDataHandlerResult::Error;
332  }
333  case CMSG_AUTH_SESSION:
334  {
335  LogOpcodeText(opcode, sessionGuard);
336  if (_authed)
337  {
338  // locking just to safely log offending user is probably overkill but we are disconnecting him anyway
339  if (sessionGuard.try_lock())
340  LOG_ERROR("network", "WorldSocket::ProcessIncoming: received duplicate CMSG_AUTH_SESSION from %s", _worldSession->GetPlayerInfo().c_str());
341  return ReadDataHandlerResult::Error;
342  }
343 
344  try
345  {
346  HandleAuthSession(packet);
347  return ReadDataHandlerResult::WaitingForQuery;
348  }
349  catch (ByteBufferException const&) { }
350 
351  LOG_ERROR("network", "WorldSocket::ReadDataHandler(): client %s sent malformed CMSG_AUTH_SESSION", GetRemoteIpAddress().to_string().c_str());
352  return ReadDataHandlerResult::Error;
353  }
354  case CMSG_KEEP_ALIVE: // todo: handle this packet in the same way of CMSG_TIME_SYNC_RESP
355  sessionGuard.lock();
356  LogOpcodeText(opcode, sessionGuard);
357  if (_worldSession)
358  _worldSession->ResetTimeOutTime(true);
359  return ReadDataHandlerResult::Ok;
360  case CMSG_TIME_SYNC_RESP:
361  packetToQueue = new WorldPacket(std::move(packet), std::chrono::steady_clock::now());
362  break;
363  default:
364  packetToQueue = new WorldPacket(std::move(packet));
365  break;
366  }
367 
368  sessionGuard.lock();
369 
370  LogOpcodeText(opcode, sessionGuard);
371 
372  if (!_worldSession)
373  {
374  LOG_ERROR("network.opcode", "ProcessIncoming: Client not authed opcode = %u", uint32(opcode));
375  delete packetToQueue;
376  return ReadDataHandlerResult::Error;
377  }
378 
379  OpcodeHandler const* handler = opcodeTable[opcode];
380  if (!handler)
381  {
382  LOG_ERROR("network.opcode", "No defined handler for opcode %s sent by %s", GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet.GetOpcode())).c_str(), _worldSession->GetPlayerInfo().c_str());
383  delete packetToQueue;
384  return ReadDataHandlerResult::Error;
385  }
386 
387  // Our Idle timer will reset on any non PING opcodes on login screen, allowing us to catch people idling.
388  _worldSession->ResetTimeOutTime(false);
389 
390  // Copy the packet to the heap before enqueuing
391  _worldSession->QueuePacket(packetToQueue);
392 
393  return ReadDataHandlerResult::Ok;
394 }

452 void WorldSocket::HandleAuthSessionCallback(std::shared_ptr<AuthSession> authSession, PreparedQueryResult result)
453 {
454  // Stop if the account is not found
455  if (!result)
456  {
457  // We can not log here, as we do not know the account. Thus, no accountId.
458  SendAuthResponseError(AUTH_UNKNOWN_ACCOUNT);
459  LOG_ERROR("network", "WorldSocket::HandleAuthSession: Sent Auth Response (unknown account).");
460  DelayedCloseSocket();
461  return;
462  }
463 
464  AccountInfo account(result->Fetch());
465 
466  // For hook purposes, we get Remoteaddress at this point.
467  std::string address = sConfigMgr->GetOption<bool>("AllowLoggingIPAddressesInDatabase", true, true) ? GetRemoteIpAddress().to_string() : "0.0.0.0";
468 
469  LoginDatabasePreparedStatement* stmt = nullptr;
470 
471  // As we don't know if attempted login process by ip works, we update last_attempt_ip right away
472  stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LAST_ATTEMPT_IP);
473  stmt->setString(0, address);
474  stmt->setString(1, authSession->Account);
475  LoginDatabase.Execute(stmt);
476  // This also allows to check for possible "hack" attempts on account
477 
478  // even if auth credentials are bad, try using the session key we have - client cannot read auth response error without it
479  _authCrypt.Init(account.SessionKey);
480 
481  // First reject the connection if packet contains invalid data or realm state doesn't allow logging in
482  if (sWorld->IsClosed())
483  {
484  SendAuthResponseError(AUTH_REJECT);
485  LOG_ERROR("network", "WorldSocket::HandleAuthSession: World closed, denying client (%s).", GetRemoteIpAddress().to_string().c_str());
486  DelayedCloseSocket();
487  return;
488  }
489 
490  if (authSession->RealmID != realm.Id.Realm)
491  {
492  SendAuthResponseError(REALM_LIST_REALM_NOT_FOUND);
493  LOG_ERROR("network", "WorldSocket::HandleAuthSession: Client %s requested connecting with realm id %u but this realm has id %u set in config.",
494  GetRemoteIpAddress().to_string().c_str(), authSession->RealmID, realm.Id.Realm);
495  DelayedCloseSocket();
496  return;
497  }
498 
499  // Must be done before WorldSession is created
500  bool wardenActive = sWorld->getBoolConfig(CONFIG_WARDEN_ENABLED);
501  if (wardenActive && account.OS != "Win" && account.OS != "OSX")
502  {
503  SendAuthResponseError(AUTH_REJECT);
504  LOG_ERROR("network", "WorldSocket::HandleAuthSession: Client %s attempted to log in using invalid client OS (%s).", address.c_str(), account.OS.c_str());
505  DelayedCloseSocket();
506  return;
507  }
508 
509  // Check that Key and account name are the same on client and server
510  uint8 t[4] = { 0x00,0x00,0x00,0x00 };
511 
512  Acore::Crypto::SHA1 sha;
513  sha.UpdateData(authSession->Account);
514  sha.UpdateData(t);
515  sha.UpdateData(authSession->LocalChallenge);
516  sha.UpdateData(_authSeed);
517  sha.UpdateData(account.SessionKey);
518  sha.Finalize();
519 
520  if (sha.GetDigest() != authSession->Digest)
521  {
522  SendAuthResponseError(AUTH_FAILED);
523  LOG_ERROR("network", "WorldSocket::HandleAuthSession: Authentication failed for account: %u ('%s') address: %s", account.Id, authSession->Account.c_str(), address.c_str());
524  DelayedCloseSocket();
525  return;
526  }
527 
528  if (IpLocationRecord const* location = sIPLocation->GetLocationRecord(address))
529  _ipCountry = location->CountryCode;
530 
532  if (account.IsLockedToIP)
533  {
534  if (account.LastIP != address)
535  {
536  SendAuthResponseError(AUTH_FAILED);
537  LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account IP differs. Original IP: %s, new IP: %s).", account.LastIP.c_str(), address.c_str());
538  // We could log on hook only instead of an additional db log, however action logger is config based. Better keep DB logging as well
539  sScriptMgr->OnFailedAccountLogin(account.Id);
540  DelayedCloseSocket();
541  return;
542  }
543  }
544  else if (!account.LockCountry.empty() && account.LockCountry != "00" && !_ipCountry.empty())
545  {
546  if (account.LockCountry != _ipCountry)
547  {
548  SendAuthResponseError(AUTH_FAILED);
549  LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account country differs. Original country: %s, new country: %s).", account.LockCountry.c_str(), _ipCountry.c_str());
550  // We could log on hook only instead of an additional db log, however action logger is config based. Better keep DB logging as well
551  sScriptMgr->OnFailedAccountLogin(account.Id);
552  DelayedCloseSocket();
553  return;
554  }
555  }
556 
558  if (account.MuteTime < 0)
559  {
560  account.MuteTime = time(nullptr) + llabs(account.MuteTime);
561 
562  auto* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_MUTE_TIME_LOGIN);
563  stmt->setInt64(0, account.MuteTime);
564  stmt->setUInt32(1, account.Id);
565  LoginDatabase.Execute(stmt);
566  }
567 
568  if (account.IsBanned)
569  {
570  SendAuthResponseError(AUTH_BANNED);
571  LOG_ERROR("network", "WorldSocket::HandleAuthSession: Sent Auth Response (Account banned).");
572  sScriptMgr->OnFailedAccountLogin(account.Id);
573  DelayedCloseSocket();
574  return;
575  }
576 
577  // Check locked state for server
578  AccountTypes allowedAccountType = sWorld->GetPlayerSecurityLimit();
579  LOG_DEBUG("network", "Allowed Level: %u Player Level %u", allowedAccountType, account.Security);
580  if (allowedAccountType > SEC_PLAYER && account.Security < allowedAccountType)
581  {
582  SendAuthResponseError(AUTH_UNAVAILABLE);
583  LOG_DEBUG("network", "WorldSocket::HandleAuthSession: User tries to login but his security level is not enough");
584  sScriptMgr->OnFailedAccountLogin(account.Id);
585  DelayedCloseSocket();
586  return;
587  }
588 
589  LOG_DEBUG("network", "WorldSocket::HandleAuthSession: Client '%s' authenticated successfully from %s.", authSession->Account.c_str(), address.c_str());
590 
591  // Update the last_ip in the database as it was successful for login
592  stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LAST_IP);
593  stmt->setString(0, address);
594  stmt->setString(1, authSession->Account);
595 
596  LoginDatabase.Execute(stmt);
597 
598  // At this point, we can safely hook a successful login
599  sScriptMgr->OnAccountLogin(account.Id);
600 
601  _authed = true;
602 
603  sScriptMgr->OnLastIpUpdate(account.Id, address);
604 
605  _worldSession = new WorldSession(account.Id, std::move(authSession->Account), shared_from_this(), account.Security,
606  account.Expansion, account.MuteTime, account.Locale, account.Recruiter, account.IsRectuiter, account.Security ? true : false, account.TotalTime);
607 
608  _worldSession->ReadAddonsInfo(authSession->AddonInfo);
609 
610  // Initialize Warden system only if it is enabled by config
611  if (wardenActive)
612  {
613  _worldSession->InitWarden(account.SessionKey, account.OS);
614  }
615 
616  sWorld->AddSession(_worldSession);
617 
618  AsyncRead();
619 }

629 bool WorldSocket::HandlePing(WorldPacket& recvPacket)
630 {
631  using namespace std::chrono;
632 
633  uint32 ping;
634  uint32 latency;
635 
636  // Get the ping packet content
637  recvPacket >> ping;
638  recvPacket >> latency;
639 
640  if (_LastPingTime == steady_clock::time_point())
641  {
642  _LastPingTime = steady_clock::now();
643  }
644  else
645  {
646  steady_clock::time_point now = steady_clock::now();
647  steady_clock::duration diff = now - _LastPingTime;
648 
649  _LastPingTime = now;
650 
651  if (diff < seconds(27))
652  {
653  ++_OverSpeedPings;
654 
655  uint32 maxAllowed = sWorld->getIntConfig(CONFIG_MAX_OVERSPEED_PINGS);
656 
657  if (maxAllowed && _OverSpeedPings > maxAllowed)
658  {
659  std::unique_lock<std::mutex> sessionGuard(_worldSessionLock);
660
