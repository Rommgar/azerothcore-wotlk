268 bool EquippedOk(Player* player, uint32 spellId)
269 {
270  SpellInfo const* spell = sSpellMgr->GetSpellInfo(spellId);
271  if (!spell)
272  return false;
273 
274  for (uint8 i = 0; i < 3; ++i)
275  {
276  uint32 reqSpell = spell->Effects[i].TriggerSpell;
277  if (!reqSpell)
278  continue;
279 
280  Item* item = nullptr;
281  for (uint8 j = EQUIPMENT_SLOT_START; j < EQUIPMENT_SLOT_END; ++j)
282  {
283  item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, j);
284  if (item && item->GetTemplate()->RequiredSpell == reqSpell)
285  {
286  //player has item equipped that require specialty. Not allow to unlearn, player has to unequip first
287  LOG_DEBUG("scripts.ai", "Player attempt to unlearn spell %u, but item %u is equipped.", reqSpell, item->GetEntry());
288  return false;
289  }
290  }
291  }
292  return true;
293 }
