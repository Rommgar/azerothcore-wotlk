309  void HandlePeriodic(AuraEffect const* aurEff)
310  {
311  PreventDefaultAction();
312  if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
313  {
314  int32 currentAmount = aurEff->GetAmount();
315  int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
316  if (newAmount != currentAmount)
317  {
318  if (aurEff->GetMiscValue() == STAT_STAMINA)
319  {
320  uint32 actStat = GetUnitOwner()->GetHealth();
321  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
322  GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
323  }
324  else
325  {
326  uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
327  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
328  GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
329  }
330  }
331  }
332  else
333  GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
334  }
