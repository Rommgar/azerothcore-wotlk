226  float* getLiquidHeightMap(adt_liquid_header* h)
227  {
228  if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
229  return nullptr;
230  if (h->offsData2b)
231  return (float*)((uint8*)this + 8 + h->offsData2b);
232  return nullptr;
233  }
234 
235  uint8* getLiquidLightMap(adt_liquid_header* h)
236  {
237  if (h->formatFlags & ADT_LIQUID_HEADER_FULL_LIGHT)
238  return nullptr;
239  if (h->offsData2b)
240  {
241  if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
242  return (uint8*)((uint8*)this + 8 + h->offsData2b);
243  return (uint8*)((uint8*)this + 8 + h->offsData2b + (h->width + 1) * (h->height + 1) * 4);
244  }
245  return nullptr;
246  }
247 
248  uint32* getLiquidFullLightMap(adt_liquid_header* h)
249  {
250  if (!(h->formatFlags & ADT_LIQUID_HEADER_FULL_LIGHT))
251  return nullptr;
252  if (h->offsData2b)
253  {
254  if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
255  return (uint32*)((uint8*)this + 8 + h->offsData2b);
256  return (uint32*)((uint8*)this + 8 + h->offsData2b + (h->width + 1) * (h->height + 1) * 4);
257  }
258  return nullptr;
259  }
