135  void UpdateAI(uint32 diff) override
136  {
137  if (!UpdateVictim() && !me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
138  return;
139 
140  events.Update(diff);
141 
142  if (me->HasUnitState(UNIT_STATE_CASTING))
143  return;
144 
145  switch(events.ExecuteEvent())
146  {
147  case 0:
148  break;
149  case EVENT_EMERGE:
150  me->SetVisible(true);
151  me->CastSpell(me, SPELL_EMERGE_0, false);
152  events.RescheduleEvent(EVENT_ATTACK, 2000);
153  break;
154  case EVENT_SUMMON_TOTEMS:
155  for (uint8 i = 0; i < 3; ++i)
156  DoSummon(NPC_TOTEM, TotemPos[i], 10 * 60 * 1000, TEMPSUMMON_TIMED_DESPAWN);
157  break;
158  case EVENT_INVOKER_SAY_1:
159  if (Player* plr = ObjectAccessor::GetPlayer(*me, InvokerGUID))
160  {
161  plr->Say("The Ice Stone has melted!", LANG_UNIVERSAL);
162  plr->CastSpell(plr, SPELL_MAKE_BONFIRE, true);
163  }
164  events.RescheduleEvent(EVENT_INVOKER_SAY_2, 2000);
165  break;
166  case EVENT_INVOKER_SAY_2:
167  if (Player* plr = ObjectAccessor::GetPlayer(*me, InvokerGUID))
168  plr->Say("Ahune, your strength grows no more!", LANG_UNIVERSAL);
169  events.RescheduleEvent(EVENT_INVOKER_SAY_3, 2000);
170  break;
171  case EVENT_INVOKER_SAY_3:
172  if (Player* plr = ObjectAccessor::GetPlayer(*me, InvokerGUID))
173  plr->Say("Your frozen reign will not come to pass!", LANG_UNIVERSAL);
174  break;
175  case EVENT_ATTACK:
176  events.Reset();
177  if (Player* plr = ObjectAccessor::GetPlayer(*me, InvokerGUID))
178  AttackStart(plr);
179  me->SetInCombatWithZone();
180  if (!me->IsInCombat())
181  {
182  EnterEvadeMode();
183  return;
184  }
185  else
186  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
187  break;
188  case EVENT_TOTEMS_ATTACK:
189  for (uint8 i = 0; i < 3; ++i)
190  if (Creature* bunny = me->FindNearestCreature(NPC_TOTEM_BUNNY_1 + i, 150.0f, true))
191  bunny->CastSpell(me, SPELL_TOTEM_BEAM, false);
192  events.RescheduleEvent(EVENT_SUBMERGE, 10000);
193  break;
194  case EVENT_SUBMERGE:
195  me->TextEmote(TEXT_RETREAT, nullptr, true);
196  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
197  me->CastSpell(me, SPELL_SUBMERGE_0, true);
198  me->CastSpell(me, SPELL_SELF_STUN, true);
199  if (Creature* c = DoSummon(NPC_FROZEN_CORE, *me, 24000, TEMPSUMMON_TIMED_DESPAWN))
200  {
201  c->SetHealth(me->GetHealth());
202  }
203  events.Reset();
204  events.RescheduleEvent(EVENT_COMBAT_EMERGE, 25000);
205  events.RescheduleEvent(EVENT_EMERGE_WARNING, 20000);
206  break;
207  case EVENT_EMERGE_WARNING:
208  me->TextEmote(TEXT_RESURFACE, nullptr, true);
209  break;
210  case EVENT_COMBAT_EMERGE:
211  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
212  me->RemoveAura(SPELL_SELF_STUN);
213  me->CastSpell(me, SPELL_EMERGE_0, false);
214  // me->CastSpell(me, SPELL_AHUNE_RESURFACES, true); // done in SummonedCreatureDespawn
