169  void UpdateAI(uint32 diff) override
170  {
171  if (!UpdateVictim())
172  return;
173 
174  events.Update(diff);
175 
176  if (me->HasUnitState(UNIT_STATE_CASTING))
177  return;
178 
179  while (uint32 eventId = events.ExecuteEvent())
180  {
181  switch (eventId)
182  {
183  case EVENT_SHADOW_WORD_PAIN:
184  DoCastVictim(SPELL_SHADOW_WORD_PAIN, true);
185  events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(5000, 7000), 0, PHASE_ONE);
186  break;
187  case EVENT_GOUGE:
188  DoCastVictim(SPELL_GOUGE, true);
189  break;
190  case EVENT_SUMMON_PROWLERS:
191  if (_summonCountA < MAX_PROWLERS_PER_SIDE)
192  {
193  if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideAGUID[urand(0, 4)]))
194  {
195  trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
196  ++_summonCountA;
197  }
198  }
199  if (_summonCountB < MAX_PROWLERS_PER_SIDE)
200  {
201  if (Unit* trigger = ObjectAccessor::GetUnit(*me, _triggersSideBGUID[urand(0, 4)]))
202  {
203  trigger->CastSpell(trigger, SPELL_SUMMON_PROWLER);
204  ++_summonCountB;
205  }
206  }
207  events.ScheduleEvent(EVENT_SUMMON_PROWLERS, 6000, 0, PHASE_ALL);
208  break;
209  case EVENT_MARK_OF_ARLOKK:
210  {
211  Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, urand(1, 3), 0.0f, false, -SPELL_MARK_OF_ARLOKK);
212  if (!target)
213  target = me->GetVictim();
214  if (target)
