148 static int fill_uchar(uchar *a,uint size,const char *str, size_t len)
149 {
150  uint i= 0;
151  const char *s, *b, *e=str+len;
152 
153  for (s=str ; s < e ; i++)
154  {
155  for ( ; (s < e) && strchr(" \t\r\n",s[0]); s++) ;
156  b=s;
157  for ( ; (s < e) && !strchr(" \t\r\n",s[0]); s++) ;
158  if (s == b || i > size)
159  break;
160  a[i]= (uchar) strtoul(b,NULL,16);
161  }
162  return 0;
163 }

216 static int cs_value(MY_XML_PARSER *st,const char *attr, size_t len)
217 {
218  struct my_cs_file_info *i= (struct my_cs_file_info *)st->user_data;
219  struct my_cs_file_section_st *s;
220  int state= (int)((s=cs_file_sec(st->attr, strlen(st->attr))) ? s->state :
221  0);
222 
223  switch (state) {
224  case _CS_ID:
225  i->cs.number= strtol(attr,(char**)NULL,10);
226  break;
227  case _CS_BINARY_ID:
228  i->cs.binary_number= strtol(attr,(char**)NULL,10);
229  break;
230  case _CS_PRIMARY_ID:
231  i->cs.primary_number= strtol(attr,(char**)NULL,10);
232  break;
233  case _CS_COLNAME:
234  i->cs.name=mstr(i->name,attr,len,MY_CS_NAME_SIZE-1);
235  break;
236  case _CS_CSNAME:
237  i->cs.csname=mstr(i->csname,attr,len,MY_CS_NAME_SIZE-1);
238  break;
239  case _CS_CSDESCRIPT:
240  i->cs.comment=mstr(i->comment,attr,len,MY_CS_CSDESCR_SIZE-1);
241  break;
242  case _CS_FLAG:
243  if (!strncmp("primary",attr,len))
244  i->cs.state|= MY_CS_PRIMARY;
245  else if (!strncmp("binary",attr,len))
246  i->cs.state|= MY_CS_BINSORT;
247  else if (!strncmp("compiled",attr,len))
248  i->cs.state|= MY_CS_COMPILED;
249  break;
250  case _CS_UPPERMAP:
251  fill_uchar(i->to_upper,MY_CS_TO_UPPER_TABLE_SIZE,attr,len);
252  i->cs.to_upper=i->to_upper;
253  break;
254  case _CS_LOWERMAP:
255  fill_uchar(i->to_lower,MY_CS_TO_LOWER_TABLE_SIZE,attr,len);
256  i->cs.to_lower=i->to_lower;
257  break;
258  case _CS_UNIMAP:
259  fill_uint16(i->tab_to_uni,MY_CS_TO_UNI_TABLE_SIZE,attr,len);
260  i->cs.tab_to_uni=i->tab_to_uni;
261  break;
262  case _CS_COLLMAP:
263  fill_uchar(i->sort_order,MY_CS_SORT_ORDER_TABLE_SIZE,attr,len);
264  i->cs.sort_order=i->sort_order;
265  break;
266  case _CS_CTYPEMAP:
267  fill_uchar(i->ctype,MY_CS_CTYPE_TABLE_SIZE,attr,len);
268  i->cs.ctype=i->ctype;
269  break;
270  case _CS_RESET:
271  case _CS_DIFF1:
272  case _CS_DIFF2:
273  case _CS_DIFF3:
274  case _CS_IDENTICAL:
275  {
276  /*
277  Convert collation description from
278  Locale Data Markup Language (LDML)
279  into ICU Collation Customization expression.
280  */
281  char arg[16];
282  const char *cmd[]= {"&","<","<<","<<<","="};
283  i->cs.tailoring= i->tailoring;
284  mstr(arg,attr,len,sizeof(arg)-1);
285  if (i->tailoring_length + 20 < sizeof(i->tailoring))
286  {
287  char *dst= i->tailoring_length + i->tailoring;
288  i->tailoring_length+= sprintf(dst," %s %s",cmd[state-_CS_RESET],arg);
289  }
290  }
291  }
292  return MY_XML_OK;
293 }

318 uint
319 my_string_repertoire(CHARSET_INFO *cs, const char *str, ulong length)
320 {
321  const char *strend= str + length;
322  if (cs->mbminlen == 1)
323  {
324  for ( ; str < strend; str++)
325  {
326  if (((uchar) *str) > 0x7F)
327  return MY_REPERTOIRE_UNICODE30;
328  }
329  }
330  else
331  {
332  my_wc_t wc;
333  int chlen;
334  for (;
335  (chlen= cs->cset->mb_wc(cs, &wc, (uchar*) str, (uchar*) strend)) > 0;
336  str+= chlen)
337  {
338  if (wc > 0x7F)
339  return MY_REPERTOIRE_UNICODE30;
340  }
341  }
342  return MY_REPERTOIRE_ASCII;
343 }

356 /*
357  Detect whether a character set is ASCII compatible.
358 
359  Returns TRUE for:
360 
361  - all 8bit character sets whose Unicode mapping of 0x7B is '{'
362  (ignores swe7 which maps 0x7B to "LATIN LETTER A WITH DIAERESIS")
363 
364  - all multi-byte character sets having mbminlen == 1
365  (ignores ucs2 whose mbminlen is 2)
366 
367  TODO:
368 
369  When merging to 5.2, this function should be changed
370  to check a new flag MY_CS_NONASCII,
371 
372  return (cs->flag & MY_CS_NONASCII) ? 0 : 1;
373 
374  This flag was previously added into 5.2 under terms
375  of WL#3759 "Optimize identifier conversion in client-server protocol"
376  especially to mark character sets not compatible with ASCII.
377 
378  We won't backport this flag to 5.0 or 5.1.
379  This function is Ok for 5.0 and 5.1, because we're not going
380  to introduce new tricky character sets between 5.0 and 5.2.
381 */
382 my_bool
383 my_charset_is_ascii_based(CHARSET_INFO *cs)
384 {
385  return
386  (cs->mbmaxlen == 1 && cs->tab_to_uni && cs->tab_to_uni['{'] == '{') ||
387  (cs->mbminlen == 1 && cs->mbmaxlen > 1);
388 }
