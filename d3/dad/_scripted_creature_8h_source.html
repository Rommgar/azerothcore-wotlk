221  /* This is called for bosses whenever an encounter is happening.
222  * - Arguments:
223  * - Position has to be passed as a constant pointer (&Position)
224  * - Axis is the X or Y axis that is used to decide the position threshold
225  * - Above decides if the boss position should be above the passed position
226  * or below.
227  * Example:
228  * Hodir is in room until his Y position is below the Door position:
229  * IsInRoom(doorPosition, AXIS_Y, false);
230  */
231  bool IsInRoom(const Position* pos, Axis axis, bool above)
232  {
233  if (!pos)
234  {
235  return true;
236  }
237 
238  switch (axis)
239  {
240  case Axis::AXIS_X:
241  if ((!above && me->GetPositionX() < pos->GetPositionX()) || me->GetPositionX() > pos->GetPositionX())
242  {
243  EnterEvadeMode();
244  return false;
245  }
246  break;
247  case Axis::AXIS_Y:
248  if ((!above && me->GetPositionY() < pos->GetPositionY()) || me->GetPositionY() > pos->GetPositionY())
249  {
250  EnterEvadeMode();
251  return false;
252  }
253 
254  break;
255  }
256 
257  return true;
258  }
