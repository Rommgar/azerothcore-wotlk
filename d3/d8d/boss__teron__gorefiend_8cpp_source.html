159  break;

162  Talk(SAY_BLOSSOM);
163 
164  me->CastSpell(me, SPELL_SUMMON_DOOM_BLOSSOM, false);

166  break;

171  events.ScheduleEvent(EVENT_SPELL_CRUSHING_SHADOWS, 15000);
172  break;
173  case EVENT_SPELL_SHADOW_OF_DEATH:
174  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, ShadowOfDeathSelector()))
175  me->CastSpell(target, SPELL_SHADOW_OF_DEATH, false);
176  events.ScheduleEvent(EVENT_SPELL_SHADOW_OF_DEATH, 30000);
177  break;
178  }
179 
180  DoMeleeAttackIfReady();
