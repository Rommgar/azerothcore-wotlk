493  alwaysAssertM(n < 0xFFFFFFFF, "This implementation does not support arrays with more than 2^32 elements, although the size in memory may be larger.");
494  if (num == n) {
495  return;
496  }

499  num = n;

506  // Once allocated, always maintain MIN_ELEMENTS elements or 32 bytes, whichever is higher.
507  const size_t minSize = G3D::max(MIN_ELEMENTS, (size_t)(MIN_BYTES / sizeof(T)));
508 
509  if ((MIN_ELEMENTS == 0) && (MIN_BYTES == 0) && (n == 0) && shrinkIfNecessary) {
510  // Deallocate the array completely
511  numAllocated = 0;
512  m_memoryManager->free(data);
513  data = NULL;
514  return;
515  }

518  // Grow the underlying array
519 
520  if (numAllocated == 0) {
521  // First allocation; grow to exactly the size requested to avoid wasting space.
522  numAllocated = n;
523  debugAssert(oldNum == 0);
524  realloc(oldNum);
525  } else {
526 
527  if (num < minSize) {
528  // Grow to at least the minimum size
529  numAllocated = minSize;
530 
531  } else {
532 
533  // Increase the underlying size of the array. Grow aggressively
534  // up to 64k, less aggressively up to 400k, and then grow relatively
535  // slowly (1.5x per resize) to avoid excessive space consumption.
536  //
537  // These numbers are tweaked according to performance tests.
538 
539  double growFactor = 3.0f;
540 
541  size_t oldSizeBytes = numAllocated * sizeof(T);
542  if (oldSizeBytes > 10000000) {
543  // Conserve memory more tightly above 10 MB
544  growFactor = 1.2f;
545  } else if (oldSizeBytes > 400000) {
546  // Avoid bloat above 400k
547  growFactor = 1.5f;
548  } else if (oldSizeBytes > 64000) {
549  // This is what std:: uses at all times
550  growFactor = 2.0f;
551  }
552 
553  numAllocated = (num - numAllocated) + (size_t)(numAllocated * growFactor);
554 
555  if (numAllocated < minSize) {
556  numAllocated = minSize;
557  }
558  }
559 
560  realloc(oldNum);
561  }
562 
563  } else if ((num <= numAllocated / 3) && shrinkIfNecessary && (num > minSize)) {
564  // Shrink the underlying array
565 
566  // Only copy over old elements that still remain after resizing
567  // (destructors were called for others if we're shrinking)
568  realloc(min(num, oldNum));

1249  void medianPartition(
1250  Array<T>& ltMedian,
1251  Array<T>& eqMedian,
1252  Array<T>& gtMedian,
1253  Array<T>& tempArray,
1254  const Comparator& comparator) const {
1255 
1256  ltMedian.fastClear();
1257  eqMedian.fastClear();
1258  gtMedian.fastClear();
1259 
1260  // Handle trivial cases first
1261  switch (size()) {
1262  case 0:
1263  // Array is empty; no parition is possible
1264  return;
1265 
1266  case 1:
1267  // One element
1268  eqMedian.append(first());
1269  return;
1270 
1271  case 2:
1272  {
1273  // Two element array; median is the smaller
1274  int c = comparator(first(), last());
1275 
1276  switch (c) {
1277  case -1:
1278  // first was bigger
1279  eqMedian.append(last());
1280  gtMedian.append(first());
1281  break;
1282 
1283  case 0:
1284  // Both equal to the median
1285  eqMedian.append(first(), last());
1286  break;
1287 
1288  case 1:
1289  // Last was bigger
1290  eqMedian.append(first());
1291  gtMedian.append(last());
1292  break;
1293  }
1294  }
1295  return;
1296  }
1297 
1298  // All other cases use a recursive randomized median
1299 
1300  // Number of values less than all in the current arrays
1301  int ltBoost = 0;
1302 
1303  // Number of values greater than all in the current arrays
1304  int gtBoost = 0;
1305 
1306  // For even length arrays, force the gt array to be one larger than the
1307  // lt array:
1308  // [1 2 3] size = 3, choose half = (s + 1) /2
1309  //
1310  int lowerHalfSize, upperHalfSize;
1311  if (isEven(size())) {
1312  lowerHalfSize = size() / 2;
1313  upperHalfSize = lowerHalfSize + 1;
1314  } else {
1315  lowerHalfSize = upperHalfSize = (size() + 1) / 2;
1316  }
1317  const T* xPtr = NULL;
1318 
1319  // Maintain pointers to the arrays; we'll switch these around during sorting
1320  // to avoid copies.
1321  const Array<T>* source = this;
1322  Array<T>* lt = &ltMedian;
1323  Array<T>* eq = &eqMedian;
1324  Array<T>* gt = &gtMedian;
1325  Array<T>* extra = &tempArray;
1326 
1327  while (true) {
1328  // Choose a random element -- choose the middle element; this is theoretically
1329  // suboptimal, but for loosly sorted array is actually the best strategy
1330 
1331  xPtr = &(source->middle());
1332  if (source->size() == 1) {
1333  // Done; there's only one element left
1334  break;
1335  }
1336  const T& x = *xPtr;
1337 
1338  // Note: partition (fast) clears the arrays for us
1339  source->partition(x, *lt, *eq, *gt, comparator);
1340 
1341  int L = lt->size() + ltBoost + eq->size();
1342  int U = gt->size() + gtBoost + eq->size();
1343  if ((L >= lowerHalfSize) &&
1344  (U >= upperHalfSize)) {
1345 
1346  // x must be the partition median
1347  break;
1348 
1349  } else if (L < lowerHalfSize) {
1350 
1351  // x must be smaller than the median. Recurse into the 'gt' array.
1352  ltBoost += lt->size() + eq->size();
1353 
1354  // The new gt array will be the old source array, unless
1355  // that was the this pointer (i.e., unless we are on the
1356  // first iteration)
1357  Array<T>* newGt = (source == this) ? extra : const_cast<Array<T>*>(source);
1358 
1359  // Now set up the gt array as the new source
1360  source = gt;
1361  gt = newGt;
1362 
1363  } else {
1364 
1365  // x must be bigger than the median. Recurse into the 'lt' array.
1366  gtBoost += gt->size() + eq->size();
1367 
1368  // The new lt array will be the old source array, unless
1369  // that was the this pointer (i.e., unless we are on the
1370  // first iteration)
1371  Array<T>* newLt = (source == this) ? extra : const_cast<Array<T>*>(source);
1372 
1373  // Now set up the lt array as the new source
1374  source = lt;
1375  lt = newLt;
1376  }
1377  }
1378 
1379  // Now that we know the median, make a copy of it (since we're about to destroy the array that it
1380  // points into).
1381  T median = *xPtr;
1382  xPtr = NULL;
1383 
1384  // Partition the original array (note that this fast clears for us)
1385  partition(median, ltMedian, eqMedian, gtMedian, comparator);
1386  }
