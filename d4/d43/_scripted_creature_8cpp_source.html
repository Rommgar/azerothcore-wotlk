293 SpellInfo const* ScriptedAI::SelectSpell(Unit* target, uint32 school, uint32 mechanic, SelectTargetType targets, uint32 powerCostMin, uint32 powerCostMax, float rangeMin, float rangeMax, SelectEffect effects)
294 {
295  //No target so we can't cast
296  if (!target)
297  return nullptr;
298 
299  //Silenced so we can't cast
300  if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SILENCED))
301  return nullptr;
302 
303  //Using the extended script system we first create a list of viable spells
304  SpellInfo const* apSpell[MAX_CREATURE_SPELLS];
305  memset(apSpell, 0, MAX_CREATURE_SPELLS * sizeof(SpellInfo*));
306 
307  uint32 spellCount = 0;
308 
309  SpellInfo const* tempSpell = nullptr;
310 
311  //Check if each spell is viable(set it to null if not)
312  for (uint32 i = 0; i < MAX_CREATURE_SPELLS; i++)
313  {
314  tempSpell = sSpellMgr->GetSpellInfo(me->m_spells[i]);
315 
316  //This spell doesn't exist
317  if (!tempSpell)
318  continue;
319 
320  // Targets and Effects checked first as most used restrictions
321  //Check the spell targets if specified
322  if (targets && !(SpellSummary[me->m_spells[i]].Targets & (1 << (targets - 1))))
323  continue;
324 
325  //Check the type of spell if we are looking for a specific spell type
326  if (effects && !(SpellSummary[me->m_spells[i]].Effects & (1 << (effects - 1))))
327  continue;
328 
329  //Check for school if specified
330  if (school && (tempSpell->SchoolMask & school) == 0)
331  continue;
332 
333  //Check for spell mechanic if specified
334  if (mechanic && tempSpell->Mechanic != mechanic)
335  continue;
336 
337  //Make sure that the spell uses the requested amount of power
338  if (powerCostMin && tempSpell->ManaCost < powerCostMin)
339  continue;
340 
341  if (powerCostMax && tempSpell->ManaCost > powerCostMax)
342  continue;
343 
344  //Continue if we don't have the mana to actually cast this spell
345  if (tempSpell->ManaCost > me->GetPower(Powers(tempSpell->PowerType)))
346  continue;
347 
348  //Check if the spell meets our range requirements
349  if (rangeMin && me->GetSpellMinRangeForTarget(target, tempSpell) < rangeMin)
350  continue;
351  if (rangeMax && me->GetSpellMaxRangeForTarget(target, tempSpell) > rangeMax)
352  continue;
353 
354  //Check if our target is in range
355  if (me->IsWithinDistInMap(target, float(me->GetSpellMinRangeForTarget(target, tempSpell))) || !me->IsWithinDistInMap(target, float(me->GetSpellMaxRangeForTarget(target, tempSpell))))
356  continue;
357 
358  //All good so lets add it to the spell list
359  apSpell[spellCount] = tempSpell;
360  ++spellCount;
361  }
362 
363  //We got our usable spells so now lets randomly pick one
364  if (!spellCount)
365  return nullptr;
366 
367  return apSpell[urand(0, spellCount - 1)];
368 }

588 bool BossAI::CheckBoundary(Unit* who)
589 {
590  if (!GetBoundary() || !who)
591  return true;
592 
593  for (BossBoundaryMap::const_iterator itr = GetBoundary()->begin(); itr != GetBoundary()->end(); ++itr)
594  {
595  switch (itr->first)
596  {
597  case BOUNDARY_N:
598  if (who->GetPositionX() > itr->second)
599  return false;
600  break;
601  case BOUNDARY_S:
602  if (who->GetPositionX() < itr->second)
603  return false;
604  break;
605  case BOUNDARY_E:
606  if (who->GetPositionY() < itr->second)
607  return false;
608  break;
609  case BOUNDARY_W:
610  if (who->GetPositionY() > itr->second)
611  return false;
612  break;
613  case BOUNDARY_NW:
614  if (who->GetPositionX() + who->GetPositionY() > itr->second)
615  return false;
616  break;
617  case BOUNDARY_SE:
618  if (who->GetPositionX() + who->GetPositionY() < itr->second)
619  return false;
620  break;
621  case BOUNDARY_NE:
622  if (who->GetPositionX() - who->GetPositionY() > itr->second)
623  return false;
624  break;
625  case BOUNDARY_SW:
626  if (who->GetPositionX() - who->GetPositionY() < itr->second)
627  return false;
628  break;
629  default:
630  break;
631  }
632  }
633 
634  return true;
635 }
