108 void Totem::UnSummon(uint32 msTime)
109 {
110  if (msTime)
111  {
112  m_Events.AddEvent(new ForcedUnsummonDelayEvent(*this), m_Events.CalculateTime(msTime));
113  return;
114  }
115 
116  CombatStop();
117  RemoveAurasDueToSpell(GetSpell(), GetGUID());
118 
119  Unit* owner = GetOwner();
120  // clear owner's totem slot
121  for (uint8 i = SUMMON_SLOT_TOTEM; i < MAX_TOTEM_SLOT; ++i)
122  {
123  if (owner->m_SummonSlot[i] == GetGUID())
124  {
125  owner->m_SummonSlot[i].Clear();
126  break;
127  }
128  }
129 
130  owner->RemoveAurasDueToSpell(GetSpell(), GetGUID());
131 
132  // Remove Sentry Totem Aura
133  if (GetEntry() == SENTRY_TOTEM_ENTRY)
134  owner->RemoveAurasDueToSpell(SENTRY_TOTEM_SPELLID);
135 
136  //remove aura all party members too
137  if (Player* player = owner->ToPlayer())
138  {
139  player->SendAutoRepeatCancel(this);
140 
141  if (SpellInfo const* spell = sSpellMgr->GetSpellInfo(GetUInt32Value(UNIT_CREATED_BY_SPELL)))
142  player->SendCooldownEvent(spell, 0, nullptr, false);
143 
144  if (Group* group = player->GetGroup())
145  {
146  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
147  {
148  Player* target = itr->GetSource();
149  if (target && target->IsInMap(player) && group->SameSubGroup(player, target))
150  target->RemoveAurasDueToSpell(GetSpell(), GetGUID());
151  }
152  }
153  }
154 
155  AddObjectToRemoveList();
156 }
