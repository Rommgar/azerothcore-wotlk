16 /* password checking routines */
17 /*****************************************************************************
18  The main idea is that no password are sent between client & server on
19  connection and that no password are saved in mysql in a decodable form.
20 
21  On connection a random string is generated and sent to the client.
22  The client generates a new string with a random generator inited with
23  the hash values from the password and the sent string.
24  This 'check' string is sent to the server where it is compared with
25  a string generated from the stored hash_value of the password and the
26  random string.
27 
28  The password is saved (in user.password) by using the PASSWORD() function in
29  mysql.
30 
31  This is .c file because it's used in libmysqlclient, which is entirely in C.
32  (we need it to be portable to a variety of systems).
33  Example:
34  update user set password=PASSWORD("hello") where user="test"
35  This saves a hashed number as a string in the password field.
36 
37  The new authentication is performed in following manner:
38 
39  SERVER: public_seed=create_random_string()
40  send(public_seed)
41 
42  CLIENT: recv(public_seed)
43  hash_stage1=sha1("password")
44  hash_stage2=sha1(hash_stage1)
45  reply=xor(hash_stage1, sha1(public_seed,hash_stage2)
46 
47  // this three steps are done in scramble()
48 
49  send(reply)
50 
51 
52  SERVER: recv(reply)
53  hash_stage1=xor(reply, sha1(public_seed,hash_stage2))
54  candidate_hash2=sha1(hash_stage1)
55  check(candidate_hash2==hash_stage2)
56 
57  // this three steps are done in check_scramble()
58 
59 *****************************************************************************/

108 /*
109  Generate binary hash from raw text string
110  Used for Pre-4.1 password handling
111  SYNOPSIS
112  hash_password()
113  result OUT store hash in this location
114  password IN plain text password to build hash
115  password_len IN password length (password may be not null-terminated)
116 */
117 
118 void hash_password(ulong *result, const char *password, uint password_len)
119 {
120  register ulong nr=1345345333L, add=7, nr2=0x12345671L;
121  ulong tmp;
122  const char *password_end= password + password_len;
123  for (; password < password_end; password++)
124  {
125  if (*password == ' ' || *password == '\t')
126  continue; /* skip space in password */
127  tmp= (ulong) (uchar) *password;
128  nr^= (((nr & 63)+add)*tmp)+ (nr << 8);
129  nr2+=(nr2 << 8) ^ nr;
130  add+=tmp;
131  }
132  result[0]=nr & (((ulong) 1L << 31) -1L); /* Don't use sign bit (str2int) */;
133  result[1]=nr2 & (((ulong) 1L << 31) -1L);
134 }

175  Used in pre 4.1 authentication phase.
176  SYNOPSIS
177  scramble_323()
178  to OUT Store scrambled message here. Buffer must be at least
179  SCRAMBLE_LENGTH_323+1 bytes long

185 void scramble_323(char *to, const char *message, const char *password)
186 {
187  struct rand_struct rand_st;
188  ulong hash_pass[2], hash_message[2];
189 
190  if (password && password[0])
191  {
192  char extra, *to_start=to;
193  const char *message_end= message + SCRAMBLE_LENGTH_323;
194  hash_password(hash_pass,password, (uint) strlen(password));
195  hash_password(hash_message, message, SCRAMBLE_LENGTH_323);
196  randominit(&rand_st,hash_pass[0] ^ hash_message[0],
197  hash_pass[1] ^ hash_message[1]);
198  for (; message < message_end; message++)
199  *to++= (char) (floor(my_rnd(&rand_st)*31)+64);
200  extra=(char) (floor(my_rnd(&rand_st)*31));
201  while (to_start != to)
202  *(to_start++)^=extra;
203  }
204  *to= 0;
205 }

208 /*
209  Check scrambled message
210  Used in pre 4.1 password handling
211  SYNOPSIS
212  check_scramble_323()
213  scrambled scrambled message to check.
214  message original random message which was used for scrambling; must
215  be exactly SCRAMBLED_LENGTH_323 bytes long and
216  NULL-terminated.
217  hash_pass password which should be used for scrambling
218  All params are IN.
219 
220  RETURN VALUE
221  0 - password correct
222  !0 - password invalid
223 */
224 
225 my_bool
226 check_scramble_323(const unsigned char *scrambled, const char *message,
227  ulong *hash_pass)
228 {
229  struct rand_struct rand_st;
230  ulong hash_message[2];
231  uchar buff[16],*to,extra; /* Big enough for check */
232  const uchar *pos;
233 
234  hash_password(hash_message, message, SCRAMBLE_LENGTH_323);
235  randominit(&rand_st,hash_pass[0] ^ hash_message[0],
236  hash_pass[1] ^ hash_message[1]);
237  to=buff;
238  DBUG_ASSERT(sizeof(buff) > SCRAMBLE_LENGTH_323);
239  for (pos=scrambled ; *pos && to < buff+sizeof(buff) ; pos++)
240  *to++=(char) (floor(my_rnd(&rand_st)*31)+64);
241  if (pos-scrambled != SCRAMBLE_LENGTH_323)
242  return 1;
243  extra=(char) (floor(my_rnd(&rand_st)*31));
244  to=buff;
245  while (*scrambled)
246  {
247  if (*scrambled++ != (uchar) (*to++ ^ extra))
248  return 1; /* Wrong password */
249  }
250  return 0;
251 }

378 /*
379  Encrypt/Decrypt function used for password encryption in authentication.
380  Simple XOR is used here but it is OK as we crypt random strings. Note,
381  that XOR(s1, XOR(s1, s2)) == s2, XOR(s1, s2) == XOR(s2, s1)
382  SYNOPSIS
383  my_crypt()
384  to OUT buffer to hold crypted string; must be at least len bytes
385  long; to and s1 (or s2) may be the same.
386  s1, s2 IN input strings (of equal length)
387  len IN length of s1 and s2

391 my_crypt(char *to, const uchar *s1, const uchar *s2, uint len)
392 {
393  const uint8 *s1_end= s1 + len;

399 /*
400  MySQL 4.1.1 password hashing: SHA conversion (see RFC 2289, 3174) twice
401  applied to the password string, and then produced octet sequence is
402  converted to hex string.
403  The result of this function is used as return value from PASSWORD() and
404  is stored in the database.
405  SYNOPSIS
406  my_make_scrambled_password()
407  buf OUT buffer of size 2*SHA1_HASH_SIZE + 2 to store hex string
408  password IN password string
409  pass_len IN length of password string
410 */
411 
412 void my_make_scrambled_password(char *to, const char *password,
413  size_t pass_len)
414 {
415  SHA1_CONTEXT sha1_context;
416  uint8 hash_stage2[SHA1_HASH_SIZE];
417 
418  mysql_sha1_reset(&sha1_context);
419  /* stage 1: hash password */
420  mysql_sha1_input(&sha1_context, (uint8 *) password, (uint) pass_len);
421  mysql_sha1_result(&sha1_context, (uint8 *) to);
422  /* stage 2: hash stage1 output */
423  mysql_sha1_reset(&sha1_context);
424  mysql_sha1_input(&sha1_context, (uint8 *) to, SHA1_HASH_SIZE);
425  /* separate buffer is used to pass 'to' in octet2hex */
426  mysql_sha1_result(&sha1_context, hash_stage2);
427  /* convert hash_stage2 to hex string */
428  *to++= PVERSION41_CHAR;
429  octet2hex(to, (const char*) hash_stage2, SHA1_HASH_SIZE);
430 }

450 /*
451  Produce an obscure octet sequence from password and random
452  string, recieved from the server. This sequence corresponds to the
453  password, but password can not be easily restored from it. The sequence
454  is then sent to the server for validation. Trailing zero is not stored
455  in the buf as it is not needed.
456  This function is used by client to create authenticated reply to the
457  server's greeting.
458  SYNOPSIS
459  scramble()
460  buf OUT store scrambled string here. The buf must be at least
461  SHA1_HASH_SIZE bytes long.
462  message IN random message, must be exactly SCRAMBLE_LENGTH long and
463  NULL-terminated.
464  password IN users' password
465 */
466 
467 void
468 scramble(char *to, const char *message, const char *password)
469 {
470  SHA1_CONTEXT sha1_context;
471  uint8 hash_stage1[SHA1_HASH_SIZE];
472  uint8 hash_stage2[SHA1_HASH_SIZE];
473 
474  mysql_sha1_reset(&sha1_context);
475  /* stage 1: hash password */
476  mysql_sha1_input(&sha1_context, (uint8 *) password, (uint) strlen(password));
477  mysql_sha1_result(&sha1_context, hash_stage1);
478  /* stage 2: hash stage 1; note that hash_stage2 is stored in the database */
479  mysql_sha1_reset(&sha1_context);
480  mysql_sha1_input(&sha1_context, hash_stage1, SHA1_HASH_SIZE);
481  mysql_sha1_result(&sha1_context, hash_stage2);
482  /* create crypt string as sha1(message, hash_stage2) */;
483  mysql_sha1_reset(&sha1_context);
484  mysql_sha1_input(&sha1_context, (const uint8 *) message, SCRAMBLE_LENGTH);
485  mysql_sha1_input(&sha1_context, hash_stage2, SHA1_HASH_SIZE);
486  /* xor allows 'from' and 'to' overlap: lets take advantage of it */
487  mysql_sha1_result(&sha1_context, (uint8 *) to);
488  my_crypt(to, (const uchar *) to, hash_stage1, SCRAMBLE_LENGTH);
489 }

492 /*
493  Check that scrambled message corresponds to the password; the function
494  is used by server to check that recieved reply is authentic.
495  This function does not check lengths of given strings: message must be
496  null-terminated, reply and hash_stage2 must be at least SHA1_HASH_SIZE
497  long (if not, something fishy is going on).
498  SYNOPSIS
499  check_scramble()
500  scramble clients' reply, presumably produced by scramble()
501  message original random string, previously sent to client
502  (presumably second argument of scramble()), must be
503  exactly SCRAMBLE_LENGTH long and NULL-terminated.
504  hash_stage2 hex2octet-decoded database entry
505  All params are IN.
506 
507  RETURN VALUE
508  0 password is correct
509  !0 password is invalid
510 */
511 
512 my_bool
513 check_scramble(const uchar *scramble_arg, const char *message,
514  const uint8 *hash_stage2)
515 {
516  SHA1_CONTEXT sha1_context;
517  uint8 buf[SHA1_HASH_SIZE];
518  uint8 hash_stage2_reassured[SHA1_HASH_SIZE];
519 
520  mysql_sha1_reset(&sha1_context);
521  /* create key to encrypt scramble */
522  mysql_sha1_input(&sha1_context, (const uint8 *) message, SCRAMBLE_LENGTH);
523  mysql_sha1_input(&sha1_context, hash_stage2, SHA1_HASH_SIZE);
524  mysql_sha1_result(&sha1_context, buf);
525  /* encrypt scramble */
526  my_crypt((char *) buf, buf, scramble_arg, SCRAMBLE_LENGTH);
527  /* now buf supposedly contains hash_stage1: so we can get hash_stage2 */
528  mysql_sha1_reset(&sha1_context);
529  mysql_sha1_input(&sha1_context, buf, SHA1_HASH_SIZE);
530  mysql_sha1_result(&sha1_context, hash_stage2_reassured);
531  return memcmp(hash_stage2, hash_stage2_reassured, SHA1_HASH_SIZE);
532 }
