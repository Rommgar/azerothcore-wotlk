288  float GetThreatMod(float dist, float /*armor*/, uint32 health, uint32 /*maxhealth*/, Unit* target)
289  {
290  float unimportant_dist = (aiType == AI_TYPE_MELEE ? 5.0f : 25.0f);
291  if (dist > unimportant_dist) dist -= unimportant_dist;
292  else dist = 0.0f;
293  const float dist_factor = (aiType == AI_TYPE_MELEE ? 15.0f : 25.0f);
294  float mod_dist = dist_factor / (dist_factor + dist); // 0.2 .. 1.0
295  float mod_health = health > 20000 ? 2.0f : (40000 - health) / 10000.0f; // 2.0 .. 4.0
296  float mod_armor = aiType == AI_TYPE_MELEE ? Unit::CalcArmorReducedDamage(me, target, 10000, nullptr) / 10000.0f : 1.0f;
297  return mod_dist * mod_health * mod_armor;
298  }

310  if (HostileReference* ref = me->getThreatMgr().getOnlineContainer().getReferenceByTarget(pUnit))
