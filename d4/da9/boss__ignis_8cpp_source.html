339  {

396  case EVENT_GRAB:
397  {
398  std::list<Creature*> icl;
399  me->GetCreaturesWithEntryInRange(icl, 300.0f, NPC_IRON_CONSTRUCT);
400 
401  GuidVector playerGUIDs;
402  Map::PlayerList const& pl = me->GetMap()->GetPlayers();
403  Player* temp = nullptr;
404 
405  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
406  {
407  temp = itr->GetSource();
408  if( !temp->IsAlive() || temp->GetExactDist2d(me) > 90.0f )
409  continue;
410  if( me->GetVictim() && temp->GetGUID() == me->GetVictim()->GetGUID() )
411  continue;
412  bool found = false;
413  for (std::list<Creature*>::iterator iterator = icl.begin(); iterator != icl.end(); ++iterator)
414  {
415  if ((*iterator)->GetVictim() && (*iterator)->GetVictim()->GetGUID() == temp->GetGUID())
416  {
417  found = true;
418  break;
419  }
420  }
421 
422  if( !found )
423  playerGUIDs.push_back(temp->GetGUID());
424  }
425 
426  if( !playerGUIDs.empty() )
427  {
428  int8 pos = urand(0, playerGUIDs.size() - 1);
429  if( Player* pTarget = ObjectAccessor::GetPlayer(*me, playerGUIDs.at(pos)) )
430  {
431  me->Yell(TEXT_SLAG_POT, LANG_UNIVERSAL);
432  me->PlayDirectSound(SOUND_SLAG_POT);
433  me->CastSpell(pTarget, SPELL_GRAB, false);
434  }
435  }
436 
437  events.RepeatEvent(24000); // +6000 below
438  events.DelayEvents(6000);
439  }
440  break;
441  }

443  DoMeleeAttackIfReady();
444  }
