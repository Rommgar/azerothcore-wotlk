279  {
280  npc_escortAI::UpdateAI(diff);
281 
282  if (HasEscortState(STATE_ESCORT_PAUSED))
283  {

289  pSpotlight->DespawnOrUnsummon();
290

292  return;
293  }
294 
295  Talk(TalkCount);
296  ++TalkCount;
297  }
298  else TalkTimer -= diff;
299  }
300 
301  if (PerformanceReady)
302  {
303  if (!RaidWiped)
304  {
305  if (WipeTimer <= diff)
306  {
307  Map* map = me->GetMap();
308  if (!map->IsDungeon())
309  return;
310 
311  Map::PlayerList const& PlayerList = map->GetPlayers();
312  if (PlayerList.isEmpty())
313  return;
314 
315  RaidWiped = true;
316  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
317  {
318  if (i->GetSource()->IsAlive() && !i->GetSource()->IsGameMaster())
319  {
320  RaidWiped = false;
321  break;
322  }
323  }
324 
325  if (RaidWiped)
326  {
327  RaidWiped = true;
328  EnterEvadeMode();
329  return;
330  }
331 
332  WipeTimer = 15000;
333  }
334  else WipeTimer -= diff;
335  }
336  }
