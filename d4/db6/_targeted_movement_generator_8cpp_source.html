49 bool ChaseMovementGenerator<T>::DoUpdate(T* owner, uint32 time_diff)
50 {
51  if (!i_target.isValid() || !i_target->IsInWorld() || !owner->IsInMap(i_target.getTarget()))
52  return false;
53 
54  if (!owner || !owner->IsAlive())
55  return false;
56 
57  Creature* cOwner = owner->ToCreature();
58 
59  // the owner might be unable to move (rooted or casting), or we have lost the target, pause movement
60  if (owner->HasUnitState(UNIT_STATE_NOT_MOVE) || HasLostTarget(owner) || (cOwner && cOwner->IsMovementPreventedByCasting()))
61  {
62  owner->StopMoving();
63  _lastTargetPosition.reset();
64  if (Creature* cOwner2 = owner->ToCreature())
65  {
66  cOwner2->SetCannotReachTarget(false);
67  }
68 
69  return true;
70  }
71 
72  bool forceDest =
73  //(cOwner && (cOwner->isWorldBoss() || cOwner->IsDungeonBoss())) || // force for all bosses, even not in instances
74  (i_target->GetTypeId() == TYPEID_PLAYER && i_target->ToPlayer()->IsGameMaster()) || // for .npc follow
75  (owner->CanFly())
76  ; // closes "bool forceDest", that way it is more appropriate, so we can comment out crap whenever we need to
77 
78  Unit* target = i_target.getTarget();
79 
80  bool const mutualChase = IsMutualChase(owner, target);
81  float const hitboxSum = owner->GetCombatReach() + target->GetCombatReach();
82  float const minTarget = (_range ? _range->MinTolerance : 0.0f) + hitboxSum;
83  float const maxRange = _range ? _range->MaxRange + hitboxSum : owner->GetMeleeRange(target); // melee range already includes hitboxes
84  float const maxTarget = _range ? _range->MaxTolerance + hitboxSum : CONTACT_DISTANCE + hitboxSum;
85  Optional<ChaseAngle> angle = mutualChase ? Optional<ChaseAngle>() : _angle;
86 
87  i_recheckDistance.Update(time_diff);
88  if (i_recheckDistance.Passed())
89  {
90  i_recheckDistance.Reset(100);
91 
92  if (i_recalculateTravel && PositionOkay(owner, target, _movingTowards ? maxTarget : Optional<float>(), angle))
93  {
94  i_recalculateTravel = false;
95  i_path = nullptr;
96  if (Creature* cOwner2 = owner->ToCreature())
97  {
98  cOwner2->SetCannotReachTarget(false);
99  }
100 
101  owner->StopMoving();
102  owner->SetInFront(target);
103  MovementInform(owner);
104  return true;
105  }
106  }
107 
108  if (owner->HasUnitState(UNIT_STATE_CHASE_MOVE) && owner->movespline->Finalized())
109  {
110  i_recalculateTravel = false;
111  i_path = nullptr;
112  owner->ClearUnitState(UNIT_STATE_CHASE_MOVE);
113  owner->SetInFront(target);
114  MovementInform(owner);
115 
116  if (owner->IsWithinMeleeRange(this->i_target.getTarget()))
117  owner->Attack(this->i_target.getTarget(), true);
118  }
119 
120  if (_lastTargetPosition && i_target->GetPosition() == _lastTargetPosition.value() && mutualChase == _mutualChase)
121  return true;
122 
123  _lastTargetPosition = i_target->GetPosition();
124 
125  if (PositionOkay(owner, target, maxRange, angle) && !owner->HasUnitState(UNIT_STATE_CHASE_MOVE))
126  return true;
127 
128  float tarX, tarY, tarZ;
129  target->GetPosition(tarX, tarY, tarZ);
130 
131  bool withinRange = owner->IsInDist(target, maxRange);
132  bool withinLOS = owner->IsWithinLOS(tarX, tarY, tarZ);
133  bool moveToward = !(withinRange && withinLOS);
134 
135  _mutualChase = mutualChase;
136 
137  if (owner->HasUnitState(UNIT_STATE_CHASE_MOVE))
138  {
139  // can we get to the target?
140  if (cOwner && !target->isInAccessiblePlaceFor(cOwner))
141  {
142  cOwner->SetCannotReachTarget(true);
143  cOwner->StopMoving();
144  i_path = nullptr;
145  return true;
146  }
147  }
148 
149  if (!i_path || moveToward != _movingTowards)
150  i_path = std::make_unique<PathGenerator>(owner);
151  else
152  i_path->Clear();
153 
154  float x, y, z;
155  bool shortenPath;
156  // if we want to move toward the target and there's no fixed angle...
157  if (moveToward && !angle)
158  {
159  // ...we'll pathfind to the center, then shorten the path
160  target->GetPosition(x, y, z);
161  shortenPath = true;
162  }
163  else
164  {
165  // otherwise, we fall back to nearpoint finding
166  target->GetNearPoint(owner, x, y, z, (moveToward ? maxTarget : minTarget) - hitboxSum, 0, angle ? target->ToAbsoluteAngle(angle->RelativeAngle) : target->GetAngle(owner));
167  shortenPath = false;
168  }
169 
170  if (owner->IsHovering())
171  owner->UpdateAllowedPositionZ(x, y, z);
172 
173  i_recalculateTravel = true;
174 
175  bool success = i_path->CalculatePath(x, y, z, forceDest);
176  if (!success || i_path->GetPathType() & PATHFIND_NOPATH)
177  {
178  if (cOwner)
179  cOwner->SetCannotReachTarget(true);
180  return true;
181  }
182 
183  if (shortenPath)
184  i_path->ShortenPathUntilDist(G3D::Vector3(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ()), maxTarget);
185 
186  if (cOwner)
187  cOwner->SetCannotReachTarget(false);
188 
189  bool walk = false;
190  if (cOwner && !cOwner->IsPet())
191  {
192  walk = owner->IsWalking();
193  }
194 
195  owner->AddUnitState(UNIT_STATE_CHASE_MOVE);
196 
197  Movement::MoveSplineInit init(owner);
198  init.MovebyPath(i_path->GetPath());
199  init.SetFacing(target);
200  init.SetWalk(walk);
201  init.Launch();
202 
203  return true;
204 }

251 static Optional<float> GetVelocity(Unit* owner, Unit* target, G3D::Vector3 const& dest, bool playerPet)
252 {
253  Optional<float> speed = {};
254  if (!owner->IsInCombat() && !owner->IsVehicle() && !owner->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_POSSESSED) &&
255  (owner->IsPet() || owner->IsGuardian() || owner->GetGUID() == target->GetCritterGUID() || owner->GetCharmerOrOwnerGUID() == target->GetGUID()))
256  {
257  UnitMoveType moveType = Movement::SelectSpeedType(target->GetUnitMovementFlags());
258  speed = std::max(target->GetSpeed(moveType), owner->GetSpeed(moveType));
259 
260  if (playerPet)
261  {
262  float distance = owner->GetDistance2d(dest.x, dest.y) - (*speed / 2.f);
263  if (distance > 0.f)
264  {
265  float multiplier = 1.f + (distance / 10.f);
266  *speed *= multiplier;
267  }
268  else
269  {
270  switch (moveType)
271  {
272  case MOVE_RUN_BACK:
273  case MOVE_SWIM_BACK:
274  case MOVE_FLIGHT_BACK:
275  break;
276  default:
277  *speed *= 0.9f;
278  break;
279  }
280  }
281  }
282  }
283 
284  return speed;
285 }

287 static Position const PredictPosition(Unit* target)
288 {
289  Position pos = target->GetPosition();
290 
291  // 0.5 - it's time (0.5 sec) between starting movement opcode (e.g. MSG_MOVE_START_FORWARD) and MSG_MOVE_HEARTBEAT sent by client
292  float speed = target->GetSpeed(Movement::SelectSpeedType(target->GetUnitMovementFlags())) * 0.5f;
293  float orientation = target->GetOrientation();
294 
295  if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_FORWARD))
296  {
297  pos.m_positionX += cos(orientation) * speed;
298  pos.m_positionY += sin(orientation) * speed;
299  }
300  else if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_BACKWARD))
301  {
302  pos.m_positionX -= cos(orientation) * speed;
303  pos.m_positionY -= sin(orientation) * speed;
304  }
305 
306  if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_STRAFE_LEFT))
307  {
308  pos.m_positionX += cos(orientation + M_PI / 2.f) * speed;
309  pos.m_positionY += sin(orientation + M_PI / 2.f) * speed;
310  }
311  else if (target->m_movementInfo.HasMovementFlag(MOVEMENTFLAG_STRAFE_RIGHT))
312  {
313  pos.m_positionX += cos(orientation - M_PI / 2.f) * speed;
314  pos.m_positionY += sin(orientation - M_PI / 2.f) * speed;
315  }
316 
317  return pos;
318 }

365 template<class T>
366 bool FollowMovementGenerator<T>::DoUpdate(T* owner, uint32 time_diff)
367 {
368  if (!i_target.isValid() || !i_target->IsInWorld() || !owner->IsInMap(i_target.getTarget()))
369  return false;
370 
371  if (!owner || !owner->IsAlive())
372  return false;
373 
374  Creature* cOwner = owner->ToCreature();
375  Unit* target = i_target.getTarget();
376 
377  // the owner might be unable to move (rooted or casting), or we have lost the target, pause movement
378  if (owner->HasUnitState(UNIT_STATE_NOT_MOVE) || (cOwner && owner->ToCreature()->IsMovementPreventedByCasting()))
379  {
380  i_path = nullptr;
381  owner->StopMoving();
382  _lastTargetPosition.reset();
383  return true;
384  }
385 
386  bool followingMaster = false;
387  Pet* oPet = owner->ToPet();
388  if (oPet)
389  {
390  if (target->GetGUID() == oPet->GetOwnerGUID())
391  followingMaster = true;
392  }
393 
394  bool forceDest =
395  (followingMaster) || // allow pets following their master to cheat while generating paths
396  (i_target->GetTypeId() == TYPEID_PLAYER && i_target->ToPlayer()->IsGameMaster()) // for .npc follow
397  ; // closes "bool forceDest", that way it is more appropriate, so we can comment out crap whenever we need to
398 
399  bool targetIsMoving = false;
400  if (PositionOkay(target, owner->IsGuardian() && target->GetTypeId() == TYPEID_PLAYER, targetIsMoving, time_diff))
401  {
402  if (owner->HasUnitState(UNIT_STATE_FOLLOW_MOVE) && owner->movespline->Finalized())
403  {
404  owner->ClearUnitState(UNIT_STATE_FOLLOW_MOVE);
405  i_path = nullptr;
406  MovementInform(owner);
407 
408  if (i_recheckPredictedDistance)
409  {
410  i_recheckPredictedDistanceTimer.Reset(1000);
411  }
412 
413  owner->SetFacingTo(target->GetOrientation());
414  }
415  }
416  else
417  {
418  Position targetPosition = target->GetPosition();
419  _lastTargetPosition = targetPosition;
420 
421  // If player is moving and their position is not updated, we need to predict position
422  if (targetIsMoving)
423  {
424  Position predictedPosition = PredictPosition(target);
425  if (_lastPredictedPosition && _lastPredictedPosition->GetExactDistSq(&predictedPosition) < 0.25f)
426  return true;
427 
428  _lastPredictedPosition = predictedPosition;
429  targetPosition = predictedPosition;
430  i_recheckPredictedDistance = true;
431  }
432  else
433  {
434  i_recheckPredictedDistance = false;
435  i_recheckPredictedDistanceTimer.Reset(0);
436  }
437 
438  if (!i_path)
439  i_path = std::make_unique<PathGenerator>(owner);
440  else
441  i_path->Clear();
442 
443  float distance = _range - target->GetCombatReach();
444 
445  float relAngle = _angle.RelativeAngle;
446  float x, y, z;
447  target->GetNearPoint(owner, x, y, z, owner->GetCombatReach(), distance, target->ToAbsoluteAngle(relAngle), 0.f, &targetPosition);
448 
449  if (owner->IsHovering())
450  owner->UpdateAllowedPositionZ(x, y, z);
451 
452  bool success = i_path->CalculatePath(x, y, z, forceDest);
453  if (!success || (i_path->GetPathType() & PATHFIND_NOPATH && !followingMaster))
454  {
455  if (!owner->IsStopped())
456  owner->StopMoving();
457 
458  return true;
459  }
460 
461  owner->AddUnitState(UNIT_STATE_FOLLOW_MOVE);
462 
463  Movement::MoveSplineInit init(owner);
464  init.MovebyPath(i_path->GetPath());
465  init.SetWalk(target->IsWalking());
466  if (Optional<float> velocity = GetVelocity(owner, target, i_path->GetActualEndPosition(), owner->IsGuardian() && target->GetTypeId() == TYPEID_PLAYER))
467  init.SetVelocity(*velocity);
468  init.Launch();
469  }
470 
471  return true;
472 }
