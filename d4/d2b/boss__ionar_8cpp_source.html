150  void Split()
151  {
152  Talk(SAY_SPLIT);
153 
154  Creature* spark;
155  for (uint8 i = 0; i < 5; ++i)
156  {
157  if ((spark = me->SummonCreature(NPC_SPARK_OF_IONAR, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0, TEMPSUMMON_TIMED_DESPAWN, 20000)))
158  {
159  summons.Summon(spark);
160  spark->CastSpell(spark, me->GetMap()->IsHeroic() ? SPELL_SPARK_VISUAL_TRIGGER_H : SPELL_SPARK_VISUAL_TRIGGER_N, true);
161  spark->CastSpell(spark, SPELL_RANDOM_LIGHTNING, true);
162  spark->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PACIFIED | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
163  spark->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0);
164 
165  if (Player* tgt = SelectTargetFromPlayerList(100))
166  spark->GetMotionMaster()->MoveFollow(tgt, 0.0f, 0.0f, MOTION_SLOT_CONTROLLED);
167  }
168  }
169 
170  me->SetVisible(false);
171  me->SetControlled(true, UNIT_STATE_STUNNED);
172 
173  events.SetPhase(2);
174  events.ScheduleEvent(EVENT_CALL_SPARKS, 15000, 0, 2);
175  }

177  void UpdateAI(uint32 diff) override
178  {
179  if (!UpdateVictim())
180  return;
181 
182  events.Update(diff);
183 
184  if (me->HasUnitState(UNIT_STATE_CASTING))
185  return;
186 
187  switch (events.ExecuteEvent())
188  {
189  case EVENT_BALL_LIGHTNING:
190  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
191  me->CastSpell(target, me->GetMap()->IsHeroic() ? SPELL_BALL_LIGHTNING_H : SPELL_BALL_LIGHTNING_N, false);
192 
193  events.RepeatEvent(10000 + rand() % 1000);
194  break;
195  case EVENT_STATIC_OVERLOAD:
196  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
197  me->CastSpell(target, me->GetMap()->IsHeroic() ? SPELL_STATIC_OVERLOAD_H : SPELL_STATIC_OVERLOAD_N, false);
198 
199  events.RepeatEvent(5000 + rand() % 1000);
200  break;
201  case EVENT_CHECK_HEALTH:
202  if (HealthBelowPct(HealthCheck))
203  me->CastSpell(me, SPELL_DISPERSE, false);
204 
205  events.RepeatEvent(1000);
206  return;
207  case EVENT_CALL_SPARKS:
208  {
209  EntryCheckPredicate pred(NPC_SPARK_OF_IONAR);
210  summons.DoAction(ACTION_CALLBACK, pred);
211  events.ScheduleEvent(EVENT_RESTORE, 2000, 0, 2);
212  return;
213  }
214  case EVENT_RESTORE:
215  EntryCheckPredicate pred(NPC_SPARK_OF_IONAR);
216  summons.DoAction(ACTION_SPARK_DESPAWN, pred);
217 
218  me->SetVisible(true);
219  me->SetControlled(false, UNIT_STATE_STUNNED);
220  ScheduleEvents(true);
221  return;
222  }
223 
224  DoMeleeAttackIfReady();
225  }
226  };
