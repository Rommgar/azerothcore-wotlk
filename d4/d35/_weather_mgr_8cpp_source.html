83  void LoadWeatherData()
84  {
85  uint32 oldMSTime = getMSTime();
86 
87  uint32 count = 0;
88 
89  QueryResult result = WorldDatabase.Query("SELECT "
90  "zone, spring_rain_chance, spring_snow_chance, spring_storm_chance,"
91  "summer_rain_chance, summer_snow_chance, summer_storm_chance,"
92  "fall_rain_chance, fall_snow_chance, fall_storm_chance,"
93  "winter_rain_chance, winter_snow_chance, winter_storm_chance,"
94  "ScriptName FROM game_weather");
95 
96  if (!result)
97  {
98  LOG_ERROR("sql.sql", ">> Loaded 0 weather definitions. DB table `game_weather` is empty.");
99  LOG_INFO("server.loading", " ");
100  return;
101  }
102 
103  do
104  {
105  Field* fields = result->Fetch();
106 
107  uint32 zone_id = fields[0].GetUInt32();
108 
109  WeatherData& wzc = mWeatherZoneMap[zone_id];
110 
111  for (uint8 season = 0; season < WEATHER_SEASONS; ++season)
112  {
113  wzc.data[season].rainChance = fields[season * (MAX_WEATHER_TYPE - 1) + 1].GetUInt8();
114  wzc.data[season].snowChance = fields[season * (MAX_WEATHER_TYPE - 1) + 2].GetUInt8();
115  wzc.data[season].stormChance = fields[season * (MAX_WEATHER_TYPE - 1) + 3].GetUInt8();
116 
117  if (wzc.data[season].rainChance > 100)
118  {
119  wzc.data[season].rainChance = 25;
120  LOG_ERROR("sql.sql", "Weather for zone %u season %u has wrong rain chance > 100%%", zone_id, season);
121  }
122 
123  if (wzc.data[season].snowChance > 100)
124  {
125  wzc.data[season].snowChance = 25;
126  LOG_ERROR("sql.sql", "Weather for zone %u season %u has wrong snow chance > 100%%", zone_id, season);
127  }
128 
129  if (wzc.data[season].stormChance > 100)
130  {
131  wzc.data[season].stormChance = 25;
132  LOG_ERROR("sql.sql", "Weather for zone %u season %u has wrong storm chance > 100%%", zone_id, season);
133  }
134  }
135 
136  wzc.ScriptId = sObjectMgr->GetScriptId(fields[13].GetCString());
137 
138  ++count;
139  } while (result->NextRow());
140 
141  LOG_INFO("server.loading", ">> Loaded %u weather definitions in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
142  LOG_INFO("server.loading", " ");
143  }
