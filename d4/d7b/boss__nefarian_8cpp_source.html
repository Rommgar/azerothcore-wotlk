250  void UpdateAI(uint32 diff) override
251  {
252  if (!UpdateVictim())
253  {
254  events.Update(diff);
255 
256  while (uint32 eventId = events.ExecuteEvent())
257  {
258  switch (eventId)
259  {
260  case EVENT_PATH_2:
261  me->GetMotionMaster()->MovePath(NEFARIUS_PATH_2, false);
262  events.ScheduleEvent(EVENT_CHAOS_1, 7000);
263  break;
264  case EVENT_CHAOS_1:
265  if (Creature* gyth = me->FindNearestCreature(NPC_GYTH, 75.0f, true))
266  {
267  me->SetFacingToObject(gyth);
268  Talk(SAY_CHAOS_SPELL);
269  }
270  events.ScheduleEvent(EVENT_CHAOS_2, 2000);
271  break;
272  case EVENT_CHAOS_2:
273  DoCast(SPELL_CHROMATIC_CHAOS);
274  me->SetFacingTo(1.570796f);
275  break;
276  case EVENT_SUCCESS_1:
277  if (Unit* player = me->SelectNearestPlayer(60.0f))
278  {
279  me->SetFacingToObject(player);
280  Talk(SAY_SUCCESS);
281  if (GameObject* portcullis1 = me->FindNearestGameObject(GO_PORTCULLIS_ACTIVE, 65.0f))
282  portcullis1->SetGoState(GO_STATE_ACTIVE);
283  if (GameObject* portcullis2 = me->FindNearestGameObject(GO_PORTCULLIS_TOBOSSROOMS, 80.0f))
284  portcullis2->SetGoState(GO_STATE_ACTIVE);
285  }
286  events.ScheduleEvent(EVENT_SUCCESS_2, 4000);
287  break;
288  case EVENT_SUCCESS_2:
289  DoCast(me, SPELL_VAELASTRASZZ_SPAWN);
290  me->DespawnOrUnsummon(1000);
291  break;
292  case EVENT_PATH_3:
293  me->GetMotionMaster()->MovePath(NEFARIUS_PATH_3, false);
294  break;
295  default:
296  break;
297  }
298  }
299  return;
300  }
301 
302  // Only do this if we haven't spawned nefarian yet
303  if (UpdateVictim() && SpawnedAdds <= 42)
304  {
305  events.Update(diff);
306 
307  if (me->HasUnitState(UNIT_STATE_CASTING))
308  return;
309 
310  while (uint32 eventId = events.ExecuteEvent())
311  {
312  switch (eventId)
313  {
314  case EVENT_SHADOW_BOLT:
315  switch (urand(0, 1))
316  {
317  case 0:
318  DoCastVictim(SPELL_SHADOWBOLT_VOLLEY);
319  break;
320  case 1:
321  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
322  DoCast(target, SPELL_SHADOWBOLT);
323  break;
324  }
325  DoResetThreat();
326  events.ScheduleEvent(EVENT_SHADOW_BOLT, urand(3000, 10000));
327  break;
328  case EVENT_FEAR:
329  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
330  DoCast(target, SPELL_FEAR);
331  events.ScheduleEvent(EVENT_FEAR, urand(10000, 20000));
332  break;
333  case EVENT_MIND_CONTROL:
334  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40, true))
335  DoCast(target, SPELL_SHADOW_COMMAND);
336  events.ScheduleEvent(EVENT_MIND_CONTROL, urand(30000, 35000));
337  break;
338  case EVENT_SPAWN_ADD:
339  for (uint8 i=0; i<2; ++i)
340  {
341  uint32 CreatureID;
342  if (urand(0, 2) == 0)
343  CreatureID = NPC_CHROMATIC_DRAKANOID;
344  else
345  CreatureID = Entry[urand(0, 4)];
346  if (Creature* dragon = me->SummonCreature(CreatureID, DrakeSpawnLoc[i]))
347  {
348  dragon->SetFaction(FACTION_DRAGONFLIGHT_BLACK);
349  dragon->AI()->AttackStart(me->GetVictim());
350  }
351 
352  if (++SpawnedAdds >= 42)
353  {
354  if (Creature* nefarian = me->SummonCreature(NPC_NEFARIAN, NefarianLoc[0]))
355  {
356  nefarian->setActive(true);
357  nefarian->SetCanFly(true);
358  nefarian->SetDisableGravity(true);
359  nefarian->CastSpell(nullptr, SPELL_SHADOWFLAME_INITIAL);
360  nefarian->GetMotionMaster()->MovePoint(1, NefarianLoc[1]);
361  }
362  events.CancelEvent(EVENT_MIND_CONTROL);
363  events.CancelEvent(EVENT_FEAR);
364  events.CancelEvent(EVENT_SHADOW_BOLT);
365  me->SetVisible(false);
366  return;
367  }
368  }
369  events.ScheduleEvent(EVENT_SPAWN_ADD, 4000);
370  break;
371  }
372 
373  if (me->HasUnitState(UNIT_STATE_CASTING))
374  return;
375  }
376  }
