229 void WorldSession::HandleGuildRankOpcode(WorldPacket& recvPacket)
230 {
231  uint32 rankId;
232  recvPacket >> rankId;
233 
234  uint32 rights;
235  recvPacket >> rights;
236 
237  std::string rankName;
238  recvPacket >> rankName;
239 
240  uint32 money;
241  recvPacket >> money;
242 
243  LOG_DEBUG("guild", "CMSG_GUILD_RANK [%s]: Rank: %s (%u)", GetPlayerInfo().c_str(), rankName.c_str(), rankId);
244 
245  Guild* guild = GetPlayer()->GetGuild();
246  if (!guild)
247  {
248  recvPacket.rpos(recvPacket.wpos());
249  return;
250  }
251 
252  // Check for overflow
253  if (rankName.length() > 15)
254  return;
255 
256  // Cleanup bad characters
257  cleanStr(rankName);
258 
259  GuildBankRightsAndSlotsVec rightsAndSlots(GUILD_BANK_MAX_TABS);
260 
261  for (uint8 tabId = 0; tabId < GUILD_BANK_MAX_TABS; ++tabId)
262  {
263  uint32 bankRights;
264  uint32 slots;
265 
266  recvPacket >> bankRights;
267  recvPacket >> slots;
268 
269  rightsAndSlots[tabId] = GuildBankRightsAndSlots(tabId, bankRights, slots);
270  }
271 
272  guild->HandleSetRankInfo(this, rankId, rankName, rights, money, rightsAndSlots);
273 }

432 void WorldSession::HandleGuildBankSwapItems(WorldPacket& recvData)
433 {
434  LOG_DEBUG("guild", "CMSG_GUILD_BANK_SWAP_ITEMS [%s]", GetPlayerInfo().c_str());
435 
436  ObjectGuid GoGuid;
437  recvData >> GoGuid;
438 
439  if (!GetPlayer()->GetGameObjectIfCanInteractWith(GoGuid, GAMEOBJECT_TYPE_GUILD_BANK))
440  {
441  recvData.rfinish(); // Prevent additional spam at rejected packet
442  return;
443  }
444 
445  Guild* guild = GetPlayer()->GetGuild();
446  if (!guild)
447  {
448  recvData.rfinish(); // Prevent additional spam at rejected packet
449  return;
450  }
451 
452  uint8 bankToBank;
453  recvData >> bankToBank;
454 
455  uint8 tabId;
456  uint8 slotId;
457  uint32 itemEntry;
458  uint32 splitedAmount = 0;
459 
460  if (bankToBank)
461  {
462  uint8 destTabId;
463  recvData >> destTabId;
464 
465  uint8 destSlotId;
466  recvData >> destSlotId;
467  recvData.read_skip<uint32>(); // Always 0
468 
469  recvData >> tabId;
470  recvData >> slotId;
471  recvData >> itemEntry;
472  recvData.read_skip<uint8>(); // Always 0
473 
474  recvData >> splitedAmount;
475 
476  guild->SwapItems(GetPlayer(), tabId, slotId, destTabId, destSlotId, splitedAmount);
477  }
478  else
479  {
480  uint8 playerBag = NULL_BAG;
481  uint8 playerSlotId = NULL_SLOT;
482  uint8 toChar = 1;
483 
484  recvData >> tabId;
485  recvData >> slotId;
486  recvData >> itemEntry;
487 
488  uint8 autoStore;
489  recvData >> autoStore;
490  if (autoStore)
491  {
492  recvData.read_skip<uint32>(); // autoStoreCount
493  recvData.read_skip<uint8>(); // ToChar (?), always and expected to be 1 (autostore only triggered in Bank -> Char)
494  recvData.read_skip<uint32>(); // Always 0
495  }
496  else
497  {
498  recvData >> playerBag;
499  recvData >> playerSlotId;
500  recvData >> toChar;
501  recvData >> splitedAmount;
502  }
503 
504  // Player <-> Bank
505  // Allow to work with inventory only
506  if (!Player::IsInventoryPos(playerBag, playerSlotId) && !(playerBag == NULL_BAG && playerSlotId == NULL_SLOT))
507  GetPlayer()->SendEquipError(EQUIP_ERR_NONE, nullptr);
508  else
509  guild->SwapItemsWithInventory(GetPlayer(), toChar, tabId, slotId, playerBag, playerSlotId, splitedAmount);
510  }
511 }
