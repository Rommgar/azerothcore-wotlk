36 /*
37  Message texts are registered into a linked list of 'my_err_head' structs.
38  Each struct contains (1.) an array of pointers to C character strings with
39  '\0' termination, (2.) the error number for the first message in the array
40  (array index 0) and (3.) the error number for the last message in the array
41  (array index (last - first)).
42  The array may contain gaps with NULL pointers and pointers to empty strings.
43  Both kinds of gaps will be translated to "Unknown error %d.", if my_error()
44  is called with a respective error number.
45  The list of header structs is sorted in increasing order of error numbers.
46  Negative error numbers are allowed. Overlap of error numbers is not allowed.
47  Not registered error numbers will be translated to "Unknown error %d.".
48 */
49 static struct my_err_head
50 {
51  struct my_err_head *meh_next; /* chain link */
52  const char** (*get_errmsgs) (); /* returns error message format */
53  int meh_first; /* error number matching array slot 0 */
54  int meh_last; /* error number matching last slot */
55 } my_errmsgs_globerrs = {NULL, get_global_errmsgs, EE_ERROR_FIRST, EE_ERROR_LAST};

71 void my_error(int nr, myf MyFlags, ...)
72 {
73  const char *format;
74  struct my_err_head *meh_p;
75  va_list args;
76  char ebuff[ERRMSGSIZE];
77  DBUG_ENTER("my_error");
78  DBUG_PRINT("my", ("nr: %d MyFlags: %d errno: %d", nr, MyFlags, errno));
79 
80  /* Search for the error messages array, which could contain the message. */
81  for (meh_p= my_errmsgs_list; meh_p; meh_p= meh_p->meh_next)
82  if (nr <= meh_p->meh_last)
83  break;
84 
85  /* get the error message string. Default, if NULL or empty string (""). */
86  if (! (format= (meh_p && (nr >= meh_p->meh_first)) ?
87  meh_p->get_errmsgs()[nr - meh_p->meh_first] : NULL) || ! *format)
88  (void) my_snprintf (ebuff, sizeof(ebuff), "Unknown error %d", nr);
89  else
90  {
91  va_start(args,MyFlags);
92  (void) my_vsnprintf_ex(&my_charset_utf8_general_ci, ebuff,
93  sizeof(ebuff), format, args);
94  va_end(args);
95  }
96  (*error_handler_hook)(nr, ebuff, MyFlags);
97  DBUG_VOID_RETURN;
98 }

167 /*
168  Register error messages for use with my_error().
169 
170  SYNOPSIS
171  my_error_register()
172  errmsgs array of pointers to error messages
173  first error number of first message in the array
174  last error number of last message in the array
175 
176  DESCRIPTION
177  The pointer array is expected to contain addresses to NUL-terminated
178  C character strings. The array contains (last - first + 1) pointers.
179  NULL pointers and empty strings ("") are allowed. These will be mapped to
180  "Unknown error" when my_error() is called with a matching error number.
181  This function registers the error numbers 'first' to 'last'.
182  No overlapping with previously registered error numbers is allowed.
183 
184  RETURN
185  0 OK
186  != 0 Error
187 */
188 
189 int my_error_register(const char** (*get_errmsgs) (), int first, int last)
190 {
191  struct my_err_head *meh_p;
192  struct my_err_head **search_meh_pp;
193 
194  /* Allocate a new header structure. */
195  if (! (meh_p= (struct my_err_head*) my_malloc(sizeof(struct my_err_head),
196  MYF(MY_WME))))
197  return 1;
198  meh_p->get_errmsgs= get_errmsgs;
199  meh_p->meh_first= first;
200  meh_p->meh_last= last;
201 
202  /* Search for the right position in the list. */
203  for (search_meh_pp= &my_errmsgs_list;
204  *search_meh_pp;
205  search_meh_pp= &(*search_meh_pp)->meh_next)
206  {
207  if ((*search_meh_pp)->meh_last > first)
208  break;
209  }
210 
211  /* Error numbers must be unique. No overlapping is allowed. */
212  if (*search_meh_pp && ((*search_meh_pp)->meh_first <= last))
213  {
214  my_free(meh_p);
215  return 1;
216  }
217 
218  /* Insert header into the chain. */
219  meh_p->meh_next= *search_meh_pp;
220  *search_meh_pp= meh_p;
221  return 0;
222 }

225 /*
226  Unregister formerly registered error messages.
227 
228  SYNOPSIS
229  my_error_unregister()
230  first error number of first message
231  last error number of last message
232 
233  DESCRIPTION
234  This function unregisters the error numbers 'first' to 'last'.
235  These must have been previously registered by my_error_register().
236  'first' and 'last' must exactly match the registration.
237  If a matching registration is present, the header is removed from the
238  list and the pointer to the error messages pointers array is returned.
239  Otherwise, NULL is returned.
240 
241  RETURN
242  non-NULL OK, returns address of error messages pointers array.
243  NULL Error, no such number range registered.
244 */
245 
246 const char **my_error_unregister(int first, int last)
247 {
248  struct my_err_head *meh_p;
249  struct my_err_head **search_meh_pp;
250  const char **errmsgs;
251 
252  /* Search for the registration in the list. */
253  for (search_meh_pp= &my_errmsgs_list;
254  *search_meh_pp;
255  search_meh_pp= &(*search_meh_pp)->meh_next)
256  {
257  if (((*search_meh_pp)->meh_first == first) &&
258  ((*search_meh_pp)->meh_last == last))
259  break;
260  }
261  if (! *search_meh_pp)
262  return NULL;
263 
264  /* Remove header from the chain. */
265  meh_p= *search_meh_pp;
266  *search_meh_pp= meh_p->meh_next;
267 
268  /* Save the return value and free the header. */
269  errmsgs= meh_p->get_errmsgs();
270  my_free(meh_p);
271 
272  return errmsgs;
273 }
