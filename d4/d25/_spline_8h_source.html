183  template<class T> inline void initLengths(T& cacher)
184  {
185  index_type i = index_lo;
186  lengths.resize(index_hi + 1);
187  length_type prev_length = 0, new_length = 0;
188  while (i < index_hi)
189  {
190  new_length = cacher(*this, i);
191  // length overflowed, assign to max positive value
192  if (new_length < 0)
193  new_length = std::numeric_limits<length_type>::max();
194  lengths[++i] = new_length;
195 
196  ASSERT(prev_length <= new_length);
197  prev_length = new_length;
198  }
199  }
200

204  [[nodiscard]] length_type length(index_type first, index_type last) const { return lengths[last] - lengths[first];}
