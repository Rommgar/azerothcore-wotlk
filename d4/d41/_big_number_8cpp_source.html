193 void BigNumber::GetBytes(uint8* buf, size_t bufsize, bool littleEndian) const
194 {
195 #if defined(OPENSSL_VERSION_NUMBER) && OPENSSL_VERSION_NUMBER < 0x10100000L
196  int nBytes = GetNumBytes();
197  ASSERT(nBytes >= 0, "Bignum has negative number of bytes (%d).", nBytes);
198  std::size_t numBytes = static_cast<std::size_t>(nBytes);
199 
200  // too large to store
201  ASSERT(numBytes <= bufsize, "Buffer of size %zu is too small to hold bignum with %zu bytes.\n", bufsize, numBytes);
202 
203  // If we need more bytes than length of BigNumber set the rest to 0
204  if (numBytes < bufsize)
205  {
206  memset((void*)buf, 0, bufsize);
207  }
208 
209  BN_bn2bin(_bn, buf + (bufsize - numBytes));
210 
211  // openssl's BN stores data internally in big endian format, reverse if little endian desired
212  if (littleEndian)
213  {
214  std::reverse(buf, buf + bufsize);
215  }
216 #else
217  int res = littleEndian ? BN_bn2lebinpad(_bn, buf, bufsize) : BN_bn2binpad(_bn, buf, bufsize);
218  ASSERT(res > 0, "Buffer of size %zu is too small to hold bignum with %d bytes.\n", bufsize, BN_num_bytes(_bn));
219 #endif
220 }
