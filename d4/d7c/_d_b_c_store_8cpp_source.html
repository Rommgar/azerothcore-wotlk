37 
38  // Check if load was sucessful, only then continue

40  return false;
41 
42  _fieldCount = dbc.GetCols();
43 
44  // load raw non-string data
45  _dataTable = dbc.AutoProduceData(_fileFormat, _indexTableSize, indexTable);
