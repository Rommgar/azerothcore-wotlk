116  void ProcessSlaughterEvent()
117  {

119  {
120  if (Creature* baron = instance->GetCreature(_baronRivendareGUID))
121  baron->AI()->Talk(SAY_BRAON_SUMMON_RAMSTEIN);
122 
123  Position pos = {4032.20f, -3378.06f, 119.75f, 4.67f};
124  instance->SummonCreature(NPC_RAMSTEIN_THE_GORGER, pos);
125  }
126  if (_slaughterProgress == 2)
127  {
128  for (uint32 i = 0; i < 33; ++i)
129  events.ScheduleEvent(EVENT_SPAWN_MINDLESS, 5000 + i * 210);
130  if (Creature* baron = instance->GetCreature(_baronRivendareGUID))
131  if (GameObject* gate = baron->FindNearestGameObject(GO_SLAUGHTER_GATE_SIDE, 200.0f))
132  gate->SetGoState(GO_STATE_ACTIVE);
133  }
134  if (_slaughterProgress == 3)
135  {
136  events.ScheduleEvent(EVENT_SPAWN_BLACK_GUARD, 20000);

139  {
140  if (Creature* baron = instance->GetCreature(_baronRivendareGUID))

373  void Update(uint32 diff) override
374  {
375  events.Update(diff);
376 
377  Map::PlayerList const& players = instance->GetPlayers();
378  // Loop over the two Gate traps, each one has up to three timers (trap reset, gate opening delay, critters spawning delay)
379  for (uint8 i = 0; i < 2; i++)
380  {
381  // if the gate is in cooldown, skip the other checks
382  if (_gateTrapsCooldown[i])
383  break;
384 
385  // Check that the trap is not on cooldown, if so check if player/pet is in range
386  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
387  {
388  if (Player* player = itr->GetSource())
389  {
390  // should pet also trigger the trap? could not find any source for it
391  if (!player->IsGameMaster() && player->IsWithinDist2d(aGateTrap[i].m_positionX, aGateTrap[i].m_positionY, 5.5f))
392  {
393  // Check if timer was not already set by another player/pet a few milliseconds before
394  if (_gateTrapsCooldown[i])
395  return;
396 
397  _gateTrapsCooldown[i] = true;
398 
399  // close the gates
400  if (_trapGatesGUIDs[2 * i])
401  DoUseDoorOrButton(_trapGatesGUIDs[2 * i]);
402  if (_trapGatesGUIDs[2 * i + 1])
403  DoUseDoorOrButton(_trapGatesGUIDs[2 * i + 1]);
404 
405  _trappedPlayerGUID = player->GetGUID();
406 
407  if (i == 0)
408  {
409  // set timer to reset the trap
410  events.ScheduleEvent(EVENT_GATE1_TRAP, 30 * MINUTE * IN_MILLISECONDS);
411  // set timer to reopen gates
412  events.ScheduleEvent(EVENT_GATE1_DELAY, 20 * IN_MILLISECONDS);
413  // set timer to spawn the plagued critters
414  events.ScheduleEvent(EVENT_GATE1_CRITTER_DELAY, 2 * IN_MILLISECONDS);
415  }
416  else if (i == 1)
417  {
418  // set timer to reset the trap
419  events.ScheduleEvent(EVENT_GATE2_TRAP, 30 * MINUTE * IN_MILLISECONDS);
420  // set timer to reopen gates
421  events.ScheduleEvent(EVENT_GATE2_DELAY, 20 * IN_MILLISECONDS);
422  // set timer to spawn the plagued critters
423  events.ScheduleEvent(EVENT_GATE2_CRITTER_DELAY, 2 * IN_MILLISECONDS);
424  }
425  }
426  }
427  }
