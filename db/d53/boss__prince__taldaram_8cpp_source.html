381  void UpdateAI(uint32 diff) override
382  {
383  if (!UpdateVictim())
384  {
385  return;
386  }
387 
388  events.Update(diff);
389 
390  if (me->HasUnitState(UNIT_STATE_CASTING))
391  {
392  return;
393  }
394 
395  while (uint32 const eventId = events.ExecuteEvent())
396  {
397  switch (eventId)
398  {
399  case EVENT_PRINCE_BLOODTHIRST:
400  {
401  DoCastSelf(SPELL_BLOODTHIRST);
402  events.RepeatEvent(10000);
403  break;
404  }
405  case EVENT_PRINCE_FLAME_SPHERES:
406  {
407  if (Unit* victim = me->GetVictim())
408  {
409  DoCast(victim, SPELL_CONJURE_FLAME_SPHERE);
410  victimSperePos = *victim;
411  }
412 
413  if (!events.GetNextEventTime(EVENT_PRINCE_VANISH))
414  {
415  events.RescheduleEvent(EVENT_PRINCE_VANISH, 14000);
416  }
417  else
418  {
419  // Make sure that Vanish won't get triggered at same time as sphere summon
420  events.DelayEvents(4000);
421  }
422 
423  events.RepeatEvent(15000);
424  break;
425  }
426  case EVENT_PRINCE_VANISH:
427  {
428  //Count alive players
429  uint8 count = 0;
430  std::list<HostileReference*> const t_list = me->getThreatMgr().getThreatList();
431  if (!t_list.empty())
432  {
433  for (HostileReference const* reference : t_list)
434  {
435  if (reference)
436  {
437  Unit const* pTarget = ObjectAccessor::GetUnit(*me, reference->getUnitGuid());
438  if (pTarget && pTarget->GetTypeId() == TYPEID_PLAYER && pTarget->IsAlive())
439  {
440  ++count;
441  }
442  }
443  }
444  }
445 
446  // He only vanishes if there are 3 or more alive players
447  if (count > 2)
448  {
449  Talk(SAY_VANISH);
450  DoCastSelf(SPELL_VANISH, false);
451  if (Unit* pEmbraceTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
452  {
453  vanishTarget_GUID = pEmbraceTarget->GetGUID();
454  }
455 
456  events.CancelEvent(EVENT_PRINCE_FLAME_SPHERES);
457  events.CancelEvent(EVENT_PRINCE_BLOODTHIRST);
458  events.ScheduleEvent(EVENT_PRINCE_VANISH_RUN, 2499);
459  }
460  break;
461  }
462  case EVENT_PRINCE_VANISH_RUN:
463  {
464  if (Unit* _vanishTarget = ObjectAccessor::GetUnit(*me, vanishTarget_GUID))
465  {
466  vanishDamage = 0;
467  DoCast(_vanishTarget, SPELL_SHADOWSTEP);
468  me->CastSpell(_vanishTarget, SPELL_EMBRACE_OF_THE_VAMPYR, false);
469  me->RemoveAura(SPELL_VANISH);
470  }
471 
472  events.ScheduleEvent(EVENT_PRINCE_RESCHEDULE, 20000);
473  break;
474  }
475  case EVENT_PRINCE_RESCHEDULE:
476  {
477  ScheduleCombatEvents();
478  break;
479  }
480  }
481 
482  if (me->HasUnitState(UNIT_STATE_CASTING))

485  }
486  }
487

492  }
