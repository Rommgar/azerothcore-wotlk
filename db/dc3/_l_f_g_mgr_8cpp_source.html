273  void LFGMgr::Update(uint32 tdiff, uint8 task)
274  {
275  if (!isOptionEnabled(LFG_OPTION_ENABLE_DUNGEON_FINDER | LFG_OPTION_ENABLE_RAID_BROWSER | LFG_OPTION_ENABLE_SEASONAL_BOSSES))
276  return;
277 
278  if (task == 0)
279  {
280  time_t currTime = time(nullptr);
281 
282  // Remove obsolete role checks
283  for (LfgRoleCheckContainer::iterator it = RoleChecksStore.begin(); it != RoleChecksStore.end();)
284  {
285  LfgRoleCheckContainer::iterator itRoleCheck = it++;
286  LfgRoleCheck& roleCheck = itRoleCheck->second;
287  if (currTime < roleCheck.cancelTime)
288  continue;
289  roleCheck.state = LFG_ROLECHECK_MISSING_ROLE;
290 
291  for (LfgRolesMap::const_iterator itRoles = roleCheck.roles.begin(); itRoles != roleCheck.roles.end(); ++itRoles)
292  {
293  ObjectGuid guid = itRoles->first;
294  RestoreState(guid, "Remove Obsolete RoleCheck");
295  SendLfgRoleCheckUpdate(guid, roleCheck);
296  if (guid == roleCheck.leader)
297  SendLfgJoinResult(guid, LfgJoinResultData(LFG_JOIN_FAILED, LFG_ROLECHECK_MISSING_ROLE));
298  }
299 
300  RestoreState(itRoleCheck->first, "Remove Obsolete RoleCheck");
301  RoleChecksStore.erase(itRoleCheck);
302  }
303 
304  // Remove obsolete proposals
305  for (LfgProposalContainer::iterator it = ProposalsStore.begin(); it != ProposalsStore.end();)
306  {
307  LfgProposalContainer::iterator itRemove = it++;
308  if (itRemove->second.cancelTime < currTime)
309  RemoveProposal(itRemove, LFG_UPDATETYPE_PROPOSAL_FAILED);
310  }
311 
312  // Remove obsolete kicks
313  for (LfgPlayerBootContainer::iterator it = BootsStore.begin(); it != BootsStore.end();)
314  {
315  LfgPlayerBootContainer::iterator itBoot = it++;
316  LfgPlayerBoot& boot = itBoot->second;
317  if (boot.cancelTime < currTime)
318  {
319  boot.inProgress = false;
320  for (LfgAnswerContainer::const_iterator itVotes = boot.votes.begin(); itVotes != boot.votes.end(); ++itVotes)
321  {
322  ObjectGuid pguid = itVotes->first;
323  if (pguid != boot.victim)
324  SendLfgBootProposalUpdate(pguid, boot);
325  SetState(pguid, LFG_STATE_DUNGEON);
326  }
327  SetState(itBoot->first, LFG_STATE_DUNGEON);
328  BootsStore.erase(itBoot);
329  }
330  }
331  }
332  else if (task == 1)
333  {
334  this->lastProposalId = m_lfgProposalId; // pussywizard: task 2 is done independantly, store previous value in LFGMgr for future use
335  uint8 newGroupsProcessed = 0;
336  // Check if a proposal can be formed with the new groups being added
337  for (LfgQueueContainer::iterator it = QueuesStore.begin(); it != QueuesStore.end(); ++it)
338  {
339  newGroupsProcessed += it->second.FindGroups();
340  if (newGroupsProcessed)
341  break;
342  }
343 
344  // Update all players status queue info
345  if (!newGroupsProcessed) // don't do this on updates that precessed groups (performance)
346  for (LfgQueueContainer::iterator it = QueuesStore.begin(); it != QueuesStore.end(); ++it)
347  it->second.UpdateQueueTimers(tdiff);
348  }
349  else if (task == 2)
350  {
351  if (lastProposalId != m_lfgProposalId)
352  {
353  // pussywizard: only one proposal can be created in World::Update (during maps update), and it has id == m_lfgProposalId, so try to find only that one, dunno why for loop here xD
354  for (LfgProposalContainer::const_iterator itProposal = ProposalsStore.find(m_lfgProposalId); itProposal != ProposalsStore.end(); ++itProposal)
355  {
356  uint32 proposalId = itProposal->first;
357  LfgProposal& proposal = ProposalsStore[proposalId];
358 
359  ObjectGuid guid;
360  for (LfgProposalPlayerContainer::const_iterator itPlayers = proposal.players.begin(); itPlayers != proposal.players.end(); ++itPlayers)
361  {
362  guid = itPlayers->first;
363  SetState(guid, LFG_STATE_PROPOSAL);
364  if (ObjectGuid gguid = GetGroup(guid))
365  {
366  SetState(gguid, LFG_STATE_PROPOSAL);
367  SendLfgUpdateParty(guid, LfgUpdateData(LFG_UPDATETYPE_PROPOSAL_BEGIN, GetSelectedDungeons(guid), GetComment(guid)));
368  }
369  else
370  SendLfgUpdatePlayer(guid, LfgUpdateData(LFG_UPDATETYPE_PROPOSAL_BEGIN, GetSelectedDungeons(guid), GetComment(guid)));
371  SendLfgUpdateProposal(guid, proposal);
372  }
373 
374  if (proposal.state == LFG_PROPOSAL_SUCCESS) // pussywizard: no idea what's the purpose of this xD
375  UpdateProposal(proposalId, guid, true);
376  }
377  }
378 
379  UpdateRaidBrowser(tdiff);
380  }

388  void LFGMgr::InitializeLockedDungeons(Player* player, uint8 level /* = 0 */)
389  {
390  ObjectGuid guid = player->GetGUID();
391  if (!level)
392  level = player->getLevel();
393  uint8 expansion = player->GetSession()->Expansion();
394  LfgDungeonSet const& dungeons = GetDungeonsByRandom(0);
395  LfgLockMap lock;
396 
397  bool onlySeasonalBosses = m_options == LFG_OPTION_ENABLE_SEASONAL_BOSSES;
398 
399  float avgItemLevel = player->GetAverageItemLevelForDF();
400 
401  for (LfgDungeonSet::const_iterator it = dungeons.begin(); it != dungeons.end(); ++it)
402  {
403  LFGDungeonData const* dungeon = GetLFGDungeon(*it);
404  if (!dungeon) // should never happen - We provide a list from sLFGDungeonStore
405  continue;
406  MapEntry const* mapEntry = sMapStore.LookupEntry(dungeon->map);
407  DungeonProgressionRequirements const* ar = sObjectMgr->GetAccessRequirement(dungeon->map, Difficulty(dungeon->difficulty));
408 
409  uint32 lockData = 0;
410  if (dungeon->expansion > expansion || (onlySeasonalBosses && !dungeon->seasonal))
411  lockData = LFG_LOCKSTATUS_INSUFFICIENT_EXPANSION;
412  else if (DisableMgr::IsDisabledFor(DISABLE_TYPE_MAP, dungeon->map, player))
413  lockData = LFG_LOCKSTATUS_RAID_LOCKED;
414  else if (DisableMgr::IsDisabledFor(DISABLE_TYPE_LFG_MAP, dungeon->map, player))
415  lockData = LFG_LOCKSTATUS_RAID_LOCKED;
416  else if (dungeon->difficulty > DUNGEON_DIFFICULTY_NORMAL && (!mapEntry || !mapEntry->IsRaid()) && sInstanceSaveMgr->PlayerIsPermBoundToInstance(player->GetGUID(), dungeon->map, Difficulty(dungeon->difficulty)))
417  lockData = LFG_LOCKSTATUS_RAID_LOCKED;
418  else if ((dungeon->minlevel > level && !sWorld->getBoolConfig(CONFIG_DUNGEON_ACCESS_REQUIREMENTS_LFG_DBC_LEVEL_OVERRIDE)) || (sWorld->getBoolConfig(CONFIG_DUNGEON_ACCESS_REQUIREMENTS_LFG_DBC_LEVEL_OVERRIDE) && ar && ar->levelMin > 0 && ar->levelMin > level))
419  lockData = LFG_LOCKSTATUS_TOO_LOW_LEVEL;
420  else if ((dungeon->maxlevel < level && !sWorld->getBoolConfig(CONFIG_DUNGEON_ACCESS_REQUIREMENTS_LFG_DBC_LEVEL_OVERRIDE)) || (sWorld->getBoolConfig(CONFIG_DUNGEON_ACCESS_REQUIREMENTS_LFG_DBC_LEVEL_OVERRIDE) && ar && ar->levelMax > 0 && ar->levelMax < level))
421  lockData = LFG_LOCKSTATUS_TOO_HIGH_LEVEL;
422  else if (dungeon->seasonal && !IsSeasonActive(dungeon->id))
423  lockData = LFG_LOCKSTATUS_NOT_IN_SEASON;
424  else if (ar)
425  {
426  // Check required items
427  for (const ProgressionRequirement* itemRequirement : ar->items)
428  {
429  if (itemRequirement->faction == TEAM_NEUTRAL || itemRequirement->faction == player->GetTeamId(true))
430  {
431  if (!player->HasItemCount(itemRequirement->id, 1))
432  {
433  lockData = LFG_LOCKSTATUS_MISSING_ITEM;
434  break;
435  }
436  }
437  }
438 
439  //Check for quests
440  for (const ProgressionRequirement* questRequirement : ar->quests)
441  {
442  if (questRequirement->faction == TEAM_NEUTRAL || questRequirement->faction == player->GetTeamId(true))
443  {
444  if (!player->GetQuestRewardStatus(questRequirement->id))
445  {
446  lockData = LFG_LOCKSTATUS_QUEST_NOT_COMPLETED;
447  break;
448  }
449  }
450  }
451 
452  //Check for ilvl
453  if (ar->reqItemLevel && (float)ar->reqItemLevel > avgItemLevel)
454  {
455  lockData = LFG_LOCKSTATUS_TOO_LOW_GEAR_SCORE;
456  }
457 
458  //Check if player has the required achievements
459  for (const ProgressionRequirement* achievementRequirement : ar->achievements)
460  {
461  if (achievementRequirement->faction == TEAM_NEUTRAL || achievementRequirement->faction == player->GetTeamId(true))
462  {
463  if (!player->HasAchieved(achievementRequirement->id))
464  {
465  lockData = LFG_LOCKSTATUS_MISSING_ACHIEVEMENT;
466  break;
467  }
468  }
469  }
470  }
471 
472  sScriptMgr->OnInitializeLockedDungeons(player, level, lockData, dungeon);
473 
474  /* TODO VoA closed if WG is not under team control (LFG_LOCKSTATUS_RAID_LOCKED)
475  lockData = LFG_LOCKSTATUS_TOO_LOW_GEAR_SCORE;
476  lockData = LFG_LOCKSTATUS_TOO_HIGH_GEAR_SCORE;
477  lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_LOW_LEVEL;
478  lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_HIGH_LEVEL;
479  */
480 
481  if (lockData)
482  lock[dungeon->Entry()] = lockData;
483  }
484 
485  sScriptMgr->OnAfterInitializeLockedDungeons(player);
486 
487  SetLockedDungeons(guid, lock);
488  }

500  void LFGMgr::JoinLfg(Player* player, uint8 roles, LfgDungeonSet& dungeons, const std::string& comment)
501  {
502  if (!player || dungeons.empty())
503  return;
504 
505  Group* grp = player->GetGroup();
506  ObjectGuid guid = player->GetGUID();
507  ObjectGuid gguid = grp ? grp->GetGUID() : guid;
508  LfgJoinResultData joinData;
509  LfgGuidSet players;
510  uint32 rDungeonId = 0;
511  bool isContinue = grp && grp->isLFGGroup() && GetState(gguid) != LFG_STATE_FINISHED_DUNGEON;
512 
513  if (grp && (grp->isBGGroup() || grp->isBFGroup()))
514  return;
515 
516  if (!sScriptMgr->CanJoinLfg(player, roles, dungeons, comment))
517  return;
518 
519  // pussywizard: can't join LFG/LFR while using LFR
520  if (GetState(player->GetGUID()) == LFG_STATE_RAIDBROWSER)
521  {
522  LfgDungeonSet tmp;
523  SendRaidBrowserJoinedPacket(player, tmp, ""); // the df "eye" can disappear in various case, resend if needed
524  return;
525  }
526 
527  // Do not allow to change dungeon in the middle of a current dungeon
528  if (isContinue)
529  {
530  dungeons.clear();
531  dungeons.insert(GetDungeon(gguid));
532  }
533 
534  LfgState state = GetState(gguid);
535  switch (state)
536  {
537  case LFG_STATE_ROLECHECK: // if joining again during rolecheck (eg. many players clicked continue inside instance)
538  if (gguid.IsGroup())
539  UpdateRoleCheck(gguid); // abort role check and remove from RoleChecksStore
540  break;
541  case LFG_STATE_QUEUED: // joining again while in a queue
542  {
543  LFGQueue& queue = GetQueue(gguid);
544  queue.RemoveFromQueue(gguid);
545  }
546  break;
547  case LFG_STATE_PROPOSAL: // if joining again during proposal
548  joinData.result = LFG_JOIN_INTERNAL_ERROR;
549  break;
550  /*case LFG_STATE_FINISHED_DUNGEON:
551  if (grp && grp->isLFGGroup())
552  joinData.result = LFG_JOIN_PARTY_NOT_MEET_REQS;
553  break;
554  */
555  default:
556  break;
557  }
558 
559  // Check if all dungeons are valid
560  bool isRaid = false;
561  if (joinData.result == LFG_JOIN_OK)
562  {
563  bool isDungeon = false;
564  for (LfgDungeonSet::const_iterator it = dungeons.begin(); it != dungeons.end() && joinData.result == LFG_JOIN_OK; ++it)
565  {
566  LfgType type = GetDungeonType(*it);
567  switch (type)
568  {
569  case LFG_TYPE_RANDOM:
570  if (dungeons.size() > 1) // Only allow 1 random dungeon
571  {
572  joinData.result = LFG_JOIN_DUNGEON_INVALID;
573  }
574  else
575  {
576  rDungeonId = (*dungeons.begin());
577  sScriptMgr->OnPlayerQueueRandomDungeon(player, rDungeonId);
578  }
579  // No break on purpose (Random can only be dungeon or heroic dungeon)
580  [[fallthrough]];
581  case LFG_TYPE_HEROIC:
582  case LFG_TYPE_DUNGEON:
583  if (isRaid)
584  {
585  joinData.result = LFG_JOIN_MIXED_RAID_DUNGEON;
586  }
587  isDungeon = true;
588  break;
589  case LFG_TYPE_RAID:
590  if (isDungeon)
591  {
592  joinData.result = LFG_JOIN_MIXED_RAID_DUNGEON;
593  }
594  isRaid = true;
595  break;
596  default:
597  LOG_ERROR("lfg", "Wrong dungeon type %u for dungeon %u", type, *it);
598  joinData.result = LFG_JOIN_DUNGEON_INVALID;
599  break;
600  }
601  }
602  }
603 
604  if (!isRaid && joinData.result == LFG_JOIN_OK)
605  {
606  // Check player or group member restrictions
607  if (player->InBattleground() || player->InArena() || player->InBattlegroundQueue())
608  joinData.result = LFG_JOIN_USING_BG_SYSTEM;
609  else if (player->HasAura(LFG_SPELL_DUNGEON_DESERTER))
610  joinData.result = LFG_JOIN_DESERTER;
611  else if (dungeons.empty())

635  }
636  }
637  else
638  players.insert(player->GetGUID());
639 
640  // Xinef: Check dungeon cooldown only for random dungeons

642  if (joinData.result == LFG_JOIN_OK && !isContinue && rDungeonId)

649  if (Player* plrg = itr->GetSource())

652  }
653  }
654  }
655

657  players.insert(player->GetGUID());

659  if (joinData.result == LFG_JOIN_OK)
660  {
661  // Expand random dungeons and check restrictions
662  if (rDungeonId)
663  dungeons = GetDungeonsByRandom(rDungeonId);

671  joinData.result = grp ? LFG_JOIN_PARTY_NOT_MEET_REQS : LFG_JOIN_NOT_MEET_REQS;
672  }
673  }

675  // pussywizard:

677  return;
678 
679  // Do not allow to change dungeon in the middle of a current dungeon

682  dungeons.clear();

706  return;

708  SetState(guid, LFG_STATE_RAIDBROWSER);
709  SendRaidBrowserJoinedPacket(player, dungeons, comment);

712 
713  std::string debugNames = "";

715  {

724

726  {
727  dungeons.clear();
728  dungeons.insert(rDungeonId);
729  }

733  LfgUpdateData updateData = LfgUpdateData(LFG_UPDATETYPE_JOIN_QUEUE, dungeons, comment);

738  ObjectGuid pguid = plrg->GetGUID();

747  }

750  UpdateRoleCheck(gguid, guid, roles);

797  void LFGMgr::LeaveLfg(ObjectGuid guid)
798  {
799  LOG_DEBUG("lfg", "LFGMgr::Leave: [%s]", guid.ToString().c_str());
800  ObjectGuid gguid = guid.IsGroup() ? guid : GetGroup(guid);
801  LfgState state = GetState(guid);
802  switch (state)
803  {
804  case LFG_STATE_QUEUED:
805  if (gguid)
806  {
807  LFGQueue& queue = GetQueue(gguid);
808  queue.RemoveFromQueue(gguid);
809  uint32 dungeonId = GetDungeon(gguid);
810  SetState(gguid, LFG_STATE_NONE);
811  const LfgGuidSet& players = GetPlayers(gguid);
812  for (LfgGuidSet::const_iterator it = players.begin(); it != players.end(); ++it)
813  {
814  SetState(*it, LFG_STATE_NONE);
815  SendLfgUpdateParty(*it, LfgUpdateData(LFG_UPDATETYPE_REMOVED_FROM_QUEUE));
816  }
817  if (Group* group = sGroupMgr->GetGroupByGUID(gguid.GetCounter()))
818  {
819  if (group->isLFGGroup())
820  {
821  SetDungeon(gguid, dungeonId);
822  }
823  }
824  }
825  else
826  {
827  LFGQueue& queue = GetQueue(guid);
828  queue.RemoveFromQueue(guid);
829  SendLfgUpdatePlayer(guid, LfgUpdateData(LFG_UPDATETYPE_REMOVED_FROM_QUEUE));
830  SetState(guid, LFG_STATE_NONE);
831  }
832  break;
833  case LFG_STATE_ROLECHECK:
834  if (gguid)
835  UpdateRoleCheck(gguid); // No player to update role = LFG_ROLECHECK_ABORTED
836  break;
837  case LFG_STATE_PROPOSAL:
838  {
839  // Remove from Proposals
840  LfgProposalContainer::iterator it = ProposalsStore.begin();
841  ObjectGuid pguid = gguid == guid ? GetLeader(gguid) : guid;
842  while (it != ProposalsStore.end())
843  {
844  LfgProposalPlayerContainer::iterator itPlayer = it->second.players.find(pguid);
845  if (itPlayer != it->second.players.end())
846  {
847  // Mark the player/leader of group who left as didn't accept the proposal
848  itPlayer->second.accept = LFG_ANSWER_DENY;
849  break;
850  }
851  ++it;
852  }
853 
854  // Remove from queue - if proposal is found, RemoveProposal will call RemoveFromQueue
855  if (it != ProposalsStore.end())
856  RemoveProposal(it, LFG_UPDATETYPE_PROPOSAL_DECLINED);
857  break;
858  }
859  case LFG_STATE_NONE:
860  break;
861  case LFG_STATE_DUNGEON:
862  case LFG_STATE_FINISHED_DUNGEON:
863  case LFG_STATE_BOOT:
864  if (guid != gguid) // Player
865  SetState(guid, LFG_STATE_NONE);
866  break;
867  case LFG_STATE_RAIDBROWSER:
868  LeaveRaidBrowser(guid);
869  SetCanOverrideRBState(guid, true);
870  SetState(guid, LFG_STATE_NONE);
871  SetCanOverrideRBState(guid, false);
872  SendLfgUpdatePlayer(guid, LfgUpdateData(LFG_UPDATETYPE_LEAVE_RAIDBROWSER));
873  SendLfgUpdateParty(guid, LfgUpdateData(LFG_UPDATETYPE_LEAVE_RAIDBROWSER));
874  break;
875  }

984  {
