305  void EnterPhase(uint32 NextPhase)
306  {
307  switch (NextPhase)
308  {
309  case 0:
310  break;
311  case 1:
312  case 2:
313  case 3:
314  case 4:
315  me->NearTeleportTo(CENTER_X, CENTER_Y, CENTER_Z, me->GetOrientation());
316  DoResetThreat();
317  me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID, 0);
318  me->RemoveAurasDueToSpell(Transform[Phase].unaura);
319  DoCast(me, Transform[Phase].spell);
320  Talk(Transform[Phase].text);
321  if (Phase > 0)
322  {
323  if (Unit* Temp = ObjectAccessor::GetUnit(*me, SpiritGUID[Phase - 1]))
324  Temp->SetUInt32Value(UNIT_FIELD_BYTES_1, UNIT_STAND_STATE_DEAD);
325  }
326  if (Unit* Temp = ObjectAccessor::GetUnit(*me, SpiritGUID[NextPhase - 1]))
327  Temp->CastSpell(me, SPELL_SIPHON_SOUL, false); // should m cast on temp
328  if (NextPhase == 2)
329  {
330  me->GetMotionMaster()->Clear();
331  DoCast(me, SPELL_ENERGY_STORM, true); // enemy aura
332  for (uint8 i = 0; i < 4; ++i)
333  {
334  Creature* Vortex = DoSpawnCreature(CREATURE_FEATHER_VORTEX, 0, 0, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
335  if (Vortex)
336  {
337  Vortex->CastSpell(Vortex, SPELL_CYCLONE_PASSIVE, true);
338  Vortex->CastSpell(Vortex, SPELL_CYCLONE_VISUAL, true);
339  Vortex->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
340  Vortex->SetSpeed(MOVE_RUN, 1.0f);
341  Vortex->AI()->AttackStart(SelectTarget(SELECT_TARGET_RANDOM, 0));
342  DoZoneInCombat(Vortex);
343  }
344  }
345  }
346  else
347  AttackStart(me->GetVictim());
348  if (NextPhase == 3)
349  {
350  me->RemoveAurasDueToSpell(SPELL_ENERGY_STORM);
351  Summons.DespawnEntry(CREATURE_FEATHER_VORTEX);
352  me->GetMotionMaster()->MoveChase(me->GetVictim());
353  }
354  break;
355  default:
356  break;
357  }
358  Phase = NextPhase;

361  void UpdateAI(uint32 diff) override
362  {
363  if (!TankGUID)
364  {
365  if (!UpdateVictim())
366  return;
367 
368  if (me->GetHealth() < health_20 * (4 - Phase))
369  EnterPhase(Phase + 1);
370  }
371 
372  if (Berserk_Timer <= diff)
373  {
374  DoCast(me, SPELL_BERSERK, true);
375  Talk(YELL_BERSERK);
376  Berserk_Timer = 60000;
377  }
378  else Berserk_Timer -= diff;
379 
380  switch (Phase)
381  {
382  case 0:
383  if (Intro_Timer)
384  {
385  if (Intro_Timer <= diff)
386  {
387  Talk(YELL_AGGRO);
388  Intro_Timer = 0;
389  }
390  else Intro_Timer -= diff;
391  }
392 
393  if (Whirlwind_Timer <= diff)
394  {
395  DoCast(me, SPELL_WHIRLWIND);
396  Whirlwind_Timer = urand(15000, 20000);
397  }
398  else Whirlwind_Timer -= diff;
399

401  {
402  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
403  DoCast(target, SPELL_GRIEVOUS_THROW, false);
404  Grievous_Throw_Timer = 10000;
405  }
406  else Grievous_Throw_Timer -= diff;
407  break;
408 
409  case 1:

412  DoCast(me, SPELL_CREEPING_PARALYSIS);
413  Creeping_Paralysis_Timer = 20000;
414  }
415  else Creeping_Paralysis_Timer -= diff;
416

421  }
422  else Overpower_Timer -= diff;
423  break;
