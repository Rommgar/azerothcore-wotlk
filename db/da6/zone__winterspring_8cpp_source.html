154 class DialogueHelper
155 {
156 public:
157  // The array MUST be terminated by {0, 0, 0}
158  DialogueHelper(DialogueEntry const* dialogueArray) :
159  _dialogueArray(dialogueArray),
160  _currentEntry(nullptr),
161  _actionTimer(0)
162  { }
163  // The array MUST be terminated by {0, 0, 0, 0, 0}
164 
167 
168  void StartNextDialogueText(int32 textEntry)
169  {
170  // Find textEntry
171  bool found = false;
172 
173  for (DialogueEntry const* entry = _dialogueArray; entry->TextEntry; ++entry)
174  {
175  if (entry->TextEntry == textEntry)
176  {
177  _currentEntry = entry;
178  found = true;
179  break;
180  }
181  }
182 
183  if (!found)
184  return;
185 
186  DoNextDialogueStep();
187  }
188 
189  void DialogueUpdate(uint32 diff)
190  {
191  if (_actionTimer)
192  {
193  if (_actionTimer <= diff)
194  DoNextDialogueStep();
195  else
196  _actionTimer -= diff;
197  }
198  }
199 
200 protected:
202  virtual void JustDidDialogueStep(int32 /*entry*/) { }
204  virtual Creature* GetSpeakerByEntry(int32 /*entry*/) { return nullptr; }
205 
206 private:
207  void DoNextDialogueStep()
208  {
209  // Last Dialogue Entry done?
210  if (!_currentEntry || !_currentEntry->TextEntry)
211  {
212  _actionTimer = 0;
213  return;
214  }
215 
216  // Get Text, SpeakerEntry and Timer
217  int32 textEntry = _currentEntry->TextEntry;
218  uint32 sayerEntry = _currentEntry->SayerEntry;
219  _actionTimer = _currentEntry->SayTimer;
220 
221  // Simulate Case
222  if (sayerEntry && textEntry >= 0)
223  {
224  // Use Speaker if directly provided
225  if (Creature* speaker = GetSpeakerByEntry(sayerEntry))
226  speaker->AI()->Talk(textEntry);
227  }
228 
229  JustDidDialogueStep(_currentEntry->TextEntry);
230 
231  // Increment position
232  ++_currentEntry;
233  }
234 
235  DialogueEntry const* _dialogueArray;
236  DialogueEntry const* _currentEntry;
237 
238  uint32 _actionTimer;
239 };
