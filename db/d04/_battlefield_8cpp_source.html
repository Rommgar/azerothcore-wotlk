74 void Battlefield::HandlePlayerEnterZone(Player* player, uint32 /*zone*/)
75 {
76  // Xinef: do not invite players on taxi
77  if (!player->IsInFlight())
78  {
79  // If battle is started,
80  // If not full of players > invite player to join the war

82  if (IsWarTime())
83  {
84  if (m_PlayersInWar[player->GetTeamId()].size() + m_InvitedPlayers[player->GetTeamId()].size() < m_MaxPlayer) // Vacant spaces
85  InvitePlayerToWar(player);
86  else // No more vacant places
87  {
88  // TODO: Send a packet to announce it to player
89  m_PlayersWillBeKick[player->GetTeamId()][player->GetGUID()] = time(nullptr) + (player->IsGameMaster() ? 30 * MINUTE : 10);
90  InvitePlayerToQueue(player);
91  }
92  }
93  else
94  {
95  // If time left is < 15 minutes invite player to join queue
96  if (m_Timer <= m_StartGroupingTimer)
97  InvitePlayerToQueue(player);
98  }
99  }
100 
101  // Add player in the list of player in zone
102  m_players[player->GetTeamId()].insert(player->GetGUID());
103  OnPlayerEnterZone(player);

135 bool Battlefield::Update(uint32 diff)
136 {
137  if (m_Timer <= diff)
138  {
139  if (!IsEnabled() || (!IsWarTime() && sWorld->GetActiveSessionCount() > 3500)) // if WG is disabled or there is more than 3500 connections, switch automaticly
140  {
141  m_isActive = true;
142  EndBattle(false);
143  return false;
144  }
145  // Battlefield ends on time
146  if (IsWarTime())
147  EndBattle(true);
148  else // Time to start a new battle!
149  StartBattle();
150  }
151  else
152  m_Timer -= diff;
153 
154  if (!IsEnabled())
155  return false;
156 
157  // Invite players a few minutes before the battle's beginning
158  if (!IsWarTime() && !m_StartGrouping && m_Timer <= m_StartGroupingTimer)
159  {
160  m_StartGrouping = true;
161  InvitePlayersInZoneToQueue();
162  OnStartGrouping();
163  }
164 
165  bool objective_changed = false;
166  if (IsWarTime())
167  {
168  if (m_uiKickAfkPlayersTimer <= diff)
169  {
170  m_uiKickAfkPlayersTimer = 20000;
171  KickAfkPlayers();
172  }
173  else
174  m_uiKickAfkPlayersTimer -= diff;
175 
176  // Kick players who chose not to accept invitation to the battle
177  if (m_uiKickDontAcceptTimer <= diff)
178  {
179  time_t now = time(nullptr);
180  for (int team = 0; team < 2; team++)
181  for (PlayerTimerMap::iterator itr = m_InvitedPlayers[team].begin(); itr != m_InvitedPlayers[team].end(); ++itr)
182  if (itr->second <= now)
183  KickPlayerFromBattlefield(itr->first);
184 
185  InvitePlayersInZoneToWar();
186  for (int team = 0; team < 2; team++)
187  for (PlayerTimerMap::iterator itr = m_PlayersWillBeKick[team].begin(); itr != m_PlayersWillBeKick[team].end(); ++itr)
188  if (itr->second <= now)
189  KickPlayerFromBattlefield(itr->first);
190 
191  m_uiKickDontAcceptTimer = 5000;
192  }
193  else
194  m_uiKickDontAcceptTimer -= diff;
195 
196  for (BfCapturePointVector::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
197  if ((*itr)->Update(diff))
198  objective_changed = true;
199  }
200 
201  if (m_LastResurectTimer <= diff)
202  {
203  for (uint8 i = 0; i < m_GraveyardList.size(); i++)
204  if (GetGraveyardById(i))
205  m_GraveyardList[i]->Resurrect();
206  m_LastResurectTimer = RESURRECTION_INTERVAL;
207  }
208  else
209  m_LastResurectTimer -= diff;
210 
211  return objective_changed;
212 }

270 {

285  if (player->getLevel() < m_MinLevel)
286  {

288  m_PlayersWillBeKick[player->GetTeamId()][player->GetGUID()] = time(nullptr) + 10;

297  m_InvitedPlayers[player->GetTeamId()][player->GetGUID()] = time(nullptr) + m_TimeForAcceptInvite;
