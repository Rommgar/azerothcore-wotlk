202  void SetData(uint32 type, uint32 data) override
203  {
204  switch(type)
205  {
206  case DATA_ACTIVATE_DEFENSE_SYSTEM:
207  {
208  if (data)
209  bDefensesUsed = true;
210  const Position pos = {1919.09546f, 812.29724f, 86.2905f, M_PI};
211  instance->SummonCreature(NPC_DEFENSE_SYSTEM, pos, 0, 6499);
212  }
213  break;
214  case DATA_START_INSTANCE:
215  if (EncounterStatus == NOT_STARTED)
216  {
217  EncounterStatus = IN_PROGRESS;
218  if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
219  c->AI()->Talk(SAY_SINCLARI_1);
220  events.RescheduleEvent(EVENT_GUARDS_FALL_BACK, 4000);
221  }
222  break;
223  case DATA_PORTAL_DEFEATED:
224  events.RescheduleEvent(EVENT_SUMMON_PORTAL, 3000);
225  break;
226  case DATA_PORTAL_LOCATION:
227  PortalLocation = data;
228  break;
229  case DATA_DECRASE_DOOR_HEALTH:
230  if (GateHealth > 0)
231  --GateHealth;
232  if (GateHealth == 0)
233  {
234  CLEANED = false;
235  InstanceCleanup();
236  }
237  DoUpdateWorldState(WORLD_STATE_VH_PRISON_STATE, (uint32)GateHealth);
238  break;
239  case DATA_RELEASE_BOSS:
240  if (WaveCount == 6)
241  StartBossEncounter(uiFirstBoss);
242  else
243  StartBossEncounter(uiSecondBoss);
244  break;
245  case DATA_BOSS_DIED:
246  if (WaveCount == 6)
247  m_auiEncounter[0] = DONE;
248  else if (WaveCount == 12)
249  m_auiEncounter[1] = DONE;
250  else if (WaveCount == 18)
251  {
252  m_auiEncounter[2] = DONE;
253  EncounterStatus = DONE;
254  HandleGameObject(GO_MainGateGUID, true);
255  DoUpdateWorldState(WORLD_STATE_VH_SHOW, 0);
256  if (Creature* c = instance->GetCreature(NPC_SinclariGUID)) { c->DespawnOrUnsummon(); c->SetRespawnTime(3); }
257  }
258  SaveToDB();
259  if (WaveCount < 18)
260  events.RescheduleEvent(EVENT_SUMMON_PORTAL, 35000);
261  break;
262  case DATA_FAILED:
263  CLEANED = false;
264  InstanceCleanup();
265  break;
266  case DATA_ACHIEV:
267  bAchiev = !!data;
268  break;
269  }

414  c->CombatStop();

416  c->GetMotionMaster()->MovePoint(0, guardMovePosition);
417  }
418  events.RescheduleEvent(EVENT_GUARDS_DISAPPEAR, 5000);
419  }
420  break;
421  case EVENT_GUARDS_DISAPPEAR:
422  {
423  for (uint8 i = 0; i < 4; ++i)
424  if (Creature* c = instance->GetCreature(NPC_GuardGUID[i]))
425  c->SetVisible(false);
426  events.RescheduleEvent(EVENT_SINCLARI_FALL_BACK, 2000);
427  }
428  break;
429  case EVENT_SINCLARI_FALL_BACK:
430  {
431  if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
432  {
433  c->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
434  c->GetMotionMaster()->MovePoint(0, sinclariOutsidePosition);
435  }
436  SetData(DATA_ACTIVATE_DEFENSE_SYSTEM, 0);
437  events.RescheduleEvent(EVENT_START_ENCOUNTER, 4000);
438  }
439  break;
440  case EVENT_START_ENCOUNTER:
441  {
442  if (Creature* c = instance->GetCreature(NPC_DoorSealGUID))
443  c->RemoveAllAuras(); // just to be sure...
444  GateHealth = 100;
445  HandleGameObject(GO_MainGateGUID, false);
446  DoUpdateWorldState(WORLD_STATE_VH_SHOW, 1);
447  DoUpdateWorldState(WORLD_STATE_VH_PRISON_STATE, (uint32)GateHealth);
448  DoUpdateWorldState(WORLD_STATE_VH_WAVE_COUNT, (uint32)WaveCount);
449 
450  for (ObjectGuid const& guid : GO_ActivationCrystalGUID)
451  if (GameObject* go = instance->GetGameObject(guid))
452  {
453  HandleGameObject(ObjectGuid::Empty, false, go); // not used yet
454  go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE); // make it useable
455  }
456  events.RescheduleEvent(EVENT_SUMMON_PORTAL, 4000);
457  }
458  break;
459  case EVENT_SUMMON_PORTAL:
460  ++WaveCount;
461  DoUpdateWorldState(WORLD_STATE_VH_WAVE_COUNT, (uint32)WaveCount);
462  SetData(DATA_PORTAL_LOCATION, (GetData(DATA_PORTAL_LOCATION) + urand(1, 5)) % 6);
463  if (Creature* c = instance->GetCreature(NPC_SinclariGUID))
464  {
465  if (WaveCount % 6 != 0)
466  c->SummonCreature(NPC_TELEPORTATION_PORTAL, PortalLocations[GetData(DATA_PORTAL_LOCATION)], TEMPSUMMON_CORPSE_DESPAWN);
467  else if (WaveCount == 6 || WaveCount == 12) // first or second boss
468  {
469  if (!uiFirstBoss || !uiSecondBoss)
470  {
471  uiFirstBoss = urand(1, 6);
472  do { uiSecondBoss = urand(1, 6); }
473  while (uiFirstBoss == uiSecondBoss);
474  SaveToDB();
475  }
476  c->SummonCreature(NPC_TELEPORTATION_PORTAL, MiddleRoomPortalSaboLocation, TEMPSUMMON_CORPSE_DESPAWN);
477  }
478  else // cyanigossa

485  }

487  }
488  }
489  break;
490  case EVENT_CYANIGOSSA_TRANSFORM:
491  if (Creature* c = instance->GetCreature(NPC_CyanigosaGUID))
492  {
493  c->RemoveAurasDueToSpell(SPELL_CYANIGOSA_BLUE_AURA);

522  bool DoNeedCleanup(bool enter)
523  {
524  uint8 aliveCount = 0;
525  Map::PlayerList const& pl = instance->GetPlayers();
526  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
527  if( Player* plr = itr->GetSource() )
528  if( plr->IsAlive() && !plr->IsGameMaster() && !plr->HasAura(27827)/*spirit of redemption aura*/ )
529  ++aliveCount;
530 
531  bool need = enter ? aliveCount <= 1 : aliveCount == 0;
532  if( !need && CLEANED )
533  CLEANED = false;
534  return need;
535  }

550

553  for (uint8 i = 0; i < 4; ++i)
554  if (Creature* c = instance->GetCreature(NPC_GuardGUID[i]))
555  {
556  c->DespawnOrUnsummon();
557  c->SetRespawnTime(3);
558  if (m_auiEncounter[MAX_ENCOUNTER - 1] == DONE)
559  c->SetVisible(false);

562  c->SetReactState(REACT_AGGRESSIVE);

564

567  c->DespawnOrUnsummon();
568  NPC_PortalGUID.Clear();
569 
570  // remove trash

572  if (Creature* c = instance->GetCreature(guid))
573  c->DespawnOrUnsummon();
574 
575  trashMobs.clear();
576 
577  // clear door seal damaging auras:

579  c->RemoveAllAuras();
580 
581  // open main gate
582  HandleGameObject(GO_MainGateGUID, true);
583 
584  if (m_auiEncounter[MAX_ENCOUNTER - 1] != DONE) // instance not finished
585  {
586  // close all cells
