230  void HandlePeriodic(AuraEffect const* aurEff)
231  {
232  PreventDefaultAction();
233  if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
234  {
235  int32 currentAmount = aurEff->GetAmount();
236  int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
237  if (newAmount != currentAmount)
238  {
239  if (aurEff->GetMiscValue() == STAT_STAMINA)
240  {
241  uint32 actStat = GetUnitOwner()->GetHealth();
242  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
243  GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
244  }
245  else
246  {
247  uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
248  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
249  GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
250  }
251  }
252  }
253  else
254  GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
255  }

414 
415  void HandleScriptEffect(SpellEffIndex /*effIndex*/)
416  {
417  Unit* caster = GetCaster();
418  if (Unit* unitTarget = GetHitUnit())
419  {
420  uint32 spellId = 0;
421  int32 basePoint = 0;
422  Unit::AuraApplicationMap& Auras = unitTarget->GetAppliedAuras();
423  for (Unit::AuraApplicationMap::iterator i = Auras.begin(); i != Auras.end(); ++i)
424  {
425  Aura* aura = i->second->GetBase();
426  if (aura->GetCasterGUID() != caster->GetGUID())
427  continue;
428 
429  // Search only Serpent Sting, Viper Sting, Scorpid Sting auras
430  flag96 familyFlag = aura->GetSpellInfo()->SpellFamilyFlags;
431  if (!(familyFlag[1] & 0x00000080 || familyFlag[0] & 0x0000C000))
432  continue;
433  if (AuraEffect* aurEff = aura->GetEffect(0))
434  {
435  // Serpent Sting - Instantly deals 40% of the damage done by your Serpent Sting.
436  if (familyFlag[0] & 0x4000)
437  {
438  int32 TickCount = aurEff->GetTotalTicks();
439  spellId = SPELL_HUNTER_CHIMERA_SHOT_SERPENT;
440  basePoint = aurEff->GetAmount();
441  ApplyPct(basePoint, TickCount * 40);
442  basePoint = unitTarget->SpellDamageBonusTaken(caster, aura->GetSpellInfo(), basePoint, DOT, aura->GetStackAmount());
443  }
444  // Viper Sting - Instantly restores mana to you equal to 60% of the total amount drained by your Viper Sting.
445  else if (familyFlag[1] & 0x00000080)
446  {
447  int32 TickCount = aura->GetEffect(0)->GetTotalTicks();
448  spellId = SPELL_HUNTER_CHIMERA_SHOT_VIPER;
449 
450  // Amount of one aura tick
451  basePoint = int32(CalculatePct(unitTarget->GetMaxPower(POWER_MANA), aurEff->GetAmount()));
452  int32 casterBasePoint = aurEff->GetAmount() * unitTarget->GetMaxPower(POWER_MANA) / 50; // TODO: WTF? caster uses unitTarget?
453  if (basePoint > casterBasePoint)
454  basePoint = casterBasePoint;
455  ApplyPct(basePoint, TickCount * 60);
456  }
457  // Scorpid Sting - Attempts to Disarm the target for 10 sec. This effect cannot occur more than once per 1 minute.
458  else if (familyFlag[0] & 0x00008000)
459  {
460  if (caster->ToPlayer()) // Scorpid Sting - Add 1 minute cooldown
461  {
462  if (caster->ToPlayer()->HasSpellCooldown(SPELL_HUNTER_CHIMERA_SHOT_SCORPID))
463  break;
464 
465  caster->ToPlayer()->AddSpellCooldown(SPELL_HUNTER_CHIMERA_SHOT_SCORPID, 0, 60000);
466  }
467 
468  spellId = SPELL_HUNTER_CHIMERA_SHOT_SCORPID;
469  }
470 
471  // Refresh aura duration
472  aura->RefreshDuration();
473  aurEff->ChangeAmount(aurEff->CalculateAmount(caster), false);
474  }
475  break;
476  }
477 
478  if (spellId)
479  caster->CastCustomSpell(unitTarget, spellId, &basePoint, 0, 0, true);
480  }
481  }

484  {
