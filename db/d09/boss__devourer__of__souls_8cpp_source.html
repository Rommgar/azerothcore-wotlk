327  void HandlePeriodicTick(AuraEffect const* aurEff)
328  {
329  PreventDefaultAction();
330  if (Unit* t = GetTarget())
331  {
332  if (aurEff->GetTickNumber() < 30)
333  {
334  // spinning, casting, etc.
335  float diff = (2 * M_PI) / (4 * 30);
336  float new_o = t->GetOrientation() + diff * dir;
337  if (new_o >= 2 * M_PI)
338  new_o -= 2 * M_PI;
339  else if (new_o < 0)
340  new_o += 2 * M_PI;
341  t->UpdateOrientation(new_o);
342  t->SetFacingTo(new_o);
343  t->CastSpell(t, SPELL_WAILING_SOULS_DMG_N, true);
344  }
345  else if (aurEff->GetTickNumber() == 33)
346  {
347  t->SetControlled(false, UNIT_STATE_ROOT);
348  t->DisableRotate(false);
349  if (t->GetTypeId() == TYPEID_UNIT)
350  t->ToCreature()->SetReactState(REACT_AGGRESSIVE);
351  if (t->GetVictim())
352  {
353  t->SetGuidValue(UNIT_FIELD_TARGET, t->GetVictim()->GetGUID());
354  t->GetMotionMaster()->MoveChase(t->GetVictim());
355  }
356  }
357  else if (aurEff->GetTickNumber() >= 34)
358  Remove(AURA_REMOVE_BY_EXPIRE);
359  }
360  }
361 
362  void Register() override
363  {
