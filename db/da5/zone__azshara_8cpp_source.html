73  {
74  if (spellhit)
75  return;
76 
77  switch (spell->Id)
78  {
79  case SPELL_POLYMORPH_RANK1:
80  case SPELL_POLYMORPH_RANK2:
81  case SPELL_POLYMORPH_RANK3:
82  case SPELL_POLYMORPH_RANK4:
83  if (Player* player = unit->ToPlayer())
84  if (player->GetQuestStatus(9364) == QUEST_STATUS_INCOMPLETE)
85  {
86  spellhit = true;

88  }
89  break;
90  default:
91  break;
92  }

95  void UpdateAI(uint32 diff) override
96  {
97  // we mustn't remove the Creature in the same round in which we cast the summon spell, otherwise there will be no summons
98  if (spellhit && morphtimer >= 5000)
99  {
100  me->DespawnOrUnsummon();
101  return;
102  }
103  // walk 5 seconds before summoning
104  if (spellhit && morphtimer < 5000)
105  {
106  morphtimer += diff;
107  if (morphtimer >= 5000)
108  {
109  DoCast(me, SPELL_POLYMORPH_BACKFIRE); // summon copies
110  DoCast(me, SPELL_REMOVE_POLYMORPH); // visual explosion
111  }
112  }
113  if (!UpdateVictim())
114  return;
115 
117  DoMeleeAttackIfReady();

281  void UpdateAI(uint32 diff) override
282  {
283  if (MustDie)
284  {
285  if (MustDieTimer <= diff)
286  {
287  me->DespawnOrUnsummon();
288  return;

290  else MustDieTimer -= diff;
291  }
292 
293  if (!Escape)
294  {
295  if (!PlayerGUID)
296  return;
297 
298  if (SpellEscapeTimer <= diff)
299  {
300  DoCast(me, SPELL_RIZZLE_ESCAPE, false);
301  SpellEscapeTimer = 10000;
302  }
303  else SpellEscapeTimer -= diff;
304 
305  if (TeleportTimer <= diff)
306  {
307  // temp solution - unit can't be teleported by core using spelleffect 5, only players
308  me->NearTeleportTo(3706.39f, -3969.15f, 35.9118f, me->GetOrientation());
309 
310  //begin swimming and summon depth charges
311  Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
312  if (!player)
313  return;
314 
315  Talk(MSG_ESCAPE_NOTICE, player);
316  DoCast(me, SPELL_PERIODIC_DEPTH_CHARGE);
317  me->SetHover(true);
318  me->SetSwim(true);
319  me->SetSpeed(MOVE_RUN, 0.85f, true);
320  me->GetMotionMaster()->MovementExpired();
321  me->GetMotionMaster()->MovePoint(CurrWP, WPs[CurrWP]);
322  Escape = true;
323  }
324  else TeleportTimer -= diff;
325 
326  return;
327  }
328 
329  if (ContinueWP)
330  {
331  me->GetMotionMaster()->MovePoint(CurrWP, WPs[CurrWP]);
332  ContinueWP = false;
333  }
334 
335  if (GrenadeTimer <= diff)
336  {
337  if (Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID))
338  {
339  Talk(SAY_RIZZLE_GRENADE, player);
340  DoCast(player, SPELL_RIZZLE_FROST_GRENADE, true);
341  }
342  GrenadeTimer = 30000;
343  }
344  else GrenadeTimer -= diff;
345 
346  if (CheckTimer <= diff)
347  {
348  Player* player = ObjectAccessor::GetPlayer(*me, PlayerGUID);
349  if (!player)

351  me->DespawnOrUnsummon();
352  return;
