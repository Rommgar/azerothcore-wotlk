33 Tokenizer::Tokenizer(const std::string& src, const char sep, uint32 vectorReserve)
34 {
35  m_str = new char[src.length() + 1];
36  memcpy(m_str, src.c_str(), src.length() + 1);
37 
38  if (vectorReserve)
39  {
40  m_storage.reserve(vectorReserve);
41  }
42 
43  char* posold = m_str;
44  char* posnew = m_str;
45 
46  for (;;)
47  {
48  if (*posnew == sep)
49  {
50  m_storage.push_back(posold);
51  posold = posnew + 1;
52 
53  *posnew = '\0';
54  }
55  else if (*posnew == '\0')
56  {
57  // Hack like, but the old code accepted these kind of broken strings,
58  // so changing it would break other things
59  if (posold != posnew)
60  {
61  m_storage.push_back(posold);
62  }
63 
64  break;
65  }
66 
67  ++posnew;
68  }
69 }

188 int32 MoneyStringToMoney(const std::string& moneyString)
189 {
190  int32 money = 0;
191 
192  if (!(std::count(moneyString.begin(), moneyString.end(), 'g') == 1 ||
193  std::count(moneyString.begin(), moneyString.end(), 's') == 1 ||
194  std::count(moneyString.begin(), moneyString.end(), 'c') == 1))
195  {
196  return 0; // Bad format
197  }
198 
199  Tokenizer tokens(moneyString, ' ');
200  for (Tokenizer::const_iterator itr = tokens.begin(); itr != tokens.end(); ++itr)
201  {
202  std::string tokenString(*itr);
203  size_t gCount = std::count(tokenString.begin(), tokenString.end(), 'g');
204  size_t sCount = std::count(tokenString.begin(), tokenString.end(), 's');
205  size_t cCount = std::count(tokenString.begin(), tokenString.end(), 'c');
206  if (gCount + sCount + cCount != 1)
207  {
208  return 0;
209  }
210 
211  uint32 amount = atoi(*itr);
212  if (gCount == 1)
213  {
214  money += amount * 100 * 100;
215  }
216  else if (sCount == 1)
217  {
218  money += amount * 100;
219  }
220  else if (cCount == 1)
221  {
222  money += amount;
223  }
224  }
225 
226  return money;
227 }

229 uint32 TimeStringToSecs(const std::string& timestring)
230 {
231  uint32 secs = 0;
232  uint32 buffer = 0;
233  uint32 multiplier = 0;
234 
235  for (std::string::const_iterator itr = timestring.begin(); itr != timestring.end(); ++itr)
236  {
237  if (isdigit(*itr))
238  {
239  buffer *= 10;
240  buffer += (*itr) - '0';
241  }
242  else
243  {
244  switch (*itr)
245  {
246  case 'd':
247  multiplier = DAY;
248  break;
249  case 'h':
250  multiplier = HOUR;
251  break;
252  case 'm':
253  multiplier = MINUTE;
254  break;
255  case 's':
256  multiplier = 1;
257  break;
258  default :
259  return 0; //bad format
260  }
261  buffer *= multiplier;
262  secs += buffer;
263  buffer = 0;
264  }
265  }
266 
267  return secs;

382  out = utf8::utf8to16(utf8str, utf8str + csize, out);
383  wsize -= out.remaining(); // remaining unused space

392  if (wsize >= errorMessageLength)

395  wsize = std::char_traits<wchar_t>::length(wstr);
396  }
397  else if (wsize > 0)
398  {
399  wstr[0] = L'\0';
400  wsize = 0;
401  }
402  else
403  {
404  wsize = 0;
405  }
406 
407  return false;
408  }
409 
410  return true;
