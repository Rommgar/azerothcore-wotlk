98 void MailDraft::prepareItems(Player* receiver, CharacterDatabaseTransaction trans)
99 {
100  if (!m_mailTemplateId || !m_mailTemplateItemsNeed)
101  return;
102 
103  m_mailTemplateItemsNeed = false;
104 
105  Loot mailLoot;
106 
107  // can be empty
108  mailLoot.FillLoot(m_mailTemplateId, LootTemplates_Mail, receiver, true, true);
109 
110  uint32 max_slot = mailLoot.GetMaxSlotInLootFor(receiver);
111  for (uint32 i = 0; m_items.size() < MAX_MAIL_ITEMS && i < max_slot; ++i)
112  {
113  if (LootItem* lootitem = mailLoot.LootItemInSlot(i, receiver))
114  {
115  if (Item* item = Item::CreateItem(lootitem->itemid, lootitem->count, receiver))
116  {
117  item->SaveToDB(trans); // save for prevent lost at next mail load, if send fail then item will deleted
118  AddItem(item);
119  }
120  }
121  }

143 void MailDraft::SendReturnToSender(uint32 /*sender_acc*/, ObjectGuid::LowType sender_guid, ObjectGuid::LowType receiver_guid, CharacterDatabaseTransaction trans)
144 {
145  Player* receiver = ObjectAccessor::FindPlayerByLowGUID(receiver_guid);
146 
147  uint32 rc_account = 0;
148  if (!receiver)
149  rc_account = sObjectMgr->GetPlayerAccountIdByGUID(receiver_guid);
150 
151  if (!receiver && !rc_account) // sender not exist
152  {
153  deleteIncludedItems(trans, true);
154  return;
155  }
156 
157  if (!m_items.empty())
158  {
159  // if item send to character at another account, then apply item delivery delay
160  //needItemDelay = sender_acc != rc_account;
161 
162  // set owner to new receiver (to prevent delete item with sender char deleting)
163  for (MailItemMap::iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
164  {
165  Item* item = mailItemIter->second;
166  item->SaveToDB(trans); // item not in inventory and can be save standalone
167  // owner in data will set at mail receive and item extracting
168  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ITEM_OWNER);
169  stmt->setUInt32(0, receiver_guid);
170  stmt->setUInt32(1, item->GetGUID().GetCounter());
171  trans->Append(stmt);
172  }
173  }
174 
175  // xinef: WowWiki: "Return mail arrives immediately."
176 
177  // will delete item or place to receiver mail list
178  SendMailTo(trans, MailReceiver(receiver, receiver_guid), MailSender(MAIL_NORMAL, sender_guid), MAIL_CHECK_MASK_RETURNED, 0);
179 }

181 void MailDraft::SendMailTo(CharacterDatabaseTransaction trans, MailReceiver const& receiver, MailSender const& sender, MailCheckMask checked, uint32 deliver_delay, uint32 custom_expiration, bool deleteMailItemsFromDB, bool sendMail, int32 auctionId)
182 {
183  sScriptMgr->OnBeforeMailDraftSendMailTo(this, receiver, sender, checked, deliver_delay, custom_expiration, deleteMailItemsFromDB, sendMail);
184 
185  if (deleteMailItemsFromDB) // can be changed in the hook
186  deleteIncludedItems(trans, true);
187 
188  if (!sendMail) // can be changed in the hook
189  return;
190 
191  Player* pReceiver = receiver.GetPlayer(); // can be nullptr
192  Player* pSender = ObjectAccessor::FindPlayerByLowGUID(sender.GetSenderId());
193 
194  if (pReceiver)
195  prepareItems(pReceiver, trans); // generate mail template items
196 
197  uint32 mailId = sObjectMgr->GenerateMailID();
198 
199  time_t deliver_time = time(nullptr) + deliver_delay;
200 
201  //expire time if COD 3 days, if no COD 30 days, if auction sale pending 1 hour
202  uint32 expire_delay;
203 
204  // auction mail without any items and money
205  if (sender.GetMailMessageType() == MAIL_AUCTION && m_items.empty() && !m_money)
206  expire_delay = sWorld->getIntConfig(CONFIG_MAIL_DELIVERY_DELAY);
207  // mail from battlemaster (rewardmarks) should last only one day
208  else if (sender.GetMailMessageType() == MAIL_CREATURE && sBattlegroundMgr->GetBattleMasterBG(sender.GetSenderId()) != BATTLEGROUND_TYPE_NONE)
209  expire_delay = DAY;
210  // default case: expire time if COD 3 days, if no COD 30 days (or 90 days if sender is a game master)
211  else
212  {
213  if (m_COD)
214  expire_delay = 3 * DAY;
215  else if (custom_expiration > 0 )
216  expire_delay = custom_expiration * DAY;
217  else
218  expire_delay = pSender && pSender->GetSession()->GetSecurity() ? 90 * DAY : 30 * DAY;
219  }
220 
221  time_t expire_time = deliver_time + expire_delay;
222 
223  // Add to DB
224  uint8 index = 0;
225  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_MAIL);
226  stmt->setUInt32( index, mailId);
227  stmt->setUInt8 (++index, uint8(sender.GetMailMessageType()));
228  stmt->setInt8 (++index, int8(sender.GetStationery()));
229  stmt->setUInt16(++index, GetMailTemplateId());
230  stmt->setUInt32(++index, sender.GetSenderId());
231  stmt->setUInt32(++index, receiver.GetPlayerGUIDLow());
232  stmt->setString(++index, GetSubject());
233  stmt->setString(++index, GetBody());
234  stmt->setBool (++index, !m_items.empty());
235  stmt->setUInt32(++index, uint32(expire_time));
236  stmt->setUInt32(++index, uint32(deliver_time));
237  stmt->setUInt32(++index, m_money);
238  stmt->setUInt32(++index, m_COD);
239  stmt->setUInt8 (++index, uint8(checked));
240  stmt->setInt32(++index, auctionId);
241  trans->Append(stmt);
242 
243  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
244  {
245  Item* pItem = mailItemIter->second;
246  stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_MAIL_ITEM);
247  stmt->setUInt32(0, mailId);
248  stmt->setUInt32(1, pItem->GetGUID().GetCounter());
249  stmt->setUInt32(2, receiver.GetPlayerGUIDLow());
250  trans->Append(stmt);
251  }
252 
253  // xinef: update global data
254  sWorld->UpdateGlobalPlayerMails(receiver.GetPlayerGUIDLow(), 1);
255 
256  // For online receiver update in game mail status and data
257  if (pReceiver)
258  {
259  pReceiver->AddNewMailDeliverTime(deliver_time);
260 
261  Mail* m = new Mail;
262  m->messageID = mailId;
263  m->mailTemplateId = GetMailTemplateId();
264  m->subject = GetSubject();
265  m->body = GetBody();
266  m->money = GetMoney();
267  m->COD = GetCOD();
268 
269  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
270  {
271  Item* item = mailItemIter->second;
272  m->AddItem(item->GetGUID().GetCounter(), item->GetEntry());
273  }
274 
275  m->messageType = sender.GetMailMessageType();
276  m->stationery = sender.GetStationery();
277  m->sender = sender.GetSenderId();
278  m->receiver = receiver.GetPlayerGUIDLow();
279  m->expire_time = expire_time;
280  m->deliver_time = deliver_time;
281  m->checked = checked;
282  m->auctionId = auctionId;
283  m->state = MAIL_STATE_UNCHANGED;
284 
285  pReceiver->AddMail(m); // to insert new mail to beginning of maillist
286 
287  if (!m_items.empty())
288  {
289  for (MailItemMap::iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
290  {
291  pReceiver->AddMItem(mailItemIter->second);
292  }
293  }
294  }
295  else if (!m_items.empty())
296  {
297  CharacterDatabaseTransaction temp = CharacterDatabaseTransaction(nullptr);
298  deleteIncludedItems(temp);
299  }
300 }
