161  void DamageTaken(Unit*, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
162  {
163  // check if venoxis is ready to transform
164  if (!_transformed && !HealthAbovePct(50))
165  {
166  _transformed = true;
167  // schedule the event that changes our phase
168  events.ScheduleEvent(EVENT_TRANSFORM, 100);
169  }
170  // we're losing health, bad, go frenzy
171  else if (!_frenzied && !HealthAbovePct(20))
172  {
173  _frenzied = true;
174  events.ScheduleEvent(EVENT_FRENZY, 100);

178  void UpdateAI(uint32 diff) override
179  {
180  if (!UpdateVictim())
181  return;
182 
183  events.Update(diff);
184 
185  // return back to main code if we're still casting
186  if (me->HasUnitState(UNIT_STATE_CASTING))
187  return;
188 
189  while (uint32 eventId = events.ExecuteEvent())
190  {
191  switch (eventId)
192  {
193  // thrash is available in all phases
194  case EVENT_THRASH:
195  DoCast(me, SPELL_THRASH, true);
196  events.ScheduleEvent(EVENT_THRASH, urand(10000, 20000));
197  break;
198 
199  // troll form spells and Actions (first part)
200  case EVENT_DISPEL_MAGIC:
201  DoCast(me, SPELL_DISPEL_MAGIC);
202  events.ScheduleEvent(EVENT_DISPEL_MAGIC, urand(15000, 20000), 0, PHASE_ONE);
203  break;
204  case EVENT_RENEW:
205  DoCast(me, SPELL_RENEW);
206  events.ScheduleEvent(EVENT_RENEW, urand(25000, 30000), 0, PHASE_ONE);
207  break;
208  case EVENT_HOLY_NOVA:
209  _inMeleeRange = 0;
210 
211  for (uint8 i = 0; i < 10; ++i)
212  {
213  if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, i))
214  // check if target is within melee-distance
215  if (me->IsWithinMeleeRange(target))
216  ++_inMeleeRange;
217  }
218 
219  // trigger spellcast only if we have 3 or more targets to affect
220  if (_inMeleeRange >= 3)
221  DoCastVictim(SPELL_HOLY_NOVA);
222 
223  events.ScheduleEvent(EVENT_HOLY_NOVA, urand(45000, 75000), 0, PHASE_ONE);
224  break;
225  case EVENT_HOLY_FIRE:
226  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
227  DoCast(target, SPELL_HOLY_FIRE);
228  events.ScheduleEvent(EVENT_HOLY_FIRE, urand(45000, 60000), 0, PHASE_ONE);
229  break;
230  case EVENT_HOLY_WRATH:
231  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
232  DoCast(target, SPELL_HOLY_WRATH);
233  events.ScheduleEvent(EVENT_HOLY_WRATH, urand(45000, 60000), 0, PHASE_ONE);
234  break;
235 
236  //
237  // snake form spells and Actions
238  //
239 
240  case EVENT_VENOM_SPIT:
241  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
242  DoCast(target, SPELL_VENOM_SPIT);
243  events.ScheduleEvent(EVENT_VENOM_SPIT, urand(5000, 15000), 0, PHASE_TWO);
244  break;
245  case EVENT_POISON_CLOUD:
246  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
247  DoCast(target, SPELL_POISON_CLOUD);
248  events.ScheduleEvent(EVENT_POISON_CLOUD, urand(15000, 20000), 0, PHASE_TWO);
249  break;
250  case EVENT_PARASITIC_SERPENT:
251  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
252  DoCast(target, SPELL_SUMMON_PARASITIC_SERPENT);
253  events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 15000, 0, PHASE_TWO);
254  break;
255  case EVENT_FRENZY:
256  // frenzy at 20% health
257  DoCast(me, SPELL_FRENZY, true);
258  break;
259 
260  //
261  // shape and phase-changing
262  //
263 
264  case EVENT_TRANSFORM:
265  // shapeshift at 50% health
266  DoCast(me, SPELL_VENOXIS_TRANSFORM);
267  Talk(SAY_VENOXIS_TRANSFORM);
268  DoResetThreat();
269 
270  // phase two events (snakeform)
271  events.ScheduleEvent(EVENT_VENOM_SPIT, 5000, 0, PHASE_TWO);
272  events.ScheduleEvent(EVENT_POISON_CLOUD, 10000, 0, PHASE_TWO);
273  events.ScheduleEvent(EVENT_PARASITIC_SERPENT, 30000, 0, PHASE_TWO);
274 
275  // transformed, start phase two
276  events.SetPhase(PHASE_TWO);
277 
278  break;
279  default:
280  break;
281  }
282  }
283 
284  DoMeleeAttackIfReady();
285  }
286 
287  private:
288  uint8 _inMeleeRange;
289  bool _transformed;
290  bool _frenzied;
291  };
