48  void LoadDisables()
49  {
50  uint32 oldMSTime = getMSTime();
51 
52  // reload case
53  for (DisableMap::iterator itr = m_DisableMap.begin(); itr != m_DisableMap.end(); ++itr)
54  itr->second.clear();
55 
56  m_DisableMap.clear();
57 
58  QueryResult result = WorldDatabase.Query("SELECT sourceType, entry, flags, params_0, params_1 FROM disables");
59 
60  uint32 total_count = 0;
61 
62  if (!result)
63  {
64  LOG_INFO("server.loading", ">> Loaded 0 disables. DB table `disables` is empty!");
65  LOG_INFO("server.loading", " ");
66  return;
67  }
68 
69  Field* fields;
70  do
71  {
72  fields = result->Fetch();
73  DisableType type = DisableType(fields[0].GetUInt32());
74  if (type >= MAX_DISABLE_TYPES)
75  {
76  LOG_ERROR("sql.sql", "Invalid type %u specified in `disables` table, skipped.", type);
77  continue;
78  }
79 
80  uint32 entry = fields[1].GetUInt32();
81  uint8 flags = fields[2].GetUInt8();
82  std::string params_0 = fields[3].GetString();
83  std::string params_1 = fields[4].GetString();
84 
85  DisableData data;
86  data.flags = flags;
87 
88  switch (type)
89  {
90  case DISABLE_TYPE_GO_LOS:
91  if (!sObjectMgr->GetGameObjectTemplate(entry))
92  {
93  LOG_ERROR("sql.sql", "Gameobject entry %u from `disables` doesn't exist in dbc, skipped.", entry);
94  continue;
95  }
96  if (flags)
97  LOG_ERROR("sql.sql", "Disable flags specified for gameobject %u, useless data.", entry);
98  break;
99  case DISABLE_TYPE_SPELL:
100  if (!(sSpellMgr->GetSpellInfo(entry) || flags & SPELL_DISABLE_DEPRECATED_SPELL))
101  {
102  LOG_ERROR("sql.sql", "Spell entry %u from `disables` doesn't exist in dbc, skipped.", entry);
103  continue;
104  }
105 
106  if (!flags || flags > MAX_SPELL_DISABLE_TYPE)
107  {
108  LOG_ERROR("sql.sql", "Disable flags for spell %u are invalid, skipped.", entry);
109  continue;
110  }
111 
112  if (flags & SPELL_DISABLE_MAP)
113  {
114  Tokenizer tokens(params_0, ',');
115  for (uint8 i = 0; i < tokens.size(); )
116  data.params[0].insert(atoi(tokens[i++]));
117  }
118 
119  if (flags & SPELL_DISABLE_AREA)
120  {
121  Tokenizer tokens(params_1, ',');
122  for (uint8 i = 0; i < tokens.size(); )
123  data.params[1].insert(atoi(tokens[i++]));
124  }
125 
126  // xinef: if spell has disabled los, add flag
127  if (flags & SPELL_DISABLE_LOS)
128  {
129  SpellInfo* spellInfo = const_cast<SpellInfo*>(sSpellMgr->GetSpellInfo(entry));
130  spellInfo->AttributesEx2 |= SPELL_ATTR2_IGNORE_LINE_OF_SIGHT;
131  }
132 
133  break;
134  // checked later
135  case DISABLE_TYPE_QUEST:
136  break;
137  case DISABLE_TYPE_MAP:
138  case DISABLE_TYPE_LFG_MAP:
139  {
140  MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
141  if (!mapEntry)
142  {
143  LOG_ERROR("sql.sql", "Map entry %u from `disables` doesn't exist in dbc, skipped.", entry);
144  continue;
145  }
146  bool isFlagInvalid = false;
147  switch (mapEntry->map_type)
148  {
149  case MAP_COMMON:
150  if (flags)
151  isFlagInvalid = true;
152  break;
153  case MAP_INSTANCE:
154  case MAP_RAID:
155  if (flags & DUNGEON_STATUSFLAG_HEROIC && !GetMapDifficultyData(entry, DUNGEON_DIFFICULTY_HEROIC))
156  isFlagInvalid = true;
157  else if (flags & RAID_STATUSFLAG_10MAN_HEROIC && !GetMapDifficultyData(entry, RAID_DIFFICULTY_10MAN_HEROIC))
158  isFlagInvalid = true;
159  else if (flags & RAID_STATUSFLAG_25MAN_HEROIC && !GetMapDifficultyData(entry, RAID_DIFFICULTY_25MAN_HEROIC))
160  isFlagInvalid = true;
161  break;
162  case MAP_BATTLEGROUND:
163  case MAP_ARENA:
164  LOG_ERROR("sql.sql", "Battleground map %u specified to be disabled in map case, skipped.", entry);
165  continue;
166  }
167  if (isFlagInvalid)
168  {
169  LOG_ERROR("sql.sql", "Disable flags for map %u are invalid, skipped.", entry);
170  continue;
171  }
172  break;
173  }
174  case DISABLE_TYPE_BATTLEGROUND:
175  if (!sBattlemasterListStore.LookupEntry(entry))
176  {
177  LOG_ERROR("sql.sql", "Battleground entry %u from `disables` doesn't exist in dbc, skipped.", entry);
178  continue;
179  }
180  if (flags)
181  LOG_ERROR("sql.sql", "Disable flags specified for battleground %u, useless data.", entry);
182  break;
183  case DISABLE_TYPE_OUTDOORPVP:
184  if (entry > MAX_OUTDOORPVP_TYPES)
185  {
186  LOG_ERROR("sql.sql", "OutdoorPvPTypes value %u from `disables` is invalid, skipped.", entry);
187  continue;
188  }
189  if (flags)
190  LOG_ERROR("sql.sql", "Disable flags specified for outdoor PvP %u, useless data.", entry);
191  break;
192  case DISABLE_TYPE_ACHIEVEMENT_CRITERIA:
193  if (!sAchievementCriteriaStore.LookupEntry(entry))
194  {
195  LOG_ERROR("sql.sql", "Achievement Criteria entry %u from `disables` doesn't exist in dbc, skipped.", entry);
196  continue;
197  }
198  if (flags)
199  LOG_ERROR("sql.sql", "Disable flags specified for Achievement Criteria %u, useless data.", entry);
200  break;
201  case DISABLE_TYPE_VMAP:
202  {
203  MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
204  if (!mapEntry)
205  {
206  LOG_ERROR("sql.sql", "Map entry %u from `disables` doesn't exist in dbc, skipped.", entry);
207  continue;
208  }
209  switch (mapEntry->map_type)
210  {
211  case MAP_COMMON:
212  if (flags & VMAP::VMAP_DISABLE_AREAFLAG)
213  LOG_INFO("disable", "Areaflag disabled for world map %u.", entry);
214  if (flags & VMAP::VMAP_DISABLE_LIQUIDSTATUS)
215  LOG_INFO("disable", "Liquid status disabled for world map %u.", entry);
216  break;
217  case MAP_INSTANCE:
218  case MAP_RAID:
219  if (flags & VMAP::VMAP_DISABLE_HEIGHT)
220  LOG_INFO("disable", "Height disabled for instance map %u.", entry);
221  if (flags & VMAP::VMAP_DISABLE_LOS)
222  LOG_INFO("disable", "LoS disabled for instance map %u.", entry);
223  break;
224  case MAP_BATTLEGROUND:
225  if (flags & VMAP::VMAP_DISABLE_HEIGHT)
226  LOG_INFO("disable", "Height disabled for battleground map %u.", entry);
227  if (flags & VMAP::VMAP_DISABLE_LOS)
228  LOG_INFO("disable", "LoS disabled for battleground map %u.", entry);
229  break;
230  case MAP_ARENA:
231  if (flags & VMAP::VMAP_DISABLE_HEIGHT)
232  LOG_INFO("disable", "Height disabled for arena map %u.", entry);
233  if (flags & VMAP::VMAP_DISABLE_LOS)
234  LOG_INFO("disable", "LoS disabled for arena map %u.", entry);
235  break;
236  default:
237  break;
238  }
239  break;
240  }
241  case DISABLE_TYPE_GAME_EVENT:
242  {
243  GameEventMgr::ActiveEvents const& activeEvents = sGameEventMgr->GetActiveEventList();
244  if (activeEvents.find(entry) != activeEvents.end())
245  {
246  sGameEventMgr->StopEvent(entry);
247  LOG_INFO("disable", "Event entry %u was stopped because it has been disabled.", entry);
248  }
249  break;
250  }
251  default:
252  break;
253  }
254 
255  m_DisableMap[type].insert(DisableTypeMap::value_type(entry, data));
256  ++total_count;
257  } while (result->NextRow());
258 
259  LOG_INFO("server.loading", ">> Loaded %u disables in %u ms", total_count, GetMSTimeDiffToNow(oldMSTime));
260  LOG_INFO("server.loading", " ");
261  }

294  bool IsDisabledFor(DisableType type, uint32 entry, Unit const* unit, uint8 flags)
295  {
296  ASSERT(type < MAX_DISABLE_TYPES);
297  if (m_DisableMap[type].empty())
298  return false;
299 
300  DisableTypeMap::iterator itr = m_DisableMap[type].find(entry);
301  if (itr == m_DisableMap[type].end()) // not disabled
302  return false;
303 
304  switch (type)
305  {
306  case DISABLE_TYPE_SPELL:
307  {
308  uint8 spellFlags = itr->second.flags;
309  if (unit)
310  {
311  if ((spellFlags & SPELL_DISABLE_PLAYER && unit->GetTypeId() == TYPEID_PLAYER) ||
312  (unit->GetTypeId() == TYPEID_UNIT && ((unit->IsPet() && spellFlags & SPELL_DISABLE_PET) || spellFlags & SPELL_DISABLE_CREATURE)))
313  {
314  if (spellFlags & SPELL_DISABLE_MAP)
315  {
316  std::set<uint32> const& mapIds = itr->second.params[0];
317  if (mapIds.find(unit->GetMapId()) != mapIds.end())
318  return true; // Spell is disabled on current map
319 
320  if (!(spellFlags & SPELL_DISABLE_AREA))
321  return false; // Spell is disabled on another map, but not this one, return false
322 
323  // Spell is disabled in an area, but not explicitly our current mapId. Continue processing.
324  }
325 
326  if (spellFlags & SPELL_DISABLE_AREA)
327  {
328  std::set<uint32> const& areaIds = itr->second.params[1];
329  if (areaIds.find(unit->GetAreaId()) != areaIds.end())
330  return true; // Spell is disabled in this area
331  return false; // Spell is disabled in another area, but not this one, return false
332  }
333  else
334  return true; // Spell disabled for all maps
335  }
336 
337  return false;
338  }
339  else if (spellFlags & SPELL_DISABLE_DEPRECATED_SPELL) // call not from spellcast
340  return true;
341 
342  break;
343  }
344  case DISABLE_TYPE_MAP:
345  case DISABLE_TYPE_LFG_MAP:
346  if (Player const* player = unit->ToPlayer())
347  {
348  MapEntry const* mapEntry = sMapStore.LookupEntry(entry);
349  if (mapEntry->IsDungeon())
350  {
351  uint8 disabledModes = itr->second.flags;
352  Difficulty targetDifficulty = player->GetDifficulty(mapEntry->IsRaid());
353  GetDownscaledMapDifficultyData(entry, targetDifficulty);
354  switch (targetDifficulty)
355  {
356  case DUNGEON_DIFFICULTY_NORMAL:
357  return disabledModes & DUNGEON_STATUSFLAG_NORMAL;
358  case DUNGEON_DIFFICULTY_HEROIC:
359  return disabledModes & DUNGEON_STATUSFLAG_HEROIC;
360  case RAID_DIFFICULTY_10MAN_HEROIC:
361  return disabledModes & RAID_STATUSFLAG_10MAN_HEROIC;
362  case RAID_DIFFICULTY_25MAN_HEROIC:
363  return disabledModes & RAID_STATUSFLAG_25MAN_HEROIC;
364  }
365  }
366  else if (mapEntry->map_type == MAP_COMMON)
367  return true;
368  }
369  return false;
370  case DISABLE_TYPE_QUEST:
371  return true;
372  case DISABLE_TYPE_BATTLEGROUND:
373  case DISABLE_TYPE_OUTDOORPVP:
374  case DISABLE_TYPE_ACHIEVEMENT_CRITERIA:
375  return true;
376  case DISABLE_TYPE_VMAP:
377  return flags & itr->second.flags;
378  case DISABLE_TYPE_GO_LOS:
379  return true;
380  case DISABLE_TYPE_GAME_EVENT:
381  return true;
382  }
383 
384  return false;
385  }
