146  void UpdateAI(uint32 diff) override
147  {
148  if (!UpdateVictim())
149  return;
150 
151  events.Update(diff);
152 
153  if (_phase == PHASE_AIR && me->GetHealthPct() < 70.0f)
154  {
155  _phase = PHASE_GROUND;
156  SetCombatMovement(true);
157  me->SetCanFly(false);
158  Position VictimPos = me->GetVictim()->GetPosition();
159  me->GetMotionMaster()->MovePoint(POINT_GROUND, VictimPos);
160  DoResetThreat();
161  events.ScheduleEvent(EVENT_LASH, urand(5000, 8000));
162  events.ScheduleEvent(EVENT_TRASH, urand(3000, 6000));
163  events.CancelEvent(EVENT_POISON_STINGER);
164  }
165  else
166  {
167  DoMeleeAttackIfReady();
168  }
169 
170  if (!_enraged && me->GetHealthPct() < 20.0f)
171  {
172  DoCast(me, SPELL_FRENZY);
173  Talk(EMOTE_FRENZY);
174  _enraged = true;
175  }
176 
177  while (uint32 eventId = events.ExecuteEvent())
178  {
179  switch (eventId)
180  {
181  case EVENT_STINGER_SPRAY:
182  DoCast(me, SPELL_STINGER_SPRAY);
183  events.ScheduleEvent(EVENT_STINGER_SPRAY, urand(15000, 20000));
184  break;
185  case EVENT_POISON_STINGER:
186  DoCastVictim(SPELL_POISON_STINGER);
187  events.ScheduleEvent(EVENT_POISON_STINGER, urand(2000, 3000));
188  break;
189  case EVENT_PARALYZE:
190  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0, true))
191  {
192  DoCast(target, SPELL_PARALYZE);
193  instance->SetGuidData(DATA_PARALYZED, target->GetGUID());
194  uint8 Index = urand(0, 1);
195  me->SummonCreature(NPC_LARVA, LarvaPos[Index], TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN, 30000);
196  }
197  events.ScheduleEvent(EVENT_PARALYZE, 15000);
198  break;
199  case EVENT_SWARMER_ATTACK:
200  for (ObjectGuid const& guid : _swarmers)
201  if (Creature* swarmer = me->GetMap()->GetCreature(guid))
202  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
203  swarmer->AI()->AttackStart(target);
204 
205  _swarmers.clear();
206  events.ScheduleEvent(EVENT_SWARMER_ATTACK, 60000);
207  break;
208  case EVENT_SUMMON_SWARMER:
209  {
210  Position Pos = me->GetRandomPoint(SwarmerPos, 80.0f);
211  me->SummonCreature(NPC_SWARMER, Pos);
212  events.ScheduleEvent(EVENT_SUMMON_SWARMER, 5000);
213  break;
214  }
215  case EVENT_TRASH:
216  DoCastVictim(SPELL_TRASH);
217  events.ScheduleEvent(EVENT_TRASH, urand(5000, 7000));
218  break;
219  case EVENT_LASH:
220  DoCastVictim(SPELL_LASH);
221  events.ScheduleEvent(EVENT_LASH, urand(8000, 15000));
222  break;
223  }
224  }
225  }
226  private:
