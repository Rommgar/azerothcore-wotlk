136 void MotionTransport::Update(uint32 diff)
137 {
138  uint32 const positionUpdateDelay = 1;
139 
140  if (AI())
141  AI()->UpdateAI(diff);
142  else if (!AIM_Initialize())
143  LOG_ERROR("entities.transport", "Could not initialize GameObjectAI for Transport");
144 
145  if (GetKeyFrames().size() <= 1)
146  return;
147 
148  if (IsMoving() || !_pendingStop)
149  SetPathProgress(GetPathProgress() + diff);
150 
151  uint32 timer = GetPathProgress() % GetPeriod();
152 
153  // Set current waypoint
154  // Desired outcome: _currentFrame->DepartureTime < timer < _nextFrame->ArriveTime
155  // ... arrive | ... delay ... | departure
156  // event / event /
157  for (;;)
158  {
159  if (timer >= _currentFrame->ArriveTime)
160  {
161  if (!_triggeredArrivalEvent)
162  {
163  DoEventIfAny(*_currentFrame, false);
164  _triggeredArrivalEvent = true;
165  }
166 
167  if (timer < _currentFrame->DepartureTime)
168  {
169  SetMoving(false);
170  if (_pendingStop && GetGoState() != GO_STATE_READY)
171  {
172  SetGoState(GO_STATE_READY);
173  SetPathProgress(GetPathProgress() / GetPeriod());
174  SetPathProgress(GetPathProgress() * GetPeriod());
175  SetPathProgress(GetPathProgress() + _currentFrame->ArriveTime);
176  }
177  break; // its a stop frame and we are waiting
178  }
179  }
180 
181  if (timer >= _currentFrame->DepartureTime && !_triggeredDepartureEvent)
182  {
183  DoEventIfAny(*_currentFrame, true); // departure event
184  _triggeredDepartureEvent = true;
185  }
186 
187  // not waiting anymore
188  SetMoving(true);
189 
190  // Enable movement
191  if (GetGOInfo()->moTransport.canBeStopped)
192  SetGoState(GO_STATE_ACTIVE);
193 
194  if (timer >= _currentFrame->DepartureTime && timer < _currentFrame->NextArriveTime)
195  break; // found current waypoint
196 
197  MoveToNextWaypoint();
198 
199  sScriptMgr->OnRelocate(this, _currentFrame->Node->index, _currentFrame->Node->mapid, _currentFrame->Node->x, _currentFrame->Node->y, _currentFrame->Node->z);
200 
201  //TC_LOG_DEBUG("entities.transport", "Transport %u (%s) moved to node %u %u %f %f %f", GetEntry(), GetName().c_str(), _currentFrame->Node->index, _currentFrame->Node->mapid, _currentFrame->Node->x, _currentFrame->Node->y, _currentFrame->Node->z);
202 
203  // Departure event
204  if (_currentFrame->IsTeleportFrame())
205  if (TeleportTransport(_nextFrame->Node->mapid, _nextFrame->Node->x, _nextFrame->Node->y, _nextFrame->Node->z, _nextFrame->InitialOrientation))
206  return; // Update more in new map thread
207  }
208 
209  // Set position
210  _positionChangeTimer.Update(diff);
211  if (_positionChangeTimer.Passed())
212  {
213  _positionChangeTimer.Reset(positionUpdateDelay);
214  if (IsMoving())
215  {
216  float t = CalculateSegmentPos(float(timer) * 0.001f);
217  G3D::Vector3 pos, dir;
218  _currentFrame->Spline->evaluate_percent(_currentFrame->Index, t, pos);
219  _currentFrame->Spline->evaluate_derivative(_currentFrame->Index, t, dir);
220  UpdatePosition(pos.x, pos.y, pos.z, NormalizeOrientation(atan2(dir.y, dir.x) + M_PI));
221  }
222  else
223  {
224  /* There are four possible scenarios that trigger loading/unloading passengers:
225  1. transport moves from inactive to active grid
226  2. the grid that transport is currently in becomes active
227  3. transport moves from active to inactive grid
228  4. the grid that transport is currently in unloads
229  */
230  if (_staticPassengers.empty() && GetMap()->IsGridLoaded(GetPositionX(), GetPositionY())) // 2.
231  LoadStaticPassengers();
232  }
233  }
234 
235  sScriptMgr->OnTransportUpdate(this, diff);
236 }

473 float MotionTransport::CalculateSegmentPos(float now)
474 {
475  KeyFrame const& frame = *_currentFrame;
476  const float speed = float(m_goInfo->moTransport.moveSpeed);
477  const float accel = float(m_goInfo->moTransport.accelRate);
478  float timeSinceStop = frame.TimeFrom + (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
479  float timeUntilStop = frame.TimeTo - (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
480  float segmentPos, dist;
481  float accelTime = _transportInfo->accelTime;
482  float accelDist = _transportInfo->accelDist;
483  // calculate from nearest stop, less confusing calculation...
484  if (timeSinceStop < timeUntilStop)
485  {
486  if (timeSinceStop < accelTime)
487  dist = 0.5f * accel * timeSinceStop * timeSinceStop;
488  else
489  dist = accelDist + (timeSinceStop - accelTime) * speed;
490  segmentPos = dist - frame.DistSinceStop;
491  }
492  else
493  {
494  if (timeUntilStop < _transportInfo->accelTime)
495  dist = 0.5f * accel * timeUntilStop * timeUntilStop;
496  else
497  dist = accelDist + (timeUntilStop - accelTime) * speed;
498  segmentPos = frame.DistUntilStop - dist;
499  }
500 
501  return segmentPos / frame.NextDistFromPrev;
502 }

534 void MotionTransport::DelayedTeleportTransport()
535 {
536  if (!_delayedTeleport)
537  return;
538 
539  _delayedTeleport = false;
540 
541  uint32 newMapId = _nextFrame->Node->mapid;
542  float x = _nextFrame->Node->x,
543  y = _nextFrame->Node->y,
544  z = _nextFrame->Node->z,
545  o = _nextFrame->InitialOrientation;
546 
547  PassengerSet _passengersCopy = _passengers;
548  for (PassengerSet::iterator itr = _passengersCopy.begin(); itr != _passengersCopy.end(); )
549  {
550  WorldObject* obj = (*itr++);
551 
552  if (_passengers.find(obj) == _passengers.end())
553  continue;
554 
555  switch (obj->GetTypeId())

558  _passengers.erase(obj);
559  if (!obj->ToCreature()->IsPet())

561  break;

565  break;

574  caster->RemoveOwnedAura(s->GetSpellInfo()->Id, caster->GetGUID());
575  }
576  obj->AddObjectToRemoveList();
577  break;
578  case TYPEID_PLAYER:
579  {
580  float destX, destY, destZ, destO;
581  obj->m_movementInfo.transport.pos.GetPosition(destX, destY, destZ, destO);
582  TransportBase::CalculatePassengerPosition(destX, destY, destZ, &destO, x, y, z, o);
583  if (!obj->ToPlayer()->TeleportTo(newMapId, destX, destY, destZ, destO, TELE_TO_NOT_LEAVE_TRANSPORT))
584  _passengers.erase(obj);
585  }
586  break;
587  default:
588  break;
589  }
590  }
591 
592  Map* newMap = sMapMgr->CreateBaseMap(newMapId);
593  GetMap()->RemoveFromMap<MotionTransport>(this, false);
594  newMap->LoadGrid(x, y); // xinef: load before adding passengers to new map
595  SetMap(newMap);

599 
600  LoadStaticPassengers();

603 void MotionTransport::UpdatePassengerPositions(PassengerSet& passengers)
604 {
605  for (PassengerSet::iterator itr = passengers.begin(); itr != passengers.end(); ++itr)
606  {
607  WorldObject* passenger = *itr;
608  // transport teleported but passenger not yet (can happen for players)
609  if (passenger->GetMap() != GetMap())
610  continue;
611 
612  // if passenger is on vehicle we have to assume the vehicle is also on transport and its the vehicle that will be updating its passengers
613  if (Unit* unit = passenger->ToUnit())
614  if (unit->GetVehicle())
615  continue;
616 
617  // Do not use Unit::UpdatePosition here, we don't want to remove auras as if regular movement occurred
618  float x, y, z, o;
619  passenger->m_movementInfo.transport.pos.GetPosition(x, y, z, o);
620  CalculatePassengerPosition(x, y, z, &o);
621 
622  // check if position is valid
623  if (!Acore::IsValidMapCoord(x, y, z))
624  continue;
625 
626  switch (passenger->GetTypeId())
627  {
628  case TYPEID_UNIT:
629  {
630  Creature* creature = passenger->ToCreature();
631  GetMap()->CreatureRelocation(creature, x, y, z, o);
632 
633  creature->GetTransportHomePosition(x, y, z, o);
634  CalculatePassengerPosition(x, y, z, &o);
635  creature->SetHomePosition(x, y, z, o);
636  }
637  break;
638  case TYPEID_PLAYER:
639  if (passenger->IsInWorld())
640  GetMap()->PlayerRelocation(passenger->ToPlayer(), x, y, z, o);
641  break;
642  case TYPEID_GAMEOBJECT:
643  GetMap()->GameObjectRelocation(passenger->ToGameObject(), x, y, z, o);
644  break;
645  case TYPEID_DYNAMICOBJECT:
646  GetMap()->DynamicObjectRelocation(passenger->ToDynObject(), x, y, z, o);
647  break;
648  default:
649  break;
650  }
651  }

863 void StaticTransport::RelocateToProgress(uint32 progress)
864 {
865  TransportAnimationEntry const* curr = nullptr, *next = nullptr;
866  float percPos;
867  if (m_goValue.Transport.AnimationInfo->GetAnimNode(progress, curr, next, percPos))
868  {
869  // curr node offset
870  G3D::Vector3 pos = G3D::Vector3(curr->X, curr->Y, curr->Z);
871 
872  // move by percentage of segment already passed
873  pos += G3D::Vector3(percPos * (next->X - curr->X), percPos * (next->Y - curr->Y), percPos * (next->Z - curr->Z));
874 
875  // rotate path by PathRotation
876  // pussywizard: PathRotation in db is only simple orientation rotation, so don't use sophisticated and not working code
877  // reminder: WorldRotation only influences model rotation, not the path
878  float sign = GetFloatValue(GAMEOBJECT_PARENTROTATION + 2) >= 0.0f ? 1.0f : -1.0f;
879  float pathRotAngle = sign * 2.0f * acos(GetFloatValue(GAMEOBJECT_PARENTROTATION + 3));
880  float cs = cos(pathRotAngle), sn = sin(pathRotAngle);
881  float nx = pos.x * cs - pos.y * sn;
882  float ny = pos.x * sn + pos.y * cs;
883  pos.x = nx;
884  pos.y = ny;
885 
886  // add stationary position to the calculated offset
887  pos += G3D::Vector3(GetStationaryX(), GetStationaryY(), GetStationaryZ());
888 
889  // rotate by AnimRotation at current segment
890  // pussywizard: AnimRotation in dbc is only simple orientation rotation, so don't use sophisticated and not working code
891  G3D::Quat currRot, nextRot;
892  float percRot;
893  m_goValue.Transport.AnimationInfo->GetAnimRotation(progress, currRot, nextRot, percRot);
894  float signCurr = currRot.z >= 0.0f ? 1.0f : -1.0f;
895  float oriRotAngleCurr = signCurr * 2.0f * acos(currRot.w);
896  float signNext = nextRot.z >= 0.0f ? 1.0f : -1.0f;
897  float oriRotAngleNext = signNext * 2.0f * acos(nextRot.w);
898  float oriRotAngle = oriRotAngleCurr + percRot * (oriRotAngleNext - oriRotAngleCurr);
899 
900  // check if position is valid
901  if (!Acore::IsValidMapCoord(pos.x, pos.y, pos.z))
902  return;
903 
904  // update position to new one
905  // also adding simplified orientation rotation here
906  UpdatePosition(pos.x, pos.y, pos.z, NormalizeOrientation(GetStationaryO() + oriRotAngle));
907  }
