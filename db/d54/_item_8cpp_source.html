30 void AddItemsSetItem(Player* player, Item* item)
31 {
32  ItemTemplate const* proto = item->GetTemplate();
33  uint32 setid = proto->ItemSet;
34 
35  ItemSetEntry const* set = sItemSetStore.LookupEntry(setid);
36 
37  if (!set)
38  {
39  LOG_ERROR("sql.sql", "Item set %u for item (id %u) not found, mods not applied.", setid, proto->ItemId);
40  return;
41  }
42 
43  if (set->required_skill_id && player->GetSkillValue(set->required_skill_id) < set->required_skill_value)
44  return;
45 
46  ItemSetEffect* eff = nullptr;
47 
48  for (size_t x = 0; x < player->ItemSetEff.size(); ++x)
49  {
50  if (player->ItemSetEff[x] && player->ItemSetEff[x]->setid == setid)
51  {
52  eff = player->ItemSetEff[x];
53  break;
54  }
55  }
56 
57  if (!eff)
58  {
59  eff = new ItemSetEffect();
60  eff->setid = setid;
61 
62  size_t x = 0;
63  for (; x < player->ItemSetEff.size(); ++x)
64  if (!player->ItemSetEff[x])
65  break;
66 
67  if (x < player->ItemSetEff.size())
68  player->ItemSetEff[x] = eff;
69  else
70  player->ItemSetEff.push_back(eff);
71  }
72 
73  ++eff->item_count;
74 
75  for (uint32 x = 0; x < MAX_ITEM_SET_SPELLS; ++x)
76  {
77  if (!set->spells [x])
78  continue;
79  //not enough for spell
80  if (set->items_to_triggerspell[x] > eff->item_count)
81  continue;
82 
83  uint32 z = 0;
84  for (; z < MAX_ITEM_SET_SPELLS; ++z)
85  if (eff->spells[z] && eff->spells[z]->Id == set->spells[x])
86  break;
87 
88  if (z < MAX_ITEM_SET_SPELLS)
89  continue;
90 
91  //new spell
92  for (uint32 y = 0; y < MAX_ITEM_SET_SPELLS; ++y)
93  {
94  if (!eff->spells[y]) // free slot
95  {
96  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(set->spells[x]);
97  if (!spellInfo)
98  {
99  LOG_ERROR("entities.item", "WORLD: unknown spell id %u in items set %u effects", set->spells[x], setid);
100  break;
101  }
102 
103  // spell casted only if fit form requirement, in other case will casted at form change
104  if (sScriptMgr->CanItemApplyEquipSpell(player, item))
105  {
106  player->ApplyEquipSpell(spellInfo, nullptr, true);
107  }
108 
109  eff->spells[y] = spellInfo;
110  break;
111  }
112  }
113  }

314 void Item::SaveToDB(CharacterDatabaseTransaction trans)
315 {
316  bool isInTransaction = static_cast<bool>(trans);
317  if (!isInTransaction)
318  trans = CharacterDatabase.BeginTransaction();
319 
320  ObjectGuid::LowType guid = GetGUID().GetCounter();
321  switch (uState)
322  {
323  case ITEM_NEW:
324  case ITEM_CHANGED:
325  {
326  uint8 index = 0;
327  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(uState == ITEM_NEW ? CHAR_REP_ITEM_INSTANCE : CHAR_UPD_ITEM_INSTANCE);
328  stmt->setUInt32( index, GetEntry());
329  stmt->setUInt32(++index, GetOwnerGUID().GetCounter());
330  stmt->setUInt32(++index, GetGuidValue(ITEM_FIELD_CREATOR).GetCounter());
331  stmt->setUInt32(++index, GetGuidValue(ITEM_FIELD_GIFTCREATOR).GetCounter());
332  stmt->setUInt32(++index, GetCount());
333  stmt->setUInt32(++index, GetUInt32Value(ITEM_FIELD_DURATION));
334 
335  std::ostringstream ssSpells;
336  for (uint8 i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
337  ssSpells << GetSpellCharges(i) << ' ';
338  stmt->setString(++index, ssSpells.str());
339 
340  stmt->setUInt32(++index, GetUInt32Value(ITEM_FIELD_FLAGS));
341 
342  std::ostringstream ssEnchants;
343  for (uint8 i = 0; i < MAX_ENCHANTMENT_SLOT; ++i)
344  {
345  ssEnchants << GetEnchantmentId(EnchantmentSlot(i)) << ' ';
346  ssEnchants << GetEnchantmentDuration(EnchantmentSlot(i)) << ' ';
347  ssEnchants << GetEnchantmentCharges(EnchantmentSlot(i)) << ' ';
348  }
349  stmt->setString(++index, ssEnchants.str());
350 
351  stmt->setInt16 (++index, GetItemRandomPropertyId());
352  stmt->setUInt16(++index, GetUInt32Value(ITEM_FIELD_DURABILITY));
353  stmt->setUInt32(++index, GetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME));
354  stmt->setString(++index, m_text);
355  stmt->setUInt32(++index, guid);
356 
357  trans->Append(stmt);
358 
359  if ((uState == ITEM_CHANGED) && HasFlag(ITEM_FIELD_FLAGS, ITEM_FIELD_FLAG_WRAPPED))
360  {
361  stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_GIFT_OWNER);
362  stmt->setUInt32(0, GetOwnerGUID().GetCounter());
363  stmt->setUInt32(1, guid);
364  trans->Append(stmt);
365  }
366  break;
367  }
368  case ITEM_REMOVED:
369  {
370  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ITEM_INSTANCE);
371  stmt->setUInt32(0, guid);
372  trans->Append(stmt);
373 
374  if (HasFlag(ITEM_FIELD_FLAGS, ITEM_FIELD_FLAG_WRAPPED))
375  {
376  stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GIFT);
377  stmt->setUInt32(0, guid);
378  trans->Append(stmt);
379  }
380 
381  if (!isInTransaction)
382  CharacterDatabase.CommitTransaction(trans);
383 
384  delete this;
385  return;
386  }
387  case ITEM_UNCHANGED:
388  break;
389  }
390 
391  SetState(ITEM_UNCHANGED);
392 
393  if (!isInTransaction)
394  CharacterDatabase.CommitTransaction(trans);
395 }

442  need_save = true;
443  }
444 
445  std::string enchants = fields[6].GetString();
446  _LoadIntoDataField(enchants.c_str(), ITEM_FIELD_ENCHANTMENT_1_1, MAX_ENCHANTMENT_SLOT * MAX_ENCHANTMENT_OFFSET);
447  SetInt32Value(ITEM_FIELD_RANDOM_PROPERTIES_ID, fields[7].GetInt16());
448  // recalculate suffix factor
449  if (GetItemRandomPropertyId() < 0)
450  UpdateItemSuffixFactor();
451 
452  uint32 durability = fields[8].GetUInt16();
453  SetUInt32Value(ITEM_FIELD_DURABILITY, durability);
454 
455  // update max durability (and durability) if need
456  // xinef: do not overwrite durability for wrapped items!!
457  SetUInt32Value(ITEM_FIELD_MAXDURABILITY, proto->MaxDurability);
458  if (durability > proto->MaxDurability && !HasFlag(ITEM_FIELD_FLAGS, ITEM_FIELD_FLAG_WRAPPED))
459  {
460  SetUInt32Value(ITEM_FIELD_DURABILITY, proto->MaxDurability);
461  need_save = true;
462  }
463 
464  SetUInt32Value(ITEM_FIELD_CREATE_PLAYED_TIME, fields[9].GetUInt32());
465  SetText(fields[10].GetString());
466 
467  if (need_save) // normal item changed state set not work at loading
468  {
469  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ITEM_INSTANCE_ON_LOAD);
470  stmt->setUInt32(0, GetUInt32Value(ITEM_FIELD_DURATION));
471  stmt->setUInt32(1, GetUInt32Value(ITEM_FIELD_FLAGS));
472  stmt->setUInt32(2, GetUInt32Value(ITEM_FIELD_DURABILITY));
473  stmt->setUInt32(3, guid);
474  CharacterDatabase.Execute(stmt);
475  }
476 
477  return true;
478 }
