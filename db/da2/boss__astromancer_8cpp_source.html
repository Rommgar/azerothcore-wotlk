125  void UpdateAI(uint32 diff) override
126  {
127  if (!UpdateVictim())
128  return;
129 
130  events.Update(diff);
131  if (me->HasUnitState(UNIT_STATE_CASTING))
132  return;
133 
134  switch (events.ExecuteEvent())
135  {
136  case EVENT_CHECK_HEALTH:
137  if (me->HealthBelowPct(21))
138  {
139  events.Reset();
140  events.ScheduleEvent(EVENT_SPELL_VOID_BOLT, 3000);
141  events.ScheduleEvent(EVENT_SPELL_PSYCHIC_SCREAM, 7000);
142  me->CastSpell(me, SPELL_SOLARIAN_TRANSFORM, true);
143  me->GetMotionMaster()->MoveChase(me->GetVictim());
144  break;
145  }
146  events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
147  break;
148  case EVENT_SPELL_ARCANE_MISSILES:
149  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 40.0f, true))
150  me->CastSpell(target, SPELL_ARCANE_MISSILES, false);
151  events.ScheduleEvent(EVENT_SPELL_ARCANE_MISSILES, 3000);
152  break;
153  case EVENT_SPELL_WRATH_OF_ASTROMANCER:
154  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
155  me->CastSpell(target, SPELL_WRATH_OF_THE_ASTROMANCER, false);
156  events.ScheduleEvent(EVENT_SPELL_WRATH_OF_ASTROMANCER, 22000);
157  break;
158  case EVENT_SPELL_BLINDING_LIGHT:
159  me->CastSpell(me, SPELL_BLINDING_LIGHT, false);
160  events.ScheduleEvent(EVENT_SPELL_BLINDING_LIGHT, 40000);
161  break;
162  case EVENT_SPAWN_PORTALS:
163  me->setAttackTimer(BASE_ATTACK, 21000);
164  me->SetModelVisible(false);
165  events.ScheduleEvent(EVENT_SPAWN_PORTALS, 50000);
166  events.DelayEvents(21000);
167  events.ScheduleEvent(EVENT_SUMMON_ADDS, 6000);
168  events.ScheduleEvent(EVENT_REAPPEAR, 20000);
169  for (uint8 i = 0; i < 3; ++i)
170  {
171  float o = rand_norm() * 2 * M_PI;
172  if (i == 0)
173  me->SummonCreature(NPC_ASTROMANCER_SOLARIAN_SPOTLIGHT, CENTER_X + cos(o)*INNER_PORTAL_RADIUS, CENTER_Y + sin(o)*INNER_PORTAL_RADIUS, CENTER_Z, 0.0f, TEMPSUMMON_TIMED_DESPAWN, 26000);
174  else
175  me->SummonCreature(NPC_ASTROMANCER_SOLARIAN_SPOTLIGHT, CENTER_X + cos(o)*OUTER_PORTAL_RADIUS, CENTER_Y + sin(o)*OUTER_PORTAL_RADIUS, PORTAL_Z, 0.0f, TEMPSUMMON_TIMED_DESPAWN, 26000);
176  }
177  break;
178  case EVENT_SUMMON_ADDS:
179  Talk(SAY_SUMMON1);
180  for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
181  {
182  if (Creature* light = ObjectAccessor::GetCreature(*me, *itr))
183  if (light->GetEntry() == NPC_ASTROMANCER_SOLARIAN_SPOTLIGHT)
184  {
185  if (light->GetDistance2d(CENTER_X, CENTER_Y) < 20.0f)
186  {
187  me->SetPosition(*light);
188  me->StopMovingOnCurrentPos();
189  }
190  for (uint8 j = 0; j < 4; ++j)
191  me->SummonCreature(NPC_SOLARIUM_AGENT, light->GetPositionX() + frand(-3.0f, 3.0f), light->GetPositionY() + frand(-3.0f, 3.0f), light->GetPositionZ(), 0.0f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
192  }
193  }
194  break;
195  case EVENT_REAPPEAR:
196  Talk(SAY_SUMMON2);
197  for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
198  {
199  if (Creature* light = ObjectAccessor::GetCreature(*me, *itr))
200  {
201  if (light->GetEntry() == NPC_ASTROMANCER_SOLARIAN_SPOTLIGHT)
202  {
203  light->RemoveAllAuras();
204  if (light->GetDistance2d(CENTER_X, CENTER_Y) < 20.0f)
205  me->SetModelVisible(true);
206  else
207  me->SummonCreature(NPC_SOLARIUM_PRIEST, light->GetPositionX() + frand(-3.0f, 3.0f), light->GetPositionY() + frand(-3.0f, 3.0f), light->GetPositionZ(), 0.0f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
208  }
209  }
210  }
211  // protection
212  if (me->GetDisplayId() != me->GetNativeDisplayId())
213  {
214  me->SetModelVisible(true);
215  me->SummonCreature(NPC_SOLARIUM_PRIEST, me->GetPositionX() + frand(-3.0f, 3.0f), me->GetPositionY() + frand(-3.0f, 3.0f), me->GetPositionZ(), 0.0f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
216  me->SummonCreature(NPC_SOLARIUM_PRIEST, me->GetPositionX() + frand(-3.0f, 3.0f), me->GetPositionY() + frand(-3.0f, 3.0f), me->GetPositionZ(), 0.0f, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
217  }
218  break;
219  case EVENT_SPELL_VOID_BOLT:
