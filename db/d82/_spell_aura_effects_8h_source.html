339  class AbsorbAuraOrderPred
340  {
341  public:
342  AbsorbAuraOrderPred() { }
343  bool operator() (AuraEffect* aurEffA, AuraEffect* aurEffB) const
344  {
345  SpellInfo const* spellProtoA = aurEffA->GetSpellInfo();
346  SpellInfo const* spellProtoB = aurEffB->GetSpellInfo();
347 
348  // Wards
349  if ((spellProtoA->SpellFamilyName == SPELLFAMILY_MAGE) ||
350  (spellProtoA->SpellFamilyName == SPELLFAMILY_WARLOCK))
351  if (spellProtoA->GetCategory() == 56)
352  return true;
353  if ((spellProtoB->SpellFamilyName == SPELLFAMILY_MAGE) ||
354  (spellProtoB->SpellFamilyName == SPELLFAMILY_WARLOCK))
355  if (spellProtoB->GetCategory() == 56)
356  return false;
357 
358  // Sacred Shield
359  if (spellProtoA->Id == 58597)
360  return true;
361  if (spellProtoB->Id == 58597)
362  return false;
363 
364  // Fel Blossom
365  if (spellProtoA->Id == 28527)
366  return true;
367  if (spellProtoB->Id == 28527)
368  return false;
369 
370  // Divine Aegis
371  if (spellProtoA->Id == 47753)
372  return true;
373  if (spellProtoB->Id == 47753)
374  return false;
375 
376  // Ice Barrier
377  if (spellProtoA->GetCategory() == 471)
378  return true;
379  if (spellProtoB->GetCategory() == 471)
380  return false;
381 
382  // Sacrifice
383  if ((spellProtoA->SpellFamilyName == SPELLFAMILY_WARLOCK) &&
384  (spellProtoA->SpellIconID == 693))
385  return true;
386  if ((spellProtoB->SpellFamilyName == SPELLFAMILY_WARLOCK) &&
387  (spellProtoB->SpellIconID == 693))
388  return false;
389 
390  return false;
391  }
