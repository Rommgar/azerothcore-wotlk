311  return GetUnitOwner()->GetTypeId() == TYPEID_PLAYER;

317  amount = -1;

320  void Absorb(AuraEffect* aurEff, DamageInfo& dmgInfo, uint32& absorbAmount)
321  {
322  Unit* victim = GetTarget();
323  int32 remainingHealth = victim->GetHealth() - dmgInfo.GetDamage();
324  uint32 allowedHealth = victim->CountPctFromMaxHealth(35);
325  // If damage kills us
326  if (remainingHealth <= 0 && !victim->ToPlayer()->HasSpellCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL))
327  {
328  // Cast healing spell, completely avoid damage
329  absorbAmount = dmgInfo.GetDamage();
330 
331  uint32 defenseSkillValue = victim->GetDefenseSkillValue();
332  // Max heal when defense skill denies critical hits from raid bosses
333  // Formula: max defense at level + 140 (raiting from gear)
334  uint32 reqDefForMaxHeal = victim->getLevel() * 5 + 140;
335  float pctFromDefense = (defenseSkillValue >= reqDefForMaxHeal)
336  ? 1.0f
337  : float(defenseSkillValue) / float(reqDefForMaxHeal);
338 
339  int32 healAmount = int32(victim->CountPctFromMaxHealth(uint32(healPct * pctFromDefense)));
340  victim->CastCustomSpell(PAL_SPELL_ARDENT_DEFENDER_HEAL, SPELLVALUE_BASE_POINT0, healAmount, victim, true, nullptr, aurEff);
341  victim->ToPlayer()->AddSpellCooldown(PAL_SPELL_ARDENT_DEFENDER_HEAL, 0, 120000);
342  }
343  else if (remainingHealth < int32(allowedHealth))
344  {
345  // Reduce damage that brings us under 35% (or full damage if we are already under 35%) by x%
346  uint32 damageToReduce = (victim->GetHealth() < allowedHealth)
347  ? dmgInfo.GetDamage()
348  : allowedHealth - remainingHealth;
349  absorbAmount = CalculatePct(damageToReduce, absorbPct);
350  }
351  }
352 
353  void Register() override
354  {
355  DoEffectCalcAmount += AuraEffectCalcAmountFn(spell_pal_ardent_defender::CalculateAmount, EFFECT_0, SPELL_AURA_SCHOOL_ABSORB);
356  OnEffectAbsorb += AuraEffectAbsorbFn(spell_pal_ardent_defender::Absorb, EFFECT_0);
357  }
