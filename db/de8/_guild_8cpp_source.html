466 // Writes information about contents of specified slot into packet.
467 bool Guild::BankTab::WriteSlotPacket(WorldPacket& data, uint8 slotId, bool ignoreEmpty /* = true */) const
468 {
469  Item* pItem = GetItem(slotId);
470  uint32 itemEntry = pItem ? pItem->GetEntry() : 0;
471 
472  if (!itemEntry && ignoreEmpty)
473  return false;
474 
475  data << uint8(slotId);
476  data << uint32(itemEntry);
477  if (itemEntry)
478  {
479  data << uint32(0); // 3.3.0 (0x00018020, 0x00018000)
480 
481  if (int32 random = pItem->GetItemRandomPropertyId())
482  {
483  data << int32(random); // Random item property id
484  data << uint32(pItem->GetItemSuffixFactor()); // SuffixFactor
485  }
486  else
487  data << int32(0);
488 
489  data << uint32(pItem->GetCount()); // ITEM_FIELD_STACK_COUNT
490  data << uint32(0);
491  data << uint8(abs(pItem->GetSpellCharges())); // Spell charges
492 
493  uint8 enchCount = 0;
494  size_t enchCountPos = data.wpos();
495 
496  data << uint8(enchCount); // Number of enchantments
497  for (uint32 i = PERM_ENCHANTMENT_SLOT; i < MAX_ENCHANTMENT_SLOT; ++i)
498  if (uint32 enchId = pItem->GetEnchantmentId(EnchantmentSlot(i)))
499  {
500  data << uint8(i);
501  data << uint32(enchId);
502  ++enchCount;
503  }
504  data.put<uint8>(enchCountPos, enchCount);
505  }
506  return true;

1029 // Tries to reserve space for source item.
1030 // If item in destination slot exists it must be the item of the same entry
1031 // and stack must have enough space to take at least one item.
1032 // Returns false if destination item specified and it cannot be used to reserve space.
1033 bool Guild::BankMoveItemData::_ReserveSpace(uint8 slotId, Item* pItem, Item* pItemDest, uint32& count)
1034 {
1035  uint32 requiredSpace = pItem->GetMaxStackCount();
1036  if (pItemDest)
1037  {
1038  // Make sure source and destination items match and destination item has space for more stacks.
1039  if (pItemDest->GetEntry() != pItem->GetEntry() || pItemDest->GetCount() >= pItem->GetMaxStackCount())
1040  return false;
1041  requiredSpace -= pItemDest->GetCount();
1042  }
1043  // Let's not be greedy, reserve only required space
1044  requiredSpace = std::min(requiredSpace, count);
1045 
1046  // Reserve space
1047  ItemPosCount pos(slotId, requiredSpace);
1048  if (!pos.isContainedIn(m_vec))
1049  {
1050  m_vec.push_back(pos);
1051  count -= requiredSpace;
1052  }
1053  return true;
1054 }

2061  // The lower rank id is considered higher rank - so promotion does rank-- and demotion does rank++
2062  // Between ranks in sequence cannot be gaps - so 0, 1, 2, 4 is impossible
2063  // Min ranks count is 5 and max is 10.

2066  if (ranks < GUILD_RANKS_MIN_COUNT || ranks > GUILD_RANKS_MAX_COUNT)
2067  {
2068  LOG_ERROR("guild", "Guild %u has invalid number of ranks, creating new...", m_id);
2069  broken_ranks = true;
2070  }
2071  else
2072  {
2073  for (uint8 rankId = 0; rankId < ranks; ++rankId)
2074  {

2076  if (rankInfo->GetId() != rankId)
2077  {
2078  LOG_ERROR("guild", "Guild %u has broken rank id %u, creating default set of ranks...", m_id, rankId);
2079  broken_ranks = true;
2080  }
2081  else
2082  {
2083  CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
2084  rankInfo->CreateMissingTabsIfNeeded(_GetPurchasedTabsSize(), trans, true);

2087  }
2088  }
