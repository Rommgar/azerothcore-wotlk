146  {
147  boss_illidari_councilAI(Creature* creature) : BossAI(creature, DATA_ILLIDARI_COUNCIL)
148  {
149  }
150 
151  ObjectGuid councilGUIDs[4];
152 
153  void Reset() override
154  {
155  BossAI::Reset();
156  Creature* member = nullptr;
157  for (uint8 i = 0; i < 4; ++i)
158  if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
159  member->AI()->EnterEvadeMode();
160  }
161 
162  void AttackStart(Unit*) override { }
163  void MoveInLineOfSight(Unit*) override { }
164 
165  void DoAction(int32 param) override
166  {
167  if (!me->isActiveObject() && param == ACTION_START_ENCOUNTER)
168  {
169  me->setActive(true);
170  councilGUIDs[0] = instance->GetGuidData(NPC_GATHIOS_THE_SHATTERER);
171  councilGUIDs[1] = instance->GetGuidData(NPC_HIGH_NETHERMANCER_ZEREVOR);
172  councilGUIDs[2] = instance->GetGuidData(NPC_LADY_MALANDE);
173  councilGUIDs[3] = instance->GetGuidData(NPC_VERAS_DARKSHADOW);
174 
175  bool spoken = false;
176  for (uint8 i = 0; i < 4; ++i)
177  {
178  if (Creature* member = ObjectAccessor::GetCreature(*me, councilGUIDs[i]))
179  {
180  if (!spoken && (roll_chance_i(33) || i == 3))
181  {
182  spoken = true;
183  member->AI()->Talk(SAY_COUNCIL_AGGRO);
184  }
185  member->SetOwnerGUID(me->GetGUID());
186  member->SetInCombatWithZone();
187  }
188  }
189  }
190  else if (param == ACTION_ENRAGE)
191  {
192  Creature* member = nullptr;
193  for (uint8 i = 0; i < 4; ++i)
194  if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
195  member->AI()->DoAction(ACTION_ENRAGE);
196  }
197  else if (param == ACTION_END_ENCOUNTER)
198  {
199  me->setActive(false);
200  Creature* member = nullptr;
201  for (uint8 i = 0; i < 4; ++i)
202  if ((member = ObjectAccessor::GetCreature(*me, councilGUIDs[i])))
203  if (member->IsAlive())
204  Unit::Kill(me, member);
205  Unit::Kill(me, me);
206  }
207  }
208 
209  void UpdateAI(uint32 /*diff*/) override
210  {
211  if (!me->isActiveObject())
212  return;
213 
214  if (!SelectTargetFromPlayerList(115.0f))
215  {
216  EnterEvadeMode();
217  return;
218  }
219 
220  me->CastSpell(me, SPELL_EMPYREAL_BALANCE, true);
221  }
222  };
