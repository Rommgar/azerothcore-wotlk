170 template<class T>
171 bool DBUpdater<T>::Create(DatabaseWorkerPool<T>& pool)
172 {
173  LOG_WARN("sql.updates", "Database \"%s\" does not exist, do you want to create it? [yes (default) / no]: ",
174  pool.GetConnectionInfo()->database.c_str());
175 
176  std::string answer;
177  std::getline(std::cin, answer);
178  if (!answer.empty() && !(answer.substr(0, 1) == "y"))
179  return false;
180 
181  LOG_INFO("sql.updates", "Creating database \"%s\"...", pool.GetConnectionInfo()->database.c_str());
182 
183  // Path of temp file
184  static Path const temp("create_table.sql");
185 
186  // Create temporary query to use external MySQL CLi
187  std::ofstream file(temp.generic_string());
188  if (!file.is_open())
189  {
190  LOG_FATAL("sql.updates", "Failed to create temporary query file \"%s\"!", temp.generic_string().c_str());
191  return false;
192  }
193 
194  file << "CREATE DATABASE `" << pool.GetConnectionInfo()->database << "` DEFAULT CHARACTER SET UTF8MB4 COLLATE utf8mb4_general_ci;\n\n";
195 
196  file.close();
197 
198  try
199  {
200  DBUpdater<T>::ApplyFile(pool, pool.GetConnectionInfo()->host, pool.GetConnectionInfo()->user, pool.GetConnectionInfo()->password,
201  pool.GetConnectionInfo()->port_or_socket, "", pool.GetConnectionInfo()->ssl, temp);
202  }
203  catch (UpdateException&)
204  {
205  LOG_FATAL("sql.updates", "Failed to create database %s! Does the user (named in *.conf) have `CREATE`, `ALTER`, `DROP`, `INSERT` and `DELETE` privileges on the MySQL server?", pool.GetConnectionInfo()->database.c_str());
206  std::filesystem::remove(temp);
207  return false;
208  }
209 
210  LOG_INFO("sql.updates", "Done.");
211  LOG_INFO("sql.updates", " ");
212  std::filesystem::remove(temp);
213  return true;
214 }

218 {
219  if (!DBUpdaterUtil::CheckExecutable())
220  return false;
221 
222  LOG_INFO("sql.updates", "Updating %s database...", DBUpdater<T>::GetTableName().c_str());
223

225 
226  if (!is_directory(sourceDirectory))
227  {
228  LOG_ERROR("sql.updates", "DBUpdater: The given source directory %s does not exist, change the path to the directory where your sql directory exists (for example c:\\source\\trinitycore). Shutting down.",
229  sourceDirectory.generic_string().c_str());

234  {

236  if (!checkTable)

252  return true;

255  return true;

259  return false;

262  [&](Path const & file) { DBUpdater<T>::ApplyFile(pool, file); },

281 
282  if (!result.updated)
283  LOG_INFO("sql.updates", ">> %s database is up-to-date! %s", DBUpdater<T>::GetTableName().c_str(), info.c_str());

287  LOG_INFO("sql.updates", " ");

353 template<class T>
354 bool DBUpdater<T>::Populate(DatabaseWorkerPool<T>& pool)
355 {
356  {
357  QueryResult const result = Retrieve(pool, "SHOW TABLES");
358  if (result && (result->GetRowCount() > 0))
359  return true;
360  }
361 
362  if (!DBUpdaterUtil::CheckExecutable())
363  return false;
364 
365  LOG_INFO("sql.updates", "Database %s is empty, auto populating it...", DBUpdater<T>::GetTableName().c_str());
366 
367  std::string const DirPathStr = DBUpdater<T>::GetBaseFilesDirectory();
368 
369  Path const DirPath(DirPathStr);
370  if (!std::filesystem::is_directory(DirPath))
371  {
372  LOG_ERROR("sql.updates", ">> Directory \"%s\" not exist", DirPath.generic_string().c_str());
373  return false;
374  }
375 
376  if (DirPath.empty())
377  {
378  LOG_ERROR("sql.updates", ">> Directory \"%s\" is empty", DirPath.generic_string().c_str());
379  return false;
380  }
381 
382  std::filesystem::directory_iterator const DirItr;
383  uint32 FilesCount = 0;
384 
385  for (std::filesystem::directory_iterator itr(DirPath); itr != DirItr; ++itr)
386  {
387  if (itr->path().extension() == ".sql")
388  FilesCount++;
389  }
390 
391  if (!FilesCount)
392  {
393  LOG_ERROR("sql.updates", ">> In directory \"%s\" not exist '*.sql' files", DirPath.generic_string().c_str());
394  return false;
395  }
396 
397  for (std::filesystem::directory_iterator itr(DirPath); itr != DirItr; ++itr)
398  {
399  if (itr->path().extension() != ".sql")
400  continue;
401 
402  LOG_INFO("sql.updates", ">> Applying \'%s\'...", itr->path().filename().generic_string().c_str());
403 
404  try
405  {
406  ApplyFile(pool, itr->path());
407  }
408  catch (UpdateException&)
409  {
410  return false;
411  }
412  }
413 
414  LOG_INFO("sql.updates", ">> Done!");
415  LOG_INFO("sql.updates", " ");
416  return true;
417 }

438 template<class T>
439 void DBUpdater<T>::ApplyFile(DatabaseWorkerPool<T>& pool, std::string const& host, std::string const& user,
440  std::string const& password, std::string const& port_or_socket, std::string const& database, std::string const& ssl, Path const& path)
441 {
442  std::vector<std::string> args;
443  args.reserve(7);
444 
445  // CLI Client connection info
446  args.emplace_back("-h" + host);
447  args.emplace_back("-u" + user);
448 
449  if (!password.empty())
450  args.emplace_back("-p" + password);
451 
452  // Check if we want to connect through ip or socket (Unix only)
453 #ifdef _WIN32
454 
455  if (host == ".")
456  args.emplace_back("--protocol=PIPE");
457  else
458  args.emplace_back("-P" + port_or_socket);
459 
460 #else
461 
462  if (!std::isdigit(port_or_socket[0]))
463  {
464  // We can't check if host == "." here, because it is named localhost if socket option is enabled
465  args.emplace_back("-P0");
466  args.emplace_back("--protocol=SOCKET");
467  args.emplace_back("-S" + port_or_socket);
468  }
469  else
470  // generic case
471  args.emplace_back("-P" + port_or_socket);
472 
473 #endif
474 
475  // Set the default charset to utf8
476  args.emplace_back("--default-character-set=utf8");
477 
478  // Set max allowed packet to 1 GB
479  args.emplace_back("--max-allowed-packet=1GB");
480 
481  if (ssl == "ssl")
482  args.emplace_back("--ssl");
483 
484  // Database
485  if (!database.empty())
486  args.emplace_back(database);
487 
488  // Invokes a mysql process which doesn't leak credentials to logs
489  int const ret = Acore::StartProcess(DBUpdaterUtil::GetCorrectedMySQLExecutable(), args,
490  "sql.updates", path.generic_string(), true);
491 
492  if (ret != EXIT_SUCCESS)
493  {
494  LOG_FATAL("sql.updates", "Applying of file \'%s\' to database \'%s\' failed!" \
495  " If you are a user, please pull the latest revision from the repository. "
496  "Also make sure you have not applied any of the databases with your sql client. "
497  "You cannot use auto-update system and import sql files from AzerothCore repository with your sql client. "
498  "If you are a developer, please fix your sql query.",
499  path.generic_string().c_str(), pool.GetConnectionInfo()->database.c_str());
500 
501  throw UpdateException("update failed");
502  }
503 }
