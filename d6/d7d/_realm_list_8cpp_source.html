54 void RealmList::LoadBuildInfo()
55 {
56  // 0 1 2 3 4 5 6
57  if (QueryResult result = LoginDatabase.Query("SELECT majorVersion, minorVersion, bugfixVersion, hotfixVersion, build, winChecksumSeed, macChecksumSeed FROM build_info ORDER BY build ASC"))
58  {
59  do
60  {
61  Field* fields = result->Fetch();
62  RealmBuildInfo& build = _builds.emplace_back();
63  build.MajorVersion = fields[0].GetUInt32();
64  build.MinorVersion = fields[1].GetUInt32();
65  build.BugfixVersion = fields[2].GetUInt32();
66  std::string hotfixVersion = fields[3].GetString();
67 
68  if (hotfixVersion.length() < build.HotfixVersion.size())
69  {
70  std::copy(hotfixVersion.begin(), hotfixVersion.end(), build.HotfixVersion.begin());
71  }
72  else
73  {
74  std::fill(hotfixVersion.begin(), hotfixVersion.end(), '\0');
75  }
76 
77  build.Build = fields[4].GetUInt32();
78  std::string windowsHash = fields[5].GetString();
79 
80  if (windowsHash.length() == build.WindowsHash.size() * 2)
81  {
82  HexStrToByteArray(windowsHash, build.WindowsHash);
83  }
84 
85  std::string macHash = fields[6].GetString();
86 
87  if (macHash.length() == build.MacHash.size() * 2)
88  {
89  HexStrToByteArray(macHash, build.MacHash);
90  }
91  } while (result->NextRow());
92  }
93 }

129 void RealmList::UpdateRealms(boost::system::error_code const& error)
130 {
131  if (error)
132  {
133  // Skip update if have errors
134  return;
135  }
136 
137  LOG_DEBUG("server.authserver", "Updating Realm List...");
138 
139  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_REALMLIST);
140  PreparedQueryResult result = LoginDatabase.Query(stmt);
141 
142  std::map<RealmHandle, std::string> existingRealms;
143  for (auto const& [handle, realm] : _realms)
144  {
145  existingRealms[handle] = realm.Name;
146  }
147 
148  _realms.clear();
149 
150  // Circle through results and add them to the realm map
151  if (result)
152  {
153  do
154  {
155  try
156  {
157  Field* fields = result->Fetch();
158  uint32 realmId = fields[0].GetUInt32();
159  std::string name = fields[1].GetString();
160  std::string externalAddressString = fields[2].GetString();
161  std::string localAddressString = fields[3].GetString();
162  std::string localSubmaskString = fields[4].GetString();
163  uint16 port = fields[5].GetUInt16();
164 
165  Optional<boost::asio::ip::tcp::endpoint> externalAddress = _resolver->Resolve(boost::asio::ip::tcp::v4(), externalAddressString, "");
166  if (!externalAddress)
167  {
168  LOG_ERROR("server.authserver", "Could not resolve address %s for realm \"%s\" id %u", externalAddressString.c_str(), name.c_str(), realmId);
169  continue;
170  }
171 
172  Optional<boost::asio::ip::tcp::endpoint> localAddress = _resolver->Resolve(boost::asio::ip::tcp::v4(), localAddressString, "");
173  if (!localAddress)
174  {
175  LOG_ERROR("server.authserver", "Could not resolve localAddress %s for realm \"%s\" id %u", localAddressString.c_str(), name.c_str(), realmId);
176  continue;
177  }
178 
179  Optional<boost::asio::ip::tcp::endpoint> localSubmask = _resolver->Resolve(boost::asio::ip::tcp::v4(), localSubmaskString, "");
180  if (!localSubmask)
181  {
182  LOG_ERROR("server.authserver", "Could not resolve localSubnetMask %s for realm \"%s\" id %u", localSubmaskString.c_str(), name.c_str(), realmId);
183  continue;
184  }
185 
186  uint8 icon = fields[6].GetUInt8();
187 
188  if (icon == REALM_TYPE_FFA_PVP)
189  {
190  icon = REALM_TYPE_PVP;
191  }
192 
193  if (icon >= MAX_CLIENT_REALM_TYPE)
194  {
195  icon = REALM_TYPE_NORMAL;
196  }
197 
198  RealmFlags flag = RealmFlags(fields[7].GetUInt8());
199  uint8 timezone = fields[8].GetUInt8();
200  uint8 allowedSecurityLevel = fields[9].GetUInt8();
201  float pop = fields[10].GetFloat();
202  uint32 build = fields[11].GetUInt32();
203 
204  RealmHandle id{ realmId };
205 
206  UpdateRealm(id, build, name, externalAddress->address(), localAddress->address(), localSubmask->address(), port, icon, flag,
207  timezone, (allowedSecurityLevel <= SEC_ADMINISTRATOR ? AccountTypes(allowedSecurityLevel) : SEC_ADMINISTRATOR), pop);
208 
209  if (!existingRealms.count(id))
210  {
211  LOG_INFO("server.authserver", "Added realm \"%s\" at %s:%u.", name.c_str(), externalAddressString.c_str(), port);
212  }
213  else
214  {
215  LOG_DEBUG("server.authserver", "Updating realm \"%s\" at %s:%u.", name.c_str(), externalAddressString.c_str(), port);
216  }
217 
218  existingRealms.erase(id);
219  }
220  catch (std::exception const& ex)
221  {
222  LOG_ERROR("server.authserver", "Realmlist::UpdateRealms has thrown an exception: %s", ex.what());
223  ABORT();
224  }
225  } while (result->NextRow());
226  }
227 
228  for (auto itr = existingRealms.begin(); itr != existingRealms.end(); ++itr)
229  LOG_INFO("server.authserver", "Removed realm \"%s\".", itr->second.c_str());
230 
231  if (_updateInterval)
232  {
233  _updateTimer->expires_from_now(boost::posix_time::seconds(_updateInterval));
234  _updateTimer->async_wait(std::bind(&RealmList::UpdateRealms, this, std::placeholders::_1));
235  }
236 }
