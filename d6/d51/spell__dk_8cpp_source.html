88 class spell_dk_wandering_plague : public SpellScript
89 {
90  PrepareSpellScript(spell_dk_wandering_plague);
91 
92  void FilterTargets(std::list<WorldObject*>& targets)
93  {
94  static const AuraType ccAuras[] =
95  {
96  SPELL_AURA_MOD_CONFUSE,
97  SPELL_AURA_MOD_FEAR,
98  SPELL_AURA_MOD_STUN,
99  SPELL_AURA_MOD_ROOT,
100  SPELL_AURA_TRANSFORM,
101  SPELL_AURA_NONE
102  };
103 
104  for (std::list<WorldObject*>::iterator itr = targets.begin(); itr != targets.end();)
105  {
106  Unit* target = (*itr)->ToUnit();
107  if (!target)
108  {
109  targets.erase(itr++);
110  continue;
111  }
112 
113  bool skip = false;
114  for (uint8 index = 0; !skip && ccAuras[index] != SPELL_AURA_NONE; ++index)
115  {
116  Unit::AuraEffectList const& auras = target->GetAuraEffectsByType(ccAuras[index]);
117  for (Unit::AuraEffectList::const_iterator i = auras.begin(); i != auras.end(); ++i)
118  if ((*i)->GetAmount())
119  {
120  skip = true;
121  break;
122  }
123  }
124 
125  if (skip)
126  targets.erase(itr++);
127  else
128  ++itr;
129  }
130  }
131 
132  void Register() override
133  {
134  OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_dk_wandering_plague::FilterTargets, EFFECT_0, TARGET_UNIT_DEST_AREA_ENEMY);
135  }

143  void HandleDummy(SpellEffIndex /*effIndex*/)
144  {
145  if (Player* unitTarget = GetHitPlayer())
146  {
147  unitTarget->CastSpell(unitTarget, GetEffectValue(), true);
148  if (Unit* ghoul = unitTarget->GetCharm())
149  {
150  //health, mana, armor and resistance
151  PetLevelInfo const* pInfo = sObjectMgr->GetPetLevelInfo(ghoul->GetEntry(), ghoul->getLevel());
152  if (pInfo) // exist in DB
153  {
154  ghoul->SetCreateHealth(pInfo->health);
155  ghoul->SetModifierValue(UNIT_MOD_HEALTH, BASE_VALUE, pInfo->health);
156  ghoul->SetModifierValue(UNIT_MOD_ARMOR, BASE_VALUE, float(pInfo->armor));
157  for (uint8 stat = 0; stat < MAX_STATS; ++stat)
158  ghoul->SetCreateStat(Stats(stat), float(pInfo->stats[stat]));
159  }
160 
161  ghoul->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, float(ghoul->getLevel() - (ghoul->getLevel() / 4)));
162  ghoul->SetBaseWeaponDamage(BASE_ATTACK, MAXDAMAGE, float(ghoul->getLevel() + (ghoul->getLevel() / 4)));
163 
164  // Avoidance, Night of the Dead
165  if (Aura* aur = ghoul->AddAura(62137, ghoul))
166  if (AuraEffect* aurEff = GetCaster()->GetAuraEffect(SPELL_AURA_ADD_FLAT_MODIFIER, SPELLFAMILY_DEATHKNIGHT, 2718, 0))
167  if (aur->GetEffect(0))
168  aur->GetEffect(0)->SetAmount(-aurEff->GetSpellInfo()->Effects[EFFECT_2].CalcValue());
169 
170  // Hit / Expertise scalling, warlock / hunter pets have this by default
171  ghoul->AddAura(SPELL_HUNTER_PET_SCALING_04, ghoul);
172 
173  // DK Ghoul haste refresh
174  float val = (GetCaster()->m_modAttackSpeedPct[BASE_ATTACK] - 1.0f) * 100.0f;
175  ghoul->m_modAttackSpeedPct[BASE_ATTACK] = GetCaster()->m_modAttackSpeedPct[BASE_ATTACK];
176  ghoul->SetFloatValue(UNIT_FIELD_BASEATTACKTIME, 2000.0f);
177  ghoul->ApplyPercentModFloatValue(UNIT_FIELD_BASEATTACKTIME, val, true); // we want to reduce attack time
178 
179  // Strength + Stamina
180  for (uint8 i = STAT_STRENGTH; i <= STAT_STAMINA; ++i)
181  {
182  Stats stat = Stats(i);
183  if (stat != STAT_STRENGTH && stat != STAT_STAMINA)
184  continue;
185 
186  float value = 0.0f;
187  float mod = (stat == STAT_STAMINA ? 0.3f : 0.7f);
188 
189  // Check just if owner has Ravenous Dead since it's effect is not an aura
190  AuraEffect const* aurEff = GetCaster()->GetAuraEffect(SPELL_AURA_MOD_TOTAL_STAT_PERCENTAGE, SPELLFAMILY_DEATHKNIGHT, 3010, 0);
191  if (aurEff)
192  {
193  SpellInfo const* spellInfo = aurEff->GetSpellInfo(); // Then get the SpellProto and add the dummy effect value
194  AddPct(mod, spellInfo->Effects[EFFECT_1].CalcValue()); // Ravenous Dead edits the original scale
195  }
196  // Glyph of the Ghoul
197  aurEff = GetCaster()->GetAuraEffect(SPELL_DK_GLYPH_OF_THE_GHOUL, EFFECT_0);
198  if (aurEff)
199  mod += CalculatePct(1.0f, aurEff->GetAmount()); // Glyph of the Ghoul adds a flat value to the scale mod
200 
201  value = float(GetCaster()->GetStat(stat)) * mod;
202  value = ghoul->GetTotalStatValue(stat, value);
203  ghoul->SetStat(stat, int32(value));
204  ghoul->ApplyStatBuffMod(stat, value, true);
205  }
206 
207  // Attack Power
208  ghoul->SetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_VALUE, 589 + ghoul->GetStat(STAT_STRENGTH) + ghoul->GetStat(STAT_AGILITY));
209  ghoul->SetInt32Value(UNIT_FIELD_ATTACK_POWER, (int32)ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_VALUE) * ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, BASE_PCT));
210  ghoul->SetInt32Value(UNIT_FIELD_ATTACK_POWER_MODS, (int32)ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, TOTAL_VALUE));
211  ghoul->SetFloatValue(UNIT_FIELD_ATTACK_POWER_MULTIPLIER, ghoul->GetModifierValue(UNIT_MOD_ATTACK_POWER, TOTAL_PCT) - 1.0f);
212 
213  // Health
214  ghoul->SetModifierValue(UNIT_MOD_HEALTH, TOTAL_VALUE, (ghoul->GetStat(STAT_STAMINA) - ghoul->GetCreateStat(STAT_STAMINA)) * 10.0f);
215 
216  // Power Energy
217  ghoul->SetModifierValue(UnitMods(UNIT_MOD_POWER_START + static_cast<uint8>(POWER_ENERGY)), BASE_VALUE, ghoul->GetCreatePowers(POWER_ENERGY));
218  ghoul->UpdateAllStats();
219  ghoul->SetFullHealth();
220 
221  // Aura Immunities
222  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_MELEE_RANGED_HASTE, true, SPELL_BLOCK_TYPE_POSITIVE);
223  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MELEE_SLOW, true, SPELL_BLOCK_TYPE_POSITIVE);
224  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_STAT, true, SPELL_BLOCK_TYPE_POSITIVE);
225  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_TOTAL_STAT_PERCENTAGE, true, SPELL_BLOCK_TYPE_POSITIVE);
226  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_ATTACK_POWER, true, SPELL_BLOCK_TYPE_POSITIVE);
227  ghoul->ApplySpellImmune(0, IMMUNITY_STATE, SPELL_AURA_MOD_ATTACK_POWER_PCT, true, SPELL_BLOCK_TYPE_POSITIVE);
228  }
229  }
230  }
