378  OpenedCoofers += 1;

385  break;
386  case DATA_GOLEM_LORD_ARGELMACH_INIT:

576  void TombOfSevenEvent()
577  {
578  if (GhostKillCount < 7 && TombBossGUIDs[TombEventCounter])
579  {
580  if (Creature* boss = instance->GetCreature(TombBossGUIDs[TombEventCounter]))
581  {
582  ++TombEventCounter;
583  boss->SetFaction(FACTION_DARK_IRON_DWARVES);
584  boss->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC);
585 
586  // find suitable target here.
587  Player* target = boss->SelectNearestPlayer(130);
588  if (target && boss->CanCreatureAttack(target, true))
589  {
590  boss->AI()->AttackStart(target);
591  boss->AI()->DoZoneInCombat();
592  tombResetTimer = TIMER_TOMB_RESET;
593  }
594  }
595  }

598  void TombOfSevenReset()
599  {
600  HandleGameObject(GoTombExitGUID, false);// close exit door
601  HandleGameObject(GoTombEnterGUID, true);// open entrance door
602  for (uint8 i = 0; i < 7; ++i)
603  {
604  if (Creature* boss = instance->GetCreature(TombBossGUIDs[i]))
605  {
606  if (!boss->IsAlive())
607  {
608  //do not call EnterEvadeMode(), it will create infinit loops
609  boss->Respawn();
610  boss->RemoveAllAuras();
611  boss->DeleteThreatList();
612  boss->CombatStop(true);
613  boss->LoadCreaturesAddon(true);
614  boss->GetMotionMaster()->MoveTargetedHome();
615  boss->SetLootRecipient(nullptr);
616  }
617  boss->SetFaction(FACTION_FRIENDLY);
618  boss->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC); // think this is useless
619  if (i == 6) // doomrel needs explicit reset
620  {
621  boss->AI()->Reset();
622  }
623  }
624  }
625 
626  GhostKillCount = 0;
627  TombEventCounter = 0;
628  TombTimer = TIMER_TOMB_START;
629  SetData(TYPE_TOMB_OF_SEVEN, NOT_STARTED);
630  }

632  bool CheckTombReset(uint32 diff)
633  {
634  bool anyBossAlive = false; // status of the bosses up until the current one
635  for (uint8 i = 0; i < TombEventCounter; i++)
636  {
637  Creature* boss = instance->GetCreature(TombBossGUIDs[i]);
638  if (boss)
639  {
640  anyBossAlive |= boss->IsAlive();
641  if (boss->IsAlive() && boss->IsInCombat())
642  {
643  tombResetTimer = TIMER_TOMB_RESET;
644  return false; // any boss in combat means we shouldn't reset.
645  }
646  }
647  }
648  if (!anyBossAlive) // no boss alive, put reset timer back up
649  {
650  tombResetTimer = TIMER_TOMB_RESET;
651  }
652  tombResetTimer -= diff;
653  return tombResetTimer < diff;
654  }
655 
656  void Update(uint32 diff) override
657  {
658  if ((GetData(TYPE_TOMB_OF_SEVEN) == IN_PROGRESS) && GhostKillCount < 7)
659  {
660  if (TombTimer <= diff)
661  {
662  TombTimer = TIMER_TOMBOFTHESEVEN;
663  TombOfSevenEvent();
664  }
665  else
666  {
667  TombTimer -= diff;
668  }
669 
670  if (CheckTombReset(diff))
671  {
672  TombOfSevenReset();
673  }
674  }
675  }
