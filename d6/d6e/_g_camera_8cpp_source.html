32  const std::string& v = toUpper(it->value);

34  m_direction = HORIZONTAL;

39  }
40  } else if (k == "COORDINATEFRAME") {
41  m_cframe = it->value;
42  } else if (k == "FOVDEGREES") {
43  m_fieldOfView = toRadians(it->value.number());

45  m_nearPlaneZ = it->value;

184 void GCamera::getProjectUnitMatrix(const Rect2D& viewport, Matrix4& P) const {
185 
186  float screenWidth = viewport.width();
187  float screenHeight = viewport.height();
188 
189  float r, l, t, b, n, f, x, y;
190 
191  float s = 1.0f;
192  if (m_direction == VERTICAL) {
193  y = -m_nearPlaneZ * tan(m_fieldOfView / 2);
194  x = y * (screenWidth / screenHeight);
195  s = screenHeight;
196  } else { //m_direction == HORIZONTAL
197  x = -m_nearPlaneZ * tan(m_fieldOfView / 2);
198  y = x * (screenHeight / screenWidth);
199  s = screenWidth;
200  }
201 
202  n = -m_nearPlaneZ;
203  f = -m_farPlaneZ;
204  r = x - m_pixelOffset.x/s;
205  l = -x - m_pixelOffset.x/s;
206  t = y + m_pixelOffset.y/s;
207  b = -y + m_pixelOffset.y/s;
208 
209  P = Matrix4::perspectiveProjection(l, r, b, t, n, f);
210 }

240 Vector3 GCamera::unproject(const Vector3& v, const Rect2D& viewport) const {
241 
242  const float n = m_nearPlaneZ;
243  const float f = m_farPlaneZ;
244 
245  float z;
246 
247  if (-f >= finf()) {
248  // Infinite far plane
249  z = 1.0f / (((-1.0f / n) * v.z) + 1.0f / n);
250  } else {
251  z = 1.0f / ((((1.0f / f) - (1.0f / n)) * v.z) + 1.0f / n);
252  }
253 
254  const Ray& ray = worldRay(v.x - m_pixelOffset.x, v.y - m_pixelOffset.y, viewport);
255 
256  // Find out where the ray reaches the specified depth.
257  const Vector3& out = ray.origin() + ray.direction() * -z / (ray.direction().dot(m_cframe.lookVector()));
258 
259  return out;
260 }

292 void GCamera::frustum(const Rect2D& viewport, Frustum& fr) const {
293 
294  // The volume is the convex hull of the vertices definining the view
295  // frustum and the light source point at infinity.
296 
297  const float x = viewportWidth(viewport) / 2;
298  const float y = viewportHeight(viewport) / 2;
299  const float zn = m_nearPlaneZ;
300  const float zf = m_farPlaneZ;
301  float xx, zz, yy;
302 
303  float halfFOV = m_fieldOfView * 0.5f;
304 
305  // This computes the normal, which is based on the complement of the
306  // halfFOV angle, so the equations are "backwards"
307  if (m_direction == VERTICAL) {
308  yy = -cosf(halfFOV);
309  xx = yy * viewport.height() / viewport.width();
310  zz = -sinf(halfFOV);
311  } else {
312  xx = -cosf(halfFOV);
313  yy = xx * viewport.width() / viewport.height();
314  zz = -sinf(halfFOV);
315  }
316 
317  // Near face (ccw from UR)
318  fr.vertexPos.append(
319  Vector4( x, y, zn, 1),
320  Vector4(-x, y, zn, 1),
321  Vector4(-x, -y, zn, 1),
322  Vector4( x, -y, zn, 1));
323 
324  // Far face (ccw from UR, from origin)
325  if (m_farPlaneZ == -finf()) {
326  fr.vertexPos.append(Vector4( x, y, zn, 0),
327  Vector4(-x, y, zn, 0),
328  Vector4(-x, -y, zn, 0),
329  Vector4( x, -y, zn, 0));
330  } else {
331  // Finite
332  const float s = zf / zn;
333  fr.vertexPos.append(Vector4( x * s, y * s, zf, 1),
334  Vector4(-x * s, y * s, zf, 1),
335  Vector4(-x * s, -y * s, zf, 1),
336  Vector4( x * s, -y * s, zf, 1));
337  }
338 
339  Frustum::Face face;
340 
341  // Near plane (wind backwards so normal faces into frustum)
342  // Recall that nearPlane, farPlane are positive numbers, so
343  // we need to negate them to produce actual z values.
344  face.plane = Plane(Vector3(0,0,-1), Vector3(0,0,m_nearPlaneZ));
345  face.vertexIndex[0] = 3;
346  face.vertexIndex[1] = 2;
347  face.vertexIndex[2] = 1;
348  face.vertexIndex[3] = 0;
349  fr.faceArray.append(face);
350 
351  // Right plane
352  face.plane = Plane(Vector3(xx, 0, zz), Vector3::zero());
353  face.vertexIndex[0] = 0;
354  face.vertexIndex[1] = 4;
355  face.vertexIndex[2] = 7;
356  face.vertexIndex[3] = 3;
357  fr.faceArray.append(face);
358 
359  // Left plane
360  face.plane = Plane(Vector3(-fr.faceArray.last().plane.normal().x, 0, fr.faceArray.last().plane.normal().z), Vector3::zero());
361  face.vertexIndex[0] = 5;
362  face.vertexIndex[1] = 1;
363  face.vertexIndex[2] = 2;
364  face.vertexIndex[3] = 6;
365  fr.faceArray.append(face);
366 
367  // Top plane
368  face.plane = Plane(Vector3(0, yy, zz), Vector3::zero());
369  face.vertexIndex[0] = 1;
370  face.vertexIndex[1] = 5;
371  face.vertexIndex[2] = 4;
372  face.vertexIndex[3] = 0;
373  fr.faceArray.append(face);
374 
375  // Bottom plane
376  face.plane = Plane(Vector3(0, -fr.faceArray.last().plane.normal().y, fr.faceArray.last().plane.normal().z), Vector3::zero());
377  face.vertexIndex[0] = 2;
378  face.vertexIndex[1] = 3;
379  face.vertexIndex[2] = 7;
380  face.vertexIndex[3] = 6;
381  fr.faceArray.append(face);
382 
383  // Far plane
384  if (-m_farPlaneZ < finf()) {
385  face.plane = Plane(Vector3(0, 0, 1), Vector3(0, 0, m_farPlaneZ));
386  face.vertexIndex[0] = 4;
387  face.vertexIndex[1] = 5;
388  face.vertexIndex[2] = 6;
389  face.vertexIndex[3] = 7;
390  fr.faceArray.append(face);
391  }
392 
393  // Transform vertices to world space
394  for (int v = 0; v < fr.vertexPos.size(); ++v) {
395  fr.vertexPos[v] = m_cframe.toWorldSpace(fr.vertexPos[v]);
396  }
397 
398  // Transform planes to world space
399  for (int p = 0; p < fr.faceArray.size(); ++p) {
400  // Since there is no scale factor, we don't have to
401  // worry about the inverse transpose of the normal.
402  Vector3 normal;
403  float d;
404 
405  fr.faceArray[p].plane.getEquation(normal, d);
406 
407  Vector3 newNormal = m_cframe.rotation * normal;
408 
409  if (isFinite(d)) {
410  d = (newNormal * -d + m_cframe.translation).dot(newNormal);
411  fr.faceArray[p].plane = Plane(newNormal, newNormal * d);
412  } else {
413  // When d is infinite, we can't multiply 0's by it without
414  // generating NaNs.
415  fr.faceArray[p].plane = Plane::fromEquation(newNormal.x, newNormal.y, newNormal.z, d);
416  }
417  }
418 }
