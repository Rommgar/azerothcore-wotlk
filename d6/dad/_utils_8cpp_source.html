193  fclose(stream);

195  }

208 
209  // And write it in the file
210  size_t wrote = fwrite(data, size, 1, disk);
211  if (wrote != 1)

213  printf("SaveToDisk: Error writing to the file while trying to save %s to disk.\n", path.c_str());
214  fclose(stream);
215  fclose(disk);
216  return;
217  }
218 
219  // Close the filestream
220  fclose(disk);
221  fclose(stream);
222 
223  // Free the used memory
224  delete[] data;
