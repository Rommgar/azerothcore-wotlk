255  void UpdateAI(uint32 diff) override
256  {
257  if( ShieldTimer <= (int32)diff )
258  {
259  me->CastSpell(me, SPELL_MINIONS_DEFEND, true);
260  ShieldTimer = 5000;
261  }
262  else
263  ShieldTimer -= diff;
264 
265  if ( !UpdateVictim() )
266  return;
267 
268  events.Update(diff);
269 
270  if( me->HasUnitState(UNIT_STATE_CASTING) )
271  return;
272 
273  switch( events.ExecuteEvent() )
274  {
275  case 0:
276  break;
277  case EVENT_MOUNT_CHARGE:
278  {
279  GuidVector LIST;
280  Map::PlayerList const& pl = me->GetMap()->GetPlayers();
281  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
282  if( Player* plr = itr->GetSource() )
283  {
284  if( me->GetExactDist(plr) < 8.0f || me->GetExactDist(plr) > 25.0f || plr->isDead() )
285  continue;
286  if( !plr->GetVehicle() )
287  LIST.push_back(plr->GetGUID());
288  else if( Vehicle* v = plr->GetVehicle() )
289  {
290  if( Unit* mount = v->GetBase() )
291  LIST.push_back(mount->GetGUID());
292  }
293  }
294  if( !LIST.empty() )
295  {
296  uint8 rnd = LIST.size() > 1 ? urand(0, LIST.size() - 1) : 0;
297  if( Unit* target = ObjectAccessor::GetUnit(*me, LIST.at(rnd)) )
298  {
299  me->getThreatMgr().resetAllAggro();
300  me->AddThreat(target, 10000.0f);
301  AttackStart(target);
302  me->CastSpell(target, SPELL_MINIONS_CHARGE, false);
303  }
304  }
305  events.RepeatEvent(urand(4500, 6000));
306  }
307  break;
308  case EVENT_SHIELD_BREAKER:
309  {
310  GuidVector LIST;
311  Map::PlayerList const& pl = me->GetMap()->GetPlayers();
312  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
313  if( Player* plr = itr->GetSource() )
314  {
315  if( me->GetExactDist(plr) < 10.0f || me->GetExactDist(plr) > 30.0f )
316  continue;
317  if( Vehicle* v = plr->GetVehicle() )
318  if( Unit* mount = v->GetBase() )
319  LIST.push_back(mount->GetGUID());
320  }
321  if( !LIST.empty() )
322  {
323  uint8 rnd = LIST.size() > 1 ? urand(0, LIST.size() - 1) : 0;
324  if( Unit* target = ObjectAccessor::GetCreature(*me, LIST.at(rnd)) )
325  me->CastSpell(target, SPELL_NPC_SHIELD_BREAKER, false);
