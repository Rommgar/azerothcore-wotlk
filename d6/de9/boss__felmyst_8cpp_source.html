254  me->SendMovementFlagUpdate();

256  break;
257  case EVENT_INTRO_3:
258  me->GetMotionMaster()->MovePoint(POINT_AIR, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ() + 10.0f, false, true);
259  events2.ScheduleEvent(EVENT_INTRO_4, 2000);
260  break;
261  case EVENT_INTRO_4:
262  events.ScheduleEvent(EVENT_LAND, 3000, 1);
263  events.ScheduleEvent(EVENT_SPELL_BERSERK, 600000);
264  me->SetInCombatWithZone();
265  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
266  me->CastSpell(me, SPELL_NOXIOUS_FUMES, true);
267  me->GetMotionMaster()->MovePoint(POINT_MISC, 1472.18f, 603.38f, 34.0f, false, true);
268  break;
269  }
270 
271  if (!events2.Empty())
272  return;
273 
274  if (!UpdateVictim())
275  return;
276 
277  events.Update(diff);
278  if (me->HasUnitState(UNIT_STATE_CASTING))
279  return;
280 
281  switch(events.ExecuteEvent())
282  {
283  case EVENT_RESTORE_COMBAT:
284  me->SetReactState(REACT_AGGRESSIVE);

301  case EVENT_SPELL_CORROSION:

308  break;
