193  LfgCompatibility LFGQueue::FindNewGroups(const ObjectGuid& newGuid)
194  {
195  // each combination of dps+heal+tank (tank*8 + heal+4 + dps) has a value assigned 0..15
196  // first 16 bits of the mask are for marking if such combination was found once, second 16 bits for marking second occurence of that combination, etc
197  uint64 foundMask = 0;
198  uint32 foundCount = 0;
199 
200  LOG_DEBUG("lfg", "FIND NEW GROUPS for: %s", newGuid.ToString().c_str());
201 
202  // we have to take into account that FindNewGroups is called every X minutes if number of compatibles is low!
203  // build set of already present compatibles for this guid
204  std::set<Lfg5Guids> currentCompatibles;
205  for (Lfg5GuidsList::iterator it = CompatibleList.begin(); it != CompatibleList.end(); ++it)
206  if (it->hasGuid(newGuid))
207  {
208  // unset roles here so they are not copied, restore after insertion
209  LfgRolesMap* r = it->roles;
210  it->roles = nullptr;
211  currentCompatibles.insert(*it);
212  it->roles = r;
213  }
214 
215  LfgCompatibility selfCompatibility = LFG_COMPATIBILITY_PENDING;
216  if (currentCompatibles.empty())
217  {
218  selfCompatibility = CheckCompatibility(Lfg5Guids(), newGuid, foundMask, foundCount, currentCompatibles);
219  if (selfCompatibility != LFG_COMPATIBLES_WITH_LESS_PLAYERS) // group is already compatible (a party of 5 players)
220  return selfCompatibility;
221  }
222 
223  for (Lfg5GuidsList::iterator it = CompatibleList.begin(); it != CompatibleList.end(); )
224  {
225  Lfg5GuidsList::iterator itr = it++;
226  if (itr->empty())
227  {
228  LOG_DEBUG("lfg", "ERASE from CompatibleList");
229  CompatibleList.erase(itr);
230  continue;
231  }
232  LfgCompatibility compatibility = CheckCompatibility(*itr, newGuid, foundMask, foundCount, currentCompatibles);
233  if (compatibility == LFG_COMPATIBLES_MATCH)
234  return LFG_COMPATIBLES_MATCH;
235  if ((foundMask & 0x3FFF3FFF3FFF3FFF) == 0x3FFF3FFF3FFF3FFF) // each combination of dps+heal+tank already found 4 times
236  break;
237  }
238 
239  return selfCompatibility;
240  }

242  LfgCompatibility LFGQueue::CheckCompatibility(Lfg5Guids const& checkWith, const ObjectGuid& newGuid, uint64& foundMask, uint32& foundCount, const std::set<Lfg5Guids>& currentCompatibles)
243  {
244  LOG_DEBUG("lfg", "CHECK CheckCompatibility: %s, new guid: %s", checkWith.toString().c_str(), newGuid.ToString().c_str());
245  Lfg5Guids check(checkWith, false); // here newGuid is at front
246  Lfg5Guids strGuids(checkWith, false); // here guids are sorted
247  check.force_insert_front(newGuid);
248  strGuids.insert(newGuid);
249 
250  if (!currentCompatibles.empty() && currentCompatibles.find(strGuids) != currentCompatibles.end())
251  return LFG_INCOMPATIBLES_TOO_MUCH_PLAYERS;
252 
253  LfgProposal proposal;
254  LfgDungeonSet proposalDungeons;
255  LfgGroupsMap proposalGroups;
256  LfgRolesMap proposalRoles;
257 
258  // Check if more than one LFG group and number of players joining
259  uint8 numPlayers = 0;
260  uint8 numLfgGroups = 0;
261  ObjectGuid guid;
262  uint64 addToFoundMask = 0;
263 
264  for (uint8 i = 0; i < 5 && !(guid = check.guids[i]).IsEmpty() && numLfgGroups < 2 && numPlayers <= MAXGROUPSIZE; ++i)
265  {
266  LfgQueueDataContainer::iterator itQueue = QueueDataStore.find(guid);
267  if (itQueue == QueueDataStore.end())
268  {
269  LOG_ERROR("lfg", "LFGQueue::CheckCompatibility: [%s] is not queued but listed as queued!", guid.ToString().c_str());
270  RemoveFromQueue(guid);
271  return LFG_COMPATIBILITY_PENDING;
272  }
273 
274  // Store group so we don't need to call Mgr to get it later (if it's player group will be 0 otherwise would have joined as group)
275  for (LfgRolesMap::const_iterator it2 = itQueue->second.roles.begin(); it2 != itQueue->second.roles.end(); ++it2)
276  proposalGroups[it2->first] = itQueue->first.IsGroup() ? itQueue->first : ObjectGuid::Empty;
277 
278  numPlayers += itQueue->second.roles.size();
279 
280  if (sLFGMgr->IsLfgGroup(guid))
281  {
282  if (!numLfgGroups)
283  proposal.group = guid;
284  ++numLfgGroups;
285  }
286  }
287 
288  if (numLfgGroups > 1)
289  return LFG_INCOMPATIBLES_MULTIPLE_LFG_GROUPS;
290 
291  // Group with less that MAXGROUPSIZE members always compatible
292  if (!sLFGMgr->IsTesting() && check.size() == 1 && numPlayers < MAXGROUPSIZE)
293  {
294  LfgQueueDataContainer::iterator itQueue = QueueDataStore.find(check.front());
295  LfgRolesMap roles = itQueue->second.roles;
296  uint8 roleCheckResult = LFGMgr::CheckGroupRoles(roles);
297  strGuids.addRoles(roles);
298  itQueue->second.bestCompatible.clear(); // this may be left after a failed proposal (not cleared, because UpdateQueueTimers would try to generate it with every update)
299  //UpdateBestCompatibleInQueue(itQueue, strGuids);
300  AddToCompatibles(strGuids);
301  if (roleCheckResult && roleCheckResult <= 15)
302  foundMask |= ( (((uint64)1) << (roleCheckResult - 1)) | (((uint64)1) << (16 + roleCheckResult - 1)) | (((uint64)1) << (32 + roleCheckResult - 1)) | (((uint64)1) << (48 + roleCheckResult - 1)) );
303  return LFG_COMPATIBLES_WITH_LESS_PLAYERS;
304  }
305 
306  if (numPlayers > MAXGROUPSIZE)
307  return LFG_INCOMPATIBLES_TOO_MUCH_PLAYERS;
308 
309  // If it's single group no need to check for duplicate players, ignores, bad roles or bad dungeons as it's been checked before joining
310  if (check.size() > 1)
311  {
312  for (uint8 i = 0; i < 5 && check.guids[i]; ++i)
313  {
314  const LfgRolesMap& roles = QueueDataStore[check.guids[i]].roles;
315  for (LfgRolesMap::const_iterator itRoles = roles.begin(); itRoles != roles.end(); ++itRoles)
316  {
317  LfgRolesMap::const_iterator itPlayer;
318  for (itPlayer = proposalRoles.begin(); itPlayer != proposalRoles.end(); ++itPlayer)
319  {
320  if (itRoles->first == itPlayer->first)
321  {
322  // pussywizard: LFG ZOMG! this means that this player was in two different LfgQueueData (in QueueDataStore), and at least one of them is a group guid, because we do checks so there aren't 2 same guids in current CHECK
323  //LOG_ERROR("lfg", "LFGQueue::CheckCompatibility: ERROR! Player multiple times in queue! [%s]", itRoles->first.ToString().c_str());
324  break;
325  }
326  else if (sLFGMgr->HasIgnore(itRoles->first, itPlayer->first))
327  break;
328  }
329  if (itPlayer == proposalRoles.end())
330  proposalRoles[itRoles->first] = itRoles->second;
331  else
332  break;
333  }
334  }
335 
336  if (numPlayers != proposalRoles.size())
337  return LFG_INCOMPATIBLES_HAS_IGNORES;
338 
339  uint8 roleCheckResult = LFGMgr::CheckGroupRoles(proposalRoles);
340  if (!roleCheckResult || roleCheckResult > 0xF)
341  return LFG_INCOMPATIBLES_NO_ROLES;
342 
343  // now, every combination can occur only 4 times (explained in FindNewGroups)
344  if (foundMask & (((uint64)1) << (roleCheckResult - 1)))
345  {
346  if (foundMask & (((uint64)1) << (16 + roleCheckResult - 1)))
347  {
348  if (foundMask & (((uint64)1) << (32 + roleCheckResult - 1)))
349  {
350  if (foundMask & (((uint64)1) << (48 + roleCheckResult - 1)))
351  {
352  if (foundCount >= 10) // but only after finding at least 10 compatibles (this helps when there are few groups)
353  return LFG_INCOMPATIBLES_NO_ROLES;
354  }
355  else
356  addToFoundMask |= (((uint64)1) << (48 + roleCheckResult - 1));
357  }
358  else
359  addToFoundMask |= (((uint64)1) << (32 + roleCheckResult - 1));
360  }
361  else
362  addToFoundMask |= (((uint64)1) << (16 + roleCheckResult - 1));
363  }
364  else
365  addToFoundMask |= (((uint64)1) << (roleCheckResult - 1));
366 
367  proposalDungeons = QueueDataStore[check.front()].dungeons;
368  for (uint8 i = 1; i < 5 && check.guids[i]; ++i)
369  {
370  LfgDungeonSet temporal;
371  LfgDungeonSet& dungeons = QueueDataStore[check.guids[i]].dungeons;
372  std::set_intersection(proposalDungeons.begin(), proposalDungeons.end(), dungeons.begin(), dungeons.end(), std::inserter(temporal, temporal.begin()));
373  proposalDungeons = temporal;
374  }
375 
376  if (proposalDungeons.empty())
377  return LFG_INCOMPATIBLES_NO_DUNGEONS;
378  }
379  else
380  {
381  ObjectGuid gguid = check.front();
382  const LfgQueueData& queue = QueueDataStore[gguid];
383  proposalDungeons = queue.dungeons;
384  proposalRoles = queue.roles;
385  LFGMgr::CheckGroupRoles(proposalRoles); // assing new roles
386  }
387 
388  // Enough players?
389  if (!sLFGMgr->IsTesting() && numPlayers != MAXGROUPSIZE)
390  {
391  strGuids.addRoles(proposalRoles);
392  for (uint8 i = 0; i < 5 && check.guids[i]; ++i)
393  {
394  LfgQueueDataContainer::iterator itr = QueueDataStore.find(check.guids[i]);
395  if (!itr->second.bestCompatible.empty()) // update if groups don't have it empty (for empty it will be generated in UpdateQueueTimers)
396  UpdateBestCompatibleInQueue(itr, strGuids);
397  }
398  AddToCompatibles(strGuids);
399  foundMask |= addToFoundMask;
400  ++foundCount;
401  return LFG_COMPATIBLES_WITH_LESS_PLAYERS;
402  }
403 
404  proposal.queues = strGuids;
405  proposal.isNew = numLfgGroups != 1 || sLFGMgr->GetOldState(proposal.group) != LFG_STATE_DUNGEON;
406 
407  if (!sLFGMgr->AllQueued(check)) // can't create proposal
408  return LFG_COMPATIBILITY_PENDING;
409 
410  // Create a new proposal
411  proposal.cancelTime = time(nullptr) + LFG_TIME_PROPOSAL;
412  proposal.state = LFG_PROPOSAL_INITIATING;
413  proposal.leader.Clear();
414  proposal.dungeonId = Acore::Containers::SelectRandomContainerElement(proposalDungeons);
415 
416  uint32 completedEncounters = 0;
417  bool leader = false;
418  for (LfgRolesMap::const_iterator itRoles = proposalRoles.begin(); itRoles != proposalRoles.end(); ++itRoles)
419  {
420  // Assing new leader
421  if (itRoles->second & PLAYER_ROLE_LEADER)
422  {
423  if (!leader || !proposal.leader || urand(0, 1))
424  proposal.leader = itRoles->first;
425  leader = true;
426  }
427  else if (!leader && (!proposal.leader || urand(0, 1)))
428  proposal.leader = itRoles->first;
429 
430  // Assing player data and roles
431  LfgProposalPlayer& data = proposal.players[itRoles->first];
432  data.role = itRoles->second;
433  data.group = proposalGroups.find(itRoles->first)->second;
434  if (!proposal.isNew && data.group && data.group == proposal.group) // Player from existing group, autoaccept
435  data.accept = LFG_ANSWER_AGREE;
436 
437  if (!completedEncounters && !proposal.isNew)
438  {
439  if (LFGDungeonEntry const* dungeon = sLFGDungeonStore.LookupEntry(proposal.dungeonId))
440  {
441  if (Player* player = ObjectAccessor::FindConnectedPlayer(itRoles->first))
442  {
443  if (player->GetMapId() == static_cast<uint32>(dungeon->map))
444  {
445  if (InstanceScript* instance = player->GetInstanceScript())
446  {
447  completedEncounters = instance->GetCompletedEncounterMask();
448  }
449  }
450  }
451  }
452  }
453  }
454 
455  proposal.encounters = completedEncounters;
456 
457  for (uint8 i = 0; i < 5 && proposal.queues.guids[i]; ++i)
458  RemoveFromQueue(proposal.queues.guids[i], true);
459 
460  sLFGMgr->AddProposal(proposal);
461 
462  return LFG_COMPATIBLES_MATCH;
463  }

465  void LFGQueue::UpdateQueueTimers(uint32 diff)
466  {
467  time_t currTime = time(nullptr);
468  bool sendQueueStatus = false;
469 
470  if (m_QueueStatusTimer > LFG_QUEUEUPDATE_INTERVAL)
471  {
472  m_QueueStatusTimer = 0;
473  sendQueueStatus = true;
474  }
475  else
476  m_QueueStatusTimer += diff;
477 
478  LOG_DEBUG("lfg", "UPDATE UpdateQueueTimers");
479  for (Lfg5GuidsList::iterator it = CompatibleList.begin(); it != CompatibleList.end(); )
480  {
481  Lfg5GuidsList::iterator itr = it++;
482  if (itr->empty())
483  {
484  LOG_DEBUG("lfg", "UpdateQueueTimers ERASE compatible");
485  CompatibleList.erase(itr);
486  }
487  }
488 
489  if (!sendQueueStatus)
490  {
491  for (LfgQueueDataContainer::iterator itQueue = QueueDataStore.begin(); itQueue != QueueDataStore.end(); )
492  {
493  if (currTime - itQueue->second.joinTime > 2 * HOUR)
494  {
495  ObjectGuid guid = itQueue->first;
496  QueueDataStore.erase(itQueue++);
497  sLFGMgr->LeaveAllLfgQueues(guid, true);
498  continue;
499  }
500  if (itQueue->second.bestCompatible.empty())
501  {
502  uint32 numOfCompatibles = FindBestCompatibleInQueue(itQueue);
503  if (numOfCompatibles /*must be positive, because proposals don't delete QueueQueueData*/ && currTime - itQueue->second.lastRefreshTime >= 60 && numOfCompatibles < (5 - itQueue->second.bestCompatible.roles->size()) * 25)
504  {
505  itQueue->second.lastRefreshTime = currTime;
506  AddToQueue(itQueue->first, false);
507  }
508  }
509  ++itQueue;
510  }
511  return;
512  }
513 
514  // LOG_TRACE("lfg", "Updating queue timers...");
515  for (LfgQueueDataContainer::iterator itQueue = QueueDataStore.begin(); itQueue != QueueDataStore.end(); ++itQueue)
516  {
517  LfgQueueData& queueinfo = itQueue->second;
518  uint32 dungeonId = (*queueinfo.dungeons.begin());
519  uint32 queuedTime = uint32(currTime - queueinfo.joinTime);
520  uint8 role = PLAYER_ROLE_NONE;
521  int32 waitTime = -1;
522  int32 wtTank = waitTimesTankStore[dungeonId].time;
523  int32 wtHealer = waitTimesHealerStore[dungeonId].time;
524  int32 wtDps = waitTimesDpsStore[dungeonId].time;
525  int32 wtAvg = waitTimesAvgStore[dungeonId].time;
526 
527  for (LfgRolesMap::const_iterator itPlayer = queueinfo.roles.begin(); itPlayer != queueinfo.roles.end(); ++itPlayer)
528  role |= itPlayer->second;
529  role &= ~PLAYER_ROLE_LEADER;
530 
531  switch (role)
532  {
533  case PLAYER_ROLE_NONE: // Should not happen - just in case
534  waitTime = -1;
535  break;
536  case PLAYER_ROLE_TANK:
537  waitTime = wtTank;
538  break;
539  case PLAYER_ROLE_HEALER:
540  waitTime = wtHealer;
541  break;
542  case PLAYER_ROLE_DAMAGE:
543  waitTime = wtDps;
544  break;
545  default:
546  waitTime = wtAvg;
547  break;
548  }
549 
550  if (queueinfo.bestCompatible.empty())
551  {
552  LOG_DEBUG("lfg", "found empty bestCompatible");
553  FindBestCompatibleInQueue(itQueue);
554  }
555 
556  LfgQueueStatusData queueData(dungeonId, waitTime, wtAvg, wtTank, wtHealer, wtDps, queuedTime, queueinfo.tanks, queueinfo.healers, queueinfo.dps);
