144  void HandleEffectRemove(AuraEffect const* /*aurEff*/, AuraEffectHandleModes /*mode*/)
145  {
146  Creature* creatureTarget = GetTarget()->ToCreature();
147  if (!creatureTarget)
148  {
149  return;
150  }
151 
152  creatureTarget->RemoveAurasDueToSpell(SPELL_PLAY_DEAD_PACIFY);
153  creatureTarget->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
154  creatureTarget->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
155  //creatureTarget->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
156  creatureTarget->SetControlled(false, UNIT_STATE_ROOT);
157  creatureTarget->SetReactState(REACT_AGGRESSIVE);
158 
159  if (!creatureTarget->IsInCombat())
160  {
161  return;
162  }
163 
164  bool shouldDie = true;
165  std::list<Creature*> hounds;
166  creatureTarget->GetCreaturesWithEntryInRange(hounds, 80.0f, NPC_CORE_HOUND);
167 
168  // Perform lambda based check to find if there is any nearby
169  if (!hounds.empty())
170  {
171  // Alive hound been found within 80 yards -> cancel suicide
172  if (std::find_if(hounds.begin(), hounds.end(), [creatureTarget](Creature const* hound)
173  {
174  return creatureTarget != hound && creatureTarget->IsWithinLOSInMap(hound) && hound->IsAlive() && hound->IsInCombat() && !hound->HasAura(SPELL_PLAY_DEAD);
175  }) != hounds.end())
176  {
177  shouldDie = false;
178  }
179  }
180 
181  if (!shouldDie)
182  {
183  if (CreatureAI* targetAI = creatureTarget->AI())
184  {
185  targetAI->DoCastSelf(SPELL_FIRE_NOVA_VISUAL, true);
186  targetAI->DoCastSelf(SPELL_FULL_HEALTH, true);
187  targetAI->Talk(EMOTE_IGNITE);
188  }
189  }
190  else
191  {
192  Unit::Kill(creatureTarget, creatureTarget);
193  }
194  }
