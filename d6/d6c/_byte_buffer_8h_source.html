387  void readPackGUID(uint64& guid)
388  {
389  if (rpos() + 1 > size())
390  {
391  throw ByteBufferPositionException(false, _rpos, 1, size());
392  }
393 
394  guid = 0;
395 
396  uint8 guidmark = 0;
397  (*this) >> guidmark;
398 
399  for (int i = 0; i < 8; ++i)
400  {
401  if (guidmark & (uint8(1) << i))
