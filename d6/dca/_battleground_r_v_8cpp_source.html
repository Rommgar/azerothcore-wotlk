54 void BattlegroundRV::CheckPositionForUnit(Unit* unit)
55 {
56  // get height at current pos, if something is wrong (below or high above) - teleport
57  if (!unit->IsFalling() && unit->IsAlive())
58  {
59  float groundZ_vmap = unit->GetMap()->GetHeight(unit->GetPositionX(), unit->GetPositionY(), 37.0f, true, 50.0f);
60  float groundZ_dyntree = unit->GetMap()->GetDynamicMapTree().getHeight(unit->GetPositionX(), unit->GetPositionY(), 37.0f, 50.0f, unit->GetPhaseMask());
61  if ((groundZ_vmap > 28.0f && groundZ_vmap < 29.0f) || (groundZ_dyntree > 28.0f && groundZ_dyntree < 37.0f))
62  {
63  float groundZ = std::max<float>(groundZ_vmap, groundZ_dyntree);
64  if (unit->GetPositionZ() < groundZ - 0.2f || unit->GetPositionZ() > groundZ + 3.5f)
65  TeleportUnitToNewZ(unit, groundZ + 1.0f, true);
66  }
67  }
68 }

70 void BattlegroundRV::PostUpdateImpl(uint32 diff)
71 {
72  if (GetStatus() != STATUS_IN_PROGRESS)
73  return;
74 
75  if (getTimer() < diff)
76  {
77  switch (getState())
78  {
79  case BG_RV_STATE_OPEN_FENCES:
80  for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
81  DoorOpen(i);
82  setTimer(BG_RV_CLOSE_FIRE_TIMER);
83  setState(BG_RV_STATE_CLOSE_FIRE);
84 
85  for (auto itr = m_Players.begin(); itr != m_Players.end(); ++itr)
86  if (Player* player = itr->second)
87  {
88  // Demonic Circle Summon
89  if (GameObject* gObj = player->GetGameObject(48018))
90  {
91  gObj->Relocate(gObj->GetPositionX(), gObj->GetPositionY(), 28.28f);
92  gObj->UpdateObjectVisibility(true);
93  }
94 
95  if (player->GetPositionZ() < 27.0f)
96  TeleportUnitToNewZ(player, 28.28f, true);
97 
98  for (uint8 i = SUMMON_SLOT_TOTEM; i < MAX_TOTEM_SLOT; ++i)
99  if (player->m_SummonSlot[i])
100  if (Creature* totem = GetBgMap()->GetCreature(player->m_SummonSlot[i]))
101  if (totem->GetPositionZ() < 28.0f)
102  TeleportUnitToNewZ(totem, 28.28f, true);
103 
104  for (auto itr2 = player->m_Controlled.begin(); itr2 != player->m_Controlled.end(); ++itr2)
105  {
106  if ((*itr2)->GetPositionZ() < 28.0f)
107  TeleportUnitToNewZ((*itr2), 28.28f, true);
108 
109  // Xinef: override stay position
110  if (CharmInfo* charmInfo = (*itr2)->GetCharmInfo())
111  if (charmInfo->IsAtStay())
112  {
113  (*itr2)->StopMovingOnCurrentPos();
114  charmInfo->SaveStayPosition(false);
115  }
116  }
117  }
118 
119  // fix ground on elevators (so aoe spells can be casted there)
120  {
121  uint32 objects[2] = {BG_RV_OBJECT_ELEVATOR_1, BG_RV_OBJECT_ELEVATOR_2};
122  for (uint8 i = 0; i < 2; ++i)
123  if (GameObject* go = GetBGObject(objects[i]))
124  go->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_TRANSPORT);
125  }
126  break;
127  case BG_RV_STATE_CLOSE_FIRE:
128  for (uint8 i = BG_RV_OBJECT_FIRE_1; i <= BG_RV_OBJECT_FIREDOOR_2; ++i)
129  DoorClose(i);
130  // Fire got closed after five seconds, leaves twenty seconds before toggling pillars
131  setTimer(BG_RV_FIRE_TO_PILLAR_TIMER);
132  setState(BG_RV_STATE_SWITCH_PILLARS);
133  break;
134  case BG_RV_STATE_SWITCH_PILLARS:
135  UpdatePillars();
136  setTimer(BG_RV_PILLAR_SWITCH_TIMER);
137  break;
138  }
139  }
140  else
141  setTimer(getTimer() - diff);
142 
143  if (getState() == BG_RV_STATE_OPEN_FENCES)
144  return;
145 
146  if (CheckPlayersTimer <= diff)
147  {
148  CheckPlayersTimer = 0;
149  for (BattlegroundPlayerMap::iterator itr = m_Players.begin(); itr != m_Players.end(); ++itr)
150  CheckPositionForUnit(itr->second);
151 
152  // maybe for pets and m_Controlled also, but not really necessary
153  }
154  else
155  CheckPlayersTimer -= diff;
156 }
