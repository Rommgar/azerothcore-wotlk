245  void UpdateAI(uint32 diff) override
246  {
247  if (timerSpawnPlate)
248  {
249  timerSpawnPlate += diff;
250  if (timerSpawnPlate >= 1000)
251  {
252  if (Vehicle* table = me->GetVehicle())
253  if (Unit* plateHolder = table->GetPassenger(6))
254  {
255  SeatMap::const_iterator itr = table->GetSeatIteratorForPassenger(me);
256  if (itr == table->Seats.end())
257  return;
258 
259  uint8 vehicleSeatId = itr->first;
260  Creature* plate = me->SummonCreature(NPC_STRUDY_PLATE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.0f);
261  if (!plate)
262  return;
263 
264  plateGUID = plate->GetGUID();
265  plate->EnterVehicle(plateHolder, vehicleSeatId);
266  timerSpawnPlate = 0;
267  timerRotateChair = 1;
268  }
269  }
270  }
271  if (timerRotateChair)
272  {
273  timerRotateChair += diff;
274  if (timerRotateChair >= 1000)
275  {
276  if (Creature* plate = GetPlate())
277  me->SetFacingToObject(plate);
278 
279  timerRotateChair = 0;
280  }
281  }
282  }
