80  bool OnGossipHello(Player* player, Creature* creature) override
81  {

89  AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_BEGIN_ALLY, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
90  else if (RageEncounter == DONE && AnetheronEncounter == NOT_STARTED)
91  AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_ANETHERON, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
92  else if (RageEncounter == DONE && AnetheronEncounter == DONE)
93  AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_RETREAT, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
94 
95  if (player->IsGameMaster())
96  AddGossipItemFor(player, GOSSIP_ICON_TRAINER, GOSSIP_ITEM_GM1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
97 
98  SendGossipMenuFor(player, 907, creature->GetGUID());
99  return true;
100  }
