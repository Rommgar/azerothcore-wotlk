215  void Load(const char* str) override
216  {
217  if (!str)
218  {
219  OUT_LOAD_INST_DATA_FAIL;
220  return;
221  }
222 
223  OUT_LOAD_INST_DATA(str);
224 
225  char dataHead1, dataHead2;
226 
227  std::istringstream loadStream(str);
228  loadStream >> dataHead1 >> dataHead2;
229 
230  if (dataHead1 == 'M' && dataHead2 == 'E')
231  {
232  for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
233  {
234  uint32 tmpState;
235  loadStream >> tmpState;
236  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
237  tmpState = NOT_STARTED;
238  SetBossState(i, EncounterState(tmpState));
239  }
240 
241  loadStream >> _passageEncounter;
242  if (_passageEncounter == ENCOUNTER_PASSAGE_DONE)
243  _passageEncounter = ENCOUNTER_PASSAGE_PHASE6;
244  }
245  else
246  OUT_LOAD_INST_DATA_FAIL;
247 
248  OUT_LOAD_INST_DATA_COMPLETE;
249  }
