29 void WorldSession::HandleGMTicketCreateOpcode(WorldPacket& recvData)
30 {
31  // Don't accept tickets if the ticket queue is disabled. (Ticket UI is greyed out but not fully dependable)
32  if (sTicketMgr->GetStatus() == GMTICKET_QUEUE_STATUS_DISABLED)
33  return;
34 
35  if (GetPlayer()->getLevel() < sWorld->getIntConfig(CONFIG_TICKET_LEVEL_REQ))
36  {
37  SendNotification(GetAcoreString(LANG_TICKET_REQ), sWorld->getIntConfig(CONFIG_TICKET_LEVEL_REQ));
38  return;
39  }
40 
41  GMTicketResponse response = GMTICKET_RESPONSE_CREATE_ERROR;
42  GmTicket* ticket = sTicketMgr->GetTicketByPlayer(GetPlayer()->GetGUID());
43 
44  if (ticket && ticket->IsCompleted())
45  sTicketMgr->CloseTicket(ticket->GetId(), GetPlayer()->GetGUID());
46 
47  // Player must not have ticket
48  if (!ticket || ticket->IsClosed())
49  {
50  uint32 mapId;
51  float x, y, z;
52  std::string message;
53  uint32 needResponse;
54  bool needMoreHelp;
55  uint32 count;
56  std::list<uint32> times;
57  uint32 decompressedSize;
58  std::string chatLog;
59 
60  recvData >> mapId;
61  recvData >> x >> y >> z;
62  recvData >> message;
63 
64  if (!ValidateHyperlinksAndMaybeKick(message))
65  {
66  return;
67  }
68 
69  recvData >> needResponse;
70  recvData >> needMoreHelp;
71 
72  recvData >> count;
73 
74  for (uint32 i = 0; i < count; i++)
75  {
76  uint32 time;
77  recvData >> time;
78  times.push_back(time);
79  }
80 
81  recvData >> decompressedSize;
82 
83  if (count && decompressedSize && decompressedSize < 0xFFFF)
84  {
85  uint32 pos = recvData.rpos();
86  ByteBuffer dest;
87  dest.resize(decompressedSize);
88 
89  uLongf realSize = decompressedSize;
90  if (uncompress(dest.contents(), &realSize, recvData.contents() + pos, recvData.size() - pos) == Z_OK)
91  {
92  dest >> chatLog;
93  }
94  else
95  {
96  LOG_ERROR("network.opcode", "CMSG_GMTICKET_CREATE possibly corrupt. Uncompression failed.");
97  recvData.rfinish();
98  return;
99  }
100 
101  recvData.rfinish(); // Will still have compressed data in buffer.
102  }
103 
104  if (!chatLog.empty() && !ValidateHyperlinksAndMaybeKick(chatLog))
105  {
106  return;
107  }
108 
109  ticket = new GmTicket(GetPlayer());
110  ticket->SetPosition(mapId, x, y, z);
111  ticket->SetMessage(message);
112  ticket->SetGmAction(needResponse, needMoreHelp);
113 
114  if (!chatLog.empty())
115  ticket->SetChatLog(times, chatLog);
116 
117  sTicketMgr->AddTicket(ticket);
118  sTicketMgr->UpdateLastChange();
119 
120  sWorld->SendGMText(LANG_COMMAND_TICKETNEW, GetPlayer()->GetName().c_str(), ticket->GetId());
121 
122  response = GMTICKET_RESPONSE_CREATE_SUCCESS;
123  }
124 
125  WorldPacket data(SMSG_GMTICKET_CREATE, 4);
126  data << uint32(response);
127  SendPacket(&data);
128 }

196 void WorldSession::HandleGMSurveySubmit(WorldPacket& recv_data)
197 {
198  uint32 nextSurveyID = sTicketMgr->GetNextSurveyID();
199  // just put the survey into the database
200  uint32 mainSurvey; // GMSurveyCurrentSurvey.dbc, column 1 (all 9) ref to GMSurveySurveys.dbc
201  recv_data >> mainSurvey;
202 
203  std::unordered_set<uint32> surveyIds;
204  CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
205 
206  // sub_survey1, r1, comment1, sub_survey2, r2, comment2, sub_survey3, r3, comment3, sub_survey4, r4, comment4, sub_survey5, r5, comment5, sub_survey6, r6, comment6, sub_survey7, r7, comment7, sub_survey8, r8, comment8, sub_survey9, r9, comment9, sub_survey10, r10, comment10,
207  for (uint8 i = 0; i < 10; i++)
208  {
209  uint32 subSurveyId; // ref to i'th GMSurveySurveys.dbc field (all fields in that dbc point to fields in GMSurveyQuestions.dbc)
210  recv_data >> subSurveyId;
211  if (!subSurveyId)
212  break;
213 
214  uint8 rank; // probably some sort of ref to GMSurveyAnswers.dbc
215  recv_data >> rank;
216  std::string comment; // comment ("Usage: GMSurveyAnswerSubmit(question, rank, comment)")
217  recv_data >> comment;
218 
219  if (!ValidateHyperlinksAndMaybeKick(comment))
220  {
221  return;
222  }
223 
224  // make sure the same sub survey is not added to DB twice
225  if (!surveyIds.insert(subSurveyId).second)
226  continue;
227 
228  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GM_SUBSURVEY);
229  stmt->setUInt32(0, nextSurveyID);
230  stmt->setUInt32(1, subSurveyId);
231  stmt->setUInt32(2, rank);
232  stmt->setString(3, comment);
233  trans->Append(stmt);
234  }
235 
236  std::string comment; // just a guess
237  recv_data >> comment;
238 
239  if (!ValidateHyperlinksAndMaybeKick(comment))
240  {
241  return;
242  }
243 
244  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_GM_SURVEY);
245  stmt->setUInt32(0, GetPlayer()->GetGUID().GetCounter());
246  stmt->setUInt32(1, nextSurveyID);
247  stmt->setUInt32(2, mainSurvey);
248  stmt->setString(3, comment);
249 
250  trans->Append(stmt);
251 
252  CharacterDatabase.CommitTransaction(trans);
253 }
