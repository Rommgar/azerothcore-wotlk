66 Guild* GuildMgr::GetGuildByName(const std::string& guildName) const
67 {
68  std::string search = guildName;

70  for (GuildContainer::const_iterator itr = GuildStore.begin(); itr != GuildStore.end(); ++itr)

72  std::string gname = itr->second->GetName();
73  std::transform(gname.begin(), gname.end(), gname.begin(), ::toupper);

76  }
