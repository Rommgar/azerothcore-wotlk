263 static inline const uchar *skip_trailing_space(const uchar *ptr,size_t len)
264 {
265  const uchar *end= ptr + len;
266 
267  if (len > 20)
268  {
269  const uchar *end_words= (const uchar *)(intptr)
270  (((ulonglong)(intptr)end) / SIZEOF_INT * SIZEOF_INT);
271  const uchar *start_words= (const uchar *)(intptr)
272  ((((ulonglong)(intptr)ptr) + SIZEOF_INT - 1) / SIZEOF_INT * SIZEOF_INT);
273 
274  DBUG_ASSERT(((ulonglong)(intptr)ptr) >= SIZEOF_INT);
275  if (end_words > ptr)
276  {
277  while (end > end_words && end[-1] == 0x20)
278  end--;
279  if (end[-1] == 0x20 && start_words < end_words)
280  while (end > start_words && ((unsigned *)end)[-1] == SPACE_INT)
281  end -= SIZEOF_INT;
282  }
283  }
284  while (end > ptr && end[-1] == 0x20)
285  end--;
286  return (end);
287 }
