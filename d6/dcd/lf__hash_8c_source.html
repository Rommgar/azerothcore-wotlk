60 /*
61  DESCRIPTION
62  Search for hashnr/key/keylen in the list starting from 'head' and
63  position the cursor. The list is ORDER BY hashnr, key
64 
65  RETURN
66  0 - not found
67  1 - found
68 
69  NOTE
70  cursor is positioned in either case
71  pins[0..2] are used, they are NOT removed on return
72 */
73 static int lfind(LF_SLIST * volatile *head, CHARSET_INFO *cs, uint32 hashnr,
74  const uchar *key, uint keylen, CURSOR *cursor, LF_PINS *pins)
75 {
76  uint32 cur_hashnr;
77  const uchar *cur_key;
78  uint cur_keylen;
79  intptr link;
80 
81 retry:
82  cursor->prev= (intptr *)head;
83  do { /* PTR() isn't necessary below, head is a dummy node */
84  cursor->curr= (LF_SLIST *)(*cursor->prev);
85  _lf_pin(pins, 1, cursor->curr);
86  } while (*cursor->prev != (intptr)cursor->curr && LF_BACKOFF);
87  for (;;)
88  {
89  if (unlikely(!cursor->curr))
90  return 0; /* end of the list */
91  do {
92  /* QQ: XXX or goto retry ? */
93  link= cursor->curr->link;
94  cursor->next= PTR(link);
95  _lf_pin(pins, 0, cursor->next);
96  } while (link != cursor->curr->link && LF_BACKOFF);
97  cur_hashnr= cursor->curr->hashnr;
98  cur_key= cursor->curr->key;
99  cur_keylen= cursor->curr->keylen;
100  if (*cursor->prev != (intptr)cursor->curr)
101  {
102  (void)LF_BACKOFF;
103  goto retry;
104  }
105  if (!DELETED(link))
106  {
107  if (cur_hashnr >= hashnr)
108  {
109  int r= 1;
110  if (cur_hashnr > hashnr ||
111  (r= my_strnncoll(cs, (uchar*) cur_key, cur_keylen, (uchar*) key,
112  keylen)) >= 0)
113  return !r;
114  }
115  cursor->prev= &(cursor->curr->link);
116  _lf_pin(pins, 2, cursor->curr);
117  }
118  else
119  {
120  /*
121  we found a deleted node - be nice, help the other thread
122  and remove this deleted node
123  */
124  if (my_atomic_casptr((void **)cursor->prev,
125  (void **)&cursor->curr, cursor->next))
126  _lf_alloc_free(pins, cursor->curr);
127  else
128  {
129  (void)LF_BACKOFF;
130  goto retry;
131  }
132  }
133  cursor->curr= cursor->next;
134  _lf_pin(pins, 1, cursor->curr);
135  }
136 }

138 /*
139  DESCRIPTION
140  insert a 'node' in the list that starts from 'head' in the correct
141  position (as found by lfind)
142 
143  RETURN
144  0 - inserted
145  not 0 - a pointer to a duplicate (not pinned and thus unusable)
146 
147  NOTE
148  it uses pins[0..2], on return all pins are removed.
149  if there're nodes with the same key value, a new node is added before them.
150 */
151 static LF_SLIST *linsert(LF_SLIST * volatile *head, CHARSET_INFO *cs,
152  LF_SLIST *node, LF_PINS *pins, uint flags)
153 {
154  CURSOR cursor;
155  int res;
156 
157  for (;;)
158  {
159  if (lfind(head, cs, node->hashnr, node->key, node->keylen,
160  &cursor, pins) &&
161  (flags & LF_HASH_UNIQUE))
162  {
163  res= 0; /* duplicate found */
164  break;
165  }
166  else
167  {
168  node->link= (intptr)cursor.curr;
169  DBUG_ASSERT(node->link != (intptr)node); /* no circular references */
170  DBUG_ASSERT(cursor.prev != &node->link); /* no circular references */
171  if (my_atomic_casptr((void **)cursor.prev, (void **)&cursor.curr, node))
172  {
173  res= 1; /* inserted ok */
174  break;
175  }
176  }
177  }
178  _lf_unpin(pins, 0);
179  _lf_unpin(pins, 1);
180  _lf_unpin(pins, 2);
181  /*
182  Note that cursor.curr is not pinned here and the pointer is unreliable,
183  the object may dissapear anytime. But if it points to a dummy node, the
184  pointer is safe, because dummy nodes are never freed - initialize_bucket()
185  uses this fact.
186  */
187  return res ? 0 : cursor.curr;
188 }

190 /*
191  DESCRIPTION
192  deletes a node as identified by hashnr/keey/keylen from the list
193  that starts from 'head'
194 
195  RETURN
196  0 - ok
197  1 - not found
198 
199  NOTE
200  it uses pins[0..2], on return all pins are removed.
201 */
202 static int ldelete(LF_SLIST * volatile *head, CHARSET_INFO *cs, uint32 hashnr,
203  const uchar *key, uint keylen, LF_PINS *pins)
204 {
205  CURSOR cursor;
206  int res;
207 
208  for (;;)
209  {
210  if (!lfind(head, cs, hashnr, key, keylen, &cursor, pins))
211  {
212  res= 1; /* not found */
213  break;
214  }
215  else
216  {
217  /* mark the node deleted */
218  if (my_atomic_casptr((void **)&(cursor.curr->link),
219  (void **)&cursor.next,
220  (void *)(((intptr)cursor.next) | 1)))
221  {
222  /* and remove it from the list */
223  if (my_atomic_casptr((void **)cursor.prev,
224  (void **)&cursor.curr, cursor.next))
225  _lf_alloc_free(pins, cursor.curr);
226  else
227  {
228  /*
229  somebody already "helped" us and removed the node ?
230  Let's check if we need to help that someone too!
231  (to ensure the number of "set DELETED flag" actions
232  is equal to the number of "remove from the list" actions)
233  */
234  lfind(head, cs, hashnr, key, keylen, &cursor, pins);
235  }
236  res= 0;
237  break;
238  }
239  }
240  }
241  _lf_unpin(pins, 0);
242  _lf_unpin(pins, 1);
243  _lf_unpin(pins, 2);
244  return res;
245 }

282 /*
283  Compute the hash key value from the raw key.
284 
285  @note, that the hash value is limited to 2^31, because we need one
286  bit to distinguish between normal and dummy nodes.
287 */
288 static inline uint calc_hash(LF_HASH *hash, const uchar *key, uint keylen)
289 {
290  ulong nr1= 1, nr2= 4;
291  hash->charset->coll->hash_sort(hash->charset, (uchar*) key, keylen,
292  &nr1, &nr2);
293  return nr1 & INT_MAX32;
294 }

300 /*
301  Initializes lf_hash, the arguments are compatible with hash_init
302 
303  @note element_size sets both the size of allocated memory block for
304  lf_alloc and a size of memcpy'ed block size in lf_hash_insert. Typically
305  they are the same, indeed. But LF_HASH::element_size can be decreased
306  after lf_hash_init, and then lf_alloc will allocate larger block that

315  CHARSET_INFO *charset)
316 {

352 /*
353  DESCRIPTION
354  inserts a new element to a hash. it will have a _copy_ of
355  data, not a pointer to it.
356 
357  RETURN
358  0 - inserted
359  1 - didn't (unique key conflict)
360  -1 - out of memory
361 
362  NOTE
363  see linsert() for pin usage notes
364 */
365 int lf_hash_insert(LF_HASH *hash, LF_PINS *pins, const void *data)
366 {
367  int csize, bucket, hashnr;
368  LF_SLIST *node, * volatile *el;
369 
370  lf_rwlock_by_pins(pins);
371  node= (LF_SLIST *)_lf_alloc_new(pins);
372  if (unlikely(!node))
373  return -1;
374  memcpy(node+1, data, hash->element_size);
375  node->key= hash_key(hash, (uchar *)(node+1), &node->keylen);
376  hashnr= calc_hash(hash, node->key, node->keylen);
377  bucket= hashnr % hash->size;
378  el= _lf_dynarray_lvalue(&hash->array, bucket);
379  if (unlikely(!el))
380  return -1;
381  if (*el == NULL && unlikely(initialize_bucket(hash, el, bucket, pins)))
382  return -1;
383  node->hashnr= my_reverse_bits(hashnr) | 1; /* normal node */
384  if (linsert(el, hash->charset, node, pins, hash->flags))
385  {
386  _lf_alloc_free(pins, node);
387  lf_rwunlock_by_pins(pins);
388  return 1;
389  }
390  csize= hash->size;
391  if ((my_atomic_add32(&hash->count, 1)+1.0) / csize > MAX_LOAD)
392  my_atomic_cas32(&hash->size, &csize, csize*2);
393  lf_rwunlock_by_pins(pins);
394  return 0;
395 }

397 /*
398  DESCRIPTION
399  deletes an element with the given key from the hash (if a hash is
400  not unique and there're many elements with this key - the "first"
401  matching element is deleted)
402  RETURN
403  0 - deleted
404  1 - didn't (not found)
405  -1 - out of memory
406  NOTE
407  see ldelete() for pin usage notes
408 */
409 int lf_hash_delete(LF_HASH *hash, LF_PINS *pins, const void *key, uint keylen)
410 {
411  LF_SLIST * volatile *el;
412  uint bucket, hashnr= calc_hash(hash, (uchar *)key, keylen);
413 
414  bucket= hashnr % hash->size;
415  lf_rwlock_by_pins(pins);
416  el= _lf_dynarray_lvalue(&hash->array, bucket);
417  if (unlikely(!el))
418  return -1;
419  /*
420  note that we still need to initialize_bucket here,
421  we cannot return "node not found", because an old bucket of that
422  node may've been split and the node was assigned to a new bucket
423  that was never accessed before and thus is not initialized.
424  */
425  if (*el == NULL && unlikely(initialize_bucket(hash, el, bucket, pins)))
426  return -1;
427  if (ldelete(el, hash->charset, my_reverse_bits(hashnr) | 1,
428  (uchar *)key, keylen, pins))
429  {
430  lf_rwunlock_by_pins(pins);
431  return 1;
432  }
433  my_atomic_add32(&hash->count, -1);
434  lf_rwunlock_by_pins(pins);
435  return 0;
436 }

473 static int initialize_bucket(LF_HASH *hash, LF_SLIST * volatile *node,
474  uint bucket, LF_PINS *pins)
475 {
476  uint parent= my_clear_highest_bit(bucket);
477  LF_SLIST *dummy= (LF_SLIST *)my_malloc(sizeof(LF_SLIST), MYF(MY_WME));
478  LF_SLIST **tmp= 0, *cur;
479  LF_SLIST * volatile *el= _lf_dynarray_lvalue(&hash->array, parent);
480  if (unlikely(!el || !dummy))
481  return -1;
482  if (*el == NULL && bucket &&
483  unlikely(initialize_bucket(hash, el, parent, pins)))
484  return -1;
485  dummy->hashnr= my_reverse_bits(bucket) | 0; /* dummy node */
486  dummy->key= dummy_key;
487  dummy->keylen= 0;
488  if ((cur= linsert(el, hash->charset, dummy, pins, LF_HASH_UNIQUE)))
489  {
490  my_free(dummy);
491  dummy= cur;
492  }
493  my_atomic_casptr((void **)node, (void **)&tmp, dummy);
494  /*
495  note that if the CAS above failed (after linsert() succeeded),
496  it would mean that some other thread has executed linsert() for
497  the same dummy node, its linsert() failed, it picked up our
498  dummy node (in "dummy= cur") and executed the same CAS as above.
499  Which means that even if CAS above failed we don't need to retry,
500  and we should not free(dummy) - there's no memory leak here
501  */
502  return 0;
503 }
