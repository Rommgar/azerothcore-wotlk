81  void UpdateAI(uint32 diff) override
82  {
83  if (!UpdateVictim())
84  {
85  return;
86  }
87 
88  // This should always process
89  if (massEruptionTimer <= diff)
90  {
91  Talk(EMOTE_MASS_ERRUPTION, me);
92  DoCastAOE(SPELL_ERUPTION_TRIGGER, true);
93  massEruptionTimer = 20000;
94  }
95  else
96  {
97  massEruptionTimer -= diff;
98  }
99 
100  events.Update(diff);
101 
102  if (me->HasUnitState(UNIT_STATE_CASTING))
103  {
104  return;
105  }
106 
107  while (uint32 const eventId = events.ExecuteEvent())
108  {
109  switch (eventId)
110  {
111  case EVENT_ANTIMAGIC_PULSE:
112  {
113  DoCastSelf(SPELL_ANTIMAGIC_PULSE);
114  events.RepeatEvent(20000);
115  break;
116  }
117  case EVENT_MAGMA_SHACKLES:
118  {
119  DoCastSelf(SPELL_MAGMA_SHACKLES);
120  events.RepeatEvent(15000);
121  break;
122  }
123  }
124 
125  if (me->HasUnitState(UNIT_STATE_CASTING))
126  {
127  return;
128  }
129  }
130 
131  DoMeleeAttackIfReady();
132  }
133 
134  private:
135  uint32 massEruptionTimer;
136  };
