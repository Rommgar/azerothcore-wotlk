133 /*
134  Request alarm after sec seconds.
135 
136  SYNOPSIS
137  thr_alarm()
138  alrm Pointer to alarm detection
139  alarm_data Structure to store in alarm queue
140 
141  NOTES
142  This function can't be called from the alarm-handling thread.
143 
144  RETURN VALUES
145  0 ok
146  1 If no more alarms are allowed (aborted by process)
147 
148  Stores in first argument a pointer to a non-zero int which is set to 0
149  when the alarm has been given
150 */

152 my_bool thr_alarm(thr_alarm_t *alrm, uint sec, ALARM *alarm_data)
153 {
154  time_t now;
155 #ifndef USE_ONE_SIGNAL_HAND
156  sigset_t old_mask;
157 #endif
158  my_bool reschedule;
159  struct st_my_thread_var *current_my_thread_var= my_thread_var;
160  DBUG_ENTER("thr_alarm");
161  DBUG_PRINT("enter",("thread: %s sec: %d",my_thread_name(),sec));
162 
163  now= my_time(0);
164 #ifndef USE_ONE_SIGNAL_HAND
165  pthread_sigmask(SIG_BLOCK,&full_signal_set,&old_mask);
166 #endif
167  mysql_mutex_lock(&LOCK_alarm); /* Lock from threads & alarms */
168  if (alarm_aborted > 0)
169  { /* No signal thread */
170  DBUG_PRINT("info", ("alarm aborted"));
171  *alrm= 0; /* No alarm */
172  mysql_mutex_unlock(&LOCK_alarm);
173 #ifndef USE_ONE_SIGNAL_HAND
174  pthread_sigmask(SIG_SETMASK,&old_mask,NULL);
175 #endif
176  DBUG_RETURN(1);
177  }
178  if (alarm_aborted < 0)
179  sec= 1; /* Abort mode */
180 
181  if (alarm_queue.elements >= max_used_alarms)
182  {
183  if (alarm_queue.elements == alarm_queue.max_elements)
184  {
185  DBUG_PRINT("info", ("alarm queue full"));
186  fprintf(stderr,"Warning: thr_alarm queue is full\n");
187  *alrm= 0; /* No alarm */
188  mysql_mutex_unlock(&LOCK_alarm);
189 #ifndef USE_ONE_SIGNAL_HAND
190  pthread_sigmask(SIG_SETMASK,&old_mask,NULL);
191 #endif
192  DBUG_RETURN(1);
193  }
194  max_used_alarms=alarm_queue.elements+1;
195  }
196  reschedule= (ulong) next_alarm_expire_time > (ulong) now + sec;
197  if (!alarm_data)
198  {
199  if (!(alarm_data=(ALARM*) my_malloc(sizeof(ALARM),MYF(MY_WME))))
200  {
201  DBUG_PRINT("info", ("failed my_malloc()"));
202  *alrm= 0; /* No alarm */
203  mysql_mutex_unlock(&LOCK_alarm);
204 #ifndef USE_ONE_SIGNAL_HAND
205  pthread_sigmask(SIG_SETMASK,&old_mask,NULL);
206 #endif
207  DBUG_RETURN(1);
208  }
209  alarm_data->malloced=1;
210  }
211  else
212  alarm_data->malloced=0;
213  alarm_data->expire_time=now+sec;
214  alarm_data->alarmed=0;
215  alarm_data->thread= current_my_thread_var->pthread_self;
216  alarm_data->thread_id= current_my_thread_var->id;
217  queue_insert(&alarm_queue,(uchar*) alarm_data);
218 
219  /* Reschedule alarm if the current one has more than sec left */
220  if (reschedule)
221  {
222  DBUG_PRINT("info", ("reschedule"));
223  if (pthread_equal(pthread_self(),alarm_thread))
224  {
225  alarm(sec); /* purecov: inspected */
226  next_alarm_expire_time= now + sec;
227  }
228  else
229  reschedule_alarms(); /* Reschedule alarms */
230  }
231  mysql_mutex_unlock(&LOCK_alarm);
232 #ifndef USE_ONE_SIGNAL_HAND
233  pthread_sigmask(SIG_SETMASK,&old_mask,NULL);
234 #endif
235  (*alrm)= &alarm_data->alarmed;
236  DBUG_RETURN(0);
237 }

245 {
246  ALARM *alarm_data;
247 #ifndef USE_ONE_SIGNAL_HAND
248  sigset_t old_mask;
249 #endif
250  uint i, found=0;
251  DBUG_ENTER("thr_end_alarm");
252 
253 #ifndef USE_ONE_SIGNAL_HAND
254  pthread_sigmask(SIG_BLOCK,&full_signal_set,&old_mask);
255 #endif
256  mysql_mutex_lock(&LOCK_alarm);
257 
258  alarm_data= (ALARM*) ((uchar*) *alarmed - offsetof(ALARM,alarmed));
259  for (i=0 ; i < alarm_queue.elements ; i++)
260  {
261  if ((ALARM*) queue_element(&alarm_queue,i) == alarm_data)
262  {

277  (long) *alarmed, alarm_queue.elements);

279  (long) *alarmed));

281  mysql_mutex_unlock(&LOCK_alarm);

296 sig_handler process_alarm(int sig __attribute__((unused)))
297 {
298  sigset_t old_mask;
299 /*
300  This must be first as we can't call DBUG inside an alarm for a normal thread
301 */
302 
303  if (thd_lib_detected == THD_LIB_LT &&
304  !pthread_equal(pthread_self(),alarm_thread))
305  {
306 #if defined(MAIN) && !defined(__bsdi__)
307  printf("thread_alarm in process_alarm\n"); fflush(stdout);
308 #endif
309 #ifdef SIGNAL_HANDLER_RESET_ON_DELIVERY
310  my_sigset(thr_client_alarm, process_alarm); /* int. thread system calls */
311 #endif
312  return;
313  }
314 
315  /*
316  We have to do do the handling of the alarm in a sub function,
317  because otherwise we would get problems with two threads calling
318  DBUG_... functions at the same time (as two threads may call
319  process_alarm() at the same time
320  */
321 
322 #ifndef USE_ALARM_THREAD
323  pthread_sigmask(SIG_SETMASK,&full_signal_set,&old_mask);
324  mysql_mutex_lock(&LOCK_alarm);
325 #endif
326  process_alarm_part2(sig);
327 #ifndef USE_ALARM_THREAD
328 #if defined(SIGNAL_HANDLER_RESET_ON_DELIVERY) && !defined(USE_ONE_SIGNAL_HAND)
329  my_sigset(THR_SERVER_ALARM,process_alarm);
330 #endif
331  mysql_mutex_unlock(&LOCK_alarm);
332  pthread_sigmask(SIG_SETMASK,&old_mask,NULL);
333 #endif

338 static sig_handler process_alarm_part2(int sig __attribute__((unused)))
339 {
340  ALARM *alarm_data;
341  DBUG_ENTER("process_alarm");
342  DBUG_PRINT("info",("sig: %d active alarms: %d",sig,alarm_queue.elements));
343 
344 #if defined(MAIN) && !defined(__bsdi__)
345  printf("process_alarm\n"); fflush(stdout);
346 #endif
347  if (alarm_queue.elements)
348  {
349  if (alarm_aborted)
350  {
351  uint i;
352  for (i=0 ; i < alarm_queue.elements ;)
353  {
354  alarm_data=(ALARM*) queue_element(&alarm_queue,i);
355  alarm_data->alarmed=1; /* Info to thread */
356  if (pthread_equal(alarm_data->thread,alarm_thread) ||
357  pthread_kill(alarm_data->thread, thr_client_alarm))
358  {
359 #ifdef MAIN
360  printf("Warning: pthread_kill couldn't find thread!!!\n");
361 #endif
362  queue_remove(&alarm_queue,i); /* No thread. Remove alarm */
363  }
364  else
365  i++; /* Signal next thread */
366  }
367 #ifndef USE_ALARM_THREAD
368  if (alarm_queue.elements)
369  alarm(1); /* Signal soon again */
370 #endif
371  }
372  else
373  {
374  ulong now=(ulong) my_time(0);
375  ulong next=now+10-(now%10);
376  while ((alarm_data=(ALARM*) queue_top(&alarm_queue))->expire_time <= now)
377  {
378  alarm_data->alarmed=1; /* Info to thread */
379  DBUG_PRINT("info",("sending signal to waiting thread"));
380  if (pthread_equal(alarm_data->thread,alarm_thread) ||
381  pthread_kill(alarm_data->thread, thr_client_alarm))
382  {
383 #ifdef MAIN
384  printf("Warning: pthread_kill couldn't find thread!!!\n");
385 #endif
386  queue_remove(&alarm_queue,0); /* No thread. Remove alarm */
387  if (!alarm_queue.elements)
388  break;
389  }
390  else
391  {
392  alarm_data->expire_time=next;
393  queue_replaced(&alarm_queue);
394  }
395  }
396 #ifndef USE_ALARM_THREAD
397  if (alarm_queue.elements)
398  {
399 #ifdef __bsdi__
400  alarm(0); /* Remove old alarm */
401 #endif
402  alarm((uint) (alarm_data->expire_time-now));
403  next_alarm_expire_time= alarm_data->expire_time;
404  }
405 #endif
406  }
407  }
408  else
409  {
410  /*
411  Ensure that next time we call thr_alarm(), we will schedule a new alarm
412  */
413  next_alarm_expire_time= ~(time_t) 0;
414  }
415  DBUG_VOID_RETURN;
416 }

419 /*
420  Schedule all alarms now and optionally free all structures
421 
422  SYNPOSIS
423  end_thr_alarm()
424  free_structures Set to 1 if we should free memory used for
425  the alarm queue.
426  When we call this we should KNOW that there
427  is no active alarms
428  IMPLEMENTATION
429  Set alarm_abort to -1 which will change the behavior of alarms as follows:
430  - All old alarms will be rescheduled at once
431  - All new alarms will be rescheduled to one second
432 */
433 
434 void end_thr_alarm(my_bool free_structures)
435 {
436  DBUG_ENTER("end_thr_alarm");
437  if (alarm_aborted != 1) /* If memory not freed */
438  {
439  mysql_mutex_lock(&LOCK_alarm);
440  DBUG_PRINT("info",("Resheduling %d waiting alarms",alarm_queue.elements));
441  alarm_aborted= -1; /* mark aborted */
442  if (alarm_queue.elements || (alarm_thread_running && free_structures))
443  {
444  if (pthread_equal(pthread_self(),alarm_thread))
445  alarm(1); /* Shut down everything soon */
446  else
447  reschedule_alarms();
448  }
449  if (free_structures)
450  {
451  struct timespec abstime;
452 
453  DBUG_ASSERT(!alarm_queue.elements);
454 
455  /* Wait until alarm thread dies */
456  set_timespec(abstime, 10); /* Wait up to 10 seconds */
457  while (alarm_thread_running)
458  {
459  int error= mysql_cond_timedwait(&COND_alarm, &LOCK_alarm, &abstime);
460  if (error == ETIME || error == ETIMEDOUT)
461  break; /* Don't wait forever */
462  }
463  delete_queue(&alarm_queue);
464  alarm_aborted= 1;
465  mysql_mutex_unlock(&LOCK_alarm);
466  if (!alarm_thread_running) /* Safety */
467  {
468  mysql_mutex_destroy(&LOCK_alarm);
469  mysql_cond_destroy(&COND_alarm);
470  }
471  }
472  else
473  mysql_mutex_unlock(&LOCK_alarm);
474  }
475  DBUG_VOID_RETURN;
476 }

542 /* set up a alarm thread with uses 'sleep' to sleep between alarms */
543 
544 #ifdef USE_ALARM_THREAD
545 static void *alarm_handler(void *arg __attribute__((unused)))
546 {
547  int error;
548  struct timespec abstime;
549 #ifdef MAIN
550  puts("Starting alarm thread");
551 #endif
552  my_thread_init();
553  alarm_thread_running= 1;
554  mysql_mutex_lock(&LOCK_alarm);
555  for (;;)
556  {
557  if (alarm_queue.elements)
558  {
559  ulong sleep_time,now= my_time(0);
560  if (alarm_aborted)
561  sleep_time=now+1;
562  else
563  sleep_time= ((ALARM*) queue_top(&alarm_queue))->expire_time;
564  if (sleep_time > now)
565  {
566  abstime.tv_sec=sleep_time;
567  abstime.tv_nsec=0;
568  next_alarm_expire_time= sleep_time;
569  if ((error= mysql_cond_timedwait(&COND_alarm, &LOCK_alarm, &abstime)) &&
570  error != ETIME && error != ETIMEDOUT)
571  {
572 #ifdef MAIN
573  printf("Got error: %d from ptread_cond_timedwait (errno: %d)\n",
574  error,errno);
575 #endif
576  }
577  }
578  }
579  else if (alarm_aborted == -1)
580  break;
581  else
582  {
583  next_alarm_expire_time= ~ (time_t) 0;
584  if ((error= mysql_cond_wait(&COND_alarm, &LOCK_alarm)))
585  {
586 #ifdef MAIN
587  printf("Got error: %d from ptread_cond_wait (errno: %d)\n",
588  error,errno);
589 #endif
590  }
591  }
592  process_alarm(0);
593  }
594  bzero((char*) &alarm_thread,sizeof(alarm_thread)); /* For easy debugging */
595  alarm_thread_running= 0;
596  mysql_cond_signal(&COND_alarm);
597  mysql_mutex_unlock(&LOCK_alarm);
598  pthread_exit(0);
599  return 0; /* Impossible */
600 }
601 #endif /* USE_ALARM_THREAD */
