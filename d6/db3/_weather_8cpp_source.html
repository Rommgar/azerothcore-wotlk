66 bool Weather::ReGenerate()
67 {
68  if (!m_weatherChances)
69  {
70  m_type = WEATHER_TYPE_FINE;
71  m_grade = 0.0f;
72  return false;
73  }
74 
80  uint32 u = urand(0, 99);
81 
82  if (u < 30)
83  return false;
84 
85  // remember old values
86  WeatherType old_type = m_type;
87  float old_grade = m_grade;
88 
89  //78 days between January 1st and March 20nd; 365/4=91 days by season
90  // season source http://aa.usno.navy.mil/data/docs/EarthSeasons.html
91  time_t gtime = sWorld->GetGameTime();
92  struct tm ltime;
93  localtime_r(&gtime, &ltime);
94  uint32 season = ((ltime.tm_yday - 78 + 365) / 91) % 4;
95 
96  static char const* seasonName[WEATHER_SEASONS] = { "spring", "summer", "fall", "winter" };
97  LOG_DEBUG("weather", "Generating a change in %s weather for zone %u.", seasonName[season], m_zone);
98 
99  if ((u < 60) && (m_grade < 0.33333334f)) // Get fair
100  {
101  m_type = WEATHER_TYPE_FINE;
102  m_grade = 0.0f;
103  }
104 
105  if ((u < 60) && (m_type != WEATHER_TYPE_FINE)) // Get better
106  {
107  m_grade -= 0.33333334f;
108  return true;
109  }
110 
111  if ((u < 90) && (m_type != WEATHER_TYPE_FINE)) // Get worse
112  {
113  m_grade += 0.33333334f;
114  return true;
115  }
116 
117  if (m_type != WEATHER_TYPE_FINE)
118  {
123 
124  if (m_grade < 0.33333334f)
125  {
126  m_grade = 0.9999f; // go nuts
127  return true;
128  }
129  else
130  {
131  if (m_grade > 0.6666667f)
132  {
133  // Severe change, but how severe?
134  uint32 rnd = urand(0, 99);
135  if (rnd < 50)
136  {
137  m_grade -= 0.6666667f;
138  return true;
139  }
140  }
141  m_type = WEATHER_TYPE_FINE; // clear up
142  m_grade = 0;
143  }
144  }
145 
146  // At this point, only weather that isn't doing anything remains but that have weather data
147  uint32 chance1 = m_weatherChances->data[season].rainChance;
148  uint32 chance2 = chance1 + m_weatherChances->data[season].snowChance;
149  uint32 chance3 = chance2 + m_weatherChances->data[season].stormChance;
150 
151  uint32 rnd = urand(0, 99);
152  if (rnd <= chance1)
153  m_type = WEATHER_TYPE_RAIN;
154  else if (rnd <= chance2)
155  m_type = WEATHER_TYPE_SNOW;
156  else if (rnd <= chance3)
157  m_type = WEATHER_TYPE_STORM;
158  else
159  m_type = WEATHER_TYPE_FINE;
160 
166 
167  if (m_type == WEATHER_TYPE_FINE)
168  {
169  m_grade = 0.0f;
170  }
171  else if (u < 90)
172  {
173  m_grade = (float)rand_norm() * 0.3333f;
174  }
175  else
176  {
177  // Severe change, but how severe?
178  rnd = urand(0, 99);
179  if (rnd < 50)
180  m_grade = (float)rand_norm() * 0.3333f + 0.3334f;
181  else
182  m_grade = (float)rand_norm() * 0.3333f + 0.6667f;
183  }
184 
185  // return true only in case weather changes
186  return m_type != old_type || m_grade != old_grade;
187 }
