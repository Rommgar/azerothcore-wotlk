20 /*
21  Sync data in file to disk
22 
23  SYNOPSIS
24  my_sync()
25  fd File descritor to sync
26  my_flags Flags (now only MY_WME is supported)
27 
28  NOTE
29  If file system supports its, only file data is synced, not inode data.
30 
31  MY_IGNORE_BADFD is useful when fd is "volatile" - not protected by a
32  mutex. In this case by the time of fsync(), fd may be already closed by
33  another thread, or even reassigned to a different file. With this flag -
34  MY_IGNORE_BADFD - such a situation will not be considered an error.
35  (which is correct behaviour, if we know that the other thread synced the
36  file before closing)
37 
38  RETURN
39  0 ok
40  -1 error
41 */
42 
43 int my_sync(File fd, myf my_flags)
44 {
45  int res;
46  DBUG_ENTER("my_sync");
47  DBUG_PRINT("my",("Fd: %d my_flags: %d", fd, my_flags));
48 
49  do
50  {
51 #if defined(F_FULLFSYNC)
52  /*
53  In Mac OS X >= 10.3 this call is safer than fsync() (it forces the
54  disk's cache and guarantees ordered writes).
55  */
56  if (!(res= fcntl(fd, F_FULLFSYNC, 0)))
57  break; /* ok */
58  /* Some file systems don't support F_FULLFSYNC and fail above: */
59  DBUG_PRINT("info",("fcntl(F_FULLFSYNC) failed, falling back"));
60 #endif
61 #if defined(HAVE_FDATASYNC) && HAVE_DECL_FDATASYNC
62  res= fdatasync(fd);
63 #elif defined(HAVE_FSYNC)
64  res= fsync(fd);
65 #elif defined(_WIN32)
66  res= my_win_fsync(fd);
67 #else
68 #error Cannot find a way to sync a file, durability in danger
69  res= 0; /* No sync (strange OS) */
70 #endif
71  } while (res == -1 && errno == EINTR);
72 
73  if (res)
74  {
75  int er= errno;
76  if (!(my_errno= er))
77  my_errno= -1; /* Unknown error */
78  if ((my_flags & MY_IGNORE_BADFD) &&
79  (er == EBADF || er == EINVAL || er == EROFS))
80  {
81  DBUG_PRINT("info", ("ignoring errno %d", er));
82  res= 0;
83  }
84  else if (my_flags & MY_WME)
85  my_error(EE_SYNC, MYF(ME_BELL+ME_WAITTANG), my_filename(fd), my_errno);
86  }
87  DBUG_RETURN(res);
88 } /* my_sync */

108 int my_sync_dir(const char *dir_name, myf my_flags)
109 {
110  File dir_fd;
111  int res= 0;
112  const char *correct_dir_name;
113  DBUG_ENTER("my_sync_dir");
114  DBUG_PRINT("my",("Dir: '%s' my_flags: %d", dir_name, my_flags));
115  /* Sometimes the path does not contain an explicit directory */
116  correct_dir_name= (dir_name[0] == 0) ? cur_dir_name : dir_name;
117  /*
118  Syncing a dir may give EINVAL on tmpfs on Linux, which is ok.
119  EIO on the other hand is very important. Hence MY_IGNORE_BADFD.
120  */
121  if ((dir_fd= my_open(correct_dir_name, O_RDONLY, MYF(my_flags))) >= 0)
122  {
123  if (my_sync(dir_fd, MYF(my_flags | MY_IGNORE_BADFD)))
124  res= 2;
125  if (my_close(dir_fd, MYF(my_flags)))
126  res= 3;
127  }
128  else
129  res= 1;
130  DBUG_RETURN(res);
131 }
