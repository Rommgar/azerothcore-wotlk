62 size_t vio_read_buff(Vio *vio, uchar* buf, size_t size)
63 {
64  size_t rc;
65 #define VIO_UNBUFFERED_READ_MIN_SIZE 2048
66  DBUG_ENTER("vio_read_buff");
67  DBUG_PRINT("enter", ("sd: %d buf: 0x%lx size: %u", vio->sd, (long) buf,
68  (uint) size));
69 
70  if (vio->read_pos < vio->read_end)
71  {
72  rc= min((size_t) (vio->read_end - vio->read_pos), size);
73  memcpy(buf, vio->read_pos, rc);
74  vio->read_pos+= rc;
75  /*
76  Do not try to read from the socket now even if rc < size:
77  vio_read can return -1 due to an error or non-blocking mode, and
78  the safest way to handle it is to move to a separate branch.
79  */
80  }
81  else if (size < VIO_UNBUFFERED_READ_MIN_SIZE)
82  {
83  rc= vio_read(vio, (uchar*) vio->read_buffer, VIO_READ_BUFFER_SIZE);
84  if (rc != 0 && rc != (size_t) -1)
85  {
86  if (rc > size)
87  {
88  vio->read_pos= vio->read_buffer + size;
89  vio->read_end= vio->read_buffer + rc;
90  rc= size;
91  }
92  memcpy(buf, vio->read_buffer, rc);
93  }
94  }
95  else
96  rc= vio_read(vio, buf, size);
97  DBUG_RETURN(rc);
98 #undef VIO_UNBUFFERED_READ_MIN_SIZE
99 }

127 int vio_blocking(Vio * vio __attribute__((unused)), my_bool set_blocking_mode,
128  my_bool *old_mode)
129 {
130  int r=0;
131  DBUG_ENTER("vio_blocking");
132 
133  *old_mode= test(!(vio->fcntl_mode & O_NONBLOCK));
134  DBUG_PRINT("enter", ("set_blocking_mode: %d old_mode: %d",
135  (int) set_blocking_mode, (int) *old_mode));
136 
137 #if !defined(__WIN__)
138 #if !defined(NO_FCNTL_NONBLOCK)
139  if (vio->sd >= 0)
140  {
141  int old_fcntl=vio->fcntl_mode;
142  if (set_blocking_mode)
143  vio->fcntl_mode &= ~O_NONBLOCK; /* clear bit */
144  else
145  vio->fcntl_mode |= O_NONBLOCK; /* set bit */
146  if (old_fcntl != vio->fcntl_mode)
147  {
148  r= fcntl(vio->sd, F_SETFL, vio->fcntl_mode);
149  if (r == -1)
150  {
151  DBUG_PRINT("info", ("fcntl failed, errno %d", errno));
152  vio->fcntl_mode= old_fcntl;
153  }
154  }
155  }
156 #else
157  r= set_blocking_mode ? 0 : 1;
158 #endif /* !defined(NO_FCNTL_NONBLOCK) */
159 #else /* !defined(__WIN__) */
160  if (vio->type != VIO_TYPE_NAMEDPIPE && vio->type != VIO_TYPE_SHARED_MEMORY)
161  {
162  ulong arg;
163  int old_fcntl=vio->fcntl_mode;
164  if (set_blocking_mode)
165  {
166  arg = 0;
167  vio->fcntl_mode &= ~O_NONBLOCK; /* clear bit */
168  }
169  else
170  {
171  arg = 1;
172  vio->fcntl_mode |= O_NONBLOCK; /* set bit */
173  }
174  if (old_fcntl != vio->fcntl_mode)
175  r = ioctlsocket(vio->sd,FIONBIO,(void*) &arg);
176  }
177  else
178  r= test(!(vio->fcntl_mode & O_NONBLOCK)) != set_blocking_mode;
179 #endif /* !defined(__WIN__) */
180  DBUG_PRINT("exit", ("%d", r));
181  DBUG_RETURN(r);
182 }

339 static void vio_get_normalized_ip(const struct sockaddr *src,
340  int src_length,
341  struct sockaddr *dst,
342  int *dst_length)
343 {
344  switch (src->sa_family) {
345  case AF_INET:
346  memcpy(dst, src, src_length);
347  *dst_length= src_length;
348  break;
349 
350 #ifdef HAVE_IPV6
351  case AF_INET6:
352  {
353  const struct sockaddr_in6 *src_addr6= (const struct sockaddr_in6 *) src;
354  const struct in6_addr *src_ip6= &(src_addr6->sin6_addr);
355  const uint32 *src_ip6_int32= (uint32 *) src_ip6->s6_addr;
356 
357  if (IN6_IS_ADDR_V4MAPPED(src_ip6) || IN6_IS_ADDR_V4COMPAT(src_ip6))
358  {
359  struct sockaddr_in *dst_ip4= (struct sockaddr_in *) dst;
360 
361  /*
362  This is an IPv4-mapped or IPv4-compatible IPv6 address. It should
363  be converted to the IPv4 form.
364  */
365 
366  *dst_length= sizeof (struct sockaddr_in);
367 
368  memset(dst_ip4, 0, *dst_length);
369  dst_ip4->sin_family= AF_INET;
370  dst_ip4->sin_port= src_addr6->sin6_port;
371 
372  /*
373  In an IPv4 mapped or compatible address, the last 32 bits represent
374  the IPv4 address. The byte orders for IPv6 and IPv4 addresses are
375  the same, so a simple copy is possible.
376  */
377  dst_ip4->sin_addr.s_addr= src_ip6_int32[3];
378  }
379  else
380  {
381  /* This is a "native" IPv6 address. */
382 
383  memcpy(dst, src, src_length);
384  *dst_length= src_length;
385  }
386 
387  break;
388  }
389 #endif /* HAVE_IPV6 */
390  }
391 }

450 my_bool vio_peer_addr(Vio *vio, char *ip_buffer, uint16 *port,
451  size_t ip_buffer_size)
452 {
453  DBUG_ENTER("vio_peer_addr");
454  DBUG_PRINT("enter", ("Client socked fd: %d", (int) vio->sd));
455 
456  if (vio->localhost)
457  {
458  /*
459  Initialize vio->remote and vio->addLen. Set vio->remote to IPv4 loopback
460  address.
461  */
462  struct in_addr *ip4= &((struct sockaddr_in *) &(vio->remote))->sin_addr;
463 
464  vio->remote.ss_family= AF_INET;
465  vio->addrLen= sizeof (struct sockaddr_in);
466 
467  ip4->s_addr= htonl(INADDR_LOOPBACK);
468 
469  /* Initialize ip_buffer and port. */
470 
471  strmov(ip_buffer, "127.0.0.1");
472  *port= 0;
473  }
474  else
475  {
476  int err_code;
477  char port_buffer[NI_MAXSERV];
478 
479  struct sockaddr_storage addr_storage;
480  struct sockaddr *addr= (struct sockaddr *) &addr_storage;
481  size_socket addr_length= sizeof (addr_storage);
482 
483  /* Get sockaddr by socked fd. */
484 
485  err_code= getpeername(vio->sd, addr, &addr_length);
486 
487  if (err_code)
488  {
489  DBUG_PRINT("exit", ("getpeername() gave error: %d", socket_errno));
490  DBUG_RETURN(TRUE);
491  }
492 
493  /* Normalize IP address. */
494 
495  vio_get_normalized_ip(addr, addr_length,
496  (struct sockaddr *) &vio->remote, &vio->addrLen);
497 
498  /* Get IP address & port number. */
499 
500  err_code= vio_getnameinfo((struct sockaddr *) &vio->remote,
501  ip_buffer, ip_buffer_size,
502  port_buffer, NI_MAXSERV,
503  NI_NUMERICHOST | NI_NUMERICSERV);
504 
505  if (err_code)
506  {
507  DBUG_PRINT("exit", ("getnameinfo() gave error: %s",
508  gai_strerror(err_code)));
509  DBUG_RETURN(TRUE);
510  }
511 
512  *port= (uint16) strtol(port_buffer, NULL, 10);
513  }
514 
515  DBUG_PRINT("exit", ("Client IP address: %s; port: %d",
516  (const char *) ip_buffer,
517  (int) *port));
518  DBUG_RETURN(FALSE);
519 }

535 static my_bool socket_poll_read(my_socket sd, uint timeout)
536 {
537 #ifdef __WIN__
538  int res;
539  my_socket fd= sd;
540  fd_set readfds, errorfds;
541  struct timeval tm;
542  DBUG_ENTER("socket_poll_read");
543  tm.tv_sec= timeout;
544  tm.tv_usec= 0;
545  FD_ZERO(&readfds);
546  FD_ZERO(&errorfds);
547  FD_SET(fd, &readfds);
548  FD_SET(fd, &errorfds);
549  /* The first argument is ignored on Windows, so a conversion to int is OK */
550  if ((res= select((int) fd, &readfds, NULL, &errorfds, &tm) <= 0))
551  {
552  DBUG_RETURN(res < 0 ? 0 : 1);
553  }
554  res= FD_ISSET(fd, &readfds) || FD_ISSET(fd, &errorfds);
555  DBUG_RETURN(!res);
556 #elif defined(HAVE_POLL)
557  struct pollfd fds;
558  int res;
559  DBUG_ENTER("socket_poll_read");
560  fds.fd=sd;
561  fds.events=POLLIN;
562  fds.revents=0;
563  if ((res=poll(&fds,1,(int) timeout*1000)) <= 0)
564  {
565  DBUG_RETURN(res < 0 ? 0 : 1); /* Don't return 1 on errors */
566  }
567  DBUG_RETURN(fds.revents & (POLLIN | POLLERR | POLLHUP) ? 0 : 1);
568 #else
569  return 0;
570 #endif
571 }

645 my_bool vio_is_connected(Vio *vio)
646 {
647  uint bytes= 0;
648  DBUG_ENTER("vio_is_connected");
649 
650  /* In the presence of errors the socket is assumed to be connected. */
651 
652  /*
653  The first step of detecting a EOF condition is veryfing
654  whether there is data to read. Data in this case would
655  be the EOF.
656  */
657  if (vio_poll_read(vio, 0))
658  DBUG_RETURN(TRUE);
659 
660  /*
661  The second step is read() or recv() from the socket returning
662  0 (EOF). Unfortunelly, it's not possible to call read directly
663  as we could inadvertently read meaningful connection data.
664  Simulate a read by retrieving the number of bytes available to
665  read -- 0 meaning EOF.
666  */
667  if (socket_peek_read(vio, &bytes))
668  DBUG_RETURN(TRUE);
669 
670 #ifdef HAVE_OPENSSL
671  /* There might be buffered data at the SSL layer. */
672  if (!bytes && vio->type == VIO_TYPE_SSL)
673  bytes= SSL_pending((SSL*) vio->ssl_arg);
674 #endif
675 
676  DBUG_RETURN(bytes ? TRUE : FALSE);
677 }

860 size_t vio_read_shared_memory(Vio * vio, uchar* buf, size_t size)
861 {
862  size_t length;
863  size_t remain_local;
864  char *current_postion;
865  HANDLE events[2];
866 
867  DBUG_ENTER("vio_read_shared_memory");
868  DBUG_PRINT("enter", ("sd: %d buf: 0x%lx size: %d", vio->sd, (long) buf,
869  size));
870 
871  remain_local = size;
872  current_postion=buf;
873 
874  events[0]= vio->event_server_wrote;
875  events[1]= vio->event_conn_closed;
876 
877  do
878  {
879  if (vio->shared_memory_remain == 0)
880  {
881  /*
882  WaitForMultipleObjects can return next values:
883  WAIT_OBJECT_0+0 - event from vio->event_server_wrote
884  WAIT_OBJECT_0+1 - event from vio->event_conn_closed. We can't read
885  anything
886  WAIT_ABANDONED_0 and WAIT_TIMEOUT - fail. We can't read anything
887  */
888  if (WaitForMultipleObjects(array_elements(events), events, FALSE,
889  vio->read_timeout_ms) != WAIT_OBJECT_0)
890  {
891  DBUG_RETURN(-1);
892  };
893 
894  vio->shared_memory_pos = vio->handle_map;
895  vio->shared_memory_remain = uint4korr((ulong*)vio->shared_memory_pos);
896  vio->shared_memory_pos+=4;
897  }
898 
899  length = size;
900 
901  if (vio->shared_memory_remain < length)
902  length = vio->shared_memory_remain;
903  if (length > remain_local)
904  length = remain_local;
905 
906  memcpy(current_postion,vio->shared_memory_pos,length);
907 
908  vio->shared_memory_remain-=length;
909  vio->shared_memory_pos+=length;
910  current_postion+=length;
911  remain_local-=length;
912 
913  if (!vio->shared_memory_remain)
914  {
915  if (!SetEvent(vio->event_client_read))
916  DBUG_RETURN(-1);
917  }
918  } while (remain_local);
919  length = size;
920 
921  DBUG_PRINT("exit", ("%lu", (ulong) length));
922  DBUG_RETURN(length);
923 }
