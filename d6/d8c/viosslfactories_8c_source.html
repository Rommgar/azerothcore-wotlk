94 static int
95 vio_set_cert_stuff(SSL_CTX *ctx, const char *cert_file, const char *key_file,
96  enum enum_ssl_init_error* error)
97 {
98  DBUG_ENTER("vio_set_cert_stuff");
99  DBUG_PRINT("enter", ("ctx: 0x%lx cert_file: %s key_file: %s",
100  (long) ctx, cert_file, key_file));
101  if (cert_file)
102  {
103  if (SSL_CTX_use_certificate_file(ctx, cert_file, SSL_FILETYPE_PEM) <= 0)
104  {
105  *error= SSL_INITERR_CERT;
106  DBUG_PRINT("error",("%s from file '%s'", sslGetErrString(*error), cert_file));
107  DBUG_EXECUTE("error", ERR_print_errors_fp(DBUG_FILE););
108  fprintf(stderr, "SSL error: %s from '%s'\n", sslGetErrString(*error),
109  cert_file);
110  fflush(stderr);
111  DBUG_RETURN(1);
112  }
113 
114  if (!key_file)
115  key_file= cert_file;
116 
117  if (SSL_CTX_use_PrivateKey_file(ctx, key_file, SSL_FILETYPE_PEM) <= 0)
118  {
119  *error= SSL_INITERR_KEY;
120  DBUG_PRINT("error", ("%s from file '%s'", sslGetErrString(*error), key_file));
121  DBUG_EXECUTE("error", ERR_print_errors_fp(DBUG_FILE););
122  fprintf(stderr, "SSL error: %s from '%s'\n", sslGetErrString(*error),
123  key_file);
124  fflush(stderr);
125  DBUG_RETURN(1);
126  }
127 
128  /*
129  If we are using DSA, we can copy the parameters from the private key
130  Now we know that a key and cert have been set against the SSL context
131  */
132  if (!SSL_CTX_check_private_key(ctx))
133  {
134  *error= SSL_INITERR_NOMATCH;
135  DBUG_PRINT("error", ("%s",sslGetErrString(*error)));
136  DBUG_EXECUTE("error", ERR_print_errors_fp(DBUG_FILE););
137  fprintf(stderr, "SSL error: %s\n", sslGetErrString(*error));
138  fflush(stderr);
139  DBUG_RETURN(1);
140  }
141  }
142  DBUG_RETURN(0);
143 }

163 /************************ VioSSLFd **********************************/
164 static struct st_VioSSLFd *
165 new_VioSSLFd(const char *key_file, const char *cert_file,
166  const char *ca_file, const char *ca_path,
167  const char *cipher, SSL_METHOD *method,
168  enum enum_ssl_init_error* error)
169 {
170  DH *dh;
171  struct st_VioSSLFd *ssl_fd;
172  DBUG_ENTER("new_VioSSLFd");
173  DBUG_PRINT("enter",
174  ("key_file: '%s' cert_file: '%s' ca_file: '%s' ca_path: '%s' "
175  "cipher: '%s'",
176  key_file ? key_file : "NULL",
177  cert_file ? cert_file : "NULL",
178  ca_file ? ca_file : "NULL",
179  ca_path ? ca_path : "NULL",
180  cipher ? cipher : "NULL"));
181 
182  check_ssl_init();
183 
184  if (!(ssl_fd= ((struct st_VioSSLFd*)
185  my_malloc(sizeof(struct st_VioSSLFd),MYF(0)))))
186  DBUG_RETURN(0);
187 
188  if (!(ssl_fd->ssl_context= SSL_CTX_new(method)))
189  {
190  *error= SSL_INITERR_MEMFAIL;
191  DBUG_PRINT("error", ("%s", sslGetErrString(*error)));
192  report_errors();
193  my_free(ssl_fd);
194  DBUG_RETURN(0);
195  }
196 
197  /*
198  Set the ciphers that can be used
199  NOTE: SSL_CTX_set_cipher_list will return 0 if
200  none of the provided ciphers could be selected
201  */
202  if (cipher &&
203  SSL_CTX_set_cipher_list(ssl_fd->ssl_context, cipher) == 0)
204  {
205  *error= SSL_INITERR_CIPHERS;
206  DBUG_PRINT("error", ("%s", sslGetErrString(*error)));
207  report_errors();
208  SSL_CTX_free(ssl_fd->ssl_context);
209  my_free(ssl_fd);
210  DBUG_RETURN(0);
211  }
212 
213  /* Load certs from the trusted ca */
214  if (SSL_CTX_load_verify_locations(ssl_fd->ssl_context, ca_file, ca_path) == 0)
215  {
216  DBUG_PRINT("warning", ("SSL_CTX_load_verify_locations failed"));
217  if (SSL_CTX_set_default_verify_paths(ssl_fd->ssl_context) == 0)
218  {
219  *error= SSL_INITERR_BAD_PATHS;
220  DBUG_PRINT("error", ("%s", sslGetErrString(*error)));
221  report_errors();
222  SSL_CTX_free(ssl_fd->ssl_context);
223  my_free(ssl_fd);
224  DBUG_RETURN(0);
225  }
226  }
227 
228  if (vio_set_cert_stuff(ssl_fd->ssl_context, cert_file, key_file, error))
229  {
230  DBUG_PRINT("error", ("vio_set_cert_stuff failed"));
231  report_errors();
232  SSL_CTX_free(ssl_fd->ssl_context);
233  my_free(ssl_fd);
234  DBUG_RETURN(0);
235  }
236 
237  /* DH stuff */
238  dh=get_dh512();
239  SSL_CTX_set_tmp_dh(ssl_fd->ssl_context, dh);
240  DH_free(dh);
241 
242  DBUG_PRINT("exit", ("OK 1"));
243 
244  DBUG_RETURN(ssl_fd);
245 }

250 new_VioSSLConnectorFd(const char *key_file, const char *cert_file,
251  const char *ca_file, const char *ca_path,

258  /*

267  {
268  return 0;

272 
273  SSL_CTX_set_verify(ssl_fd->ssl_context, verify, NULL);
