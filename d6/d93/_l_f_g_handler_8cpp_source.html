58  recvData.rfinish();
59  return;
60  }
61 
62  uint8 numDungeons;

64 
65  recvData >> roles;

67  recvData >> numDungeons;
68  if (!numDungeons)

209  data << uint32(quest->XPValue(GetPlayer()));

311 void WorldSession::SendLfgUpdatePlayer(lfg::LfgUpdateData const& updateData)
312 {
313  bool queued = false;
314  uint8 size = uint8(updateData.dungeons.size());
315 
316  switch (updateData.updateType)
317  {
318  case lfg::LFG_UPDATETYPE_JOIN_QUEUE:
319  case lfg::LFG_UPDATETYPE_ADDED_TO_QUEUE:
320  queued = true;
321  break;
322  case lfg::LFG_UPDATETYPE_UPDATE_STATUS:
323  queued = updateData.state == lfg::LFG_STATE_QUEUED;
324  break;
325  default:
326  break;
327  }
328 
329  LOG_DEBUG("lfg", "SMSG_LFG_UPDATE_PLAYER %s updatetype: %u",
330  GetPlayerInfo().c_str(), updateData.updateType);
331  WorldPacket data(SMSG_LFG_UPDATE_PLAYER, 1 + 1 + (size > 0 ? 1 : 0) * (1 + 1 + 1 + 1 + size * 4 + updateData.comment.length()));
332  data << uint8(updateData.updateType); // Lfg Update type
333  data << uint8(size > 0); // Extra info
334  if (size)
335  {
336  data << uint8(queued); // Join the queue
337  data << uint8(0); // unk - Always 0
338  data << uint8(0); // unk - Always 0
339 
340  data << uint8(size);
341  for (lfg::LfgDungeonSet::const_iterator it = updateData.dungeons.begin(); it != updateData.dungeons.end(); ++it)
342  data << uint32(*it);
343  data << updateData.comment;
344  }
345  SendPacket(&data);

348 void WorldSession::SendLfgUpdateParty(lfg::LfgUpdateData const& updateData)
349 {
350  bool join = false;
351  bool queued = false;
352  uint8 size = uint8(updateData.dungeons.size());
353 
354  switch (updateData.updateType)
355  {
356  case lfg::LFG_UPDATETYPE_ADDED_TO_QUEUE: // Rolecheck Success
357  queued = true;
358  [[fallthrough]];
359  case lfg::LFG_UPDATETYPE_PROPOSAL_BEGIN:
360  join = true;
361  break;
362  case lfg::LFG_UPDATETYPE_UPDATE_STATUS:
363  join = updateData.state != lfg::LFG_STATE_ROLECHECK && updateData.state != lfg::LFG_STATE_NONE;
364  queued = updateData.state == lfg::LFG_STATE_QUEUED;
365  break;
366  default:
367  break;
368  }
369 
370  LOG_DEBUG("lfg", "SMSG_LFG_UPDATE_PARTY %s updatetype: %u",
371  GetPlayerInfo().c_str(), updateData.updateType);
372  WorldPacket data(SMSG_LFG_UPDATE_PARTY, 1 + 1 + (size > 0 ? 1 : 0) * (1 + 1 + 1 + 1 + 1 + size * 4 + updateData.comment.length()));
373  data << uint8(updateData.updateType); // Lfg Update type
374  data << uint8(size > 0); // Extra info
375  if (size)
376  {
377  data << uint8(join); // LFG Join
378  data << uint8(queued); // Join the queue
379  data << uint8(0); // unk - Always 0
380  data << uint8(0); // unk - Always 0

550 void WorldSession::SendLfgUpdateProposal(lfg::LfgProposal const& proposal)
551 {
552  ObjectGuid guid = GetPlayer()->GetGUID();
553  ObjectGuid gguid = proposal.players.find(guid)->second.group;
554  bool silent = !proposal.isNew && gguid == proposal.group;
555  uint32 dungeonEntry = proposal.dungeonId;
556 
557  LOG_DEBUG("network", "SMSG_LFG_PROPOSAL_UPDATE [%s state: %u", guid.ToString().c_str(), proposal.state);
558 
559  // show random dungeon if player selected random dungeon and it's not lfg group
560  if (!silent)
561  {
562  lfg::LfgDungeonSet const& playerDungeons = sLFGMgr->GetSelectedDungeons(guid);
563  if (playerDungeons.find(proposal.dungeonId) == playerDungeons.end())
564  dungeonEntry = (*playerDungeons.begin());
565  }
566 
567  dungeonEntry = sLFGMgr->GetLFGDungeonEntry(dungeonEntry);
568 
569  WorldPacket data(SMSG_LFG_PROPOSAL_UPDATE, 4 + 1 + 4 + 4 + 1 + 1 + proposal.players.size() * (4 + 1 + 1 + 1 + 1 + 1));
570  data << uint32(dungeonEntry); // Dungeon
571  data << uint8(proposal.state); // Proposal state
572  data << uint32(proposal.id); // Proposal ID
573  data << uint32(proposal.encounters); // encounters done
574  data << uint8(silent); // Show proposal window
575  data << uint8(proposal.players.size()); // Group size
576 
577  for (lfg::LfgProposalPlayerContainer::const_iterator it = proposal.players.begin(); it != proposal.players.end(); ++it)
578  {
579  lfg::LfgProposalPlayer const& player = it->second;
580  data << uint32(player.role); // Role
581  data << uint8(it->first == guid); // Self player
582  if (!player.group) // Player not it a group
583  {
584  data << uint8(0); // Not in dungeon
585  data << uint8(0); // Not same group
586  }
587  else
588  {
589  data << uint8(player.group == proposal.group); // In dungeon (silent)
590  data << uint8(player.group == gguid); // Same Group than player
591  }
592  data << uint8(player.accept != lfg::LFG_ANSWER_PENDING);// Answered
593  data << uint8(player.accept == lfg::LFG_ANSWER_AGREE); // Accepted
594  }
595  SendPacket(&data);
596 }
