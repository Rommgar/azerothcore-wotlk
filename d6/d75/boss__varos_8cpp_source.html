158  void UpdateAI(uint32 diff) override
159  {
160  if( !UpdateVictim() )
161  return;
162 
163  events.Update(diff);
164 
165  if( me->HasUnitState(UNIT_STATE_CASTING) )
166  return;
167 
168  DoMeleeAttackIfReady();
169 
170  switch( events.ExecuteEvent() )
171  {
172  case 0:
173  break;
174  case EVENT_AMPLIFY_MAGIC:
175  {
176  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true) )
177  me->CastSpell(target, SPELL_AMPLIFY_MAGIC, false);
178  events.RepeatEvent(urand(17500, 22500));
179  }
180  break;
181  case EVENT_CALL_AZURE_RING_CAPTAIN_1:
182  case EVENT_CALL_AZURE_RING_CAPTAIN_2:
183  case EVENT_CALL_AZURE_RING_CAPTAIN_3:
184  case EVENT_CALL_AZURE_RING_CAPTAIN_4:
185  {
186  Talk(SAY_AZURE);
187  Talk(SAY_AZURE_EMOTE);
188  switch( events.ExecuteEvent() )
189  {
190  case EVENT_CALL_AZURE_RING_CAPTAIN_1:
191  me->CastSpell(me, SPELL_CALL_AZURE_RING_CAPTAIN_1, true);
192  events.ScheduleEvent(EVENT_CALL_AZURE_RING_CAPTAIN_2, 16000);
193  break;
194  case EVENT_CALL_AZURE_RING_CAPTAIN_2:
195  me->CastSpell(me, SPELL_CALL_AZURE_RING_CAPTAIN_2, true);
196  events.ScheduleEvent(EVENT_CALL_AZURE_RING_CAPTAIN_3, 16000);
197  break;
198  case EVENT_CALL_AZURE_RING_CAPTAIN_3:
199  me->CastSpell(me, SPELL_CALL_AZURE_RING_CAPTAIN_3, true);
200  events.ScheduleEvent(EVENT_CALL_AZURE_RING_CAPTAIN_4, 16000);
201  break;
202  case EVENT_CALL_AZURE_RING_CAPTAIN_4:
203  me->CastSpell(me, SPELL_CALL_AZURE_RING_CAPTAIN_4, true);
204  events.ScheduleEvent(EVENT_CALL_AZURE_RING_CAPTAIN_1, 16000);
205  break;
206  }
207  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true) )
208  {
209  if( Creature* trigger = me->SummonCreature(NPC_ARCANE_BEAM, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), 0.0f, TEMPSUMMON_TIMED_DESPAWN, 13000) )
210  {
211  if( Creature* c = me->FindNearestCreature(NPC_AZURE_RING_CAPTAIN, 500.0f, true) )
212  c->CastSpell(trigger, SPELL_ARCANE_BEAM_VISUAL, true);
213  trigger->GetMotionMaster()->MoveChase(target, 0.1f);
214  trigger->CastSpell(me, SPELL_ARCANE_BEAM_PERIODIC_DAMAGE, true);
215  }
216  }
217  }
218  break;
219  case EVENT_ENERGIZE_CORES_THIN:
220  {
221  me->SetControlled(false, UNIT_STATE_ROOT);
222  me->DisableRotate(false);
223  me->SetOrientation(ZapAngle);
224  me->CastSpell(me, SPELL_ENERGIZE_CORES_THIN, true);
225  events.ScheduleEvent(EVENT_ENERGIZE_CORES_DAMAGE, 4500);
226  }
227  break;

229  {
230  me->SetOrientation(ZapAngle);

232  me->DisableSpline();

234  me->SetControlled(true, UNIT_STATE_ROOT);
235  me->CastSpell((Unit*)nullptr, SPELL_ENERGIZE_CORES, false);

239  events.ScheduleEvent(EVENT_ENERGIZE_CORES_THIN, 2000);
240  }
241  break;
242  }
243  }
