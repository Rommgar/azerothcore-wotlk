24 // KillRewarder incapsulates logic of rewarding player upon kill with:
25 // * XP;
26 // * honor;
27 // * reputation;
28 // * kill credit (for quest objectives).
29 // Rewarding is initiated in two cases: when player kills unit in Unit::Kill()
30 // and on battlegrounds in Battleground::RewardXPAtKill().
31 //
32 // Rewarding algorithm is:
33 // 1. Initialize internal variables to default values.
34 // 2. In case when player is in group, initialize variables necessary for group calculations:
35 // 2.1. _count - number of alive group members within reward distance;
36 // 2.2. _aliveSumLevel - sum of levels of alive group members within reward distance;
37 // 2.3. _sumLevel - sum of levels of group members within reward distance;
38 // 2.4. _maxLevel - maximum level of alive group member within reward distance;
39 // 2.5. _maxNotGrayMember - maximum level of alive group member within reward distance,
40 // for whom victim is not gray;
41 // 2.6. _isFullXP - flag identifying that for all group members victim is not gray,
42 // so 100% XP will be rewarded (50% otherwise).
43 // 3. Reward killer (and group, if necessary).
44 // 3.1. If killer is in group, reward group.
45 // 3.1.1. Initialize initial XP amount based on maximum level of group member,
46 // for whom victim is not gray.
47 // 3.1.2. Alter group rate if group is in raid (not for battlegrounds).
48 // 3.1.3. Reward each group member (even dead) within reward distance (see 4. for more details).
49 // 3.2. Reward single killer (not group case).
50 // 3.2.1. Initialize initial XP amount based on killer's level.
51 // 3.2.2. Reward killer (see 4. for more details).
52 // 4. Reward player.
53 // 4.1. Give honor (player must be alive and not on BG).
54 // 4.2. Give XP.
55 // 4.2.1. If player is in group, adjust XP:
56 // * set to 0 if player's level is more than maximum level of not gray member;
57 // * cut XP in half if _isFullXP is false.
58 // 4.2.2. Apply auras modifying rewarded XP.
59 // 4.2.3. Give XP to player.
60 // 4.2.4. If player has pet, reward pet with XP (100% for single player, 50% for group case).
61 // 4.3. Give reputation (player must not be on BG).
62 // 4.4. Give kill credit (player must not be in group, or he must be alive or without corpse).
63 // 5. Credit instance encounter.
64 KillRewarder::KillRewarder(Player* killer, Unit* victim, bool isBattleGround) :
65 // 1. Initialize internal variables to default values.
66  _killer(killer), _victim(victim), _group(killer->GetGroup()),
67  _groupRate(1.0f), _maxNotGrayMember(nullptr), _count(0), _aliveSumLevel(0), _sumLevel(0), _xp(0),
68  _isFullXP(false), _maxLevel(0), _isBattleGround(isBattleGround), _isPvP(false)
69 {
70  // mark the credit as pvp if victim is player
71  if (victim->GetTypeId() == TYPEID_PLAYER)
72  _isPvP = true;
73  // or if its owned by player and its not a vehicle
74  else if (victim->GetCharmerOrOwnerGUID().IsPlayer())
75  _isPvP = !victim->IsVehicle();
76 
77  _InitGroupData();
78 }

80 inline void KillRewarder::_InitGroupData()
81 {
82  if (_group)
83  {
84  // 2. In case when player is in group, initialize variables necessary for group calculations:
85  for (GroupReference* itr = _group->GetFirstMember(); itr != nullptr; itr = itr->next())
86  if (Player* member = itr->GetSource())
87  if ((_killer == member || member->IsAtGroupRewardDistance(_victim)))
88  {
89  const uint8 lvl = member->getLevel();
90  if (member->IsAlive())
91  {
92  // 2.1. _count - number of alive group members within reward distance;
93  ++_count;
94  // 2.2. _aliveSumLevel - sum of levels of alive group members within reward distance;
95  _aliveSumLevel += lvl;
96  // 2.3. _maxLevel - maximum level of alive group member within reward distance;
97  if (_maxLevel < lvl)
98  {
99  _maxLevel = lvl;
100  }
101  // 2.4. _maxNotGrayMember - maximum level of alive group member within reward distance,
102  // for whom victim is not gray;
103  uint32 grayLevel = Acore::XP::GetGrayLevel(lvl);
104  if (_victim->getLevel() > grayLevel && (!_maxNotGrayMember || _maxNotGrayMember->getLevel() < lvl))
105  {
106  _maxNotGrayMember = member;
107  }
108  }
109  // 2.5. _sumLevel - sum of levels of group members within reward distance;
110  _sumLevel += lvl;
111  }
112  // 2.6. _isFullXP - flag identifying that for all group members victim is not gray,
113  // so 100% XP will be rewarded (50% otherwise).
114  _isFullXP = _maxNotGrayMember && (_maxLevel == _maxNotGrayMember->getLevel());
115  }
116  else
117  _count = 1;
118 }

120 inline void KillRewarder::_InitXP(Player* player)
121 {
122  // Get initial value of XP for kill.
123  // XP is given:
124  // * on battlegrounds;
125  // * otherwise, not in PvP;
126  // * not if killer is on vehicle.
127  if (_victim && (_isBattleGround || (!_isPvP && !_killer->GetVehicle())))
128  _xp = Acore::XP::Gain(player, _victim, _isBattleGround);
129 
130  if (_xp && !_isBattleGround && _victim) // pussywizard: npcs with relatively low hp give lower exp
131  if (_victim->GetTypeId() == TYPEID_UNIT)
132  if (const CreatureTemplate* ct = _victim->ToCreature()->GetCreatureTemplate())
133  if (ct->ModHealth <= 0.75f && ct->ModHealth >= 0.0f)
134  _xp = uint32(_xp * ct->ModHealth);
135 }

144 inline void KillRewarder::_RewardXP(Player* player, float rate)
145 {
146  uint32 xp(_xp);
147  if (_group)
148  {
149  // 4.2.1. If player is in group, adjust XP:
150  // * set to 0 if player's level is more than maximum level of not gray member;
151  // * cut XP in half if _isFullXP is false.
152  if (_maxNotGrayMember && player->IsAlive() &&
153  _maxNotGrayMember->getLevel() >= player->getLevel())
154  xp = _isFullXP ?
155  uint32(xp * rate) : // Reward FULL XP if all group members are not gray.
156  uint32(xp * rate / 2) + 1; // Reward only HALF of XP if some of group members are gray.
157  else
158  xp = 0;
159  }
160  if (xp)
161  {
162  // 4.2.2. Apply auras modifying rewarded XP (SPELL_AURA_MOD_XP_PCT).
163  Unit::AuraEffectList const& auras = player->GetAuraEffectsByType(SPELL_AURA_MOD_XP_PCT);
164  for (Unit::AuraEffectList::const_iterator i = auras.begin(); i != auras.end(); ++i)
165  AddPct(xp, (*i)->GetAmount());
166 
167  // 4.2.3. Give XP to player.
168  player->GiveXP(xp, _victim, _groupRate);
169  if (Pet* pet = player->GetPet())
170  // 4.2.4. If player has pet, reward pet with XP (100% for single player, 50% for group case).
171  pet->GivePetXP(_group ? xp / 2 : xp);
172  }
173 }

193 void KillRewarder::_RewardPlayer(Player* player, bool isDungeon)
194 {
195  // 4. Reward player.
196  if (!_isBattleGround)
197  {
198  // 4.1. Give honor (player must be alive and not on BG).
199  _RewardHonor(player);
200  // 4.1.1 Send player killcredit for quests with PlayerSlain
201  if (_victim->GetTypeId() == TYPEID_PLAYER)
202  player->KilledPlayerCredit();
203  }
204  // Give XP only in PvE or in battlegrounds.
205  // Give reputation and kill credit only in PvE.
206  if (!_isPvP || _isBattleGround)
207  {
208  float xpRate = _group ? _groupRate * float(player->getLevel()) / _aliveSumLevel : /*Personal rate is 100%.*/ 1.0f; // Group rate depends on the sum of levels.
209  float reputationRate = _group ? _groupRate * float(player->getLevel()) / _sumLevel : /*Personal rate is 100%.*/ 1.0f; // Group rate depends on the sum of levels.
210  sScriptMgr->OnRewardKillRewarder(player, isDungeon, xpRate); // Personal rate is 100%.
211 
212  if (_xp)
213  {
214  // 4.2. Give XP.
215  _RewardXP(player, xpRate);
216  }
217  if (!_isBattleGround)
218  {
219  // If killer is in dungeon then all members receive full reputation at kill.
220  _RewardReputation(player, isDungeon ? 1.0f : reputationRate);
221  _RewardKillCredit(player);
222  }
223  }
224 }

226 void KillRewarder::_RewardGroup()
227 {
228  if (_maxLevel)
229  {
230  if (_maxNotGrayMember)
231  // 3.1.1. Initialize initial XP amount based on maximum level of group member,
232  // for whom victim is not gray.
233  _InitXP(_maxNotGrayMember);
234  // To avoid unnecessary calculations and calls,
235  // proceed only if XP is not ZERO or player is not on battleground
236  // (battleground rewards only XP, that's why).
237  if (!_isBattleGround || _xp)
238  {
239  const bool isDungeon = !_isPvP && sMapStore.LookupEntry(_killer->GetMapId())->IsDungeon();
240  if (!_isBattleGround)
241  {
242  // 3.1.2. Alter group rate if group is in raid (not for battlegrounds).
243  const bool isRaid = !_isPvP && sMapStore.LookupEntry(_killer->GetMapId())->IsRaid() && _group->isRaidGroup();
244  _groupRate = Acore::XP::xp_in_group_rate(_count, isRaid);
245  }
246 
247  // 3.1.3. Reward each group member (even dead or corpse) within reward distance.
248  for (GroupReference* itr = _group->GetFirstMember(); itr != nullptr; itr = itr->next())
249  {
250  if (Player* member = itr->GetSource())
251  {
252  if (_killer == member || member->IsAtGroupRewardDistance(_victim))
253  {
254  _RewardPlayer(member, isDungeon);
255  // Xinef: only count players
256  //if (_victim->GetTypeId() == TYPEID_PLAYER)
257  // member->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_SPECIAL_PVP_KILL, 1, 0, _victim);
258  }
259  }
260  }
261  }
262  }

265 void KillRewarder::Reward()
266 {
267  // 3. Reward killer (and group, if necessary).
268  if (_group)
269  // 3.1. If killer is in group, reward group.
270  _RewardGroup();
271  else
272  {
273  // 3.2. Reward single killer (not group case).
274  // 3.2.1. Initialize initial XP amount based on killer's level.
275  _InitXP(_killer);
276  // To avoid unnecessary calculations and calls,
277  // proceed only if XP is not ZERO or player is not on battleground
278  // (battleground rewards only XP, that's why).
279  if (!_isBattleGround || _xp)
280  // 3.2.2. Reward killer.
281  if (_killer->IsInMap(_victim)) // pussywizard: killer may be on other map (crashfix), when killing in a group same map is required, so its not a problem
282  _RewardPlayer(_killer, false);
283  }
284 
285  // 5. Credit instance encounter.
286  if (Creature* victim = _victim->ToCreature())
287  if (victim->IsDungeonBoss())
288  if (Map* map = _victim->FindMap())
289  map->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, _victim->GetEntry(), _victim);
