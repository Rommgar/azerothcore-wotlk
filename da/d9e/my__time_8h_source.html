99 /*
100  Function to check sanity of a TIMESTAMP value
101 
102  DESCRIPTION
103  Check if a given MYSQL_TIME value fits in TIMESTAMP range.
104  This function doesn't make precise check, but rather a rough
105  estimate.
106 
107  RETURN VALUES
108  FALSE The value seems sane
109  TRUE The MYSQL_TIME value is definitely out of range
110 */
111 
112 static inline my_bool validate_timestamp_range(const MYSQL_TIME *t)
113 {
114  if ((t->year > TIMESTAMP_MAX_YEAR || t->year < TIMESTAMP_MIN_YEAR) ||
115  (t->year == TIMESTAMP_MAX_YEAR && (t->month > 1 || t->day > 19)) ||
116  (t->year == TIMESTAMP_MIN_YEAR && (t->month < 12 || t->day < 31)))
117  return FALSE;
118 
119  return TRUE;
120 }
