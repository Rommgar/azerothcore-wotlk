304  {

318  bool CheckPhaseMinions()
319  {
320  summons.RemoveNotExisting();
321  if (summons.empty())
322  {
323  ResetPlayersPhaseMask();
324  return true;
325  }
326 
327  uint32 phase = 1;
328  for (ObjectGuid const& summonGUID : summons)
329  {
330  if (Creature* summon = ObjectAccessor::GetCreature(*me, summonGUID))
331  {
332  phase |= summon->GetPhaseMask();
333  }
334  }
335 
336  Map::PlayerList const& players = me->GetMap()->GetPlayers();
337  for (auto const& i : players)
338  {
339  Player* pPlayer = i.GetSource();
340  if (pPlayer && !(pPlayer->GetPhaseMask() & phase))
341  {
342  pPlayer->RemoveAurasDueToSpell(GetPlrInsanityAuraId(pPlayer->GetPhaseMask()));
343  }
344  }
345 
346  return false;
347  }

368  void HandleDummyEffect(std::list<WorldObject*>& targets)
369  {
370  Unit* caster = GetCaster();
371  if (!caster)
372  {
373  targets.clear();
374  return;
375  }
376 
377  if (!targets.empty())
378  {
379  targets.remove_if([this](WorldObject* targetObj) -> bool
380  {
381  return !targetObj || targetObj->GetTypeId() != TYPEID_PLAYER || !targetObj->ToPlayer()->IsInCombatWith(GetCaster()) ||
382  targetObj->GetDistance(GetCaster()) >= (MAX_VISIBILITY_DISTANCE * 2);
383  });
384  }
385 
386  if (targets.empty())
387  {
388  return;
389  }
390 
391  // Start channel visual and set self as unnattackable
392  caster->ToCreature()->AI()->Talk(SAY_INSANITY);
393  caster->CastSpell(caster, SPELL_WHISPER_INSANITY, true);
394  caster->RemoveAllAuras();
395  caster->CastSpell(caster, INSANITY_VISUAL, true);
396  caster->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
397  caster->SetControlled(true, UNIT_STATE_STUNNED);
398 
399  // Handle phase effect
400  uint32 insanityCounter = 0;
401  std::list<WorldObject*>::const_iterator itr = targets.begin();
402  while (itr != targets.end() && insanityCounter < MAX_INSANITY_TARGETS)
403  {
404  WorldObject* targetObj = *itr;
405  if (!targetObj)
406  {
407  continue;
408  }
409 
410  Player* plrTarget = targetObj->ToPlayer();
411  // This should never happen, spell has attribute SPELL_ATTR3_ONLY_TARGET_PLAYERS
412  if (!plrTarget)
413  {
414  continue;
415  }
416 
417  // phase mask
418  plrTarget->CastSpell(plrTarget, InsanitySpells.at(insanityCounter), true);
419 
420  // Summon clone
421  for (std::list<WorldObject*>::const_iterator itr2 = targets.begin(); itr2 != targets.end(); ++itr2)
422  {
423  // Should not make clone of current player target
424  Player const* plrClone = *itr2 ? (*itr2)->ToPlayer() : nullptr;
425  if (!plrClone || plrClone == plrTarget)
426  {
427  continue;
428  }
429 
430  if (Unit* summon = caster->SummonCreature(NPC_TWISTED_VISAGE, plrClone->GetPosition(), TEMPSUMMON_CORPSE_DESPAWN, 0))
431  {
432  summon->AddThreat(plrTarget, 0.0f);
433  summon->SetInCombatWith(plrTarget);
434  plrTarget->SetInCombatWith(summon);
435 
436  plrTarget->CastSpell(summon, SPELL_CLONE_PLAYER, true);
437  summon->SetPhaseMask(1 | (1 << (4 + insanityCounter)), true);
438  summon->SetUInt32Value(UNIT_FIELD_MINDAMAGE, plrClone->GetUInt32Value(UNIT_FIELD_MINDAMAGE));
439  summon->SetUInt32Value(UNIT_FIELD_MAXDAMAGE, plrClone->GetUInt32Value(UNIT_FIELD_MAXDAMAGE));
440  }
441  }
442 
443  ++insanityCounter;
444  ++itr;
445  }
446  }
