266 /*
267  We define my_sigset() and use that instead of the system sigset() so that
268  we can favor an implementation based on sigaction(). On some systems, such
269  as Mac OS X, sigset() results in flags such as SA_RESTART being set, and
270  we want to make sure that no such flags are set.
271 */
272 #if defined(HAVE_SIGACTION) && !defined(my_sigset)
273 #define my_sigset(A,B) do { struct sigaction l_s; sigset_t l_set; \
274  DBUG_ASSERT((A) != 0); \
275  sigemptyset(&l_set); \
276  l_s.sa_handler = (B); \
277  l_s.sa_mask = l_set; \
278  l_s.sa_flags = 0; \
279  sigaction((A), &l_s, NULL); \
280  } while (0)
281 #elif defined(HAVE_SIGSET) && !defined(my_sigset)
282 #define my_sigset(A,B) sigset((A),(B))
