138  // Enable/Disable accepting whispers (for GM)
139  static bool HandleWhispersCommand(ChatHandler* handler, Optional<Variant<bool, EXACT_SEQUENCE("remove")>> operationArg, Optional<std::string> playerNameArg)
140  {
141  if (!operationArg)
142  {
143  handler->PSendSysMessage(LANG_COMMAND_WHISPERACCEPTING, handler->GetSession()->GetPlayer()->isAcceptWhispers() ? handler->GetAcoreString(LANG_ON) : handler->GetAcoreString(LANG_OFF));
144  return true;
145  }
146 
147  if (operationArg->holds_alternative<bool>())
148  {
149  if (operationArg->get<bool>())
150  {
151  handler->GetSession()->GetPlayer()->SetAcceptWhispers(true);
152  handler->SendSysMessage(LANG_COMMAND_WHISPERON);
153  return true;
154  }
155  else
156  {
157  // Remove all players from the Gamemaster's whisper whitelist
158  handler->GetSession()->GetPlayer()->ClearWhisperWhiteList();
159  handler->GetSession()->GetPlayer()->SetAcceptWhispers(false);
160  handler->SendSysMessage(LANG_COMMAND_WHISPEROFF);
161  return true;
162  }
163  }
164 
165  if (operationArg->holds_alternative<EXACT_SEQUENCE("remove")>())
166  {
167  if (!playerNameArg)
168  return false;
169 
170  if (normalizePlayerName(*playerNameArg))
171  {
172  if (Player* player = ObjectAccessor::FindPlayerByName(*playerNameArg))
173  {
174  handler->GetSession()->GetPlayer()->RemoveFromWhisperWhiteList(player->GetGUID());
175  handler->PSendSysMessage(LANG_COMMAND_WHISPEROFFPLAYER, playerNameArg->c_str());
176  return true;
177  }
178  else
179  {
180  handler->PSendSysMessage(LANG_PLAYER_NOT_FOUND, playerNameArg->c_str());
181  handler->SetSentErrorMessage(true);
182  return false;
183  }
184  }
185  }
186  handler->SendSysMessage(LANG_USE_BOL);
187  handler->SetSentErrorMessage(true);
188  return false;
189  }
