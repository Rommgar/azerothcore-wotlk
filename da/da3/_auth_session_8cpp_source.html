139 void AccountInfo::LoadResult(Field* fields)

141  // 0 1 2 3 4 5
142  // SELECT a.id, a.username, a.locked, a.lock_country, a.last_ip, a.failed_logins,
143  // 6 7
144  // ab.unbandate > UNIX_TIMESTAMP() OR ab.unbandate = ab.bandate, ab.unbandate = ab.bandate,
145  // 8 9
146  // ipb.unbandate > UNIX_TIMESTAMP() OR ipb.unbandate = ipb.bandate, ipb.unbandate = ipb.bandate,
147  // 10
148  // aa.gmlevel (, more query-specific fields)
149  // FROM account a LEFT JOIN account_access aa ON a.id = aa.id LEFT JOIN account_banned ab ON ab.id = a.id AND ab.active = 1 LEFT JOIN ip_banned ipb ON ipb.ip = ? WHERE a.username = ?
150 
151  Id = fields[0].GetUInt32();
152  Login = fields[1].GetString();
153  IsLockedToIP = fields[2].GetBool();
154  LockCountry = fields[3].GetString();
155  LastIP = fields[4].GetString();
156  FailedLogins = fields[5].GetUInt32();
157  IsBanned = fields[6].GetBool() || fields[8].GetBool();
158  IsPermanentlyBanned = fields[7].GetBool() || fields[9].GetBool();
159  SecurityLevel = static_cast<AccountTypes>(fields[10].GetUInt8()) > SEC_CONSOLE ? SEC_CONSOLE : static_cast<AccountTypes>(fields[10].GetUInt8());

161  // Use our own uppercasing of the account name instead of using UPPER() in mysql query

220 void AuthSession::ReadHandler()
221 {
222  MessageBuffer& packet = GetReadBuffer();
223 
224  while (packet.GetActiveSize())
225  {
226  uint8 cmd = packet.GetReadPointer()[0];
227  auto itr = Handlers.find(cmd);
228  if (itr == Handlers.end())
229  {
230  // well we dont handle this, lets just ignore it
231  packet.Reset();
232  break;
233  }
234 
235  if (_status != itr->second.status)
236  {
237  CloseSocket();
238  return;
239  }
240 
241  uint16 size = uint16(itr->second.packetSize);
242  if (packet.GetActiveSize() < size)
243  break;
244 
245  if (cmd == AUTH_LOGON_CHALLENGE || cmd == AUTH_RECONNECT_CHALLENGE)
246  {
247  sAuthLogonChallenge_C* challenge = reinterpret_cast<sAuthLogonChallenge_C*>(packet.GetReadPointer());
248  size += challenge->size;
249  if (size > MAX_ACCEPTED_CHALLENGE_SIZE)
250  {
251  CloseSocket();
252  return;
253  }
254  }
255 
256  if (packet.GetActiveSize() < size)
257  break;
258 
259  if (!(*this.*itr->second.handler)())
260  {
261  CloseSocket();
262  return;
263  }
264 
265  packet.ReadCompleted(size);
266  }
267 
268  AsyncRead();
269 }

284 bool AuthSession::HandleLogonChallenge()
285 {
286  _status = STATUS_CLOSED;
287 
288  sAuthLogonChallenge_C* challenge = reinterpret_cast<sAuthLogonChallenge_C*>(GetReadBuffer().GetReadPointer());
289  if (challenge->size - (sizeof(sAuthLogonChallenge_C) - AUTH_LOGON_CHALLENGE_INITIAL_SIZE - 1) != challenge->I_len)
290  return false;
291 
292  std::string login((char const*)challenge->I, challenge->I_len);
293  LOG_DEBUG("server.authserver", "[AuthChallenge] '%s'", login.c_str());
294 
295  _build = challenge->build;
296  _expversion = uint8(AuthHelper::IsPostBCAcceptedClientBuild(_build) ? POST_BC_EXP_FLAG : (AuthHelper::IsPreBCAcceptedClientBuild(_build) ? PRE_BC_EXP_FLAG : NO_VALID_EXP_FLAG));
297  std::array<char, 5> os;
298  os.fill('\0');
299  memcpy(os.data(), challenge->os, sizeof(challenge->os));
300  _os = os.data();
301 
302  // Restore string order as its byte order is reversed
303  std::reverse(_os.begin(), _os.end());
304 
305  _localizationName.resize(4);
306  for (int i = 0; i < 4; ++i)
307  _localizationName[i] = challenge->country[4 - i - 1];
308 
309  // Get the account details from the account table
310  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_LOGONCHALLENGE);
311  stmt->setString(0, GetRemoteIpAddress().to_string());
312  stmt->setString(1, login);
313 
314  _queryProcessor.AddCallback(LoginDatabase.AsyncQuery(stmt).WithPreparedCallback(std::bind(&AuthSession::LogonChallengeCallback, this, std::placeholders::_1)));
315  return true;
316 }

318 void AuthSession::LogonChallengeCallback(PreparedQueryResult result)
319 {
320  ByteBuffer pkt;
321  pkt << uint8(AUTH_LOGON_CHALLENGE);
322  pkt << uint8(0x00);
323 
324  if (!result)
325  {
326  pkt << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
327  SendPacket(pkt);
328  return;
329  }
330 
331  Field* fields = result->Fetch();
332 
333  _accountInfo.LoadResult(fields);
334 
335  std::string ipAddress = GetRemoteIpAddress().to_string();
336  uint16 port = GetRemotePort();
337 
338  // If the IP is 'locked', check that the player comes indeed from the correct IP address
339  if (_accountInfo.IsLockedToIP)
340  {
341  LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is locked to IP - '%s' is logging in from '%s'", _accountInfo.Login.c_str(), _accountInfo.LastIP.c_str(), ipAddress.c_str());
342  if (_accountInfo.LastIP != ipAddress)
343  {
344  pkt << uint8(WOW_FAIL_LOCKED_ENFORCED);
345  SendPacket(pkt);
346  return;
347  }
348  }
349  else
350  {
351  if (IpLocationRecord const* location = sIPLocation->GetLocationRecord(ipAddress))
352  _ipCountry = location->CountryCode;
353 
354  LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is not locked to ip", _accountInfo.Login.c_str());
355  if (_accountInfo.LockCountry.empty() || _accountInfo.LockCountry == "00")
356  LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is not locked to country", _accountInfo.Login.c_str());
357  else if (!_ipCountry.empty())
358  {
359  LOG_DEBUG("server.authserver", "[AuthChallenge] Account '%s' is locked to country: '%s' Player country is '%s'", _accountInfo.Login.c_str(), _accountInfo.LockCountry.c_str(), _ipCountry.c_str());
360  if (_ipCountry != _accountInfo.LockCountry)
361  {
362  pkt << uint8(WOW_FAIL_UNLOCKABLE_LOCK);
363  SendPacket(pkt);
364  return;
365  }
366  }
367  }
368 
369  // If the account is banned, reject the logon attempt
370  if (_accountInfo.IsBanned)
371  {
372  if (_accountInfo.IsPermanentlyBanned)
373  {
374  pkt << uint8(WOW_FAIL_BANNED);
375  SendPacket(pkt);
376  LOG_INFO("server.authserver.banned", "'%s:%d' [AuthChallenge] Banned account %s tried to login!", ipAddress.c_str(), port, _accountInfo.Login.c_str());
377  return;
378  }
379  else
380  {
381  pkt << uint8(WOW_FAIL_SUSPENDED);
382  SendPacket(pkt);
383  LOG_INFO("server.authserver.banned", "'%s:%d' [AuthChallenge] Temporarily banned account %s tried to login!", ipAddress.c_str(), port, _accountInfo.Login.c_str());
384  return;
385  }
386  }
387 
388  uint8 securityFlags = 0;
389 
390  // Check if a TOTP token is needed
391  if (!fields[11].IsNull())
392  {
393  securityFlags = 4;
394  _totpSecret = fields[11].GetBinary();
395 
396  if (auto const& secret = sSecretMgr->GetSecret(SECRET_TOTP_MASTER_KEY))
397  {
398  bool success = Acore::Crypto::AEDecrypt<Acore::Crypto::AES>(*_totpSecret, *secret);
399  if (!success)
400  {
401  pkt << uint8(WOW_FAIL_DB_BUSY);
402  LOG_ERROR("server.authserver", "[AuthChallenge] Account '%s' has invalid ciphertext for TOTP token key stored", _accountInfo.Login.c_str());
403  SendPacket(pkt);
404  return;
405  }
406  }
407  }
408 
409  _srp6.emplace(
410  _accountInfo.Login,
411  fields[12].GetBinary<Acore::Crypto::SRP6::SALT_LENGTH>(),
412  fields[13].GetBinary<Acore::Crypto::SRP6::VERIFIER_LENGTH>());
413 
414  // Fill the response packet with the result
415  if (AuthHelper::IsAcceptedClientBuild(_build))
416  {
417  pkt << uint8(WOW_SUCCESS);
418 
419  pkt.append(_srp6->B);
420  pkt << uint8(1);
421  pkt.append(_srp6->g);
422  pkt << uint8(32);
423  pkt.append(_srp6->N);
424  pkt.append(_srp6->s);
425  pkt.append(VersionChallenge.data(), VersionChallenge.size());
426  pkt << uint8(securityFlags); // security flags (0x0...0x04)
427 
428  if (securityFlags & 0x01) // PIN input
429  {
430  pkt << uint32(0);
431  pkt << uint64(0) << uint64(0); // 16 bytes hash?
432  }
433 
434  if (securityFlags & 0x02) // Matrix input
435  {
436  pkt << uint8(0);
437  pkt << uint8(0);
438  pkt << uint8(0);
439  pkt << uint8(0);
440  pkt << uint64(0);
441  }
442 
443  if (securityFlags & 0x04) // Security token input
444  pkt << uint8(1);
445 
446  LOG_DEBUG("server.authserver", "'%s:%d' [AuthChallenge] account %s is using '%s' locale (%u)",
447  ipAddress.c_str(), port, _accountInfo.Login.c_str(), _localizationName.c_str(), GetLocaleByName(_localizationName));
448 
449  _status = STATUS_LOGON_PROOF;
450  }
451  else
452  pkt << uint8(WOW_FAIL_VERSION_INVALID);
453 
454  SendPacket(pkt);
455 }

457 // Logon Proof command handler
458 bool AuthSession::HandleLogonProof()
459 {
460  LOG_DEBUG("server.authserver", "Entering _HandleLogonProof");
461  _status = STATUS_CLOSED;
462 
463  // Read the packet
464  sAuthLogonProof_C* logonProof = reinterpret_cast<sAuthLogonProof_C*>(GetReadBuffer().GetReadPointer());
465 
466  // If the client has no valid version
467  if (_expversion == NO_VALID_EXP_FLAG)
468  {
469  // Check if we have the appropriate patch on the disk
470  LOG_DEBUG("network", "Client with invalid version, patching is not implemented");
471  return false;
472  }
473 
474  // Check if SRP6 results match (password is correct), else send an error
475  if (Optional<SessionKey> K = _srp6->VerifyChallengeResponse(logonProof->A, logonProof->clientM))
476  {
477  _sessionKey = *K;
478  // Check auth token
479  bool tokenSuccess = false;
480  bool sentToken = (logonProof->securityFlags & 0x04);
481  if (sentToken && _totpSecret)
482  {
483  uint8 size = *(GetReadBuffer().GetReadPointer() + sizeof(sAuthLogonProof_C));
484  std::string token(reinterpret_cast<char*>(GetReadBuffer().GetReadPointer() + sizeof(sAuthLogonProof_C) + sizeof(size)), size);
485  GetReadBuffer().ReadCompleted(sizeof(size) + size);
486 
487  uint32 incomingToken = atoi(token.c_str());
488  tokenSuccess = Acore::Crypto::TOTP::ValidateToken(*_totpSecret, incomingToken);
489  memset(_totpSecret->data(), 0, _totpSecret->size());
490  }
491  else if (!sentToken && !_totpSecret)
492  tokenSuccess = true;
493 
494  if (!tokenSuccess)
495  {
496  ByteBuffer packet;
497  packet << uint8(AUTH_LOGON_PROOF);
498  packet << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
499  packet << uint16(0); // LoginFlags, 1 has account message
500  SendPacket(packet);
501  return true;
502  }
503 
504  if (!VerifyVersion(logonProof->A.data(), logonProof->A.size(), logonProof->crc_hash, false))
505  {
506  ByteBuffer packet;
507  packet << uint8(AUTH_LOGON_PROOF);
508  packet << uint8(WOW_FAIL_VERSION_INVALID);
509  SendPacket(packet);
510  return true;
511  }
512 
513  LOG_DEBUG("server.authserver", "'%s:%d' User '%s' successfully authenticated", GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo.Login.c_str());
514 
515  // Update the sessionkey, last_ip, last login time and reset number of failed logins in the account table for this account
516  // No SQL injection (escaped user name) and IP address as received by socket
517 
518  std::string address = sConfigMgr->GetOption<bool>("AllowLoggingIPAddressesInDatabase", true, true) ? GetRemoteIpAddress().to_string() : "0.0.0.0";
519  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_LOGONPROOF);
520  stmt->setBinary(0, _sessionKey);
521  stmt->setString(1, address);
522  stmt->setUInt32(2, GetLocaleByName(_localizationName));
523  stmt->setString(3, _os);
524  stmt->setString(4, _accountInfo.Login);
525  LoginDatabase.DirectExecute(stmt);
526 
527  // Finish SRP6 and send the final result to the client
528  Acore::Crypto::SHA1::Digest M2 = Acore::Crypto::SRP6::GetSessionVerifier(logonProof->A, logonProof->clientM, _sessionKey);
529 
530  ByteBuffer packet;
531  if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
532  {
533  sAuthLogonProof_S proof;
534  proof.M2 = M2;
535  proof.cmd = AUTH_LOGON_PROOF;
536  proof.error = 0;
537  proof.AccountFlags = 0x00800000; // 0x01 = GM, 0x08 = Trial, 0x00800000 = Pro pass (arena tournament)
538  proof.SurveyId = 0;
539  proof.LoginFlags = 0; // 0x1 = has account message
540 
541  packet.resize(sizeof(proof));
542  std::memcpy(packet.contents(), &proof, sizeof(proof));
543  }
544  else
545  {
546  sAuthLogonProof_S_Old proof;
547  proof.M2 = M2;
548  proof.cmd = AUTH_LOGON_PROOF;
549  proof.error = 0;
550  proof.unk2 = 0x00;
551 
552  packet.resize(sizeof(proof));
553  std::memcpy(packet.contents(), &proof, sizeof(proof));
554  }
555 
556  SendPacket(packet);
557  _status = STATUS_AUTHED;
558  }
559  else
560  {
561  ByteBuffer packet;
562  packet << uint8(AUTH_LOGON_PROOF);
563  packet << uint8(WOW_FAIL_UNKNOWN_ACCOUNT);
564  packet << uint16(0); // LoginFlags, 1 has account message
565  SendPacket(packet);
566 
567  LOG_INFO("server.authserver.hack", "'%s:%d' [AuthChallenge] account %s tried to login with invalid password!",
568  GetRemoteIpAddress().to_string().c_str(), GetRemotePort(), _accountInfo.Login.c_str());
569 
570  uint32 MaxWrongPassCount = sConfigMgr->GetOption<int32>("WrongPass.MaxCount", 0);
571 
572  // We can not include the failed account login hook. However, this is a workaround to still log this.
573  if (sConfigMgr->GetOption<bool>("WrongPass.Logging", false))
574  {
575  LoginDatabasePreparedStatement* logstmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_FALP_IP_LOGGING);
576  logstmt->setUInt32(0, _accountInfo.Id);
577  logstmt->setString(1, GetRemoteIpAddress().to_string());
578  logstmt->setString(2, "Login to WoW Failed - Incorrect Password");
579 
580  LoginDatabase.Execute(logstmt);
581  }
582 
583  if (MaxWrongPassCount > 0)
584  {
585  //Increment number of failed logins by one and if it reaches the limit temporarily ban that account or IP
586  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_UPD_FAILEDLOGINS);
587  stmt->setString(0, _accountInfo.Login);
588  LoginDatabase.Execute(stmt);
589 
590  if (++_accountInfo.FailedLogins >= MaxWrongPassCount)
591  {
592  uint32 WrongPassBanTime = sConfigMgr->GetOption<int32>("WrongPass.BanTime", 600);
593  bool WrongPassBanType = sConfigMgr->GetOption<bool>("WrongPass.BanType", false);
594 
595  if (WrongPassBanType)
596  {
597  stmt = LoginDatabase.GetPreparedStatement(LOGIN_INS_ACCOUNT_AUTO_BANNED);
598  stmt->setUInt32(0, _accountInfo.Id);
599  stmt->setUInt32(1, WrongPassBanTime);

718  pkt << uint16(0); // LoginFlags, 1 has account message
719  SendPacket(pkt);

743 void AuthSession::RealmListCallback(PreparedQueryResult result)
744 {
745  std::map<uint32, uint8> characterCounts;
746  if (result)
747  {
748  do
749  {
750  Field* fields = result->Fetch();
751  characterCounts[fields[0].GetUInt32()] = fields[1].GetUInt8();
752  } while (result->NextRow());
753  }
754 
755  // Circle through realms in the RealmList and construct the return packet (including # of user characters in each realm)
756  ByteBuffer pkt;
757 
758  size_t RealmListSize = 0;
759  for (auto const& [realmHandle, realm] : sRealmList->GetRealms())
760  {
761  // don't work with realms which not compatible with the client
762  bool okBuild = ((_expversion & POST_BC_EXP_FLAG) && realm.Build == _build) || ((_expversion & PRE_BC_EXP_FLAG) && !AuthHelper::IsPreBCAcceptedClientBuild(realm.Build));
763 
764  // No SQL injection. id of realm is controlled by the database.
765  uint32 flag = realm.Flags;
766  RealmBuildInfo const* buildInfo = sRealmList->GetBuildInfo(realm.Build);
767  if (!okBuild)
768  {
769  if (!buildInfo)
770  continue;
771 
772  flag |= REALM_FLAG_OFFLINE | REALM_FLAG_SPECIFYBUILD; // tell the client what build the realm is for
773  }
774 
775  if (!buildInfo)
776  flag &= ~REALM_FLAG_SPECIFYBUILD;
777 
778  std::string name = realm.Name;
779  if (_expversion & PRE_BC_EXP_FLAG && flag & REALM_FLAG_SPECIFYBUILD)
780  {
781  std::ostringstream ss;
782  ss << name << " (" << buildInfo->MajorVersion << '.' << buildInfo->MinorVersion << '.' << buildInfo->BugfixVersion << ')';
783  name = ss.str();
784  }
785 
786  uint8 lock = (realm.AllowedSecurityLevel > _accountInfo.SecurityLevel) ? 1 : 0;
787 
788  pkt << uint8(realm.Type); // realm type
789  if (_expversion & POST_BC_EXP_FLAG) // only 2.x and 3.x clients
790  pkt << uint8(lock); // if 1, then realm locked
791 
792  pkt << uint8(flag); // RealmFlags
793  pkt << name;
794  pkt << boost::lexical_cast<std::string>(realm.GetAddressForClient(GetRemoteIpAddress()));
795  pkt << float(realm.PopulationLevel);
796  pkt << uint8(characterCounts[realm.Id.Realm]);
797  pkt << uint8(realm.Timezone); // realm category
798 
799  if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
800  pkt << uint8(realm.Id.Realm);
801  else
802  pkt << uint8(0x0); // 1.12.1 and 1.12.2 clients
803 
804  if (_expversion & POST_BC_EXP_FLAG && flag & REALM_FLAG_SPECIFYBUILD)
805  {
806  pkt << uint8(buildInfo->MajorVersion);
807  pkt << uint8(buildInfo->MinorVersion);
808  pkt << uint8(buildInfo->BugfixVersion);
809  pkt << uint16(buildInfo->Build);
810  }
811 
812  ++RealmListSize;
813  }
814 
815  if (_expversion & POST_BC_EXP_FLAG) // 2.x and 3.x clients
816  {
817  pkt << uint8(0x10);
818  pkt << uint8(0x00);
819  }
820  else // 1.12.1 and 1.12.2 clients
821  {
822  pkt << uint8(0x00);
823  pkt << uint8(0x02);
824  }
825 
826  // make a ByteBuffer which stores the RealmList's size
827  ByteBuffer RealmListSizeBuffer;
828  RealmListSizeBuffer << uint32(0);
829 
830  if (_expversion & POST_BC_EXP_FLAG) // only 2.x and 3.x clients
831  RealmListSizeBuffer << uint16(RealmListSize);
832  else
833  RealmListSizeBuffer << uint32(RealmListSize);
834 
835  ByteBuffer hdr;
836  hdr << uint8(REALM_LIST);
837  hdr << uint16(pkt.size() + RealmListSizeBuffer.size());
838  hdr.append(RealmListSizeBuffer); // append RealmList's size buffer
839  hdr.append(pkt); // append realms in the realmlist
840  SendPacket(hdr);
841 
842  _status = STATUS_AUTHED;
843 }

845 bool AuthSession::VerifyVersion(uint8 const* a, int32 aLength, Acore::Crypto::SHA1::Digest const& versionProof, bool isReconnect)
846 {
847  if (!sConfigMgr->GetOption<bool>("StrictVersionCheck", false))
848  return true;
849 
850  Acore::Crypto::SHA1::Digest zeros;
851  Acore::Crypto::SHA1::Digest const* versionHash = nullptr;
852 
853  if (!isReconnect)
854  {
855  RealmBuildInfo const* buildInfo = sRealmList->GetBuildInfo(_build);
856  if (!buildInfo)
857  return false;
858 
859  if (_os == "Win")
860  versionHash = &buildInfo->WindowsHash;
861  else if (_os == "OSX")
862  versionHash = &buildInfo->MacHash;
863 
864  if (!versionHash)
865  return false;
866 
867  if (zeros == *versionHash)
868  return true; // not filled serverside
869  }
870  else
871  versionHash = &zeros;
872 
873  Acore::Crypto::SHA1 version;
874  version.UpdateData(a, aLength);
875  version.UpdateData(*versionHash);
876  version.Finalize();
877 
878  return (versionProof == version.GetDigest());
879 }
