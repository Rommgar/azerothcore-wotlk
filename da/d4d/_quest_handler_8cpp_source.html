38 void WorldSession::HandleQuestgiverStatusQueryOpcode(WorldPacket& recvData)
39 {
40  ObjectGuid guid;
41  recvData >> guid;
42  uint32 questStatus = DIALOG_STATUS_NONE;
43 
44  Object* questGiver = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
45  if (!questGiver)
46  {
47  LOG_DEBUG("network.opcode", "Error in CMSG_QUESTGIVER_STATUS_QUERY, called for not found questgiver (%s)", guid.ToString().c_str());
48  return;
49  }
50 
51  switch (questGiver->GetTypeId())
52  {
53  case TYPEID_UNIT:
54  {
55  LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for npc %s", guid.ToString().c_str());
56  if (!questGiver->ToCreature()->IsHostileTo(_player)) // do not show quest status to enemies
57  questStatus = _player->GetQuestDialogStatus(questGiver);
58  break;
59  }
60  case TYPEID_GAMEOBJECT:
61  {
62  LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_STATUS_QUERY for GameObject %s", guid.ToString().c_str());
63  questStatus = _player->GetQuestDialogStatus(questGiver);
64  break;
65  }
66  default:
67  LOG_ERROR("network.opcode", "QuestGiver called for unexpected type %u", questGiver->GetTypeId());
68  break;
69  }
70 
71  // inform client about status of quest
72  _player->PlayerTalkClass->SendQuestGiverStatus(uint8(questStatus), guid);
73 }

110 void WorldSession::HandleQuestgiverAcceptQuestOpcode(WorldPacket& recvData)
111 {
112  ObjectGuid guid;
113  uint32 questId;
114  uint32 unk1;
115  recvData >> guid >> questId >> unk1;
116 
117  LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_ACCEPT_QUEST npc %s, quest = %u, unk1 = %u", guid.ToString().c_str(), questId, unk1);
118 
119  Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT | TYPEMASK_ITEM | TYPEMASK_PLAYER);
120 
121  // no or incorrect quest giver
122  if (!object || object == _player || (object->GetTypeId() != TYPEID_PLAYER && !object->hasQuest(questId)) ||
123  (object->GetTypeId() == TYPEID_PLAYER && !object->ToPlayer()->CanShareQuest(questId)))
124  {
125  _player->PlayerTalkClass->SendCloseGossip();
126  _player->SetDivider();
127  return;
128  }
129 
130  // some kind of WPE protection
131  if (!_player->CanInteractWithQuestGiver(object))
132  return;
133 
134  if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
135  {
136  // pussywizard: exploit fix, can't share quests that give items to be sold
137  if (object->GetTypeId() == TYPEID_PLAYER)
138  if (uint32 itemId = quest->GetSrcItemId())
139  if (const ItemTemplate* srcItem = sObjectMgr->GetItemTemplate(itemId))
140  if (srcItem->SellPrice > 0)
141  return;
142 
143  // prevent cheating
144  if (!GetPlayer()->CanTakeQuest(quest, true))
145  {
146  _player->PlayerTalkClass->SendCloseGossip();
147  _player->SetDivider();
148  return;
149  }
150 
151  if (_player->GetDivider())
152  {
153  Player* player = ObjectAccessor::GetPlayer(*_player, _player->GetDivider());
154  if (player)
155  {
156  player->SendPushToPartyResponse(_player, QUEST_PARTY_MSG_ACCEPT_QUEST);
157  _player->SetDivider();
158  }
159  }
160 
161  if (_player->CanAddQuest(quest, true))
162  {
163  _player->AddQuestAndCheckCompletion(quest, object);
164 
165  if (quest->HasFlag(QUEST_FLAGS_PARTY_ACCEPT))
166  {
167  if (Group* group = _player->GetGroup())
168  {
169  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
170  {
171  Player* itrPlayer = itr->GetSource();
172  if (!itrPlayer || itrPlayer == _player || !itrPlayer->IsAtGroupRewardDistance(_player) || itrPlayer->HasPendingBind()) // xinef: check range
173  continue;
174 
175  if (itrPlayer->CanTakeQuest(quest, false))
176  {
177  itrPlayer->SetDivider(_player->GetGUID());
178 
179  // need confirmation that any gossip window will close
180  itrPlayer->PlayerTalkClass->SendCloseGossip();
181 
182  _player->SendQuestConfirmAccept(quest, itrPlayer);
183  }
184  }
185  }
186  }
187 
188  _player->PlayerTalkClass->SendCloseGossip();
189 
190  if (quest->GetSrcSpell() > 0)
191  _player->CastSpell(_player, quest->GetSrcSpell(), true);
192 
193  return;
194  }
195  }
196 
197  _player->PlayerTalkClass->SendCloseGossip();
198 }

247 void WorldSession::HandleQuestgiverChooseRewardOpcode(WorldPacket& recvData)
248 {
249  uint32 questId, reward;
250  ObjectGuid guid;
251  recvData >> guid >> questId >> reward;
252 
253  if (reward >= QUEST_REWARD_CHOICES_COUNT)
254  {
255  LOG_ERROR("network.opcode", "Error in CMSG_QUESTGIVER_CHOOSE_REWARD: player %s (%s) tried to get invalid reward (%u) (probably packet hacking)",
256  _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), reward);
257  return;
258  }
259 
260  LOG_DEBUG("network", "WORLD: Received CMSG_QUESTGIVER_CHOOSE_REWARD npc %s, quest = %u, reward = %u", guid.ToString().c_str(), questId, reward);
261 
262  Object* object = ObjectAccessor::GetObjectByTypeMask(*_player, guid, TYPEMASK_UNIT | TYPEMASK_GAMEOBJECT);
263  if (!object || !object->hasInvolvedQuest(questId))
264  return;
265 
266  // some kind of WPE protection
267  if (!_player->CanInteractWithQuestGiver(object))
268  return;
269 
270  if (Quest const* quest = sObjectMgr->GetQuestTemplate(questId))
271  {
272  if ((!_player->CanSeeStartQuest(quest) && _player->GetQuestStatus(questId) == QUEST_STATUS_NONE) ||
273  (_player->GetQuestStatus(questId) != QUEST_STATUS_COMPLETE && !quest->IsAutoComplete() && quest->GetQuestMethod()))
274  {
275  LOG_ERROR("network.opcode", "HACK ALERT: Player %s (%s) is trying to complete quest (id: %u) but he has no right to do it!",
276  _player->GetName().c_str(), _player->GetGUID().ToString().c_str(), questId);
277  return;
278  }
279  if (_player->CanRewardQuest(quest, reward, true))
280  {
281  _player->RewardQuest(quest, reward, object);
282 
283  // Special dialog status update (client does not query this)
284  if (!quest->GetQuestMethod())
285  {
286  _player->PlayerTalkClass->SendQuestGiverStatus(uint8(_player->GetQuestDialogStatus(object)), guid);
287  }
288 
289  switch (object->GetTypeId())
290  {
291  case TYPEID_UNIT:
292  {
293  Creature* questgiver = object->ToCreature();
294  if (!sScriptMgr->OnQuestReward(_player, questgiver, quest, reward))
295  {
296  // Send next quest
297  if (Quest const* nextQuest = _player->GetNextQuest(guid, quest))
298  {
299  if (_player->CanTakeQuest(nextQuest, false))
300  {
301  if (nextQuest->IsAutoAccept())
302  {
303  // QUEST_FLAGS_AUTO_ACCEPT was not used by Blizzard.
304  if (_player->CanAddQuest(nextQuest, false))
305  {
306  _player->AddQuestAndCheckCompletion(nextQuest, object);
307  }
308  else
309  {
310  // Auto accept is set for a custom quest and there is no inventory space
311  _player->PlayerTalkClass->SendCloseGossip();
312  break;
313  }
314  }
315  _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, guid, true);
316  }
317  }
318 
319  questgiver->AI()->sQuestReward(_player, quest, reward);
320  }
321  break;
322  }
323  case TYPEID_GAMEOBJECT:
324  {
325  GameObject* questGiver = object->ToGameObject();
326  if (!sScriptMgr->OnQuestReward(_player, questGiver, quest, reward))
327  {
328  // Send next quest
329  if (Quest const* nextQuest = _player->GetNextQuest(guid, quest))
330  {
331  if (_player->CanAddQuest(nextQuest, false) && _player->CanTakeQuest(nextQuest, false))
332  {
333  if (nextQuest->IsAutoAccept())
334  _player->AddQuestAndCheckCompletion(nextQuest, object);
335  _player->PlayerTalkClass->SendQuestGiverQuestDetails(nextQuest, guid, true);
336  }
337  }
338 
339  questGiver->AI()->QuestReward(_player, quest, reward);
340  }
341  break;
342  }
343  default:
344  break;
345  }
346  }
347  else
348  _player->PlayerTalkClass->SendQuestGiverOfferReward(quest, guid, true);
349  }
350 }

538 void WorldSession::HandlePushQuestToParty(WorldPacket& recvPacket)

546  LOG_DEBUG("network", "WORLD: Received CMSG_PUSHQUESTTOPARTY quest = %u", questId);
547

549  {
550  if (Group* group = _player->GetGroup())
551  {
552  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
553  {
554  Player* player = itr->GetSource();
555 
556  if (!player || player == _player || !player->IsInMap(_player)) // skip self
557  continue;
558

560  {
561  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_HAVE_QUEST);
562  continue;
563  }
564

566  {
567  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_FINISH_QUEST);
568  continue;
569  }
570 
571  if (!player->CanTakeQuest(quest, false))
572  {
573  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_CANT_TAKE_QUEST);
574  continue;
575  }
576 
577  if (!player->SatisfyQuestLog(false))
578  {
579  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_LOG_FULL);
580  continue;
581  }
582 
583  // Check if Quest Share in BG is enabled
584  if (sWorld->getBoolConfig(CONFIG_BATTLEGROUND_DISABLE_QUEST_SHARE_IN_BG))
585  {
586  // Check if player is in BG
587  if (_player->InBattleground())
588  {
589  _player->GetSession()->SendNotification(LANG_BG_SHARE_QUEST_ERROR);

591  }
592  }
593 
594  if (player->GetDivider())
595  {
596  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_BUSY);
597  continue;
598  }
599 
600  _player->SendPushToPartyResponse(player, QUEST_PARTY_MSG_SHARING_QUEST);
601 
602  if (quest->IsAutoAccept() && player->CanAddQuest(quest, true) && player->CanTakeQuest(quest, true))
603  player->AddQuestAndCheckCompletion(quest, _player);
604 
605  if (quest->IsAutoComplete() || !quest->GetQuestMethod())
606  player->PlayerTalkClass->SendQuestGiverRequestItems(quest, _player->GetGUID(), player->CanCompleteRepeatableQuest(quest), true);
607  else
608  {
609  player->SetDivider(_player->GetGUID());
610  player->PlayerTalkClass->SendQuestGiverQuestDetails(quest, player->GetGUID(), true);
611  }
612  }
613  }
