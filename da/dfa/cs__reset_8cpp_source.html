94  static bool HandleResetStatsOrLevelHelper(Player* player)
95  {
96  ChrClassesEntry const* classEntry = sChrClassesStore.LookupEntry(player->getClass());
97  if (!classEntry)
98  {
99  LOG_ERROR("dbc", "Class %u not found in DBC (Wrong DBC files?)", player->getClass());
100  return false;
101  }
102

104 
105  // reset m_form if no aura
106  if (!player->HasAuraType(SPELL_AURA_MOD_SHAPESHIFT))
107  player->SetShapeshiftForm(FORM_NONE);
108 
109  player->SetFactionForRace(player->getRace());
110 
111  player->SetUInt32Value(UNIT_FIELD_BYTES_0, ((player->getRace()) | (player->getClass() << 8) | (player->getGender() << 16) | (powerType << 24)));
112 
113  // reset only if player not in some form;
114  if (player->GetShapeshiftForm() == FORM_NONE)
115  player->InitDisplayIds();
116 
117  player->SetByteValue(UNIT_FIELD_BYTES_2, 1, UNIT_BYTE2_FLAG_PVP);
118 
119  player->SetUInt32Value(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED);
120 
121  //-1 is default value
122  player->SetUInt32Value(PLAYER_FIELD_WATCHED_FACTION_INDEX, uint32(-1));

269  static bool HandleResetAllCommand(ChatHandler* handler, char const* args)
270  {
271  if (!*args)
272  return false;
273 
274  std::string caseName = args;
275 
276  AtLoginFlags atLogin;
277 
278  // Command specially created as single command to prevent using short case names
279  if (caseName == "spells")
280  {
281  atLogin = AT_LOGIN_RESET_SPELLS;
282  sWorld->SendWorldText(LANG_RESETALL_SPELLS);
283  if (!handler->GetSession())
284  handler->SendSysMessage(LANG_RESETALL_SPELLS);
285  }
286  else if (caseName == "talents")
287  {
288  atLogin = AtLoginFlags(AT_LOGIN_RESET_TALENTS | AT_LOGIN_RESET_PET_TALENTS);
289  sWorld->SendWorldText(LANG_RESETALL_TALENTS);
290  if (!handler->GetSession())
291  handler->SendSysMessage(LANG_RESETALL_TALENTS);
292  }
293  else
294  {
295  handler->PSendSysMessage(LANG_RESETALL_UNKNOWN_CASE, args);
296  handler->SetSentErrorMessage(true);
297  return false;
298  }
299 
300  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ALL_AT_LOGIN_FLAGS);
301  stmt->setUInt16(0, uint16(atLogin));
302  CharacterDatabase.Execute(stmt);
303 
304  std::shared_lock<std::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
305  HashMapHolder<Player>::MapType const& plist = ObjectAccessor::GetPlayers();
306  for (HashMapHolder<Player>::MapType::const_iterator itr = plist.begin(); itr != plist.end(); ++itr)
307  itr->second->SetAtLoginFlag(atLogin);
308 
309  return true;
310  }
