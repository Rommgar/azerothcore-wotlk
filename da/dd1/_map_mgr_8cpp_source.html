136 Map::EnterState MapMgr::PlayerCannotEnter(uint32 mapid, Player* player, bool loginCheck)
137 {
138  MapEntry const* entry = sMapStore.LookupEntry(mapid);
139  if (!entry)
140  return Map::CANNOT_ENTER_NO_ENTRY;
141 
142  if (!entry->IsDungeon())
143  return Map::CAN_ENTER;
144 
145  InstanceTemplate const* instance = sObjectMgr->GetInstanceTemplate(mapid);
146  if (!instance)
147  return Map::CANNOT_ENTER_UNINSTANCED_DUNGEON;
148 
149  Difficulty targetDifficulty, requestedDifficulty;
150  targetDifficulty = requestedDifficulty = player->GetDifficulty(entry->IsRaid());
151  // Get the highest available difficulty if current setting is higher than the instance allows
152  MapDifficulty const* mapDiff = GetDownscaledMapDifficultyData(entry->MapID, targetDifficulty);
153  if (!mapDiff)
154  {
155  player->SendTransferAborted(mapid, TRANSFER_ABORT_DIFFICULTY, requestedDifficulty);
156  return Map::CANNOT_ENTER_DIFFICULTY_UNAVAILABLE;
157  }
158 
159  //Bypass checks for GMs
160  if (player->IsGameMaster())
161  return Map::CAN_ENTER;
162 
163  char const* mapName = entry->name[player->GetSession()->GetSessionDbcLocale()];
164 
165  if (!sScriptMgr->CanEnterMap(player, entry, instance, mapDiff, loginCheck))
166  return Map::CANNOT_ENTER_UNSPECIFIED_REASON;
167 
168  Group* group = player->GetGroup();
169  if (entry->IsRaid())
170  {
171  // can only enter in a raid group
172  if ((!group || !group->isRaidGroup()) && !sWorld->getBoolConfig(CONFIG_INSTANCE_IGNORE_RAID))
173  {
174  // probably there must be special opcode, because client has this string constant in GlobalStrings.lua
175  // TODO: this is not a good place to send the message
176  player->GetSession()->SendAreaTriggerMessage(player->GetSession()->GetAcoreString(LANG_INSTANCE_RAID_GROUP_ONLY), mapName);
177  LOG_DEBUG("maps", "MAP: Player '%s' must be in a raid group to enter instance '%s'", player->GetName().c_str(), mapName);
178  return Map::CANNOT_ENTER_NOT_IN_RAID;
179  }
180  }
181 
182  // xinef: dont allow LFG Group to enter other instance that is selected
183  if (group)
184  if (group->isLFGGroup())
185  if (!sLFGMgr->inLfgDungeonMap(group->GetGUID(), mapid, targetDifficulty))
186  {
187  player->SendTransferAborted(mapid, TRANSFER_ABORT_MAP_NOT_ALLOWED);
188  return Map::CANNOT_ENTER_UNSPECIFIED_REASON;
189  }
190 
191  if (!player->IsAlive())
192  {
193  if (player->HasCorpse())
194  {
195  // let enter in ghost mode in instance that connected to inner instance with corpse
196  uint32 corpseMap = player->GetCorpseLocation().GetMapId();
197  do
198  {
199  if (corpseMap == mapid)
200  break;
201 
202  InstanceTemplate const* corpseInstance = sObjectMgr->GetInstanceTemplate(corpseMap);
203  corpseMap = corpseInstance ? corpseInstance->Parent : 0;
204  } while (corpseMap);
205 
206  if (!corpseMap)
207  {
208  WorldPacket data(SMSG_CORPSE_NOT_IN_INSTANCE, 0);
209  player->GetSession()->SendPacket(&data);
210  LOG_DEBUG("maps", "MAP: Player '%s' does not have a corpse in instance '%s' and cannot enter.", player->GetName().c_str(), mapName);
211  return Map::CANNOT_ENTER_CORPSE_IN_DIFFERENT_INSTANCE;
212  }
213  LOG_DEBUG("maps", "MAP: Player '%s' has corpse in instance '%s' and can enter.", player->GetName().c_str(), mapName);
214  }
215  else
216  {
217  LOG_DEBUG("maps", "Map::PlayerCannotEnter - player '%s' is dead but does not have a corpse!", player->GetName().c_str());
218  }
219  }
220 
221  // if map exists - check for being full, etc.
222  if (!loginCheck) // for login this is done by the calling function
223  {
224  uint32 destInstId = sInstanceSaveMgr->PlayerGetDestinationInstanceId(player, mapid, targetDifficulty);
225  if (destInstId)
226  if (Map* boundMap = sMapMgr->FindMap(mapid, destInstId))
227  if (Map::EnterState denyReason = boundMap->CannotEnter(player, loginCheck))
228  return denyReason;
229  }
230 
231  // players are only allowed to enter 5 instances per hour
232  if (entry->IsDungeon() && (!group || !group->isLFGGroup() || !group->IsLfgRandomInstance()))
233  {
234  uint32 instaceIdToCheck = 0;
235  if (InstanceSave* save = sInstanceSaveMgr->PlayerGetInstanceSave(player->GetGUID(), mapid, player->GetDifficulty(entry->IsRaid())))
236  instaceIdToCheck = save->GetInstanceId();
237 
238  // instaceIdToCheck can be 0 if save not found - means no bind so the instance is new
239  if (!player->CheckInstanceCount(instaceIdToCheck) && !player->isDead())
240  {
241  player->SendTransferAborted(mapid, TRANSFER_ABORT_TOO_MANY_INSTANCES);
242  return Map::CANNOT_ENTER_TOO_MANY_INSTANCES;
243  }
244  }
245 
246  //Other requirements
247  return player->Satisfy(sObjectMgr->GetAccessRequirement(mapid, targetDifficulty), mapid, true) ? Map::CAN_ENTER : Map::CANNOT_ENTER_UNSPECIFIED_REASON;
248 }

250 void MapMgr::Update(uint32 diff)
251 {
252  for (uint8 i = 0; i < 4; ++i)
253  i_timer[i].Update(diff);
254 
255  // pussywizard: lfg compatibles update, schedule before maps so it is processed from the very beginning
256  //if (mapUpdateStep == 0)
257  {
258  if (m_updater.activated())
259  m_updater.schedule_lfg_update(diff);
260  else
261  {
262  uint32 startTime = getMSTime();
263  sLFGMgr->Update(diff, 1);
264  uint32 totalTime = getMSTimeDiff(startTime, getMSTime());
265  lfgDiffTracker.Update(10000 + totalTime); // +10k to mark it was NOT multithreaded
266  }
267  }
268 
269  MapMapType::iterator iter = i_maps.begin();
270  for (; iter != i_maps.end(); ++iter)
271  {
272  bool full = mapUpdateStep < 3 && ((mapUpdateStep == 0 && !iter->second->IsBattlegroundOrArena() && !iter->second->IsDungeon()) || (mapUpdateStep == 1 && iter->second->IsBattlegroundOrArena()) || (mapUpdateStep == 2 && iter->second->IsDungeon()));
273  if (m_updater.activated())
274  m_updater.schedule_update(*iter->second, uint32(full ? i_timer[mapUpdateStep].GetCurrent() : 0), diff);
275  else
276  iter->second->Update(uint32(full ? i_timer[mapUpdateStep].GetCurrent() : 0), diff);
277  }
278 
279  if (m_updater.activated())
280  m_updater.wait();
281 
282  if (mapUpdateStep < 3)
283  {
284  for (iter = i_maps.begin(); iter != i_maps.end(); ++iter)
285  {
286  bool full = ((mapUpdateStep == 0 && !iter->second->IsBattlegroundOrArena() && !iter->second->IsDungeon()) || (mapUpdateStep == 1 && iter->second->IsBattlegroundOrArena()) || (mapUpdateStep == 2 && iter->second->IsDungeon()));
287  if (full)
288  iter->second->DelayedUpdate(uint32(i_timer[mapUpdateStep].GetCurrent()));
289  }
290 
291  i_timer[mapUpdateStep].SetCurrent(0);
292  ++mapUpdateStep;
293  }
294 
295  if (mapUpdateStep == 3 && i_timer[3].Passed())
296  {
297  mapUpdateStep = 0;
298  i_timer[3].SetCurrent(0);
299  }
300 }

362 void MapMgr::GetNumPlayersInInstances(uint32& dungeons, uint32& battlegrounds, uint32& arenas, uint32& spectators)
363 {
364  for (MapMapType::iterator itr = i_maps.begin(); itr != i_maps.end(); ++itr)

367  if (!map->Instanceable())

371  {
372  if (mitr->second->IsDungeon()) dungeons += ((InstanceMap*)mitr->second)->GetPlayers().getSize();
373  else if (mitr->second->IsBattleground()) battlegrounds += ((InstanceMap*)mitr->second)->GetPlayers().getSize();
374  else if (mitr->second->IsBattleArena())
375  {
376  uint32 spect = 0;
377  if (BattlegroundMap* bgmap = mitr->second->ToBattlegroundMap())
378  if (Battleground* bg = bgmap->GetBG())
379  spect = bg->GetSpectators().size();
380 
381  arenas += ((InstanceMap*)mitr->second)->GetPlayers().getSize() - spect;
382  spectators += spect;
383  }
