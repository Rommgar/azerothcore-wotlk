183  void SummonHostage(uint8 num)
184  {
185  if (!QuestMinute)
186  return;
187 
188  Map::PlayerList const& PlayerList = instance->GetPlayers();
189  if (PlayerList.isEmpty())
190  return;
191 
192  Map::PlayerList::const_iterator i = PlayerList.begin();
193  if (Player* i_pl = i->GetSource())
194  {
195  if (Unit* Hostage = i_pl->SummonCreature(HostageInfo[num].npc, HostageInfo[num].x, HostageInfo[num].y, HostageInfo[num].z, HostageInfo[num].o, TEMPSUMMON_DEAD_DESPAWN, 0))
196  {
197  Hostage->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
198  Hostage->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
199  }
200  }

280  SaveToDB();
281  }
282  break;
283  case DATA_JANALAIEVENT:
284  m_auiEncounter[DATA_JANALAIEVENT] = data;
285  if (data == DONE)
286  SummonHostage(2);
287  SaveToDB();
288  break;
289  case DATA_HALAZZIEVENT:
290  m_auiEncounter[DATA_HALAZZIEVENT] = data;
291  HandleGameObject(HalazziDoorGUID, data != IN_PROGRESS);
292  if (data == DONE) SummonHostage(3);
293  SaveToDB();
294  break;
295  case DATA_HEXLORDEVENT:
296  m_auiEncounter[DATA_HEXLORDEVENT] = data;
297  if (data == IN_PROGRESS)
298  HandleGameObject(HexLordGateGUID, false);
299  else if (data == NOT_STARTED)
300  CheckInstanceStatus();
301  SaveToDB();
302  break;
303  case DATA_ZULJINEVENT:
304  m_auiEncounter[DATA_ZULJINEVENT] = data;
305  HandleGameObject(ZulJinDoorGUID, data != IN_PROGRESS);
306  SaveToDB();
307  break;
308  case DATA_CHESTLOOTED:
309  ++ChestLooted;
310  SaveToDB();
311  break;
312  case TYPE_RAND_VENDOR_1:
313  RandVendor[0] = data;
314  break;
315  case TYPE_RAND_VENDOR_2:
316  RandVendor[1] = data;
317  break;
318  }
319 
320  if (data == DONE)
321  {
322  ++BossKilled;
323  if (QuestMinute && BossKilled >= DATA_HALAZZIEVENT)
324  {
325  QuestMinute = 0;
326  DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 0);
327  }
328  CheckInstanceStatus();
329  SaveToDB();
330  }
331  }

362  void Update(uint32 diff) override
363  {
364  if (QuestMinute)
365  {
366  if (QuestTimer <= diff)
367  {
368  QuestMinute--;
369  SaveToDB();
370  QuestTimer += 60000;
371  if (QuestMinute)
372  {
373  DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 1);
374  DoUpdateWorldState(WORLDSTATE_TIME_TO_SACRIFICE, QuestMinute);
375  }
376  else DoUpdateWorldState(WORLDSTATE_SHOW_TIMER, 0);
377  }
378  QuestTimer -= diff;
379  }
