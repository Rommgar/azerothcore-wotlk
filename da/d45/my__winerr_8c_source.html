80 /* size of the table */
81 #define ERRTABLESIZE (sizeof(errtable)/sizeof(errtable[0]))

83 /* The following two constants must be the minimum and maximum
84 values in the (contiguous) range of Exec Failure errors. */
85 #define MIN_EXEC_ERROR ERROR_INVALID_STARTING_CODESEG
86 #define MAX_EXEC_ERROR ERROR_INFLOOP_IN_RELOC_CHAIN

88 /* These are the low and high value in the range of errors that are
89 access violations */

91 #define MAX_EACCES_RANGE ERROR_SHARING_BUFFER_EXCEEDED

94 static int get_errno_from_oserr(unsigned long oserrno)
95 {
96  int i;
97 
98  /* check the table for the OS error code */
99  for (i= 0; i < ERRTABLESIZE; ++i)
100  {
101  if (oserrno == errtable[i].oscode)
102  {
103  return errtable[i].sysv_errno;
104  }
105  }
106 
107  /* The error code wasn't in the table. We check for a range of */
108  /* EACCES errors or exec failure errors (ENOEXEC). Otherwise */
109  /* EINVAL is returned. */
110 
111  if (oserrno >= MIN_EACCES_RANGE && oserrno <= MAX_EACCES_RANGE)
112  return EACCES;
113  else if (oserrno >= MIN_EXEC_ERROR && oserrno <= MAX_EXEC_ERROR)
114  return ENOEXEC;
115  else
116  return EINVAL;
117 }

119 /* Set errno corresponsing to GetLastError() value */
120 void my_osmaperr ( unsigned long oserrno)
121 {
122  errno= get_errno_from_oserr(oserrno);
123 }
