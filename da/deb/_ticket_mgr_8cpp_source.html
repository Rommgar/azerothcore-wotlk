114 {
115  data << uint32(GMTICKET_STATUS_HASTEXT);

117  data << _message;
118  data << uint8(_needMoreHelp);
119  data << GetAge(_lastModifiedTime);

121  data << GetAge(ticket->GetLastModifiedTime());

126  data << GetAge(sTicketMgr->GetLastChange());

132 void GmTicket::SendResponse(WorldSession* session) const
133 {
134  WorldPacket data(SMSG_GMRESPONSE_RECEIVED);
135  data << uint32(1); // responseID
136  data << uint32(_id); // ticketID
137  data << _message.c_str();
138 
139  size_t len = _response.size();
140  char const* s = _response.c_str();
141 
142  for (int i = 0; i < 4; i++)
143  {
144  if (len)
145  {
146  size_t writeLen = std::min<size_t>(len, 3999);
147  data.append(s, writeLen);
148 
149  len -= writeLen;
150  s += writeLen;
151  }
152 
153  data << uint8(0);
154  }
155 
156  session->SendPacket(&data);
157 }

291 void TicketMgr::LoadTickets()
292 {
293  uint32 oldMSTime = getMSTime();
294 
295  for (GmTicketList::const_iterator itr = _ticketList.begin(); itr != _ticketList.end(); ++itr)
296  delete itr->second;
297  _ticketList.clear();
298 
299  _lastTicketId = 0;
300  _openTicketCount = 0;
301 
302  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_GM_TICKETS);
303  PreparedQueryResult result = CharacterDatabase.Query(stmt);
304  if (!result)
305  {
306  LOG_INFO("server.loading", ">> Loaded 0 GM tickets. DB table `gm_ticket` is empty!");
307 
308  return;
309  }
310 
311  uint32 count = 0;
312  do
313  {
314  Field* fields = result->Fetch();
315  GmTicket* ticket = new GmTicket();
316  if (!ticket->LoadFromDB(fields))
317  {
318  delete ticket;
319  continue;
320  }
321  if (!ticket->IsClosed())
322  ++_openTicketCount;
323 
324  // Update max ticket id if necessary
325  uint32 id = ticket->GetId();
326  if (_lastTicketId < id)
327  _lastTicketId = id;
328 
329  _ticketList[id] = ticket;
330  ++count;
331  } while (result->NextRow());
332 
333  LOG_INFO("server.loading", ">> Loaded %u GM tickets in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
334  LOG_INFO("server.loading", " ");
335 }
