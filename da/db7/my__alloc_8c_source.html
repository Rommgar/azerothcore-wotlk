25 /*
26  Initialize memory root
27 
28  SYNOPSIS
29  init_alloc_root()
30  mem_root - memory root to initialize
31  block_size - size of chunks (blocks) used for memory allocation
32  (It is external size of chunk i.e. it should include
33  memory required for internal structures, thus it
34  should be no less than ALLOC_ROOT_MIN_BLOCK_SIZE)
35  pre_alloc_size - if non-0, then size of block that should be
36  pre-allocated during memory root initialization.
37 
38  DESCRIPTION
39  This function prepares memory root for further use, sets initial size of
40  chunk for memory allocation and pre-allocates first block if specified.
41  Altough error can happen during execution of this function if
42  pre_alloc_size is non-0 it won't be reported. Instead it will be
43  reported as error in first alloc_root() on this memory root.
44 */
45 
46 void init_alloc_root(MEM_ROOT *mem_root, size_t block_size,
47  size_t pre_alloc_size __attribute__((unused)))
48 {
49  DBUG_ENTER("init_alloc_root");
50  DBUG_PRINT("enter",("root: 0x%lx", (long) mem_root));
51 
52  mem_root->free= mem_root->used= mem_root->pre_alloc= 0;
53  mem_root->min_malloc= 32;
54  mem_root->block_size= block_size - ALLOC_ROOT_MIN_BLOCK_SIZE;
55  mem_root->error_handler= 0;
56  mem_root->block_num= 4; /* We shift this with >>2 */
57  mem_root->first_block_usage= 0;
58 
59 #if !(defined(HAVE_purify) && defined(EXTRA_DEBUG))
60  if (pre_alloc_size)
61  {
62  if ((mem_root->free= mem_root->pre_alloc=
63  (USED_MEM*) my_malloc(pre_alloc_size+ ALIGN_SIZE(sizeof(USED_MEM)),
64  MYF(0))))
65  {
66  mem_root->free->size= pre_alloc_size+ALIGN_SIZE(sizeof(USED_MEM));
67  mem_root->free->left= pre_alloc_size;
68  mem_root->free->next= 0;
69  }
70  }
71 #endif
72  DBUG_VOID_RETURN;
73 }

76 /*
77  SYNOPSIS
78  reset_root_defaults()
79  mem_root memory root to change defaults of
80  block_size new value of block size. Must be greater or equal
81  than ALLOC_ROOT_MIN_BLOCK_SIZE (this value is about
82  68 bytes and depends on platform and compilation flags)
83  pre_alloc_size new size of preallocated block. If not zero,
84  must be equal to or greater than block size,
85  otherwise means 'no prealloc'.
86  DESCRIPTION
87  Function aligns and assigns new value to block size; then it tries to
88  reuse one of existing blocks as prealloc block, or malloc new one of
89  requested size. If no blocks can be reused, all unused blocks are freed
90  before allocation.
91 */
92 
93 void reset_root_defaults(MEM_ROOT *mem_root, size_t block_size,
94  size_t pre_alloc_size __attribute__((unused)))
95 {
96  DBUG_ASSERT(alloc_root_inited(mem_root));
97 
98  mem_root->block_size= block_size - ALLOC_ROOT_MIN_BLOCK_SIZE;
99 #if !(defined(HAVE_purify) && defined(EXTRA_DEBUG))
100  if (pre_alloc_size)
101  {
102  size_t size= pre_alloc_size + ALIGN_SIZE(sizeof(USED_MEM));
103  if (!mem_root->pre_alloc || mem_root->pre_alloc->size != size)
104  {
105  USED_MEM *mem, **prev= &mem_root->free;
106  /*
107  Free unused blocks, so that consequent calls
108  to reset_root_defaults won't eat away memory.
109  */
110  while (*prev)
111  {
112  mem= *prev;
113  if (mem->size == size)
114  {
115  /* We found a suitable block, no need to do anything else */
116  mem_root->pre_alloc= mem;
117  return;
118  }
119  if (mem->left + ALIGN_SIZE(sizeof(USED_MEM)) == mem->size)
120  {
121  /* remove block from the list and free it */
122  *prev= mem->next;
123  my_free(mem);
124  }
125  else
126  prev= &mem->next;
127  }
128  /* Allocate new prealloc block and add it to the end of free list */
129  if ((mem= (USED_MEM *) my_malloc(size, MYF(0))))
130  {
131  mem->size= size;
132  mem->left= pre_alloc_size;
133  mem->next= *prev;
134  *prev= mem_root->pre_alloc= mem;
135  }
136  else
137  {
138  mem_root->pre_alloc= 0;
139  }
140  }
141  }
142  else
143 #endif
144  mem_root->pre_alloc= 0;
145 }

148 void *alloc_root(MEM_ROOT *mem_root, size_t length)
149 {
150 #if defined(HAVE_purify) && defined(EXTRA_DEBUG)
151  reg1 USED_MEM *next;
152  DBUG_ENTER("alloc_root");
153  DBUG_PRINT("enter",("root: 0x%lx", (long) mem_root));
154 
155  DBUG_ASSERT(alloc_root_inited(mem_root));
156 
157  DBUG_EXECUTE_IF("simulate_out_of_memory",
158  {
159  if (mem_root->error_handler)
160  (*mem_root->error_handler)();
161  DBUG_SET("-d,simulate_out_of_memory");
162  DBUG_RETURN((void*) 0); /* purecov: inspected */
163  });
164 
165  length+=ALIGN_SIZE(sizeof(USED_MEM));
166  if (!(next = (USED_MEM*) my_malloc(length,MYF(MY_WME | ME_FATALERROR))))
167  {
168  if (mem_root->error_handler)
169  (*mem_root->error_handler)();
170  DBUG_RETURN((uchar*) 0); /* purecov: inspected */
171  }
172  next->next= mem_root->used;
173  next->size= length;
174  mem_root->used= next;
175  DBUG_PRINT("exit",("ptr: 0x%lx", (long) (((char*) next)+
176  ALIGN_SIZE(sizeof(USED_MEM)))));
177  DBUG_RETURN((uchar*) (((char*) next)+ALIGN_SIZE(sizeof(USED_MEM))));
178 #else
179  size_t get_size, block_size;
180  uchar* point;
181  reg1 USED_MEM *next= 0;
182  reg2 USED_MEM **prev;
183  DBUG_ENTER("alloc_root");
184  DBUG_PRINT("enter",("root: 0x%lx", (long) mem_root));
185  DBUG_ASSERT(alloc_root_inited(mem_root));
186 
187  DBUG_EXECUTE_IF("simulate_out_of_memory",
188  {
189  /* Avoid reusing an already allocated block */
190  if (mem_root->error_handler)
191  (*mem_root->error_handler)();
192  DBUG_SET("-d,simulate_out_of_memory");
193  DBUG_RETURN((void*) 0); /* purecov: inspected */
194  });
195  length= ALIGN_SIZE(length);
196  if ((*(prev= &mem_root->free)) != NULL)
197  {
198  if ((*prev)->left < length &&
199  mem_root->first_block_usage++ >= ALLOC_MAX_BLOCK_USAGE_BEFORE_DROP &&
200  (*prev)->left < ALLOC_MAX_BLOCK_TO_DROP)
201  {
202  next= *prev;
203  *prev= next->next; /* Remove block from list */
204  next->next= mem_root->used;
205  mem_root->used= next;
206  mem_root->first_block_usage= 0;
207  }
208  for (next= *prev ; next && next->left < length ; next= next->next)
209  prev= &next->next;
210  }
211  if (! next)
212  { /* Time to alloc new block */
213  block_size= mem_root->block_size * (mem_root->block_num >> 2);
214  get_size= length+ALIGN_SIZE(sizeof(USED_MEM));
215  get_size= max(get_size, block_size);
216 
217  if (!(next = (USED_MEM*) my_malloc(get_size,MYF(MY_WME | ME_FATALERROR))))
218  {
219  if (mem_root->error_handler)
220  (*mem_root->error_handler)();
221  DBUG_RETURN((void*) 0); /* purecov: inspected */
222  }
223  mem_root->block_num++;
224  next->next= *prev;
225  next->size= get_size;
226  next->left= get_size-ALIGN_SIZE(sizeof(USED_MEM));
227  *prev=next;
228  }
229 
230  point= (uchar*) ((char*) next+ (next->size-next->left));
231  /*TODO: next part may be unneded due to mem_root->first_block_usage counter*/
232  if ((next->left-= length) < mem_root->min_malloc)
233  { /* Full block */
234  *prev= next->next; /* Remove block from list */
235  next->next= mem_root->used;
236  mem_root->used= next;
237  mem_root->first_block_usage= 0;
238  }
239  DBUG_PRINT("exit",("ptr: 0x%lx", (ulong) point));
240  DBUG_RETURN((void*) point);
241 #endif
242 }

299 static inline void mark_blocks_free(MEM_ROOT* root)
300 {
301  reg1 USED_MEM *next;
302  reg2 USED_MEM **last;
303 
304  /* iterate through (partially) free blocks, mark them free */
305  last= &root->free;
306  for (next= root->free; next; next= *(last= &next->next))
307  {
308  next->left= next->size - ALIGN_SIZE(sizeof(USED_MEM));
309  TRASH_MEM(next);
310  }
311 
312  /* Combine the free and the used list */
313  *last= next=root->used;
314 
315  /* now go through the used blocks and mark them free */
316  for (; next; next= next->next)
317  {
318  next->left= next->size - ALIGN_SIZE(sizeof(USED_MEM));
319  TRASH_MEM(next);
320  }
321 
322  /* Now everything is set; Indicate that nothing is used anymore */
323  root->used= 0;
324  root->first_block_usage= 0;
325 }

329  Deallocate everything used by alloc_root or just move
330  used blocks to free list if called with MY_USED_TO_FREE
331 
332  SYNOPSIS
333  free_root()
334  root Memory root
335  MyFlags Flags for what should be freed:
336

339  preallocated block

347 void free_root(MEM_ROOT *root, myf MyFlags)
348 {
349  reg1 USED_MEM *next,*old;
350  DBUG_ENTER("free_root");
351  DBUG_PRINT("enter",("root: 0x%lx flags: %u", (long) root, (uint) MyFlags));
352 
353  if (MyFlags & MY_MARK_BLOCKS_FREE)
354  {
355  mark_blocks_free(root);
356  DBUG_VOID_RETURN;
357  }
358  if (!(MyFlags & MY_KEEP_PREALLOC))
359  root->pre_alloc=0;
360 
361  for (next=root->used; next ;)
362  {
363  old=next; next= next->next ;
364  if (old != root->pre_alloc)
365  my_free(old);
366  }
367  for (next=root->free ; next ;)
368  {
369  old=next; next= next->next;
370  if (old != root->pre_alloc)
371  my_free(old);
372  }
373  root->used=root->free=0;
374  if (root->pre_alloc)
375  {
376  root->free=root->pre_alloc;
377  root->free->left=root->pre_alloc->size-ALIGN_SIZE(sizeof(USED_MEM));
378  TRASH_MEM(root->pre_alloc);
379  root->free->next=0;
380  }
381  root->block_num= 4;
382  root->first_block_usage= 0;
383  DBUG_VOID_RETURN;
384 }
