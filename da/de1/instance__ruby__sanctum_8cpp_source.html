219  void Load(char const* str) override
220  {
221  if (!str)
222  {
223  OUT_LOAD_INST_DATA_FAIL;
224  return;
225  }
226 
227  OUT_LOAD_INST_DATA(str);
228 
229  char dataHead1, dataHead2;
230 
231  std::istringstream loadStream(str);
232  loadStream >> dataHead1 >> dataHead2;
233 
234  if (dataHead1 == 'R' && dataHead2 == 'S')
235  {
236  for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
237  {
238  uint32 tmpState;
239  loadStream >> tmpState;
240  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
241  tmpState = NOT_STARTED;
242 
243  SetBossState(i, EncounterState(tmpState));
244  }
245 
246  // Xinef: additional check
247  if (GetBossState(DATA_HALION_INTRO_DONE) != DONE)
248  {
249  SetBossState(DATA_HALION_INTRO1, NOT_STARTED);
250  SetBossState(DATA_HALION_INTRO2, NOT_STARTED);
251  }
252  }
253  else
254  OUT_LOAD_INST_DATA_FAIL;
255 
256  OUT_LOAD_INST_DATA_COMPLETE;
257  }
258
