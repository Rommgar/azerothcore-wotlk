144 static char *backtick_string(CHARSET_INFO *cs, char *to, char *end,
145  char *par, size_t par_len, char quote_char)
146 {
147  uint char_len;
148  char *start= to;
149  char *par_end= par + par_len;
150  size_t buff_length= (size_t) (end - to);
151 
152  if (buff_length <= par_len)
153  goto err;
154  *start++= quote_char;
155 
156  for ( ; par < par_end; par+= char_len)
157  {
158  uchar c= *(uchar *) par;
159  if (!(char_len= my_mbcharlen(cs, c)))
160  char_len= 1;
161  if (char_len == 1 && c == (uchar) quote_char )
162  {
163  if (start + 1 >= end)
164  goto err;
165  *start++= quote_char;
166  }
167  if (start + char_len >= end)
168  goto err;
169  start= strnmov(start, par, char_len);
170  }
171 
172  if (start + 1 >= end)
173  goto err;
174  *start++= quote_char;
175  return start;
176 
177 err:
178  *to='\0';
179  return to;
180 }

187 static char *process_str_arg(CHARSET_INFO *cs, char *to, char *end,
188  size_t width, char *par, uint print_type)
189 {
190  int well_formed_error;
191  size_t plen, left_len= (size_t) (end - to) + 1;
192  if (!par)
193  par = (char*) "(null)";
194 
195  plen= strnlen(par, width);
196  if (left_len <= plen)
197  plen = left_len - 1;
198  plen= cs->cset->well_formed_len(cs, par, par + plen,
199  width, &well_formed_error);

201  to= backtick_string(cs, to, end, par, plen, '`');
202  else
203  to= strnmov(to,par,plen);
204  return to;

248 static char *process_int_arg(char *to, char *end, size_t length,
249  longlong par, char arg_type, uint print_type)
250 {
251  size_t res_length, to_length;
252  char *store_start= to, *store_end;
253  char buff[32];
254 
255  if ((to_length= (size_t) (end-to)) < 16 || length)
256  store_start= buff;
257 
258  if (arg_type == 'd' || arg_type == 'i')
259  store_end= longlong10_to_str(par, store_start, -10);
260  else if (arg_type == 'u')
261  store_end= longlong10_to_str(par, store_start, 10);
262  else if (arg_type == 'p')
263  {
264  store_start[0]= '0';
265  store_start[1]= 'x';
266  store_end= ll2str(par, store_start + 2, 16, 0);
267  }
268  else if (arg_type == 'o')
269  {
270  store_end= ll2str(par, store_start, 8, 0);
271  }
272  else
273  {
274  DBUG_ASSERT(arg_type == 'X' || arg_type =='x');
275  store_end= ll2str(par, store_start, 16, (arg_type == 'X'));
276  }
277 
278  if ((res_length= (size_t) (store_end - store_start)) > to_length)
279  return to; /* num doesn't fit in output */
280  /* If %#d syntax was used, we have to pre-zero/pre-space the string */
281  if (store_start == buff)
282  {
283  length= min(length, to_length);
284  if (res_length < length)
285  {
286  size_t diff= (length- res_length);
287  bfill(to, diff, (print_type & PREZERO_ARG) ? '0' : ' ');
288  if (arg_type == 'p' && print_type & PREZERO_ARG)
289  {
290  if (diff > 1)
291  to[1]= 'x';
292  else
293  store_start[0]= 'x';
294  store_start[1]= '0';
295  }
296  to+= diff;
297  }
298  bmove(to, store_start, res_length);
299  }
300  to+= res_length;
301  return to;
302 }

319 static char *process_args(CHARSET_INFO *cs, char *to, char *end,
320  const char* fmt, size_t arg_index, va_list ap)
321 {
322  ARGS_INFO args_arr[MAX_ARGS];
323  PRINT_INFO print_arr[MAX_PRINT_INFO];
324  uint idx= 0, arg_count= arg_index;
325 
326 start:
327  /* Here we are at the beginning of positional argument, right after $ */
328  arg_index--;
329  print_arr[idx].flags= 0;
330  if (*fmt == '`')
331  {
332  print_arr[idx].flags|= ESCAPED_ARG;
333  fmt++;
334  }
335  if (*fmt == '-')
336  fmt++;
337  print_arr[idx].length= print_arr[idx].width= 0;
338  /* Get print length */
339  if (*fmt == '*')
340  {
341  fmt++;
342  fmt= get_length(fmt, &print_arr[idx].length, &print_arr[idx].flags);
343  print_arr[idx].length--;
344  DBUG_ASSERT(*fmt == '$' && print_arr[idx].length < MAX_ARGS);
345  args_arr[print_arr[idx].length].arg_type= 'd';
346  print_arr[idx].flags|= LENGTH_ARG;
347  arg_count= max(arg_count, print_arr[idx].length + 1);
348  fmt++;
349  }
350  else
351  fmt= get_length(fmt, &print_arr[idx].length, &print_arr[idx].flags);
352 
353  if (*fmt == '.')
354  {
355  fmt++;
356  /* Get print width */
357  if (*fmt == '*')
358  {
359  fmt++;
360  fmt= get_width(fmt, &print_arr[idx].width);
361  print_arr[idx].width--;
362  DBUG_ASSERT(*fmt == '$' && print_arr[idx].width < MAX_ARGS);
363  args_arr[print_arr[idx].width].arg_type= 'd';
364  print_arr[idx].flags|= WIDTH_ARG;
365  arg_count= max(arg_count, print_arr[idx].width + 1);
366  fmt++;
367  }
368  else
369  fmt= get_width(fmt, &print_arr[idx].width);
370  }
371  else
372  print_arr[idx].width= SIZE_T_MAX;
373 
374  fmt= check_longlong(fmt, &args_arr[arg_index].have_longlong);
375  if (*fmt == 'p')
376  args_arr[arg_index].have_longlong= (sizeof(void *) == sizeof(longlong));
377  args_arr[arg_index].arg_type= print_arr[idx].arg_type= *fmt;
378 
379  print_arr[idx].arg_idx= arg_index;
380  print_arr[idx].begin= ++fmt;
381 
382  while (*fmt && *fmt != '%')
383  fmt++;
384 
385  if (!*fmt) /* End of format string */
386  {
387  uint i;
388  print_arr[idx].end= fmt;
389  /* Obtain parameters from the list */
390  for (i= 0 ; i < arg_count; i++)
391  {
392  switch (args_arr[i].arg_type) {
393  case 's':
394  case 'b':
395  args_arr[i].str_arg= va_arg(ap, char *);
396  break;
397  case 'f':
398  case 'g':
399  args_arr[i].double_arg= va_arg(ap, double);
400  break;
401  case 'd':
402  case 'i':
403  case 'u':
404  case 'x':
405  case 'X':
406  case 'o':
407  case 'p':
408  if (args_arr[i].have_longlong)
409  args_arr[i].longlong_arg= va_arg(ap,longlong);
410  else if (args_arr[i].arg_type == 'd' || args_arr[i].arg_type == 'i')
411  args_arr[i].longlong_arg= va_arg(ap, int);
412  else
413  args_arr[i].longlong_arg= va_arg(ap, uint);
414  break;
415  case 'c':
416  args_arr[i].longlong_arg= va_arg(ap, int);
417  break;
418  default:
419  DBUG_ASSERT(0);
420  }
421  }
422  /* Print result string */
423  for (i= 0; i <= idx; i++)
424  {
425  size_t width= 0, length= 0;
426  switch (print_arr[i].arg_type) {
427  case 's':
428  {
429  char *par= args_arr[print_arr[i].arg_idx].str_arg;
430  width= (print_arr[i].flags & WIDTH_ARG)
431  ? (size_t)args_arr[print_arr[i].width].longlong_arg
432  : print_arr[i].width;
433  to= process_str_arg(cs, to, end, width, par, print_arr[i].flags);
434  break;
435  }
436  case 'b':
437  {
438  char *par = args_arr[print_arr[i].arg_idx].str_arg;
439  width= (print_arr[i].flags & WIDTH_ARG)
440  ? (size_t)args_arr[print_arr[i].width].longlong_arg
441  : print_arr[i].width;
442  to= process_bin_arg(to, end, width, par);
443  break;
444  }
445  case 'c':
446  {
447  if (to == end)
448  break;
449  *to++= (char) args_arr[print_arr[i].arg_idx].longlong_arg;
450  break;
451  }
452  case 'f':
453  case 'g':
454  {
455  double d= args_arr[print_arr[i].arg_idx].double_arg;
456  width= (print_arr[i].flags & WIDTH_ARG) ?
457  (uint)args_arr[print_arr[i].width].longlong_arg : print_arr[i].width;
458  to= process_dbl_arg(to, end, width, d, print_arr[i].arg_type);
459  break;
460  }
461  case 'd':
462  case 'i':
463  case 'u':
464  case 'x':
465  case 'X':
466  case 'o':
467  case 'p':
468  {
469  /* Integer parameter */
470  longlong larg;
471  length= (print_arr[i].flags & LENGTH_ARG)
472  ? (size_t)args_arr[print_arr[i].length].longlong_arg
473  : print_arr[i].length;
474 
475  if (args_arr[print_arr[i].arg_idx].have_longlong)
476  larg = args_arr[print_arr[i].arg_idx].longlong_arg;
477  else if (print_arr[i].arg_type == 'd' || print_arr[i].arg_type == 'i' )
478  larg = (int) args_arr[print_arr[i].arg_idx].longlong_arg;
479  else
480  larg= (uint) args_arr[print_arr[i].arg_idx].longlong_arg;
481 
482  to= process_int_arg(to, end, length, larg, print_arr[i].arg_type,
483  print_arr[i].flags);
484  break;
485  }
486  default:
487  break;
488  }
489 
490  if (to == end)
491  break;
492 
493  length= min(end - to , print_arr[i].end - print_arr[i].begin);
494  if (to + length < end)
495  length++;
496  to= strnmov(to, print_arr[i].begin, length);
497  }
498  DBUG_ASSERT(to <= end);
499  *to='\0'; /* End of errmessage */
500  return to;
501  }
502  else
503  {
504  /* Process next positional argument*/
505  DBUG_ASSERT(*fmt == '%');
506  print_arr[idx].end= fmt - 1;
507  idx++;
508  fmt++;
509  arg_index= 0;
510  fmt= get_width(fmt, &arg_index);
511  DBUG_ASSERT(*fmt == '$');
512  fmt++;
513  arg_count= max(arg_count, arg_index);
514  goto start;
515  }
516 
517  return 0;
518 }

537 size_t my_vsnprintf_ex(CHARSET_INFO *cs, char *to, size_t n,
538  const char* fmt, va_list ap)
539 {
540  char *start=to, *end=to+n-1;
541  size_t length, width;
542  uint print_type, have_longlong;
543 
544  for (; *fmt ; fmt++)
545  {
546  if (*fmt != '%')
547  {
548  if (to == end) /* End of buffer */
549  break;
550  *to++= *fmt; /* Copy ordinary char */
551  continue;
552  }
553  fmt++; /* skip '%' */
554 
555  length= width= 0;
556  print_type= 0;
557 
558  /* Read max fill size (only used with %d and %u) */
559  if (my_isdigit(&my_charset_latin1, *fmt))
560  {
561  fmt= get_length(fmt, &length, &print_type);
562  if (*fmt == '$')
563  {
564  to= process_args(cs, to, end, (fmt+1), length, ap);
565  return (size_t) (to - start);
566  }
567  }
568  else
569  {
570  if (*fmt == '`')
571  {
572  print_type|= ESCAPED_ARG;
573  fmt++;
574  }
575  if (*fmt == '-')
576  fmt++;
577  if (*fmt == '*')
578  {
579  fmt++;
580  length= va_arg(ap, int);
581  }
582  else
583  fmt= get_length(fmt, &length, &print_type);
584  }
585 
586  if (*fmt == '.')
587  {
588  fmt++;
589  if (*fmt == '*')
590  {
591  fmt++;
592  width= va_arg(ap, int);
593  }
594  else
595  fmt= get_width(fmt, &width);
596  }
597  else
598  width= SIZE_T_MAX;
599 
600  fmt= check_longlong(fmt, &have_longlong);
601 
602  if (*fmt == 's') /* String parameter */
603  {
604  reg2 char *par= va_arg(ap, char *);
605  to= process_str_arg(cs, to, end, width, par, print_type);
606  continue;
607  }
608  else if (*fmt == 'b') /* Buffer parameter */
609  {
610  char *par = va_arg(ap, char *);
611  to= process_bin_arg(to, end, width, par);
612  continue;
613  }
614  else if (*fmt == 'f' || *fmt == 'g')
615  {
616  double d= va_arg(ap, double);
617  to= process_dbl_arg(to, end, width, d, *fmt);
618  continue;
619  }
620  else if (*fmt == 'd' || *fmt == 'i' || *fmt == 'u' || *fmt == 'x' ||
621  *fmt == 'X' || *fmt == 'p' || *fmt == 'o')
622  {
623  /* Integer parameter */
624  longlong larg;
625  if (*fmt == 'p')
626  have_longlong= (sizeof(void *) == sizeof(longlong));
627 
628  if (have_longlong)
629  larg = va_arg(ap,longlong);
630  else if (*fmt == 'd' || *fmt == 'i')
631  larg = va_arg(ap, int);
632  else
633  larg= va_arg(ap, uint);
634 
635  to= process_int_arg(to, end, length, larg, *fmt, print_type);
636  continue;
637  }
638  else if (*fmt == 'c') /* Character parameter */
639  {
640  register int larg;
641  if (to == end)
642  break;
643  larg = va_arg(ap, int);
644  *to++= (char) larg;
645  continue;
646  }
647 
648  /* We come here on '%%', unknown code or too long parameter */
649  if (to == end)
650  break;
651  *to++='%'; /* % used as % or unknown code */
652  }
653  DBUG_ASSERT(to <= end);
654  *to='\0'; /* End of errmessage */
655  return (size_t) (to - start);
656 }
