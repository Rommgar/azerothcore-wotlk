791 void System::sleep(RealTime t) {
792 
793  // Overhead of calling this function, measured from a previous run.
794  static const RealTime OVERHEAD = 0.00006f;
795 
796  RealTime now = time();
797  RealTime wakeupTime = now + t - OVERHEAD;
798 
799  RealTime remainingTime = wakeupTime - now;
800  RealTime sleepTime = 0;
801 
802  // On Windows, a "time slice" is measured in quanta of 3-5 ms (http://support.microsoft.com/kb/259025)
803  // Sleep(0) yields the remainder of the time slice, which could be a long time.
804  // A 1 ms minimum time experimentally kept the "Empty GApp" at nearly no CPU load at 100 fps,
805  // yet nailed the frame timing perfectly.
806  static RealTime minRealSleepTime = 3 * units::milliseconds();
807 
808  while (remainingTime > 0) {
809 
810  if (remainingTime > minRealSleepTime * 2.5) {
811  // Safe to use Sleep with a time... sleep for half the remaining time
812  sleepTime = max(remainingTime * 0.5, 0.0005);
813  } else if (remainingTime > minRealSleepTime) {
814  // Safe to use Sleep with a zero time;
815  // causes the program to yield only
816  // the current time slice, and then return.
817  sleepTime = 0;
818  } else {
819  // Not safe to use Sleep; busy wait
820  sleepTime = -1;
821  }
822 
823  if (sleepTime >= 0) {
824  #ifdef G3D_WINDOWS
825  // Translate to milliseconds
826  Sleep((int)(sleepTime * 1e3));
827  #else
828  // Translate to microseconds
829  usleep((int)(sleepTime * 1e6));
830  #endif
831  }
832 
833  now = time();
834  remainingTime = wakeupTime - now;
835  }
836 }

1484 void* System::alignedMalloc(size_t bytes, size_t alignment) {
1485 
1486  alwaysAssertM(isPow2((uint32)alignment), "alignment must be a power of 2");
1487 
1488  // We must align to at least a word boundary.
1489  alignment = max(alignment, sizeof(void *));
1490 
1491  // Pad the allocation size with the alignment size and the size of
1492  // the redirect pointer. This is the worst-case size we'll need.
1493  // Since the alignment size is at least teh word size, we don't
1494  // need to allocate space for the redirect pointer. We repeat the max here
1495  // for clarity.
1496  size_t totalBytes = bytes + max(alignment, sizeof(void*));
1497 
1498  size_t truePtr = (size_t)System::malloc(totalBytes);
1499 
1500  if (truePtr == 0) {
1501  // malloc returned NULL
1502  return NULL;
1503  }
1504 
1505  debugAssert(isValidHeapPointer((void*)truePtr));
1506  #ifdef G3D_WINDOWS
1507  // The blocks we return will not be valid Win32 debug heap
1508  // pointers because they are offset
1509  // debugAssert(_CrtIsValidPointer((void*)truePtr, totalBytes, TRUE) );
1510  #endif
1511 
1512 
1513  // We want alignedPtr % alignment == 0, which we'll compute with a
1514  // binary AND because 2^n - 1 has the form 1111... in binary.
1515  const size_t bitMask = (alignment - 1);
1516 
1517  // The return pointer will be the next aligned location that is at
1518  // least sizeof(void*) after the true pointer. We need the padding
1519  // to have a place to write the redirect pointer.
1520  size_t alignedPtr = truePtr + sizeof(void*);
1521 
1522  const size_t remainder = alignedPtr & bitMask;
