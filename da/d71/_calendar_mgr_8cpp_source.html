57 void CalendarMgr::LoadFromDB()
58 {
59  uint32 count = 0;
60  _maxEventId = 0;
61  _maxInviteId = 0;
62 
63  // 0 1 2 3 4 5 6 7 8
64  if (QueryResult result = CharacterDatabase.Query("SELECT id, creator, title, description, type, dungeon, eventtime, flags, time2 FROM calendar_events"))
65  do
66  {
67  Field* fields = result->Fetch();
68 
69  uint64 eventId = fields[0].GetUInt64();
70  ObjectGuid creatorGUID = ObjectGuid::Create<HighGuid::Player>(fields[1].GetUInt32());
71  std::string title = fields[2].GetString();
72  std::string description = fields[3].GetString();
73  CalendarEventType type = CalendarEventType(fields[4].GetUInt8());
74  int32 dungeonId = fields[5].GetInt32();
75  uint32 eventTime = fields[6].GetUInt32();
76  uint32 flags = fields[7].GetUInt32();
77  uint32 timezoneTime = fields[8].GetUInt32();
78  uint32 guildId = 0;
79 
80  if (flags & CALENDAR_FLAG_GUILD_EVENT || flags & CALENDAR_FLAG_WITHOUT_INVITES)
81  guildId = Player::GetGuildIdFromStorage(creatorGUID.GetCounter());
82 
83  CalendarEvent* calendarEvent = new CalendarEvent(eventId, creatorGUID, guildId, type, dungeonId, time_t(eventTime), flags, time_t(timezoneTime), title, description);
84  _events.insert(calendarEvent);
85 
86  _maxEventId = std::max(_maxEventId, eventId);
87 
88  ++count;
89  } while (result->NextRow());
90 
91  LOG_INFO("server.loading", ">> Loaded %u calendar events", count);
92  count = 0;
93 
94  // 0 1 2 3 4 5 6 7
95  if (QueryResult result = CharacterDatabase.Query("SELECT id, event, invitee, sender, status, statustime, `rank`, text FROM calendar_invites"))
96  do
97  {
98  Field* fields = result->Fetch();
99 
100  uint64 inviteId = fields[0].GetUInt64();
101  uint64 eventId = fields[1].GetUInt64();
102  ObjectGuid invitee = ObjectGuid::Create<HighGuid::Player>(fields[2].GetUInt32());
103  ObjectGuid senderGUID = ObjectGuid::Create<HighGuid::Player>(fields[3].GetUInt32());
104  CalendarInviteStatus status = CalendarInviteStatus(fields[4].GetUInt8());
105  uint32 statusTime = fields[5].GetUInt32();
106  CalendarModerationRank rank = CalendarModerationRank(fields[6].GetUInt8());
107  std::string text = fields[7].GetString();
108 
109  CalendarInvite* invite = new CalendarInvite(inviteId, eventId, invitee, senderGUID, time_t(statusTime), status, rank, text);
110  _invites[eventId].push_back(invite);
111 
112  _maxInviteId = std::max(_maxInviteId, inviteId);
113 
114  ++count;
115  } while (result->NextRow());
116 
117  LOG_INFO("server.loading", ">> Loaded %u calendar invites", count);
118  LOG_INFO("server.loading", " ");
119 
120  for (uint64 i = 1; i < _maxEventId; ++i)
121  if (!GetEvent(i))
122  _freeEventIds.push_back(i);
123 
124  for (uint64 i = 1; i < _maxInviteId; ++i)
125  if (!GetInvite(i))
126  _freeInviteIds.push_back(i);
127 }

164 void CalendarMgr::RemoveEvent(CalendarEvent* calendarEvent, ObjectGuid remover)
165 {
166  if (!calendarEvent)
167  {
168  SendCalendarCommandResult(remover, CALENDAR_ERROR_EVENT_INVALID);
169  return;
170  }
171 
172  SendCalendarEventRemovedAlert(*calendarEvent);
173 
174  CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
175  CharacterDatabasePreparedStatement* stmt;
176  MailDraft mail(calendarEvent->BuildCalendarMailSubject(remover), calendarEvent->BuildCalendarMailBody());
177 
178  CalendarInviteStore& eventInvites = _invites[calendarEvent->GetEventId()];
179  for (size_t i = 0; i < eventInvites.size(); ++i)
180  {
181  CalendarInvite* invite = eventInvites[i];
182  stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_INVITE);
183  stmt->setUInt64(0, invite->GetInviteId());
184  trans->Append(stmt);
185 
186  // guild events only? check invite status here?
187  // When an event is deleted, all invited (accepted/declined? - verify) guildies are notified via in-game mail. (wowwiki)
188  if (remover && invite->GetInviteeGUID() != remover)
189  mail.SendMailTo(trans, MailReceiver(invite->GetInviteeGUID().GetCounter()), calendarEvent, MAIL_CHECK_MASK_COPIED);
190 
191  delete invite;
192  }
193 
194  _invites.erase(calendarEvent->GetEventId());
195 
196  stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_CALENDAR_EVENT);
197  stmt->setUInt64(0, calendarEvent->GetEventId());
198  trans->Append(stmt);
199  CharacterDatabase.CommitTransaction(trans);
200 
201  delete calendarEvent;
202  _events.erase(calendarEvent);
203  return;
204 }

473 void CalendarMgr::SendCalendarEventInvite(CalendarInvite const& invite)
474 {
475  CalendarEvent* calendarEvent = GetEvent(invite.GetEventId());
476  time_t statusTime = invite.GetStatusTime();
477  bool hasStatusTime = statusTime != 946684800; // 01/01/2000 00:00:00
478 
479  ObjectGuid invitee = invite.GetInviteeGUID();
480  Player* player = ObjectAccessor::FindConnectedPlayer(invitee);
481 
482  uint8 level = player ? player->getLevel() : Player::GetLevelFromStorage(invitee.GetCounter());
483 
484  WorldPacket data(SMSG_CALENDAR_EVENT_INVITE, 8 + 8 + 8 + 1 + 1 + 1 + (statusTime ? 4 : 0) + 1);
485  data << invitee.WriteAsPacked();
486  data << uint64(invite.GetEventId());
487  data << uint64(invite.GetInviteId());
488  data << uint8(level);
489  data << uint8(invite.GetStatus());
490  data << uint8(hasStatusTime);
491  if (hasStatusTime)
492  data.AppendPackedTime(statusTime);
493  data << uint8(invite.GetSenderGUID() != invite.GetInviteeGUID()); // false only if the invite is sign-up
494 
495  if (!calendarEvent) // Pre-invite
496  {
497  if (Player* player = ObjectAccessor::FindConnectedPlayer(invite.GetSenderGUID()))
498  player->SendDirectMessage(&data);
499  }
500  else
501  {
502  if (calendarEvent->GetCreatorGUID() != invite.GetInviteeGUID()) // correct?
503  SendPacketToAllEventRelatives(data, *calendarEvent);
504  }
505 }

626

628  uint32 inviteeGuildId = invitee ? invitee->GetGuildId() : Player::GetGuildIdFromStorage(inviteeGuid.GetCounter());
629 
630  data << inviteeGuid.WriteAsPacked();
631  data << uint8(inviteeLevel);
632  data << uint8(calendarInvite->GetStatus());
633  data << uint8(calendarInvite->GetRank());
634  data << uint8(calendarEvent.IsGuildEvent() && calendarEvent.GetGuildId() == inviteeGuildId);
635  data << uint64(calendarInvite->GetInviteId());
636  data.AppendPackedTime(calendarInvite->GetStatusTime());
