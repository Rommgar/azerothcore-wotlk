26 bool DBCFileLoader::Load(char const* filename, char const* fmt)
27 {
28  uint32 header;
29  if (data)
30  {
31  delete [] data;
32  data = nullptr;
33  }
34 
35  FILE* f = fopen(filename, "rb");
36  if (!f)
37  {
38  return false;
39  }
40 
41  if (fread(&header, 4, 1, f) != 1) // Number of records
42  {
43  fclose(f);
44  return false;
45  }
46 
47  EndianConvert(header);
48 
49  if (header != 0x43424457) //'WDBC'
50  {
51  fclose(f);
52  return false;
53  }
54 
55  if (fread(&recordCount, 4, 1, f) != 1) // Number of records
56  {
57  fclose(f);
58  return false;
59  }
60 
61  EndianConvert(recordCount);
62 
63  if (fread(&fieldCount, 4, 1, f) != 1) // Number of fields
64  {
65  fclose(f);
66  return false;
67  }
68 
69  EndianConvert(fieldCount);
70 
71  if (fread(&recordSize, 4, 1, f) != 1) // Size of a record
72  {
73  fclose(f);
74  return false;
75  }
76 
77  EndianConvert(recordSize);
78 
79  if (fread(&stringSize, 4, 1, f) != 1) // String size
80  {
81  fclose(f);
82  return false;
83  }
84 
85  EndianConvert(stringSize);
86 
87  fieldsOffset = new uint32[fieldCount];
88  fieldsOffset[0] = 0;
89 
90  for (uint32 i = 1; i < fieldCount; ++i)
91  {
92  fieldsOffset[i] = fieldsOffset[i - 1];
93  if (fmt[i - 1] == 'b' || fmt[i - 1] == 'X') // byte fields
94  {
95  fieldsOffset[i] += sizeof(uint8);
96  }
97  else // 4 byte fields (int32/float/strings)
98  {
99  fieldsOffset[i] += sizeof(uint32);
100  }
101  }
102 
103  data = new unsigned char[recordSize * recordCount + stringSize];
104  stringTable = data + recordSize * recordCount;
105 
106  if (fread(data, recordSize * recordCount + stringSize, 1, f) != 1)
107  {
108  fclose(f);
109  return false;
110  }
111 
112  fclose(f);
113 
114  return true;
115 }

130 uint32 DBCFileLoader::GetFormatRecordSize(char const* format, int32* index_pos)
131 {
132  uint32 recordsize = 0;
133  int32 i = -1;
134 
135  for (uint32 x = 0; format[x]; ++x)
136  {
137  switch (format[x])
138  {
139  case FT_FLOAT:
140  recordsize += sizeof(float);
141  break;
142  case FT_INT:
143  recordsize += sizeof(uint32);
144  break;
145  case FT_STRING:
146  recordsize += sizeof(char*);
147  break;
148  case FT_SORT:
149  i = x;
150  break;
151  case FT_IND:
152  i = x;
153  recordsize += sizeof(uint32);
154  break;
155  case FT_BYTE:
156  recordsize += sizeof(uint8);
157  break;
158  case FT_NA:
159  case FT_NA_BYTE:
160  break;
161  case FT_LOGIC:
162  ASSERT(false && "Attempted to load DBC files that do not have field types that match what is in the core. Check DBCfmt.h or your DBC files.");
163  break;
164  default:
165  ASSERT(false && "Unknown field format character in DBCfmt.h");
166  break;
167  }
168  }
169 
170  if (index_pos)
171  {
172  *index_pos = i;
173  }
174 
175  return recordsize;
176 }

178 char* DBCFileLoader::AutoProduceData(char const* format, uint32& records, char**& indexTable)
179 {
180  /*
181  format STRING, NA, FLOAT, NA, INT <=>
182  struct{
183  char* field0,
184  float field1,
185  int field2
186  }entry;
187 
188  this func will generate entry[rows] data;
189  */
190 
191  typedef char* ptr;
192  if (strlen(format) != fieldCount)
193  {
194  return nullptr;
195  }
196 
197  //get struct size and index pos
198  int32 i;
199  uint32 recordsize = GetFormatRecordSize(format, &i);
200 
201  if (i >= 0)
202  {
203  uint32 maxi = 0;
204  //find max index
205  for (uint32 y = 0; y < recordCount; ++y)
206  {
207  uint32 ind = getRecord(y).getUInt(i);
208  if (ind > maxi)
209  {
210  maxi = ind;
211  }
212  }
213 
214  ++maxi;
215  records = maxi;
216  indexTable = new ptr[maxi];
217  memset(indexTable, 0, maxi * sizeof(ptr));
218  }
219  else
220  {
221  records = recordCount;
222  indexTable = new ptr[recordCount];
223  }
224 
225  char* dataTable = new char[recordCount * recordsize];
226 
227  uint32 offset = 0;
228 
229  for (uint32 y = 0; y < recordCount; ++y)
230  {
231  if (i >= 0)
232  {
233  indexTable[getRecord(y).getUInt(i)] = &dataTable[offset];
234  }
235  else
236  {
237  indexTable[y] = &dataTable[offset];
238  }
239 
240  for (uint32 x = 0; x < fieldCount; ++x)
241  {
242  switch (format[x])
243  {
244  case FT_FLOAT:
245  *((float*)(&dataTable[offset])) = getRecord(y).getFloat(x);
246  offset += sizeof(float);
247  break;
248  case FT_IND:
249  case FT_INT:
250  *((uint32*)(&dataTable[offset])) = getRecord(y).getUInt(x);
251  offset += sizeof(uint32);
252  break;
253  case FT_BYTE:
254  *((uint8*)(&dataTable[offset])) = getRecord(y).getUInt8(x);
255  offset += sizeof(uint8);
256  break;
257  case FT_STRING:
258  *((char**)(&dataTable[offset])) = nullptr; // will replace non-empty or "" strings in AutoProduceStrings
259  offset += sizeof(char*);
260  break;
261  case FT_LOGIC:
262  ASSERT(false && "Attempted to load DBC files that do not have field types that match what is in the core. Check DBCfmt.h or your DBC files.");
263  break;
264  case FT_NA:
265  case FT_NA_BYTE:
266  case FT_SORT:
267  break;
268  default:
269  ASSERT(false && "Unknown field format character in DBCfmt.h");
270  break;
271  }
272  }
273  }
274 
275  return dataTable;
276 }

278 char* DBCFileLoader::AutoProduceStrings(char const* format, char* dataTable)
279 {
280  if (strlen(format) != fieldCount)

288  uint32 offset = 0;
289 
290  for (uint32 y = 0; y < recordCount; ++y)
291  {
292  for (uint32 x = 0; x < fieldCount; ++x)
293  {
294  switch (format[x])
295  {
296  case FT_FLOAT:
297  offset += sizeof(float);
298  break;
299  case FT_IND:
300  case FT_INT:
301  offset += sizeof(uint32);
302  break;
303  case FT_BYTE:
304  offset += sizeof(uint8);
305  break;
306  case FT_STRING:
307  {
308  // fill only not filled entries
309  char** slot = (char**)(&dataTable[offset]);
310  if (!*slot || !** slot)
311  {
312  const char* st = getRecord(y).getString(x);
313  *slot = stringPool + (st - (char const*)stringTable);
314  }
315  offset += sizeof(char*);
316  break;
317  }
318  case FT_LOGIC:
319  ASSERT(false && "Attempted to load DBC files that does not have field types that match what is in the core. Check DBCfmt.h or your DBC files.");
320  break;
321  case FT_NA:
322  case FT_NA_BYTE:
323  case FT_SORT:
324  break;
325  default:
326  ASSERT(false && "Unknown field format character in DBCfmt.h");
327  break;
328  }
329  }
330  }
