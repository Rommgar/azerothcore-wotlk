128  void Reset() override
129  {
130  _feralDied = false;

133  events.Reset();
134  EntryCheckPredicate pred(NPC_FERAL_DEFENDER);
135  summons.DoAction(ACTION_DESPAWN_ADDS, pred);

142  me->SummonCreature(NPC_SANCTUM_SENTRY, me->GetPositionX() + urand(4, 12), me->GetPositionY() + urand(4, 12), me->GetPositionZ());

145  }

223  void DoAction(int32 param) override
224  {
225  if (param == ACTION_FERAL_DEATH_WITH_STACK)
226  events.ScheduleEvent(EVENT_RESPAWN_FERAL_DEFENDER, 25000);
227  else if (param == ACTION_FERAL_DEATH)

231  void UpdateAI(uint32 diff) override
232  {
233  if (!UpdateVictim())
234  return;
235 
236  events.Update(diff);
237  if (me->HasUnitState(UNIT_STATE_CASTING))
238  return;
239 
240  switch (events.ExecuteEvent())
241  {
242  case EVENT_SUMMON_FERAL_DEFENDER:
243  me->TextEmote("Auriaya begins to activate Feral Defender.", nullptr, true);
244  me->CastSpell(me, SPELL_ACTIVATE_FERAL_DEFENDER, true);
245  me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, true);
246  events.ScheduleEvent(EVENT_REMOVE_IMMUNE, 3000);
247  break;
248  case EVENT_REMOVE_IMMUNE:
249  me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, false);
250  break;
251  case EVENT_TERRIFYING_SCREECH:
252  me->TextEmote("Auriaya begins to cast Terrifying Screech.", nullptr, true);
253  me->CastSpell(me, SPELL_TERRIFYING_SCREECH, false);
254  events.RepeatEvent(35000);
255  break;
256  case EVENT_SONIC_SCREECH:
257  me->CastSpell(me, SPELL_SONIC_SCREECH, false);
258  events.RepeatEvent(50000);
259  break;
260  case EVENT_GUARDIAN_SWARM:
261  me->CastSpell(me->GetVictim(), SPELL_GUARDIAN_SWARM, false);
262  events.RepeatEvent(40000);
263  break;
264  case EVENT_SENTINEL_BLAST:
265  me->CastSpell(me, SPELL_SENTINEL_BLAST, false);
266  events.RepeatEvent(35000);
267  events.DelayEvents(5000, 0);
268  break;
269  case EVENT_RESPAWN_FERAL_DEFENDER:
270  {
271  EntryCheckPredicate pred(NPC_FERAL_DEFENDER);
272  summons.DoAction(ACTION_FERAL_RESPAWN, pred);
273  break;
274  }
275  case EVENT_ENRAGE:
276  me->TextEmote("You waste my time!", nullptr, true);
277  me->PlayDirectSound(SOUND_BERSERK);
278  me->CastSpell(me, SPELL_ENRAGE, true);
279  break;
280  }
281 
282  DoMeleeAttackIfReady();
283  }
284  };

384  cr->AI()->DoAction(_feralEssenceStack ? ACTION_FERAL_DEATH_WITH_STACK : ACTION_FERAL_DEATH);
385 
386  if (_feralEssenceStack)
387  {
388  if (Creature* cr = me->SummonCreature(NPC_SEEPING_FERAL_ESSENCE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 0.0f))
389  summons.Summon(cr);

391  --_feralEssenceStack;

407  }
408 
409  if (_feralEssenceStack)

411  aur->SetStackAmount(_feralEssenceStack);
412  }
413  else if (param == ACTION_DESPAWN_ADDS)
414  summons.DespawnAll();
415  }
