88  events.Update(diff);
89 
90  if (me->HasUnitState(UNIT_STATE_CASTING))
91  return;
92 
93  while (uint32 eventId = events.ExecuteEvent())
94  {
95  switch (eventId)
96  {
97  case EVENT_MORTAL_WOUND:
98  DoCastVictim(SPELL_MORTALWOUND);
99  events.ScheduleEvent(EVENT_MORTAL_WOUND, 8000);
100  break;
101  case EVENT_SANDTRAP:
102  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
103  target->CastSpell(target, SPELL_SANDTRAP, true);
104  else if (Unit* victim = me->GetVictim())
105  victim->CastSpell(victim, SPELL_SANDTRAP, true);
106  events.ScheduleEvent(EVENT_SANDTRAP, urand(5000, 15000));
107  break;
108  case EVENT_WIDE_SLASH:
109  DoCast(me, SPELL_WIDE_SLASH);
110  events.ScheduleEvent(EVENT_WIDE_SLASH, 11000);
111  break;
112  case EVENT_TRASH:
113  DoCast(me, SPELL_TRASH);
114  events.ScheduleEvent(EVENT_WIDE_SLASH, 16000);
115  break;
116  default:
117  break;
118  }
119  }
