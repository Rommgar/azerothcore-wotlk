108 void ChatHandler::SendSysMessage(std::string_view str, bool escapeCharacters)
109 {
110  std::string msg{ str };
111 
112  // Replace every "|" with "||" in msg
113  if (escapeCharacters && msg.find('|') != std::string::npos)
114  {
115  std::vector<std::string_view> tokens = Acore::Tokenize(msg, '|', true);
116  std::ostringstream stream;
117 
118  for (size_t i = 0; i < tokens.size() - 1; ++i)
119  stream << tokens[i] << "||";
120 
121  stream << tokens[tokens.size() - 1];
122 
123  msg = stream.str();
124  }
125 
126  WorldPacket data;
127  for (std::string_view line : Acore::Tokenize(str, '\n', true))
128  {
129  BuildChatPacket(data, CHAT_MSG_SYSTEM, LANG_UNIVERSAL, nullptr, nullptr, line);
130  m_session->SendPacket(&data);
131  }

174 bool ChatHandler::ParseCommands(std::string_view text)
175 {
176  ASSERT(!text.empty());
177 
178  // chat case (.command or !command format)
179  if ((text[0] != '!') && (text[0] != '.'))
180  return false;
181 
182  // ignore single . and ! in line
183  if (text.length() < 2)
184  return false;
185 
186  // ignore messages staring from many dots.
187  if (text[1] == text[0])
188  return false;
189 
190  // ignore messages with separator after .
191  if (text[1] == Acore::Impl::ChatCommands::COMMAND_DELIMITER)
192  return false;
193 
194  return _ParseCommands(text.substr(1));

197 size_t ChatHandler::BuildChatPacket(WorldPacket& data, ChatMsg chatType, Language language, ObjectGuid senderGUID, ObjectGuid receiverGUID, std::string_view message, uint8 chatTag,
198  std::string const& senderName /*= ""*/, std::string const& receiverName /*= ""*/,
199  uint32 achievementId /*= 0*/, bool gmMessage /*= false*/, std::string const& channelName /*= ""*/)
200 {
201  size_t receiverGUIDPos = 0;
202  data.Initialize(!gmMessage ? SMSG_MESSAGECHAT : SMSG_GM_MESSAGECHAT);
203  data << uint8(chatType);
204  data << int32(language);
205  data << senderGUID;
206  data << uint32(0); // some flags
207  switch (chatType)
208  {
209  case CHAT_MSG_MONSTER_SAY:
210  case CHAT_MSG_MONSTER_PARTY:
211  case CHAT_MSG_MONSTER_YELL:
212  case CHAT_MSG_MONSTER_WHISPER:
213  case CHAT_MSG_MONSTER_EMOTE:
214  case CHAT_MSG_RAID_BOSS_EMOTE:
215  case CHAT_MSG_RAID_BOSS_WHISPER:
216  case CHAT_MSG_BATTLENET:
217  data << uint32(senderName.length() + 1);
218  data << senderName;
219  receiverGUIDPos = data.wpos();
220  data << receiverGUID;
221  if (receiverGUID && !receiverGUID.IsPlayer() && !receiverGUID.IsPet())
222  {
223  data << uint32(receiverName.length() + 1);
224  data << receiverName;
225  }
226  break;
227  case CHAT_MSG_WHISPER_FOREIGN:
228  data << uint32(senderName.length() + 1);
229  data << senderName;
230  receiverGUIDPos = data.wpos();
231  data << receiverGUID;
232  break;
233  case CHAT_MSG_BG_SYSTEM_NEUTRAL:
234  case CHAT_MSG_BG_SYSTEM_ALLIANCE:
235  case CHAT_MSG_BG_SYSTEM_HORDE:
236  receiverGUIDPos = data.wpos();
237  data << receiverGUID;
238  if (receiverGUID && !receiverGUID.IsPlayer())
239  {
240  data << uint32(receiverName.length() + 1);
241  data << receiverName;
242  }
243  break;
244  case CHAT_MSG_ACHIEVEMENT:
245  case CHAT_MSG_GUILD_ACHIEVEMENT:
246  receiverGUIDPos = data.wpos();
247  data << receiverGUID;
248  break;
249  default:
250  if (gmMessage)
251  {
252  data << uint32(senderName.length() + 1);
253  data << senderName;
254  }
255 
256  if (chatType == CHAT_MSG_CHANNEL)
257  {
258  ASSERT(channelName.length() > 0);
259  data << channelName;
260  }
261 
262  receiverGUIDPos = data.wpos();
263  data << receiverGUID;
264  break;
265  }
266 
267  data << uint32(message.length() + 1);
268  data << message;
269  data << uint8(chatTag);
270 
271  if (chatType == CHAT_MSG_ACHIEVEMENT || chatType == CHAT_MSG_GUILD_ACHIEVEMENT)
272  data << uint32(achievementId);
273 
274  return receiverGUIDPos;
275 }

368 char* ChatHandler::extractKeyFromLink(char* text, char const* linkType, char** something1)
369 {
370  // skip empty
371  if (!text)
372  return nullptr;
373 
374  // skip spaces
375  while (*text == ' ' || *text == '\t' || *text == '\b')
376  ++text;
377 
378  if (!*text)
379  return nullptr;
380 
381  // return non link case
382  if (text[0] != '|')
383  return strtok(text, " ");
384 
385  // [name] Shift-click form |color|linkType:key|h[name]|h|r
386  // or
387  // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
388 
389  char* check = strtok(text, "|"); // skip color
390  if (!check)
391  return nullptr; // end of data
392 
393  char* cLinkType = strtok(nullptr, ":"); // linktype
394  if (!cLinkType)
395  return nullptr; // end of data
396 
397  if (strcmp(cLinkType, linkType) != 0)
398  {
399  strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after retturn from function
400  SendSysMessage(LANG_WRONG_LINK_TYPE);
401  return nullptr;
402  }
403 
404  char* cKeys = strtok(nullptr, "|"); // extract keys and values
405  char* cKeysTail = strtok(nullptr, "");
406 
407  char* cKey = strtok(cKeys, ":|"); // extract key
408  if (something1)
409  *something1 = strtok(nullptr, ":|"); // extract something
410 
411  strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
412  strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
413  return cKey;
414 }

416 char* ChatHandler::extractKeyFromLink(char* text, char const* const* linkTypes, int* found_idx, char** something1)
417 {
418  // skip empty
419  if (!text)
420  return nullptr;
421 
422  // skip spaces
423  while (*text == ' ' || *text == '\t' || *text == '\b')
424  ++text;
425 
426  if (!*text)
427  return nullptr;
428 
429  // return non link case
430  if (text[0] != '|')
431  return strtok(text, " ");
432 
433  // [name] Shift-click form |color|linkType:key|h[name]|h|r
434  // or
435  // [name] Shift-click form |color|linkType:key:something1:...:somethingN|h[name]|h|r
436  // or
437  // [name] Shift-click form |linkType:key|h[name]|h|r
438 
439  char* tail;
440 
441  if (text[1] == 'c')
442  {
443  char* check = strtok(text, "|"); // skip color
444  if (!check)
445  return nullptr; // end of data
446 
447  tail = strtok(nullptr, ""); // tail
448  }
449  else
450  tail = text + 1; // skip first |
451 
452  char* cLinkType = strtok(tail, ":"); // linktype
453  if (!cLinkType)
454  return nullptr; // end of data
455 
456  for (int i = 0; linkTypes[i]; ++i)
457  {
458  if (strcmp(cLinkType, linkTypes[i]) == 0)
459  {
460  char* cKeys = strtok(nullptr, "|"); // extract keys and values
461  char* cKeysTail = strtok(nullptr, "");
462 
463  char* cKey = strtok(cKeys, ":|"); // extract key
464  if (something1)
465  *something1 = strtok(nullptr, ":|"); // extract something
466 
467  strtok(cKeysTail, "]"); // restart scan tail and skip name with possible spaces
468  strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
469  if (found_idx)
470  *found_idx = i;
471  return cKey;
472  }
473  }
474 
475  strtok(nullptr, " "); // skip link tail (to allow continue strtok(nullptr, s) use after return from function
476  SendSysMessage(LANG_WRONG_LINK_TYPE);
477  return nullptr;
478 }

543 uint32 ChatHandler::extractSpellIdFromLink(char* text)
544 {
545  // number or [name] Shift-click form |color|Henchant:recipe_spell_id|h[prof_name: recipe_name]|h|r
546  // number or [name] Shift-click form |color|Hglyph:glyph_slot_id:glyph_prop_id|h[%s]|h|r
547  // number or [name] Shift-click form |color|Hspell:spell_id|h[name]|h|r
548  // number or [name] Shift-click form |color|Htalent:talent_id, rank|h[name]|h|r
549  // number or [name] Shift-click form |color|Htrade:spell_id, skill_id, max_value, cur_value|h[name]|h|r
550  int type = 0;
551  char* param1_str = nullptr;
552  char* idS = extractKeyFromLink(text, spellKeys, &type, &param1_str);
553  if (!idS)
554  return 0;
555 
556  uint32 id = (uint32)atol(idS);
557 
558  switch (type)
559  {
560  case SPELL_LINK_SPELL:
561  return id;
562  case SPELL_LINK_TALENT:
563  {
564  // talent
565  TalentEntry const* talentEntry = sTalentStore.LookupEntry(id);
566  if (!talentEntry)
567  return 0;
568 
569  int32 rank = param1_str ? (uint32)atol(param1_str) : 0;
570  if (rank >= MAX_TALENT_RANK)
571  return 0;
572 
573  if (rank < 0)
574  rank = 0;
575 
576  return talentEntry->RankID[rank];
577  }
578  case SPELL_LINK_ENCHANT:
579  case SPELL_LINK_TRADE:
580  return id;
581  case SPELL_LINK_GLYPH:
582  {
583  uint32 glyph_prop_id = param1_str ? (uint32)atol(param1_str) : 0;
584 
585  GlyphPropertiesEntry const* glyphPropEntry = sGlyphPropertiesStore.LookupEntry(glyph_prop_id);
586  if (!glyphPropEntry)
587  return 0;
588 
589  return glyphPropEntry->SpellId;
590  }
591  }
592 
593  // unknown type?
594  return 0;
595 }
