92  static bool HandleGameObjectAddCommand(ChatHandler* handler, GameObjectEntry objectId, Optional<int32> spawnTimeSecs)
93  {
94  if (!objectId)
95  return false;
96 
97  GameObjectTemplate const* objectInfo = sObjectMgr->GetGameObjectTemplate(objectId);
98  if (!objectInfo)
99  {
100  handler->PSendSysMessage(LANG_GAMEOBJECT_NOT_EXIST, uint32(objectId));
101  handler->SetSentErrorMessage(true);
102  return false;
103  }
104 
105  if (objectInfo->displayId && !sGameObjectDisplayInfoStore.LookupEntry(objectInfo->displayId))
106  {
107  // report to DB errors log as in loading case
108  LOG_ERROR("sql.sql", "Gameobject (Entry %u GoType: %u) have invalid displayId (%u), not spawned.", *objectId, objectInfo->type, objectInfo->displayId);
109  handler->PSendSysMessage(LANG_GAMEOBJECT_HAVE_INVALID_DATA, uint32(objectId));
110  handler->SetSentErrorMessage(true);
111  return false;
112  }
113 
114  Player* player = handler->GetSession()->GetPlayer();
115  float x = float(player->GetPositionX());
116  float y = float(player->GetPositionY());
117  float z = float(player->GetPositionZ());
118  float o = float(player->GetOrientation());
119  Map* map = player->GetMap();
120 
121  GameObject* object = sObjectMgr->IsGameObjectStaticTransport(objectInfo->entry) ? new StaticTransport() : new GameObject();
122  ObjectGuid::LowType guidLow = map->GenerateLowGuid<HighGuid::GameObject>();
123 
124  if (!object->Create(guidLow, objectInfo->entry, map, player->GetPhaseMaskForSpawn(), x, y, z, o, G3D::Quat(), 0, GO_STATE_READY))
125  {
126  delete object;
127  return false;
128  }
129 
130  if (spawnTimeSecs)
131  object->SetRespawnTime(*spawnTimeSecs);
132 
133  // fill the gameobject data and save to the db
134  object->SaveToDB(map->GetId(), (1 << map->GetSpawnMode()), player->GetPhaseMaskForSpawn());
135  guidLow = object->GetSpawnId();
136 
137  // delete the old object and do a clean load from DB with a fresh new GameObject instance.
138  // this is required to avoid weird behavior and memory leaks
139  delete object;
140 
141  object = sObjectMgr->IsGameObjectStaticTransport(objectInfo->entry) ? new StaticTransport() : new GameObject();
142  // this will generate a new guid if the object is in an instance
143  if (!object->LoadGameObjectFromDB(guidLow, map, true))
144  {
145  delete object;
146  return false;
147  }
148 
150  sObjectMgr->AddGameobjectToGrid(guidLow, sObjectMgr->GetGOData(guidLow));
151 
152  handler->PSendSysMessage(LANG_GAMEOBJECT_ADD, uint32(objectId), objectInfo->name.c_str(), guidLow, player->GetPositionX(), player->GetPositionY(), player->GetPositionZ());
153  return true;
154  }

187

190  if (objectId->holds_alternative<GameObjectEntry>())
191  {
192  result = WorldDatabase.PQuery("SELECT guid, id, position_x, position_y, position_z, orientation, map, phaseMask, (POW(position_x - '%f', 2) + POW(position_y - '%f', 2) + POW(position_z - '%f', 2)) AS order_ FROM gameobject WHERE map = '%i' AND id = '%u' ORDER BY order_ ASC LIMIT 1",
193  player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetMapId(), static_cast<uint32>(objectId->get<GameObjectEntry>()));
194  }
195  else
196  {
197  std::string name = std::string(objectId->get<std::string_view>());
198  WorldDatabase.EscapeString(name);
199  result = WorldDatabase.PQuery(
200  "SELECT guid, id, position_x, position_y, position_z, orientation, map, phaseMask, (POW(position_x - %f, 2) + POW(position_y - %f, 2) + POW(position_z - %f, 2)) AS order_ "
201  "FROM gameobject LEFT JOIN gameobject_template ON gameobject_template.entry = gameobject.id WHERE map = %i AND name LIKE '%%%s%%' ORDER BY order_ ASC LIMIT 1",
202  player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetMapId(), name.c_str());
203  }
204  }
205  else
206  {
207  std::ostringstream eventFilter;
208  eventFilter << " AND (eventEntry IS NULL ";
209  bool initString = true;
210 
211  for (GameEventMgr::ActiveEvents::const_iterator itr = activeEventsList.begin(); itr != activeEventsList.end(); ++itr)
212  {
213  if (initString)
214  {
215  eventFilter << "OR eventEntry IN (" << *itr;
216  initString = false;
217  }
218  else
219  eventFilter << ',' << *itr;
220  }
221 
222  if (!initString)
223  eventFilter << "))";
224  else
225  eventFilter << ')';
226 
227  result = WorldDatabase.PQuery("SELECT gameobject.guid, id, position_x, position_y, position_z, orientation, map, phaseMask, "
228  "(POW(position_x - %f, 2) + POW(position_y - %f, 2) + POW(position_z - %f, 2)) AS order_ FROM gameobject "
229  "LEFT OUTER JOIN game_event_gameobject on gameobject.guid = game_event_gameobject.guid WHERE map = '%i' %s ORDER BY order_ ASC LIMIT 10",
230  handler->GetSession()->GetPlayer()->GetPositionX(), handler->GetSession()->GetPlayer()->GetPositionY(), handler->GetSession()->GetPlayer()->GetPositionZ(),
231  handler->GetSession()->GetPlayer()->GetMapId(), eventFilter.str().c_str());
232  }
233 
234  if (!result)
235  {
236  handler->SendSysMessage(LANG_COMMAND_TARGETOBJNOTFOUND);
237  return true;
238  }
239 
240  bool found = false;
241  float x, y, z, o;
242  ObjectGuid::LowType guidLow;
243  uint32 id, phase;
244  uint16 mapId;
245  uint32 poolId;
246 
247  do
248  {
249  Field* fields = result->Fetch();
250  guidLow = fields[0].GetUInt32();
251  id = fields[1].GetUInt32();
252  x = fields[2].GetFloat();
253  y = fields[3].GetFloat();
254  z = fields[4].GetFloat();
255  o = fields[5].GetFloat();
256  mapId = fields[6].GetUInt16();
257  phase = fields[7].GetUInt32();
258  poolId = sPoolMgr->IsPartOfAPool<GameObject>(guidLow);
259  if (!poolId || sPoolMgr->IsSpawnedObject<GameObject>(guidLow))
260  found = true;
261  } while (result->NextRow() && !found);
262 
263  if (!found)
264  {
265  handler->PSendSysMessage(LANG_GAMEOBJECT_NOT_EXIST, id);
266  return false;
267  }
268 
269  GameObjectTemplate const* objectInfo = sObjectMgr->GetGameObjectTemplate(id);
270 
271  if (!objectInfo)
272  {
273  handler->PSendSysMessage(LANG_GAMEOBJECT_NOT_EXIST, id);
274  return false;
275  }
276 
277  GameObject* target = handler->GetObjectFromPlayerMapByDbGuid(guidLow);
278 
279  handler->PSendSysMessage(LANG_GAMEOBJECT_DETAIL, guidLow, objectInfo->name.c_str(), guidLow, id, x, y, z, mapId, o, phase);
280 
281  if (target)
282  {
283  int32 curRespawnDelay = int32(target->GetRespawnTimeEx() - time(nullptr));
284  if (curRespawnDelay < 0)
285  curRespawnDelay = 0;
286 
287  std::string curRespawnDelayStr = secsToTimeString(curRespawnDelay, true);
288  std::string defRespawnDelayStr = secsToTimeString(target->GetRespawnDelay(), true);
289 
290  handler->PSendSysMessage(LANG_COMMAND_RAWPAWNTIMES, defRespawnDelayStr.c_str(), curRespawnDelayStr.c_str());
291  }
292  return true;
293  }

587  }
588  handler->PSendSysMessage("Set gobject type %d state %u", objectType, *objectState);
