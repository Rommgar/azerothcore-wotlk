115  {
116  BossAI::EnterEvadeMode();

170  void UpdateAI(uint32 diff) override
171  {
172  if (!UpdateVictim())
173  return;
174 
175  events.Update(diff);
176  if (me->HasUnitState(UNIT_STATE_CASTING))
177  return;
178 
179  switch (events.ExecuteEvent())
180  {
181  case EVENT_SPELL_ENRAGE:
182  Talk(YELL_ENRAGE);
183  me->CastSpell(me, SPELL_ENRAGE, true);
184  break;
185  case EVENT_SPELL_CONFOUNDING_BLOW:
186  me->CastSpell(me->GetVictim(), SPELL_CONFOUNDING_BLOW, false);
187  events.ScheduleEvent(EVENT_SPELL_CONFOUNDING_BLOW, urand(20000, 25000));
188  break;
189  case EVENT_SPELL_SHADOW_BLADES:
190  me->CastSpell(me, SPELL_SHADOW_BLADES, false);
191  events.ScheduleEvent(EVENT_SPELL_SHADOW_BLADES, 10000);
192  break;
193  case EVENT_SPELL_SHADOW_NOVA:
194  {
195  Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1, 100.0f);
196  if (!target)
197  target = me->GetVictim();
198  Talk(EMOTE_SHADOW_NOVA, target);
199  Talk(YELL_SHADOW_NOVA);
200  me->CastSpell(target, SPELL_SHADOW_NOVA, false);
201  events.ScheduleEvent(EVENT_SPELL_SHADOW_NOVA, urand(30000, 35000));
202  break;
203  }
204  case EVENT_SHADOW_IMAGE:
205  me->SummonCreature(NPC_SHADOW_IMAGE, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation(), TEMPSUMMON_TIMED_DESPAWN, 12000);
206  events.ScheduleEvent(EVENT_SHADOW_IMAGE, 6000);
207  break;
208  case EVENT_SPELL_CONFLAGRATION:
209  {
210  Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1, 100.0f);
211  if (!target)
212  target = me->GetVictim();
213  me->CastSpell(target, SPELL_CONFLAGRATION, false);
214  events.ScheduleEvent(EVENT_SPELL_CONFLAGRATION, urand(30000, 35000));
215  break;
216  }
217  }
218 
219  DoMeleeAttackIfReady();
220  }
221  };
