211  const G3D::Vector3& endPos, float& maxDist) const

223 bool DynamicMapTree::GetObjectHitPos(const uint32 phasemask, const G3D::Vector3& startPos,
224  const G3D::Vector3& endPos, G3D::Vector3& resultHit,
225  float modifyDist) const
226 {
227  bool result = false;
228  float maxDist = (endPos - startPos).magnitude();
229  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
230  ASSERT(maxDist < std::numeric_limits<float>::max());
231  // prevent NaN values which can cause BIH intersection to enter infinite loop
232  if (maxDist < 1e-10f)
233  {
234  resultHit = endPos;
235  return false;
236  }
237  G3D::Vector3 dir = (endPos - startPos) / maxDist; // direction with length of 1
238  G3D::Ray ray(startPos, dir);
239  float dist = maxDist;
240  if (GetIntersectionTime(phasemask, ray, endPos, dist))
241  {
242  resultHit = startPos + dir * dist;
243  if (modifyDist < 0)
244  {
245  if ((resultHit - startPos).magnitude() > -modifyDist)
246  {
247  resultHit = resultHit + dir * modifyDist;
248  }
249  else
250  {
251  resultHit = startPos;
252  }
253  }
254  else
255  {
256  resultHit = resultHit + dir * modifyDist;
257  }
258 
259  result = true;
260  }
261  else
262  {
263  resultHit = endPos;
264  result = false;
265  }
266  return result;
267 }

269 bool DynamicMapTree::isInLineOfSight(float x1, float y1, float z1, float x2, float y2, float z2, uint32 phasemask) const
270 {
271  G3D::Vector3 v1(x1, y1, z1), v2(x2, y2, z2);
272 
273  float maxDist = (v2 - v1).magnitude();
274 
275  if (!G3D::fuzzyGt(maxDist, 0) )
276  {
277  return true;
278  }
279 
280  G3D::Ray r(v1, (v2 - v1) / maxDist);
281  DynamicTreeIntersectionCallback callback(phasemask);
282  impl->intersectRay(r, callback, maxDist, v2, true);
283 
284  return !callback.did_hit;
