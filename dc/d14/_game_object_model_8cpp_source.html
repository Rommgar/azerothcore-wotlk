44 void LoadGameObjectModelList(std::string const& dataPath)
45 {
46  uint32 oldMSTime = getMSTime();
47 
48  FILE* model_list_file = fopen((dataPath + "vmaps/" + VMAP::GAMEOBJECT_MODELS).c_str(), "rb");
49  if (!model_list_file)
50  {
51  LOG_ERROR("maps", "Unable to open '%s' file.", VMAP::GAMEOBJECT_MODELS);
52  return;
53  }
54 
55  char magic[8];
56  if (fread(magic, 1, 8, model_list_file) != 8 || memcmp(magic, VMAP::VMAP_MAGIC, 8) != 0)
57  {
58  LOG_ERROR("maps", "File '%s' has wrong header, expected %s.", VMAP::GAMEOBJECT_MODELS, VMAP::VMAP_MAGIC);
59  return;
60  }
61 
62  uint32 name_length, displayId;
63  uint8 isWmo;
64  char buff[500];
65  while (true)
66  {
67  Vector3 v1, v2;
68  if (fread(&displayId, sizeof(uint32), 1, model_list_file) != 1)
69  if (feof(model_list_file)) // EOF flag is only set after failed reading attempt
70  {
71  break;
72  }
73 
74  if (fread(&isWmo, sizeof(uint8), 1, model_list_file) != 1
75  || fread(&name_length, sizeof(uint32), 1, model_list_file) != 1
76  || name_length >= sizeof(buff)
77  || fread(&buff, sizeof(char), name_length, model_list_file) != name_length
78  || fread(&v1, sizeof(Vector3), 1, model_list_file) != 1
79  || fread(&v2, sizeof(Vector3), 1, model_list_file) != 1)
80  {
81  LOG_ERROR("maps", "File '%s' seems to be corrupted!", VMAP::GAMEOBJECT_MODELS);
82  fclose(model_list_file);
83  break;
84  }
85 
86  if (v1.isNaN() || v2.isNaN())
87  {
88  LOG_ERROR("maps", "File '%s' Model '%s' has invalid v1%s v2%s values!",
89  VMAP::GAMEOBJECT_MODELS, std::string(buff, name_length).c_str(), v1.toString().c_str(), v2.toString().c_str());
90  continue;
91  }
92 
93  model_list.emplace(std::piecewise_construct, std::forward_as_tuple(displayId), std::forward_as_tuple(&buff[0], name_length, v1, v2, isWmo != 0));
94  }
95 
96  fclose(model_list_file);
97 
98  LOG_INFO("server.loading", ">> Loaded %u GameObject models in %u ms", uint32(model_list.size()), GetMSTimeDiffToNow(oldMSTime));
99  LOG_INFO("server.loading", " ");
100 }

110 bool GameObjectModel::initialize(std::unique_ptr<GameObjectModelOwnerBase> modelOwner, std::string const& dataPath)
111 {
112  ModelList::const_iterator it = model_list.find(modelOwner->GetDisplayId());
113  if (it == model_list.end())
114  {
115  return false;
116  }
117 
118  G3D::AABox mdl_box(it->second.bound);
119  // ignore models with no bounds
120  if (mdl_box == G3D::AABox::zero())
121  {
122  LOG_ERROR("maps", "GameObject model %s has zero bounds, loading skipped", it->second.name.c_str());
123  return false;
124  }
125 
126  iModel = VMAP::VMapFactory::createOrGetVMapMgr()->acquireModelInstance(dataPath + "vmaps/", it->second.name);
127 
128  if (!iModel)
129  {
130  return false;
131  }
132 
133  name = it->second.name;
134  iPos = modelOwner->GetPosition();
135  phasemask = modelOwner->GetPhaseMask();
136  iScale = modelOwner->GetScale();
137  iInvScale = 1.f / iScale;
138 
139  G3D::Matrix3 iRotation = G3D::Matrix3::fromEulerAnglesZYX(modelOwner->GetOrientation(), 0, 0);
140  iInvRot = iRotation.inverse();
141  // transform bounding box:
142  mdl_box = AABox(mdl_box.low() * iScale, mdl_box.high() * iScale);
143  AABox rotated_bounds;
144  for (int i = 0; i < 8; ++i)
145  {
146  rotated_bounds.merge(iRotation * mdl_box.corner(i));
147  }
148 
149  iBound = rotated_bounds + iPos;
150 
151 #ifdef SPAWN_CORNERS
152  // test:
153  for (int i = 0; i < 8; ++i)
154  {
155  Vector3 pos(iBound.corner(i));
156  modelOwner->DebugVisualizeCorner(pos);
157  }
158 #endif
159 
160  owner = std::move(modelOwner);
161  isWmo = it->second.isWmo;
162  return true;
163 }

177 bool GameObjectModel::intersectRay(const G3D::Ray& ray, float& MaxDist, bool StopAtFirstHit, uint32 ph_mask) const
178 {
179  if (!(phasemask & ph_mask) || !owner->IsSpawned())
180  {
181  return false;
182  }
183 
184  float time = ray.intersectionTime(iBound);
185  if (time == G3D::inf())
186  {
187  return false;
188  }
189 
190  // child bounds are defined in object space:
191  Vector3 p = iInvRot * (ray.origin() - iPos) * iInvScale;
192  Ray modRay(p, iInvRot * ray.direction());
193  float distance = MaxDist * iInvScale;
194  bool hit = iModel->IntersectRay(modRay, distance, StopAtFirstHit);
195  if (hit)
196  {
197  distance *= iScale;
198  MaxDist = distance;
199  }
200  return hit;
201 }

203 void GameObjectModel::IntersectPoint(G3D::Vector3 const& point, VMAP::AreaInfo& info, uint32 ph_mask) const
204 {
205  if (!(phasemask & ph_mask) || !owner->IsSpawned() || !IsMapObject())
206  return;
207 
208  if (!iBound.contains(point))
209  return;
210 
211  // child bounds are defined in object space:
212  Vector3 pModel = iInvRot * (point - iPos) * iInvScale;
213  Vector3 zDirModel = iInvRot * Vector3(0.f, 0.f, -1.f);
214  float zDist;
215  if (iModel->IntersectPoint(pModel, zDirModel, zDist, info))
216  {
217  Vector3 modelGround = pModel + zDist * zDirModel;
218  float world_Z = ((modelGround * iInvRot) * iScale + iPos).z;
219  if (info.ground_Z < world_Z)
220  info.ground_Z = world_Z;
221  }
222 }
