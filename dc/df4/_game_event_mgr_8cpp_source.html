47 bool GameEventMgr::CheckOneGameEvent(uint16 entry) const
48 {
49  switch (mGameEvent[entry].state)
50  {
51  default:
52  case GAMEEVENT_NORMAL:
53  {
54  time_t currenttime = time(nullptr);
55  // Get the event information
56  return mGameEvent[entry].start < currenttime
57  && currenttime < mGameEvent[entry].end
58  && (currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE) < mGameEvent[entry].length * MINUTE;
59  }
60  // if the state is conditions or nextphase, then the event should be active
61  case GAMEEVENT_WORLD_CONDITIONS:
62  case GAMEEVENT_WORLD_NEXTPHASE:
63  return true;
64  // finished world events are inactive
65  case GAMEEVENT_WORLD_FINISHED:
66  case GAMEEVENT_INTERNAL:
67  return false;
68  // if inactive world event, check the prerequisite events
69  case GAMEEVENT_WORLD_INACTIVE:
70  {
71  time_t currenttime = time(nullptr);
72  for (std::set<uint16>::const_iterator itr = mGameEvent[entry].prerequisite_events.begin(); itr != mGameEvent[entry].prerequisite_events.end(); ++itr)
73  {
74  if ((mGameEvent[*itr].state != GAMEEVENT_WORLD_NEXTPHASE && mGameEvent[*itr].state != GAMEEVENT_WORLD_FINISHED) || // if prereq not in nextphase or finished state, then can't start this one
75  mGameEvent[*itr].nextstart > currenttime) // if not in nextphase state for long enough, can't start this one
76  return false;
77  }
78  // all prerequisite events are met
79  // but if there are no prerequisites, this can be only activated through gm command
80  return !(mGameEvent[entry].prerequisite_events.empty());
81  }
82  }

85 uint32 GameEventMgr::NextCheck(uint16 entry) const
86 {
87  time_t currenttime = time(nullptr);
88 
89  // for NEXTPHASE state world events, return the delay to start the next event, so the followup event will be checked correctly
90  if ((mGameEvent[entry].state == GAMEEVENT_WORLD_NEXTPHASE || mGameEvent[entry].state == GAMEEVENT_WORLD_FINISHED) && mGameEvent[entry].nextstart >= currenttime)
91  return uint32(mGameEvent[entry].nextstart - currenttime);
92 
93  // for CONDITIONS state world events, return the length of the wait period, so if the conditions are met, this check will be called again to set the timer as NEXTPHASE event
94  if (mGameEvent[entry].state == GAMEEVENT_WORLD_CONDITIONS)
95  {
96  if (mGameEvent[entry].length)
97  return mGameEvent[entry].length * 60;
98  else
99  return max_ge_check_delay;
100  }
101 
102  // outdated event: we return max
103  if (currenttime > mGameEvent[entry].end)
104  return max_ge_check_delay;
105 
106  // never started event, we return delay before start
107  if (mGameEvent[entry].start > currenttime)
108  return uint32(mGameEvent[entry].start - currenttime);
109 
110  uint32 delay;
111  // in event, we return the end of it
112  if ((((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * 60)) < (mGameEvent[entry].length * 60)))
113  // we return the delay before it ends
114  delay = (mGameEvent[entry].length * MINUTE) - ((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE));
115  else // not in window, we return the delay before next start
116  delay = (mGameEvent[entry].occurence * MINUTE) - ((currenttime - mGameEvent[entry].start) % (mGameEvent[entry].occurence * MINUTE));
117  // In case the end is before next check
118  if (mGameEvent[entry].end < time_t(currenttime + delay))
119  return uint32(mGameEvent[entry].end - currenttime);
120  else
121  return delay;
122 }

138 bool GameEventMgr::StartEvent(uint16 event_id, bool overwrite)
139 {
140  if (DisableMgr::IsDisabledFor(DISABLE_TYPE_GAME_EVENT, event_id, nullptr) && !overwrite)
141  {
142  return false;
143  }
144 
145  GameEventData& data = mGameEvent[event_id];
146  if (data.state == GAMEEVENT_NORMAL || data.state == GAMEEVENT_INTERNAL)
147  {
148  AddActiveEvent(event_id);
149  ApplyNewEvent(event_id);
150  if (overwrite)
151  {
152  mGameEvent[event_id].start = time(nullptr);
153  if (data.end <= data.start)
154  data.end = data.start + data.length;
155  }
156 
157  if (IsActiveEvent(event_id))
158  sScriptMgr->OnGameEventStart(event_id);
159 
160  return false;
161  }
162  else
163  {
164  if (data.state == GAMEEVENT_WORLD_INACTIVE)
165  // set to conditions phase
166  data.state = GAMEEVENT_WORLD_CONDITIONS;
167 
168  // add to active events
169  AddActiveEvent(event_id);
170  // add spawns
171  ApplyNewEvent(event_id);
172 
173  // check if can go to next state
174  bool conditions_met = CheckOneGameEventConditions(event_id);
175  // save to db
176  SaveWorldEventStateToDB(event_id);
177  // force game event update to set the update timer if conditions were met from a command
178  // this update is needed to possibly start events dependent on the started one
179  // or to scedule another update where the next event will be started
180  if (overwrite && conditions_met)
181  sWorld->ForceGameEventUpdate();
182 
183  if (IsActiveEvent(event_id))
184  sScriptMgr->OnGameEventStart(event_id);
185 
186  return conditions_met;
187  }

190 void GameEventMgr::StopEvent(uint16 event_id, bool overwrite)
191 {
192  GameEventData& data = mGameEvent[event_id];
193  bool serverwide_evt = data.state != GAMEEVENT_NORMAL && data.state != GAMEEVENT_INTERNAL;
194 
195  RemoveActiveEvent(event_id);
196  UnApplyEvent(event_id);
197 
198  if (overwrite && !serverwide_evt)
199  {
200  data.start = time(nullptr) - data.length * MINUTE;
201  if (data.end <= data.start)
202  data.end = data.start + data.length;
203  }
204  else if (serverwide_evt)
205  {
206  // if finished world event, then only gm command can stop it
207  if (overwrite || data.state != GAMEEVENT_WORLD_FINISHED)
208  {
209  // reset conditions
210  data.nextstart = 0;
211  data.state = GAMEEVENT_WORLD_INACTIVE;
212  GameEventConditionMap::iterator itr;
213  for (itr = data.conditions.begin(); itr != data.conditions.end(); ++itr)
214  itr->second.done = 0;
215 
216  CharacterDatabaseTransaction trans = CharacterDatabase.BeginTransaction();
217  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_ALL_GAME_EVENT_CONDITION_SAVE);
218  stmt->setUInt8(0, uint8(event_id));
219  trans->Append(stmt);
220 
221  stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_GAME_EVENT_SAVE);
222  stmt->setUInt8(0, uint8(event_id));
223  trans->Append(stmt);
224 
225  CharacterDatabase.CommitTransaction(trans);
226  }
227  }
228 
229  if (!IsActiveEvent(event_id))
230  sScriptMgr->OnGameEventStop(event_id);
231 }

1106 uint32 GameEventMgr::Update() // return the next event delay in ms
1107 {
1108  time_t currenttime = time(nullptr);
1109  uint32 nextEventDelay = max_ge_check_delay; // 1 day
1110  uint32 calcDelay;
1111  std::set<uint16> activate, deactivate;
1112  for (uint16 itr = 1; itr < mGameEvent.size(); ++itr)
1113  {
1114  // must do the activating first, and after that the deactivating
1115  // so first queue it
1116  //LOG_ERROR("sql.sql", "Checking event %u", itr);
1117  if (CheckOneGameEvent(itr))
1118  {
1119  // if the world event is in NEXTPHASE state, and the time has passed to finish this event, then do so
1120  if (mGameEvent[itr].state == GAMEEVENT_WORLD_NEXTPHASE && mGameEvent[itr].nextstart <= currenttime)
1121  {
1122  // set this event to finished, null the nextstart time
1123  mGameEvent[itr].state = GAMEEVENT_WORLD_FINISHED;
1124  mGameEvent[itr].nextstart = 0;
1125  // save the state of this gameevent
1126  SaveWorldEventStateToDB(itr);
1127  // queue for deactivation
1128  if (IsActiveEvent(itr))
1129  deactivate.insert(itr);
1130  // go to next event, this no longer needs an event update timer
1131  continue;
1132  }
1133  else if (mGameEvent[itr].state == GAMEEVENT_WORLD_CONDITIONS && CheckOneGameEventConditions(itr))
1134  // changed, save to DB the gameevent state, will be updated in next update cycle
1135  SaveWorldEventStateToDB(itr);
1136 
1137  // queue for activation
1138  if (!IsActiveEvent(itr))
1139  activate.insert(itr);
1140  }
1141  else
1142  {
1143  if (IsActiveEvent(itr))
1144  {
1145  // Xinef: do not deactivate internal events on whim
1146  if (mGameEvent[itr].state != GAMEEVENT_INTERNAL)
1147  deactivate.insert(itr);
1148  }
1149  else
1150  {
1151  if (!isSystemInit)
1152  {
1153  int16 event_nid = (-1) * (itr);
1154  // spawn all negative ones for this event
1155  GameEventSpawn(event_nid);
1156  }
1157  }
1158  }
1159  calcDelay = NextCheck(itr);
1160  if (calcDelay < nextEventDelay)
1161  nextEventDelay = calcDelay;
1162  }
1163  // now activate the queue
1164  // a now activated event can contain a spawn of a to-be-deactivated one
1165  // following the activate - deactivate order, deactivating the first event later will leave the spawn in (wont disappear then reappear clientside)
1166  for (std::set<uint16>::iterator itr = activate.begin(); itr != activate.end(); ++itr)
1167  // start the event
1168  // returns true the started event completed
1169  // in that case, initiate next update in 1 second
1170  if (StartEvent(*itr))
1171  nextEventDelay = 0;
1172  for (std::set<uint16>::iterator itr = deactivate.begin(); itr != deactivate.end(); ++itr)
1173  StopEvent(*itr);
1174 
1175  LOG_DEBUG("gameevent", "Next game event check in %u seconds.", nextEventDelay + 1);
1176  return (nextEventDelay + 1) * IN_MILLISECONDS; // Add 1 second to be sure event has started/stopped at next call
1177 }
