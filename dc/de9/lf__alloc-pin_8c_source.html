17 /*
18  wait-free concurrent allocator based on pinning addresses
19 
20  It works as follows: every thread (strictly speaking - every CPU, but
21  it's too difficult to do) has a small array of pointers. They're called
22  "pins". Before using an object its address must be stored in this array
23  (pinned). When an object is no longer necessary its address must be
24  removed from this array (unpinned). When a thread wants to free() an
25  object it scans all pins of all threads to see if somebody has this
26  object pinned. If yes - the object is not freed (but stored in a
27  "purgatory"). To reduce the cost of a single free() pins are not scanned
28  on every free() but only added to (thread-local) purgatory. On every
29  LF_PURGATORY_SIZE free() purgatory is scanned and all unpinned objects
30  are freed.
31 
32  Pins are used to solve ABA problem. To use pins one must obey
33  a pinning protocol:
34 
35  1. Let's assume that PTR is a shared pointer to an object. Shared means
36  that any thread may modify it anytime to point to a different object
37  and free the old object. Later the freed object may be potentially
38  allocated by another thread. If we're unlucky that other thread may
39  set PTR to point to this object again. This is ABA problem.
40  2. Create a local pointer LOCAL_PTR.
41  3. Pin the PTR in a loop:
42  do
43  {
44  LOCAL_PTR= PTR;
45  pin(PTR, PIN_NUMBER);
46  } while (LOCAL_PTR != PTR)
47  4. It is guaranteed that after the loop has ended, LOCAL_PTR
48  points to an object (or NULL, if PTR may be NULL), that
49  will never be freed. It is not guaranteed though
50  that LOCAL_PTR == PTR (as PTR can change any time)
51  5. When done working with the object, remove the pin:
52  unpin(PIN_NUMBER)
53  6. When copying pins (as in the list traversing loop:
54  pin(CUR, 1);
55  while ()
56  {
57  do // standard
58  { // pinning
59  NEXT=CUR->next; // loop
60  pin(NEXT, 0); // see #3
61  } while (NEXT != CUR->next); // above
62  ...
63  ...
64  CUR=NEXT;
65  pin(CUR, 1); // copy pin[0] to pin[1]
66  }
67  which keeps CUR address constantly pinned), note than pins may be
68  copied only upwards (!!!), that is pin[N] to pin[M], M > N.
69  7. Don't keep the object pinned longer than necessary - the number of
70  pins you have is limited (and small), keeping an object pinned
71  prevents its reuse and cause unnecessary mallocs.
72 
73  Explanations:
74 
75  3. The loop is important. The following can occur:
76  thread1> LOCAL_PTR= PTR
77  thread2> free(PTR); PTR=0;
78  thread1> pin(PTR, PIN_NUMBER);
79  now thread1 cannot access LOCAL_PTR, even if it's pinned,
80  because it points to a freed memory. That is, it *must*
81  verify that it has indeed pinned PTR, the shared pointer.
82 
83  6. When a thread wants to free some LOCAL_PTR, and it scans
84  all lists of pins to see whether it's pinned, it does it
85  upwards, from low pin numbers to high. Thus another thread
86  must copy an address from one pin to another in the same
87  direction - upwards, otherwise the scanning thread may
88  miss it.
89 
90  Implementation details:
91 
92  Pins are given away from a "pinbox". Pinbox is stack-based allocator.
93  It used dynarray for storing pins, new elements are allocated by dynarray
94  as necessary, old are pushed in the stack for reuse. ABA is solved by
95  versioning a pointer - because we use an array, a pointer to pins is 16 bit,
96  upper 16 bits are used for a version.
97 
98  It is assumed that pins belong to a THD and are not transferable
99  between THD's (LF_PINS::stack_ends_here being a primary reason
100  for this limitation).
101 */

132 /*
133  Get pins from a pinbox. Usually called via lf_alloc_get_pins() or
134  lf_hash_get_pins().
135 
136  SYNOPSYS
137  pinbox -
138 
139  DESCRIPTION
140  get a new LF_PINS structure from a stack of unused pins,
141  or allocate a new one out of dynarray.
142 
143  NOTE
144  It is assumed that pins belong to a thread and are not transferable
145  between threads.
146 */
147 LF_PINS *_lf_pinbox_get_pins(LF_PINBOX *pinbox)
148 {
149  uint32 pins, next, top_ver;
150  LF_PINS *el;
151  /*
152  We have an array of max. 64k elements.
153  The highest index currently allocated is pinbox->pins_in_array.
154  Freed elements are in a lifo stack, pinstack_top_ver.
155  pinstack_top_ver is 32 bits; 16 low bits are the index in the
156  array, to the first element of the list. 16 high bits are a version
157  (every time the 16 low bits are updated, the 16 high bits are
158  incremented). Versioniong prevents the ABA problem.
159  */
160  top_ver= pinbox->pinstack_top_ver;
161  do
162  {
163  if (!(pins= top_ver % LF_PINBOX_MAX_PINS))
164  {
165  /* the stack of free elements is empty */
166  pins= my_atomic_add32((int32 volatile*) &pinbox->pins_in_array, 1)+1;
167  if (unlikely(pins >= LF_PINBOX_MAX_PINS))
168  return 0;
169  /*
170  note that the first allocated element has index 1 (pins==1).
171  index 0 is reserved to mean "NULL pointer"
172  */
173  el= (LF_PINS *)_lf_dynarray_lvalue(&pinbox->pinarray, pins);
174  if (unlikely(!el))
175  return 0;
176  break;
177  }
178  el= (LF_PINS *)_lf_dynarray_value(&pinbox->pinarray, pins);
179  next= el->link;
180  } while (!my_atomic_cas32((int32 volatile*) &pinbox->pinstack_top_ver,
181  (int32*) &top_ver,
182  top_ver-pins+next+LF_PINBOX_MAX_PINS));
183  /*
184  set el->link to the index of el in the dynarray (el->link has two usages:
185  - if element is allocated, it's its own index
186  - if element is free, it's its next element in the free stack
187  */
188  el->link= pins;
189  el->purgatory_count= 0;
190  el->pinbox= pinbox;
191  el->stack_ends_here= & my_thread_var->stack_ends_here;
192  return el;
193 }

195 /*
196  Put pins back to a pinbox. Usually called via lf_alloc_put_pins() or
197  lf_hash_put_pins().
198 
199  DESCRIPTION
200  empty the purgatory (XXX deadlock warning below!),
201  push LF_PINS structure to a stack
202 */
203 void _lf_pinbox_put_pins(LF_PINS *pins)
204 {
205  LF_PINBOX *pinbox= pins->pinbox;
206  uint32 top_ver, nr;
207  nr= pins->link;
208 #ifdef MY_LF_EXTRA_DEBUG
209  {
210  int i;
211  for (i= 0; i < LF_PINBOX_PINS; i++)
212  DBUG_ASSERT(pins->pin[i] == 0);
213  }
214 #endif
215  /*
216  XXX this will deadlock if other threads will wait for
217  the caller to do something after _lf_pinbox_put_pins(),
218  and they would have pinned addresses that the caller wants to free.
219  Thus: only free pins when all work is done and nobody can wait for you!!!
220  */
221  while (pins->purgatory_count)
222  {
223  _lf_pinbox_real_free(pins);
224  if (pins->purgatory_count)
225  {
226  my_atomic_rwlock_wrunlock(&pins->pinbox->pinarray.lock);
227  pthread_yield();
228  my_atomic_rwlock_wrlock(&pins->pinbox->pinarray.lock);
229  }
230  }
231  top_ver= pinbox->pinstack_top_ver;
232  do
233  {
234  pins->link= top_ver % LF_PINBOX_MAX_PINS;
235  } while (!my_atomic_cas32((int32 volatile*) &pinbox->pinstack_top_ver,
236  (int32*) &top_ver,
237  top_ver-pins->link+nr+LF_PINBOX_MAX_PINS));
238  return;
239 }

274 /*
275  callback for _lf_dynarray_iterate:
276  scan all pins of all threads and accumulate all pins
277 */
278 static int harvest_pins(LF_PINS *el, struct st_harvester *hv)
279 {
280  int i;
281  LF_PINS *el_end= el+min(hv->npins, LF_DYNARRAY_LEVEL_LENGTH);
282  for (; el < el_end; el++)
283  {
284  for (i= 0; i < LF_PINBOX_PINS; i++)
285  {
286  void *p= el->pin[i];
287  if (p)
288  *hv->granary++= p;
289  }
290  }
291  /*
292  hv->npins may become negative below, but it means that
293  we're on the last dynarray page and harvest_pins() won't be
294  called again. We don't bother to make hv->npins() correct
295  (that is 0) in this case.
296  */
297  hv->npins-= LF_DYNARRAY_LEVEL_LENGTH;
298  return 0;
299 }

301 /*
302  callback for _lf_dynarray_iterate:
303  scan all pins of all threads and see if addr is present there
304 */
305 static int match_pins(LF_PINS *el, void *addr)
306 {
307  int i;
308  LF_PINS *el_end= el+LF_DYNARRAY_LEVEL_LENGTH;
309  for (; el < el_end; el++)
310  for (i= 0; i < LF_PINBOX_PINS; i++)
311  if (el->pin[i] == addr)
312  return 1;
313  return 0;

325 /*
326  Scan the purgatory and free everything that can be freed
327 */
328 static void _lf_pinbox_real_free(LF_PINS *pins)
329 {
330  int npins, alloca_size;
331  void *list, **addr;
332  void *first= NULL, *last= NULL;
333  LF_PINBOX *pinbox= pins->pinbox;
334 
335  npins= pinbox->pins_in_array+1;
336 
337 #ifdef HAVE_ALLOCA
338  alloca_size= sizeof(void *)*LF_PINBOX_PINS*npins;
339  /* create a sorted list of pinned addresses, to speed up searches */
340  if (available_stack_size(&pinbox, *pins->stack_ends_here) > alloca_size)
341  {
342  struct st_harvester hv;
343  addr= (void **) alloca(alloca_size);
344  hv.granary= addr;
345  hv.npins= npins;
346  /* scan the dynarray and accumulate all pinned addresses */
347  _lf_dynarray_iterate(&pinbox->pinarray,
348  (lf_dynarray_func)harvest_pins, &hv);
349 
350  npins= hv.granary-addr;
351  /* and sort them */
352  if (npins)
353  qsort(addr, npins, sizeof(void *), (qsort_cmp)ptr_cmp);
354  }
355  else
356 #endif
357  addr= 0;
358 
359  list= pins->purgatory;
360  pins->purgatory= 0;
361  pins->purgatory_count= 0;
362  while (list)
363  {
364  void *cur= list;
365  list= *(void **)((char *)cur+pinbox->free_ptr_offset);
366  if (npins)
367  {
368  if (addr) /* use binary search */
369  {
370  void **a, **b, **c;
371  for (a= addr, b= addr+npins-1, c= a+(b-a)/2; (b-a) > 1; c= a+(b-a)/2)
372  if (cur == *c)
373  a= b= c;
374  else if (cur > *c)
375  a= c;
376  else
377  b= c;
378  if (cur == *a || cur == *b)
379  goto found;
380  }
381  else /* no alloca - no cookie. linear search here */
382  {
383  if (_lf_dynarray_iterate(&pinbox->pinarray,
384  (lf_dynarray_func)match_pins, cur))
385  goto found;
386  }
387  }
388  /* not pinned - freeing */
389  if (last)
390  last= next_node(pinbox, last)= (uchar *)cur;
391  else
392  first= last= (uchar *)cur;
393  continue;
394 found:
395  /* pinned - keeping */
396  add_to_purgatory(pins, cur);
397  }
398  if (last)
399  pinbox->free_func(first, last, pinbox->free_func_arg);
400 }

406 /*
407  callback for _lf_pinbox_real_free to free a list of unpinned objects -
408  add it back to the allocator stack
409 
410  DESCRIPTION
411  'first' and 'last' are the ends of the linked list of nodes:
412  first->el->el->....->el->last. Use first==last to free only one element.
413 */
414 static void alloc_free(uchar *first,
415  uchar volatile *last,
416  LF_ALLOCATOR *allocator)
417 {
418  /*
419  we need a union here to access type-punned pointer reliably.
420  otherwise gcc -fstrict-aliasing will not see 'tmp' changed in the loop
421  */
422  union { uchar * node; void *ptr; } tmp;
423  tmp.node= allocator->top;
424  do
425  {
426  anext_node(last)= tmp.node;
427  } while (!my_atomic_casptr((void **)(char *)&allocator->top,
428  (void **)&tmp.ptr, first) && LF_BACKOFF);
429 }

454 /*
455  destroy the allocator, free everything that's in it
456 
457  NOTE
458  As every other init/destroy function here and elsewhere it
459  is not thread safe. No, this function is no different, ensure
460  that no thread needs the allocator before destroying it.
461  We are not responsible for any damage that may be caused by
462  accessing the allocator when it is being or has been destroyed.
463  Oh yes, and don't put your cat in a microwave.
464 */
465 void lf_alloc_destroy(LF_ALLOCATOR *allocator)
466 {
467  uchar *node= allocator->top;
468  while (node)
469  {
470  uchar *tmp= anext_node(node);
471  my_free(node);
472  node= tmp;
473  }
474  lf_pinbox_destroy(&allocator->pinbox);
475  allocator->top= 0;
