36 size_t my_read(File Filedes, uchar *Buffer, size_t Count, myf MyFlags)
37 {
38  size_t readbytes, save_count;
39  DBUG_ENTER("my_read");
40  DBUG_PRINT("my",("fd: %d Buffer: %p Count: %lu MyFlags: %d",
41  Filedes, Buffer, (ulong) Count, MyFlags));
42  save_count= Count;
43 
44  for (;;)
45  {
46  errno= 0; /* Linux, Windows don't reset this on EOF/success */
47 #ifdef _WIN32
48  readbytes= my_win_read(Filedes, Buffer, Count);
49 #else
50  readbytes= read(Filedes, Buffer, Count);
51 #endif
52 
53  if (readbytes != Count)
54  {
55  my_errno= errno;
56  if (errno == 0 || (readbytes != (size_t) -1 &&
57  (MyFlags & (MY_NABP | MY_FNABP))))
58  my_errno= HA_ERR_FILE_TOO_SHORT;
59  DBUG_PRINT("warning",("Read only %d bytes off %lu from %d, errno: %d",
60  (int) readbytes, (ulong) Count, Filedes,
61  my_errno));
62 
63  if ((readbytes == 0 || (int) readbytes == -1) && errno == EINTR)
64  {
65  DBUG_PRINT("debug", ("my_read() was interrupted and returned %ld",
66  (long) readbytes));
67  continue; /* Interrupted */
68  }
69 
70  if (MyFlags & (MY_WME | MY_FAE | MY_FNABP))
71  {
72  if (readbytes == (size_t) -1)
73  my_error(EE_READ, MYF(ME_BELL+ME_WAITTANG),
74  my_filename(Filedes),my_errno);
75  else if (MyFlags & (MY_NABP | MY_FNABP))
76  my_error(EE_EOFERR, MYF(ME_BELL+ME_WAITTANG),
77  my_filename(Filedes),my_errno);
78  }
79  if (readbytes == (size_t) -1 ||
80  ((MyFlags & (MY_FNABP | MY_NABP)) && !(MyFlags & MY_FULL_IO)))
81  DBUG_RETURN(MY_FILE_ERROR); /* Return with error */
82  if (readbytes != (size_t) -1 && (MyFlags & MY_FULL_IO))
83  {
84  Buffer+= readbytes;
85  Count-= readbytes;
86  continue;
87  }
88  }
89 
90  if (MyFlags & (MY_NABP | MY_FNABP))
91  readbytes= 0; /* Ok on read */
92  else if (MyFlags & MY_FULL_IO)
93  readbytes= save_count;
94  break;
95  }
96  DBUG_RETURN(readbytes);
