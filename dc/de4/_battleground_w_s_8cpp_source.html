51 void BattlegroundWS::PostUpdateImpl(uint32 diff)
52 {
53  if (GetStatus() == STATUS_IN_PROGRESS)
54  {
55  _bgEvents.Update(diff);
56  switch (_bgEvents.ExecuteEvent())
57  {
58  case BG_WS_EVENT_UPDATE_GAME_TIME:
59  UpdateWorldState(BG_WS_STATE_TIMER, GetMatchTime());
60  _bgEvents.ScheduleEvent(BG_WS_EVENT_UPDATE_GAME_TIME, ((BG_WS_TOTAL_GAME_TIME - GetStartTime()) % (MINUTE * IN_MILLISECONDS)) + 1);
61  break;
62  case BG_WS_EVENT_NO_TIME_LEFT:
63  if (GetTeamScore(TEAM_ALLIANCE) == GetTeamScore(TEAM_HORDE))
64  EndBattleground(_lastFlagCaptureTeam);
65  else
66  EndBattleground(GetTeamScore(TEAM_HORDE) > GetTeamScore(TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE);
67  break;
68  case BG_WS_EVENT_RESPAWN_BOTH_FLAGS:
69  SpawnBGObject(BG_WS_OBJECT_H_FLAG, RESPAWN_IMMEDIATELY);
70  SpawnBGObject(BG_WS_OBJECT_A_FLAG, RESPAWN_IMMEDIATELY);
71  SendMessageToAll(LANG_BG_WS_F_PLACED, CHAT_MSG_BG_SYSTEM_NEUTRAL);
72  PlaySoundToAll(BG_WS_SOUND_FLAGS_RESPAWNED);
73  break;
74  case BG_WS_EVENT_ALLIANCE_DROP_FLAG:
75  RespawnFlagAfterDrop(TEAM_ALLIANCE);
76  break;
77  case BG_WS_EVENT_HORDE_DROP_FLAG:
78  RespawnFlagAfterDrop(TEAM_HORDE);
79  break;
80  case BG_WS_EVENT_BOTH_FLAGS_KEPT10:
81  if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_ALLIANCE)))
82  player->CastSpell(player, BG_WS_SPELL_FOCUSED_ASSAULT, true);
83  if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_HORDE)))
84  player->CastSpell(player, BG_WS_SPELL_FOCUSED_ASSAULT, true);
85  break;
86  case BG_WS_EVENT_BOTH_FLAGS_KEPT15:
87  if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_ALLIANCE)))
88  {
89  player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
90  player->CastSpell(player, BG_WS_SPELL_BRUTAL_ASSAULT, true);
91  }
92  if (Player* player = ObjectAccessor::GetPlayer(FindBgMap(), GetFlagPickerGUID(TEAM_HORDE)))
93  {
94  player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
95  player->CastSpell(player, BG_WS_SPELL_BRUTAL_ASSAULT, true);
96  }
97  break;
98  }
99  }

386 void BattlegroundWS::HandleAreaTrigger(Player* player, uint32 trigger)
387 {
388  if (GetStatus() != STATUS_IN_PROGRESS || !player->IsAlive())
389  return;
390 
391  switch (trigger)
392  {
393  case BG_WS_TRIGGER_ALLIANCE_FLAG_SPAWN: // Alliance Flag spawn
394  if (GetFlagState(TEAM_ALLIANCE) == BG_WS_FLAG_STATE_ON_BASE && GetFlagPickerGUID(TEAM_HORDE) == player->GetGUID())
395  EventPlayerCapturedFlag(player);
396  break;
397  case BG_WS_TRIGGER_HORDE_FLAG_SPAWN: // Horde Flag spawn
398  if (GetFlagState(TEAM_HORDE) == BG_WS_FLAG_STATE_ON_BASE && GetFlagPickerGUID(TEAM_ALLIANCE) == player->GetGUID())
399  EventPlayerCapturedFlag(player);
400  break;
401  case 3649: // Not used
402  case 3688: // Not used
403  case 4628: // Not used
404  case 4629: // Not used
405  case BG_WS_TRIGGER_ALLIANCE_ELIXIR_SPEED_SPAWN: // Alliance elixir of speed spawn
406  case BG_WS_TRIGGER_HORDE_ELIXIR_SPEED_SPAWN: // Horde elixir of speed spawn
407  case BG_WS_TRIGGER_ALLIANCE_ELIXIR_REGEN_SPAWN: // Alliance elixir of regeneration spawn
408  case BG_WS_TRIGGER_HORDE_ELIXIR_REGEN_SPAWN: // Horde elixir of regeneration spawn
409  case BG_WS_TRIGGER_ALLIANCE_ELIXIR_BERSERK_SPAWN: // Alliance elixir of berserk spawn
410  case BG_WS_TRIGGER_HORDE_ELIXIR_BERSERK_SPAWN: // Horde elixir of berserk spawn
411  break;
412  }
