159  bool StaticMapTree::isInLineOfSight(const Vector3& pos1, const Vector3& pos2) const
160  {
161  float maxDist = (pos2 - pos1).magnitude();
162  // return false if distance is over max float, in case of cheater teleporting to the end of the universe
163  if (maxDist == std::numeric_limits<float>::max() || !std::isfinite(maxDist))
164  {
165  return false;
166  }
167 
168  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
169  ASSERT(maxDist < std::numeric_limits<float>::max());
170  // prevent NaN values which can cause BIH intersection to enter infinite loop
171  if (maxDist < 1e-10f)
172  {
173  return true;
174  }
175  // direction with length of 1
176  G3D::Ray ray = G3D::Ray::fromOriginAndDirection(pos1, (pos2 - pos1) / maxDist);
177 
178  if (GetIntersectionTime(ray, maxDist, true))
179  {
180 
181  return false;
182  }
183 
184  return true;
185  }

192  bool StaticMapTree::GetObjectHitPos(const Vector3& pPos1, const Vector3& pPos2, Vector3& pResultHitPos, float pModifyDist) const
193  {
194  bool result = false;
195  float maxDist = (pPos2 - pPos1).magnitude();
196  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
197  ASSERT(maxDist < std::numeric_limits<float>::max());
198  // prevent NaN values which can cause BIH intersection to enter infinite loop
199  if (maxDist < 1e-10f)
200  {
201  pResultHitPos = pPos2;
202  return false;
203  }
204  Vector3 dir = (pPos2 - pPos1) / maxDist; // direction with length of 1
205  G3D::Ray ray(pPos1, dir);
206  float dist = maxDist;
207  if (GetIntersectionTime(ray, dist, false))
208  {
209  pResultHitPos = pPos1 + dir * dist;
210  if (pModifyDist < 0)
211  {
212  if ((pResultHitPos - pPos1).magnitude() > -pModifyDist)
213  {
214  pResultHitPos = pResultHitPos + dir * pModifyDist;
215  }
216  else
217  {
218  pResultHitPos = pPos1;
219  }
220  }
221  else
222  {
223  pResultHitPos = pResultHitPos + dir * pModifyDist;
224  }
225  result = true;
226  }
227  else
228  {
229  pResultHitPos = pPos2;
230  result = false;
231  }
232  return result;
233  }

366  bool StaticMapTree::LoadMapTile(uint32 tileX, uint32 tileY, VMapMgr2* vm)
367  {
368  if (!iIsTiled)
369  {
370  // currently, core creates grids for all maps, whether it has terrain tiles or not
371  // so we need "fake" tile loads to know when we can unload map geometry
372  iLoadedTiles[packTileID(tileX, tileY)] = false;
373  return true;
374  }
375  if (!iTreeValues)
376  {
377  LOG_ERROR("maps", "StaticMapTree::LoadMapTile() : tree has not been initialized [%u, %u]", tileX, tileY);
378  return false;
379  }
380  bool result = true;
381 
382  std::string tilefile = iBasePath + getTileFileName(iMapID, tileX, tileY);
383  FILE* tf = fopen(tilefile.c_str(), "rb");
384  if (tf)
385  {
386  char chunk[8];
387 
388  if (!readChunk(tf, chunk, VMAP_MAGIC, 8))
389  {
390  result = false;
391  }
392  uint32 numSpawns = 0;
393  if (result && fread(&numSpawns, sizeof(uint32), 1, tf) != 1)
394  {
395  result = false;
396  }
397  for (uint32 i = 0; i < numSpawns && result; ++i)
398  {
399  // read model spawns
400  ModelSpawn spawn;
401  result = ModelSpawn::readFromFile(tf, spawn);
402  if (result)
403  {
404  // acquire model instance
405  WorldModel* model = vm->acquireModelInstance(iBasePath, spawn.name);
406  if (!model)
407  {
408  LOG_ERROR("maps", "StaticMapTree::LoadMapTile() : could not acquire WorldModel pointer [%u, %u]", tileX, tileY);
409  }
410 
411  // update tree
412  uint32 referencedVal;
413 
414  if (fread(&referencedVal, sizeof(uint32), 1, tf) == 1)
415  {
416  if (!iLoadedSpawns.count(referencedVal))
417  {
418 #if defined(VMAP_DEBUG)
419  if (referencedVal > iNTreeValues)
420  {
421  LOG_DEBUG("maps", "StaticMapTree::LoadMapTile() : invalid tree element (%u/%u)", referencedVal, iNTreeValues);
422  continue;
423  }
424 #endif
425  iTreeValues[referencedVal] = ModelInstance(spawn, model);
426  iLoadedSpawns[referencedVal] = 1;
427  }
428  else
429  {
430  ++iLoadedSpawns[referencedVal];
431 #if defined(VMAP_DEBUG)
432  if (iTreeValues[referencedVal].ID != spawn.ID)
433  {
434  LOG_DEBUG("maps", "StaticMapTree::LoadMapTile() : trying to load wrong spawn in node");
435  }
436  else if (iTreeValues[referencedVal].name != spawn.name)
437  {
438  LOG_DEBUG("maps", "StaticMapTree::LoadMapTile() : name collision on GUID=%u", spawn.ID);
439  }
440 #endif
441  }
442  }
443  else
444  {
445  result = false;
446  }
447  }
448  }
449  iLoadedTiles[packTileID(tileX, tileY)] = true;
450  fclose(tf);
451  }
452  else
453  {
454  iLoadedTiles[packTileID(tileX, tileY)] = false;
455  }
456 
457  METRIC_EVENT("map_events", "LoadMapTile",
458  "Map: " + std::to_string(iMapID) + " TileX: " + std::to_string(tileX) + " TileY: " + std::to_string(tileY));
459 
460  return result;
461  }

496  if (result)
