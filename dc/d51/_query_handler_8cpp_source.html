228 
229  WorldLocation corpseLocation = _player->GetCorpseLocation();

234  float z = corpseLocation.GetPositionZ();

236  // if corpse at different map

248  x = corpseMapEntry->entrance_x;
249  y = corpseMapEntry->entrance_y;

251  }
252  }
253  }
254  }
255 
256  WorldPacket data(MSG_CORPSE_QUERY, 1 + (6 * 4));
257  data << uint8(1); // corpse found

259  data << float(x);

409 void WorldSession::HandleQuestPOIQuery(WorldPacket& recvData)
410 {
411  uint32 count;
412  recvData >> count; // quest count, max=25
413 
414  if (count > MAX_QUEST_LOG_SIZE)
415  {
416  recvData.rfinish();
417  return;
418  }
419 
420  // Read quest ids and add the in a unordered_set so we don't send POIs for the same quest multiple times
421  std::unordered_set<uint32> questIds;
422  for (uint32 i = 0; i < count; ++i)
423  questIds.insert(recvData.read<uint32>()); // quest id
424 
425  WorldPacket data(SMSG_QUEST_POI_QUERY_RESPONSE, 4 + (4 + 4)*questIds.size());
426  data << uint32(questIds.size()); // count
427 
428  for (std::unordered_set<uint32>::const_iterator itr = questIds.begin(); itr != questIds.end(); ++itr)
429  {
430  uint32 questId = *itr;
431  bool questOk = false;
432 
433  uint16 questSlot = _player->FindQuestSlot(questId);
434 
435  if (questSlot != MAX_QUEST_LOG_SIZE)
436  questOk = _player->GetQuestSlotQuestId(questSlot) == questId;
437 
438  if (questOk)
439  {
440  QuestPOIVector const* POI = sObjectMgr->GetQuestPOIVector(questId);
441 
442  if (POI)
443  {
444  data << uint32(questId); // quest ID
445  data << uint32(POI->size()); // POI count
446 
447  for (QuestPOIVector::const_iterator itr = POI->begin(); itr != POI->end(); ++itr)
448  {
449  data << uint32(itr->Id); // POI index
450  data << int32(itr->ObjectiveIndex); // objective index
451  data << uint32(itr->MapId); // mapid
452  data << uint32(itr->AreaId); // areaid
453  data << uint32(itr->FloorId); // floorid
454  data << uint32(itr->Unk3); // unknown
455  data << uint32(itr->Unk4); // unknown
456  data << uint32(itr->points.size()); // POI points count
457 
458  for (std::vector<QuestPOIPoint>::const_iterator itr2 = itr->points.begin(); itr2 != itr->points.end(); ++itr2)
459  {
460  data << int32(itr2->x); // POI point x
461  data << int32(itr2->y); // POI point y
462  }
463  }
464  }
465  else
466  {
467  data << uint32(questId); // quest ID
468  data << uint32(0); // POI count
469  }
470  }
471  else
472  {
473  data << uint32(questId); // quest ID
474  data << uint32(0); // POI count
475  }
476  }
477 
478  SendPacket(&data);
479 }
