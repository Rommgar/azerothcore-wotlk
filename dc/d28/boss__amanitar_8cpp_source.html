156  void ExecuteEvent(uint32 eventId) override
157  {
158  switch (eventId)
159  {
160  case EVENT_RESPAWN:
161  {
162  while (!_mushroomsDeque.empty())
163  {
164  SummonMushroom(_mushroomsDeque.front());
165  _mushroomsDeque.pop_front();
166  }
167 
168  events.RepeatEvent(urand(40000, 60000));
169  break;
170  }
171  case EVENT_ROOTS:
172  {
173  if (Unit* pTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
174  {
175  DoCast(pTarget, SPELL_ENTANGLING_ROOTS, false);
176  }
177 
178  events.RepeatEvent(urand(10000, 15000));
179  break;
180  }
181  case EVENT_BASH:
182  {
183  DoCastVictim(SPELL_BASH, false);
184  events.RepeatEvent(urand(15000, 20000));
185  break;
186  }
187  case EVENT_BOLT:
188  {
189  if (Unit* pTarget = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
190  {
191  DoCast(pTarget, SPELL_VENOM_BOLT_VOLLEY, false);
192  }
193 
194  events.RepeatEvent(urand(15000, 20000));
195  break;
196  }
197  case EVENT_REMOVE_MUSHROOM_POWER:
198  {
199  DoCastAOE(SPELL_REMOVE_MUSHROOM_POWER, true);
200  events.RescheduleEvent(EVENT_MINI, 1000);
201  break;
202  }
203  case EVENT_MINI:
204  {
205  if (!mushroomsSummoned)
206  {
207  mushroomsSummoned = true;
208  for (uint8 i = 0; i < MAX_MUSHROOMS_COUNT; ++i)
209  {
210  SummonMushroom(MushroomPositions[i]);
211  }
212  }
213 
214  if (SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true, -SPELL_MINI))
215  {
216  DoCastSelf(SPELL_REMOVE_MUSHROOM_POWER, true);
217  DoCastAOE(SPELL_MINI);
218  events.RescheduleEvent(EVENT_REMOVE_MUSHROOM_POWER, 29000);
219  }
220  else
221  {
222  events.RepeatEvent(1000);
223  }
224 
225  break;
226  }
227  }
228  }
229 
230  private:
231  std::deque<Position> _mushroomsDeque;
232  bool mushroomsSummoned;
233

236  me->SummonCreature(roll_chance_i(40) ? NPC_HEALTHY_MUSHROOM : NPC_POISONOUS_MUSHROOM, pos, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 4000);
237  }
238  };

298  void UpdateAI(uint32 diff) override
299  {
300  if (events.Empty())
301  return;
302 
303  events.Update(diff);
304  while (uint32 const eventId = events.ExecuteEvent())
305  {
306  switch (eventId)
307  {
308  case EVENT_GROW:
309  {
310  DoCastSelf(SPELL_GROW);
311  break;
312  }
313  case EVENT_CHECK_PLAYER:
314  {
315  if (Player* plr = me->SelectNearestPlayer(2.0f))
316  {
317  plr->RemoveAurasDueToSpell(SPELL_HEALTHY_MUSHROOM_POTENT_FUNGUS);
318  DoCastSelf(SPELL_POISONOUS_MUSHROOM_VISUAL_AREA);
319  DoCastSelf(SPELL_POISONOUS_MUSHROOM_POISON_CLOUD);
320  DoCastSelf(SPELL_SHRINK);
321  events.ScheduleEvent(EVENT_KILLSELF, 4000);
322  }
323  else
324  {
325  events.RepeatEvent(250);
326  }
327 
328  break;
329  }
330  case EVENT_KILLSELF:
331  {
332  me->DisappearAndDie();
333  break;
334  }
335  }
336  }
337  }
338 
339  private:
340  EventMap events;
341  };
