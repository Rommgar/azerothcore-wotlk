64 void WorldModelRoot::ReadDoodadInstances()
65 {
66  Chunk* chunk = Data->GetChunkByName("MODD");
67  Chunk* nameChunk = Data->GetChunkByName("MODN");
68  if (!chunk || !nameChunk)
69  return;
70 
71  const uint32 instanceSize = 40;
72  uint32 countInstances = chunk->Length / instanceSize;
73  DoodadInstances.reserve(countInstances);
74  for (uint32 i = 0; i < countInstances; i++)
75  {
76  FILE* stream = chunk->GetStream();
77  fseek(stream, instanceSize * i, SEEK_CUR);
78  DoodadInstance instance = DoodadInstance::Read(stream);
79  FILE* nameStream = nameChunk->GetStream();
80  if (instance.FileOffset >= nameChunk->Length)
81  continue;
82  fseek(nameStream, instance.FileOffset, SEEK_CUR);
83  instance.File = Utils::ReadString(nameStream);
84  DoodadInstances.push_back(instance);
85  }
86 }
