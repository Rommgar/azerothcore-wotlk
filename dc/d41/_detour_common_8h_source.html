452 inline unsigned int dtIlog2(unsigned int v)
453 {
454  unsigned int r;
455  unsigned int shift;
456  r = (v > 0xffff) << 4; v >>= r;
457  shift = (v > 0xff) << 3; v >>= shift; r |= shift;
458  shift = (v > 0xf) << 2; v >>= shift; r |= shift;
459  shift = (v > 0x3) << 1; v >>= shift; r |= shift;
460  r |= (v >> 1);
461  return r;
462 }
