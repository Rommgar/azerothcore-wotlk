27 template <typename Container, typename T = typename std::decay<decltype(*std::begin(std::declval<Container>()))>::type>
28 inline T standard_deviation(Container&& c)
29 {
30  auto b = std::begin(c), e = std::end(c);
31  auto size = std::distance(b, e);
32  auto sum = std::accumulate(b, e, T());
33  auto mean = sum / size;
34 
35  if (size == 1)
36  {
37  return (T) 0;
38  }
39 
40  T accum = T();
41  for (const auto d : c)
42  {
43  accum += (d - mean) * (d - mean);
44  }
45  return std::sqrt(accum / (size - 1));
46 }
