146  void UpdateAI(uint32 diff) override
147  {
148  dropSludgeTimer += diff;
149  if (!me->IsInCombat() && dropSludgeTimer >= 5000)
150  {
151  if (me->IsWithinDist3d(3178, -3305, 319, 5.0f) && !summons.HasEntry(NPC_SEWAGE_SLIME))
152  {
153  me->CastSpell(3128.96f + irand(-20, 20), -3312.96f + irand(-20, 20), 293.25f, SPELL_BOMBARD_SLIME, false);
154  }
155  dropSludgeTimer = 0;
156  }
157 
158  if (!UpdateVictim())
159  return;
160 
161  events.Update(diff);
162  if (me->HasUnitState(UNIT_STATE_CASTING))
163  return;
164 
165  switch (events.ExecuteEvent())
166  {
167  case EVENT_POISON_CLOUD:
168  me->CastSpell(me, SPELL_POISON_CLOUD, true);
169  events.RepeatEvent(15000);
170  break;
171  case EVENT_BERSERK:
172  me->CastSpell(me, SPELL_BERSERK, true);
173  break;
174  case EVENT_SLIME_SPRAY:
175  Talk(EMOTE_SLIME);
176  me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_SLIME_SPRAY_10, SPELL_SLIME_SPRAY_25), false);
177  events.RepeatEvent(20000);
178  break;
179  case EVENT_MUTATING_INJECTION:
180  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 100.0f, true, -SPELL_MUTATING_INJECTION))
181  {
182  me->CastSpell(target, SPELL_MUTATING_INJECTION, false);
183  }
184  events.RepeatEvent(6000 + uint32(120 * me->GetHealthPct()));
185  break;
186  }
187  DoMeleeAttackIfReady();
188  }
189  };
