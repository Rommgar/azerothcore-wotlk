148 /*
149  A hack around a race condition in the implementation of freopen.
150 
151  The race condition steams from the fact that the current fd of
152  the stream is closed before its number is used to duplicate the
153  new file descriptor. This defeats the desired atomicity of the
154  close and duplicate of dup2().
155 
156  See PR number 79887 for reference:
157  http://www.freebsd.org/cgi/query-pr.cgi?pr=79887
158 */
159 
160 static FILE *my_freebsd_freopen(const char *path, const char *mode, FILE *stream)
161 {
162  int old_fd;
163  FILE *result;
164 
165  flockfile(stream);
166 
167  old_fd= fileno(stream);
168 
169  /* Use a no operation close hook to avoid having the fd closed. */
170  stream->_close= no_close;
171 
172  /* Relies on the implicit dup2 to close old_fd. */
173  result= freopen(path, mode, stream);
174 
175  /* If successful, the _close hook was replaced. */
176 
177  if (result == NULL)
178  close(old_fd);
179  else
180  funlockfile(result);
181 
182  return result;
183 }

306 /*
307  Make a fopen() typestring from a open() type bitmap
308 
309  SYNOPSIS
310  make_ftype()
311  to String for fopen() is stored here
312  flag Flag used by open()
313 
314  IMPLEMENTATION
315  This routine attempts to find the best possible match
316  between a numeric option and a string option that could be
317  fed to fopen. There is not a 1 to 1 mapping between the two.
318 
319  NOTE
320  On Unix, O_RDONLY is usually 0
321 
322  MAPPING
323  r == O_RDONLY
324  w == O_WRONLY|O_TRUNC|O_CREAT
325  a == O_WRONLY|O_APPEND|O_CREAT
326  r+ == O_RDWR
327  w+ == O_RDWR|O_TRUNC|O_CREAT
328  a+ == O_RDWR|O_APPEND|O_CREAT
329 */
330 
331 static void make_ftype(register char * to, register int flag)
332 {
333  /* check some possible invalid combinations */
334  DBUG_ASSERT((flag & (O_TRUNC | O_APPEND)) != (O_TRUNC | O_APPEND));
335  DBUG_ASSERT((flag & (O_WRONLY | O_RDWR)) != (O_WRONLY | O_RDWR));
336 
337  if ((flag & (O_RDONLY|O_WRONLY)) == O_WRONLY)
338  *to++= (flag & O_APPEND) ? 'a' : 'w';
339  else if (flag & O_RDWR)
340  {
341  /* Add '+' after theese */
342  if (flag & (O_TRUNC | O_CREAT))
343  *to++= 'w';
344  else if (flag & O_APPEND)
345  *to++= 'a';
346  else
347  *to++= 'r';
348  *to++= '+';
349  }
350  else
351  *to++= 'r';
352 
353 #if FILE_BINARY /* If we have binary-files */
354  if (flag & FILE_BINARY)
355  *to++='b';
356 #endif
357  *to='\0';
358 } /* make_ftype */
