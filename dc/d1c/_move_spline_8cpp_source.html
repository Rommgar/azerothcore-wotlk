26  Location MoveSpline::ComputePosition() const
27  {
28  ASSERT(Initialized());
29 
30  float u = 1.f;
31  int32 seg_time = spline.length(point_Idx, point_Idx + 1);
32  if (seg_time > 0)
33  u = (time_passed - spline.length(point_Idx)) / (float)seg_time;
34  Location c;
35  c.orientation = initialOrientation;
36  spline.evaluate_percent(point_Idx, u, c);
37 
38  if (splineflags.animation)
39  ;// MoveSplineFlag::Animation disables falling or parabolic movement
40  else if (splineflags.parabolic)
41  computeParabolicElevation(c.z);
42  else if (splineflags.falling)
43  computeFallElevation(c.z);
44 
45  if (splineflags.done && splineflags.isFacing())
46  {
47  if (splineflags.final_angle)
48  c.orientation = facing.angle;
49  else if (splineflags.final_point)
50  c.orientation = atan2(facing.f.y - c.y, facing.f.x - c.x);
51  //nothing to do for MoveSplineFlag::Final_Target flag
52  }
53  else
54  {
55  if (!splineflags.hasFlag(MoveSplineFlag::OrientationFixed | MoveSplineFlag::Falling))
56  {
57  Vector3 hermite;
58  spline.evaluate_derivative(point_Idx, u, hermite);
59  c.orientation = atan2(hermite.y, hermite.x);
60  }
61 
62  if (splineflags.orientationInversed)
63  c.orientation = -c.orientation;
64  }
65  return c;
66  }

68  void MoveSpline::computeParabolicElevation(float& el) const
69  {
70  if (time_passed > effect_start_time)
71  {
72  float t_passedf = MSToSec(time_passed - effect_start_time);
73  float t_durationf = MSToSec(Duration() - effect_start_time); //client use not modified duration here
74 
75  // -a*x*x + bx + c:
76  //(dur * v3->z_acceleration * dt)/2 - (v3->z_acceleration * dt * dt)/2 + Z;
77  el += (t_durationf - t_passedf) * 0.5f * vertical_acceleration * t_passedf;
78  }
79  }

156  void MoveSpline::Initialize(MoveSplineInitArgs const& args)
157  {
158  splineflags = args.flags;

162  initialOrientation = args.initialOrientation;

164  time_passed = 0;
165  vertical_acceleration = 0.f;
166  effect_start_time = 0;
167 
168  // Check if its a stop spline
169  if (args.flags.done)
170  {
171  spline.clear();
172  return;
173  }
174 
175  init_spline(args);
176 
177  // init parabolic / animation
178  // spline initialized, duration known and i able to compute parabolic acceleration
179  if (args.flags & (MoveSplineFlag::Parabolic | MoveSplineFlag::Animation))
180  {
181  effect_start_time = Duration() * args.time_perc;
182  if (args.flags.parabolic && effect_start_time < Duration())
183  {
184  float f_duration = MSToSec(Duration() - effect_start_time);
185  vertical_acceleration = args.parabolic_amplitude * 8.f / (f_duration * f_duration);
186  }
187  }

218  // MONSTER_MOVE packet format limitation for not CatmullRom movement:
219  // each vertex offset packed into 11 bytes
220  bool MoveSplineInitArgs::_checkPathBounds() const
221  {
222  if (!(flags & MoveSplineFlag::Mask_CatmullRom) && path.size() > 2)
223  {
224  constexpr auto MAX_OFFSET = (1 << 11) / 2;
225  Vector3 middle = (path.front() + path.back()) / 2;
226  Vector3 offset;
227  for (uint32 i = 1; i < path.size() - 1; ++i)
228  {
229  offset = path[i] - middle;
230  if (fabs(offset.x) >= MAX_OFFSET || fabs(offset.y) >= MAX_OFFSET || fabs(offset.z) >= MAX_OFFSET)
231  {
232  LOG_ERROR("movement", "MoveSplineInitArgs::_checkPathBounds check failed");
233  return false;
234  }
235  }
236  }
237  return true;

242  MoveSpline::UpdateResult MoveSpline::_updateState(int32& ms_time_diff)
243  {
244  if (Finalized())
245  {
246  ms_time_diff = 0;
247  return Result_Arrived;
248  }
249 
250  UpdateResult result = Result_None;
251 
252  int32 minimal_diff = std::min(ms_time_diff, segment_time_elapsed());
253  if (minimal_diff < 0)
254  minimal_diff = 0;
255 
256  ASSERT(minimal_diff >= 0);
257  time_passed += minimal_diff;
258  ms_time_diff -= minimal_diff;
259 
260  if (time_passed >= next_timestamp())
261  {
262  ++point_Idx;
263  if (point_Idx < spline.last())
264  {
265  result = Result_NextSegment;
266  }
267  else
268  {
269  if (spline.isCyclic())
270  {
271  point_Idx = spline.first();
272  time_passed = time_passed % Duration();
273  result = Movement::MoveSpline::UpdateResult(Result_NextCycle | Result_JustArrived);
274  }
275  else
276  {
277  _Finalize();
278  ms_time_diff = 0;
279  result = Movement::MoveSpline::UpdateResult(Result_Arrived | Result_JustArrived);
280  }
281  }
282  }
283 
284  return result;
285  }
