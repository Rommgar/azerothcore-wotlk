226  void UpdateAI(uint32 diff) override
227  {
228  if (!UpdateVictim())
229  {
230  checkTimer += diff;
231  if (checkTimer >= 5000)
232  {
233  checkTimer = 0;
234  CheckChannelers();
235  if (!me->HasUnitState(UNIT_STATE_CASTING))
236  me->CastSpell(me, SPELL_EVOCATION, false);
237  }
238  return;
239  }
240 
241  events.Update(diff);
242  if (me->HasUnitState(UNIT_STATE_CASTING))
243  return;
244 
245  switch (events.ExecuteEvent())
246  {
247  case EVENT_SPELL_VOLLEY:
248  me->CastSpell(me, SPELL_SHADOW_BOLT_VOLLEY, false);
249  events.RepeatEvent(urand(8000, 13000));
250  break;
251  case EVENT_SPELL_CORRUPTION:
252  me->CastSpell(me, SPELL_CORRUPTION, false);
253  events.RepeatEvent(urand(30000, 50000));
254  break;
255  case EVENT_SPELL_BURNING_NOVA:
256  Talk(SAY_NOVA);
257 
258  ApplyImmunities(false);
259  me->AddAura(SPELL_BURNING_NOVA, me);
260  ApplyImmunities(true);
261 
262  if (IsHeroic())
263  DoTeleportAll(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), me->GetOrientation());
264 
265  events.DelayEvents(6000, 0);
266  events.RepeatEvent(urand(25000, 32000));
267  events.ScheduleEvent(EVENT_SPELL_FIRE_NOVA, 5000);
268  break;
269  case EVENT_SPELL_FIRE_NOVA:
270  me->CastSpell(me, SPELL_FIRE_NOVA, true);
271  break;
272  }
273 
274  DoMeleeAttackIfReady();
275  }
