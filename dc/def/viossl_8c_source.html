89 int vio_ssl_close(Vio *vio)
90 {
91  int r= 0;
92  SSL *ssl= (SSL*)vio->ssl_arg;
93  DBUG_ENTER("vio_ssl_close");
94 
95  if (ssl)
96  {
97  /*
98  THE SSL standard says that SSL sockets must send and receive a close_notify
99  alert on socket shutdown to avoid truncation attacks. However, this can
100  cause problems since we often hold a lock during shutdown and this IO can
101  take an unbounded amount of time to complete. Since our packets are self
102  describing with length, we aren't vunerable to these attacks. Therefore,
103  we just shutdown by closing the socket (quiet shutdown).
104  */
105  SSL_set_quiet_shutdown(ssl, 1);
106 
107  switch ((r= SSL_shutdown(ssl))) {
108  case 1:
109  /* Shutdown successful */
110  break;
111  case 0:
112  /*
113  Shutdown not yet finished - since the socket is going to
114  be closed there is no need to call SSL_shutdown() a second
115  time to wait for the other side to respond
116  */
117  break;
118  default: /* Shutdown failed */
119  DBUG_PRINT("vio_error", ("SSL_shutdown() failed, error: %d",
120  SSL_get_error(ssl, r)));
121  break;
122  }
123  }
124  DBUG_RETURN(vio_close(vio));
125 }

146 static int ssl_do(struct st_VioSSLFd *ptr, Vio *vio, long timeout,
147  int (*connect_accept_func)(SSL*))
148 {
149  SSL *ssl;
150  my_bool unused;
151  my_bool was_blocking;
152 
153  DBUG_ENTER("ssl_do");
154  DBUG_PRINT("enter", ("ptr: 0x%lx, sd: %d ctx: 0x%lx",
155  (long) ptr, vio->sd, (long) ptr->ssl_context));
156 
157  /* Set socket to blocking if not already set */
158  vio_blocking(vio, 1, &was_blocking);
159 
160  if (!(ssl= SSL_new(ptr->ssl_context)))
161  {
162  DBUG_PRINT("error", ("SSL_new failure"));
163  report_errors(ssl);
164  vio_blocking(vio, was_blocking, &unused);
165  DBUG_RETURN(1);
166  }
167  DBUG_PRINT("info", ("ssl: 0x%lx timeout: %ld", (long) ssl, timeout));
168  SSL_clear(ssl);
169  SSL_SESSION_set_timeout(SSL_get_session(ssl), timeout);
170  SSL_set_fd(ssl, vio->sd);
171 
172  if (connect_accept_func(ssl) < 1)
173  {
174  DBUG_PRINT("error", ("SSL_connect/accept failure"));
175  report_errors(ssl);
176  SSL_free(ssl);
177  vio_blocking(vio, was_blocking, &unused);
178  DBUG_RETURN(1);
179  }
180 
181  /*
182  Connection succeeded. Install new function handlers,
183  change type, set sd to the fd used when connecting
184  and set pointer to the SSL structure
185  */
186  vio_reset(vio, VIO_TYPE_SSL, SSL_get_fd(ssl), 0, 0);
187  vio->ssl_arg= (void*)ssl;
188 
189 #ifndef DBUG_OFF
190  {
191  /* Print some info about the peer */
192  X509 *cert;
193  char buf[512];
194 
195  DBUG_PRINT("info",("SSL connection succeeded"));
196  DBUG_PRINT("info",("Using cipher: '%s'" , SSL_get_cipher_name(ssl)));
197 
198  if ((cert= SSL_get_peer_certificate (ssl)))
199  {
200  DBUG_PRINT("info",("Peer certificate:"));
201  X509_NAME_oneline(X509_get_subject_name(cert), buf, sizeof(buf));
202  DBUG_PRINT("info",("\t subject: '%s'", buf));
203  X509_NAME_oneline(X509_get_issuer_name(cert), buf, sizeof(buf));
204  DBUG_PRINT("info",("\t issuer: '%s'", buf));
205  X509_free(cert);
206  }
207  else
208  DBUG_PRINT("info",("Peer does not have certificate."));
209 
210  if (SSL_get_shared_ciphers(ssl, buf, sizeof(buf)))
211  {
212  DBUG_PRINT("info",("shared_ciphers: '%s'", buf));
213  }
214  else
215  DBUG_PRINT("info",("no shared ciphers!"));
216  }
217 #endif
218 
219  DBUG_RETURN(0);
220 }
