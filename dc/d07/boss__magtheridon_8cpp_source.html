147  void UpdateAI(uint32 diff) override
148  {
149  events2.Update(diff);
150  switch (events2.ExecuteEvent())
151  {
152  case EVENT_RANDOM_TAUNT:
153  Talk(SAY_TAUNT);
154  events2.ScheduleEvent(EVENT_RANDOM_TAUNT, 90000);
155  break;
156  case EVENT_CHECK_GRASP:
157  if (me->GetAuraCount(SPELL_SHADOW_GRASP_VISUAL) == 5)
158  {
159  Talk(SAY_BANISH);
160  me->InterruptNonMeleeSpells(true);
161  break;
162  }
163  events2.ScheduleEvent(EVENT_CHECK_GRASP, 0);
164  break;
165  }
166 
167  if (!UpdateVictim() || !CheckInRoom())
168  return;
169 
170  events.Update(diff);
171  if (me->HasUnitState(UNIT_STATE_CASTING))
172  return;
173 
174  switch (events.ExecuteEvent())
175  {
176  case EVENT_EMOTE1:
177  Talk(SAY_EMOTE_BEGIN);
178  break;
179  case EVENT_EMOTE2:
180  Talk(SAY_EMOTE_NEARLY);
181  break;
182  case EVENT_EMOTE3:
183  Talk(SAY_EMOTE_FREE);
184  Talk(SAY_FREE);
185  break;
186  case EVENT_ENTER_COMBAT:
187  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_IMMUNE_TO_PC);
188  me->SetReactState(REACT_AGGRESSIVE);
189  events.ScheduleEvent(EVENT_CLEAVE, 9000);
190  events.ScheduleEvent(EVENT_BLAST_NOVA, 60000);
191  events.ScheduleEvent(EVENT_BLAZE, 10000);
192  events.ScheduleEvent(EVENT_QUAKE, 40000);
193  events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
194  events.ScheduleEvent(EVENT_ENRAGE, 22 * MINUTE * IN_MILLISECONDS);
195 
196  instance->SetData(DATA_ACTIVATE_CUBES, 1);
197  me->RemoveAurasDueToSpell(SPELL_SHADOW_CAGE);
198  break;
199  case EVENT_CLEAVE:
200  me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
201  events.ScheduleEvent(EVENT_CLEAVE, 10000);
202  break;
203  case EVENT_BLAST_NOVA:
204  me->CastSpell(me, SPELL_BLAST_NOVA, false);
205  events.ScheduleEvent(EVENT_BLAST_NOVA, 60000);
206  events.ScheduleEvent(EVENT_CANCEL_GRASP_CHECK, 12000);
207  events2.ScheduleEvent(EVENT_CHECK_GRASP, 0);
208  break;
209  case EVENT_BLAZE:
210  me->CastCustomSpell(SPELL_BLAZE, SPELLVALUE_MAX_TARGETS, 1);
211  events.ScheduleEvent(EVENT_BLAZE, 30000);
212  break;
213  case EVENT_ENRAGE:
214  me->CastSpell(me, SPELL_BERSERK, true);
215  break;
216  case EVENT_CANCEL_GRASP_CHECK:
217  events2.Reset();
218  break;
219  case EVENT_QUAKE:
220  me->CastSpell(me, SPELL_QUAKE, false);
221  events.ScheduleEvent(EVENT_QUAKE, 50000);
222  break;
223  case EVENT_CHECK_HEALTH:
224  if (me->HealthBelowPct(30))
225  {
226  Talk(SAY_PHASE3);
227  events.SetPhase(1);
228  events.DelayEvents(18000);
229  events.ScheduleEvent(EVENT_COLLAPSE_CEIL, 8000);
230  events.ScheduleEvent(EVENT_COLLAPSE_DAMAGE, 15000);
231  break;
232  }
233  events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
234  break;
235  case EVENT_COLLAPSE_CEIL:
236  me->CastSpell(me, SPELL_CAMERA_SHAKE, true);
237  instance->SetData(DATA_COLLAPSE, GO_STATE_ACTIVE);
238  break;
239  case EVENT_COLLAPSE_DAMAGE:
240  me->CastSpell(me, SPELL_COLLAPSE_DAMAGE, true);
241  me->resetAttackTimer();
242  events.SetPhase(0);
243  events.ScheduleEvent(EVENT_DEBRIS, 20000);
244  break;
245  case EVENT_DEBRIS:
246  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
247  {
248  target->CastSpell(target, SPELL_DEBRIS_VISUAL, true, nullptr, nullptr, me->GetGUID());
249  me->m_Events.AddEvent(new DealDebrisDamage(*me, target->GetGUID()), me->m_Events.CalculateTime(5000));
250  }
251  events.ScheduleEvent(EVENT_DEBRIS, 20000);
252  break;
253  }
254 
255  if (!events.IsInPhase(1))
256  DoMeleeAttackIfReady();
257  }
