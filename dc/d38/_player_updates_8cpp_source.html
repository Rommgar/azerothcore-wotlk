837 float getProbabilityOfLevelUp(uint32 SkillValue)
838 {
839  /* According to El's Extreme Angling page, from 1 to 115 the probability of
840  * a skill level up is 100% since 100/1 = 100. From 115 - 135 should average
841  * 2 catches per skill up so that means 100/2 = 50%. This returns the
842  * probability depending on the player's SkillValue.
843  */
844  if (!SkillValue)
845  {
846  return 0.0f;
847  }
848 
849  std::array<uint32, 10> bounds {115, 135, 160, 190, 215,
850  295, 315, 355, 425, 450};
851  std::array<float, 11> dens {1.0f, 2.0f, 3.0f, 4.0f, 5.0f, 6.0f,
852  9.0f, 10.0f, 11.0f, 12.0f, 1.0f};
853  auto it =
854  std::lower_bound(std::begin(bounds), std::end(bounds), SkillValue);
855  return 100 / dens[std::distance(std::begin(bounds), it)];
856 }
