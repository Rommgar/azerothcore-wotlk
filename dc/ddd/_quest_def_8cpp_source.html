187 uint32 Quest::XPValue(Player* player) const
188 {
189  if (player)
190  {
191  int32 quest_level = (Level == -1 ? player->getLevel() : Level);
192  const QuestXPEntry* xpentry = sQuestXPStore.LookupEntry(quest_level);
193  if (!xpentry)
194  return 0;
195 
196  int32 diffFactor = 2 * (quest_level - player->getLevel()) + 20;
197  if (diffFactor < 1)
198  diffFactor = 1;
199  else if (diffFactor > 10)
200  diffFactor = 10;
201 
202  uint32 xp = diffFactor * xpentry->Exp[RewardXPDifficulty] / 10;
203  if (xp <= 100)
204  xp = 5 * ((xp + 2) / 5);
205  else if (xp <= 500)
206  xp = 10 * ((xp + 5) / 10);
207  else if (xp <= 1000)
208  xp = 25 * ((xp + 12) / 25);
209  else
210  xp = 50 * ((xp + 25) / 50);
211 
212  return xp;
213  }
214 
215  return 0;
216 }
