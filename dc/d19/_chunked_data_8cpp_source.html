40 void ChunkedData::Load( uint32 maxLength, uint32 chunksHint )
41 {
42  if (!maxLength)
43  maxLength = Utils::Size(Stream);
44  Chunks.reserve(chunksHint);
45  uint32 baseOffset = ftell(Stream);
46  uint32 calcOffset = 0;
47  while ((calcOffset + baseOffset) < Utils::Size(Stream) && (calcOffset < maxLength))
48  {
49  char nameBytes[5];
50  uint32 read = fread(&nameBytes, sizeof(char), 4, Stream);
51  nameBytes[read] = '\0';
52  std::string name = std::string(nameBytes);
53  // Utils::Reverse(nameBytes);
54  name = std::string(name.rbegin(), name.rend());
55  uint32 length;
56  if (fread(&length, sizeof(uint32), 1, Stream) != 1)
57  continue;
58  calcOffset += 8;
59  Chunks.push_back(new Chunk(name.c_str(), length, calcOffset + baseOffset, Stream));
60  calcOffset += length;
61  // save an extra seek at the end
62  if ((calcOffset + baseOffset) < Utils::Size(Stream) && calcOffset < maxLength)
63  fseek(Stream, length, SEEK_CUR);
64  }
65 }
