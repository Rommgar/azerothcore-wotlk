103  void DamageTaken(Unit* attacker, uint32& /*damage*/, DamageEffectType, SpellSchoolMask) override
104  {
105  if (!attacker || _phase != PHASE_MELEE)
106  return;
107 
108  ++_hitcounter;
109 
110  if (attacker->HasUnitState(UNIT_STATE_MELEE_ATTACKING) && _hitcounter >= HITCOUNTER_EXPLODE)
111  {
112  Talk(EMOTE_EXPLODE);
113  events.Reset();
114  _phase = PHASE_GLOB;
115  DoCast(me, SPELL_VISCIDUS_EXPLODE);
116  me->SetVisible(false);
117  me->RemoveAura(SPELL_TOXIN);
118  me->RemoveAura(SPELL_VISCIDUS_FREEZE);
119 
120  uint8 NumGlobes = me->GetHealthPct() / 5.0f;
121  for (uint8 i = 0; i < NumGlobes; ++i)
122  {
123  float Angle = i * 2 * M_PI / NumGlobes;
124  float X = ViscidusCoord.GetPositionX() + std::cos(Angle) * RoomRadius;
125  float Y = ViscidusCoord.GetPositionY() + std::sin(Angle) * RoomRadius;
126  float Z = -35.0f;
127 
128  if (TempSummon* Glob = me->SummonCreature(NPC_GLOB_OF_VISCIDUS, X, Y, Z))
129  {
130  Glob->UpdateAllowedPositionZ(X, Y, Z);
131  Glob->NearTeleportTo(X, Y, Z, 0.0f);
132  Glob->GetMotionMaster()->MovePoint(ROOM_CENTER, ViscidusCoord);
133  }
134  }
135  }
136  else if (_hitcounter == HITCOUNTER_SHATTER)
137  Talk(EMOTE_SHATTER);
138  else if (_hitcounter == HITCOUNTER_CRACK)
139  Talk(EMOTE_CRACK);
140  }
141 
142  void SpellHit(Unit* /*caster*/, SpellInfo const* spell) override
143  {
144  if ((spell->GetSchoolMask() & SPELL_SCHOOL_MASK_FROST) && _phase == PHASE_FROST && me->GetHealthPct() > 5.0f)
145  {
146  ++_hitcounter;
147 
148  if (_hitcounter >= HITCOUNTER_FREEZE)
149  {
150  _hitcounter = 0;
151  Talk(EMOTE_FROZEN);
152  _phase = PHASE_MELEE;
153  DoCast(me, SPELL_VISCIDUS_FREEZE);
154  me->RemoveAura(SPELL_VISCIDUS_SLOWED_MORE);
155  events.ScheduleEvent(EVENT_RESET_PHASE, 15000);
156  }
157  else if (_hitcounter >= HITCOUNTER_SLOW_MORE)
158  {
159  Talk(EMOTE_FREEZE);
160  me->RemoveAura(SPELL_VISCIDUS_SLOWED);
161  DoCast(me, SPELL_VISCIDUS_SLOWED_MORE);
162  }
163  else if (_hitcounter >= HITCOUNTER_SLOW)
164  {
165  Talk(EMOTE_SLOW);
166  DoCast(me, SPELL_VISCIDUS_SLOWED);
167  }
168  }
169  }

197  void UpdateAI(uint32 diff) override
198  {
199  if (!UpdateVictim())
200  return;
201 
202  if (_phase == PHASE_GLOB && summons.empty())
203  {
204  DoResetThreat();
205  me->NearTeleportTo(ViscidusCoord.GetPositionX(),
206  ViscidusCoord.GetPositionY(),
207  ViscidusCoord.GetPositionZ(),
208  ViscidusCoord.GetOrientation());
209 
210  _hitcounter = 0;
211  _phase = PHASE_FROST;
212  InitSpells();
213  me->SetVisible(true);
214  }
215 
216  events.Update(diff);
217 
218  while (uint32 eventId = events.ExecuteEvent())
219  {
220  switch (eventId)
221  {
222  case EVENT_POISONBOLT_VOLLEY:
223  DoCast(me, SPELL_POISONBOLT_VOLLEY);
224  events.ScheduleEvent(EVENT_POISONBOLT_VOLLEY, urand(10000, 15000));
225  break;
226  case EVENT_POISON_SHOCK:
227  DoCast(me, SPELL_POISON_SHOCK);
228  events.ScheduleEvent(EVENT_POISON_SHOCK, urand(7000, 12000));
229  break;
230  case EVENT_RESET_PHASE:
231  _hitcounter = 0;
232  _phase = PHASE_FROST;
233  break;
234  default:
235  break;
236  }
237  }
238 
239  if (_phase != PHASE_GLOB)
240  DoMeleeAttackIfReady();
241  }
242 
243  private:
244  uint8 _hitcounter;
245  Phases _phase;
246  };
