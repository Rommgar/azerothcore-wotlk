152  One time initialization. For simplicity, we assume initializer thread
153  does not exit within init_routine().

155 int my_pthread_once(my_pthread_once_t *once_control,
156  void (*init_routine)(void))
157 {
158  LONG state;
159 
160  /*
161  Do "dirty" read to find out if initialization is already done, to
162  save an interlocked operation in common case. Memory barriers are ensured by
163  Visual C++ volatile implementation.
164  */
165  if (*once_control == MY_PTHREAD_ONCE_DONE)
166  return 0;
167 
168  state= InterlockedCompareExchange(once_control, MY_PTHREAD_ONCE_INPROGRESS,
169  MY_PTHREAD_ONCE_INIT);
170 
171  switch(state)
172  {
173  case MY_PTHREAD_ONCE_INIT:
174  /* This is initializer thread */
175  (*init_routine)();
176  *once_control= MY_PTHREAD_ONCE_DONE;
177  break;
178 
179  case MY_PTHREAD_ONCE_INPROGRESS:
180  /* init_routine in progress. Wait for its completion */
181  while(*once_control == MY_PTHREAD_ONCE_INPROGRESS)
182  {
183  Sleep(1);
184  }
185  break;
186  case MY_PTHREAD_ONCE_DONE:
187  /* Nothing to do */
188  break;
189  }
190  return 0;
191 }
