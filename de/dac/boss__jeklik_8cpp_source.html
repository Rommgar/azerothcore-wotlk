131  void UpdateAI(uint32 diff) override
132  {
133  if (!UpdateVictim())
134  return;
135 
136  events.Update(diff);
137 
138  if (me->HasUnitState(UNIT_STATE_CASTING))
139  return;
140 
141  while (uint32 eventId = events.ExecuteEvent())
142  {
143  switch (eventId)
144  {
145  case EVENT_CHARGE_JEKLIK:
146  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
147  {
148  DoCast(target, SPELL_CHARGE);
149  AttackStart(target);
150  }
151  events.ScheduleEvent(EVENT_CHARGE_JEKLIK, urand(15000, 30000), 0, PHASE_ONE);
152  break;
153  case EVENT_SONIC_BURST:
154  DoCastVictim(SPELL_SONICBURST);
155  events.ScheduleEvent(EVENT_SONIC_BURST, urand(8000, 13000), 0, PHASE_ONE);
156  break;
157  case EVENT_SCREECH:
158  DoCastVictim(SPELL_SCREECH);
159  events.ScheduleEvent(EVENT_SCREECH, urand(18000, 26000), 0, PHASE_ONE);
160  break;
161  case EVENT_SPAWN_BATS:
162  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
163  for (uint8 i = 0; i < 6; ++i)
164  if (Creature* bat = me->SummonCreature(NPC_BLOODSEEKER_BAT, SpawnBat[i], TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
165  bat->AI()->AttackStart(target);
166  events.ScheduleEvent(EVENT_SPAWN_BATS, 60000, 0, PHASE_ONE);
167  break;
168  case EVENT_SHADOW_WORD_PAIN:
169  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
170  DoCast(target, SPELL_SHADOW_WORD_PAIN);
171  events.ScheduleEvent(EVENT_SHADOW_WORD_PAIN, urand(12000, 18000), 0, PHASE_TWO);
172  break;
173  case EVENT_MIND_FLAY:
174  DoCastVictim(SPELL_MIND_FLAY);
175  events.ScheduleEvent(EVENT_MIND_FLAY, 16000, 0, PHASE_TWO);
176  break;
177  case EVENT_CHAIN_MIND_FLAY:
178  me->InterruptNonMeleeSpells(false);
179  DoCastVictim(SPELL_CHAIN_MIND_FLAY);
180  events.ScheduleEvent(EVENT_CHAIN_MIND_FLAY, urand(15000, 30000), 0, PHASE_TWO);
181  break;
182  case EVENT_GREATER_HEAL:
183  me->InterruptNonMeleeSpells(false);
184  DoCast(me, SPELL_GREATERHEAL);
185  events.ScheduleEvent(EVENT_GREATER_HEAL, urand(25000, 35000), 0, PHASE_TWO);
186  break;
187  case EVENT_SPAWN_FLYING_BATS:
188  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
189  if (Creature* flyingBat = me->SummonCreature(NPC_FRENZIED_BAT, target->GetPositionX(), target->GetPositionY(), target->GetPositionZ() + 15.0f, 0.0f, TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 15000))
190  flyingBat->AI()->AttackStart(target);
191  events.ScheduleEvent(EVENT_SPAWN_FLYING_BATS, urand(10000, 15000), 0, PHASE_TWO);
192  break;
193  default:
194  break;
195  }
196  }
197 
198  DoMeleeAttackIfReady();
199  }
200  };
