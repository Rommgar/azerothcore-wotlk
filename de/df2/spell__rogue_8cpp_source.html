193 {
194  PrepareSpellScript(spell_rog_deadly_poison);
195 
196  bool Load() override
197  {
198  _stackAmount = 0;

200  return GetCaster()->GetTypeId() == TYPEID_PLAYER && GetCastItem();
201  }
202 
203  void HandleBeforeHit(SpellMissInfo missInfo)
204  {
205  if (missInfo != SPELL_MISS_NONE)
206  {
207  return;
208  }
209 
210  if (Unit* target = GetHitUnit())
211  // Deadly Poison
212  if (AuraEffect const* aurEff = target->GetAuraEffect(SPELL_AURA_PERIODIC_DAMAGE, SPELLFAMILY_ROGUE, 0x10000, 0x80000, 0, GetCaster()->GetGUID()))
213  _stackAmount = aurEff->GetBase()->GetStackAmount();
214  }
215 
216  void HandleAfterHit()
217  {
218  if (_stackAmount < 5)
219  return;
220 
221  Player* player = GetCaster()->ToPlayer();
222 
223  if (Unit* target = GetHitUnit())
224  {
225  Item* item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_MAINHAND);
226 
227  if (item == GetCastItem())
228  item = player->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_OFFHAND);
229 
230  if (!item)
231  return;
232 
233  // item combat enchantments
234  for (uint8 slot = 0; slot < MAX_ENCHANTMENT_SLOT; ++slot)
235  {
236  SpellItemEnchantmentEntry const* enchant = sSpellItemEnchantmentStore.LookupEntry(item->GetEnchantmentId(EnchantmentSlot(slot)));
237  if (!enchant)
238  continue;
239 
240  for (uint8 s = 0; s < 3; ++s)
241  {
242  if (enchant->type[s] != ITEM_ENCHANTMENT_TYPE_COMBAT_SPELL)
243  continue;
244 
245  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(enchant->spellid[s]);
246  if (!spellInfo)
247  {
248  LOG_ERROR("misc", "Player::CastItemCombatSpell Enchant %i, player (Name: %s, %s) cast unknown spell %i",
249  enchant->ID, player->GetName().c_str(), player->GetGUID().ToString().c_str(), enchant->spellid[s]);
250  continue;
251  }
252 
253  // Proc only rogue poisons
254  if (spellInfo->SpellFamilyName != SPELLFAMILY_ROGUE || spellInfo->Dispel != DISPEL_POISON)
255  continue;
256 
257  // Do not reproc deadly
258  if (spellInfo->SpellFamilyFlags.IsEqual(0x10000, 0x80000, 0))
259  continue;
260 
261  if (spellInfo->IsPositive())
262  player->CastSpell(player, enchant->spellid[s], true, item);
263  else
264  player->CastSpell(target, enchant->spellid[s], true, item);
265  }
266  }
267  }
268  }
269 
270  void Register() override
271  {
272  BeforeHit += BeforeSpellHitFn(spell_rog_deadly_poison::HandleBeforeHit);
273  AfterHit += SpellHitFn(spell_rog_deadly_poison::HandleAfterHit);
274  }
275 
276  uint8 _stackAmount;
