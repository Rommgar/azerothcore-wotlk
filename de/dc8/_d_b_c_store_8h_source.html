187 #ifdef ELUNA
188  void SetEntry(uint32 id, T* t)
189  {
190  if (id >= _indexTableSize)
191  {
192  // Resize
193  typedef char* ptr;
194  size_t newSize = id + 1;
195  ptr* newArr = new ptr[newSize];
196  memset(newArr, 0, newSize * sizeof(ptr));
197  memcpy(newArr, _indexTable.AsChar, _indexTableSize * sizeof(ptr));
198  delete[] reinterpret_cast<char*>(_indexTable.AsT);
199  _indexTable.AsChar = newArr;
200  _indexTableSize = newSize;
201  }
202 
203  delete _indexTable.AsT[id];
204  _indexTable.AsT[id] = t;
205  }
206 #endif

208  [[nodiscard]] uint32 GetNumRows() const { return _indexTableSize; }
