121  void UpdateAI(uint32 diff) override
122  {
123  if (!UpdateVictim())
124  return;
125 
126  events.Update(diff);
127  if (me->HasUnitState(UNIT_STATE_CASTING))
128  return;
129 
130  switch (events.ExecuteEvent())
131  {
132  case EVENT_CHECK_HEALTH:
133  if (HealthBelowPct(10))
134  {
135  me->CastSpell(me, SPELL_FRENZY, true);
136  Talk(SAY_EMOTE_FRENZY);
137  break;
138  }
139  events.ScheduleEvent(EVENT_CHECK_HEALTH, 1000);
140  break;
141  case EVENT_SPELL_ENRAGE:
142  me->CastSpell(me, SPELL_ENRAGE, true);
143  Talk(SAY_ENRAGE);
144  break;
145  case EVENT_RANDOM_TALK:
146  Talk(SAY_TAUNT);
147  events.ScheduleEvent(EVENT_RANDOM_TALK, urand(60000, 120000));
148  break;
149  case EVENT_SPELL_SABER_LASH:
150  me->CastSpell(me->GetVictim(), SPELL_SABER_LASH, false);
151  events.ScheduleEvent(EVENT_SPELL_SABER_LASH, 30000);
152  break;
153  case EVENT_SPELL_PRISMATIC_AURA:
154  me->CastSpell(me, RAND(SPELL_PRISMATIC_AURA_SHADOW, SPELL_PRISMATIC_AURA_FIRE, SPELL_PRISMATIC_AURA_NATURE, SPELL_PRISMATIC_AURA_ARCANE, SPELL_PRISMATIC_AURA_FROST, SPELL_PRISMATIC_AURA_HOLY), false);
155  events.ScheduleEvent(EVENT_SPELL_PRISMATIC_AURA, 15000);
156  break;
157  case EVENT_SPELL_SILENCING_SHRIEK:
158  me->CastSpell(me, SPELL_SILENCING_SHRIEK, false);
159  events.ScheduleEvent(EVENT_SPELL_SILENCING_SHRIEK, 30000);
160  break;
161  case EVENT_SPELL_FATAL_ATTRACTION:
162  Talk(SAY_SPELL);
163  me->CastCustomSpell(SPELL_FATAL_ATTRACTION, SPELLVALUE_MAX_TARGETS, 3, me, false);
164  events.ScheduleEvent(EVENT_SPELL_FATAL_ATTRACTION, 60000);
165  break;
166  }
167 
168  DoMeleeAttackIfReady();
169  }
170  };
