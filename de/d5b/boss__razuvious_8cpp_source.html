151  if (me->HasUnitState(UNIT_STATE_CASTING))
152  return;
153 
154  switch (events.ExecuteEvent())
155  {
156  case EVENT_UNBALANCING_STRIKE:
157  me->CastSpell(me->GetVictim(), SPELL_UNBALANCING_STRIKE, false);
158  events.RepeatEvent(20000);
159  break;
160  case EVENT_DISRUPTING_SHOUT:
161  me->CastSpell(me, RAID_MODE(SPELL_DISRUPTING_SHOUT_10, SPELL_DISRUPTING_SHOUT_25), false);
162  events.RepeatEvent(15000);
163  break;
164  case EVENT_JAGGED_KNIFE:
165  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 45.0f))
166  {
167  me->CastSpell(target, SPELL_JAGGED_KNIFE, false);
168  }
169  events.RepeatEvent(10000);

171  }
