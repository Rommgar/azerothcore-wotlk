33 /*
34  int my_pthread_auto_mutex_lock(HANDLE* ph, const char* name,
35  int id, int time)
36  NOTES
37  creates a mutex with given name and tries to lock it time msec.
38  mutex name is appended with id to allow system wide or process wide
39  locks. Handle to created mutex returned in ph argument.
40 
41  RETURN
42  0 thread owns mutex
43  <>0 error
44 */

46 static
47 int my_pthread_auto_mutex_lock(HANDLE* ph, const char* name, int id, int time)
48 {
49  int res;
50  char tname[FN_REFLEN];
51 
52  sprintf(tname, "%s-%08X", name, id);
53 
54  *ph= CreateMutex(NULL, FALSE, tname);
55  if (*ph == NULL)
56  return GetLastError();
57 
58  res= WaitForSingleObject(*ph, time);
59 
60  if (res == WAIT_TIMEOUT)
61  return ERROR_SEM_TIMEOUT;
62 
63  if (res == WAIT_FAILED)
64  return GetLastError();
65 
66  return 0;
67 }

105  char* my_cgets()
106 
107  NOTES
108  Replaces _cgets from libc to support input of more than 255 chars.

124 char* my_cgets(char *buffer, size_t clen, size_t* plen)
125 {

127  char *result;
128  DWORD plen_res;
129  CONSOLE_SCREEN_BUFFER_INFO csbi;
130 
131  pthread_auto_mutex_decl(my_conio_cs);
132 
133  /* lock the console for the current process*/
134  if (pthread_auto_mutex_lock(my_conio_cs, GetCurrentProcessId(), INFINITE))
135  {
136  /* can not lock console */
137  pthread_auto_mutex_free(my_conio_cs);
138  return NULL;
139  }
140 
141  /* init console input */
142  if (my_coninpfh == 0)
143  {
144  /* same handle will be used until process termination */
145  my_coninpfh= CreateFile("CONIN$", GENERIC_READ | GENERIC_WRITE,
146  FILE_SHARE_READ | FILE_SHARE_WRITE,
147  NULL, OPEN_EXISTING, 0, NULL);
148  }
149 
150  if (my_coninpfh == INVALID_HANDLE_VALUE)
151  {
152  /* unlock the console */
153  pthread_auto_mutex_free(my_conio_cs);
154  return(NULL);
155  }
156 
157  GetConsoleMode((HANDLE)my_coninpfh, &state);
158  SetConsoleMode((HANDLE)my_coninpfh, ENABLE_LINE_INPUT |
159  ENABLE_PROCESSED_INPUT | ENABLE_ECHO_INPUT);
160 
161  GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
162 
163  /*
164  there is no known way to determine allowed buffer size for input
165  though it is known it should not be more than 64K
166  so we cut 64K and try first size of screen buffer
167  if it is still to large we cut half of it and try again
168  later we may want to cycle from min(clen, 65535) to allowed size
169  with small decrement to determine exact allowed buffer
170  */
171  clen= min(clen, 65535);
172  do
173  {
174  clen= min(clen, (size_t) csbi.dwSize.X*csbi.dwSize.Y);
175  if (!ReadConsole((HANDLE)my_coninpfh, (LPVOID)buffer, (DWORD) clen - 1, &plen_res,
176  NULL))
177  {
178  result= NULL;
179  clen>>= 1;
180  }
181  else
182  {
183  result= buffer;
184  break;
185  }
186  }
187  while (GetLastError() == ERROR_NOT_ENOUGH_MEMORY);
188  *plen= plen_res;
189 
190  /* We go here on error reading the string (Ctrl-C for example) */
191  if (!*plen)
192  result= NULL; /* purecov: inspected */
193 
194  if (result != NULL)
195  {
196  if (*plen > 1 && buffer[*plen - 2] == '\r')
197  {
198  *plen= *plen - 2;
199  }
200  else
201  {
202  if (*plen > 0 && buffer[*plen - 1] == '\r')
203  {
204  char tmp[3];
205  int tmplen= sizeof(tmp);
206 
207  *plen= *plen - 1;
208  /* read /n left in the buffer */
209  ReadConsole((HANDLE)my_coninpfh, (LPVOID)tmp, tmplen, &tmplen, NULL);
210  }
211  }
212  buffer[*plen]= '\0';
213  }
214 
215  SetConsoleMode((HANDLE)my_coninpfh, state);
216  /* unlock the console */
217  pthread_auto_mutex_free(my_conio_cs);
218 
219  return result;
220 }
