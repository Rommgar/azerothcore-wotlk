248  void EnfeebleHealthEffect()
249  {
250  const SpellInfo* info = sSpellMgr->GetSpellInfo(SPELL_ENFEEBLE_EFFECT);
251  if (!info)
252  return;
253 
254  ThreatContainer::StorageType const& t_list = me->getThreatMgr().getThreatList();
255  std::vector<Unit*> targets;
256 
257  if (t_list.empty())
258  return;
259 
260  //begin + 1, so we don't target the one with the highest threat
261  ThreatContainer::StorageType::const_iterator itr = t_list.begin();
262  std::advance(itr, 1);
263  for (; itr != t_list.end(); ++itr) //store the threat list in a different container
264  if (Unit* target = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
265  if (target->IsAlive() && target->GetTypeId() == TYPEID_PLAYER)
266  targets.push_back(target);
267 
268  //cut down to size if we have more than 5 targets
269  while (targets.size() > 5)
270  targets.erase(targets.begin() + rand() % targets.size());
271 
272  uint32 i = 0;
273  for (std::vector<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter, ++i)
274  if (Unit* target = *iter)
275  {
276  enfeeble_targets[i] = target->GetGUID();
277  enfeeble_health[i] = target->GetHealth();
278 
279  me->CastSpell(target, SPELL_ENFEEBLE, true, 0, 0, me->GetGUID());
280  target->SetHealth(1);
281  }
282  }
283 
284  void EnfeebleResetHealth()
285  {
286  for (uint8 i = 0; i < 5; ++i)
287  {
288  Unit* target = ObjectAccessor::GetUnit(*me, enfeeble_targets[i]);
289  if (target && target->IsAlive())
290  target->SetHealth(enfeeble_health[i]);
291  enfeeble_targets[i].Clear();
292  enfeeble_health[i] = 0;
293  }
294  }

349  void UpdateAI(uint32 diff) override
350  {
351  if (!UpdateVictim())
352  return;
353 
354  if (EnfeebleResetTimer && EnfeebleResetTimer <= diff) // Let's not forget to reset that
355  {
356  EnfeebleResetHealth();
357  EnfeebleResetTimer = 0;
358  }
359  else EnfeebleResetTimer -= diff;
360 
361  if (InfernalTimer <= diff)
362  {
363  SummonInfernal();
364  InfernalTimer = phase == 3 ? 14500 : 44500; // 15 secs in phase 3, 45 otherwise
365  }
366  else InfernalTimer -= diff;
367 
368  if (phase != 2)
369  {
370  if (SWPainTimer <= diff)
371  {
372 
373  // if phase == 1 target the tank, otherwise anyone but the tank
374  Unit* target = phase == 1
375  ? me->GetVictim()
376  : SelectTarget(SELECT_TARGET_RANDOM, 1, 100, true);
377 
378  if (target)
379  {
380  DoCast(target, SPELL_SW_PAIN);
381  }
382 
383  SWPainTimer = 20000;
384  }
385  else
386  SWPainTimer -= diff;
387  }
388 
389  if (ShadowNovaTimer <= diff)
390  {
391  DoCast(SPELL_SHADOWNOVA);
392  ShadowNovaTimer = 35500;
393  }
394  else
395  ShadowNovaTimer -= diff;
396 
397  if (phase == 1)
398  {
399  if (HealthBelowPct(60))
400  {
401  Phase2();
402  }
403  }
404
