65  bool MMapMgr::loadMapData(uint32 mapId)
66  {
67  // we already have this map loaded?
68  MMapDataSet::iterator itr = loadedMMaps.find(mapId);
69  if (itr != loadedMMaps.end())
70  {
71  if (itr->second)
72  {
73  return true;
74  }
75  }
76  else
77  {
78  if (thread_safe_environment)
79  {
80  itr = loadedMMaps.insert(MMapDataSet::value_type(mapId, nullptr)).first;
81  }
82  else
83  {
84  ASSERT(false, "Invalid mapId %u passed to MMapMgr after startup in thread unsafe environment", mapId);
85  }
86  }
87 
88  // load and init dtNavMesh - read parameters from file
89  std::string fileName = Acore::StringFormat(MAP_FILE_NAME_FORMAT, sConfigMgr->GetOption<std::string>("DataDir", ".").c_str(), mapId);
90 
91  FILE* file = fopen(fileName.c_str(), "rb");
92  if (!file)
93  {
94  LOG_DEBUG("maps", "MMAP:loadMapData: Error: Could not open mmap file '%s'", fileName.c_str());
95  return false;
96  }
97 
98  dtNavMeshParams params;
99  uint32 count = uint32(fread(&params, sizeof(dtNavMeshParams), 1, file));
100  fclose(file);
101  if (count != 1)
102  {
103  LOG_DEBUG("maps", "MMAP:loadMapData: Error: Could not read params from file '%s'", fileName.c_str());
104  return false;
105  }
106 
107  dtNavMesh* mesh = dtAllocNavMesh();
108  ASSERT(mesh);
109  if (DT_SUCCESS != mesh->init(&params))
110  {
111  dtFreeNavMesh(mesh);
112  LOG_ERROR("maps", "MMAP:loadMapData: Failed to initialize dtNavMesh for mmap %03u from file %s", mapId, fileName.c_str());
113  return false;
114  }
115 
116  LOG_DEBUG("maps", "MMAP:loadMapData: Loaded %03i.mmap", mapId);
117 
118  // store inside our map list
119  MMapData* mmap_data = new MMapData(mesh);
120  itr->second = mmap_data;
121  return true;
122  }

129  bool MMapMgr::loadMap(uint32 mapId, int32 x, int32 y)
130  {
131  // make sure the mmap is loaded and ready to load tiles
132  if (!loadMapData(mapId))
133  {
134  return false;
135  }
136 
137  // get this mmap data
138  MMapData* mmap = loadedMMaps[mapId];
139  ASSERT(mmap->navMesh);
140 
141  // check if we already have this tile loaded
142  uint32 packedGridPos = packTileID(x, y);
143  if (mmap->loadedTileRefs.find(packedGridPos) != mmap->loadedTileRefs.end())
144  {
145  LOG_ERROR("maps", "MMAP:loadMap: Asked to load already loaded navmesh tile. %03u%02i%02i.mmtile", mapId, x, y);
146  return false;
147  }
148 
149  // load this tile :: mmaps/MMMXXYY.mmtile
150  std::string fileName = Acore::StringFormat(TILE_FILE_NAME_FORMAT, sConfigMgr->GetOption<std::string>("DataDir", ".").c_str(), mapId, x, y);
151  FILE* file = fopen(fileName.c_str(), "rb");
152  if (!file)
153  {
154  LOG_DEBUG("maps", "MMAP:loadMap: Could not open mmtile file '%s'", fileName.c_str());
155  return false;
156  }
157 
158  // read header
159  MmapTileHeader fileHeader;
160  if (fread(&fileHeader, sizeof(MmapTileHeader), 1, file) != 1 || fileHeader.mmapMagic != MMAP_MAGIC)
161  {
162  LOG_ERROR("maps", "MMAP:loadMap: Bad header in mmap %03u%02i%02i.mmtile", mapId, x, y);
163  fclose(file);
164  return false;
165  }
166 
167  if (fileHeader.mmapVersion != MMAP_VERSION)
168  {
169  LOG_ERROR("maps", "MMAP:loadMap: %03u%02i%02i.mmtile was built with generator v%i, expected v%i",
170  mapId, x, y, fileHeader.mmapVersion, MMAP_VERSION);
171  fclose(file);
172  return false;
173  }
174 
175  unsigned char* data = (unsigned char*)dtAlloc(fileHeader.size, DT_ALLOC_PERM);
176  ASSERT(data);
177 
178  size_t result = fread(data, fileHeader.size, 1, file);
179  if (!result)
180  {
181  LOG_ERROR("maps", "MMAP:loadMap: Bad header or data in mmap %03u%02i%02i.mmtile", mapId, x, y);
182  fclose(file);
183  return false;
184  }
185 
186  fclose(file);
187 
188  dtTileRef tileRef = 0;
189 
190  // memory allocated for data is now managed by detour, and will be deallocated when the tile is removed
191  if (dtStatusSucceed(mmap->navMesh->addTile(data, fileHeader.size, DT_TILE_FREE_DATA, 0, &tileRef)))
192  {
193  mmap->loadedTileRefs.insert(std::pair<uint32, dtTileRef>(packedGridPos, tileRef));
194  ++loadedTiles;
195  dtMeshHeader* header = (dtMeshHeader*)data;
196  LOG_DEBUG("maps", "MMAP:loadMap: Loaded mmtile %03i[%02i,%02i] into %03i[%02i,%02i]", mapId, x, y, mapId, header->x, header->y);
197  return true;
198  }
199 
200  LOG_ERROR("maps", "MMAP:loadMap: Could not load %03u%02i%02i.mmtile into navmesh", mapId, x, y);
201  dtFree(data);
202  return false;
203  }

205  bool MMapMgr::unloadMap(uint32 mapId, int32 x, int32 y)
206  {
207  // check if we have this map loaded
208  MMapDataSet::const_iterator itr = GetMMapData(mapId);
209  if (itr == loadedMMaps.end())
210  {
211  // file may not exist, therefore not loaded
212  LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh map. %03u%02i%02i.mmtile", mapId, x, y);
213  return false;
214  }
215 
216  MMapData* mmap = itr->second;
217 
218  // check if we have this tile loaded
219  uint32 packedGridPos = packTileID(x, y);
220  if (mmap->loadedTileRefs.find(packedGridPos) == mmap->loadedTileRefs.end())
221  {
222  // file may not exist, therefore not loaded
223  LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh tile. %03u%02i%02i.mmtile", mapId, x, y);
224  return false;
225  }
226 
227  dtTileRef tileRef = mmap->loadedTileRefs[packedGridPos];
228 
229  // unload, and mark as non loaded
230  if (dtStatusFailed(mmap->navMesh->removeTile(tileRef, nullptr, nullptr)))
231  {
232  // this is technically a memory leak
233  // if the grid is later reloaded, dtNavMesh::addTile will return error but no extra memory is used
234  // we cannot recover from this error - assert out
235  LOG_ERROR("maps", "MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
236  ABORT();
237  }
238 
239  mmap->loadedTileRefs.erase(packedGridPos);
240  --loadedTiles;
241  LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded mmtile %03i[%02i,%02i] from %03i", mapId, x, y, mapId);
242  return true;
243  }

245  bool MMapMgr::unloadMap(uint32 mapId)
246  {
247  MMapDataSet::iterator itr = loadedMMaps.find(mapId);
248  if (itr == loadedMMaps.end() || !itr->second)
249  {
250  // file may not exist, therefore not loaded
251  LOG_DEBUG("maps", "MMAP:unloadMap: Asked to unload not loaded navmesh map %03u", mapId);
252  return false;
253  }
254 
255  // unload all tiles from given map
256  MMapData* mmap = itr->second;
257  for (auto i : mmap->loadedTileRefs)
258  {
259  uint32 x = (i.first >> 16);
260  uint32 y = (i.first & 0x0000FFFF);
261 
262  if (dtStatusFailed(mmap->navMesh->removeTile(i.second, nullptr, nullptr)))
263  {
264  LOG_ERROR("maps", "MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
265  }
266  else
267  {
268  --loadedTiles;
269  LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded mmtile %03i[%02i,%02i] from %03i", mapId, x, y, mapId);
270  }
271  }
272 
273  delete mmap;
274  itr->second = nullptr;
275  LOG_DEBUG("maps", "MMAP:unloadMap: Unloaded %03i.mmap", mapId);
276 
277  return true;
278  }

318  dtNavMeshQuery const* MMapMgr::GetNavMeshQuery(uint32 mapId, uint32 instanceId)
319  {
320  MMapDataSet::const_iterator itr = GetMMapData(mapId);
321  if (itr == loadedMMaps.end())
322  {
323  return nullptr;
324  }
325 
326  MMapData* mmap = itr->second;
327  if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
328  {
329  // check again after acquiring mutex
330  if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
331  {
332  // allocate mesh query
333  dtNavMeshQuery* query = dtAllocNavMeshQuery();
334  ASSERT(query);
335 
336  if (dtStatusFailed(query->init(mmap->navMesh, 1024)))
337  {
338  dtFreeNavMeshQuery(query);
339  LOG_ERROR("maps", "MMAP:GetNavMeshQuery: Failed to initialize dtNavMeshQuery for mapId %03u instanceId %u", mapId, instanceId);
340  return nullptr;
341  }
342 
343  LOG_DEBUG("maps", "MMAP:GetNavMeshQuery: created dtNavMeshQuery for mapId %03u instanceId %u", mapId, instanceId);
344  mmap->navMeshQueries.insert(std::pair<uint32, dtNavMeshQuery*>(instanceId, query));
345  }
346  }
347 
348  return mmap->navMeshQueries[instanceId];
