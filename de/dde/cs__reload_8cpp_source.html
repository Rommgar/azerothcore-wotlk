425  static bool HandleReloadCreatureTemplateCommand(ChatHandler* handler, std::string_view args)
426  {
427  if (args.empty())
428  return false;
429 
430  for (std::string_view entryStr : Acore::Tokenize(args, ' ', false))
431  {
432  uint32 entry = Acore::StringTo<uint32>(entryStr).value_or(0);
433 
434  WorldDatabasePreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_SEL_CREATURE_TEMPLATE);
435  stmt->setUInt32(0, entry);
436  PreparedQueryResult result = WorldDatabase.Query(stmt);
437 
438  if (!result)
439  {
440  handler->PSendSysMessage(LANG_COMMAND_CREATURETEMPLATE_NOTFOUND, entry);
441  continue;
442  }
443 
444  CreatureTemplate* cInfo = const_cast<CreatureTemplate*>(sObjectMgr->GetCreatureTemplate(entry));
445  if (!cInfo)
446  {
447  handler->PSendSysMessage(LANG_COMMAND_CREATURESTORAGE_NOTFOUND, entry);
448  continue;
449  }
450 
451  LOG_INFO("server.loading", "Reloading creature template entry %u", entry);
452 
453  Field* fields = result->Fetch();
454 
455  sObjectMgr->LoadCreatureTemplate(fields);
456  sObjectMgr->CheckCreatureTemplate(cInfo);
457  }
458 
459  handler->SendGlobalGMSysMessage("Creature template reloaded.");
460  return true;
461  }
