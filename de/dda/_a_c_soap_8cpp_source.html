25 void ACSoapThread(const std::string& host, uint16 port)
26 {
27  struct soap soap;
28  soap_init(&soap);
29  soap_set_imode(&soap, SOAP_C_UTFSTRING);
30  soap_set_omode(&soap, SOAP_C_UTFSTRING);
31 
32  // check every 3 seconds if world ended
33  soap.accept_timeout = 3;
34  soap.recv_timeout = 5;
35  soap.send_timeout = 5;
36 
37  if (!soap_valid_socket(soap_bind(&soap, host.c_str(), port, 100)))
38  {
39  LOG_ERROR("network.soap", "ACSoap: couldn't bind to %s:%d", host.c_str(), port);
40  exit(-1);
41  }
42 
43  LOG_INFO("network.soap", "ACSoap: bound to http://%s:%d", host.c_str(), port);
44 
45  while (!World::IsStopped())
46  {
47  if (!soap_valid_socket(soap_accept(&soap)))
48  continue; // ran into an accept timeout
49 
50  LOG_DEBUG("network.soap", "ACSoap: accepted connection from IP=%d.%d.%d.%d", (int)(soap.ip >> 24) & 0xFF, (int)(soap.ip >> 16) & 0xFF, (int)(soap.ip >> 8) & 0xFF, (int)soap.ip & 0xFF);
51  struct soap* thread_soap = soap_copy(&soap);// make a safe copy
52 
53  process_message(thread_soap);
54  }
55 
56  soap_destroy(&soap);
57  soap_end(&soap);
58  soap_done(&soap);
59 }

75 int ns1__executeCommand(soap* soap, char* command, char** result)
76 {
77  // security check
78  if (!soap->userid || !soap->passwd)
79  {
80  LOG_DEBUG("network.soap", "ACSoap: Client didn't provide login information");
81  return 401;
82  }
83 
84  uint32 accountId = AccountMgr::GetId(soap->userid);
85  if (!accountId)
86  {
87  LOG_DEBUG("network", "ACSoap: Client used invalid username '%s'", soap->userid);
88  return 401;
89  }
90 
91  if (!AccountMgr::CheckPassword(accountId, soap->passwd))
92  {
93  LOG_DEBUG("network.soap", "ACSoap: invalid password for account '%s'", soap->userid);
94  return 401;
95  }
96 
97  if (AccountMgr::GetSecurity(accountId) < SEC_ADMINISTRATOR)
98  {
99  LOG_DEBUG("network.soap", "ACSoap: %s's gmlevel is too low", soap->userid);
100  return 403;
101  }
102 
103  if (!command || !*command)
104  return soap_sender_fault(soap, "Command can not be empty", "The supplied command was an empty string");
105 
106  LOG_DEBUG("network.soap", "ACSoap: got command '%s'", command);
107  SOAPCommand connection;
108 
109  // commands are executed in the world thread. We have to wait for them to be completed
110  {
111  // CliCommandHolder will be deleted from world, accessing after queueing is NOT save
112  CliCommandHolder* cmd = new CliCommandHolder(&connection, command, &SOAPCommand::print, &SOAPCommand::commandFinished);
113  sWorld->QueueCliCommand(cmd);
114  }
115 
116  // Wait until the command has finished executing
117  connection.finishedPromise.get_future().wait();
118 
119  // The command has finished executing already
120  char* printBuffer = soap_strdup(soap, connection.m_printBuffer.c_str());
121  if (connection.hasCommandSucceeded())
122  {
123  *result = printBuffer;
124  return SOAP_OK;
125  }
126  else
127  return soap_sender_fault(soap, printBuffer, printBuffer);
128 }
