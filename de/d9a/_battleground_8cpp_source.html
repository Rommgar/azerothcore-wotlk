1319 {

1321  return 0;

1325  return (GetInvitedCount(teamId) < GetMaxPlayersPerTeam()) ? GetMaxPlayersPerTeam() - GetInvitedCount(teamId) : 0;

1334  // default: allow 0

1341  diff = 1;
