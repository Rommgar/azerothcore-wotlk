184 Vio *vio_new(my_socket sd, enum enum_vio_type type, uint flags)
185 {
186  Vio *vio;
187  DBUG_ENTER("vio_new");
188  DBUG_PRINT("enter", ("sd: %d", sd));
189  if ((vio = (Vio*) my_malloc(sizeof(*vio),MYF(MY_WME))))
190  {
191  vio_init(vio, type, sd, 0, flags);
192  sprintf(vio->desc,
193  (vio->type == VIO_TYPE_SOCKET ? "socket (%d)" : "TCP/IP (%d)"),
194  vio->sd);
195 #if !defined(__WIN__)
196 #if !defined(NO_FCNTL_NONBLOCK)
197  /*
198  We call fcntl() to set the flags and then immediately read them back
199  to make sure that we and the system are in agreement on the state of
200  things.
201 
202  An example of why we need to do this is FreeBSD (and apparently some
203  other BSD-derived systems, like Mac OS X), where the system sometimes
204  reports that the socket is set for non-blocking when it really will
205  block.
206  */
207  fcntl(sd, F_SETFL, 0);
208  vio->fcntl_mode= fcntl(sd, F_GETFL);
209 #elif defined(HAVE_SYS_IOCTL_H) /* hpux */
210  /* Non blocking sockets doesn't work good on HPUX 11.0 */
211  (void) ioctl(sd,FIOSNBIO,0);
212  vio->fcntl_mode &= ~O_NONBLOCK;
213 #endif
214 #else /* !defined(__WIN__) */
215  {
216  /* set to blocking mode by default */
217  ulong arg=0, r;
218  r = ioctlsocket(sd,FIONBIO,(void*) &arg);
219  vio->fcntl_mode &= ~O_NONBLOCK;
220  }
221 #endif
222  }
223  DBUG_RETURN(vio);
224 }
