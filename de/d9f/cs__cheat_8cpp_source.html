172  static bool HandleTaxiCheatCommand(ChatHandler* handler, Optional<bool> enableArg)
173  {
174  Player* chr = handler->getSelectedPlayer();
175 
176  if (!chr)
177  chr = handler->GetSession()->GetPlayer();
178  else if (handler->HasLowerSecurity(chr, ObjectGuid::Empty)) // check online security
179  return false;
180 
181  bool enable = !chr->isTaxiCheater();
182  if (enableArg)
183  enable = *enableArg;
184 
185  if (enable)
186  {
187  chr->SetTaxiCheater(true);
188  handler->PSendSysMessage(LANG_YOU_GIVE_TAXIS, handler->GetNameLink(chr).c_str());
189  if (handler->needReportToTarget(chr))
190  ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_TAXIS_ADDED, handler->GetNameLink().c_str());
191  }
192  else
193  {
194  chr->SetTaxiCheater(false);
195  handler->PSendSysMessage(LANG_YOU_REMOVE_TAXIS, handler->GetNameLink(chr).c_str());
196  if (handler->needReportToTarget(chr))
197  ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_TAXIS_REMOVED, handler->GetNameLink().c_str());
198  }
199 
200  return true;
201  }

203  static bool HandleExploreCheatCommand(ChatHandler* handler, bool reveal)
204  {
205  Player* chr = handler->getSelectedPlayer();
206  if (!chr)
207  {
208  handler->SendSysMessage(LANG_NO_CHAR_SELECTED);
209  handler->SetSentErrorMessage(true);
210  return false;
211  }
212 
213  if (reveal)
214  {
215  handler->PSendSysMessage(LANG_YOU_SET_EXPLORE_ALL, handler->GetNameLink(chr).c_str());
216  if (handler->needReportToTarget(chr))
217  ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_EXPLORE_SET_ALL, handler->GetNameLink().c_str());
218  }
219  else
220  {
221  handler->PSendSysMessage(LANG_YOU_SET_EXPLORE_NOTHING, handler->GetNameLink(chr).c_str());
222  if (handler->needReportToTarget(chr))
223  ChatHandler(chr->GetSession()).PSendSysMessage(LANG_YOURS_EXPLORE_SET_NOTHING, handler->GetNameLink().c_str());
224  }
225 
226  for (uint8 i = 0; i < PLAYER_EXPLORED_ZONES_SIZE; ++i)
227  {
228  if (reveal)
229  handler->GetSession()->GetPlayer()->SetFlag(PLAYER_EXPLORED_ZONES_1+i, 0xFFFFFFFF);
230  else
231  handler->GetSession()->GetPlayer()->SetFlag(PLAYER_EXPLORED_ZONES_1+i, 0);
232  }
233 
234  return true;
235  }
