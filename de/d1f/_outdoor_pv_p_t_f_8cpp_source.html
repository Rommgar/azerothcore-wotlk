63 
64  data << TF_UI_TOWER_COUNT_H << m_HordeTowersControlled;
65  data << TF_UI_TOWER_COUNT_A << m_AllianceTowersControlled;

70  data << TF_UI_LOCKED_TIME_HOURS << hours_left;
71 
72  data << TF_UI_LOCKED_DISPLAY_NEUTRAL << uint32(m_IsLocked && !m_HordeTowersControlled && !m_AllianceTowersControlled);
73  data << TF_UI_LOCKED_DISPLAY_HORDE << uint32(m_IsLocked && (m_HordeTowersControlled > m_AllianceTowersControlled));
74  data << TF_UI_LOCKED_DISPLAY_ALLIANCE << uint32(m_IsLocked && (m_HordeTowersControlled < m_AllianceTowersControlled));
75 
76  for (OPvPCapturePointMap::iterator itr = m_capturePoints.begin(); itr != m_capturePoints.end(); ++itr)
77  {
78  itr->second->FillInitialWorldStates(data);
79  }

134 bool OutdoorPvPTF::Update(uint32 diff)
135 {
136  bool changed = OutdoorPvP::Update(diff);
137 
138  if (changed)
139  {
140  if (m_AllianceTowersControlled == TF_TOWER_NUM)
141  {
142  TeamApplyBuff(TEAM_ALLIANCE, TF_CAPTURE_BUFF);
143  m_IsLocked = true;
144  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
145  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
146  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(1));
147  SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
148  }
149  else if (m_HordeTowersControlled == TF_TOWER_NUM)
150  {
151  TeamApplyBuff(TEAM_HORDE, TF_CAPTURE_BUFF);
152  m_IsLocked = true;
153  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
154  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(1));
155  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
156  SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(0));
157  }
158  else
159  {
160  TeamCastSpell(TEAM_ALLIANCE, -TF_CAPTURE_BUFF);
161  TeamCastSpell(TEAM_HORDE, -TF_CAPTURE_BUFF);
162  }
163  SendUpdateWorldState(TF_UI_TOWER_COUNT_A, m_AllianceTowersControlled);
164  SendUpdateWorldState(TF_UI_TOWER_COUNT_H, m_HordeTowersControlled);
165  }
166  if (m_IsLocked)
167  {
168  // lock timer is down, release lock
169  if (m_LockTimer < diff)
170  {
171  m_LockTimer = TF_LOCK_TIME;
172  m_LockTimerUpdate = 0;
173  m_IsLocked = false;
174  SendUpdateWorldState(TF_UI_TOWERS_CONTROLLED_DISPLAY, uint32(1));
175  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_NEUTRAL, uint32(0));
176  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_HORDE, uint32(0));
177  SendUpdateWorldState(TF_UI_LOCKED_DISPLAY_ALLIANCE, uint32(0));
178  }
179  else
180  {
181  // worldstateui update timer is down, update ui with new time data
182  if (m_LockTimerUpdate < diff)
183  {
184  m_LockTimerUpdate = TF_LOCK_TIME_UPDATE;
185  uint32 minutes_left = m_LockTimer / 60000;
186  hours_left = minutes_left / 60;
187  minutes_left -= hours_left * 60;
188  second_digit = minutes_left % 10;
189  first_digit = minutes_left / 10;
190 
191  SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_FIRST_DIGIT, first_digit);
192  SendUpdateWorldState(TF_UI_LOCKED_TIME_MINUTES_SECOND_DIGIT, second_digit);
193  SendUpdateWorldState(TF_UI_LOCKED_TIME_HOURS, hours_left);
194  }
195  else m_LockTimerUpdate -= diff;
196  m_LockTimer -= diff;
197  }
198  }
199  return changed;
200 }

276 bool OPvPCapturePointTF::Update(uint32 diff)
277 {
278  // can update even in locked state if gathers the controlling faction
279  bool canupdate = ((((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled() > 0) && m_activePlayers[0].size() > m_activePlayers[1].size()) ||
280  ((((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled() > 0) && m_activePlayers[0].size() < m_activePlayers[1].size());
281  // if gathers the other faction, then only update if the pvp is unlocked
282  canupdate = canupdate || !((OutdoorPvPTF*)m_PvP)->IsLocked();
283  return canupdate && OPvPCapturePoint::Update(diff);
284 }

286 void OPvPCapturePointTF::ChangeState()
287 {
288  // if changing from controlling alliance to horde
289  if (m_OldState == OBJECTIVESTATE_ALLIANCE)
290  {
291  if (uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled())
292  ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(--alliance_towers);
293  sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_LOSE_A));
294  }
295  // if changing from controlling horde to alliance
296  else if (m_OldState == OBJECTIVESTATE_HORDE)
297  {
298  if (uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled())
299  ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(--horde_towers);
300  sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_LOSE_H));
301  }
302 
303  uint32 artkit = 21;
304 
305  switch (m_State)
306  {
307  case OBJECTIVESTATE_ALLIANCE:
308  {
309  m_TowerState = TF_TOWERSTATE_A;
310  artkit = 2;
311  uint32 alliance_towers = ((OutdoorPvPTF*)m_PvP)->GetAllianceTowersControlled();
312  if (alliance_towers < TF_TOWER_NUM)
313  ((OutdoorPvPTF*)m_PvP)->SetAllianceTowersControlled(++alliance_towers);
314 
315  sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_CAPTURE_A));
316 
317  for (PlayerSet::iterator itr = m_activePlayers[0].begin(); itr != m_activePlayers[0].end(); ++itr)
318  if (Player* player = ObjectAccessor::FindPlayer(*itr))
319  player->AreaExploredOrEventHappens(TF_ALLY_QUEST);
320  break;
321  }
322  case OBJECTIVESTATE_HORDE:
323  {
324  m_TowerState = TF_TOWERSTATE_H;
325  artkit = 1;
326  uint32 horde_towers = ((OutdoorPvPTF*)m_PvP)->GetHordeTowersControlled();
327  if (horde_towers < TF_TOWER_NUM)
328  ((OutdoorPvPTF*)m_PvP)->SetHordeTowersControlled(++horde_towers);
329 
330  sWorld->SendZoneText(OutdoorPvPTFBuffZones[0], sObjectMgr->GetAcoreStringForDBCLocale(LANG_OPVP_TF_CAPTURE_H));
331 
332  for (PlayerSet::iterator itr = m_activePlayers[1].begin(); itr != m_activePlayers[1].end(); ++itr)
333  if (Player* player = ObjectAccessor::FindPlayer(*itr))
334  player->AreaExploredOrEventHappens(TF_HORDE_QUEST);
335  break;
336  }
337  case OBJECTIVESTATE_NEUTRAL:
338  case OBJECTIVESTATE_NEUTRAL_ALLIANCE_CHALLENGE:
339  case OBJECTIVESTATE_NEUTRAL_HORDE_CHALLENGE:
340  case OBJECTIVESTATE_ALLIANCE_HORDE_CHALLENGE:
341  case OBJECTIVESTATE_HORDE_ALLIANCE_CHALLENGE:
342  m_TowerState = TF_TOWERSTATE_N;
343  break;
344  }
345 
346  auto bounds = sMapMgr->FindMap(530, 0)->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
347  for (auto itr = bounds.first; itr != bounds.second; ++itr)
348  itr->second->SetGoArtKit(artkit);
349 
350  UpdateTowerState();
351 }
