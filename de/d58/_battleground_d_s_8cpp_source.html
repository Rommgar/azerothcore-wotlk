53 void BattlegroundDS::PostUpdateImpl(uint32 diff)
54 {
55  if (GetStatus() != STATUS_IN_PROGRESS)
56  return;
57 
58  if (getPipeKnockBackCount() < BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
59  {
60  if (getPipeKnockBackTimer() < diff)
61  {
62  for (uint32 i = BG_DS_NPC_PIPE_KNOCKBACK_1; i <= BG_DS_NPC_PIPE_KNOCKBACK_2; ++i)
63  if (Creature* waterSpout = GetBgMap()->GetCreature(BgCreatures[i]))
64  waterSpout->CastSpell(waterSpout, BG_DS_SPELL_FLUSH, true);
65 
66  setPipeKnockBackCount(getPipeKnockBackCount() + 1);
67  setPipeKnockBackTimer(BG_DS_PIPE_KNOCKBACK_DELAY);
68  }
69  else
70  setPipeKnockBackTimer(getPipeKnockBackTimer() - diff);
71  }
72 
73  if (getWaterFallStatus() == BG_DS_WATERFALL_STATUS_ON) // Repeat knockback while the waterfall still active
74  {
75  if (getWaterFallKnockbackTimer() < diff)
76  {
77  if (Creature* waterSpout = GetBgMap()->GetCreature(BgCreatures[BG_DS_NPC_WATERFALL_KNOCKBACK]))
78  waterSpout->CastSpell(waterSpout, BG_DS_SPELL_WATER_SPOUT, true);
79 
80  setWaterFallKnockbackTimer(BG_DS_WATERFALL_KNOCKBACK_TIMER);
81  }
82  else
83  setWaterFallKnockbackTimer(getWaterFallKnockbackTimer() - diff);
84  }
85 
86  if (getWaterFallTimer() < diff)
87  {
88  if (getWaterFallStatus() == BG_DS_WATERFALL_STATUS_OFF) // Add the water
89  {
90  DoorClose(BG_DS_OBJECT_WATER_2);
91  setWaterFallTimer(BG_DS_WATERFALL_WARNING_DURATION);
92  setWaterFallStatus(BG_DS_WATERFALL_STATUS_WARNING);
93  }
94  else if (getWaterFallStatus() == BG_DS_WATERFALL_STATUS_WARNING) // Active collision and perform knockback
95  {
96  if (GameObject* gob = GetBgMap()->GetGameObject(BgObjects[BG_DS_OBJECT_WATER_1]))
97  gob->SetGoState(GO_STATE_READY);
98 
99  setWaterFallTimer(BG_DS_WATERFALL_DURATION);
100  setWaterFallStatus(BG_DS_WATERFALL_STATUS_ON);
101  setWaterFallKnockbackTimer(BG_DS_WATERFALL_KNOCKBACK_TIMER);
102  }
103  else //if (getWaterFallStatus() == BG_DS_WATERFALL_STATUS_ON) // Remove collision and water
104  {
105  // turn off collision
106  if (GameObject* gob = GetBgMap()->GetGameObject(BgObjects[BG_DS_OBJECT_WATER_1]))
107  gob->SetGoState(GO_STATE_ACTIVE);
108 
109  DoorOpen(BG_DS_OBJECT_WATER_2);
110  setWaterFallTimer(urand(BG_DS_WATERFALL_TIMER_MIN, BG_DS_WATERFALL_TIMER_MAX));
111  setWaterFallStatus(BG_DS_WATERFALL_STATUS_OFF);
112  }
113  }
114  else
115  setWaterFallTimer(getWaterFallTimer() - diff);
116 }

184 void BattlegroundDS::HandleAreaTrigger(Player* player, uint32 trigger)
185 {
186  if (GetStatus() != STATUS_IN_PROGRESS)
187  return;
188 
189  switch (trigger)
190  {
191  case 5347:
192  case 5348:
193  // Remove effects of Demonic Circle Summon
194  if (player->HasAura(48018))
195  player->RemoveAurasDueToSpell(48018);
196 
197  // Someone has get back into the pipes and the knockback has already been performed,
198  // so we reset the knockback count for kicking the player again into the arena.
199  if (getPipeKnockBackCount() >= BG_DS_PIPE_KNOCKBACK_TOTAL_COUNT)
200  setPipeKnockBackCount(0);
201  break;
202  // OUTSIDE OF ARENA, TELEPORT!
203  case 5328:
