183 Sphere GLight::effectSphere(float cutoff) const {
184  if (position.w == 0) {
185  // Directional light
186  return Sphere(Vector3::zero(), finf());
187  } else {
188  // Avoid divide by zero
189  cutoff = max(cutoff, 0.00001f);
190  float maxIntensity = max(color.r, max(color.g, color.b));
191 
192  float radius = finf();
193 
194  if (attenuation[2] != 0) {
195 
196  // Solve I / attenuation.dot(1, r, r^2) < cutoff for r
197  //
198  // a[0] + a[1] r + a[2] r^2 > I/cutoff
199  //
200 
201  float a = attenuation[2];
202  float b = attenuation[1];
203  float c = attenuation[0] - maxIntensity / cutoff;
204 
205  float discrim = square(b) - 4 * a * c;
206 
207  if (discrim >= 0) {
208  discrim = sqrt(discrim);
209 
210  float r1 = (-b + discrim) / (2 * a);
211  float r2 = (-b - discrim) / (2 * a);
212 
213  if (r1 < 0) {
214  if (r2 > 0) {
215  radius = r2;
216  }
217  } else if (r2 > 0) {
218  radius = min(r1, r2);
219  } else {
220  radius = r1;
221  }
222  }
223 
224  } else if (attenuation[1] != 0) {
225 
226  // Solve I / attenuation.dot(1, r) < cutoff for r
227  //
228  // r * a[1] + a[0] = I / cutoff
229  // r = (I / cutoff - a[0]) / a[1]
230 
231  float radius = (maxIntensity / cutoff - attenuation[0]) / attenuation[1];
232  radius = max(radius, 0.0f);
233  }
234 
235  return Sphere(position.xyz(), radius);
236 
237  }

241 CoordinateFrame GLight::frame() const {
242  CoordinateFrame f;
243  if (rightDirection == Vector3::zero()) {
244  // No specified right direction; choose one automatically
245  if (position.w == 0) {
246  // Directional light
247  f.lookAt(-position.xyz());
248  } else {
249  // Spot light
250  f.lookAt(spotDirection);
251  }
252  } else {
253  const Vector3& Z = -spotDirection.direction();
254  Vector3 X = rightDirection.direction();
255 
256  // Ensure the vectors are not too close together
257  while (abs(X.dot(Z)) > 0.9f) {
258  X = Vector3::random();
259  }
260 
261  // Ensure perpendicular
