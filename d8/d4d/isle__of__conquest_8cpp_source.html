236  void CheckRageBuff()
237  {
238  if (!rage)
239  {
240  if (me->GetDistance(me->GetHomePosition()) > 40.0f)
241  {

245  }
246  else

248  if (me->GetDistance(me->GetHomePosition()) < 40.0f && abs(me->GetPositionZ() - me->GetHomePosition().GetPositionZ()) < 5.0f)

252  }
253  }
254  }

256  void EnterCombat(Unit* /*who*/) override
257  {
258  events.ScheduleEvent(EVENT_CHECK_RAGE, 2000);
259  events.ScheduleEvent(EVENT_BRUTAL_STRIKE, 6000);
260  events.ScheduleEvent(EVENT_CRUSHING_LEAP, 22000);
261  events.ScheduleEvent(EVENT_DAGGER_THROW, 10000);
262  }
263 
264  void UpdateAI(uint32 diff) override
265  {
266  if (!UpdateVictim())
267  return;
268 
269  events.Update(diff);
270  if (me->HasUnitState(UNIT_STATE_CASTING))
271  return;
272 
273  switch (events.ExecuteEvent())
274  {
275  case EVENT_CHECK_RAGE:
276  CheckRageBuff();
277  events.RepeatEvent(2000);
278  break;
279  case EVENT_BRUTAL_STRIKE:
280  me->CastSpell(me->GetVictim(), SPELL_IOCBOSS_BRUTAL_STRIKE, false);
281  events.RepeatEvent(6000);
282  break;
283  case EVENT_CRUSHING_LEAP:
284  me->CastSpell(me, SPELL_IOCBOSS_CRUSHING_LEAP, false);
285  events.RepeatEvent(22000);
286  break;
287  case EVENT_DAGGER_THROW:
288  if (Unit* tgt = SelectTarget(SELECT_TARGET_RANDOM))
289  me->CastSpell(tgt, SPELL_IOCBOSS_DAGGER_THROW, false);
290 
291  events.RepeatEvent(10000);
292  break;
293  }
294 
295  DoMeleeAttackIfReady();
296  }
