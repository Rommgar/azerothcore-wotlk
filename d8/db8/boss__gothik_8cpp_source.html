350  bool CheckGroupSplitted()
351  {
352  Map::PlayerList const& PlayerList = me->GetMap()->GetPlayers();
353  if (!PlayerList.isEmpty())
354  {
355  bool checklife = false;
356  bool checkdead = false;
357  for (const auto& i : PlayerList)
358  {
359  Player* player = i.GetSource();
360  if (player->IsAlive() &&
361  player->GetPositionX() <= POS_X_NORTH &&
362  player->GetPositionX() >= POS_X_SOUTH &&
363  player->GetPositionY() <= POS_Y_GATE &&
364  player->GetPositionY() >= POS_Y_EAST)
365  {
366  checklife = true;
367  }
368  else if (player->IsAlive() &&
369  player->GetPositionX() <= POS_X_NORTH &&
370  player->GetPositionX() >= POS_X_SOUTH &&
371  player->GetPositionY() >= POS_Y_GATE &&
372  player->GetPositionY() <= POS_Y_WEST)
373  {
374  checkdead = true;
375  }
376 
377  if (checklife && checkdead)
378  return true;
379  }
380  }
381  return false;

411  void UpdateAI(uint32 diff) override

429  Talk(SAY_INTRO_3);

431  case EVENT_INTRO_4:

435  me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_SHADOW_BOLT_10, SPELL_SHADOW_BOLT_25), false);

437  break;
438  case EVENT_HARVEST_SOUL:
439  me->CastSpell(me, SPELL_HARVEST_SOUL, false);
440  events.RepeatEvent(15000);
441  break;

443  me->AttackStop();

445  {
446  me->CastSpell(me, SPELL_TELEPORT_DEAD, false);

456  AttackStart(pTarget);
457  }
458  events.RepeatEvent(20000);
459  break;
460  case EVENT_CHECK_HEALTH:
461  if (me->HealthBelowPct(30) && pInstance)
462  {
463  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_GOTHIK_INNER_GATE)))

509  }
