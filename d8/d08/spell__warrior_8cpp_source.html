558  void CalculateAmount(AuraEffect const* aurEff, int32& amount, bool& canBeRecalculated)
559  {

561  {
562  canBeRecalculated = false;
563 
564  // $0.2 * (($MWB + $mwb) / 2 + $AP / 14 * $MWS) bonus per tick
565  float ap = caster->GetTotalAttackPowerValue(BASE_ATTACK);
566  int32 mws = caster->GetAttackTime(BASE_ATTACK);
567  float mwbMin = caster->GetWeaponDamageRange(BASE_ATTACK, MINDAMAGE);
