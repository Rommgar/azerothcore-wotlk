28 enum GameEventState
29 {
30  GAMEEVENT_NORMAL = 0, // standard game events
31  GAMEEVENT_WORLD_INACTIVE = 1, // not yet started
32  GAMEEVENT_WORLD_CONDITIONS = 2, // condition matching phase
33  GAMEEVENT_WORLD_NEXTPHASE = 3, // conditions are met, now 'length' timer to start next event
34  GAMEEVENT_WORLD_FINISHED = 4, // next events are started, unapply this one
35  GAMEEVENT_INTERNAL = 5, // never handled in update
36 };

55 struct GameEventData
56 {
57  GameEventData() { }
58  uint32 eventId;
59  time_t start{1}; // occurs after this time
60  time_t end{0}; // occurs before this time
61  time_t nextstart{0}; // after this time the follow-up events count this phase completed
62  uint32 occurence{0}; // time between end and start
63  uint32 length{0}; // length of the event (minutes) after finishing all conditions
64  HolidayIds holiday_id{HOLIDAY_NONE};
65  uint8 holidayStage;
66  GameEventState state{GAMEEVENT_NORMAL}; // state of the game event, these are saved into the game_event table on change!
67  GameEventConditionMap conditions; // conditions to finish
68  std::set<uint16 /*gameevent id*/> prerequisite_events; // events that must be completed before starting this event
69  std::string description;
70  uint8 announce; // if 0 dont announce, if 1 announce, if 2 take config value
71 
72  [[nodiscard]] bool isValid() const { return length > 0 || state > GAMEEVENT_NORMAL; }
73 };
