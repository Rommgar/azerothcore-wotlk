102 /* function to detect decryption key. */
103 int32_t libmpq__decrypt_key(uint8_t *in_buf, uint32_t in_size, uint32_t block_size, uint32_t *key) {
104 
105  /* some common variables. */
106  uint32_t saveseed1;
107 
108  /* temp = seed1 + seed2 */
109  uint32_t temp;
110  uint32_t i = 0;
111 
112  /* temp = seed1 + buffer[0x400 + (seed1 & 0xFF)] */
113  temp = (*(uint32_t *)in_buf ^ in_size) - 0xEEEEEEEE;
114 
115  /* try all 255 possibilities. */
116  for (i = 0; i < 0x100; i++) {
117 
118  /* some common variables. */
119  uint32_t seed1;
120  uint32_t seed2 = 0xEEEEEEEE;
121  uint32_t ch;
122  uint32_t ch2;
123 
124  /* try the first uint32_t's (we exactly know the value). */
125  seed1 = temp - crypt_buf[0x400 + i];
126  seed2 += crypt_buf[0x400 + (seed1 & 0xFF)];
127  ch = ((uint32_t *)in_buf)[0] ^ (seed1 + seed2);
128 
129  if (ch != in_size) {
130  continue;
131  }
132 
133  /* add one because we are decrypting block positions. */
134  saveseed1 = seed1 + 1;
135  ch2 = ch;
136 
137  /*
138  * if ok, continue and test the second value. we don't know exactly the value,
139  * but we know that the second one has lower 16 bits set to zero (no compressed
140  * block is larger than 0xFFFF bytes)
141  */
142  seed1 = ((~seed1 << 0x15) + 0x11111111) | (seed1 >> 0x0B);
143  seed2 = ch + seed2 + (seed2 << 5) + 3;
144  seed2 += crypt_buf[0x400 + (seed1 & 0xFF)];
145  ch = ((uint32_t *)in_buf)[1] ^ (seed1 + seed2);
146 
147  /* check if we found the file seed. */
148  if ((ch - ch2) <= block_size) {
149 
150  /* file seed found, so return it. */
151  *key = saveseed1;
152  return LIBMPQ_SUCCESS;
153  }
154  }
155 
156  /* if no file seed was found return with error. */
157  return LIBMPQ_ERROR_DECRYPT;
158 }

162

164  int32_t tb = 0;
165 
166  /* check if buffer is not compressed. */
167  if (compression_type == LIBMPQ_FLAG_COMPRESS_NONE) {
168 
169  /* no compressed data, so copy input buffer to output buffer. */
170  memcpy(out_buf, in_buf, out_size);
171 
172  /* store number of bytes copied. */
173  tb = out_size;
174  }
175 
176  /* check if one compression mode is used. */
177  else if (compression_type == LIBMPQ_FLAG_COMPRESS_PKZIP ||
178  compression_type == LIBMPQ_FLAG_COMPRESS_MULTI) {
179 
180  /* check if block is really compressed, some blocks have set the compression flag, but are not compressed. */
181  if (in_size < out_size) {
182 
183  /* check if we are using pkzip compression algorithm. */
184  if (compression_type == LIBMPQ_FLAG_COMPRESS_PKZIP) {
185 
186  /* decompress using pkzip. */
187  if ((tb = libmpq__decompress_pkzip(in_buf, in_size, out_buf, out_size)) < 0) {
188 
189  /* something on decompression failed. */
190  return tb;
191  }
192  }
193 
194  /* check if we are using multiple compression algorithm. */
195  else if (compression_type == LIBMPQ_FLAG_COMPRESS_MULTI) {
196 
197  /*
198  * check if it is a file compressed by blizzard's multiple compression, note that storm.dll
199  * version 1.0.9 distributed with warcraft 3 passes the full path name of the opened archive
200  * as the new last parameter.
201  */
202  if ((tb = libmpq__decompress_multi(in_buf, in_size, out_buf, out_size)) < 0) {
203 
204  /* something on decompression failed. */
205  return tb;
206  }
207  }
208  } else {
209 
210  /* block has set compression flag, but is not compressed, so copy data to output buffer. */
211  memcpy(out_buf, in_buf, out_size);
212 
213  /* save the number of transferred bytes. */
214  tb = in_size;
215  }
216  }
217 
218  /* if no error was found, return transferred bytes. */
219  return tb;
220 }
