187 bool BattlefieldWG::Update(uint32 diff)
188 {
189  bool m_return = Battlefield::Update(diff);
190  if (m_saveTimer <= diff)
191  {
192  sWorld->setWorldState(BATTLEFIELD_WG_WORLD_STATE_ACTIVE, m_isActive);
193  sWorld->setWorldState(BATTLEFIELD_WG_WORLD_STATE_DEFENDER, m_DefenderTeam);
194  sWorld->setWorldState(ClockWorldState[0], m_Timer);
195  m_saveTimer = 60 * IN_MILLISECONDS;
196  }
197  else
198  m_saveTimer -= diff;
199 
200  // Update Tenacity
201  if (IsWarTime())
202  {
203  if (m_tenacityUpdateTimer <= diff)
204  {
205  m_tenacityUpdateTimer = 10000;
206  if (!m_updateTenacityList.empty())
207  UpdateTenacity();
208  m_updateTenacityList.clear();
209  }
210  else
211  m_tenacityUpdateTimer -= diff;
212  }
213 
214  return m_return;
215 }

480 }
