35 bool WorldSession::CanOpenMailBox(ObjectGuid guid)
36 {
37  if (guid == _player->GetGUID())
38  {
39  if (_player->GetSession()->GetSecurity() < SEC_MODERATOR)
40  {
41  LOG_ERROR("network.opcode", "%s attempt open mailbox in cheating way.", _player->GetName().c_str());
42  return false;
43  }
44  }
45  else if (guid.IsGameObject())
46  {
47  if (!_player->GetGameObjectIfCanInteractWith(guid, GAMEOBJECT_TYPE_MAILBOX))
48  return false;
49  }
50  else if (guid.IsAnyTypeCreature())
51  {
52  if (!_player->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_MAILBOX))
53  return false;
54  }
55  else
56  return false;
57 
58  return true;
59 }

433  if (item)

435 
436  player->RemoveMItem(itr2->item_guid);
437  }
438  }

442  CharacterDatabase.CommitTransaction(trans);
443 
444  delete m; //we can deallocate old mail
445  player->SendMailResult(mailId, MAIL_RETURNED_TO_SENDER, MAIL_OK);
446 
447  // xinef: update global data
448  sWorld->UpdateGlobalPlayerMails(player->GetGUID().GetCounter(), -1);
449 }

593 void WorldSession::HandleGetMailList(WorldPacket& recvData)
594 {
595  ObjectGuid mailbox;
596  recvData >> mailbox;
597 
598  if (!CanOpenMailBox(mailbox))
599  return;
600 
601  Player* player = _player;
602 
603  uint8 mailsCount = 0;
604  uint32 realCount = 0;
605 
606  WorldPacket data(SMSG_MAIL_LIST_RESULT, (200)); // guess size
607  data << uint32(0); // real mail's count
608  data << uint8(0); // mail's count
609  time_t cur_time = time(nullptr);
610 
611  for (Mail const* mail : player->GetMails())
612  {
613  // prevent client storage overflow
614  if (mailsCount >= MAX_INBOX_CLIENT_CAPACITY)
615  {
616  ++realCount;
617  continue;
618  }
619 
620  // skip deleted or not delivered (deliver delay not expired) mails
621  if (mail->state == MAIL_STATE_DELETED || cur_time < mail->deliver_time)
622  {
623  continue;
624  }
625 
626  uint8 item_count = uint8(mail->items.size()); // max count is MAX_MAIL_ITEMS (12)
627 
628  size_t next_mail_size = 2 + 4 + 1 + (mail->messageType == MAIL_NORMAL ? 8 : 4) + 4 * 8 + (mail->subject.size() + 1) + (mail->body.size() + 1) + 1 + item_count * (1 + 4 + 4 + MAX_INSPECTED_ENCHANTMENT_SLOT * 3 * 4 + 4 + 4 + 4 + 4 + 4 + 4 + 1);
629 
630  if (data.wpos() + next_mail_size > MAX_NETCLIENT_PACKET_SIZE)
631  {
632  ++realCount;
633  continue;
634  }
635 
636  data << uint16(next_mail_size); // Message size
637  data << uint32(mail->messageID); // Message ID
