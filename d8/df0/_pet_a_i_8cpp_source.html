44 bool PetAI::_needToStop()
45 {
46  // This is needed for charmed creatures, as once their target was reset other effects can trigger threat
47  if (me->IsCharmed() && me->GetVictim() == me->GetCharmer())
48  return true;
49 
50  // xinef: dont allow to follow targets out of visibility range
51  if (me->GetExactDist(me->GetVictim()) > me->GetVisibilityRange() - 5.0f)
52  return true;
53 
54  // dont allow pets to follow targets far away from owner
55  if (Unit* owner = me->GetCharmerOrOwner())
56  if (owner->GetExactDist(me) >= (owner->GetVisibilityRange() - 10.0f))
57  return true;
58 
59  if (!me->_CanDetectFeignDeathOf(me->GetVictim()))
60  return true;
61 
62  return !me->CanCreatureAttack(me->GetVictim());
63 }

93 bool PetAI::_canMeleeAttack()
94 {
95  combatRange = 0.f;
96  switch (me->GetEntry())
97  {
98  case ENTRY_IMP:
99  case ENTRY_WATER_ELEMENTAL:
100  case ENTRY_WATER_ELEMENTAL_PERM:
101  {
102  for (uint8 i = 0; i < me->GetPetAutoSpellSize(); ++i)
103  {
104  uint32 spellID = me->GetPetAutoSpellOnPos(i);
105  switch (spellID)
106  {
107  case IMP_FIREBOLT_RANK_1:
108  case IMP_FIREBOLT_RANK_2:
109  case IMP_FIREBOLT_RANK_3:
110  case IMP_FIREBOLT_RANK_4:
111  case IMP_FIREBOLT_RANK_5:
112  case IMP_FIREBOLT_RANK_6:
113  case IMP_FIREBOLT_RANK_7:
114  case IMP_FIREBOLT_RANK_8:
115  case IMP_FIREBOLT_RANK_9:
116  case WATER_ELEMENTAL_WATERBOLT_1:
117  case WATER_ELEMENTAL_WATERBOLT_2:
118  {
119  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellID);
120  int32 mana = me->GetPower(POWER_MANA);
121 
122  if (mana >= spellInfo->CalcPowerCost(me, spellInfo->GetSchoolMask()))
123  {
124  combatRange = spellInfo->GetMaxRange();
125  return true;
126  }
127  }
128  default:
129  break;
130  }
131  }
132  return false;
133  }
134  default:
135  break;
136  }
137 
138  return true;

141 void PetAI::UpdateAI(uint32 diff)
142 {
143  if (!me->IsAlive() || !me->GetCharmInfo())
144  return;
145 
146  Unit* owner = me->GetCharmerOrOwner();
147 
148  if (m_updateAlliesTimer <= diff)
149  // UpdateAllies self set update timer
150  UpdateAllies();
151  else
152  m_updateAlliesTimer -= diff;
153 
154  if (me->GetVictim() && me->GetVictim()->IsAlive())
155  {
156  // is only necessary to stop casting, the pet must not exit combat
157  if (me->GetVictim()->HasBreakableByDamageCrowdControlAura(me))
158  {
159  me->InterruptNonMeleeSpells(false);
160  return;
161  }
162 
163  if (_needToStop())
164  {
165  LOG_DEBUG("entities.unit.ai", "Pet AI stopped attacking [%s]", me->GetGUID().ToString().c_str());
166  _stopAttack();
167  return;
168  }
169 
170  // Check before attacking to prevent pets from leaving stay position
171  if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
172  {
173  if (me->GetCharmInfo()->IsCommandAttack() || (me->GetCharmInfo()->IsAtStay() && me->IsWithinMeleeRange(me->GetVictim())))
174  _doMeleeAttack();
175  }
176  else
177  _doMeleeAttack();
178  }
179  else if (!me->GetCharmInfo() || (!me->GetCharmInfo()->GetForcedSpell() && !me->HasUnitState(UNIT_STATE_CASTING)))
180  {
181  if (me->HasReactState(REACT_AGGRESSIVE) || me->GetCharmInfo()->IsAtStay())
182  {
183  // Every update we need to check targets only in certain cases
184  // Aggressive - Allow auto select if owner or pet don't have a target
185  // Stay - Only pick from pet or owner targets / attackers so targets won't run by
186  // while chasing our owner. Don't do auto select.
187  // All other cases (ie: defensive) - Targets are assigned by AttackedBy(), OwnerAttackedBy(), OwnerAttacked(), etc.
188  Unit* nextTarget = SelectNextTarget(me->HasReactState(REACT_AGGRESSIVE));
189 
190  if (nextTarget)
191  AttackStart(nextTarget);
192  else
193  HandleReturnMovement();
194  }
195  else
196  HandleReturnMovement();
197  }
198 
199  // xinef: charm info must be always available
200  if (!me->GetCharmInfo())
201  return;
202 
203  // Autocast (casted only in combat or persistent spells in any state)
204  if (!me->HasUnitState(UNIT_STATE_CASTING))
205  {
206  if (owner && owner->GetTypeId() == TYPEID_PLAYER && me->GetCharmInfo()->GetForcedSpell() && me->GetCharmInfo()->GetForcedTarget())
207  {
208  owner->ToPlayer()->GetSession()->HandlePetActionHelper(me, me->GetGUID(), abs(me->GetCharmInfo()->GetForcedSpell()), ACT_ENABLED, me->GetCharmInfo()->GetForcedTarget());
209 
210  // xinef: if spell was casted properly and we are in passive mode, handle return
211  if (!me->GetCharmInfo()->GetForcedSpell() && me->HasReactState(REACT_PASSIVE))
212  {
213  if (me->HasUnitState(UNIT_STATE_CASTING))
214  {
215  me->GetMotionMaster()->Clear(false);
216  me->StopMoving();
217  }
218  else
219  _stopAttack();
220  }
221  return;
222  }
223 
224  // xinef: dont allow ghouls to cast spells below 75 energy
225  if (me->IsPet() && me->ToPet()->IsPetGhoul() && me->GetPower(POWER_ENERGY) < 75)
226  return;
227 
228  typedef std::vector<std::pair<Unit*, Spell*> > TargetSpellList;
229  TargetSpellList targetSpellStore;
230 
231  for (uint8 i = 0; i < me->GetPetAutoSpellSize(); ++i)
232  {
233  uint32 spellID = me->GetPetAutoSpellOnPos(i);
234  if (!spellID)
235  continue;
236 
237  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellID);
238  if (!spellInfo)
239  continue;
240 
241  if (me->GetCharmInfo()->GetGlobalCooldownMgr().HasGlobalCooldown(spellInfo))
242  continue;
243 
244  // check spell cooldown, this should be checked in CheckCast...
245  if (me->HasSpellCooldown(spellInfo->Id))
246  continue;
247 
248  if (spellInfo->IsPositive())
249  {
250  if (spellInfo->CanBeUsedInCombat())
251  {
252  // Check if we're in combat or commanded to attack (exlude auras with infinity duration)
253  if (!me->IsInCombat() && !me->GetCharmInfo()->IsCommandAttack() && spellInfo->GetMaxDuration() != -1)
254  continue;
255  }
256 
257  Spell* spell = new Spell(me, spellInfo, TRIGGERED_NONE);
258  spell->LoadScripts(); // xinef: load for CanAutoCast (calling CheckPetCast)
259  bool spellUsed = false;
260 
261  // Some spells can target enemy or friendly (DK Ghoul's Leap)
262  // Check for enemy first (pet then owner)
263  Unit* target = me->getAttackerForHelper();
264  if (!target && owner)
265  target = owner->getAttackerForHelper();
266 
267  if (target)
268  {
269  if (CanAttack(target) && spell->CanAutoCast(target))
270  {
271  targetSpellStore.push_back(std::make_pair(target, spell));
272  spellUsed = true;
273  }
274  }
275 
276  // No enemy, check friendly
277  if (!spellUsed)
278  {
279  for (ObjectGuid const& guid : m_AllySet)
280  {
281  Unit* ally = ObjectAccessor::GetUnit(*me, guid);
282 
283  //only buff targets that are in combat, unless the spell can only be cast while out of combat
284  if (!ally)
285  continue;
286 
287  if (spell->CanAutoCast(ally))
288  {
289  targetSpellStore.push_back(std::make_pair(ally, spell));
290  spellUsed = true;
291  break;
292  }
293  }
294  }
295 
296  // No valid targets at all
297  if (!spellUsed)
298  delete spell;
299  }
300  else if (me->GetVictim() && CanAttack(me->GetVictim(), spellInfo) && spellInfo->CanBeUsedInCombat())
301  {
302  Spell* spell = new Spell(me, spellInfo, TRIGGERED_NONE);
303  if (spell->CanAutoCast(me->GetVictim()))
304  targetSpellStore.push_back(std::make_pair(me->GetVictim(), spell));
305  else
306  delete spell;
307  }
308  }
309 
310  //found units to cast on to
311  if (!targetSpellStore.empty())
312  {
313  uint32 index = urand(0, targetSpellStore.size() - 1);
314 
315  Spell* spell = targetSpellStore[index].second;
316  Unit* target = targetSpellStore[index].first;
317 
318  targetSpellStore.erase(targetSpellStore.begin() + index);
319 
320  SpellCastTargets targets;
321  targets.SetUnitTarget(target);
322 
323  if (!me->HasInArc(M_PI, target))
324  {
325  me->SetInFront(target);
326  if (target && target->GetTypeId() == TYPEID_PLAYER)
327  me->SendUpdateToPlayer(target->ToPlayer());
328 
329  if (owner && owner->GetTypeId() == TYPEID_PLAYER)
330  me->SendUpdateToPlayer(owner->ToPlayer());
331  }
332 
333  me->AddSpellCooldown(spell->m_spellInfo->Id, 0, 0);
334 
335  spell->prepare(&targets);
336  }
337 
338  // deleted cached Spell objects
339  for (TargetSpellList::const_iterator itr = targetSpellStore.begin(); itr != targetSpellStore.end(); ++itr)
340  delete itr->second;
341  }
342 }

344 void PetAI::UpdateAllies()
345 {
346  Unit* owner = me->GetCharmerOrOwner();
347  Group* group = nullptr;
348 
349  m_updateAlliesTimer = 10 * IN_MILLISECONDS; //update friendly targets every 10 seconds, lesser checks increase performance
350 
351  if (!owner)
352  return;
353  else if (owner->GetTypeId() == TYPEID_PLAYER)
354  group = owner->ToPlayer()->GetGroup();
355 
356  //only pet and owner/not in group->ok
357  if (m_AllySet.size() == 2 && !group)
358  return;
359  //owner is in group; group members filled in already (no raid -> subgroupcount = whole count)
360  if (group && !group->isRaidGroup() && m_AllySet.size() == (group->GetMembersCount() + 2))
361  return;
362 
363  m_AllySet.clear();
364  m_AllySet.insert(me->GetGUID());
365  if (group) //add group
366  {
367  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
368  {
369  Player* Target = itr->GetSource();
370  if (!Target || !Target->IsInMap(owner) || !group->SameSubGroup(owner->ToPlayer(), Target))
371  continue;
372 
373  if (Target->GetGUID() == owner->GetGUID())
374  continue;
375 
376  m_AllySet.insert(Target->GetGUID());
377  }
378  }
379  else //remove group
380  m_AllySet.insert(owner->GetGUID());
381 }

383 void PetAI::KilledUnit(Unit* victim)
384 {
385  // Called from Unit::Kill() in case where pet or owner kills something
386  // if owner killed this victim, pet may still be attacking something else
387  if (me->GetVictim() && me->GetVictim() != victim)
388  return;
389 
390  // Xinef: if pet is channeling a spell and owner killed something different, dont interrupt it
391  if (me->HasUnitState(UNIT_STATE_CASTING) && me->GetGuidValue(UNIT_FIELD_CHANNEL_OBJECT) && me->GetGuidValue(UNIT_FIELD_CHANNEL_OBJECT) != victim->GetGUID())
392  return;
393

461 Unit* PetAI::SelectNextTarget(bool allowAutoSelect) const
462 {
463  // Provides next target selection after current target death.
464  // This function should only be called internally by the AI
465  // Targets are not evaluated here for being valid targets, that is done in _CanAttack()
466  // The parameter: allowAutoSelect lets us disable aggressive pet auto targeting for certain situations
467 
468  // Passive pets don't do next target selection
469  if (me->HasReactState(REACT_PASSIVE))
470  return nullptr;
471 
472  // Check pet attackers first so we don't drag a bunch of targets to the owner
473  if (Unit* myAttacker = me->getAttackerForHelper())
474  if (!myAttacker->HasBreakableByDamageCrowdControlAura() && me->_CanDetectFeignDeathOf(myAttacker) && me->CanCreatureAttack(myAttacker))
475  return myAttacker;
476 
477  // Check pet's attackers first to prevent dragging mobs back to owner
478  if (me->HasAuraType(SPELL_AURA_MOD_TAUNT))
479  {
480  const Unit::AuraEffectList& tauntAuras = me->GetAuraEffectsByType(SPELL_AURA_MOD_TAUNT);
481  if (!tauntAuras.empty())
482  for (Unit::AuraEffectList::const_reverse_iterator itr = tauntAuras.rbegin(); itr != tauntAuras.rend(); ++itr)
483  if (Unit* caster = (*itr)->GetCaster())
484  if (me->_CanDetectFeignDeathOf(caster) && me->CanCreatureAttack(caster) && !caster->HasAuraTypeWithCaster(SPELL_AURA_IGNORED, me->GetGUID()))
485  return caster;
486  }
487 
488  // Not sure why we wouldn't have an owner but just in case...
489  Unit* owner = me->GetCharmerOrOwner();
490  if (!owner)
491  return nullptr;
492 
493  // Check owner attackers
494  if (Unit* ownerAttacker = owner->getAttackerForHelper())
495  if (!ownerAttacker->HasBreakableByDamageCrowdControlAura() && me->_CanDetectFeignDeathOf(ownerAttacker) && me->CanCreatureAttack(ownerAttacker))
496  return ownerAttacker;
497 
498  // Check owner victim
499  // 3.0.2 - Pets now start attacking their owners victim in defensive mode as soon as the hunter does
500  if (Unit* ownerVictim = owner->GetVictim())
501  if (me->_CanDetectFeignDeathOf(ownerVictim) && me->CanCreatureAttack(ownerVictim))
502  return ownerVictim;
503 
504  // Neither pet or owner had a target and aggressive pets can pick any target
505  // To prevent aggressive pets from chain selecting targets and running off, we
506  // only select a random target if certain conditions are met.
507  if (allowAutoSelect)
508  if (!me->GetCharmInfo()->IsReturning() || me->GetCharmInfo()->IsFollowing() || me->GetCharmInfo()->IsAtStay())
509  if (Unit* nearTarget = me->ToCreature()->SelectNearestTargetInAttackDistance(MAX_AGGRO_RADIUS))
510  return nearTarget;
511 
512  // Default - no valid targets
513  return nullptr;
514 }

516 void PetAI::HandleReturnMovement()
517 {
518  // Handles moving the pet back to stay or owner
519 
520  // Prevent activating movement when under control of spells
521  // such as "Eyes of the Beast"
522  if (me->isPossessed())
523  return;
524 
525  if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
526  {
527  if (!me->GetCharmInfo()->IsAtStay() && !me->GetCharmInfo()->IsReturning())
528  {
529  if (me->GetCharmInfo()->HasStayPosition())
530  {
531  // Return to previous position where stay was clicked
532  if (me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
533  {
534  float x, y, z;
535  me->GetCharmInfo()->GetStayPosition(x, y, z);
536  ClearCharmInfoFlags();
537  me->GetCharmInfo()->SetIsReturning(true);
538  me->GetMotionMaster()->Clear();
539  me->GetMotionMaster()->MovePoint(me->GetGUID().GetCounter(), x, y, z);
540  }
541  }
542  }
543  }
544  else // COMMAND_FOLLOW
545  {
546  if (!me->GetCharmInfo()->IsFollowing() && !me->GetCharmInfo()->IsReturning())
547  {
548  if (me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
549  {
550  ClearCharmInfoFlags();
551  me->GetCharmInfo()->SetIsReturning(true);
552  me->GetMotionMaster()->Clear();
553  me->GetMotionMaster()->MoveFollow(me->GetCharmerOrOwner(), PET_FOLLOW_DIST, me->GetFollowAngle());
554  }
555  }
556  }
557 
558  me->GetCharmInfo()->SetForcedSpell(0);
559  me->GetCharmInfo()->SetForcedTargetGUID();
560 
561  // xinef: remember that npcs summoned by npcs can also be pets
562  me->DeleteThreatList();
563  me->ClearInPetCombat();
564 }

577 void PetAI::DoAttack(Unit* target, bool chase)
578 {
579  // Handles attack with or without chase and also resets flags
580  // for next update / creature kill
581 
582  if (me->Attack(target, true))
583  {
584  // xinef: properly fix fake combat after pet is sent to attack
585  if (Unit* owner = me->GetOwner())
586  owner->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PET_IN_COMBAT);
587 
588  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PET_IN_COMBAT);
589 
590  // Play sound to let the player know the pet is attacking something it picked on its own
591  if (me->HasReactState(REACT_AGGRESSIVE) && !me->GetCharmInfo()->IsCommandAttack())
592  me->SendPetAIReaction(me->GetGUID());
593 
594  if (CharmInfo* ci = me->GetCharmInfo())
595  {
596  ci->SetIsAtStay(false);
597  ci->SetIsCommandFollow(false);
598  ci->SetIsFollowing(false);
599  ci->SetIsReturning(false);
600  }
601 
602  if (chase)
603  {
604  if (_canMeleeAttack())
605  {
606  float angle = combatRange == 0.f && target->GetTypeId() != TYPEID_PLAYER && !target->IsPet() ? float(M_PI) : 0.f;
607  float tolerance = combatRange == 0.f ? float(M_PI_4) : float(M_PI * 2);
608  me->GetMotionMaster()->MoveChase(target, ChaseRange(0.f, combatRange), ChaseAngle(angle, tolerance));
609  }
610  }
611  else // (Stay && ((Aggressive || Defensive) && In Melee Range)))
612  {
613  me->GetCharmInfo()->SetIsAtStay(true);
614  me->GetMotionMaster()->MovementExpiredOnSlot(MOTION_SLOT_ACTIVE, false);
615  me->GetMotionMaster()->MoveIdle();
616  }
617  }

620 void PetAI::MovementInform(uint32 moveType, uint32 data)
621 {
622  // Receives notification when pet reaches stay or follow owner
623  switch (moveType)
624  {
625  case POINT_MOTION_TYPE:
626  {
627  // Pet is returning to where stay was clicked. data should be
628  // pet's GUIDLow since we set that as the waypoint ID
629  if (data == me->GetGUID().GetCounter() && me->GetCharmInfo()->IsReturning())
630  {
631  ClearCharmInfoFlags();
632  me->GetCharmInfo()->SetIsAtStay(true);
633  me->GetMotionMaster()->Clear();
634  me->GetMotionMaster()->MoveIdle();
635  }
636  break;
637  }
638  case FOLLOW_MOTION_TYPE:
639  {
640  // If data is owner's GUIDLow then we've reached follow point,
641  // otherwise we're probably chasing a creature
642  if (me->GetCharmerOrOwner() && me->GetCharmInfo() && data == me->GetCharmerOrOwner()->GetGUID().GetCounter() && me->GetCharmInfo()->IsReturning())
643  {
644  ClearCharmInfoFlags();
645  me->GetCharmInfo()->SetIsFollowing(true);
646  }
647  break;
648  }
649  default:
650  break;
651  }

654 bool PetAI::CanAttack(Unit* target, const SpellInfo* spellInfo)
655 {
656  // Evaluates wether a pet can attack a specific target based on CommandState, ReactState and other flags
657  // IMPORTANT: The order in which things are checked is important, be careful if you add or remove checks
658 
659  // Hmmm...
660  if (!target)
661  return false;
662 
663  if (!target->IsAlive())
664  {
665  // xinef: if target is invalid, pet should evade automaticly
666  // Clear target to prevent getting stuck on dead targets
667  //me->AttackStop();
668  //me->InterruptNonMeleeSpells(false);
669  return false;
670  }
671 
672  // xinef: check unit states of pet
673  if (me->HasUnitState(UNIT_STATE_LOST_CONTROL))
674  return false;
675 
676  // xinef: pets of mounted players have stunned flag only, check this also
677  if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED))
678  return false;
679 
680  // pussywizard: ZOMG! TEMP!
681  if (!me->GetCharmInfo())
682  {
683  LOG_INFO("misc", "PetAI::CanAttack (A1) - %u, %s", me->GetEntry(), me->GetOwnerGUID().ToString().c_str());
684  return false;
685  }
686 
687  // Passive - passive pets can attack if told to
688  if (me->HasReactState(REACT_PASSIVE))
689  return me->GetCharmInfo()->IsCommandAttack();
690 
691  // CC - mobs under crowd control can be attacked if owner commanded
692  if (target->HasBreakableByDamageCrowdControlAura() && (!spellInfo || !spellInfo->HasAttribute(SPELL_ATTR4_REACTIVE_DAMAGE_PROC)))
693  return me->GetCharmInfo()->IsCommandAttack();
694 
695  // Returning - pets ignore attacks only if owner clicked follow
696  if (me->GetCharmInfo()->IsReturning())
697  return !me->GetCharmInfo()->IsCommandFollow();
698 
699  // Stay - can attack if target is within range or commanded to
700  if (me->GetCharmInfo()->HasCommandState(COMMAND_STAY))
701  return (me->IsWithinMeleeRange(target) || me->GetCharmInfo()->IsCommandAttack());
702 
703  // Pets attacking something (or chasing) should only switch targets if owner tells them to
704  if (me->GetVictim() && me->GetVictim() != target)
705  {
706  // Check if our owner selected this target and clicked "attack"
707  Unit* ownerTarget = nullptr;
708  if (Player* owner = me->GetCharmerOrOwner()->ToPlayer())
709  ownerTarget = owner->GetSelectedUnit();
710  else
711  ownerTarget = me->GetCharmerOrOwner()->GetVictim();
712 
713  if (ownerTarget && me->GetCharmInfo()->IsCommandAttack())
714  return (target->GetGUID() == ownerTarget->GetGUID());
715  }
716 
717  // Follow
718  if (me->GetCharmInfo()->HasCommandState(COMMAND_FOLLOW))
719  return !me->GetCharmInfo()->IsReturning();

722  return false;
