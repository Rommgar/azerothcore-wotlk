31 // Checks if object meets the condition
32 // Can have CONDITION_SOURCE_TYPE_NONE && !mReferenceId if called from a special event (ie: eventAI)
33 bool Condition::Meets(ConditionSourceInfo& sourceInfo)
34 {
35  // ASSERT(ConditionTarget < MAX_CONDITION_TARGETS);
36  if (ConditionTarget >= MAX_CONDITION_TARGETS)
37  {
38  LOG_ERROR("condition", "ConditionTarget %d for for condition (Entry: %u Type: %u Group: %u) is greater or equal than MAX_CONDITION_TARGETS", ConditionTarget, SourceEntry, SourceType, SourceGroup);
39  return false;
40  }
41 
42  WorldObject* object = sourceInfo.mConditionTargets[ConditionTarget];
43  // object not present, return false
44  if (!object)
45  {
46  LOG_DEBUG("condition", "Condition object not found for condition (Entry: %u Type: %u Group: %u)", SourceEntry, SourceType, SourceGroup);
47  return false;
48  }
49  bool condMeets = false;
50  switch (ConditionType)
51  {
52  case CONDITION_NONE:
53  condMeets = true; // empty condition, always met
54  break;
55  case CONDITION_AURA:
56  {
57  if (Unit* unit = object->ToUnit())
58  condMeets = unit->HasAuraEffect(ConditionValue1, ConditionValue2);
59  break;
60  }
61  case CONDITION_ITEM:
62  {
63  if (Player* player = object->ToPlayer())
64  {
65  // don't allow 0 items (it's checked during table load)
66  ASSERT(ConditionValue2);
67  bool checkBank = !!ConditionValue3;
68  condMeets = player->HasItemCount(ConditionValue1, ConditionValue2, checkBank);
69  }
70  break;
71  }
72  case CONDITION_ITEM_EQUIPPED:
73  {
74  if (Player* player = object->ToPlayer())
75  condMeets = player->HasItemOrGemWithIdEquipped(ConditionValue1, 1);
76  break;
77  }
78  case CONDITION_ZONEID:
79  condMeets = object->GetZoneId() == ConditionValue1;
80  break;
81  case CONDITION_REPUTATION_RANK:
82  {
83  if (Player* player = object->ToPlayer())
84  {
85  if (FactionEntry const* faction = sFactionStore.LookupEntry(ConditionValue1))
86  condMeets = (ConditionValue2 & (1 << player->GetReputationMgr().GetRank(faction)));
87  }
88  break;
89  }
90  case CONDITION_ACHIEVEMENT:
91  {
92  if (Player* player = object->ToPlayer())
93  condMeets = player->HasAchieved(ConditionValue1);
94  break;
95  }
96  case CONDITION_TEAM:
97  {
98  if (Player* player = object->ToPlayer())
99  {
100  // Xinef: DB Data compatibility...
101  uint32 teamOld = player->GetTeamId() == TEAM_ALLIANCE ? ALLIANCE : HORDE;
102  condMeets = teamOld == ConditionValue1;
103  }
104  break;
105  }
106  case CONDITION_CLASS:
107  {
108  if (Unit* unit = object->ToUnit())
109  condMeets = unit->getClassMask() & ConditionValue1;
110  break;
111  }
112  case CONDITION_RACE:
113  {
114  if (Unit* unit = object->ToUnit())
115  condMeets = unit->getRaceMask() & ConditionValue1;
116  break;
117  }
118  case CONDITION_GENDER:
119  {
120  if (Player* player = object->ToPlayer())
121  condMeets = player->getGender() == ConditionValue1;
122  break;
123  }
124  case CONDITION_SKILL:
125  {
126  if (Player* player = object->ToPlayer())
127  condMeets = player->HasSkill(ConditionValue1) && player->GetBaseSkillValue(ConditionValue1) >= ConditionValue2;
128  break;
129  }
130  case CONDITION_QUESTREWARDED:
131  {
132  if (Player* player = object->ToPlayer())
133  condMeets = player->GetQuestRewardStatus(ConditionValue1);
134  break;
135  }
136  case CONDITION_QUESTTAKEN:
137  {
138  if (Player* player = object->ToPlayer())
139  {
140  QuestStatus status = player->GetQuestStatus(ConditionValue1);
141  condMeets = (status == QUEST_STATUS_INCOMPLETE);
142  }
143  break;
144  }
145  case CONDITION_QUEST_COMPLETE:
146  {
147  if (Player* player = object->ToPlayer())
148  {
149  QuestStatus status = player->GetQuestStatus(ConditionValue1);
150  condMeets = (status == QUEST_STATUS_COMPLETE && !player->GetQuestRewardStatus(ConditionValue1));
151  }
152  break;
153  }
154  case CONDITION_QUEST_NONE:
155  {
156  if (Player* player = object->ToPlayer())
157  {
158  QuestStatus status = player->GetQuestStatus(ConditionValue1);
159  condMeets = (status == QUEST_STATUS_NONE);
160  }
161  break;
162  }
163  case CONDITION_QUEST_SATISFY_EXCLUSIVE:
164  {
165  if (Player* player = object->ToPlayer())
166  {
167  // Xinef: cannot be null, checked at loading
168  const Quest* quest = sObjectMgr->GetQuestTemplate(ConditionValue1);
169  condMeets = !player->IsQuestRewarded(ConditionValue1) && player->SatisfyQuestExclusiveGroup(quest, false);
170  }
171  break;
172  }
173  case CONDITION_ACTIVE_EVENT:
174  condMeets = sGameEventMgr->IsActiveEvent(ConditionValue1);
175  break;
176  case CONDITION_INSTANCE_INFO:
177  {
178  Map* map = object->GetMap();
179  if (map->IsDungeon())
180  {
181  if (InstanceScript const* instance = map->ToInstanceMap()->GetInstanceScript())
182  {
183  switch (ConditionValue3)
184  {
185  case INSTANCE_INFO_DATA:
186  condMeets = instance->GetData(ConditionValue1) == ConditionValue2;
187  break;
188  case INSTANCE_INFO_GUID_DATA:
189  condMeets = instance->GetGuidData(ConditionValue1) == ObjectGuid(uint64(ConditionValue2));
190  break;
191  case INSTANCE_INFO_BOSS_STATE:
192  condMeets = instance->GetBossState(ConditionValue1) == EncounterState(ConditionValue2);
193  break;
194  case INSTANCE_INFO_DATA64:
195  condMeets = instance->GetData64(ConditionValue1) == ConditionValue2;
196  break;
197  }
198  }
199  }
200  break;
201  }
202  case CONDITION_MAPID:
203  condMeets = object->GetMapId() == ConditionValue1;
204  break;
205  case CONDITION_AREAID:
206  condMeets = object->GetAreaId() == ConditionValue1;
207  break;
208  case CONDITION_SPELL:
209  {
210  if (Player* player = object->ToPlayer())
211  condMeets = player->HasSpell(ConditionValue1);
212  break;
213  }
214  case CONDITION_LEVEL:
215  {
216  if (Unit* unit = object->ToUnit())
217  condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), static_cast<uint32>(unit->getLevel()), ConditionValue1);
218  break;
219  }
220  case CONDITION_DRUNKENSTATE:
221  {
222  if (Player* player = object->ToPlayer())
223  condMeets = (uint32) Player::GetDrunkenstateByValue(player->GetDrunkValue()) >= ConditionValue1;
224  break;
225  }
226  case CONDITION_NEAR_CREATURE:
227  {
228  condMeets = static_cast<bool>(GetClosestCreatureWithEntry(object, ConditionValue1, static_cast<float>(ConditionValue2), !ConditionValue3));
229  break;
230  }
231  case CONDITION_NEAR_GAMEOBJECT:
232  {
233  condMeets = static_cast<bool>(GetClosestGameObjectWithEntry(object, ConditionValue1, static_cast<float>(ConditionValue2)));
234  break;
235  }
236  case CONDITION_OBJECT_ENTRY_GUID:
237  {
238  if (ConditionValue3 == 1 && object->ToUnit()) // pussywizard: if == 1, ignore not attackable/selectable targets
239  if (object->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE))
240  break;
241 
242  if (uint32(object->GetTypeId()) == ConditionValue1)
243  {
244  condMeets = !ConditionValue2 || (object->GetEntry() == ConditionValue2);
245 
246  if (ConditionValue3 > 1)
247  {
248  switch (object->GetTypeId())
249  {
250  case TYPEID_UNIT:
251  condMeets &= object->ToCreature()->GetSpawnId() == ConditionValue3;
252  break;
253  case TYPEID_GAMEOBJECT:
254  condMeets &= object->ToGameObject()->GetSpawnId() == ConditionValue3;
255  break;
256  default:
257  break;
258  }
259  }
260  }
261  break;
262  }
263  case CONDITION_TYPE_MASK:
264  {
265  condMeets = object->isType(ConditionValue1);
266  break;
267  }
268  case CONDITION_RELATION_TO:
269  {
270  if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
271  {
272  Unit* toUnit = toObject->ToUnit();
273  Unit* unit = object->ToUnit();
274  if (toUnit && unit)
275  {
276  switch (ConditionValue2)
277  {
278  case RELATION_SELF:
279  condMeets = unit == toUnit;
280  break;
281  case RELATION_IN_PARTY:
282  condMeets = unit->IsInPartyWith(toUnit);
283  break;
284  case RELATION_IN_RAID_OR_PARTY:
285  condMeets = unit->IsInRaidWith(toUnit);
286  break;
287  case RELATION_OWNED_BY:
288  condMeets = unit->GetOwnerGUID() == toUnit->GetGUID();
289  break;
290  case RELATION_PASSENGER_OF:
291  condMeets = unit->IsOnVehicle(toUnit);
292  break;
293  case RELATION_CREATED_BY:
294  condMeets = unit->GetCreatorGUID() == toUnit->GetGUID();
295  break;
296  }
297  }
298  }
299  break;
300  }
301  case CONDITION_REACTION_TO:
302  {
303  if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
304  {
305  Unit* toUnit = toObject->ToUnit();
306  Unit* unit = object->ToUnit();
307  if (toUnit && unit)
308  condMeets = (1 << unit->GetReactionTo(toUnit)) & ConditionValue2;
309  }
310  break;
311  }
312  case CONDITION_DISTANCE_TO:
313  {
314  if (WorldObject* toObject = sourceInfo.mConditionTargets[ConditionValue1])
315  condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue3), object->GetDistance(toObject), static_cast<float>(ConditionValue2));
316  break;
317  }
318  case CONDITION_ALIVE:
319  {
320  if (Unit* unit = object->ToUnit())
321  condMeets = unit->IsAlive();
322  break;
323  }
324  case CONDITION_HP_VAL:
325  {
326  if (Unit* unit = object->ToUnit())
327  condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), unit->GetHealth(), static_cast<uint32>(ConditionValue1));
328  break;
329  }
330  case CONDITION_HP_PCT:
331  {
332  if (Unit* unit = object->ToUnit())
333  condMeets = CompareValues(static_cast<ComparisionType>(ConditionValue2), unit->GetHealthPct(), static_cast<float>(ConditionValue1));
334  break;
335  }
336  case CONDITION_WORLD_STATE:
337  {
338  condMeets = ConditionValue2 == sWorld->getWorldState(ConditionValue1);
339  break;
340  }
341  case CONDITION_PHASEMASK:
342  {
343  condMeets = object->GetPhaseMask() & ConditionValue1;
344  break;
345  }
346  case CONDITION_TITLE:
347  {
348  if (Player* player = object->ToPlayer())
349  condMeets = player->HasTitle(ConditionValue1);
350  break;
351  }
352  case CONDITION_SPAWNMASK:
353  {
354  condMeets = ((1 << object->GetMap()->GetSpawnMode()) & ConditionValue1);
355  break;
356  }
357  case CONDITION_UNIT_STATE:
358  {
359  if (Unit* unit = object->ToUnit())
360  condMeets = unit->HasUnitState(ConditionValue1);
361  break;
362  }
363  case CONDITION_CREATURE_TYPE:
364  {
365  if (Creature* creature = object->ToCreature())
366  condMeets = creature->GetCreatureTemplate()->type == ConditionValue1;
367  break;
368  }
369  case CONDITION_REALM_ACHIEVEMENT:
370  {
371  AchievementEntry const* achievement = sAchievementStore.LookupEntry(ConditionValue1);
372  if (achievement && sAchievementMgr->IsRealmCompleted(achievement))
373  condMeets = true;
374  break;
375  }
376  case CONDITION_IN_WATER:
377  {
378  if (Unit* unit = object->ToUnit())
379  condMeets = unit->IsInWater();
380  break;
381  }
382  case CONDITION_QUESTSTATE:
383  {
384  if (Player* player = object->ToPlayer())
385  {
386  uint32 queststateConditionValue1 = player->GetQuestStatus(ConditionValue1);
387  if (((ConditionValue2 & (1 << QUEST_STATUS_NONE)) && (queststateConditionValue1 == QUEST_STATUS_NONE)) || ((ConditionValue2 & (1 << QUEST_STATUS_COMPLETE)) && (queststateConditionValue1 == QUEST_STATUS_COMPLETE)) || ((ConditionValue2 & (1 << QUEST_STATUS_INCOMPLETE)) && (queststateConditionValue1 == QUEST_STATUS_INCOMPLETE)) || ((ConditionValue2 & (1 << QUEST_STATUS_FAILED)) && (queststateConditionValue1 == QUEST_STATUS_FAILED)) ||
388  ((ConditionValue2 & (1 << QUEST_STATUS_REWARDED)) && player->GetQuestRewardStatus(ConditionValue1)))
389  {
390  condMeets = true;
391  }
392  }
393  break;
394  }
395  case CONDITION_DAILY_QUEST_DONE:
396  {
397  if (Player* player = object->ToPlayer())
398  {
399  condMeets = player->IsDailyQuestDone(ConditionValue1);
400  }
401  break;
402  }
403  case CONDITION_QUEST_OBJECTIVE_PROGRESS:
404  {
405  if (Player* player = object->ToPlayer())
406  {
407  const Quest* quest = ASSERT_NOTNULL(sObjectMgr->GetQuestTemplate(ConditionValue1));
408  uint16 log_slot = player->FindQuestSlot(quest->GetQuestId());
409  if (log_slot >= MAX_QUEST_LOG_SIZE)
410  break;
411  if (player->GetQuestSlotCounter(log_slot, ConditionValue2) == ConditionValue3)
412  condMeets = true;
413  }
414  break;
415  }
416  case CONDITION_HAS_AURA_TYPE:
417  {
418  if (Unit* unit = object->ToUnit())
419  condMeets = unit->HasAuraType(AuraType(ConditionValue1));
420  break;
421  }
422  default:
423  condMeets = false;
424  break;
425  }
426 
427  if (NegativeCondition)
428  condMeets = !condMeets;
429 
430  if (!condMeets)
431  sourceInfo.mLastFailedCondition = this;
432 
433  // bool script = sScriptMgr->OnConditionCheck(this, sourceInfo); // Returns true by default. // pussywizard: optimization
434  return condMeets; // && script;
435 }

658 uint32 ConditionMgr::GetSearcherTypeMaskForConditionList(ConditionList const& conditions)
659 {
660  if (conditions.empty())
661  return GRID_MAP_TYPE_MASK_ALL;
662  // groupId, typeMask
663  std::map<uint32, uint32> ElseGroupStore;
664  for (ConditionList::const_iterator i = conditions.begin(); i != conditions.end(); ++i)
665  {
666  // no point of having not loaded conditions in list
667  ASSERT((*i)->isLoaded() && "ConditionMgr::GetSearcherTypeMaskForConditionList - not yet loaded condition found in list");
668  std::map<uint32, uint32>::const_iterator itr = ElseGroupStore.find((*i)->ElseGroup);
669  // group not filled yet, fill with widest mask possible
670  if (itr == ElseGroupStore.end())
671  ElseGroupStore[(*i)->ElseGroup] = GRID_MAP_TYPE_MASK_ALL;
672  // no point of checking anymore, empty mask
673  else if (!(*itr).second)
674  continue;

677  {

681  }
682  else // handle normal condition
683  {
684  // object will match conditions in one ElseGroupStore only when it matches all of them
685  // so, let's find a smallest possible mask which satisfies all conditions
686  ElseGroupStore[(*i)->ElseGroup] &= (*i)->GetSearcherTypeMaskForCondition();
687  }
688  }

690  // so, let's include all possible masks
691  uint32 mask = 0;
692  for (std::map<uint32, uint32>::const_iterator i = ElseGroupStore.begin(); i != ElseGroupStore.end(); ++i) mask |= i->second;
693 
694  return mask;

1190  bool found = false;
1191  for (std::list<uint32>::iterator itr = sharedMasks.begin(); itr != sharedMasks.end(); ++itr)
1192  {
1193  if ((1 << i) & *itr)
1194  {
1195  found = true;

1197  }
1198  }
1199  if (found)
1200  continue;
1201 
1202  // build new shared mask with found effect
1203  uint32 sharedMask = (1 << i);
1204  ConditionList* cmp = spellInfo->Effects[i].ImplicitTargetConditions;
1205  for (uint8 effIndex = i + 1; effIndex < MAX_SPELL_EFFECTS; ++effIndex)
1206  {
1207  if (spellInfo->Effects[effIndex].ImplicitTargetConditions == cmp)
1208  sharedMask |= 1 << effIndex;
1209  }
1210  sharedMasks.push_back(sharedMask);
1211  }
1212 
1213  for (std::list<uint32>::iterator itr = sharedMasks.begin(); itr != sharedMasks.end(); ++itr)
1214  {
1215  // some effect indexes should have same data
1216  if (uint32 commonMask = *itr & conditionEffMask)
1217  {
1218  uint8 firstEffIndex = 0;
1219  for (; firstEffIndex < MAX_SPELL_EFFECTS; ++firstEffIndex)
