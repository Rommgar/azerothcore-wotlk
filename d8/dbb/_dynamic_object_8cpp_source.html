99 bool DynamicObject::CreateDynamicObject(ObjectGuid::LowType guidlow, Unit* caster, uint32 spellId, Position const& pos, float radius, DynamicObjectType type)
100 {
101  SetMap(caster->GetMap());
102  Relocate(pos);
103  if (!IsPositionValid())
104  {
105  LOG_ERROR("dyobject", "DynamicObject (spell %u) not created. Suggested coordinates isn't valid (X: %f Y: %f)", spellId, GetPositionX(), GetPositionY());
106  return false;
107  }
108 
109  WorldObject::_Create(guidlow, HighGuid::DynamicObject, caster->GetPhaseMask());
110 
111  UpdatePositionData();
112 
113  SetEntry(spellId);
114  SetObjectScale(1);
115  SetGuidValue(DYNAMICOBJECT_CASTER, caster->GetGUID());
116 
117  // The lower word of DYNAMICOBJECT_BYTES must be 0x0001. This value means that the visual radius will be overriden
118  // by client for most of the "ground patch" visual effect spells and a few "skyfall" ones like Hurricane.
119  // If any other value is used, the client will _always_ use the radius provided in DYNAMICOBJECT_RADIUS, but
120  // precompensation is necessary (eg radius *= 2) for many spells. Anyway, blizz sends 0x0001 for all the spells
121  // I saw sniffed...
122  SetByteValue(DYNAMICOBJECT_BYTES, 0, type);
123  SetUInt32Value(DYNAMICOBJECT_SPELLID, spellId);
124  SetFloatValue(DYNAMICOBJECT_RADIUS, radius);
125  SetUInt32Value(DYNAMICOBJECT_CASTTIME, World::GetGameTimeMS());
126 
127  if (IsWorldObject())
128  setActive(true); //must before add to map to be put in world container
129 
130  if (!GetMap()->AddToMap(this, true))
131  {
132  // Returning false will cause the object to be deleted - remove from transport
133  return false;
134  }
135 
136  return true;
137 }

139 void DynamicObject::Update(uint32 p_time)
140 {
141  // caster has to be always available and in the same map
142  ASSERT(_caster);
143  ASSERT(_caster->GetMap() == GetMap());
144 
145  bool expired = false;
146 
147  if (_aura)
148  {
149  if (!_aura->IsRemoved())
150  _aura->UpdateOwner(p_time, this);
151 
152  // _aura may be set to null in Aura::UpdateOwner call
153  if (_aura && (_aura->IsRemoved() || _aura->IsExpired()))
154  expired = true;
155  }
156  else
157  {
158  if (GetDuration() > int32(p_time))
159  _duration -= p_time;
160  else
161  expired = true;
162  }
163 
164  if (expired)
165  Remove();
166  else
167  sScriptMgr->OnDynamicObjectUpdate(this, p_time);
168 }
