294 bool WorldSession::Update(uint32 diff, PacketFilter& updater)
295 {
299  if (sWorld->getBoolConfig(CONFIG_CLOSE_IDLE_CONNECTIONS) && IsConnectionIdle() && m_Socket)
300  m_Socket->CloseSocket();
301 
302  if (updater.ProcessUnsafe())
303  UpdateTimeOutTime(diff);
304 
305  HandleTeleportTimeout(updater.ProcessUnsafe());
306 
309  WorldPacket* packet = nullptr;
310 
312  bool deletePacket = true;
313  std::vector<WorldPacket*> requeuePackets;
314  uint32 processedPackets = 0;
315  time_t currentTime = time(nullptr);
316 
317  while (m_Socket && _recvQueue.next(packet, updater))
318  {
319  OpcodeClient opcode = static_cast<OpcodeClient>(packet->GetOpcode());
320  ClientOpcodeHandler const* opHandle = opcodeTable[opcode];
321 
322  METRIC_DETAILED_TIMER("worldsession_update_opcode_time", METRIC_TAG("opcode", opHandle->Name));
323 
324  try
325  {
326  switch (opHandle->Status)
327  {
328  case STATUS_LOGGEDIN:
329  if (!_player)
330  {
331  // pussywizard: such packets were sent to do something for a character that has already logged out, skip them
332  }
333  else if (!_player->IsInWorld())
334  {
335  // pussywizard: such packets may do something important and the player is just being teleported, move to the end of the queue
336  // pussywizard: previously such were skipped, so leave it as it is xD proper code below if we wish to change that
337 
338  // pussywizard: requeue only important packets not related to maps (PROCESS_THREADUNSAFE)
339  /*if (opHandle.packetProcessing == PROCESS_THREADUNSAFE)
340  {
341  if (!firstDelayedPacket)
342  firstDelayedPacket = packet;
343  deletePacket = false;
344  QueuePacket(packet);
345  }*/
346  }
347  else if (_player->IsInWorld() && AntiDOS.EvaluateOpcode(*packet, currentTime))
348  {
349  sScriptMgr->OnPacketReceive(this, *packet);
350 #ifdef ELUNA
351  if (!sEluna->OnPacketReceive(this, *packet))
352  break;
353 #endif
354  opHandle->Call(this, *packet);
355  LogUnprocessedTail(packet);
356  }
357  break;
358  case STATUS_TRANSFER:
359  if (_player && !_player->IsInWorld() && AntiDOS.EvaluateOpcode(*packet, currentTime))
360  {
361  sScriptMgr->OnPacketReceive(this, *packet);
362 #ifdef ELUNA
363  if (!sEluna->OnPacketReceive(this, *packet))
364  break;
365 #endif
366  opHandle->Call(this, *packet);
367  LogUnprocessedTail(packet);
368  }
369  break;
370  case STATUS_AUTHED:
371  if (m_inQueue) // prevent cheating
372  break;
373 
374  if (AntiDOS.EvaluateOpcode(*packet, currentTime))
375  {
376  sScriptMgr->OnPacketReceive(this, *packet);
377 #ifdef ELUNA
378  if (!sEluna->OnPacketReceive(this, *packet))
379  break;
380 #endif
381  opHandle->Call(this, *packet);
382  LogUnprocessedTail(packet);
383  }
384  break;
385  case STATUS_NEVER:
386  LOG_ERROR("network.opcode", "Received not allowed opcode %s from %s",
387  GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet->GetOpcode())).c_str(), GetPlayerInfo().c_str());
388  break;
389  case STATUS_UNHANDLED:
390  LOG_DEBUG("network.opcode", "Received not handled opcode %s from %s",
391  GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet->GetOpcode())).c_str(), GetPlayerInfo().c_str());
392  break;
393  }
394  }
395  catch (WorldPackets::InvalidHyperlinkException const& ihe)
396  {
397  LOG_ERROR("network", "%s sent %s with an invalid link:\n%s", GetPlayerInfo().c_str(),
398  GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet->GetOpcode())).c_str(), ihe.GetInvalidValue().c_str());
399 
400  if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_KICK))
401  {
402  KickPlayer("WorldSession::Update Invalid chat link");
403  }
404  }
405  catch (WorldPackets::IllegalHyperlinkException const& ihe)
406  {
407  LOG_ERROR("network", "%s sent %s which illegally contained a hyperlink:\n%s", GetPlayerInfo().c_str(),
408  GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet->GetOpcode())).c_str(), ihe.GetInvalidValue().c_str());
409 
410  if (sWorld->getIntConfig(CONFIG_CHAT_STRICT_LINK_CHECKING_KICK))
411  {
412  KickPlayer("WorldSession::Update Illegal chat link");
413  }
414  }
415  catch (WorldPackets::PacketArrayMaxCapacityException const& pamce)
416  {
417  LOG_ERROR("network", "PacketArrayMaxCapacityException: %s while parsing %s from %s.",
418  pamce.what(), GetOpcodeNameForLogging(static_cast<OpcodeClient>(packet->GetOpcode())).c_str(), GetPlayerInfo().c_str());
419  }
420  catch (ByteBufferException const&)
421  {
422  LOG_ERROR("network", "WorldSession::Update ByteBufferException occured while parsing a packet (opcode: %u) from client %s, accountid=%i. Skipped packet.", packet->GetOpcode(), GetRemoteAddress().c_str(), GetAccountId());
423  if (sLog->ShouldLog("network", LogLevel::LOG_LEVEL_DEBUG))
424  {
425  LOG_DEBUG("network", "Dumping error causing packet:");
426  packet->hexlike();
427  }
428  }
429 
430  if (deletePacket)
431  delete packet;
432 
433  deletePacket = true;
434 
435 #define MAX_PROCESSED_PACKETS_IN_SAME_WORLDSESSION_UPDATE 150
436  processedPackets++;
437 
438  //process only a max amout of packets in 1 Update() call.
439  //Any leftover will be processed in next update
440  if (processedPackets > MAX_PROCESSED_PACKETS_IN_SAME_WORLDSESSION_UPDATE)
441  break;
442  }
443 
444  _recvQueue.readd(requeuePackets.begin(), requeuePackets.end());
445 
446  METRIC_VALUE("processed_packets", processedPackets);
447 
448  if (!updater.ProcessUnsafe()) // <=> updater is of type MapSessionFilter
449  {
450  // Send time sync packet every 10s.
451  if (_timeSyncTimer > 0)
452  {
453  if (diff >= _timeSyncTimer)
454  {
455  SendTimeSync();
456  }
457  else
458  {
459  _timeSyncTimer -= diff;
460  }
461  }
462  }
463 
464  ProcessQueryCallbacks();
465 
466  //check if we are safe to proceed with logout
467  //logout procedure should happen only in World::UpdateSessions() method!!!
468  if (updater.ProcessUnsafe())
469  {
470  if (m_Socket && m_Socket->IsOpen() && _warden)
471  {
472  _warden->Update(diff);
473  }
474 
475  if (ShouldLogOut(currentTime) && !m_playerLoading)
476  {
477  LogoutPlayer(true);
478  }
479 
480  if (m_Socket && !m_Socket->IsOpen())
481  {
482  if (GetPlayer() && _warden)
483  _warden->Update(diff);
484 
485  m_Socket = nullptr;
486  }
487 
488  if (!m_Socket)
489  {
490  return false;
491  }
492  }
493 
494  return true;
495 }
