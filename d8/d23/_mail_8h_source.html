196  bool RemoveItem(ObjectGuid::LowType item_guid)
197  {
198  for (MailItemInfoVec::iterator itr = items.begin(); itr != items.end(); ++itr)
199  {
200  if (itr->item_guid == item_guid)
201  {
202  items.erase(itr);
203  return true;
204  }
205  }
206  return false;
207  }
