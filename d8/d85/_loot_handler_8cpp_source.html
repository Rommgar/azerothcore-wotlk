36 void WorldSession::HandleAutostoreLootItemOpcode(WorldPacket& recvData)
37 {
38  LOG_DEBUG("network", "WORLD: CMSG_AUTOSTORE_LOOT_ITEM");
39  Player* player = GetPlayer();
40  ObjectGuid lguid = player->GetLootGUID();
41  Loot* loot = nullptr;
42  uint8 lootSlot = 0;
43 
44  recvData >> lootSlot;
45 
46  if (lguid.IsGameObject())
47  {
48  GameObject* go = player->GetMap()->GetGameObject(lguid);
49  // xinef: cheating protection
50  //if (player->GetGroup() && player->GetGroup()->GetLootMethod() == MASTER_LOOT && player->GetGUID() != player->GetGroup()->GetMasterLooterGuid())
51  // go = nullptr;
52 
53  // not check distance for GO in case owned GO (fishing bobber case, for example) or Fishing hole GO
54  if (!go || ((go->GetOwnerGUID() != _player->GetGUID() && go->GetGoType() != GAMEOBJECT_TYPE_FISHINGHOLE) && !go->IsWithinDistInMap(_player)))
55  {
56  player->SendLootRelease(lguid);
57  return;
58  }
59 
60  loot = &go->loot;
61  }
62  else if (lguid.IsItem())
63  {
64  Item* pItem = player->GetItemByGuid(lguid);
65 
66  if (!pItem)
67  {
68  player->SendLootRelease(lguid);
69  return;
70  }
71 
72  loot = &pItem->loot;
73  }
74  else if (lguid.IsCorpse())
75  {
76  Corpse* bones = ObjectAccessor::GetCorpse(*player, lguid);
77  if (!bones)
78  {
79  player->SendLootRelease(lguid);
80  return;
81  }
82 
83  loot = &bones->loot;
84  }
85  else
86  {
87  Creature* creature = GetPlayer()->GetMap()->GetCreature(lguid);
88 
89  bool lootAllowed = creature && creature->IsAlive() == (player->getClass() == CLASS_ROGUE && creature->loot.loot_type == LOOT_PICKPOCKETING);
90  if (!lootAllowed || !creature->IsWithinDistInMap(_player, INTERACTION_DISTANCE))
91  {
92  player->SendLootError(lguid, lootAllowed ? LOOT_ERROR_TOO_FAR : LOOT_ERROR_DIDNT_KILL);
93  return;
94  }
95 
96  loot = &creature->loot;
97  }
98 
99  player->StoreLootItem(lootSlot, loot);
100 
101  // If player is removing the last LootItem, delete the empty container.
102  if (loot->isLooted() && lguid.IsItem())
103  DoLootRelease(lguid);
104 }

106 void WorldSession::HandleLootMoneyOpcode(WorldPacket& /*recvData*/)
107 {
108  LOG_DEBUG("network", "WORLD: CMSG_LOOT_MONEY");
109 
110  Player* player = GetPlayer();
111  ObjectGuid guid = player->GetLootGUID();
112  if (!guid)
113  return;
114 
115  Loot* loot = nullptr;
116  bool shareMoney = true;
117 
118  switch (guid.GetHigh())
119  {
120  case HighGuid::GameObject:
121  {
122  GameObject* go = GetPlayer()->GetMap()->GetGameObject(guid);
123 
124  // do not check distance for GO if player is the owner of it (ex. fishing bobber)
125  if (go && ((go->GetOwnerGUID() == player->GetGUID() || go->IsWithinDistInMap(player))))
126  {
127  loot = &go->loot;
128  }
129 
130  break;
131  }
132  case HighGuid::Corpse: // remove insignia ONLY in BG
133  {
134  Corpse* bones = ObjectAccessor::GetCorpse(*player, guid);
135 
136  if (bones && bones->IsWithinDistInMap(player, INTERACTION_DISTANCE))
137  {
138  loot = &bones->loot;
139  shareMoney = false;
140  }
141 
142  break;
143  }
144  case HighGuid::Item:
145  {
146  if (Item* item = player->GetItemByGuid(guid))
147  {
148  loot = &item->loot;
149  shareMoney = false;
150  }
151  break;
152  }
153  case HighGuid::Unit:
154  case HighGuid::Vehicle:
155  {
156  Creature* creature = player->GetMap()->GetCreature(guid);
157  bool lootAllowed = creature && creature->IsAlive() == (player->getClass() == CLASS_ROGUE && creature->loot.loot_type == LOOT_PICKPOCKETING);
158  if (lootAllowed && creature->IsWithinDistInMap(player, INTERACTION_DISTANCE))
159  {
160  loot = &creature->loot;
161  if (creature->IsAlive())
162  shareMoney = false;
163  }
164  else
165  player->SendLootError(guid, lootAllowed ? LOOT_ERROR_TOO_FAR : LOOT_ERROR_DIDNT_KILL);
166  break;
167  }
168  default:
169  return; // unlootable type
170  }
171 
172  if (loot)
173  {
174  loot->NotifyMoneyRemoved();
175  if (shareMoney && player->GetGroup()) //item, pickpocket and players can be looted only single player
176  {
177  Group* group = player->GetGroup();
178 
179  std::vector<Player*> playersNear;
180  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
181  {
182  Player* member = itr->GetSource();
183  if (!member)
184  continue;
185 
186  if (player->IsAtLootRewardDistance(member))
187  playersNear.push_back(member);
188  }
189 
190  uint32 goldPerPlayer = uint32((loot->gold) / (playersNear.size()));
191 
192  for (std::vector<Player*>::const_iterator i = playersNear.begin(); i != playersNear.end(); ++i)
193  {
194  (*i)->ModifyMoney(goldPerPlayer);
195  (*i)->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_LOOT_MONEY, goldPerPlayer);
196 
197  WorldPacket data(SMSG_LOOT_MONEY_NOTIFY, 4 + 1);
198  data << uint32(goldPerPlayer);
199  data << uint8(playersNear.size() > 1 ? 0 : 1); // Controls the text displayed in chat. 0 is "Your share is..." and 1 is "You loot..."
200  (*i)->GetSession()->SendPacket(&data);
201  }
202  }
203  else
204  {
205  player->ModifyMoney(loot->gold);
206  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_LOOT_MONEY, loot->gold);
207 
208  WorldPacket data(SMSG_LOOT_MONEY_NOTIFY, 4 + 1);
209  data << uint32(loot->gold);
210  data << uint8(1); // "You loot..."
211  SendPacket(&data);
212  }
213 #ifdef ELUNA
214  sEluna->OnLootMoney(player, loot->gold);
215 #endif
216  loot->gold = 0;
217 
218  // Delete the money loot record from the DB
219  if (loot->containerGUID)
220  sLootItemStorage->RemoveStoredLootMoney(loot->containerGUID, loot);
221 
222  // Delete container if empty
223  if (loot->isLooted() && guid.IsItem())
224  DoLootRelease(guid);
225  }
226 }

260 void WorldSession::DoLootRelease(ObjectGuid lguid)
261 {
262  Player* player = GetPlayer();
263  Loot* loot;
264 
265  player->SetLootGUID(ObjectGuid::Empty);
266  player->SendLootRelease(lguid);
267 
268  player->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_LOOTING);
269 
270  if (!player->IsInWorld())
271  return;
272 
273  if (lguid.IsGameObject())
274  {
275  GameObject* go = GetPlayer()->GetMap()->GetGameObject(lguid);
276 
277  // not check distance for GO in case owned GO (fishing bobber case, for example) or Fishing hole GO
278  if (!go || ((go->GetOwnerGUID() != _player->GetGUID() && go->GetGoType() != GAMEOBJECT_TYPE_FISHINGHOLE) && !go->IsWithinDistInMap(_player)))
279  {
280  return;
281  }
282 
283  loot = &go->loot;
284 
285  if (go->GetGoType() == GAMEOBJECT_TYPE_DOOR)
286  {
287  // locked doors are opened with spelleffect openlock, prevent remove its as looted
288  go->UseDoorOrButton();
289  }
290  else if (loot->isLooted() || go->GetGoType() == GAMEOBJECT_TYPE_FISHINGNODE)
291  {
292  if (go->GetGoType() == GAMEOBJECT_TYPE_FISHINGHOLE)
293  {
294  // The fishing hole used once more
295  go->AddUse(); // if the max usage is reached, will be despawned in next tick
296  if (go->GetUseCount() >= go->GetGOValue()->FishingHole.MaxOpens)
297  go->SetLootState(GO_JUST_DEACTIVATED);
298  else
299  go->SetLootState(GO_READY);
300  }
301  else
302  {
303  go->SetLootState(GO_JUST_DEACTIVATED);
304 
305  // Xinef: moved event execution to loot release (after everything is looted)
306  // Xinef: 99% sure that this worked like this on blizz
307  // Xinef: prevents exploits with just opening GO and spawning bilions of npcs, which can crash core if you know what you're doin ;)
308  if (go->GetGoType() == GAMEOBJECT_TYPE_CHEST && go->GetGOInfo()->chest.eventId)
309  {
310  LOG_DEBUG("spells.aura", "Chest ScriptStart id %u for GO %u", go->GetGOInfo()->chest.eventId, go->GetSpawnId());
311  player->GetMap()->ScriptsStart(sEventScripts, go->GetGOInfo()->chest.eventId, player, go);
312  }
313  }
314 
315  loot->clear();
316  }
317  else
318  {
319  // not fully looted object
320  go->SetLootState(GO_ACTIVATED, player);
321 
322  // if the round robin player release, reset it.
323  if (player->GetGUID() == loot->roundRobinPlayer)
324  loot->roundRobinPlayer.Clear();
325  }
326  }
327  else if (lguid.IsCorpse()) // ONLY remove insignia at BG
328  {
329  Corpse* corpse = ObjectAccessor::GetCorpse(*player, lguid);
330  if (!corpse || !corpse->IsWithinDistInMap(_player, INTERACTION_DISTANCE))
331  return;
332 
333  loot = &corpse->loot;
334 
335  // Xinef: Buggs client? (Opening loot after closing)
336  //if (loot->isLooted())
337  {
338  loot->clear();
339  corpse->RemoveFlag(CORPSE_FIELD_DYNAMIC_FLAGS, CORPSE_DYNFLAG_LOOTABLE);
340  }
341  }
342  else if (lguid.IsItem())
343  {
344  Item* pItem = player->GetItemByGuid(lguid);
345  if (!pItem)
346  return;
347 
348  loot = &pItem->loot;
349  ItemTemplate const* proto = pItem->GetTemplate();
350 
351  // destroy only 5 items from stack in case prospecting and milling
352  if (proto->Flags & (ITEM_FLAG_IS_PROSPECTABLE | ITEM_FLAG_IS_MILLABLE))
353  {
354  pItem->m_lootGenerated = false;
355  pItem->loot.clear();
356 
357  uint32 count = pItem->GetCount();
358 
359  // >=5 checked in spell code, but will work for cheating cases also with removing from another stacks.
360  if (count > 5)
361  count = 5;
362 
363  player->DestroyItemCount(pItem, count, true);
364  }
365  else if (pItem->loot.isLooted() || !(proto->Flags & ITEM_FLAG_HAS_LOOT))
366  {
367  player->DestroyItem(pItem->GetBagSlot(), pItem->GetSlot(), true);
368  return;
369  }
370  }
371  else
372  {
373  Creature* creature = GetPlayer()->GetMap()->GetCreature(lguid);
374 
375  bool lootAllowed = creature && creature->IsAlive() == (player->getClass() == CLASS_ROGUE && creature->loot.loot_type == LOOT_PICKPOCKETING);
376  if (!lootAllowed || !creature->IsWithinDistInMap(_player, INTERACTION_DISTANCE))
377  return;
378 
379  loot = &creature->loot;
380  if (loot->isLooted())
381  {
382  // skip pickpocketing loot for speed, skinning timer reduction is no-op in fact
383  if (!creature->IsAlive())
384  creature->AllLootRemovedFromCorpse();
385 
386  creature->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_LOOTABLE);
387  loot->clear();
388  }
389  else
390  {
391  // if the round robin player release, reset it.
392  if (player->GetGUID() == loot->roundRobinPlayer)
393  {
394  loot->roundRobinPlayer.Clear();
395 
396  if (Group* group = player->GetGroup())
397  group->SendLooter(creature, nullptr);
398  }
399  // force dynflag update to update looter and lootable info
400  creature->ForceValuesUpdateAtIndex(UNIT_DYNAMIC_FLAGS);
401  }
402  }
403 
404  //Player is not looking at loot list, he doesn't need to see updates on the loot list
405  loot->RemoveLooter(player->GetGUID());
406 }

408 void WorldSession::HandleLootMasterGiveOpcode(WorldPacket& recvData)
409 {
410  uint8 slotid;
411  ObjectGuid lootguid, target_playerguid;
412 
413  recvData >> lootguid >> slotid >> target_playerguid;
414 
415  if (!_player->GetGroup() || _player->GetGroup()->GetMasterLooterGuid() != _player->GetGUID() || _player->GetGroup()->GetLootMethod() != MASTER_LOOT)
416  {
417  _player->SendLootError(lootguid, LOOT_ERROR_DIDNT_KILL);
418  return;
419  }
420 
421  Player* target = ObjectAccessor::GetPlayer(*_player, target_playerguid);
422  if (!target)
423  {
424  _player->SendLootError(lootguid, LOOT_ERROR_PLAYER_NOT_FOUND);
425  return;
426  }
427 
428  LOG_DEBUG("network", "WorldSession::HandleLootMasterGiveOpcode (CMSG_LOOT_MASTER_GIVE, 0x02A3) Target = [%s].", target->GetName().c_str());
429 
430  if (_player->GetLootGUID() != lootguid)
431  {
432  _player->SendLootError(lootguid, LOOT_ERROR_DIDNT_KILL);
433  return;
434  }
435 
436  if (!_player->IsInRaidWith(target))
437  {
438  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_OTHER);
439  //LOG_DEBUG("network", "MasterLootItem: Player %s tried to give an item to ineligible player %s !", GetPlayer()->GetName().c_str(), target->GetName().c_str());
440  return;
441  }
442 
443  Loot* loot = nullptr;
444 
445  if (GetPlayer()->GetLootGUID().IsCreatureOrVehicle())
446  {
447  Creature* creature = GetPlayer()->GetMap()->GetCreature(lootguid);
448  if (!creature)
449  return;
450 
451  loot = &creature->loot;
452  }
453  else if (GetPlayer()->GetLootGUID().IsGameObject())
454  {
455  GameObject* pGO = GetPlayer()->GetMap()->GetGameObject(lootguid);
456  if (!pGO)
457  return;
458 
459  loot = &pGO->loot;
460  }
461 
462  if (!loot)
463  return;
464 
465  if (slotid >= loot->items.size() + loot->quest_items.size())
466  {
467  LOG_DEBUG("loot", "MasterLootItem: Player %s might be using a hack! (slot %d, size %lu)", GetPlayer()->GetName().c_str(), slotid, (unsigned long)loot->items.size());
468  return;
469  }
470 
471  LootItem& item = slotid >= loot->items.size() ? loot->quest_items[slotid - loot->items.size()] : loot->items[slotid];
472 
473  ItemPosCountVec dest;
474  InventoryResult msg = target->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, item.itemid, item.count);
475  if (!item.AllowedForPlayer(target, true))
476  msg = EQUIP_ERR_YOU_CAN_NEVER_USE_THAT_ITEM;
477  if (msg != EQUIP_ERR_OK)
478  {
479  if (msg == EQUIP_ERR_CANT_CARRY_MORE_OF_THIS)
480  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_UNIQUE_ITEM);
481  else if (msg == EQUIP_ERR_INVENTORY_FULL)
482  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_INV_FULL);
483  else
484  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_OTHER);
485 
486  return;
487  }
488 
489  // list of players allowed to receive this item in trade
490  AllowedLooterSet looters = item.GetAllowedLooters();
491 
492  // not move item from loot to target inventory
493  Item* newitem = target->StoreNewItem(dest, item.itemid, true, item.randomPropertyId, looters);
494  target->SendNewItem(newitem, uint32(item.count), false, false, true);
495  target->UpdateLootAchievements(&item, loot);
496 
497 #ifdef ELUNA
498  sEluna->OnLootItem(target, newitem, item.count, lootguid);
499 #endif
500 
501  // mark as looted
502  item.count = 0;
503  item.is_looted = true;
504 
505  loot->NotifyItemRemoved(slotid);
506  --loot->unlootedCount;
507 }
