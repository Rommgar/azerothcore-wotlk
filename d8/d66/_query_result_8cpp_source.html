27 static uint32 SizeForType(MYSQL_FIELD* field)
28 {
29  switch (field->type)
30  {
31  case MYSQL_TYPE_NULL:
32  return 0;
33  case MYSQL_TYPE_TINY:
34  return 1;
35  case MYSQL_TYPE_YEAR:
36  case MYSQL_TYPE_SHORT:
37  return 2;
38  case MYSQL_TYPE_INT24:
39  case MYSQL_TYPE_LONG:
40  case MYSQL_TYPE_FLOAT:
41  return 4;
42  case MYSQL_TYPE_DOUBLE:
43  case MYSQL_TYPE_LONGLONG:
44  case MYSQL_TYPE_BIT:
45  return 8;
46 
47  case MYSQL_TYPE_TIMESTAMP:
48  case MYSQL_TYPE_DATE:
49  case MYSQL_TYPE_TIME:
50  case MYSQL_TYPE_DATETIME:
51  return sizeof(MYSQL_TIME);
52 
53  case MYSQL_TYPE_TINY_BLOB:
54  case MYSQL_TYPE_MEDIUM_BLOB:
55  case MYSQL_TYPE_LONG_BLOB:
56  case MYSQL_TYPE_BLOB:
57  case MYSQL_TYPE_STRING:
58  case MYSQL_TYPE_VAR_STRING:
59  return field->max_length + 1;
60 
61  case MYSQL_TYPE_DECIMAL:
62  case MYSQL_TYPE_NEWDECIMAL:
63  return 64;
64 
65  case MYSQL_TYPE_GEOMETRY:
66  /*
67  Following types are not sent over the wire:
68  MYSQL_TYPE_ENUM:
69  MYSQL_TYPE_SET:
70  */
71  default:
72  LOG_WARN("sql.sql", "SQL::SizeForType(): invalid field type %u", uint32(field->type));
73  return 0;
74  }
75 }

182 PreparedResultSet::PreparedResultSet(MySQLStmt* stmt, MySQLResult* result, uint64 rowCount, uint32 fieldCount) :
183 m_rowCount(rowCount),
184 m_rowPosition(0),
185 m_fieldCount(fieldCount),
186 m_rBind(nullptr),
187 m_stmt(stmt),
188 m_metadataResult(result)
189 {
190  if (!m_metadataResult)
191  return;
192 
193  if (m_stmt->bind_result_done)
194  {
195  delete[] m_stmt->bind->length;
196  delete[] m_stmt->bind->is_null;
197  }
198 
199  m_rBind = new MySQLBind[m_fieldCount];
200 
201  //- for future readers wondering where the fuck this is freed - mysql_stmt_bind_result moves pointers to these
202  // from m_rBind to m_stmt->bind and it is later freed by the `if (m_stmt->bind_result_done)` block just above here
203  // MYSQL_STMT lifetime is equal to connection lifetime
204  MySQLBool* m_isNull = new MySQLBool[m_fieldCount];
205  unsigned long* m_length = new unsigned long[m_fieldCount];
206 
207  memset(m_isNull, 0, sizeof(MySQLBool) * m_fieldCount);
208  memset(m_rBind, 0, sizeof(MySQLBind) * m_fieldCount);
209  memset(m_length, 0, sizeof(unsigned long) * m_fieldCount);
210 
211  //- This is where we store the (entire) resultset
212  if (mysql_stmt_store_result(m_stmt))
213  {
214  LOG_WARN("sql.sql", "%s:mysql_stmt_store_result, cannot bind result from MySQL server. Error: %s", __FUNCTION__, mysql_stmt_error(m_stmt));
215  delete[] m_rBind;
216  delete[] m_isNull;
217  delete[] m_length;
218  return;
219  }
220 
221  m_rowCount = mysql_stmt_num_rows(m_stmt);
222 
223  //- This is where we prepare the buffer based on metadata
224  MySQLField* field = reinterpret_cast<MySQLField*>(mysql_fetch_fields(m_metadataResult));
225  m_fieldMetadata.resize(m_fieldCount);
226  std::size_t rowSize = 0;
227  for (uint32 i = 0; i < m_fieldCount; ++i)
228  {
229  uint32 size = SizeForType(&field[i]);
230  rowSize += size;
231 
232  InitializeDatabaseFieldMetadata(&m_fieldMetadata[i], &field[i], i);
233 
234  m_rBind[i].buffer_type = field[i].type;
235  m_rBind[i].buffer_length = size;
236  m_rBind[i].length = &m_length[i];
237  m_rBind[i].is_null = &m_isNull[i];
238  m_rBind[i].error = nullptr;
239  m_rBind[i].is_unsigned = field[i].flags & UNSIGNED_FLAG;
240  }
241 
242  char* dataBuffer = new char[rowSize * m_rowCount];
243  for (uint32 i = 0, offset = 0; i < m_fieldCount; ++i)
244  {
245  m_rBind[i].buffer = dataBuffer + offset;
246  offset += m_rBind[i].buffer_length;
247  }
248 
249  //- This is where we bind the bind the buffer to the statement
250  if (mysql_stmt_bind_result(m_stmt, m_rBind))
251  {
252  LOG_WARN("sql.sql", "%s:mysql_stmt_bind_result, cannot bind result from MySQL server. Error: %s", __FUNCTION__, mysql_stmt_error(m_stmt));
253  mysql_stmt_free_result(m_stmt);
254  CleanUp();
255  delete[] m_isNull;
256  delete[] m_length;
257  return;
258  }
259 
260  m_rows.resize(uint32(m_rowCount) * m_fieldCount);
261  while (_NextRow())
262  {
263  for (uint32 fIndex = 0; fIndex < m_fieldCount; ++fIndex)
264  {
265  m_rows[uint32(m_rowPosition) * m_fieldCount + fIndex].SetMetadata(&m_fieldMetadata[fIndex]);
266 
267  unsigned long buffer_length = m_rBind[fIndex].buffer_length;
268  unsigned long fetched_length = *m_rBind[fIndex].length;
269  if (!*m_rBind[fIndex].is_null)
270  {
271  void* buffer = m_stmt->bind[fIndex].buffer;
272  switch (m_rBind[fIndex].buffer_type)
273  {
274  case MYSQL_TYPE_TINY_BLOB:
275  case MYSQL_TYPE_MEDIUM_BLOB:
276  case MYSQL_TYPE_LONG_BLOB:
277  case MYSQL_TYPE_BLOB:
278  case MYSQL_TYPE_STRING:
279  case MYSQL_TYPE_VAR_STRING:
280  // warning - the string will not be null-terminated if there is no space for it in the buffer
281  // when mysql_stmt_fetch returned MYSQL_DATA_TRUNCATED
282  // we cannot blindly null-terminate the data either as it may be retrieved as binary blob and not specifically a string
283  // in this case using Field::GetCString will result in garbage
284  // TODO: remove Field::GetCString and use std::string_view in C++17
285  if (fetched_length < buffer_length)
286  *((char*)buffer + fetched_length) = '\0';
287  break;
288  default:
289  break;
290  }
291 
292  m_rows[uint32(m_rowPosition) * m_fieldCount + fIndex].SetByteValue(
293  (char const*)buffer,
294  fetched_length);
295 
296  // move buffer pointer to next part
297  m_stmt->bind[fIndex].buffer = (char*)buffer + rowSize;
298  }
299  else
300  {
301  m_rows[uint32(m_rowPosition) * m_fieldCount + fIndex].SetByteValue(
302  nullptr,
303  *m_rBind[fIndex].length);
304  }
305  }
306  m_rowPosition++;
307  }
308  m_rowPosition = 0;
309 
311  mysql_stmt_free_result(m_stmt);
312 }

324 bool ResultSet::NextRow()
325 {
326  MYSQL_ROW row;
327 
328  if (!_result)
329  return false;
330 
331  row = mysql_fetch_row(_result);
332  if (!row)
333  {
334  CleanUp();
335  return false;
336  }
337 
338  unsigned long* lengths = mysql_fetch_lengths(_result);
339  if (!lengths)
340  {
341  LOG_WARN("sql.sql", "%s:mysql_fetch_lengths, cannot retrieve value lengths. Error %s.", __FUNCTION__, mysql_error(_result->handle));
342  CleanUp();
343  return false;
344  }
345 
346  for (uint32 i = 0; i < _fieldCount; i++)
347  _currentRow[i].SetStructuredValue(row[i], lengths[i]);
348 
349  return true;
350 }
