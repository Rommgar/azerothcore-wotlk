78 uint8 _SpellScript::EffectHook::GetAffectedEffectsMask(SpellInfo const* spellEntry)
79 {
80  uint8 mask = 0;
81  if ((effIndex == EFFECT_ALL) || (effIndex == EFFECT_FIRST_FOUND))
82  {
83  for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
84  {
85  if ((effIndex == EFFECT_FIRST_FOUND) && mask)
86  return mask;
87  if (CheckEffect(spellEntry, i))
88  mask |= (uint8)1 << i;
89  }
90  }
91  else
92  {
93  if (CheckEffect(spellEntry, effIndex))
94  mask |= (uint8)1 << effIndex;
95  }
96  return mask;
97 }

237 bool SpellScript::TargetHook::CheckEffect(SpellInfo const* spellEntry, uint8 effIndex)
238 {
239  if (!targetType)
240  return false;
241 
242  if (spellEntry->Effects[effIndex].TargetA.GetTarget() != targetType &&
243  spellEntry->Effects[effIndex].TargetB.GetTarget() != targetType)
244  return false;
245 
246  SpellImplicitTargetInfo targetInfo(targetType);
247  switch (targetInfo.GetSelectionCategory())
248  {
249  case TARGET_SELECT_CATEGORY_CHANNEL: // SINGLE
250  return !area;
251  case TARGET_SELECT_CATEGORY_NEARBY: // BOTH
252  return true;
253  case TARGET_SELECT_CATEGORY_CONE: // AREA
254  case TARGET_SELECT_CATEGORY_AREA: // AREA
255  return area;
256  case TARGET_SELECT_CATEGORY_DEFAULT:
257  switch (targetInfo.GetObjectType())
258  {
259  case TARGET_OBJECT_TYPE_SRC: // EMPTY
260  return false;
261  case TARGET_OBJECT_TYPE_DEST: // DEST
262  return dest;
263  default:
264  switch (targetInfo.GetReferenceType())
265  {
266  case TARGET_REFERENCE_TYPE_CASTER: // SINGLE
267  return !area;
268  case TARGET_REFERENCE_TYPE_TARGET: // BOTH
269  return true;
270  default:
271  break;
272  }
273  break;
274  }
275  break;
276  default:
277  break;
278  }
279 
280  return false;
281 }
