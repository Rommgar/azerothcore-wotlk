209  void UpdateAI(uint32 diff) override
210  {
211  if( !UpdateVictim() )
212  return;
213 
214  if( !berserk && (me->GetPositionX() < 1720.0f || me->GetPositionX() > 1940.0f || me->GetPositionY() < 20.0f || me->GetPositionY() > 210.0f) )
215  events.RescheduleEvent(EVENT_BERSERK, 1);
216 
217  events.Update(diff);
218 
219  if( me->HasUnitState(UNIT_STATE_CASTING) )
220  return;
221 
222  switch( events.ExecuteEvent() )
223  {
224  case 0:
225  break;
226  case EVENT_BERSERK:
227  berserk = true;
228  me->CastSpell(me, SPELL_VEZAX_BERSERK, true);
229  me->Yell(TEXT_VEZAX_BERSERK, LANG_UNIVERSAL);
230  me->PlayDirectSound(SOUND_VEZAX_BERSERK, 0);
231  break;
232  case EVENT_SPELL_VEZAX_SHADOW_CRASH:
233  {
234  events.RepeatEvent(10000);
235 
236  std::vector<Player*> players;
237  Map::PlayerList const& pl = me->GetMap()->GetPlayers();
238  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
239  {
240  Player* temp = itr->GetSource();
241  if( temp->IsAlive() && temp->GetDistance(me) > 15.0f )
242  players.push_back(temp);
243  }
244  if (!players.empty())
245  {
246  me->setAttackTimer(BASE_ATTACK, 2000);
247  Player* target = players.at(urand(0, players.size() - 1));
248  me->SetGuidValue(UNIT_FIELD_TARGET, target->GetGUID());
249  me->CastSpell(target, SPELL_VEZAX_SHADOW_CRASH, false);
250  events.ScheduleEvent(EVENT_RESTORE_TARGET, 750);
251  }
252  }
253  break;
254  case EVENT_RESTORE_TARGET:
255  if (me->GetVictim())
256  me->SetGuidValue(UNIT_FIELD_TARGET, me->GetVictim()->GetGUID());
257  break;
258  case EVENT_SPELL_SEARING_FLAMES:
259  if(!me->HasAura(SPELL_SARONITE_BARRIER))
260  me->CastSpell(me->GetVictim(), SPELL_SEARING_FLAMES, false);
261  events.RepeatEvent( me->GetMap()->Is25ManRaid() ? 8000 : 15000 );
262  break;
263  case EVENT_SPELL_SURGE_OF_DARKNESS:
264  me->Yell(TEXT_VEZAX_SURGE, LANG_UNIVERSAL);
265  me->PlayDirectSound(SOUND_VEZAX_SURGE, 0);
266  me->CastSpell(me, SPELL_SURGE_OF_DARKNESS, false);
267  events.RepeatEvent(63000);
268  events.DelayEvents(10000, 1);
269  break;
270  case EVENT_SPELL_MARK_OF_THE_FACELESS:
271  {
272  std::vector<Player*> outside;
273  std::vector<Player*> inside;
274  Map::PlayerList const& pl = me->GetMap()->GetPlayers();
275  for( Map::PlayerList::const_iterator itr = pl.begin(); itr != pl.end(); ++itr )
276  if( Player* tmp = itr->GetSource() )
277  if( tmp->IsAlive() )
278  {
279  if( tmp->GetDistance(me) > 15.0f )
280  outside.push_back(tmp);
281  else
282  inside.push_back(tmp);
283  }
284 
285  Player* t = nullptr;
286  if( outside.size() >= uint8(me->GetMap()->Is25ManRaid() ? 9 : 4) )
287  t = outside.at(urand(0, outside.size() - 1));
288  else if( !inside.empty() )
289  t = inside.at(urand(0, inside.size() - 1));
290 
291  if (t)
292  me->CastSpell(t, SPELL_MARK_OF_THE_FACELESS_AURA, false);
293 
294  events.RepeatEvent(40000);
295  }
296  break;
297  case EVENT_SPELL_SUMMON_SARONITE_VAPORS:
298  {
299  vaporsCount++;
300  me->CastSpell(me, SPELL_SUMMON_SARONITE_VAPORS, false);
301  me->TextEmote("A cloud of saronite vapors coalesces nearby!", nullptr, true);
302 
303  if( vaporsCount < 6 || !hardmodeAvailable )
304  events.RepeatEvent(30000);
305  else
306  {
307  for (ObjectGuid const& guid : summons)
308  if (Creature* sv = ObjectAccessor::GetCreature(*me, guid))
309  {
310  sv->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
311  sv->GetMotionMaster()->MoveIdle();
312  sv->GetMotionMaster()->MoveCharge(1852.78f, 81.38f, 342.461f, 28.0f);
313  }
314 
315  events.DelayEvents(12000, 0);
316  events.DelayEvents(12000, 1);
317  events.ScheduleEvent(EVENT_SARONITE_VAPORS_SWIRL, 6000);
318  }
319  }
320  break;
321  case EVENT_SARONITE_VAPORS_SWIRL:
322  if (summons.size())
323  {
324  me->TextEmote("The saronite vapors mass and swirl violently, merging into a monstrous form!", nullptr, true);
325  if( Creature* sv = ObjectAccessor::GetCreature(*me, *(summons.begin())) )
326  sv->CastSpell(sv, SPELL_SARONITE_ANIMUS_FORMATION_VISUAL, true);
327 
328  events.ScheduleEvent(EVENT_SPELL_SUMMON_SARONITE_ANIMUS, 2000);
329  break;
330  }
331  break;
332  case EVENT_SPELL_SUMMON_SARONITE_ANIMUS:
333  if (summons.size())
334  {
335  me->TextEmote("A saronite barrier appears around General Vezax!", nullptr, true);
336  me->Yell(TEXT_VEZAX_HARDMODE, LANG_UNIVERSAL);
337  me->PlayDirectSound(SOUND_VEZAX_HARDMODE, 0);
338 
339  me->CastSpell(me, SPELL_SARONITE_BARRIER, true);
340  if( Creature* sv = ObjectAccessor::GetCreature(*me, *(summons.begin())) )
341  sv->CastSpell(sv, SPELL_SUMMON_SARONITE_ANIMUS, true);
342 
343  events.ScheduleEvent(EVENT_DESPAWN_SARONITE_VAPORS, 2500);
344  break;
345  }
346  break;
347  case EVENT_DESPAWN_SARONITE_VAPORS:
348  summons.DespawnEntry(NPC_SARONITE_VAPORS);
349  break;
350  }
351 
352  DoMeleeAttackIfReady();
353  }
