25 /*
26  Copy contents of an IO_CACHE to a file.
27 
28  SYNOPSIS
29  my_b_copy_to_file()
30  cache IO_CACHE to copy from
31  file File to copy to
32 
33  DESCRIPTION
34  Copy the contents of the cache to the file. The cache will be
35  re-inited to a read cache and will read from the beginning of the
36  cache.
37 
38  If a failure to write fully occurs, the cache is only copied
39  partially.
40 
41  TODO
42  Make this function solid by handling partial reads from the cache
43  in a correct manner: it should be atomic.
44 
45  RETURN VALUE
46  0 All OK
47  1 An error occured
48 */
49 int
50 my_b_copy_to_file(IO_CACHE *cache, FILE *file)
51 {
52  size_t bytes_in_cache;
53  DBUG_ENTER("my_b_copy_to_file");
54 
55  /* Reinit the cache to read from the beginning of the cache */
56  if (reinit_io_cache(cache, READ_CACHE, 0L, FALSE, FALSE))
57  DBUG_RETURN(1);
58  bytes_in_cache= my_b_bytes_in_cache(cache);
59  do
60  {
61  if (my_fwrite(file, cache->read_pos, bytes_in_cache,
62  MYF(MY_WME | MY_NABP)) == (size_t) -1)
63  DBUG_RETURN(1);
64  cache->read_pos= cache->read_end;
65  } while ((bytes_in_cache= my_b_fill(cache)));
66  DBUG_RETURN(0);
67 }

70 my_off_t my_b_append_tell(IO_CACHE* info)
71 {
72  /*
73  Sometimes we want to make sure that the variable is not put into
74  a register in debugging mode so we can see its value in the core
75  */
76 #ifndef DBUG_OFF
77 # define dbug_volatile volatile
78 #else
79 # define dbug_volatile
80 #endif
81 
82  /*
83  Prevent optimizer from putting res in a register when debugging
84  we need this to be able to see the value of res when the assert fails
85  */
86  dbug_volatile my_off_t res;
87 
88  /*
89  We need to lock the append buffer mutex to keep flush_io_cache()
90  from messing with the variables that we need in order to provide the
91  answer to the question.
92  */
93  mysql_mutex_lock(&info->append_buffer_lock);
94 
95 #ifndef DBUG_OFF
96  /*
97  Make sure EOF is where we think it is. Note that we cannot just use
98  my_tell() because we have a reader thread that could have left the
99  file offset in a non-EOF location
100  */
101  {
102  volatile my_off_t save_pos;
103  save_pos = my_tell(info->file,MYF(0));
104  my_seek(info->file,(my_off_t)0,MY_SEEK_END,MYF(0));
105  /*
106  Save the value of my_tell in res so we can see it when studying coredump
107  */
108  DBUG_ASSERT(info->end_of_file - (info->append_read_pos-info->write_buffer)
109  == (res=my_tell(info->file,MYF(0))));
110  my_seek(info->file,save_pos,MY_SEEK_SET,MYF(0));
111  }
112 #endif
113  res = info->end_of_file + (info->write_pos-info->append_read_pos);
114  mysql_mutex_unlock(&info->append_buffer_lock);
115  return res;
116 }

130 void my_b_seek(IO_CACHE *info,my_off_t pos)
131 {
132  my_off_t offset;
133  DBUG_ENTER("my_b_seek");
134  DBUG_PRINT("enter",("pos: %lu", (ulong) pos));
135 
136  /*
137  TODO:
138  Verify that it is OK to do seek in the non-append
139  area in SEQ_READ_APPEND cache
140  a) see if this always works
141  b) see if there is a better way to make it work
142  */
143  if (info->type == SEQ_READ_APPEND)
144  (void) flush_io_cache(info);
145 
146  offset=(pos - info->pos_in_file);
147 
148  if (info->type == READ_CACHE || info->type == SEQ_READ_APPEND)
149  {
150  /* TODO: explain why this works if pos < info->pos_in_file */
151  if ((ulonglong) offset < (ulonglong) (info->read_end - info->buffer))
152  {
153  /* The read is in the current buffer; Reuse it */
154  info->read_pos = info->buffer + offset;
155  DBUG_VOID_RETURN;
156  }
157  else
158  {
159  /* Force a new read on next my_b_read */
160  info->read_pos=info->read_end=info->buffer;
161  }
162  }
163  else if (info->type == WRITE_CACHE)
164  {
165  /* If write is in current buffer, reuse it */
166  if ((ulonglong) offset <
167  (ulonglong) (info->write_end - info->write_buffer))
168  {
169  info->write_pos = info->write_buffer + offset;
170  DBUG_VOID_RETURN;
171  }
172  (void) flush_io_cache(info);
173  /* Correct buffer end so that we write in increments of IO_SIZE */
174  info->write_end=(info->write_buffer+info->buffer_length-
175  (pos & (IO_SIZE-1)));
176  }
177  info->pos_in_file=pos;
178  info->seek_not_done=1;
179  DBUG_VOID_RETURN;
180 }

185 
186  NOTES
187  This assumes that you have already used all characters in the CACHE,

196 size_t my_b_fill(IO_CACHE *info)
197 {
198  my_off_t pos_in_file=(info->pos_in_file+
199  (size_t) (info->read_end - info->buffer));
200  size_t diff_length, length, max_length;
201 
202  if (info->seek_not_done)
203  { /* File touched, do seek */
204  if (my_seek(info->file,pos_in_file,MY_SEEK_SET,MYF(0)) ==
205  MY_FILEPOS_ERROR)
206  {
207  info->error= 0;
208  return 0;
209  }
210  info->seek_not_done=0;
211  }
212  diff_length=(size_t) (pos_in_file & (IO_SIZE-1));
213  max_length=(info->read_length-diff_length);
214  if (max_length >= (info->end_of_file - pos_in_file))
215  max_length= (size_t) (info->end_of_file - pos_in_file);
216 
217  if (!max_length)
218  {
219  info->error= 0;
220  return 0; /* EOF */
221  }
222  if ((length= my_read(info->file,info->buffer,max_length,
223  info->myflags)) == (size_t) -1)
224  {
225  info->error= -1;
226  return 0;
227  }
228  info->read_pos=info->buffer;
229  info->read_end=info->buffer+length;
230  info->pos_in_file=pos_in_file;
231  return length;
232 }

235 /*
236  Read a string ended by '\n' into a buffer of 'max_length' size.
237  Returns number of characters read, 0 on error.
238  last byte is set to '\0'
239  If buffer is full then to[max_length-1] will be set to \0.
240 */

242 size_t my_b_gets(IO_CACHE *info, char *to, size_t max_length)
243 {
244  char *start = to;
245  size_t length;
246  max_length--; /* Save place for end \0 */
247 
248  /* Calculate number of characters in buffer */
249  if (!(length= my_b_bytes_in_cache(info)) &&
250  !(length= my_b_fill(info)))
251  return 0;
252 
253  for (;;)
254  {
255  uchar *pos, *end;
256  if (length > max_length)
257  length=max_length;
258  for (pos=info->read_pos,end=pos+length ; pos < end ;)
259  {
260  if ((*to++ = *pos++) == '\n')
261  {
262  info->read_pos=pos;
263  *to='\0';
264  return (size_t) (to-start);
265  }
266  }
267  if (!(max_length-=length))
268  {
269  /* Found enough charcters; Return found string */
270  info->read_pos=pos;
271  *to='\0';
272  return (size_t) (to-start);
273  }
274  if (!(length=my_b_fill(info)))
275  return 0;
276  }
277 }

308 {

311  uint minimum_width_sign;
312  uint precision; /* as yet unimplemented for anything but %b */
313  my_bool is_zero_padded;
314 
315  /*
316  Store the location of the beginning of a format directive, for the
317  case where we learn we shouldn't have been parsing a format string
318  at all, and we don't want to lose the flag/precision/width/size
319  information.
320  */
321  const char* backtrack;
322 
323  for (; *fmt != '\0'; fmt++)
324  {
325  /* Copy everything until '%' or end of string */
326  const char *start=fmt;
327  size_t length;
328 
329  for (; (*fmt != '\0') && (*fmt != '%'); fmt++) ;
330 
331  length= (size_t) (fmt - start);
332  out_length+=length;
333  if (my_b_write(info, (const uchar*) start, length))
334  goto err;
335 
336  if (*fmt == '\0') /* End of format */
337  return out_length;
338 
339  /*
340  By this point, *fmt must be a percent; Keep track of this location and
341  skip over the percent character.
342  */
343  DBUG_ASSERT(*fmt == '%');
344  backtrack= fmt;
345  fmt++;
346 
347  is_zero_padded= FALSE;
348  minimum_width_sign= 1;
349  minimum_width= 0;
350  precision= 0;
351  /* Skip if max size is used (to be compatible with printf) */
352 
353 process_flags:
354  switch (*fmt)
355  {
356  case '-':
357  minimum_width_sign= -1; fmt++; goto process_flags;
358  case '0':
359  is_zero_padded= TRUE; fmt++; goto process_flags;
360  case '#': fmt++; goto process_flags;
362  case ' ': fmt++; goto process_flags;
364  case '+': fmt++; goto process_flags;
366  }
367 
368  if (*fmt == '*')
369  {
370  precision= (int) va_arg(args, int);
371  fmt++;
372  }
373  else
374  {
375  while (my_isdigit(&my_charset_latin1, *fmt)) {
376  minimum_width=(minimum_width * 10) + (*fmt - '0');
377  fmt++;
378  }
379  }
380  minimum_width*= minimum_width_sign;
381 
382  if (*fmt == '.')
383  {
384  fmt++;
385  if (*fmt == '*') {
386  precision= (int) va_arg(args, int);
387  fmt++;
388  }
389  else
390  {
391  while (my_isdigit(&my_charset_latin1, *fmt)) {
392  precision=(precision * 10) + (*fmt - '0');
393  fmt++;
394  }
395  }
396  }
397 
398  if (*fmt == 's') /* String parameter */
399  {
400  reg2 char *par = va_arg(args, char *);
401  size_t length2 = strlen(par);
402  /* TODO: implement precision */
403  out_length+= length2;
404  if (my_b_write(info, (uchar*) par, length2))
405  goto err;
406  }
407  else if (*fmt == 'b') /* Sized buffer parameter, only precision makes sense */
408  {
409  char *par = va_arg(args, char *);
410  out_length+= precision;
411  if (my_b_write(info, (uchar*) par, precision))
412  goto err;
413  }
414  else if (*fmt == 'd' || *fmt == 'u') /* Integer parameter */
415  {
416  register int iarg;
417  size_t length2;
418  char buff[17];
419 
420  iarg = va_arg(args, int);
421  if (*fmt == 'd')
422  length2= (size_t) (int10_to_str((long) iarg,buff, -10) - buff);
423  else
424  length2= (uint) (int10_to_str((long) (uint) iarg,buff,10)- buff);
425 
426  /* minimum width padding */
427  if (minimum_width > length2)
428  {
429  char *buffz;
430 
431  buffz= my_alloca(minimum_width - length2);
432  if (is_zero_padded)
433  memset(buffz, '0', minimum_width - length2);
434  else
435  memset(buffz, ' ', minimum_width - length2);
436  my_b_write(info, buffz, minimum_width - length2);
437  my_afree(buffz);
438  }
439 
440  out_length+= length2;
441  if (my_b_write(info, (uchar*) buff, length2))
442  goto err;
443  }
444  else if ((*fmt == 'l' && fmt[1] == 'd') || fmt[1] == 'u')
445  /* long parameter */
446  {
447  register long iarg;
448  size_t length2;
449  char buff[17];
450 
451  iarg = va_arg(args, long);
452  if (*++fmt == 'd')
453  length2= (size_t) (int10_to_str(iarg,buff, -10) - buff);
454  else
455  length2= (size_t) (int10_to_str(iarg,buff,10)- buff);
456  out_length+= length2;
457  if (my_b_write(info, (uchar*) buff, length2))
458  goto err;
459  }
460  else
461  {
462  /* %% or unknown code */
463  if (my_b_write(info, (uchar*) backtrack, (size_t) (fmt-backtrack)))
464  goto err;
465  out_length+= fmt-backtrack;
466  }
467  }
468  return out_length;
469

471  return (size_t) -1;
