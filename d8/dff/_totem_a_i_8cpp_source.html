58 void TotemAI::UpdateAI(uint32 /*diff*/)
59 {
60  if (me->ToTotem()->GetTotemType() != TOTEM_ACTIVE)
61  return;
62 
63  if (!me->IsAlive() || me->IsNonMeleeSpellCast(false))
64  return;
65 
66  // Search spell
67  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(me->ToTotem()->GetSpell());
68  if (!spellInfo)
69  return;
70 
71  // Get spell range
72  float max_range = spellInfo->GetMaxRange(false);
73 
74  // SPELLMOD_RANGE not applied in this place just because not existence range mods for attacking totems
75 
76  // pointer to appropriate target if found any
77  Unit* victim = i_victimGuid ? ObjectAccessor::GetUnit(*me, i_victimGuid) : nullptr;
78 
79  // Search victim if no, not attackable, or out of range, or friendly (possible in case duel end)
80  if (!victim ||
81  !victim->isTargetableForAttack(true, me) || !me->IsWithinDistInMap(victim, max_range) ||
82  me->IsFriendlyTo(victim) || !me->CanSeeOrDetect(victim))
83  {
84  victim = nullptr;
85  Acore::NearestAttackableUnitInObjectRangeCheck u_check(me, me, max_range);
86  Acore::UnitLastSearcher<Acore::NearestAttackableUnitInObjectRangeCheck> checker(me, victim, u_check);
87  Cell::VisitAllObjects(me, checker, max_range);
88  }
89 
90  if (!victim && me->GetCharmerOrOwnerOrSelf()->IsInCombat())
91  {
92  victim = me->GetCharmerOrOwnerOrSelf()->getAttackerForHelper();
93  }
94 
95  // If have target
96  if (victim)
97  {
98  // remember
99  i_victimGuid = victim->GetGUID();
100 
101  // attack
102  me->SetInFront(victim); // client change orientation by self
103  me->CastSpell(victim, me->ToTotem()->GetSpell(), false);
104  }
105  else
106  i_victimGuid.Clear();
107 }
