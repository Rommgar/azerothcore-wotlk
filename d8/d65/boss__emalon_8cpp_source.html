160  void UpdateAI(uint32 diff) override
161  {
162  if (!UpdateVictim())
163  return;
164 
165  events.Update(diff);
166  if (me->HasUnitState(UNIT_STATE_CASTING))
167  return;
168 
169  switch (events.ExecuteEvent())
170  {
171  case EVENT_CHAIN_LIGHTNING:
172  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
173  me->CastSpell(target, RAID_MODE(SPELL_CHAIN_LIGHTNING_10, SPELL_CHAIN_LIGHTNING_25), false);
174  events.RepeatEvent(25000);
175  break;
176  case EVENT_LIGHTNING_NOVA:
177  me->CastSpell(me, RAID_MODE(SPELL_LIGHTNING_NOVA_10, SPELL_LIGHTNING_NOVA_25), false);
178  events.RepeatEvent(40000);
179  break;
180  case EVENT_OVERCHARGE:
181  if (!summons.empty())
182  me->CastCustomSpell(SPELL_OVERCHARGE, SPELLVALUE_MAX_TARGETS, 1, me, true);
183  Talk(EMOTE_OVERCHARGE);
184  events.RepeatEvent(40000);
185  break;
186  case EVENT_BERSERK:
187  me->CastSpell(me, SPELL_BERSERK, true);
188  Talk(EMOTE_BERSERK);
189  break;
190  case EVENT_SUMMON_NEXT_MINION:
191  me->SummonCreature(NPC_TEMPEST_MINION, TempestMinions[urand(0, 3)], TEMPSUMMON_CORPSE_DESPAWN, 0);
192  break;
193  default:
194  break;
195  }
196 
197  DoMeleeAttackIfReady();
198  }
199  };
