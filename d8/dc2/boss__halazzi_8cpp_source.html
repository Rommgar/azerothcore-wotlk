142  void EnterPhase(PhaseHalazzi NextPhase)
143  {
144  switch (NextPhase)
145  {
146  case PHASE_LYNX:
147  case PHASE_ENRAGE:
148  if (Phase == PHASE_MERGE)
149  {
150  DoCast(me, SPELL_TRANSFORM_MERGE, true);
151  me->Attack(me->GetVictim(), true);
152  me->GetMotionMaster()->MoveChase(me->GetVictim());
153  }
154  if (Creature* Lynx = ObjectAccessor::GetCreature(*me, LynxGUID))
155  Lynx->DisappearAndDie();
156  me->SetMaxHealth(600000);
157  me->SetHealth(600000 - 150000 * TransformCount);
158  FrenzyTimer = 16000;
159  SaberlashTimer = 20000;
160  ShockTimer = 10000;
161  TotemTimer = 12000;
162  break;
163  case PHASE_SPLIT:
164  Talk(SAY_SPLIT);
165  DoCast(me, SPELL_TRANSFORM_SPLIT, true);
166  break;
167  case PHASE_HUMAN:
168  //DoCast(me, SPELL_SUMMON_LYNX, true);
169  DoSpawnCreature(NPC_SPIRIT_LYNX, 5, 5, 0, 0, TEMPSUMMON_CORPSE_DESPAWN, 0);
170  me->SetMaxHealth(400000);
171  me->SetHealth(400000);
172  ShockTimer = 10000;
173  TotemTimer = 12000;
174  break;
175  case PHASE_MERGE:
176  if (Unit* pLynx = ObjectAccessor::GetUnit(*me, LynxGUID))
177  {
178  Talk(SAY_MERGE);
179  pLynx->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
180  pLynx->GetMotionMaster()->Clear();
181  pLynx->GetMotionMaster()->MoveFollow(me, 0, 0);
182  me->GetMotionMaster()->Clear();
183  me->GetMotionMaster()->MoveFollow(pLynx, 0, 0);
184  ++TransformCount;
185  }
186  break;
187  default:
188  break;
189  }
190  Phase = NextPhase;
191  }

193  void UpdateAI(uint32 diff) override
194  {
195  if (!UpdateVictim())
196  return;
197 
198  if (BerserkTimer <= diff)
199  {
200  Talk(SAY_BERSERK);
201  DoCast(me, SPELL_BERSERK, true);
202  BerserkTimer = 60000;
203  }
204  else BerserkTimer -= diff;
205 
206  if (Phase == PHASE_LYNX || Phase == PHASE_ENRAGE)
207  {
208  if (SaberlashTimer <= diff)
209  {
210  // A tank with more than 490 defense skills should receive no critical hit
211  //DoCast(me, 41296, true);
212  DoCastVictim(SPELL_SABER_LASH, true);
213  //me->RemoveAurasDueToSpell(41296);
214  SaberlashTimer = 30000;
215  }
216  else SaberlashTimer -= diff;
217 
218  if (FrenzyTimer <= diff)
219  {
220  DoCast(me, SPELL_FRENZY);
221  FrenzyTimer = urand(10000, 15000);
222  }
223  else FrenzyTimer -= diff;
224 
225  if (Phase == PHASE_LYNX)
226  {
227  if (CheckTimer <= diff)
228  {
229  if (HealthBelowPct(25 * (3 - TransformCount)))
230  EnterPhase(PHASE_SPLIT);
231  CheckTimer = 1000;
232  }
233  else CheckTimer -= diff;
234  }
235  }
236 
237  if (Phase == PHASE_HUMAN || Phase == PHASE_ENRAGE)
238  {
239  if (TotemTimer <= diff)
240  {
241  DoCast(me, SPELL_SUMMON_TOTEM);
242  TotemTimer = 20000;
243  }
244  else TotemTimer -= diff;
245 
246  if (ShockTimer <= diff)
247  {
248  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
249  {
250  if (target->IsNonMeleeSpellCast(false))
251  DoCast(target, SPELL_EARTHSHOCK);
252  else
253  DoCast(target, SPELL_FLAMESHOCK);
254  ShockTimer = urand(10000, 15000);
255  }
256  }
257  else ShockTimer -= diff;
258 
259  if (Phase == PHASE_HUMAN)
260  {
261  if (CheckTimer <= diff)
262  {
263  if (!HealthAbovePct(20) /*HealthBelowPct(10)*/)
264  EnterPhase(PHASE_MERGE);
265  else
266  {
267  Unit* Lynx = ObjectAccessor::GetUnit(*me, LynxGUID);
268  if (Lynx && !Lynx->HealthAbovePct(20) /*Lynx->HealthBelowPct(10)*/)
269  EnterPhase(PHASE_MERGE);
270  }
271  CheckTimer = 1000;
272  }
273  else CheckTimer -= diff;
274  }
275  }
276 
277  if (Phase == PHASE_MERGE)
278  {
279  if (CheckTimer <= diff)
280  {
281  Unit* Lynx = ObjectAccessor::GetUnit(*me, LynxGUID);
282  if (Lynx)
283  {
284  Lynx->GetMotionMaster()->MoveFollow(me, 0, 0);
285  me->GetMotionMaster()->MoveFollow(Lynx, 0, 0);
286  if (me->IsWithinDistInMap(Lynx, 6.0f))
287  {
288  if (TransformCount < 3)

291  EnterPhase(PHASE_ENRAGE);
