35 float ThreatCalcHelper::calcThreat(Unit* hatedUnit, Unit* /*hatingUnit*/, float threat, SpellSchoolMask schoolMask, SpellInfo const* threatSpell)
36 {
37  if (threatSpell)
38  {
39  if (SpellThreatEntry const* threatEntry = sSpellMgr->GetSpellThreatEntry(threatSpell->Id))
40  if (threatEntry->pctMod != 1.0f)
41  threat *= threatEntry->pctMod;
42 
43  // Energize is not affected by Mods
44  for (uint8 i = 0; i < MAX_SPELL_EFFECTS; i++)
45  if (threatSpell->Effects[i].Effect == SPELL_EFFECT_ENERGIZE || threatSpell->Effects[i].ApplyAuraName == SPELL_AURA_PERIODIC_ENERGIZE)
46  return threat;
47 
48  if (Player* modOwner = hatedUnit->GetSpellModOwner())
49  modOwner->ApplySpellMod(threatSpell->Id, SPELLMOD_THREAT, threat);
50  }
51 
52  return hatedUnit->ApplyTotalThreatModifier(threat, schoolMask);
53 }

55 bool ThreatCalcHelper::isValidProcess(Unit* hatedUnit, Unit* hatingUnit, SpellInfo const* threatSpell)
56 {
57  //function deals with adding threat and adding players and pets into ThreatList
58  //mobs, NPCs, guards have ThreatList and HateOfflineList
59  //players and pets have only InHateListOf
60  //HateOfflineList is used co contain unattackable victims (in-flight, in-water, GM etc.)
61 
62  if (!hatedUnit || !hatingUnit)
63  return false;
64 
65  // not to self
66  if (hatedUnit == hatingUnit)
67  return false;
68 
69  // not to GM
70  if (hatedUnit->GetTypeId() == TYPEID_PLAYER && hatedUnit->ToPlayer()->IsGameMaster())
71  return false;
72 
73  // not to dead and not for dead
74  if (!hatedUnit->IsAlive() || !hatingUnit->IsAlive())
75  return false;
76 
77  // not in same map or phase
78  if (!hatedUnit->IsInMap(hatingUnit) || !hatedUnit->InSamePhase(hatingUnit))
79  return false;
80 
81  // spell not causing threat
82  if (threatSpell && threatSpell->HasAttribute(SPELL_ATTR1_NO_THREAT))
83  return false;
84 
85  ASSERT(hatingUnit->GetTypeId() == TYPEID_UNIT);
86 
87  return true;
88 }

178 void HostileReference::updateOnlineStatus()
179 {
180  bool online = false;
181 
182  if (!isValid())
183  if (Unit* target = ObjectAccessor::GetUnit(*GetSourceUnit(), getUnitGuid()))
184  link(target, GetSource());
185 
186  // only check for online status if
187  // ref is valid
188  // target is no player or not gamemaster
189  // target is not in flight
190  if (isValid()
191  && (getTarget()->GetTypeId() != TYPEID_PLAYER || !getTarget()->ToPlayer()->IsGameMaster())
192  && !getTarget()->IsInFlight()
193  && getTarget()->IsInMap(GetSourceUnit())
194  && getTarget()->InSamePhase(GetSourceUnit())
195  )
196  {
197  Creature* creature = GetSourceUnit()->ToCreature();
198  online = getTarget()->isInAccessiblePlaceFor(creature);
199  if (!online)
200  {
201  if (creature->IsWithinCombatRange(getTarget(), creature->m_CombatDistance))
202  online = true; // not accessible but stays online
203  }
204  }
205 
206  setOnlineOfflineState(online);

545 void ThreatMgr::processThreatEvent(ThreatRefStatusChangeEvent* threatRefStatusChangeEvent)
546 {
547  threatRefStatusChangeEvent->setThreatMgr(this); // now we can set the threat manager
548 
549  HostileReference* hostilRef = threatRefStatusChangeEvent->getReference();
550 
551  switch (threatRefStatusChangeEvent->getType())
552  {
553  case UEV_THREAT_REF_THREAT_CHANGE:
554  if ((getCurrentVictim() == hostilRef && threatRefStatusChangeEvent->getFValue() < 0.0f) ||
555  (getCurrentVictim() != hostilRef && threatRefStatusChangeEvent->getFValue() > 0.0f))
556  setDirty(true); // the order in the threat list might have changed
557  break;
558  case UEV_THREAT_REF_ONLINE_STATUS:
559  if (!hostilRef->isOnline())
560  {
561  if (hostilRef == getCurrentVictim())
562  {
563  setCurrentVictim(nullptr);
564  setDirty(true);
565  }
566  if (GetOwner() && GetOwner()->IsInWorld())
567  if (Unit* target = ObjectAccessor::GetUnit(*GetOwner(), hostilRef->getUnitGuid()))
568  if (GetOwner()->IsInMap(target))
569  GetOwner()->SendRemoveFromThreatListOpcode(hostilRef);
570  iThreatContainer.remove(hostilRef);
571  iThreatOfflineContainer.addReference(hostilRef);
572  }
573  else
574  {
575  if (getCurrentVictim() && hostilRef->getThreat() > (1.1f * getCurrentVictim()->getThreat()))
576  setDirty(true);
577  iThreatContainer.addReference(hostilRef);
578  iThreatOfflineContainer.remove(hostilRef);
579  }
580  break;
581  case UEV_THREAT_REF_REMOVE_FROM_LIST:
582  if (hostilRef == getCurrentVictim())
583  {
584  setCurrentVictim(nullptr);
585  setDirty(true);
586  }
587  iOwner->SendRemoveFromThreatListOpcode(hostilRef);
588  if (hostilRef->isOnline())
589  iThreatContainer.remove(hostilRef);
590  else
591  iThreatOfflineContainer.remove(hostilRef);
592  break;
593  }
