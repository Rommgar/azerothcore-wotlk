85 /*static*/ void Acore::Impl::ChatCommands::ChatCommandNode::LoadCommandMap()
86 {
87  InvalidateCommandMap();
88  LoadCommandsIntoMap(nullptr, COMMAND_MAP, sScriptMgr->GetChatCommands());
89 
90  if (PreparedQueryResult result = WorldDatabase.Query(WorldDatabase.GetPreparedStatement(WORLD_SEL_COMMANDS)))
91  {
92  do
93  {
94  Field* fields = result->Fetch();
95  std::string_view const name = fields[0].GetStringView();
96  std::string_view const help = fields[2].GetStringView();
97  uint32 const secLevel = fields[1].GetUInt8();
98 
99  ChatCommandNode* cmd = nullptr;
100  ChatSubCommandMap* map = &COMMAND_MAP;
101 
102  for (std::string_view key : Acore::Tokenize(name, COMMAND_DELIMITER, false))
103  {
104  auto it = map->find(key);
105  if (it != map->end())
106  {
107  cmd = &it->second;
108  map = &cmd->_subCommands;
109  }
110  else
111  {
112  LOG_ERROR("sql.sql", "Table `command` contains data for non-existant command '" STRING_VIEW_FMT "'. Skipped.", STRING_VIEW_FMT_ARG(name));
113  cmd = nullptr;
114  break;
115  }
116  }
117 
118  if (!cmd)
119  continue;
120 
121  if (cmd->_invoker && (cmd->_permission.RequiredLevel != secLevel))
122  {
123  LOG_WARN("sql.sql", "Table `command` has permission %u for '" STRING_VIEW_FMT "' which does not match the core (%u). Overriding.",
124  secLevel, STRING_VIEW_FMT_ARG(name), cmd->_permission.RequiredLevel);
125 
126  cmd->_permission.RequiredLevel = secLevel;
127  }
128 
129  if (std::holds_alternative<std::string>(cmd->_help))
130  LOG_ERROR("sql.sql", "Table `command` contains duplicate data for command '" STRING_VIEW_FMT "'. Skipped.", STRING_VIEW_FMT_ARG(name));
131 
132  if (std::holds_alternative<std::monostate>(cmd->_help))
133  cmd->_help.emplace<std::string>(help);
134  else
135  LOG_ERROR("sql.sql", "Table `command` contains legacy help text for command '" STRING_VIEW_FMT "', which uses `trinity_string`. Skipped.", STRING_VIEW_FMT_ARG(name));
136  } while (result->NextRow());
137  }
138 
139  for (auto& [name, cmd] : COMMAND_MAP)
140  cmd.ResolveNames(std::string(name));
141 }

278 /*static*/ bool Acore::Impl::ChatCommands::ChatCommandNode::TryExecuteCommand(ChatHandler& handler, std::string_view cmdStr)
279 {
280  ChatCommandNode const* cmd = nullptr;
281  ChatSubCommandMap const* map = &GetTopLevelMap();
282 
283  while (!cmdStr.empty() && (cmdStr.front() == COMMAND_DELIMITER))
284  cmdStr.remove_prefix(1);
285 
286  while (!cmdStr.empty() && (cmdStr.back() == COMMAND_DELIMITER))
287  cmdStr.remove_suffix(1);
288 
289  std::string_view oldTail = cmdStr;
290  while (!oldTail.empty())
291  {
292  /* oldTail = token DELIMITER newTail */
293  auto [token, newTail] = tokenize(oldTail);
294  ASSERT(!token.empty());
295 
296  FilteredCommandListIterator it1(*map, handler, token);
297  if (!it1)
298  break; /* no matching subcommands found */
299 
300  if (!StringEqualI(it1->first, token))
301  { /* ok, so it1 points at a partially matching subcommand - let's see if there are others */
302  auto it2 = it1;
303  ++it2;
304 
305  if (it2)
306  { /* there are multiple matching subcommands - print possibilities and return */
307  if (cmd)
308  handler.PSendSysMessage(LANG_SUBCMD_AMBIGUOUS, STRING_VIEW_FMT_ARG(cmd->_name), COMMAND_DELIMITER, STRING_VIEW_FMT_ARG(token));
309  else
310  handler.PSendSysMessage(LANG_CMD_AMBIGUOUS, STRING_VIEW_FMT_ARG(token));
311 
312  handler.PSendSysMessage(it1->second.HasVisibleSubCommands(handler) ? LANG_SUBCMDS_LIST_ENTRY_ELLIPSIS : LANG_SUBCMDS_LIST_ENTRY, STRING_VIEW_FMT_ARG(it1->first));
313  do
314  {
315  handler.PSendSysMessage(it2->second.HasVisibleSubCommands(handler) ? LANG_SUBCMDS_LIST_ENTRY_ELLIPSIS : LANG_SUBCMDS_LIST_ENTRY, STRING_VIEW_FMT_ARG(it2->first));
316  } while (++it2);
317 
318  return true;
319  }
320  }
321 
322  /* now we matched exactly one subcommand, and it1 points to it; go down the rabbit hole */
323  cmd = &it1->second;
324  map = &cmd->_subCommands;
325 
326  oldTail = newTail;
327  }
328 
329  if (cmd)
330  { /* if we matched a command at some point, invoke it */
331  handler.SetSentErrorMessage(false);
332  if (cmd->IsInvokerVisible(handler) && cmd->_invoker(&handler, oldTail))
333  { /* invocation succeeded, log this */
334  if (!handler.IsConsole())
335  LogCommandUsage(*handler.GetSession(), cmdStr);
336  }
337  else if (!handler.HasSentErrorMessage())
338  { /* invocation failed, we should show usage */
339 #ifdef ELUNA
340  if (!sEluna->OnCommand(handler.IsConsole() ? nullptr : handler.GetSession()->GetPlayer(), std::string(cmdStr).c_str()))
341  {
342  return true;
343  }
344 #endif
345 
346  cmd->SendCommandHelp(handler);
347  handler.SetSentErrorMessage(true);
348  }
349  return true;
350  }
351 
352 #ifdef ELUNA
353  if (!sEluna->OnCommand(handler.IsConsole() ? nullptr : handler.GetSession()->GetPlayer(), std::string(cmdStr).c_str()))
354  {
355  return true;
356  }
357 #endif
358 
359  return false;
360 }
