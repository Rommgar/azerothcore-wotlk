171  events.ScheduleEvent(EVENT_POISON_NOVA, 15000);

173  case EVENT_POWERFULL_BITE:
174  me->CastSpell(me->GetVictim(), SPELL_POWERFULL_BITE, false);
175  events.ScheduleEvent(EVENT_POWERFULL_BITE, 10000);
176  break;
177  case EVENT_VENOM_BOLT:
178  me->CastSpell(me->GetVictim(), SPELL_VENOM_BOLT, false);
179  events.ScheduleEvent(EVENT_VENOM_BOLT, 10000);
180  break;
181  case EVENT_SUMMON1:
182  for (uint8 i = MAX_CONSTRICTOR; i < MAX_SUMMONS; ++i)
183  me->SummonCreature(NPC_SLADRAN_VIPER, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20 * IN_MILLISECONDS);
184  events.ScheduleEvent(EVENT_SUMMON1, 8000);

188  me->SummonCreature(NPC_SLADRAN_CONSTRICTORS, SpawnLoc[i], TEMPSUMMON_CORPSE_TIMED_DESPAWN, 20 * IN_MILLISECONDS);
189  events.ScheduleEvent(EVENT_SUMMON2, urand(3000, 5000));
190  break;
191  }
192 
193  DoMeleeAttackIfReady();
194  }
