82 my_bool my_thread_basic_global_init(void)
83 {
84  int pth_ret;
85 
86  if (my_thread_basic_global_init_done)
87  return 0;
88  my_thread_basic_global_init_done= 1;
89 
90 #ifdef PTHREAD_ADAPTIVE_MUTEX_INITIALIZER_NP
91  /*
92  Set mutex type to "fast" a.k.a "adaptive"
93 
94  In this case the thread may steal the mutex from some other thread
95  that is waiting for the same mutex. This will save us some
96  context switches but may cause a thread to 'starve forever' while
97  waiting for the mutex (not likely if the code within the mutex is
98  short).
99  */
100  pthread_mutexattr_init(&my_fast_mutexattr);
101  pthread_mutexattr_settype(&my_fast_mutexattr,
102  PTHREAD_MUTEX_ADAPTIVE_NP);
103 #endif
104 
105 #ifdef PTHREAD_ERRORCHECK_MUTEX_INITIALIZER_NP
106  /*
107  Set mutex type to "errorcheck"
108  */
109  pthread_mutexattr_init(&my_errorcheck_mutexattr);
110  pthread_mutexattr_settype(&my_errorcheck_mutexattr,
111  PTHREAD_MUTEX_ERRORCHECK);
112 #endif
113 
114  mysql_mutex_init(key_THR_LOCK_malloc, &THR_LOCK_malloc, MY_MUTEX_INIT_FAST);
115  mysql_mutex_init(key_THR_LOCK_open, &THR_LOCK_open, MY_MUTEX_INIT_FAST);
116  mysql_mutex_init(key_THR_LOCK_charset, &THR_LOCK_charset, MY_MUTEX_INIT_FAST);
117  mysql_mutex_init(key_THR_LOCK_threads, &THR_LOCK_threads, MY_MUTEX_INIT_FAST);
118 
119  if ((pth_ret= pthread_key_create(&THR_KEY_mysys, NULL)) != 0)
120  {
121  fprintf(stderr, "Can't initialize threads: error %d\n", pth_ret);
122  return 1;
123  }
124 
125  if (my_thread_init())
126  return 1;
127 
128  return 0;
129 }

182 my_bool my_thread_global_init(void)
183 {
184  if (my_thread_basic_global_init())
185  return 1;
186 
187  thd_lib_detected= get_thread_lib();
188 
189 #ifdef TARGET_OS_LINUX
190  /*
191  BUG#24507: Race conditions inside current NPTL pthread_exit()
192  implementation.
193 
194  To avoid a possible segmentation fault during concurrent
195  executions of pthread_exit(), a dummy thread is spawned which
196  initializes internal variables of pthread lib. See bug description
197  for a full explanation.
198 
199  TODO: Remove this code when fixed versions of glibc6 are in common
200  use.
201  */
202  if (thd_lib_detected == THD_LIB_NPTL)
203  {
204  pthread_t dummy_thread;
205  pthread_attr_t dummy_thread_attr;
206 
207  pthread_attr_init(&dummy_thread_attr);
208  pthread_attr_setdetachstate(&dummy_thread_attr, PTHREAD_CREATE_JOINABLE);
209 
210  if (pthread_create(&dummy_thread,&dummy_thread_attr,
211  nptl_pthread_exit_hack_handler, NULL) == 0)
212  (void)pthread_join(dummy_thread, NULL);
213  }
214 #endif /* TARGET_OS_LINUX */

217  mysql_mutex_init(key_THR_LOCK_isam, &THR_LOCK_isam, MY_MUTEX_INIT_SLOW);

246 void my_thread_global_end(void)
247 {
248  struct timespec abstime;
249  my_bool all_threads_killed= 1;
250 
251  set_timespec(abstime, my_thread_end_wait_time);
252  mysql_mutex_lock(&THR_LOCK_threads);
253  while (THR_thread_count > 0)
254  {
255  int error= mysql_cond_timedwait(&THR_COND_threads, &THR_LOCK_threads,
256  &abstime);
257  if (error == ETIMEDOUT || error == ETIME)
258  {
259 #ifdef HAVE_PTHREAD_KILL
260  /*
261  We shouldn't give an error here, because if we don't have
262  pthread_kill(), programs like mysqld can't ensure that all threads
263  are killed when we enter here.
264  */
265  if (THR_thread_count)
266  fprintf(stderr,
267  "Error in my_thread_global_end(): %d threads didn't exit\n",
268  THR_thread_count);
269 #endif
270  all_threads_killed= 0;
271  break;
272  }
273  }
274  mysql_mutex_unlock(&THR_LOCK_threads);
275 
276  pthread_key_delete(THR_KEY_mysys);
277 #ifdef PTHREAD_ADAPTIVE_MUTEX_INITIALIZER_NP
278  pthread_mutexattr_destroy(&my_fast_mutexattr);
279 #endif
280 #ifdef PTHREAD_ERRORCHECK_MUTEX_INITIALIZER_NP
281  pthread_mutexattr_destroy(&my_errorcheck_mutexattr);
282 #endif
283  mysql_mutex_destroy(&THR_LOCK_malloc);
284  mysql_mutex_destroy(&THR_LOCK_open);
285  mysql_mutex_destroy(&THR_LOCK_lock);
286  mysql_mutex_destroy(&THR_LOCK_isam);
287  mysql_mutex_destroy(&THR_LOCK_myisam);
288  mysql_mutex_destroy(&THR_LOCK_myisam_mmap);
289  mysql_mutex_destroy(&THR_LOCK_heap);
290  mysql_mutex_destroy(&THR_LOCK_net);
291  mysql_mutex_destroy(&THR_LOCK_time);
292  mysql_mutex_destroy(&THR_LOCK_charset);
293  if (all_threads_killed)
294  {
295  mysql_mutex_destroy(&THR_LOCK_threads);
296  mysql_cond_destroy(&THR_COND_threads);
297  }
298 #if !defined(HAVE_LOCALTIME_R) || !defined(HAVE_GMTIME_R)
299  mysql_mutex_destroy(&LOCK_localtime_r);
300 #endif
301 #ifndef HAVE_GETHOSTBYNAME_R
302  mysql_mutex_destroy(&LOCK_gethostbyname_r);
303 #endif
304 
305  my_thread_basic_global_init_done= 0;
306 }

330 my_bool my_thread_init(void)
331 {
332  struct st_my_thread_var *tmp;
333  my_bool error=0;
334 
335 #ifdef EXTRA_DEBUG_THREADS
336  fprintf(stderr,"my_thread_init(): thread_id: 0x%lx\n",
337  (ulong) pthread_self());
338 #endif
339 
340  if (my_pthread_getspecific(struct st_my_thread_var *,THR_KEY_mysys))
341  {
342 #ifdef EXTRA_DEBUG_THREADS
343  fprintf(stderr,"my_thread_init() called more than once in thread 0x%lx\n",
344  (long) pthread_self());
345 #endif
346  goto end;
347  }
348 
349 #ifdef _MSC_VER
350  install_sigabrt_handler();
351 #endif
352 
353  if (!(tmp= (struct st_my_thread_var *) calloc(1, sizeof(*tmp))))
354  {
355  error= 1;
356  goto end;
357  }
358  pthread_setspecific(THR_KEY_mysys,tmp);
359  tmp->pthread_self= pthread_self();
360  mysql_mutex_init(key_my_thread_var_mutex, &tmp->mutex, MY_MUTEX_INIT_FAST);
361  mysql_cond_init(key_my_thread_var_suspend, &tmp->suspend, NULL);
362 
363  tmp->stack_ends_here= (char*)&tmp +
364  STACK_DIRECTION * (long)my_thread_stack_size;
365 
366  mysql_mutex_lock(&THR_LOCK_threads);
367  tmp->id= ++thread_id;
368  ++THR_thread_count;
369  mysql_mutex_unlock(&THR_LOCK_threads);
370  tmp->init= 1;
371 #ifndef DBUG_OFF
372  /* Generate unique name for thread */
373  (void) my_thread_name();
374 #endif
375 
376 end:
377  return error;
378 }

381 /*
382  Deallocate memory used by the thread for book-keeping
383 
384  SYNOPSIS
385  my_thread_end()
386 
387  NOTE
388  This may be called multiple times for a thread.
389  This happens for example when one calls 'mysql_server_init()'
390  mysql_server_end() and then ends with a mysql_end().
391 */
392 
393 void my_thread_end(void)
394 {
395  struct st_my_thread_var *tmp;
396  tmp= my_pthread_getspecific(struct st_my_thread_var*,THR_KEY_mysys);
397 
398 #ifdef EXTRA_DEBUG_THREADS
399  fprintf(stderr,"my_thread_end(): tmp: 0x%lx pthread_self: 0x%lx thread_id: %ld\n",
400  (long) tmp, (long) pthread_self(), tmp ? (long) tmp->id : 0L);
401 #endif
402 
403 #ifdef HAVE_PSI_INTERFACE
404  /*
405  Remove the instrumentation for this thread.
406  This must be done before trashing st_my_thread_var,
407  because the LF_HASH depends on it.
408  */
409  if (PSI_server)
410  PSI_server->delete_current_thread();
411 #endif
412 
413  if (tmp && tmp->init)
414  {
415 #if !defined(DBUG_OFF)
416  /* tmp->dbug is allocated inside DBUG library */
417  if (tmp->dbug)
418  {
419  DBUG_POP();
420  free(tmp->dbug);
421  tmp->dbug=0;
422  }
423 #endif
424 #if !defined(__bsdi__) && !defined(__OpenBSD__)
425  /* bsdi and openbsd 3.5 dumps core here */
426  mysql_cond_destroy(&tmp->suspend);
427 #endif
428  mysql_mutex_destroy(&tmp->mutex);
429  free(tmp);
430 
431  /*
432  Decrement counter for number of running threads. We are using this
433  in my_thread_global_end() to wait until all threads have called
434  my_thread_end and thus freed all memory they have allocated in
435  my_thread_init() and DBUG_xxxx
436  */
437  mysql_mutex_lock(&THR_LOCK_threads);
438  DBUG_ASSERT(THR_thread_count != 0);
439  if (--THR_thread_count == 0)
440  mysql_cond_signal(&THR_COND_threads);
441  mysql_mutex_unlock(&THR_LOCK_threads);
442  }
443  pthread_setspecific(THR_KEY_mysys,0);
444 }
