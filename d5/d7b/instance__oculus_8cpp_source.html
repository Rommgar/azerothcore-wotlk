158  eregos->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
159  break;
160  case DATA_EREGOS:
161  m_auiEncounter[DATA_EREGOS] = data;
162  if (data == DONE)
163  DoRespawnGameObject(EregosCacheGUID, 7 * DAY);
164  break;
165  case DATA_CC_COUNT:
166  if( CentrifugeCount < 10 )
167  {
168  ++CentrifugeCount;
169  DoUpdateWorldState(WORLD_STATE_CENTRIFUGE_CONSTRUCT_AMOUNT, 10 - CentrifugeCount);
170  }
171  if( CentrifugeCount >= 10 )
172  if( Creature* varos = instance->GetCreature(uiVarosGUID) )
173  {
174  varos->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
175  varos->InterruptNonMeleeSpells(false);
176  varos->RemoveAura(50053);
177  }
178  break;
179  case DATA_AMBER_VOID:
180  bAmberVoid = !!data;
181  break;
182  case DATA_EMERALD_VOID:
183  bEmeraldVoid = !!data;
184  break;
185  case DATA_RUBY_VOID:
186  bRubyVoid = !!data;
187  break;
188  }
189 
190  if( data == DONE )
191  SaveToDB();
192  }
