112  static bool GetDeletedCharacterInfoList(DeletedInfoList& foundList, std::string searchString)
113  {
114  PreparedQueryResult result;
115  CharacterDatabasePreparedStatement* stmt = nullptr;
116  if (!searchString.empty())
117  {
118  // search by GUID
119  if (isNumeric(searchString.c_str()))
120  {
121  stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHAR_DEL_INFO_BY_GUID);
122  stmt->setUInt32(0, uint32(atoi(searchString.c_str())));
123  result = CharacterDatabase.Query(stmt);
124  }
125  // search by name
126  else
127  {
128  if (!normalizePlayerName(searchString))
129  return false;
130 
131  stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHAR_DEL_INFO_BY_NAME);
132  stmt->setString(0, searchString);
133  result = CharacterDatabase.Query(stmt);
134  }
135  }
136  else
137  {
138  stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHAR_DEL_INFO);
139  result = CharacterDatabase.Query(stmt);
140  }
141 
142  if (result)
143  {
144  do
145  {
146  Field* fields = result->Fetch();
147 
148  DeletedInfo info;
149 
150  info.lowGuid = fields[0].GetUInt32();
151  info.name = fields[1].GetString();
152  info.accountId = fields[2].GetUInt32();
153 
154  // account name will be empty for nonexisting account
155  AccountMgr::GetName(info.accountId, info.accountName);
156  info.deleteDate = time_t(fields[3].GetUInt32());
157  foundList.push_back(info);
158  } while (result->NextRow());
159  }
160 
161  return true;
162  }

269  CharacterDatabase.Execute(stmt);
270 
271  // xinef: update global storage

296  CharTitlesEntry const* titleInfo = sCharTitlesStore.LookupEntry(id);
297 
298  if (titleInfo && target->HasTitle(titleInfo))

302  name = (target->getGender() == GENDER_MALE ? titleInfo->nameMale[sWorld->GetDefaultDbcLocale()] : titleInfo->nameFemale[sWorld->GetDefaultDbcLocale()]);
303 
304  if (!*name)
305  continue;
306 
307  char const* activeStr = "";
308  if (target->GetUInt32Value(PLAYER_CHOSEN_TITLE) == titleInfo->bit_index)
309  activeStr = handler->GetAcoreString(LANG_ACTIVE);
310 
311  std::string titleName = Acore::StringFormat(name, player->GetName().c_str());
312 
313  // send title in "id (idx:idx) - [namedlink locale]" format

325  static bool HandleCharacterRenameCommand(ChatHandler* handler, Optional<PlayerIdentifier> player, Optional<bool> reserveName, Optional<std::string_view> newNameV)
326  {
327  if (!player && newNameV)
328  return false;
329 
330  if (!player)
331  player = PlayerIdentifier::FromTarget(handler);
332 
333  if (!player)
334  return false;
335 
336  if (handler->HasLowerSecurity(nullptr, player->GetGUID()))
337  return false;
338 
339  if (newNameV)
340  {
341  std::string newName{ *newNameV };
342  if (!normalizePlayerName(newName))
343  {
344  handler->SendSysMessage(LANG_BAD_VALUE);
345  handler->SetSentErrorMessage(true);
346  return false;
347  }
348 
349  if (ObjectMgr::CheckPlayerName(newName, true) != CHAR_NAME_SUCCESS)
350  {
351  handler->SendSysMessage(LANG_BAD_VALUE);
352  handler->SetSentErrorMessage(true);
353  return false;
354  }
355 
356  if (sObjectMgr->IsReservedName(newName))
357  {
358  handler->SendSysMessage(LANG_RESERVED_NAME);
359  handler->SetSentErrorMessage(true);
360  return false;
361  }
362 
363  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_CHECK_NAME);
364  stmt->setString(0, newName);
365  PreparedQueryResult result = CharacterDatabase.Query(stmt);
366  if (result)
367  {
368  handler->PSendSysMessage(LANG_RENAME_PLAYER_ALREADY_EXISTS, newName.c_str());
369  handler->SetSentErrorMessage(true);
370  return false;
371  }
372 
373  // Remove declined name from db
374  stmt = CharacterDatabase.GetPreparedStatement(CHAR_DEL_DECLINED_NAME);
375  stmt->setUInt32(0, player->GetGUID().GetCounter());
376  CharacterDatabase.Execute(stmt);
377 
378  if (Player* target = player->GetConnectedPlayer())
379  {
380  target->SetName(newName);
381 
382  if (WorldSession* session = target->GetSession())
383  session->KickPlayer("HandleCharacterRenameCommand GM Command renaming character");
384  }
385  else
386  {
387  stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_NAME_BY_GUID);
388  stmt->setString(0, newName);
389  stmt->setUInt32(1, player->GetGUID().GetCounter());
390  CharacterDatabase.Execute(stmt);
391  }
392 
393  sWorld->UpdateGlobalNameData(player->GetGUID().GetCounter(), player->GetName().c_str(), newName);
394  sWorld->UpdateGlobalPlayerData(player->GetGUID().GetCounter(), PLAYER_UPDATE_DATA_NAME, newName);
395 
396  handler->PSendSysMessage(LANG_RENAME_PLAYER_WITH_NEW_NAME, player->GetName().c_str(), newName.c_str());
397  }
398  else
399  {
400  if (Player* target = player->GetConnectedPlayer())
401  {
402  handler->PSendSysMessage(LANG_RENAME_PLAYER, handler->GetNameLink(target).c_str());
403  target->SetAtLoginFlag(AT_LOGIN_RENAME);
404  }
405  else
406  {
407  // check offline security
408  if (handler->HasLowerSecurity(nullptr, player->GetGUID()))
409  return false;
410 
411  handler->PSendSysMessage(LANG_RENAME_PLAYER_GUID, handler->playerLink(*player).c_str(), player->GetGUID().GetCounter());
412 
413  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_ADD_AT_LOGIN_FLAG);
414  stmt->setUInt16(0, uint16(AT_LOGIN_RENAME));
415  stmt->setUInt32(1, player->GetGUID().GetCounter());
416  CharacterDatabase.Execute(stmt);
417  }
418  }
419 
420  if (reserveName)
421  {
422  sObjectMgr->AddReservedPlayerName(player->GetName());
423  }
424 
425  return true;
426  }
