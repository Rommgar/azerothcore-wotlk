108  void SetData(uint32 type, uint32 data) override
109  {
110  switch (type)
111  {
112  case DATA_SHOW_CRATES:
113  DoUpdateWorldState(WORLDSTATE_SHOW_CRATES, data);
114  return;
115  case DATA_SHOW_INFINITE_TIMER:
116  if (!instance->IsHeroic() || !_guardianTimer)
117  return;
118  DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, data);
119  DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN, uint32(_guardianTimer / (MINUTE * IN_MILLISECONDS)));
120  if (data == 0)
121  {
122  _guardianTimer = 0;
123  SaveToDB();
124  }
125  else if (!_infiniteGUID)
126  instance->SummonCreature(NPC_INFINITE, EventPos[EVENT_SRC_CORRUPTOR]);
127  return;
128  case DATA_START_WAVES:
129  DoUpdateWorldState(WORLDSTATE_WAVE_COUNT, 1);
130  if (instance->IsHeroic())
131  {
132  DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, true);
133  _guardianTimer = 26 * MINUTE * IN_MILLISECONDS;
134  if (!_infiniteGUID)
135  instance->SummonCreature(NPC_INFINITE, EventPos[EVENT_SRC_CORRUPTOR]);
136  }
137  return;
138  case DATA_CRATE_COUNT:
139  _crateCount++;
140  if (_crateCount == 5)
141  {
142  Map::PlayerList const& PlayerList = instance->GetPlayers();
143  if (!PlayerList.isEmpty())
144  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
145  i->GetSource()->KilledMonsterCredit(NPC_GRAIN_CREATE_TRIGGER);
146 
147  _showCrateTimer++;
148  if (GetData(DATA_ARTHAS_EVENT) < COS_PROGRESS_CRATES_FOUND)
149  SetData(DATA_ARTHAS_EVENT, COS_PROGRESS_CRATES_FOUND);
150  }
151 
152  DoUpdateWorldState(WORLDSTATE_CRATES_REVEALED, _crateCount);
153  return;
154  case DATA_ARTHAS_EVENT:
155  // Start Event
156  _encounterState = data;
157  if (data == COS_PROGRESS_START_INTRO)
158  {
159  if (Creature* arthas = instance->GetCreature(_arthasGUID))
160  arthas->AI()->DoAction(ACTION_START_EVENT);
161  }
162  else if (data == COS_PROGRESS_KILLED_SALRAMM)
163  {
164  if (Creature* arthas = instance->GetCreature(_arthasGUID))
165  arthas->AI()->DoAction(ACTION_KILLED_SALRAMM);
166  }
167  break;
168  case DATA_ARTHAS_REPOSITION:
169  if (data == 2)
170  _respawnAndReposition = true;
171  else if (Creature* arthas = instance->GetCreature(_arthasGUID))
172  Reposition(arthas);
173  return;
174  }
175 
176  if (type == DATA_ARTHAS_EVENT)
177  SaveToDB();
178  }

207  void Update(uint32 diff) override
208  {
209  if (_loadTimer)
210  {
211  _loadTimer += diff;
212  if (_loadTimer > 3000)
213  {
214  UpdateEventState();
215  _loadTimer = 0;
216  }
217  }
218  // Used when arthas dies
219  if (_respawnAndReposition)
220  {
221  if (Creature* arthas = instance->GetCreature(_arthasGUID))
222  {
223  if (!arthas->IsAlive())
224  {
225  EnsureGridLoaded();
226  arthas->setDeathState(DEAD);
227  arthas->Respawn();
228  }
229  else
230  {
231  arthas->AI()->Reset();
232  _respawnAndReposition = false;
233  }
234  }
235  }
236 
237  // Used after 5-th crates is revealed
238  if (_showCrateTimer)
239  {
240  _showCrateTimer += diff;
241  if (_showCrateTimer >= 20000)
242  {
243  UpdateEventState();
244  _showCrateTimer = 0; // just to be sure
245  }
246  }
247 
248  // Used to display how much time players have
249  if (_guardianTimer)
250  {
251  uint32 div = uint32(_guardianTimer / (MINUTE * IN_MILLISECONDS));
252  _guardianTimer -= diff;
253  uint32 divAfter = uint32(_guardianTimer / (MINUTE * IN_MILLISECONDS));
254 
255  if (divAfter == 0)
256  {
257  _guardianTimer = 0;
258  DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN_SHOW, 0);
259 
260  // Inform infinite we run out of time
261  if (instance->IsHeroic() && _infiniteGUID)
262  if (Creature* cr = instance->GetCreature(_infiniteGUID))
263  cr->AI()->DoAction(ACTION_RUN_OUT_OF_TIME);
264  }
265  else if (div > divAfter)
266  {
267  if (divAfter == 5)
268  ChromieWhisper(1);
269  else if (divAfter == 1)
270  ChromieWhisper(2);
271 
272  DoUpdateWorldState(WORLDSTATE_TIME_GUARDIAN, divAfter);
273  SaveToDB();
274  }
275  }
276  }

285 
286  if (_encounterState == COS_PROGRESS_CRATES_FOUND ||

289  ChromieWhisper(0);
290 
291  // hide crates count
292  DoUpdateWorldState(WORLDSTATE_SHOW_CRATES, 0);
