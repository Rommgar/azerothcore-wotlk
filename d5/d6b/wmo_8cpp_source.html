269 int WMOGroup::ConvertToVMAPGroupWmo(FILE* output, WMORoot* rootWMO, bool preciseVectorData)
270 {
271  fwrite(&mogpFlags, sizeof(uint32), 1, output);
272  fwrite(&groupWMOID, sizeof(uint32), 1, output);
273  // group bound
274  fwrite(bbcorn1, sizeof(float), 3, output);
275  fwrite(bbcorn2, sizeof(float), 3, output);
276  fwrite(&liquflags, sizeof(uint32), 1, output);
277  int nColTriangles = 0;
278  if (preciseVectorData)
279  {
280  char GRP[] = "GRP ";
281  fwrite(GRP, 1, 4, output);
282 
283  int k = 0;
284  int moba_batch = moba_size / 12;
285  MobaEx = new int[moba_batch * 4];
286  for (int i = 8; i < moba_size; i += 12)
287  {
288  MobaEx[k++] = MOBA[i];
289  }
290  int moba_size_grp = moba_batch * 4 + 4;
291  fwrite(&moba_size_grp, 4, 1, output);
292  fwrite(&moba_batch, 4, 1, output);
293  fwrite(MobaEx, 4, k, output);
294  delete [] MobaEx;
295 
296  uint32 nIdexes = nTriangles * 3;
297 
298  if (fwrite("INDX", 4, 1, output) != 1)
299  {
300  printf("Error while writing file nbraches ID");
301  exit(0);
302  }
303  int wsize = sizeof(uint32) + sizeof(unsigned short) * nIdexes;
304  if (fwrite(&wsize, sizeof(int), 1, output) != 1)
305  {
306  printf("Error while writing file wsize");
307  // no need to exit?
308  }
309  if (fwrite(&nIdexes, sizeof(uint32), 1, output) != 1)
310  {
311  printf("Error while writing file nIndexes");
312  exit(0);
313  }
314  if (nIdexes > 0)
315  {
316  if (fwrite(MOVI, sizeof(unsigned short), nIdexes, output) != nIdexes)
317  {
318  printf("Error while writing file indexarray");
319  exit(0);
320  }
321  }
322 
323  if (fwrite("VERT", 4, 1, output) != 1)
324  {
325  printf("Error while writing file nbraches ID");
326  exit(0);
327  }
328  wsize = sizeof(int) + sizeof(float) * 3 * nVertices;
329  if (fwrite(&wsize, sizeof(int), 1, output) != 1)
330  {
331  printf("Error while writing file wsize");
332  // no need to exit?
333  }
334  if (fwrite(&nVertices, sizeof(int), 1, output) != 1)
335  {
336  printf("Error while writing file nVertices");
337  exit(0);
338  }
339  if (nVertices > 0)
340  {
341  if (fwrite(MOVT, sizeof(float) * 3, nVertices, output) != nVertices)
342  {
343  printf("Error while writing file vectors");
344  exit(0);
345  }
346  }
347 
348  nColTriangles = nTriangles;
349  }
350  else
351  {
352  char GRP[] = "GRP ";
353  fwrite(GRP, 1, 4, output);
354  int k = 0;
355  int moba_batch = moba_size / 12;
356  MobaEx = new int[moba_batch * 4];
357  for (int i = 8; i < moba_size; i += 12)
358  {
359  MobaEx[k++] = MOBA[i];
360  }
361 
362  int moba_size_grp = moba_batch * 4 + 4;
363  fwrite(&moba_size_grp, 4, 1, output);
364  fwrite(&moba_batch, 4, 1, output);
365  fwrite(MobaEx, 4, k, output);
366  delete [] MobaEx;
367 
368  //-------INDX------------------------------------
369  //-------MOPY--------
370  MoviEx = new uint16[nTriangles * 3]; // "worst case" size...
371  int* IndexRenum = new int[nVertices];
372  memset(IndexRenum, 0xFF, nVertices * sizeof(int));
373  for (int i = 0; i < nTriangles; ++i)
374  {
375  // Skip no collision triangles
376  bool isRenderFace = (MOPY[2 * i] & WMO_MATERIAL_RENDER) && !(MOPY[2 * i] & WMO_MATERIAL_DETAIL);
377  bool isCollision = MOPY[2 * i] & WMO_MATERIAL_COLLISION || isRenderFace;
378  if (!isCollision)
379  continue;
380  // Use this triangle
381  for (int j = 0; j < 3; ++j)
382  {
383  IndexRenum[MOVI[3 * i + j]] = 1;
384  MoviEx[3 * nColTriangles + j] = MOVI[3 * i + j];
385  }
386  ++nColTriangles;
387  }
388 
389  // assign new vertex index numbers
390  int nColVertices = 0;
391  for (uint32 i = 0; i < nVertices; ++i)
392  {
393  if (IndexRenum[i] == 1)
394  {
395  IndexRenum[i] = nColVertices;
396  ++nColVertices;
397  }
398  }
399 
400  // translate triangle indices to new numbers
401  for (int i = 0; i < 3 * nColTriangles; ++i)
402  {
403  assert(MoviEx[i] < nVertices);
404  MoviEx[i] = IndexRenum[MoviEx[i]];
405  }
406 
407  // write triangle indices
408  int INDX[] = {0x58444E49, nColTriangles * 6 + 4, nColTriangles * 3};
409  fwrite(INDX, 4, 3, output);
410  fwrite(MoviEx, 2, nColTriangles * 3, output);
411 
412  // write vertices
413  int VERT[] = {0x54524556, nColVertices * 3 * static_cast<int>(sizeof(float)) + 4, nColVertices}; // "VERT"
414  int check = 3 * nColVertices;
415  fwrite(VERT, 4, 3, output);
416  for (uint32 i = 0; i < nVertices; ++i)
417  if (IndexRenum[i] >= 0)
418  check -= fwrite(MOVT + 3 * i, sizeof(float), 3, output);
419 
420  assert(check == 0);
421 
422  delete [] MoviEx;
423  delete [] IndexRenum;
424  }
425 
426  //------LIQU------------------------
427  if (LiquEx_size != 0)
428  {
429  int LIQU_h[] = { 0x5551494C, static_cast<int>(sizeof(WMOLiquidHeader) + LiquEx_size) + hlq->xtiles * hlq->ytiles }; // "LIQU"
430  fwrite(LIQU_h, 4, 2, output);
431 
432  // according to WoW.Dev Wiki:
433  uint32 liquidEntry;
434  if (rootWMO->liquidType & 4)
435  liquidEntry = liquidType;
436  else if (liquidType == 15)
437  liquidEntry = 0;
438  else
439  liquidEntry = liquidType + 1;
440 
441  if (!liquidEntry)
442  {
443  int v1; // edx@1
444  int v2; // eax@1
445 
446  v1 = hlq->xtiles * hlq->ytiles;
447  v2 = 0;
448  if (v1 > 0)
449  {
450  while ((LiquBytes[v2] & 0xF) == 15)
451  {
452  ++v2;
453  if (v2 >= v1)
454  break;
455  }
456 
457  if (v2 < v1 && (LiquBytes[v2] & 0xF) != 15)
458  liquidEntry = (LiquBytes[v2] & 0xF) + 1;
459  }
460  }
461 
462  if (liquidEntry && liquidEntry < 21)
463  {
464  switch ((liquidEntry - 1) & 3)
465  {
466  case 0:
467  liquidEntry = ((mogpFlags & 0x80000) != 0) + 13;
468  break;
469  case 1:
470  liquidEntry = 14;
471  break;
472  case 2:
473  liquidEntry = 19;
474  break;
475  case 3:
476  liquidEntry = 20;
477  break;
478  }
479  }
480 
481  hlq->type = liquidEntry;
482 
483  /* std::ofstream llog("Buildings/liquid.log", ios_base::out | ios_base::app);
484  llog << filename;
485  llog << ":\nliquidEntry: " << liquidEntry << " type: " << hlq->type << " (root:" << rootWMO->liquidType << " group:" << liquidType << ")\n";
486  llog.close(); */
487 
488  fwrite(hlq, sizeof(WMOLiquidHeader), 1, output);
489  // only need height values, the other values are unknown anyway
490  for (uint32 i = 0; i < LiquEx_size / sizeof(WMOLiquidVert); ++i)
491  fwrite(&LiquEx[i].height, sizeof(float), 1, output);
492  // todo: compress to bit field
493  fwrite(LiquBytes, 1, hlq->xtiles * hlq->ytiles, output);
494  }
495 
496  return nColTriangles;
497 }
