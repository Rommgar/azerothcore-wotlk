169 void InstanceScript::UpdateMinionState(Creature* minion, EncounterState state)
170 {
171  switch (state)
172  {
173  case NOT_STARTED:
174  if (!minion->IsAlive())
175  minion->Respawn();
176  else if (minion->IsInCombat())
177  minion->AI()->EnterEvadeMode();
178  break;
179  case IN_PROGRESS:
180  if (!minion->IsAlive())
181  minion->Respawn();
182  else if (!minion->GetVictim())
183  minion->AI()->DoZoneInCombat(nullptr, 100.0f);
184  break;
185  default:
186  break;
187  }

190 void InstanceScript::UpdateDoorState(GameObject* door)
191 {
192  DoorInfoMapBounds range = doors.equal_range(door->GetEntry());
193  if (range.first == range.second)
194  return;
195 
196  // xinef: doors can be assigned to few bosses, if any of them demands doors closed - they should be closed (added & operator for assigment)
197  bool open = true;
198  for (; range.first != range.second && open; ++range.first)
199  {
200  DoorInfo const& info = range.first->second;
201  switch (info.type)
202  {
203  case DOOR_TYPE_ROOM:
204  open &= (info.bossInfo->state != IN_PROGRESS);
205  break;
206  case DOOR_TYPE_PASSAGE:
207  open &= (info.bossInfo->state == DONE);
208  break;
209  case DOOR_TYPE_SPAWN_HOLE:
210  open &= (info.bossInfo->state == IN_PROGRESS);
211  break;
212  default:
213  break;
214  }
215  }
216 
217  door->SetGoState(open ? GO_STATE_ACTIVE : GO_STATE_READY);
218 }

254 void InstanceScript::AddDoor(GameObject* door, bool add)
255 {
256  DoorInfoMapBounds range = doors.equal_range(door->GetEntry());
257  if (range.first == range.second)
258  return;
259 
260  for (; range.first != range.second; ++range.first)
261  {
262  DoorInfo const& data = range.first->second;
263 
264  if (add)
265  {
266  data.bossInfo->door[data.type].insert(door);
267  switch (data.boundary)
268  {
269  default:
270  case BOUNDARY_NONE:
271  break;
272  case BOUNDARY_N:
273  case BOUNDARY_S:
274  data.bossInfo->boundary[data.boundary] = door->GetPositionX();
275  break;
276  case BOUNDARY_E:
277  case BOUNDARY_W:
278  data.bossInfo->boundary[data.boundary] = door->GetPositionY();
279  break;
280  case BOUNDARY_NW:
281  case BOUNDARY_SE:
282  data.bossInfo->boundary[data.boundary] = door->GetPositionX() + door->GetPositionY();
283  break;
284  case BOUNDARY_NE:
285  case BOUNDARY_SW:
286  data.bossInfo->boundary[data.boundary] = door->GetPositionX() - door->GetPositionY();
287  break;
288  }
289  }
290  else
291  data.bossInfo->door[data.type].erase(door);
292  }
293 
294  if (add)
295  UpdateDoorState(door);
296 }

310 bool InstanceScript::SetBossState(uint32 id, EncounterState state)
311 {
312  if (id < bosses.size())
313  {
314  BossInfo* bossInfo = &bosses[id];
315  if (bossInfo->state == TO_BE_DECIDED) // loading
316  {
317  bossInfo->state = state;
318  return false;
319  }
320  else
321  {
322  if (bossInfo->state == state)
323  return false;
324 
325  if (state == DONE)
326  for (MinionSet::iterator i = bossInfo->minion.begin(); i != bossInfo->minion.end(); ++i)
327  if ((*i)->isWorldBoss() && (*i)->IsAlive())
328  return false;
329 
330  bossInfo->state = state;
331  SaveToDB();
332  }
333 
334  for (uint32 type = 0; type < MAX_DOOR_TYPES; ++type)
335  for (DoorSet::iterator i = bossInfo->door[type].begin(); i != bossInfo->door[type].end(); ++i)
336  UpdateDoorState(*i);
337 
338  for (MinionSet::iterator i = bossInfo->minion.begin(); i != bossInfo->minion.end(); ++i)
339  UpdateMinionState(*i, state);
340 
341  return true;
342  }
343  return false;
344 }
