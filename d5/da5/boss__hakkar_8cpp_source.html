184  bool _OnTrigger(Player* player, const AreaTrigger* /*at*/) override
185  {
186  if (InstanceScript* instance = player->GetInstanceScript())
187  {
188  // Instance map's enormous, Hakkar's GRID is not loaded by the time players enter.
189  // Without this, the creature never says anything, because it doesn't load in time.
190  player->GetMap()->LoadGrid(-11783.99f, -1655.27f);
191 
192  if (Creature* hakkar = ObjectAccessor::GetCreature(*player, instance->GetGuidData(DATA_HAKKAR)))
193  {
194  hakkar->setActive(true);
195  if (hakkar->GetAI())
196  {
197  hakkar->AI()->Talk(SAY_PROTECT_ALTAR);
198  }
199  }
200  return false;
201  }
202  return false;
203  }
