99  void UpdateAI(uint32 diff) override
100  {
101  if (!UpdateVictim())
102  return;
103 
104  _events.Update(diff);
105 
106  if (me->HasUnitState(UNIT_STATE_CASTING))
107  return;
108 
109  while (uint32 eventId = _events.ExecuteEvent())
110  {
111  switch (eventId)
112  {
113  case EVENT_ENRAGE:
114  if (!HealthAbovePct(20))
115  {
116  DoCast(me, SPELL_ENRAGE);
117  _events.ScheduleEvent(EVENT_ENRAGE, 6000);
118  _inEnrage = true;
119  }
120  break;
121  case EVENT_OVERRUN:
122  Talk(SAY_OVERRUN);
123  DoCastVictim(SPELL_OVERRUN);
124  _events.ScheduleEvent(EVENT_OVERRUN, urand(25000, 40000));
125  break;
126  case EVENT_QUAKE:
127  if (urand(0, 1))
128  return;
129 
130  Talk(SAY_EARTHQUAKE);
131 
132  //remove enrage before casting earthquake because enrage + earthquake = 16000dmg over 8sec and all dead
133  if (_inEnrage)
134  me->RemoveAurasDueToSpell(SPELL_ENRAGE);
135 
136  DoCast(me, SPELL_EARTHQUAKE);
137  _events.ScheduleEvent(EVENT_QUAKE, urand(30000, 55000));
138  break;
139  case EVENT_CHAIN:
140  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
141  DoCast(target, SPELL_CHAIN_LIGHTNING);
142  _events.ScheduleEvent(EVENT_CHAIN, urand(7000, 27000));
143  break;
144  case EVENT_ARMOR:
145  DoCastVictim(SPELL_SUNDER_ARMOR);
146  _events.ScheduleEvent(EVENT_ARMOR, urand(10000, 25000));
147  break;
148  default:
149  break;
150  }
151  }
152  DoMeleeAttackIfReady();
153  }
