116 void ContinentBuilder::Build()

131  std::vector<BuilderThread*> Threads;

133  if (TileMap->IsGlobalModel)
134  {

136 
137  TileBuilder* builder = new TileBuilder(this, Continent, 0, 0, MapId);
138  builder->AddGeometry(TileMap->Model, TileMap->ModelDefinition);

140  if (nav)

149  fwrite(&params, sizeof(dtNavMeshParams), 1, mmap);
150  fclose(mmap);
151 
152  char buff[100];

154  FILE* f = fopen(buff, "wb");
155  if (!f)
156  {
157  printf("Could not create file %s. Check that you have write permissions to the destination folder and try again\n", buff);
158  return;
159  }
160 
161  MmapTileHeader mheader;
162  mheader.size = builder->DataSize;
163  fwrite(&mheader, sizeof(MmapTileHeader), 1, f);
164  fwrite(nav, sizeof(unsigned char), builder->DataSize, f);
165  fclose(f);
166  }
167 
168  dtFree(nav);
169  delete builder;
170  }
171  else
172  {
173  params.maxPolys = 32768;
174  params.maxTiles = 4096;
175  rcVcopy(params.orig, Constants::Origin);

182  Threads.push_back(new BuilderThread(this, params));
183  printf("Map %s ( %u ) has %u tiles. Building them with %u threads\n", Continent.c_str(), MapId, uint32(TileMap->TileTable.size()), NumberOfThreads);
184  for (std::vector<TilePos>::iterator itr = TileMap->TileTable.begin(); itr != TileMap->TileTable.end(); ++itr)
185  {
