147  void Load(const char* str) override
148  {
149  if (!str)
150  {
151  OUT_LOAD_INST_DATA_FAIL;
152  return;
153  }
154 
155  OUT_LOAD_INST_DATA(str);
156 
157  char dataHead1, dataHead2;
158 
159  std::istringstream loadStream(str);
160  loadStream >> dataHead1 >> dataHead2;
161 
162  if (dataHead1 == 'Z' && dataHead2 == 'G')
163  {
164  for (uint32 i = 0; i < EncounterCount; ++i)
165  {
166  uint32 tmpState;
167  loadStream >> tmpState;
168  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
169  tmpState = NOT_STARTED;
170  SetBossState(i, EncounterState(tmpState));
171  }
172  }
173  else
174  OUT_LOAD_INST_DATA_FAIL;
175 
176  OUT_LOAD_INST_DATA_COMPLETE;
177  }
