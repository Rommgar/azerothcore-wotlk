100  void UpdateAI(uint32 diff) override
101  {
102  if (!UpdateVictim())
103  return;
104 
105  events.Update(diff);
106  if (me->HasUnitState(UNIT_STATE_CASTING))
107  return;
108 
109  switch (events.ExecuteEvent())
110  {
111  case EVENT_GROWTH:
112  Talk(EMOTE_GROW);
113  DoCast(me, SPELL_GROWTH);
114  events.ScheduleEvent(EVENT_GROWTH, 30000);
115  break;
116  case EVENT_CAVE_IN:
117  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
118  me->CastSpell(target, SPELL_CAVE_IN, false);
119  if (_caveInTimer >= 4000)
120  _caveInTimer -= 1500;
121  events.ScheduleEvent(EVENT_CAVE_IN, _caveInTimer);
122  break;
123  case EVENT_REVERBERATION:
124  me->CastSpell(me, SPELL_REVERBERATION, false);
125  events.ScheduleEvent(EVENT_REVERBERATION, 22000);
126  break;
127  case EVENT_HURTFUL_STRIKE:
128  if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 1, 5.0f))
129  {
130  me->CastSpell(target, SPELL_HURTFUL_STRIKE, false);
131  }
132  else
133  {
134  me->CastSpell(me->GetVictim(), SPELL_HURTFUL_STRIKE, false);
135  }
136  events.ScheduleEvent(EVENT_HURTFUL_STRIKE, 15000);
137  break;
138  case EVENT_GROUND_SLAM:
139  Talk(SAY_SLAM);
140  me->CastSpell(me, SPELL_GROUND_SLAM, false);
141  events.DelayEvents(8001);
142  events.ScheduleEvent(EVENT_GROUND_SLAM, 60000);
143  events.ScheduleEvent(EVENT_SHATTER, 8000);
144  me->SetControlled(true, UNIT_STATE_ROOT);
145  break;
146  case EVENT_SHATTER:
147  Talk(SAY_SHATTER);
148  me->SetControlled(false, UNIT_STATE_ROOT);
149  me->CastSpell(me, SPELL_SHATTER, false);
150  break;
151  }
152 
153  if (!me->HasUnitState(UNIT_STATE_ROOT))
154  {
155  DoMeleeAttackIfReady();
156  }
157  }
158 
159  private:
160  uint32 _caveInTimer;
161  };
