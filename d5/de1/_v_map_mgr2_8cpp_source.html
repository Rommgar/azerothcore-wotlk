118  bool VMapMgr2::_loadMap(uint32 mapId, const std::string& basePath, uint32 tileX, uint32 tileY)
119  {
120  InstanceTreeMap::iterator instanceTree = iInstanceMapTrees.find(mapId);
121  if (instanceTree == iInstanceMapTrees.end())
122  {
123  if (thread_safe_environment)
124  {
125  instanceTree = iInstanceMapTrees.insert(InstanceTreeMap::value_type(mapId, nullptr)).first;
126  }
127  else
128  ASSERT(false, "Invalid mapId %u tile [%u, %u] passed to VMapMgr2 after startup in thread unsafe environment",
129  mapId, tileX, tileY);
130  }
131 
132  if (!instanceTree->second)
133  {
134  std::string mapFileName = getMapFileName(mapId);
135  StaticMapTree* newTree = new StaticMapTree(mapId, basePath);
136  if (!newTree->InitMap(mapFileName, this))
137  {
138  delete newTree;
139  return false;
140  }
141  instanceTree->second = newTree;
142  }
143 
144  return instanceTree->second->LoadMapTile(tileX, tileY, this);
145  }

277  bool VMapMgr2::GetLiquidLevel(uint32 mapId, float x, float y, float z, uint8 reqLiquidType, float& level, float& floor, uint32& type, uint32& mogpFlags) const
278  {
279 #if defined(ENABLE_VMAP_CHECKS)

284  if (instanceTree != iInstanceMapTrees.end())
285  {
286  LocationInfo info;

292  type = info.hitModel->GetLiquidType(); // entry from LiquidType.dbc
293  mogpFlags = info.hitModel->GetMogpFlags();
294  if (reqLiquidType && !(GetLiquidFlagsPtr(type) & reqLiquidType))
295  {
296  return false;
297  }
298  if (info.hitInstance->GetLiquidLevel(pos, info, level))
299  {
300  return true;
301  }
302  }
303  }
304  }
305 
306  return false;

309  void VMapMgr2::GetAreaAndLiquidData(uint32 mapId, float x, float y, float z, uint8 reqLiquidType, AreaAndLiquidData& data) const
310  {
311  if (IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_LIQUIDSTATUS))
312  {
313  data.floorZ = z;
314  int32 adtId, rootId, groupId;
315  uint32 flags;
316  if (GetAreaInfo(mapId, x, y, data.floorZ, flags, adtId, rootId, groupId))
317  data.areaInfo.emplace(adtId, rootId, groupId, flags);
318  return;
319  }
320 
321  InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
322  if (instanceTree != iInstanceMapTrees.end())
323  {
324  LocationInfo info;
325  Vector3 pos = convertPositionToInternalRep(x, y, z);
326  if (instanceTree->second->GetLocationInfo(pos, info))
327  {
328  data.floorZ = info.ground_Z;
329  uint32 liquidType = info.hitModel->GetLiquidType();
330  float liquidLevel;
331  if (!reqLiquidType || (GetLiquidFlagsPtr(liquidType) & reqLiquidType))
332  if (info.hitInstance->GetLiquidLevel(pos, info, liquidLevel))
333  data.liquidInfo.emplace(liquidType, liquidLevel);
334 
335  if (!IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_AREAFLAG))
336  data.areaInfo.emplace(info.hitInstance->adtId, info.rootId, info.hitModel->GetWmoID(), info.hitModel->GetMogpFlags());
337  }
338  }

341  WorldModel* VMapMgr2::acquireModelInstance(const std::string& basepath, const std::string& filename)
342  {
344  std::lock_guard<std::mutex> lock(LoadedModelFilesLock);
345 
346  ModelFileMap::iterator model = iLoadedModelFiles.find(filename);
347  if (model == iLoadedModelFiles.end())
348  {
349  WorldModel* worldmodel = new WorldModel();
350  if (!worldmodel->readFile(basepath + filename + ".vmo"))
351  {
352  LOG_ERROR("maps", "VMapMgr2: could not load '%s%s.vmo'", basepath.c_str(), filename.c_str());
353  delete worldmodel;
354  return nullptr;
355  }
356  LOG_DEBUG("maps", "VMapMgr2: loading file '%s%s'", basepath.c_str(), filename.c_str());
357  model = iLoadedModelFiles.insert(std::pair<std::string, ManagedModel>(filename, ManagedModel())).first;
358  model->second.setModel(worldmodel);
359  }
360 
361  return model->second.getModel();
362  }

364  void VMapMgr2::releaseModelInstance(const std::string& filename)
365  {
367  std::lock_guard<std::mutex> lock(LoadedModelFilesLock);
368 
369  ModelFileMap::iterator model = iLoadedModelFiles.find(filename);
370  if (model == iLoadedModelFiles.end())
371  {
372  LOG_ERROR("maps", "VMapMgr2: trying to unload non-loaded file '%s'", filename.c_str());

377  LOG_DEBUG("maps", "VMapMgr2: unloading file '%s'", filename.c_str());
378  delete model->second.getModel();
