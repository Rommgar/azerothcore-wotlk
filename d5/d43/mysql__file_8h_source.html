501 static inline char *
502 inline_mysql_file_fgets(
503 #ifdef HAVE_PSI_INTERFACE
504  const char *src_file, uint src_line,
505 #endif
506  char *str, int size, MYSQL_FILE *file)
507 {
508  char *result;
509 #ifdef HAVE_PSI_INTERFACE
510  struct PSI_file_locker *locker= NULL;
511  PSI_file_locker_state state;
512  if (likely(PSI_server && file->m_psi))
513  {
514  locker= PSI_server->get_thread_file_stream_locker(&state, file->m_psi,
515  PSI_FILE_READ);
516  if (likely(locker != NULL))
517  PSI_server->start_file_wait(locker, (size_t) size, src_file, src_line);
518  }
519 #endif
520  result= fgets(str, size, file->m_file);
521 #ifdef HAVE_PSI_INTERFACE
522  if (likely(locker != NULL))
523  PSI_server->end_file_wait(locker, result ? strlen(result) : 0);
524 #endif
525  return result;
526 }
