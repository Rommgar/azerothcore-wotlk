309  {

333  me->PlayDirectSound(XT_SOUND_HEART_OPEN);
334 
335  events.CancelEventGroup(1);
336  events.ScheduleEvent(EVENT_START_SECOND_PHASE, 5000);
337  return;
338  }
339  events.RepeatEvent(1000);
340  break;
341  case EVENT_CHECK_ROOM:
342  events.RepeatEvent(5000);
