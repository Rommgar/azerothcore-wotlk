364  void DoAction(int32 action) override
365  {
366  switch (action)
367  {
368  case ACTION_STAND_UP:
369  summons.DespawnEntry(WORLD_TRIGGER);
370  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PREVENT_EMOTES_FROM_CHAT_TEXT | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
371  me->RemoveFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
372  me->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
373  me->SetReactState(REACT_AGGRESSIVE);
374  me->ForceValuesUpdateAtIndex(UNIT_NPC_FLAGS); // was in sniff. don't ask why
375  me->m_Events.AddEvent(new StandUpEvent(*me), me->m_Events.CalculateTime(1000));
376  DoAction(ACTION_REMOVE_INVOCATION);
377  me->SetHealth(1);
378  break;
379  case ACTION_CAST_INVOCATION:
380  Talk(SAY_KELESETH_INVOCATION);
381  me->CastSpell(me, SPELL_INVOCATION_OF_BLOOD_KELESETH, true);
382  me->CastSpell(me, SPELL_INVOCATION_VISUAL_ACTIVE, true);
383  _isEmpowered = true;
384  break;
385  case ACTION_REMOVE_INVOCATION:
386  _isEmpowered = false;
387  me->RemoveAurasDueToSpell(SPELL_INVOCATION_VISUAL_ACTIVE);
388  me->RemoveAurasDueToSpell(SPELL_INVOCATION_OF_BLOOD_KELESETH);
389  break;
390  default:
391  break;
392  }
393  }
