341  void UpdateAI(uint32 diff) override
342  {
343  if (!UpdateVictim())
344  return;
345 
346  events.Update(diff);
347  if (!me->HasAura(SPELL_KELTHUZAD_CHANNEL))
348  {
349  if (me->HasUnitState(UNIT_STATE_CASTING))
350  return;
351  }
352 
353  switch (events.ExecuteEvent())
354  {
355  case EVENT_FLOOR_CHANGE:
356  if (pInstance)
357  {
358  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_KELTHUZAD_FLOOR)))
359  {
360  events.ScheduleEvent(EVENT_FLOOR_CHANGE, 15000);
361  go->SetGoState(GO_STATE_READY);
362  go->SetPhaseMask(2, true);
363  }
364  }
365  break;
366  case EVENT_SPAWN_POOL:
367  SpawnHelpers();
368  break;
369  case EVENT_SUMMON_SOLDIER:
370  SummonHelper(NPC_SOLDIER_OF_THE_FROZEN_WASTES, 1);
371  events.RepeatEvent(3100);
372  break;
373  case EVENT_SUMMON_UNSTOPPABLE_ABOMINATION:
374  SummonHelper(NPC_UNSTOPPABLE_ABOMINATION, 1);
375  events.RepeatEvent(18500);
376  break;
377  case EVENT_SUMMON_SOUL_WEAVER:
378  SummonHelper(NPC_SOUL_WEAVER, 1);
379  events.RepeatEvent(30000);
380  break;
381  case EVENT_PHASE_2:
382  Talk(EMOTE_PHASE_TWO);
383  Talk(SAY_AGGRO);
384  events.Reset();
385  summons.DoAction(ACTION_SECOND_PHASE);
386  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_DISABLE_MOVE);
387  me->GetMotionMaster()->MoveChase(me->GetVictim());
388  me->RemoveAura(SPELL_KELTHUZAD_CHANNEL);
389  me->SetReactState(REACT_AGGRESSIVE);
390  events.ScheduleEvent(EVENT_FROST_BOLT_SINGLE, urand(2000, 10000));
391  events.ScheduleEvent(EVENT_FROST_BOLT_MULTI, urand(15000, 30000));
392  events.ScheduleEvent(EVENT_DETONATE_MANA, 30000);
393  events.ScheduleEvent(EVENT_PHASE_3, 1000);
394  events.ScheduleEvent(EVENT_SHADOW_FISSURE, 25000);
395  events.ScheduleEvent(EVENT_FROST_BLAST, 45000);
396  if (Is25ManRaid())
397  {
398  events.ScheduleEvent(EVENT_CHAINS, 90000);
399  }
400  break;
401  case EVENT_ENRAGE:
402  me->CastSpell(me, SPELL_BERSERK, true);
403  break;
404  case EVENT_FROST_BOLT_SINGLE:
405  me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_FROST_BOLT_SINGLE_10, SPELL_FROST_BOLT_SINGLE_25), false);
406  events.RepeatEvent(urand(2000, 10000));
407  break;
408  case EVENT_FROST_BOLT_MULTI:
409  me->CastSpell(me, RAID_MODE(SPELL_FROST_BOLT_MULTI_10, SPELL_FROST_BOLT_MULTI_25), false);
410  events.RepeatEvent(urand(15000, 30000));
411  break;
412  case EVENT_SHADOW_FISSURE:
413  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100.0f, true))
414  {
415  me->CastSpell(target, SPELL_SHADOW_FISURE, false);
416  }
417  events.RepeatEvent(25000);
418  break;
419  case EVENT_FROST_BLAST:
420  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, RAID_MODE(1, 0), 0, true))
421  {
422  me->CastSpell(target, SPELL_FROST_BLAST, false);
423  }
424  Talk(SAY_FROST_BLAST);
425  events.RepeatEvent(45000);
426  break;
427  case EVENT_CHAINS:
428  for (uint8 i = 0; i < 3; ++i)
429  {
430  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 200, true, -SPELL_CHAINS_OF_KELTHUZAD))
431  {
432  me->CastSpell(target, SPELL_CHAINS_OF_KELTHUZAD, true);
433  }
434  }
435  Talk(SAY_CHAIN);
436  events.RepeatEvent(90000);
437  break;
438  case EVENT_DETONATE_MANA:
439  {
440  std::vector<Unit*> unitList;
441  ThreatContainer::StorageType const& threatList = me->getThreatMgr().getThreatList();
442  for (auto itr : threatList)
443  {
444  if (itr->getTarget()->GetTypeId() == TYPEID_PLAYER
445  && itr->getTarget()->getPowerType() == POWER_MANA
446  && itr->getTarget()->GetPower(POWER_MANA))
447  {
448  unitList.push_back(itr->getTarget());
449  }
450  }
451  if (!unitList.empty())
452  {
453  auto itr = unitList.begin();
454  advance(itr, urand(0, unitList.size() - 1));
455  me->CastSpell(*itr, SPELL_DETONATE_MANA, false);
456  Talk(SAY_SPECIAL);
457  }
458  events.RepeatEvent(30000);
459  break;
460  }
461  case EVENT_PHASE_3:
462  if (me->HealthBelowPct(45))
463  {
464  Talk(SAY_REQUEST_AID);
465  events.DelayEvents(5500);
466  events.ScheduleEvent(EVENT_P3_LICH_KING_SAY, 5000);
467  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_KELTHUZAD_PORTAL_1)))
468  {
469  go->SetGoState(GO_STATE_ACTIVE);
470  }
471  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_KELTHUZAD_PORTAL_2)))
472  {
473  go->SetGoState(GO_STATE_ACTIVE);
474  }
475  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_KELTHUZAD_PORTAL_3)))
476  {
477  go->SetGoState(GO_STATE_ACTIVE);
478  }
479  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_KELTHUZAD_PORTAL_4)))
480  {
481  go->SetGoState(GO_STATE_ACTIVE);
482  }
483  break;
484  }
485  events.RepeatEvent(1000);
486  break;
487  case EVENT_P3_LICH_KING_SAY:
488  if (pInstance)
489  {
490  if (Creature* cr = ObjectAccessor::GetCreature(*me, pInstance->GetGuidData(DATA_LICH_KING_BOSS)))
491  {
492  cr->AI()->Talk(SAY_ANSWER_REQUEST);
493  }
494  }
495  for (uint8 i = 0 ; i < RAID_MODE(2, 4); ++i)
496  {
497  events.ScheduleEvent(EVENT_SUMMON_GUARDIAN_OF_ICECROWN, 10000 + (i * 5000));
498  }
499  break;
500  case EVENT_SUMMON_GUARDIAN_OF_ICECROWN:
501  if (Creature* cr = me->SummonCreature(NPC_GUARDIAN_OF_ICECROWN, SpawnPool[RAND(0, 1, 3, 4)]))
502  {
503  cr->AI()->Talk(EMOTE_GUARDIAN_APPEAR);
504  cr->AI()->AttackStart(me->GetVictim());
505  }
506  break;
507  }
508  if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE))
509  DoMeleeAttackIfReady();
510  }
511  };
