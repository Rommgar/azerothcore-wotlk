83 static int ptr_compare(size_t *compare_length, uchar **a, uchar **b)
84 {
85  reg3 int length= *compare_length;
86  reg1 uchar *first,*last;
87 
88  first= *a; last= *b;
89  while (--length)
90  {
91  if (*first++ != *last++)
92  return (int) first[-1] - (int) last[-1];
93  }
94  return (int) first[0] - (int) last[0];
95 }
