238 void WorldSession::HandleCalendarAddEvent(WorldPacket& recvData)
239 {
240  ObjectGuid guid = _player->GetGUID();
241 
242  std::string title;
243  std::string description;
244  uint8 type;
245  uint8 repeatable;
246  uint32 maxInvites;
247  int32 dungeonId;
248  uint32 eventPackedTime;
249  uint32 unkPackedTime;
250  uint32 flags;
251 
252  recvData >> title >> description >> type >> repeatable >> maxInvites >> dungeonId;
253  recvData.ReadPackedTime(eventPackedTime);
254  recvData.ReadPackedTime(unkPackedTime);
255  recvData >> flags;
256 
257  // prevent attacks with non-utf8 chars -> with multiple packets it will hang up the db due to errors.
258  if (!validUtf8String(recvData, title, "create", guid) || title.size() > 31 || !validUtf8String(recvData, description, "create", guid) || description.size() > 255)
259  return;
260 
261  // prevent events in the past
262  // To Do: properly handle timezones and remove the "- time_t(86400L)" hack
263  if (time_t(eventPackedTime) < (time(nullptr) - time_t(86400L)))
264  {
265  recvData.rfinish();
266  sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENT_PASSED);
267  return;
268  }
269 
270  // If the event is a guild event, check if the player is in a guild
271  if (CalendarEvent::IsGuildEvent(flags) || CalendarEvent::IsGuildAnnouncement(flags))
272  {
273  if (!_player->GetGuildId())
274  {
275  recvData.rfinish();
276  sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_GUILD_PLAYER_NOT_IN_GUILD);
277  return;
278  }
279  }
280 
281  // Check if the player reached the max number of events allowed to create
282  if (CalendarEvent::IsGuildEvent(flags) || CalendarEvent::IsGuildAnnouncement(flags))
283  {
284  if (sCalendarMgr->GetGuildEvents(_player->GetGuildId()).size() >= CALENDAR_MAX_GUILD_EVENTS)
285  {
286  recvData.rfinish();
287  sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_GUILD_EVENTS_EXCEEDED);
288  return;
289  }
290  }
291  else
292  {
293  if (sCalendarMgr->GetEventsCreatedBy(guid).size() >= CALENDAR_MAX_EVENTS)
294  {
295  recvData.rfinish();
296  sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_EVENTS_EXCEEDED);
297  return;
298  }
299  }
300 
301  if (GetCalendarEventCreationCooldown() > time(nullptr))
302  {
303  recvData.rfinish();
304  sCalendarMgr->SendCalendarCommandResult(guid, CALENDAR_ERROR_INTERNAL);
305  return;
306  }
307  SetCalendarEventCreationCooldown(time(nullptr) + CALENDAR_CREATE_EVENT_COOLDOWN);
308 
309  CalendarEvent* calendarEvent = new CalendarEvent(sCalendarMgr->GetFreeEventId(), guid, 0, CalendarEventType(type), dungeonId,
310  time_t(eventPackedTime), flags, time_t(unkPackedTime), title, description);
311 
312  if (calendarEvent->IsGuildEvent() || calendarEvent->IsGuildAnnouncement())
313  if (Player* creator = ObjectAccessor::FindConnectedPlayer(guid))
314  calendarEvent->SetGuildId(creator->GetGuildId());
315 
316  if (calendarEvent->IsGuildAnnouncement())
317  {
318  // 946684800 is 01/01/2000 00:00:00 - default response time
319  CalendarInvite* invite = new CalendarInvite(0, calendarEvent->GetEventId(), ObjectGuid::Empty, guid, 946684800, CALENDAR_STATUS_NOT_SIGNED_UP, CALENDAR_RANK_PLAYER, "");
320  sCalendarMgr->AddInvite(calendarEvent, invite);
321  }
322  else
323  {
324  uint32 inviteCount;
325  ObjectGuid invitee[CALENDAR_MAX_INVITES];
326  uint8 status[CALENDAR_MAX_INVITES];
327  uint8 rank[CALENDAR_MAX_INVITES];
328 
329  memset(status, 0, sizeof(status));
330  memset(rank, 0, sizeof(rank));
331 
332  try
333  {
334  recvData >> inviteCount;
335 
336  for (uint32 i = 0; i < inviteCount && i < CALENDAR_MAX_INVITES; ++i)
337  {
338  recvData >> invitee[i].ReadAsPacked();
339  recvData >> status[i] >> rank[i];
340  }
341  }
342  catch (ByteBufferException const&)
343  {
344  delete calendarEvent;
345  calendarEvent = nullptr;
346  throw;
347  }
348 
349  CharacterDatabaseTransaction trans;
350  if (inviteCount > 1)
351  trans = CharacterDatabase.BeginTransaction();
352 
353  for (uint32 i = 0; i < inviteCount && i < CALENDAR_MAX_INVITES; ++i)
354  {
355  // 946684800 is 01/01/2000 00:00:00 - default response time
356  CalendarInvite* invite = new CalendarInvite(sCalendarMgr->GetFreeInviteId(), calendarEvent->GetEventId(), invitee[i], guid, 946684800, CalendarInviteStatus(status[i]), CalendarModerationRank(rank[i]), "");
357  sCalendarMgr->AddInvite(calendarEvent, invite, trans);
358  }
359 
360  if (inviteCount > 1)
361  CharacterDatabase.CommitTransaction(trans);
362  }
363 
364  sCalendarMgr->AddEvent(calendarEvent, CALENDAR_SENDTYPE_ADD);
365 }
