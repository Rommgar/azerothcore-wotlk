136  void UpdateAI(uint32 diff) override
137  {

139  return;
140 
141  events.Update(diff);
142 
143  while (uint32 eventId = events.ExecuteEvent())
144  {
145  switch (eventId)
146  {
147  case EVENT_ARCANE_EXPLOSION:
148  DoCastAOE(SPELL_ARCANE_EXPLOSION, true);
149  events.ScheduleEvent(EVENT_ARCANE_EXPLOSION, urand(8000, 18000));
150  break;
151  case EVENT_FULLFILMENT:
153  // Spell actually works, tested in duel
154  DoCast(SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true), SPELL_TRUE_FULFILLMENT, true);
155  events.ScheduleEvent(EVENT_FULLFILMENT, urand(20000, 30000));
156  break;
157  case EVENT_BLINK:
158  DoCast(me, BlinkSpells[urand(0, 2)]);
159  DoResetThreat();
160  me->SetVisible(true);
161  events.ScheduleEvent(EVENT_BLINK, urand(10000, 30000));
162  break;
163  case EVENT_EARTH_SHOCK:
164  DoCastVictim(SPELL_EARTH_SHOCK);
165  events.ScheduleEvent(EVENT_EARTH_SHOCK, 2000);
166  break;
167  }
168  }
169 
170  if (!me->IsSummon() && me->GetHealthPct() < _hpct)
171  {
172  DoCast(me, SPELL_SUMMON_IMAGES);
173  Talk(SAY_SPLIT);
174  _hpct -= 25.0f;
175  me->SetVisible(false);
176  events.RescheduleEvent(EVENT_BLINK, 2000);
177  }
178 
179  if (me->IsWithinMeleeRange(me->GetVictim()))
180  {
181  events.RescheduleEvent(EVENT_EARTH_SHOCK, 2000);
182  DoMeleeAttackIfReady();
183  }
184  }
