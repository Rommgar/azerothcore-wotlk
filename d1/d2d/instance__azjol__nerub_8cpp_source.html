109  void Load(const char* in) override
110  {
111  if( !in )
112  return;
113 
114  char dataHead1, dataHead2;
115  std::istringstream loadStream(in);
116  loadStream >> dataHead1 >> dataHead2;
117  if (dataHead1 == 'A' && dataHead2 == 'N')
118  {
119  for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
120  {
121  uint32 tmpState;
122  loadStream >> tmpState;
123  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
124  tmpState = NOT_STARTED;
125  SetBossState(i, EncounterState(tmpState));
126  }
127  }
128  }
