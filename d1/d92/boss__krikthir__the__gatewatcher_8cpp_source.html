163  void UpdateAI(uint32 diff) override
164  {
165  events2.Update(diff);
166  switch (events2.ExecuteEvent())
167  {
168  case EVENT_KRIK_START_WAVE:
169  me->CastCustomSpell(SPELL_SUBBOSS_AGGRO_TRIGGER, SPELLVALUE_MAX_TARGETS, 1, me, true);
170  Talk(SAY_SEND_GROUP);
171  break;
172  case EVENT_KRIK_ENTER_COMBAT:
173  me->SetInCombatWithZone();
174  break;
175  case EVENT_KRIK_CHECK_EVADE:
176  if (!SelectTargetFromPlayerList(100.0f))
177  {
178  EnterEvadeMode();
179  return;
180  }
181  events2.ScheduleEvent(EVENT_KRIK_CHECK_EVADE, 5000);
182  break;
183  }
184 
185  if (!UpdateVictim())
186  return;
187 
188  events.Update(diff);
189  if (me->HasUnitState(UNIT_STATE_CASTING))
190  return;
191 
192  switch (events.ExecuteEvent())
193  {
194  case EVENT_KRIK_HEALTH_CHECK:
195  if (HealthBelowPct(10))
196  {
197  me->CastSpell(me, SPELL_FRENZY, true);
198  break;
199  }
200  events.ScheduleEvent(EVENT_KRIK_HEALTH_CHECK, 1000);
201  break;
202  case EVENT_KRIK_SUMMON:
203  Talk(SAY_SWARM);
204  me->CastSpell(me, SPELL_SWARM, false);
205  events.ScheduleEvent(EVENT_KRIK_SUMMON, 20000);
206  break;
207  case EVENT_KRIK_MIND_FLAY:
208  me->CastSpell(me->GetVictim(), SPELL_MIND_FLAY, false);
209  events.ScheduleEvent(EVENT_KRIK_MIND_FLAY, 15000);
210  break;
211  case EVENT_KRIK_CURSE:
212  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
213  me->CastSpell(target, SPELL_CURSE_OF_FATIGUE, true);
214  events.ScheduleEvent(EVENT_KRIK_CURSE, 10000);
215  break;
216  case EVENT_CALL_ADDS:
217  summons.DoZoneInCombat();
218  break;
219  }
220 
221  DoMeleeAttackIfReady();
222  }
