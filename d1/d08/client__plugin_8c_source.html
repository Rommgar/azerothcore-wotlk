57 /*
58  Loaded plugins are stored in a linked list.
59  The list is append-only, the elements are added to the head (like in a stack).
60  The elements are added under a mutex, but the list can be read and traversed
61  without any mutex because once an element is added to the list, it stays
62  there. The main purpose of a mutex is to prevent two threads from
63  loading the same plugin twice in parallel.
64 */
65 struct st_client_plugin_int *plugin_list[MYSQL_CLIENT_MAX_PLUGINS];
66 static pthread_mutex_t LOCK_load_client_plugin;

90 find_plugin(const char *name, int type)
91 {
92  struct st_client_plugin_int *p;
93 
94  DBUG_ASSERT(initialized);
95  DBUG_ASSERT(type >= 0 && type < MYSQL_CLIENT_MAX_PLUGINS);
96  if (type < 0 || type >= MYSQL_CLIENT_MAX_PLUGINS)
97  return 0;
98 
99  for (p= plugin_list[type]; p; p= p->next)
100  {
101  if (strcmp(p->plugin->name, name) == 0)
102  return p->plugin;
103  }
104  return NULL;
105 }

119 static struct st_mysql_client_plugin *
120 add_plugin(MYSQL *mysql, struct st_mysql_client_plugin *plugin, void *dlhandle,
121  int argc, va_list args)
122 {
123  const char *errmsg;
124  struct st_client_plugin_int plugin_int, *p;
125  char errbuf[1024];
126 
127  DBUG_ASSERT(initialized);
128 
129  plugin_int.plugin= plugin;
130  plugin_int.dlhandle= dlhandle;
131 
132  if (plugin->type >= MYSQL_CLIENT_MAX_PLUGINS)
133  {
134  errmsg= "Unknown client plugin type";
135  goto err1;
136  }
137 
138  if (plugin->interface_version < plugin_version[plugin->type] ||
139  (plugin->interface_version >> 8) >
140  (plugin_version[plugin->type] >> 8))
141  {
142  errmsg= "Incompatible client plugin interface";
143  goto err1;
144  }
145 
146  /* Call the plugin initialization function, if any */
147  if (plugin->init && plugin->init(errbuf, sizeof(errbuf), argc, args))
148  {
149  errmsg= errbuf;
150  goto err1;
151  }
152 
153  p= (struct st_client_plugin_int *)
154  memdup_root(&mem_root, &plugin_int, sizeof(plugin_int));
155 
156  if (!p)
157  {
158  errmsg= "Out of memory";
159  goto err2;
160  }
161 
162  safe_mutex_assert_owner(&LOCK_load_client_plugin);
163 
164  p->next= plugin_list[plugin->type];
165  plugin_list[plugin->type]= p;
166 
167  return plugin;
168 
169 err2:
170  if (plugin->deinit)
171  plugin->deinit();
172 err1:
173  set_mysql_extended_error(mysql, CR_AUTH_PLUGIN_CANNOT_LOAD, unknown_sqlstate,
174  ER(CR_AUTH_PLUGIN_CANNOT_LOAD), plugin->name,
175  errmsg);
176  if (dlhandle)
177  dlclose(dlhandle);
178  return NULL;
179 }

226 int mysql_client_plugin_init()
227 {
228  MYSQL mysql;
229  struct st_mysql_client_plugin **builtin;
230 
231  if (initialized)
232  return 0;
233 
234  bzero(&mysql, sizeof(mysql)); /* dummy mysql for set_mysql_extended_error */
235 
236  pthread_mutex_init(&LOCK_load_client_plugin, MY_MUTEX_INIT_SLOW);
237  init_alloc_root(&mem_root, 128, 128);
238 
239  bzero(&plugin_list, sizeof(plugin_list));
240 
241  initialized= 1;
242 
243  pthread_mutex_lock(&LOCK_load_client_plugin);
244 
245  for (builtin= mysql_client_builtins; *builtin; builtin++)
246  add_plugin(&mysql, *builtin, 0, 0, 0);
247 
248  pthread_mutex_unlock(&LOCK_load_client_plugin);
249 
250  load_env_plugins(&mysql);
251 
252  return 0;
253 }

260 void mysql_client_plugin_deinit()
261 {
262  int i;
263  struct st_client_plugin_int *p;
264 
265  if (!initialized)
266  return;
267 
268  for (i=0; i < MYSQL_CLIENT_MAX_PLUGINS; i++)
269  for (p= plugin_list[i]; p; p= p->next)
270  {
271  if (p->plugin->deinit)
272  p->plugin->deinit();
273  if (p->dlhandle)
274  dlclose(p->dlhandle);
275  }
276 
277  bzero(&plugin_list, sizeof(plugin_list));
278  initialized= 0;
279  free_root(&mem_root, MYF(0));
280  pthread_mutex_destroy(&LOCK_load_client_plugin);
281 }

312 mysql_load_plugin_v(MYSQL *mysql, const char *name, int type,
313  int argc, va_list args)
314 {
315  const char *errmsg;
316  char dlpath[FN_REFLEN+1];
317  void *sym, *dlhandle;
318  struct st_mysql_client_plugin *plugin;
319 #ifdef _WIN32
320  char win_errormsg[2048];
321 #endif
322 
323  DBUG_ENTER ("mysql_load_plugin_v");
324  DBUG_PRINT ("entry", ("name=%s type=%d int argc=%d", name, type, argc));
325  if (is_not_initialized(mysql, name))
326  {
327  DBUG_PRINT ("leave", ("mysql not initialized"));
328  DBUG_RETURN (NULL);
329  }
330 
331  pthread_mutex_lock(&LOCK_load_client_plugin);
332 
333  /* make sure the plugin wasn't loaded meanwhile */
334  if (type >= 0 && find_plugin(name, type))
335  {
336  errmsg= "it is already loaded";
337  goto err;
338  }
339 
340  /* Compile dll path */
341  strxnmov(dlpath, sizeof(dlpath) - 1,
342  mysql->options.extension && mysql->options.extension->plugin_dir ?
343  mysql->options.extension->plugin_dir : PLUGINDIR, "/",
344  name, SO_EXT, NullS);
345 
346  DBUG_PRINT ("info", ("dlopeninig %s", dlpath));
347  /* Open new dll handle */
348  if (!(dlhandle= dlopen(dlpath, RTLD_NOW)))
349  {
350 #if defined(__APPLE__)
351  /* Apple supports plugins with .so also, so try this as well */
352  strxnmov(dlpath, sizeof(dlpath) - 1,
353  mysql->options.extension && mysql->options.extension->plugin_dir ?
354  mysql->options.extension->plugin_dir : PLUGINDIR, "/",
355  name, ".so", NullS);
356  if ((dlhandle= dlopen(dlpath, RTLD_NOW)))
357  goto have_plugin;
358 #endif
359 
360  DBUG_PRINT ("info", ("failed to dlopen"));
361 #ifdef _WIN32
362  if(FormatMessage(FORMAT_MESSAGE_FROM_SYSTEM,
363  0, GetLastError(), 0, win_errormsg, 2048, NULL))
364  errmsg= win_errormsg;
365  else
366  errmsg= "";
367 #else
368  errmsg= dlerror();
369 #endif
370  goto err;
371  }
372 
373 #if defined(__APPLE__)
374 have_plugin:
375 #endif
376  if (!(sym= dlsym(dlhandle, plugin_declarations_sym)))
377  {
378  errmsg= "not a plugin";
379  dlclose(dlhandle);
380  goto err;
381  }
382 
383  plugin= (struct st_mysql_client_plugin*)sym;
384 
385  if (type >=0 && type != plugin->type)
386  {
387  errmsg= "type mismatch";
388  goto err;
389  }
390 
391  if (strcmp(name, plugin->name))
392  {
393  errmsg= "name mismatch";
394  goto err;
395  }
396 
397  if (type < 0 && find_plugin(name, plugin->type))
398  {
399  errmsg= "it is already loaded";
400  goto err;
401  }
402 
403  plugin= add_plugin(mysql, plugin, dlhandle, argc, args);
404 
405  pthread_mutex_unlock(&LOCK_load_client_plugin);
406 
407  DBUG_PRINT ("leave", ("plugin loaded ok"));
408  DBUG_RETURN (plugin);
409 
410 err:
411  pthread_mutex_unlock(&LOCK_load_client_plugin);
412  DBUG_PRINT ("leave", ("plugin load error : %s", errmsg));
413  set_mysql_extended_error(mysql, CR_AUTH_PLUGIN_CANNOT_LOAD, unknown_sqlstate,
414  ER(CR_AUTH_PLUGIN_CANNOT_LOAD), name, errmsg);
415  DBUG_RETURN (NULL);
416 }
