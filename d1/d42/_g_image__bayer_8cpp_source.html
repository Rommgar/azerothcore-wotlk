72 static uint8 applyFilter(
73  const uint8* I,
74  int x,
75  int y,
76  int w,
77  int h,
78  const float filter[5][5]) {
79 
80  debugAssert(isEven(w));
81  debugAssert(isEven(h));
82 
83  float sum = 0.0f;
84  float denom = 0.0f;
85 
86  for (int dy = 0; dy < 5; ++dy) {
87  int offset = ((y + dy + h - 2) % h) * w;
88 
89  for (int dx = 0; dx < 5; ++dx) {
90  float f = filter[dy][dx];
91  sum += f * I[((x + dx + w - 2) % w) + offset];
92  denom += f;
93  }
94  }
95 
96  return (uint8)iClamp(iRound(sum / denom), 0, 255);

102 //
103 
104 // There are two kinds of rows (GR and BG).
105 // In each row, there are two kinds of pixels (G/R, B/G).
106 // We express the four kinds of INPUT pixels as:
107 // GRG, GRG, BGB, BGG
108 //
109 // There are three kinds of OUTPUT pixels: R, G, B.
110 // Thus there are nominally 12 different I/O combinations,
111 // but several are impulses because needed output at that
112 // location *is* the input (e.g., G_GRG and G_BGG).
113 //
114 // The following 5x5 row-major filters are named as output_input.
