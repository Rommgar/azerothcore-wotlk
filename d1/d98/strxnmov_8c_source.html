18 /* File : strxnmov.c

23  strxnmov(dst, len, src1, ..., srcn, NullS)
24  moves the first len characters of the concatenation of src1,...,srcn
25  to dst and add a closing NUL character.
26  It is just like strnmov except that it concatenates multiple sources.
27  Beware: the last argument should be the null character pointer.
28  Take VERY great care not to omit it! Also be careful to use NullS
29  and NOT to use 0, as on some machines 0 is not the same size as a
30  character pointer, or not the same bit pattern as NullS.
31 
32  NOTE
33  strxnmov is like strnmov in that it moves up to len
34  characters; dst will be padded on the right with one '\0' character.
35  if total-string-length >= length then dst[length] will be set to \0
36 */

42 char *strxnmov(char *dst, size_t len, const char *src, ...)
43 {
44  va_list pvar;
45  char *end_of_dst=dst+len;
46 
47  va_start(pvar,src);
48  while (src != NullS)
49  {
50  do
51  {
52  if (dst == end_of_dst)
53  goto end;
54  }
55  while ((*dst++ = *src++));
56  dst--;
57  src = va_arg(pvar, char *);
58  }
59 end:
60  *dst=0;
61  va_end(pvar);
62  return dst;
