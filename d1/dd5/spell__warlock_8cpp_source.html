333  void HandlePeriodic(AuraEffect const* aurEff)
334  {
335  PreventDefaultAction();
336  if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
337  {
338  int32 currentAmount = aurEff->GetAmount();
339  int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
340  if (newAmount != currentAmount)
341  {
342  if (aurEff->GetMiscValue() == STAT_STAMINA)
343  {
344  uint32 actStat = GetUnitOwner()->GetHealth();
345  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
346  GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
347  }
348  else
349  {
350  uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
351  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
352  GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
353  }
354  }
355  }
356  else
357  GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
358  }
