121  {

123  return;
124 
125  events.Update(diff);
126

134  case EVENT_MORTALCLEAVE:

136  events.ScheduleEvent(EVENT_MORTALCLEAVE, urand(15000, 20000), 0, PHASE_ONE);
137  break;

139  DoCastVictim(SPELL_SILENCE, true);
140  events.ScheduleEvent(EVENT_SILENCE, urand(20000, 25000), 0, PHASE_ONE);
141  break;
142  case EVENT_RESURRECT_TIMER:
143  //Thekal will transform to Tiger if he died and was not resurrected after 10 seconds.
144  if (WasDead)
145  {
146  DoCast(me, SPELL_TIGER_FORM); // SPELL_AURA_TRANSFORM
147  me->SetObjectScale(2.00f);
148  me->SetStandState(UNIT_STAND_STATE_STAND);
149  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
150  /*
151  const CreatureTemplate* cinfo = me->GetCreatureTemplate();
152  me->SetBaseWeaponDamage(BASE_ATTACK, MINDAMAGE, (cinfo->mindmg +((cinfo->mindmg/100) * 40)));

157  DoResetThreat();
158  events.ScheduleEvent(EVENT_FRENZY, 30000, 0, PHASE_TWO); // Phase 2
159  events.ScheduleEvent(EVENT_FORCEPUNCH, 4000, 0, PHASE_TWO); // Phase 2

162  events.ScheduleEvent(EVENT_SUMMONTIGERS, 25000, 0, PHASE_TWO); // Phase 2
163  events.SetPhase(PHASE_TWO);
164  }
165  events.ScheduleEvent(EVENT_RESURRECT_TIMER, 10000, 0, PHASE_ONE);
166  break;
167  case EVENT_CHECK_TIMER:
168  //Check_Timer for the death of LorKhan and Zath.
169  if (!WasDead)
170  {
171  if (instance->GetBossState(DATA_LORKHAN) == SPECIAL)
172  {
173  //Resurrect LorKhan
174  if (Unit* pLorKhan = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_LORKHAN)))
175  {
176  pLorKhan->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
177  pLorKhan->SetFaction(FACTION_MONSTER);

179  pLorKhan->SetFullHealth();
180  instance->SetData(DATA_LORKHAN, DONE);
181  }
182  }
183 
184  if (instance->GetBossState(DATA_ZATH) == SPECIAL)
185  {
186  //Resurrect Zath
187  if (Unit* pZath = ObjectAccessor::GetUnit(*me, instance->GetGuidData(DATA_ZATH)))
188  {
189  pZath->SetUInt32Value(UNIT_FIELD_BYTES_1, 0);
190  pZath->SetFaction(FACTION_MONSTER);
191  pZath->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
192  pZath->SetFullHealth();
193  instance->SetBossState(DATA_ZATH, DONE);
194  }
195  }
196  }
197  events.ScheduleEvent(EVENT_CHECK_TIMER, 5000, 0, PHASE_ONE);
198  break;
199  case EVENT_FRENZY:
200  DoCast(me, SPELL_FRENZY);
201  events.ScheduleEvent(EVENT_FRENZY, 30000, 0, PHASE_TWO);
202  break;
203  case EVENT_FORCEPUNCH:
204  DoCastVictim(SPELL_FORCEPUNCH, true);
205  events.ScheduleEvent(EVENT_FORCEPUNCH, urand(16000, 21000), 0, PHASE_TWO);
206  break;
207  case EVENT_CHARGE:
208  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
209  {
210  DoCast(target, SPELL_CHARGE);
211  DoResetThreat();
212  AttackStart(target);
213  }
214  events.ScheduleEvent(EVENT_CHARGE, urand(15000, 22000), 0, PHASE_TWO);
215  break;
216  case EVENT_ENRAGE:
217  if (HealthBelowPct(11) && !Enraged)
218  {
219  DoCast(me, SPELL_ENRAGE);
220  Enraged = true;
221  }
222  events.ScheduleEvent(EVENT_ENRAGE, 30000);
223  break;
224  case EVENT_SUMMONTIGERS:
225  DoCastVictim(SPELL_SUMMONTIGERS, true);
226  events.ScheduleEvent(EVENT_SUMMONTIGERS, urand(10000, 14000), 0, PHASE_TWO);
227  break;
228  default:
229  break;
230  }
231 
232  if (me->IsFullHealth() && WasDead)
233  WasDead = false;
234 
235  if ((events.IsInPhase(PHASE_ONE)) && !WasDead && !HealthAbovePct(5))
236  {
237  me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE_PERCENT);
238  me->RemoveAurasByType(SPELL_AURA_PERIODIC_DAMAGE);

241  me->SetStandState(UNIT_STAND_STATE_SLEEP);
242  me->AttackStop();
243  instance->SetBossState(DATA_THEKAL, SPECIAL);
244  WasDead = true;
245  }
246  }
247  DoMeleeAttackIfReady();
248  }
