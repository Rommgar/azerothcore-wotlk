119 void SmartAI::GenerateWayPointArray(Movement::PointsArray* points)
120 {
121  if (!mWayPoints || mWayPoints->empty())
122  return;
123 
124  // Flying unit, just fill array
125  if (me->m_movementInfo.HasMovementFlag((MovementFlags)(MOVEMENTFLAG_CAN_FLY | MOVEMENTFLAG_DISABLE_GRAVITY)))
126  {
127  // xinef: first point in vector is unit real position
128  points->clear();
129  points->push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
130  uint32 wpCounter = mCurrentWPID;
131  WPPath::const_iterator itr;
132  while ((itr = mWayPoints->find(wpCounter++)) != mWayPoints->end())
133  {
134  WayPoint* wp = (*itr).second;
135  points->push_back(G3D::Vector3(wp->x, wp->y, wp->z));
136  }
137  }
138  else
139  {
140  for (float size = 1.0f; size; size *= 0.5f)
141  {
142  std::vector<G3D::Vector3> pVector;
143  // xinef: first point in vector is unit real position
144  pVector.push_back(G3D::Vector3(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ()));
145  uint32 length = (mWayPoints->size() - mCurrentWPID) * size;
146 
147  uint32 cnt = 0;
148  uint32 wpCounter = mCurrentWPID;
149  WPPath::const_iterator itr;
150  while ((itr = mWayPoints->find(wpCounter++)) != mWayPoints->end() && cnt++ <= length)
151  {
152  WayPoint* wp = (*itr).second;
153  pVector.push_back(G3D::Vector3(wp->x, wp->y, wp->z));
154  }
155 
156  if (pVector.size() > 2) // more than source + dest
157  {
158  G3D::Vector3 middle = (pVector[0] + pVector[pVector.size() - 1]) / 2.f;
159  G3D::Vector3 offset;
160 
161  bool continueLoop = false;
162  for (uint32 i = 1; i < pVector.size() - 1; ++i)
163  {
164  offset = middle - pVector[i];
165  if (fabs(offset.x) >= 0xFF || fabs(offset.y) >= 0xFF || fabs(offset.z) >= 0x7F)
166  {
167  // offset is too big, split points
168  continueLoop = true;
169  break;
170  }
171  }
172  if (continueLoop)
173  continue;
174  }
175  // everything ok
176  *points = pVector;
177  break;
178  }
179  }

283 void SmartAI::EndPath(bool fail)
284 {
285  RemoveEscortState(SMART_ESCORT_ESCORTING | SMART_ESCORT_PAUSED | SMART_ESCORT_RETURNING);
286  mWayPoints = nullptr;
287  mLastWP = nullptr;
288  mWPPauseTimer = 0;
289 
290  if (mEscortNPCFlags)
291  {
292  me->SetUInt32Value(UNIT_NPC_FLAGS, mEscortNPCFlags);
293  mEscortNPCFlags = 0;
294  }
295 
296  ObjectList* targets = GetScript()->GetTargetList(SMART_ESCORT_TARGETS);
297  if (targets && mEscortQuestID)
298  {
299  if (targets->size() == 1 && GetScript()->IsPlayer((*targets->begin())))
300  {
301  Player* player = (*targets->begin())->ToPlayer();
302  if (Group* group = player->GetGroup())
303  {
304  for (GroupReference* groupRef = group->GetFirstMember(); groupRef != nullptr; groupRef = groupRef->next())
305  {
306  Player* groupGuy = groupRef->GetSource();
307  if (!groupGuy || !player->IsInMap(groupGuy))
308  continue;
309 
310  if (!fail && groupGuy->IsAtGroupRewardDistance(me) && !groupGuy->HasCorpse())
311  groupGuy->AreaExploredOrEventHappens(mEscortQuestID);
312  else if (fail && groupGuy->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
313  groupGuy->FailQuest(mEscortQuestID);
314  }
315  }
316  else
317  {
318  if (!fail && player->IsAtGroupRewardDistance(me) && !player->HasCorpse())
319  player->GroupEventHappens(mEscortQuestID, me);
320  else if (fail && player->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
321  player->FailQuest(mEscortQuestID);
322  }
323  }
324  else
325  {
326  for (ObjectList::iterator iter = targets->begin(); iter != targets->end(); ++iter)
327  {
328  if (GetScript()->IsPlayer((*iter)))
329  {
330  Player* player = (*iter)->ToPlayer();
331  if (!fail && player->IsAtGroupRewardDistance(me) && !player->HasCorpse())
332  player->AreaExploredOrEventHappens(mEscortQuestID);
333  else if (fail && player->GetQuestStatus(mEscortQuestID) == QUEST_STATUS_INCOMPLETE)
334  player->FailQuest(mEscortQuestID);
335  }
336  }
337  }
338  }
339 
340  // Xinef: if the escort failed - DO NOT PROCESS ANYTHING, ITS RETARDED
341  // Xinef: End Path events should be only processed if it was SUCCESSFUL stop or stop called by SMART_ACTION_WAYPOINT_STOP
342  if (fail)
343  {
344  mCurrentWPID = 0;
345  return;
346  }
347 
348  GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_ENDED, nullptr, mCurrentWPID, GetScript()->GetPathId());
349  mCurrentWPID = 0;
350 
351  if (mCanRepeatPath)
352  StartPath(mRun, GetScript()->GetPathId(), mCanRepeatPath);
353  else
354  GetScript()->SetPathId(0);
355 
356  if (mDespawnState == 1)
357  StartDespawn();
358 }

381 void SmartAI::UpdatePath(const uint32 diff)
382 {
383  if (!HasEscortState(SMART_ESCORT_ESCORTING))
384  return;
385 
386  if (mEscortInvokerCheckTimer < diff)
387  {
388  // Xinef: Escort failed - no players in range
389  // Xinef: Despawn immediately
390  if (!IsEscortInvokerInRange())
391  {
392  StopPath(0, mEscortQuestID, true);
393 
394  // Xinef: allow to properly hook out of range despawn action, which in most cases should perform the same operation as dying
395  GetScript()->ProcessEventsFor(SMART_EVENT_DEATH, me);
396  me->DespawnOrUnsummon(1);
397  return;
398  }
399  mEscortInvokerCheckTimer = 1000;
400  }
401  else
402  mEscortInvokerCheckTimer -= diff;
403 
404  // handle pause
405  if (HasEscortState(SMART_ESCORT_PAUSED))
406  {
407  if (mWPPauseTimer < diff)
408  {
409  if (!me->IsInCombat() && !HasEscortState(SMART_ESCORT_RETURNING) && (mWPReached || mForcedPaused))
410  {
411  GetScript()->ProcessEventsFor(SMART_EVENT_WAYPOINT_RESUMED, nullptr, mCurrentWPID, GetScript()->GetPathId());
412  RemoveEscortState(SMART_ESCORT_PAUSED);
413  if (mForcedPaused)// if paused between 2 wps resend movement
414  {
415  mWPReached = false;
416  mForcedPaused = false;
417  ResumePath();
418  }
419 
420  mWPPauseTimer = 0;
421  }
422  }
423  else
424  mWPPauseTimer -= diff;
425  }
426 
427  if (HasEscortState(SMART_ESCORT_RETURNING))
428  {
429  if (mOOCReached)//reached OOC WP
430  {
431  mOOCReached = false;
432  RemoveEscortState(SMART_ESCORT_RETURNING);
433  if (!HasEscortState(SMART_ESCORT_PAUSED))
434  ResumePath();
435  }
436  }
437 
438  if ((me->GetVictim() && me->IsInCombat()) || HasEscortState(SMART_ESCORT_PAUSED | SMART_ESCORT_RETURNING))
439  return;
440 
441  // handle next wp
442  if (!me->HasUnitState(UNIT_STATE_NOT_MOVE) && me->movespline->Finalized())//reached WP
443  {
444  if (!mWPReached)
445  {
446  ResumePath();
447  return;
448  }
449 
450  mWPReached = false;
451  if (mCurrentWPID == GetWPCount())
452  EndPath();
453  else if (GetNextWayPoint())
454  {
455  SetRun(mRun);
456  // xinef: if we have reached waypoint, and there is no working spline movement it means our splitted array has ended, make new one
457  if (me->movespline->Finalized())
458  ResumePath();
459  }
460  }
