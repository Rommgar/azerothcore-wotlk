208  void UpdateAI(uint32 diff) override
209  {
210  if (!UpdateVictim())
211  return;
212 
213  events.Update(diff);
214 
215  if (me->HasUnitState(UNIT_STATE_CASTING))
216  return;
217 
218  while (uint32 eventId = events.ExecuteEvent())
219  {
220  switch (eventId)
221  {
222  case EVENT_SHIMMER:
223  {
224  // Remove old vulnerabilty spell
225  if (CurrentVurln_Spell)
226  me->RemoveAurasDueToSpell(CurrentVurln_Spell);
227 
228  // Cast new random vulnerabilty on self
229  uint32 spell = RAND(SPELL_FIRE_VULNERABILITY, SPELL_FROST_VULNERABILITY, SPELL_SHADOW_VULNERABILITY, SPELL_NATURE_VULNERABILITY, SPELL_ARCANE_VULNERABILITY);
230  DoCast(me, spell);
231  CurrentVurln_Spell = spell;
232  Talk(EMOTE_SHIMMER);
233  events.ScheduleEvent(EVENT_SHIMMER, 45000);
234  break;
235  }
236  case EVENT_BREATH_1:
237  DoCastVictim(Breath1_Spell);
238  events.ScheduleEvent(EVENT_BREATH_1, 60000);
239  break;
240  case EVENT_BREATH_2:
241  DoCastVictim(Breath2_Spell);
242  events.ScheduleEvent(EVENT_BREATH_2, 60000);
243  break;
244  case EVENT_AFFLICTION:
245  {
246  Map::PlayerList const& players = me->GetMap()->GetPlayers();
247  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
248  {
249  if (Player* player = itr->GetSource()->ToPlayer())
250  {
251  DoCast(player, RAND(SPELL_BROODAF_BLUE, SPELL_BROODAF_BLACK, SPELL_BROODAF_RED, SPELL_BROODAF_BRONZE, SPELL_BROODAF_GREEN), true);
252 
253  if (player->HasAura(SPELL_BROODAF_BLUE) &&
254  player->HasAura(SPELL_BROODAF_BLACK) &&
255  player->HasAura(SPELL_BROODAF_RED) &&
256  player->HasAura(SPELL_BROODAF_BRONZE) &&
257  player->HasAura(SPELL_BROODAF_GREEN))
258  {
259  DoCast(player, SPELL_CHROMATIC_MUT_1);
260  }
261  }
262  }
263  }
264  events.ScheduleEvent(EVENT_AFFLICTION, 10000);
265  break;
266  case EVENT_FRENZY:
267  DoCast(me, SPELL_FRENZY);
268  events.ScheduleEvent(EVENT_FRENZY, 10000, 15000);
269  break;
270  }
271 
272  if (me->HasUnitState(UNIT_STATE_CASTING))
273  return;
274  }
275 
276  // Enrage if not already enraged and below 20%
277  if (!Enraged && HealthBelowPct(20))
278  {
279  DoCast(me, SPELL_ENRAGE);
280  Enraged = true;
281  }
282 
283  DoMeleeAttackIfReady();
284  }
285 
286  private:
287  uint32 Breath1_Spell;
288  uint32 Breath2_Spell;
289  uint32 CurrentVurln_Spell;
290  bool Enraged;
291  };
