330  void UpdateAI(uint32 diff) override
331  {
332  // THIS IS USED ONLY FOR FIRST ENGAGE!
333  if (_startTimer)
334  {
335  _startTimer += diff;
336  if (_startTimer >= 4000)
337  {
338  // Colossus dead, players in range
339  if (me->FindNearestCreature(NPC_ULDUAR_COLOSSUS, 250.0f, true) || !SelectTargetFromPlayerList(250.0f))
340  _startTimer = 1;
341  else
342  {
343  _startTimer = 0;
344  _speakTimer = 1;
345  }
346  }
347  return;
348  }
349 
350  if (_speakTimer)
351  {
352  _speakTimer += diff;
353  if (_speakTimer <= 10000)
354  {
355  RadioSay("You've done it! You've broken the defenses of Ulduar. In a few moments, we will be dropping in to...", RSOUND_L1);
356  _speakTimer = 10000;
357  }
358  else if (_speakTimer > 16000 && _speakTimer < 20000)
359  {
360  _speakTimer = 20000;
361  RadioSay("What is that? Be careful! Something's headed your way!", RSOUND_L2);
362  }
363  else if (_speakTimer > 24000 && _speakTimer < 40000)
364  {
365  _speakTimer = 40000;
366  RadioSay("Quicly! Evasive action! Evasive act--", RSOUND_L3);
367  }
368  else if (_speakTimer > 41000 && _speakTimer < 60000)
369  {
370  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
371  me->SendMonsterMove(380.4f, -14.3f, 409.8f, 2000);
372  me->UpdatePosition(380.4f, -14.3f, 409.8f, me->GetOrientation());
373  _speakTimer = 60000;
374  }
375  else if (_speakTimer > 61500)
376  {
377  me->SetInCombatWithZone();
378  if (!me->GetVictim())
379  {
380  me->CastSpell(me, SPELL_PURSUED, false);
381  events.RescheduleEvent(EVENT_PURSUE, 31000);
382  }
383  _speakTimer = 0;
384  }
385  return;
386  }
387 
388  if (!UpdateVictim())
389  return;
390 
391  events.Update(diff);
392  if (me->HasUnitState(UNIT_STATE_CASTING))
393  return;
394 
395  switch (events.ExecuteEvent())
396  {
397  case EVENT_POSITION_CHECK:
398  if (me->GetPositionX() > 450 || me->GetPositionX() < 120)
399  {
400  EnterEvadeMode();
401  return;
402  }
403  events.RepeatEvent(5000);
404  break;
405  case EVENT_PURSUE:
406  Talk(FLAME_LEVIATHAN_SAY_PURSUE);
407  me->CastSpell(me, SPELL_PURSUED, false);
408  events.RescheduleEvent(EVENT_PURSUE, 31000);
409  return;
410  case EVENT_SPEED:
411  me->CastSpell(me, SPELL_GATHERING_SPEED, false);
412  events.RepeatEvent(15000);
413  return;
414  case EVENT_MISSILE:
415  me->CastSpell(me, SPELL_MISSILE_BARRAGE, true);
416  events.RepeatEvent(4000);
417  return;
418  case EVENT_VENT:
419  me->CastSpell(me, SPELL_FLAME_VENTS, false);
420  events.RepeatEvent(20000);
421  return;
422  case EVENT_SUMMON:
423  if(summons.size() < 20)
424  if (Creature* lift = DoSummonFlyer(NPC_MECHANOLIFT, me, 30.0f, 50.0f, 0))
425  lift->GetMotionMaster()->MoveRandom(100);
426 
427  events.RepeatEvent(4000);
428  return;
429  case EVENT_SOUND_BEGINNING:
430  if (_towersCount)
431  Talk(FLAME_LEVIATHAN_SAY_HARDMODE);
432  else
433  Talk(FLAME_LEVIATHAN_SAY_TOWER_NONE);
434  return;
435  case EVENT_REINSTALL:
436  for (uint8 i = RAID_MODE(0, 2); i < 4; ++i)
437  if (Unit* seat = vehicle->GetPassenger(i))
438  if (seat->GetTypeId() == TYPEID_UNIT)
439  seat->ToCreature()->AI()->EnterEvadeMode();
440  me->TextEmote("Flame Leviathan reactivated. Resumming combat functions.", nullptr, true);
441  return;
442  case EVENT_THORIMS_HAMMER:
443  SummonTowerHelpers(TOWER_OF_STORMS);
444  events.RepeatEvent(60000 + rand() % 60000);
445  me->TextEmote("Flame Leviathan activates Thorim's Hammer.", nullptr, true);
446  Talk(FLAME_LEVIATHAN_SAY_TOWER_STORM);
447  return;
448  case EVENT_FREYA:
449  SummonTowerHelpers(TOWER_OF_LIFE);
450  me->TextEmote("Flame Leviathan activates Freya's Ward.", nullptr, true);
451  Talk(FLAME_LEVIATHAN_SAY_TOWER_NATURE);
452  return;
453  case EVENT_MIMIRONS_INFERNO:
454  SummonTowerHelpers(TOWER_OF_FLAMES);
455  me->TextEmote("Flame Leviathan activates Mimiron's Inferno.", nullptr, true);
456  Talk(FLAME_LEVIATHAN_SAY_TOWER_FLAME);

458  case EVENT_HODIRS_FURY:
459  SummonTowerHelpers(TOWER_OF_FROST);
460  me->TextEmote("Flame Leviathan activates Hodir's Fury.", nullptr, true);
461  Talk(FLAME_LEVIATHAN_SAY_TOWER_FROST);

463  }
464 
465  if(me->isAttackReady() && !me->HasUnitState(UNIT_STATE_STUNNED))
466  {
467  if(me->IsWithinCombatRange(me->GetVictim(), 15.0f))
