138 // Loads a *_loot_template DB table into loot store
139 // All checks of the loaded template are called from here, no error reports at loot generation required
140 uint32 LootStore::LoadLootTable()
141 {
142  LootTemplateMap::const_iterator tab;
143 
144  // Clearing store (for reloading case)
145  Clear();
146 
147  // 0 1 2 3 4 5 6
148  QueryResult result = WorldDatabase.PQuery("SELECT Entry, Item, Reference, Chance, QuestRequired, LootMode, GroupId, MinCount, MaxCount FROM %s", GetName());
149 
150  if (!result)
151  return 0;
152 
153  uint32 count = 0;
154 
155  do
156  {
157  Field* fields = result->Fetch();
158 
159  uint32 entry = fields[0].GetUInt32();
160  uint32 item = fields[1].GetUInt32();
161  int32 reference = fields[2].GetInt32();
162  float chance = fields[3].GetFloat();
163  bool needsquest = fields[4].GetBool();
164  uint16 lootmode = fields[5].GetUInt16();
165  uint8 groupid = fields[6].GetUInt8();
166  int32 mincount = fields[7].GetUInt8();
167  int32 maxcount = fields[8].GetUInt8();
168 
169  if (maxcount > std::numeric_limits<uint8>::max())
170  {
171  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: MaxCount value (%u) to large. must be less %u - skipped", GetName(), entry, item, maxcount, std::numeric_limits<uint8>::max());
172  continue; // error already printed to log/console.
173  }
174 
175  if (lootmode == 0)
176  {
177  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: LootMode is equal to 0, item will never drop - setting mode 1", GetName(), entry, item);
178  lootmode = 1;
179  }
180 
181  LootStoreItem* storeitem = new LootStoreItem(item, reference, chance, needsquest, lootmode, groupid, mincount, maxcount);
182 
183  if (!storeitem->IsValid(*this, entry)) // Validity checks
184  {
185  delete storeitem;
186  continue;
187  }
188 
189  // Looking for the template of the entry
190  // often entries are put together
191  if (m_LootTemplates.empty() || tab->first != entry)
192  {
193  // Searching the template (in case template Id changed)
194  tab = m_LootTemplates.find(entry);
195  if (tab == m_LootTemplates.end())
196  {
197  std::pair< LootTemplateMap::iterator, bool > pr = m_LootTemplates.insert(LootTemplateMap::value_type(entry, new LootTemplate()));
198  tab = pr.first;
199  }
200  }
201  // else is empty - template Id and iter are the same
202  // finally iter refers to already existed or just created <entry, LootTemplate>
203 
204  // Adds current row to the template
205  tab->second->AddEntry(storeitem);
206  ++count;
207  } while (result->NextRow());
208 
209  Verify(); // Checks validity of the loot store
210 
211  return count;
212 }

302 bool LootStoreItem::Roll(bool rate, Player const* player, Loot& loot, LootStore const& store) const
303 {
304  float _chance = chance;
305 
306  if (!sScriptMgr->OnItemRoll(player, this, _chance, loot, store))
307  return false;
308 
309  if (_chance >= 100.0f)
310  return true;
311 
312  if (reference) // reference case
313  return roll_chance_f(_chance * (rate ? sWorld->getRate(RATE_DROP_ITEM_REFERENCED) : 1.0f));
314

316

322 // Checks correctness of values
323 bool LootStoreItem::IsValid(LootStore const& store, uint32 entry) const
324 {
325  if (groupid >= 1 << 7) // it stored in 7 bit field
326  {
327  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: GroupId (%u) must be less %u - skipped", store.GetName(), entry, itemid, groupid, 1 << 7);
328  return false;
329  }
330 
331  if (mincount == 0)
332  {
333  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: wrong MinCount (%d) - skipped", store.GetName(), entry, itemid, mincount);
334  return false;
335  }
336 
337  if (!reference) // item (quest or non-quest) entry, maybe grouped
338  {
339  ItemTemplate const* proto = sObjectMgr->GetItemTemplate(itemid);
340  if (!proto)
341  {
342  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: item entry not listed in `item_template` - skipped", store.GetName(), entry, itemid);
343  return false;
344  }
345 
346  if (chance == 0 && groupid == 0) // Zero chance is allowed for grouped entries only
347  {
348  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: equal-chanced grouped entry, but group not defined - skipped", store.GetName(), entry, itemid);
349  return false;
350  }
351 
352  if (chance != 0 && chance < 0.000001f) // loot with low chance
353  {
354  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: low chance (%f) - skipped",
355  store.GetName(), entry, itemid, chance);
356  return false;
357  }
358 
359  if (maxcount < mincount) // wrong max count
360  {
361  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: MaxCount (%u) less that MinCount (%i) - skipped", store.GetName(), entry, itemid, int32(maxcount), mincount);
362  return false;
363  }
364  }
365  else // if reference loot
366  {
367  if (needs_quest)
368  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: quest required will be ignored", store.GetName(), entry, itemid);
369  else if (chance == 0) // no chance for the reference
370  {
371  LOG_ERROR("sql.sql", "Table '%s' Entry %d Item %d: zero chance is specified for a reference, skipped", store.GetName(), entry, itemid);
372  return false;
373  }
374  }
375  return true; // Referenced template existence is checked at whole store level
376 }

405 // Basic checks for player/item compatibility - if false no chance to see the item in the loot
406 bool LootItem::AllowedForPlayer(Player const* player, bool isGivenByMasterLooter /*= false*/, bool allowQuestLoot /*= true*/) const
407 {
408  ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(itemid);
409  if (!pProto)
410  {
411  return false;
412  }
413 
414  bool isMasterLooter = player->GetGroup() && player->GetGroup()->GetMasterLooterGuid() == player->GetGUID();
415 
416  // DB conditions check
417  if (!sConditionMgr->IsObjectMeetToConditions(const_cast<Player*>(player), conditions))
418  {
419  // Master Looter can see conditioned recipes
420  if (!isGivenByMasterLooter && isMasterLooter)
421  {
422  if ((pProto->Flags & ITEM_FLAG_HIDE_UNUSABLE_RECIPE) || (pProto->Class == ITEM_CLASS_RECIPE && pProto->Bonding == BIND_WHEN_PICKED_UP && pProto->Spells[1].SpellId != 0))
423  {
424  return true;
425  }
426  }
427 
428  return false;
429  }
430 
431  // Checking for unique or unique(XX) objects
432  // master looter should still be able to see the loot to give to people.
433  // casting to avoid warnings, it's unlikely there's an item with unique but 2^31 items allowed.
434  if (!isMasterLooter && pProto->MaxCount > 0 && ((int32)player->GetItemCount(itemid, true) >= pProto->MaxCount))
435  {
436  return false;
437  }
438 
439  // not show loot for not own team
440  if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_HORDE_ONLY) && player->GetTeamId(true) != TEAM_HORDE)
441  {
442  return false;
443  }
444 
445  if ((pProto->Flags2 & ITEM_FLAGS_EXTRA_ALLIANCE_ONLY) && player->GetTeamId(true) != TEAM_ALLIANCE)
446  {
447  return false;
448  }
449 
450  // Master looter can see all items even if the character can't loot them
451  if (!isGivenByMasterLooter && isMasterLooter && allowQuestLoot)
452  {
453  return true;
454  }
455 
456  // Don't allow loot for players without profession or those who already know the recipe
457  if ((pProto->Flags & ITEM_FLAG_HIDE_UNUSABLE_RECIPE) && (!player->HasSkill(pProto->RequiredSkill) || player->HasSpell(pProto->Spells[1].SpellId)))
458  {
459  return false;
460  }
461 
462  // Don't allow to loot soulbound recipes that the player has already learned
463  if (pProto->Class == ITEM_CLASS_RECIPE && pProto->Bonding == BIND_WHEN_PICKED_UP && pProto->Spells[1].SpellId != 0 && player->HasSpell(pProto->Spells[1].SpellId))
464  {
465  return false;
466  }
467 
468  // check quest requirements
469  if (!(pProto->FlagsCu & ITEM_FLAGS_CU_IGNORE_QUEST_STATUS) && ((needs_quest || (pProto->StartQuest && player->GetQuestStatus(pProto->StartQuest) != QUEST_STATUS_NONE)) && !player->HasQuestForItem(itemid)))
470  {
471  return false;
472  }
473 
474  return true;
475 }

487 void Loot::AddItem(LootStoreItem const& item)
488 {
489  ItemTemplate const* proto = sObjectMgr->GetItemTemplate(item.itemid);
490  if (!proto)
491  return;
492 
493  uint32 count = urand(item.mincount, item.maxcount);
494  uint32 stacks = count / proto->GetMaxStackSize() + (count % proto->GetMaxStackSize() ? 1 : 0);
495 
496  std::vector<LootItem>& lootItems = item.needs_quest ? quest_items : items;
497  uint32 limit = item.needs_quest ? MAX_NR_QUEST_ITEMS : MAX_NR_LOOT_ITEMS;
498 
499  for (uint32 i = 0; i < stacks && lootItems.size() < limit; ++i)
500  {
501  LootItem generatedLoot(item);
502  generatedLoot.count = std::min(count, proto->GetMaxStackSize());
503  generatedLoot.itemIndex = lootItems.size();
504  lootItems.push_back(generatedLoot);
505  count -= proto->GetMaxStackSize();
506 
507  // In some cases, a dropped item should be visible/lootable only for some players in group
508  bool canSeeItemInLootWindow = false;
509  if (auto player = ObjectAccessor::FindPlayer(lootOwnerGUID))
510  {
511  if (auto group = player->GetGroup())
512  {
513  for (auto itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
514  {
515  if (auto member = itr->GetSource())
516  {
517  if (generatedLoot.AllowedForPlayer(member))
518  {
519  canSeeItemInLootWindow = true;
520  break;
521  }
522  }
523  }
524  }
525  else if (generatedLoot.AllowedForPlayer(player))
526  {
527  canSeeItemInLootWindow = true;
528  }
529  }
530 
531  if (!canSeeItemInLootWindow)
532  {
533  LOG_DEBUG("loot", "Skipping ++unlootedCount for unlootable item: %u", item.itemid);
534  continue;
535  }
536 
537  // non-conditional one-player only items are counted here,
538  // free for all items are counted in FillFFALoot(),
539  // non-ffa conditionals are counted in FillNonQuestNonFFAConditionalLoot()
540  if (!item.needs_quest && item.conditions.empty() && !(proto->Flags & ITEM_FLAG_MULTI_DROP))
541  ++unlootedCount;
542  }
