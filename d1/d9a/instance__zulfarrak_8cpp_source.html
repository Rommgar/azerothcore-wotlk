255  void Update(uint32 diff) override
256  {
257  switch (PyramidPhase)
258  {
259  case PYRAMID_NOT_STARTED:
260  case PYRAMID_KILLED_ALL_TROLLS:
261  case PYRAMID_MOVED_DOWNSTAIRS:
262  case PYRAMID_DESTROY_GATES:
263  case PYRAMID_GATES_DESTROYED:
264  case PYRAMID_DONE:
265  break;
266  case PYRAMID_ARRIVED_AT_STAIR:
267  SpawnPyramidWave(1);
268  SetData(DATA_PYRAMID, PYRAMID_WAVE_1);
269  major_wave_Timer = 2 * MINUTE * IN_MILLISECONDS;
270  minor_wave_Timer = 0;
271  addGroupSize = 2;
272  break;
273  case PYRAMID_WAVE_1:
274  if (IsWaveAllDead())
275  {
276  SetData(DATA_PYRAMID, PYRAMID_PRE_WAVE_2);
277  major_wave_Timer = 10 * IN_MILLISECONDS; //give players a few seconds before wave 2 starts to rebuff
278  }
279  else
280  {
281  if (minor_wave_Timer <= diff)
282  {
283  SendAddsUpStairs(addGroupSize++);
284  minor_wave_Timer = 10 * IN_MILLISECONDS;
285  }
286  else
287  {
288  minor_wave_Timer -= diff;
289  }
290  }
291  break;
292  case PYRAMID_PRE_WAVE_2:
293  if (major_wave_Timer <= diff)
294  {
295  // beginning 2nd wave!
296  SpawnPyramidWave(2);
297  SetData(DATA_PYRAMID, PYRAMID_WAVE_2);
298  minor_wave_Timer = 0;
299  addGroupSize = 2;
300  }
301  else
302  {
303  major_wave_Timer -= diff;
304  }
305  break;
306  case PYRAMID_WAVE_2:
307  if (IsWaveAllDead())
308  {
309  SpawnPyramidWave(3);
310  SetData(DATA_PYRAMID, PYRAMID_PRE_WAVE_3);
311  major_wave_Timer = 5 * IN_MILLISECONDS; //give NPCs time to return to their home spots
312  }
313  else
314  {
315  if (minor_wave_Timer <= diff)
316  {
317  SendAddsUpStairs(addGroupSize++);
318  minor_wave_Timer = 10 * IN_MILLISECONDS;
319  }
320  else
321  {
322  minor_wave_Timer -= diff;
323  }
324  }
325  break;
326  case PYRAMID_PRE_WAVE_3:
327  if (major_wave_Timer <= diff)
328  {
329  // move NPCs to bottom of stair
330  MoveNPCIfAlive(NPC_BLY, 1887.92f, 1228.179f, 9.98f, 4.78f);
331  MoveNPCIfAlive(NPC_MURTA, 1891.57f, 1228.68f, 9.69f, 4.78f);
332  MoveNPCIfAlive(NPC_ORO, 1897.23f, 1228.34f, 9.43f, 4.78f);
333  MoveNPCIfAlive(NPC_RAVEN, 1883.68f, 1227.95f, 9.543f, 4.78f);
334  MoveNPCIfAlive(NPC_WEEGLI, 1878.02f, 1227.65f, 9.485f, 4.78f);
335  SetData(DATA_PYRAMID, PYRAMID_WAVE_3);
336  if (Creature* sergeantBlye = instance->GetCreature(BlyGUID))
337  {
338  sergeantBlye->AI()->Talk(SAY_BLY_FORWARD);
339  }
340  }

342  {
343  major_wave_Timer -= diff;
344  }

375  void SpawnPyramidWave(uint32 wave)
376  {
377  for (uint32 i = 0; i < pyramidSpawnTotal; i++)
378  {
379  if (pyramidSpawns[i].waveID == wave)
380  {
381  float orientation = 4.78f;
382  if (pyramidSpawns[i].creatureID == NPC_SHADOWPRIEST_SEZZZIZ || pyramidSpawns[i].creatureID == NPC_NEKRUM_GUTCHEWER)
383  {
384  orientation = 1.32f;
385  }
386 
387  Position pos = { pyramidSpawns[i].pos.GetPositionX(), pyramidSpawns[i].pos.GetPositionY(), 8.87f, orientation };
388  if (TempSummon* ts = instance->SummonCreature(pyramidSpawns[i].creatureID, pos))
389  {
390  addsAtBase.push_back(ts->GetGUID());
391 
392  if (pyramidSpawns[i].creatureID != NPC_SHADOWPRIEST_SEZZZIZ && pyramidSpawns[i].creatureID != NPC_NEKRUM_GUTCHEWER)

394  ts->GetMotionMaster()->MoveRandom(10);
