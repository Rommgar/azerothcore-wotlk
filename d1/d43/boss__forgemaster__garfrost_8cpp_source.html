210  void UpdateAI(uint32 diff) override
211  {

213  return;

218  me->GetVictim()->GetPosition(x, y, z);

220  {
221  me->SetHealth(me->GetMaxHealth());
222  EnterEvadeMode();
223  if (CreatureGroup* f = me->GetFormation())
224  {
225  const CreatureGroup::CreatureGroupMemberType& m = f->GetMembers();
226  for (CreatureGroup::CreatureGroupMemberType::const_iterator itr = m.begin(); itr != m.end(); ++itr)
227  if (itr->first->IsAlive() && itr->first->IsInCombat() && !itr->first->IsInEvadeMode() && itr->first->IsAIEnabled)
228  itr->first->AI()->EnterEvadeMode();
229  }
230  return;
231  }
232  }
233 
234  events.Update(diff);
235 
236  if (me->HasUnitState(UNIT_STATE_CASTING))
237  return;
238 
239  switch(events.ExecuteEvent())
240  {
241  case 0:
242  break;
243  case EVENT_SPELL_THROW_SARONITE:
244  bCanSayBoulderHit = true;
245  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 140.0f, true))

250  me->CastSpell(target, SPELL_THROW_SARONITE, false);
