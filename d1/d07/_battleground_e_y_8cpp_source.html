57 void BattlegroundEY::PostUpdateImpl(uint32 diff)
58 {
59  if (GetStatus() == STATUS_IN_PROGRESS)
60  {
61  _bgEvents.Update(diff);
62  while (uint32 eventId = _bgEvents.ExecuteEvent())
63  switch (eventId)
64  {
65  case BG_EY_EVENT_ADD_POINTS:
66  if (_ownedPointsCount[TEAM_ALLIANCE] > 0)
67  AddPoints(TEAM_ALLIANCE, BG_EY_TickPoints[_ownedPointsCount[TEAM_ALLIANCE] - 1]);
68  if (_ownedPointsCount[TEAM_HORDE] > 0)
69  AddPoints(TEAM_HORDE, BG_EY_TickPoints[_ownedPointsCount[TEAM_HORDE] - 1]);
70  _bgEvents.ScheduleEvent(BG_EY_EVENT_ADD_POINTS, BG_EY_FPOINTS_TICK_TIME - (World::GetGameTimeMS() % BG_EY_FPOINTS_TICK_TIME));
71  break;
72  case BG_EY_EVENT_FLAG_ON_GROUND:
73  RespawnFlagAfterDrop();
74  break;
75  case BG_EY_EVENT_RESPAWN_FLAG:
76  RespawnFlag();
77  break;
78  case BG_EY_EVENT_CHECK_CPOINTS:
79  UpdatePointsState();
80  _bgEvents.ScheduleEvent(BG_EY_EVENT_CHECK_CPOINTS, BG_EY_FPOINTS_CHECK_TIME - (World::GetGameTimeMS() % BG_EY_FPOINTS_CHECK_TIME));
81  break;
82  }
83  }
84 }

112 void BattlegroundEY::AddPoints(TeamId teamId, uint32 points)
113 {
114  uint8 honorRewards = uint8(m_TeamScores[teamId] / _honorTics);
115  m_TeamScores[teamId] += points;
116 
117  for (; honorRewards < uint8(m_TeamScores[teamId] / _honorTics); ++honorRewards)
118  RewardHonorToTeam(GetBonusHonorFromKill(1), teamId);
119 
120  UpdateWorldState(teamId == TEAM_ALLIANCE ? EY_ALLIANCE_RESOURCES : EY_HORDE_RESOURCES, std::min<uint32>(m_TeamScores[teamId], BG_EY_MAX_TEAM_SCORE));
121  if (m_TeamScores[teamId] >= BG_EY_MAX_TEAM_SCORE)
122  EndBattleground(teamId);

125 void BattlegroundEY::UpdatePointsState()
126 {
127  std::vector<GameObject*> pointsVec;
128  for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
129  {
130  pointsVec.push_back(GetBGObject(BG_EY_OBJECT_TOWER_CAP_FEL_REAVER + point));
131  _capturePointInfo[point]._playersCount[TEAM_ALLIANCE] = 0;
132  _capturePointInfo[point]._playersCount[TEAM_HORDE] = 0;
133  }
134 
135  const BattlegroundPlayerMap& bgPlayerMap = GetPlayers();
136  for (BattlegroundPlayerMap::const_iterator itr = bgPlayerMap.begin(); itr != bgPlayerMap.end(); ++itr)
137  {
138  UpdateWorldStateForPlayer(PROGRESS_BAR_SHOW, BG_EY_PROGRESS_BAR_DONT_SHOW, itr->second);
139  for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
140  if (GameObject* pointObject = pointsVec[point])
141  if (itr->second->CanCaptureTowerPoint() && itr->second->IsWithinDistInMap(pointObject, BG_EY_POINT_RADIUS))
142  {
143  UpdateWorldStateForPlayer(PROGRESS_BAR_SHOW, BG_EY_PROGRESS_BAR_SHOW, itr->second);
144  UpdateWorldStateForPlayer(PROGRESS_BAR_PERCENT_GREY, BG_EY_PROGRESS_BAR_PERCENT_GREY, itr->second);
145  UpdateWorldStateForPlayer(PROGRESS_BAR_STATUS, _capturePointInfo[point]._barStatus, itr->second);
146  ++_capturePointInfo[point]._playersCount[itr->second->GetTeamId()];
147 
148  // Xinef: ugly hax... area trigger is no longer called by client...
149  if (pointObject->GetEntry() == BG_OBJECT_FR_TOWER_CAP_EY_ENTRY && itr->second->GetDistance2d(2043.96f, 1729.68f) < 3.0f)
150  HandleAreaTrigger(itr->second, AT_FEL_REAVER_POINT);
151  }
152  }
153 
154  for (uint8 point = 0; point < EY_POINTS_MAX; ++point)
155  {
156  _capturePointInfo[point]._barStatus += std::max<int8>(std::min<int8>(_capturePointInfo[point]._playersCount[TEAM_ALLIANCE] - _capturePointInfo[point]._playersCount[TEAM_HORDE], BG_EY_POINT_MAX_CAPTURERS_COUNT), -BG_EY_POINT_MAX_CAPTURERS_COUNT);
157  _capturePointInfo[point]._barStatus = std::max<int8>(std::min<int8>(_capturePointInfo[point]._barStatus, BG_EY_PROGRESS_BAR_ALI_CONTROLLED), BG_EY_PROGRESS_BAR_HORDE_CONTROLLED);
158 
159  TeamId pointOwnerTeamId = TEAM_NEUTRAL;
160  if (_capturePointInfo[point]._barStatus <= BG_EY_PROGRESS_BAR_NEUTRAL_LOW)
161  pointOwnerTeamId = TEAM_HORDE;
162  else if (_capturePointInfo[point]._barStatus >= BG_EY_PROGRESS_BAR_NEUTRAL_HIGH)
163  pointOwnerTeamId = TEAM_ALLIANCE;
164 
165  if (pointOwnerTeamId != _capturePointInfo[point]._ownerTeamId)
166  {
167  if (_capturePointInfo[point].IsUncontrolled())
168  EventTeamCapturedPoint(pointOwnerTeamId, point);
169 
170  if (pointOwnerTeamId == TEAM_NEUTRAL && _capturePointInfo[point].IsUnderControl())
171  EventTeamLostPoint(pointOwnerTeamId, point);
172  }
173  }
174 }

218 void BattlegroundEY::HandleAreaTrigger(Player* player, uint32 trigger)
219 {
220  if (GetStatus() != STATUS_IN_PROGRESS || !player->IsAlive())
221  return;
222 
223  switch (trigger)
224  {
225  case AT_BLOOD_ELF_POINT:
226  if (_capturePointInfo[POINT_BLOOD_ELF].IsUnderControl(player->GetTeamId()))
227  if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
228  EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_BLOOD_ELF);
229  break;
230  case AT_FEL_REAVER_POINT:
231  if (_capturePointInfo[POINT_FEL_REAVER].IsUnderControl(player->GetTeamId()))
232  if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
233  EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_FEL_REAVER);
234  break;
235  case AT_MAGE_TOWER_POINT:
236  if (_capturePointInfo[POINT_MAGE_TOWER].IsUnderControl(player->GetTeamId()))
237  if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
238  EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_MAGE_TOWER);
239  break;
240  case AT_DRAENEI_RUINS_POINT:
241  if (_capturePointInfo[POINT_DRAENEI_RUINS].IsUnderControl(player->GetTeamId()))
242  if (_flagState == BG_EY_FLAG_STATE_ON_PLAYER && GetFlagPickerGUID() == player->GetGUID())
243  EventPlayerCapturedFlag(player, BG_EY_OBJECT_FLAG_DRAENEI_RUINS);
244  break;
245  case 4512:
246  case 4515:
247  case 4517:
248  case 4519:
249  case 4530:
250  case 4531:
251  case 5866:
252  case AT_BLOOD_ELF_BUFF:
253  case AT_FEL_REAVER_BUFF:
254  case AT_MAGE_TOWER_BUFF:
255  case AT_DRAENEI_RUINS_BUFF:
256  break;
257  }
258 }
