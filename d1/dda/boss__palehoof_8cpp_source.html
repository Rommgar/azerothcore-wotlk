120  struct boss_palehoofAI : public ScriptedAI
121  {
122  boss_palehoofAI(Creature* pCreature) : ScriptedAI(pCreature), summons(me)
123  {
124  m_pInstance = pCreature->GetInstanceScript();
125  }

127  InstanceScript* m_pInstance;
128  EventMap events;
129  SummonList summons;
130  ObjectGuid OrbGUID;
131  uint8 Counter;
132  uint8 RandomUnfreeze[4];
133 
134  void Reset() override
135  {
136  for (uint8 i = 0; i < 4; ++i)
137  {
138  bool good;
139  do
140  {
141  good = true;
142  RandomUnfreeze[i] = urand(0, 3);
143 
144  for (uint8 j = 0; j < i; ++j)
145  if (RandomUnfreeze[i] == RandomUnfreeze[j])
146  {
147  good = false;
148  break;
149  }
150  } while (!good);
151  }
152 
153  events.Reset();
154  summons.DoAction(ACTION_DESPAWN_ADDS);
155  summons.DespawnAll();
156  OrbGUID.Clear();
157  Counter = 0;
158  me->CastSpell(me, SPELL_FREEZE, true);
159  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
160  me->SetControlled(false, UNIT_STATE_STUNNED);
161 
162  if (m_pInstance)
163  {
164  m_pInstance->SetData(DATA_GORTOK_PALEHOOF, NOT_STARTED);
165 
166  // Reset statue
167  if (GameObject* statisGenerator = m_pInstance->instance->GetGameObject(m_pInstance->GetGuidData(STATIS_GENERATOR)))
168  {
169  statisGenerator->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
170  statisGenerator->SetGoState(GO_STATE_READY);
171  }
172 
173  // Reset mini bosses
174  for(uint8 i = 0; i < 4; ++i)
175  {
176  if(Creature* Animal = ObjectAccessor::GetCreature(*me, m_pInstance->GetGuidData(DATA_NPC_FRENZIED_WORGEN + i)))
177  {
178  Animal->SetPosition(Animal->GetHomePosition());
179  Animal->StopMovingOnCurrentPos();
180  if(Animal->isDead())
181  Animal->Respawn(true);
182 
183  Animal->CastSpell(Animal, SPELL_FREEZE, true);
184  }
185  }
186  }
187  }
188 
189  void DoAction(int32 param) override
190  {
191  if (param == ACTION_START_EVENT)
192  {
193  if (Creature* cr = me->SummonCreature(NPC_ORB_TRIGGER, 238.608f, -460.71f, 109.567f))
194  {
195  OrbGUID = cr->GetGUID();
196  cr->AddAura(SPELL_ORB_VISUAL, cr);
197  summons.Summon(cr);
198  cr->SetDisableGravity(true);
199  cr->GetMotionMaster()->MovePoint(0, 275.4f, -453, 110); // ROOM CENTER
200  events.RescheduleEvent(EVENT_UNFREEZE_MONSTER, 10000);
201  me->SetInCombatWithZone();
202  me->SetControlled(true, UNIT_STATE_STUNNED);
203  }
204  }
205  else if (param == ACTION_MINIBOSS_DIED)
206  {
207  if (Counter > (IsHeroic() ? 3 : 1))
208  events.RescheduleEvent(EVENT_PALEHOOF_START, 3000);
209  else
210  events.RescheduleEvent(EVENT_UNFREEZE_MONSTER, 3000);
211  }
212  }

232  void UpdateAI(uint32 diff) override
233  {
234  if (!UpdateVictim())
235  return;
236 
237  events.Update(diff);
238  switch (events.ExecuteEvent())
239  {
240  case EVENT_UNFREEZE_MONSTER:
241  {
242  if (Creature* orb = ObjectAccessor::GetCreature(*me, OrbGUID))
243  {
244  if (Creature* miniBoss = ObjectAccessor::GetCreature(*me, m_pInstance->GetGuidData(DATA_NPC_FRENZIED_WORGEN + RandomUnfreeze[Counter])))
245  {
246  Counter++;
247  miniBoss->AI()->DoAction(ACTION_UNFREEZE);
248  orb->CastSpell(miniBoss, SPELL_AWAKEN_SUBBOSS, true);
249  events.ScheduleEvent(EVENT_UNFREEZE_MONSTER2, 6000);
250  }
251  else
252  EnterEvadeMode();
253  }
254  break;
255  }
256  case EVENT_UNFREEZE_MONSTER2:
257  {
258  if (Creature* orb = ObjectAccessor::GetCreature(*me, OrbGUID))
259  {
260  if (Creature* miniBoss = ObjectAccessor::GetCreature(*me, m_pInstance->GetGuidData(DATA_NPC_FRENZIED_WORGEN + RandomUnfreeze[Counter - 1])))
261  {
262  miniBoss->AI()->DoAction(ACTION_UNFREEZE2);
263  orb->RemoveAurasDueToSpell(SPELL_AWAKEN_SUBBOSS);
264  }
265  else
266  EnterEvadeMode();
267  }
268  break;
269  }
270  case EVENT_PALEHOOF_START:
271  {
272  if (Creature* orb = ObjectAccessor::GetCreature(*me, OrbGUID))
273  {
274  orb->CastSpell(me, SPELL_AWAKEN_SUBBOSS, true);
275  events.ScheduleEvent(EVENT_PALEHOOF_START2, 6000);
276  }
277  break;
278  }
279  case EVENT_PALEHOOF_START2:
280  {
281  Talk(SAY_AGGRO);
282  if (Creature* orb = ObjectAccessor::GetCreature(*me, OrbGUID))
283  orb->RemoveAurasDueToSpell(SPELL_AWAKEN_SUBBOSS);
284 
285  me->RemoveAurasDueToSpell(SPELL_FREEZE);
286  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
287  me->SetControlled(false, UNIT_STATE_STUNNED);
288  // SETINCOMBATWITHZONE
289 
290  // schedule combat events
291  events.ScheduleEvent(EVENT_PALEHOOF_WITHERING_ROAR, 10000);
292  events.ScheduleEvent(EVENT_PALEHOOF_IMPALE, 12000);
293  events.ScheduleEvent(EVENT_PALEHOOF_ARCING_SMASH, 15000);
294  break;
295  }
296  case EVENT_PALEHOOF_WITHERING_ROAR:
297  {
298  me->CastSpell(me, IsHeroic() ? SPELL_WITHERING_ROAR_H : SPELL_WITHERING_ROAR_N, false);
299  events.RepeatEvent(8000 + rand() % 4000);
300  break;
301  }
302  case EVENT_PALEHOOF_IMPALE:
303  {
304  if (Unit* tgt = SelectTarget(SELECT_TARGET_RANDOM, 0))
305  me->CastSpell(tgt, IsHeroic() ? SPELL_IMPALE_H : SPELL_IMPALE_N, false);
306 
307  events.RepeatEvent(8000 + rand() % 4000);
308  break;
309  }
310  case EVENT_PALEHOOF_ARCING_SMASH:
311  {
312  me->CastSpell(me->GetVictim(), SPELL_ARCING_SMASH, false);
313  events.RepeatEvent(13000 + rand() % 4000);
314  break;
315  }
316  }
317 
318  DoMeleeAttackIfReady();
319  }
320 
321  void JustDied(Unit* /*pKiller*/) override
322  {
323  me->PlayDirectSound(SOUND_DEATH);
324  if(m_pInstance)
325  m_pInstance->SetData(DATA_GORTOK_PALEHOOF, DONE);
326  }
327 
328  void KilledUnit(Unit* victim) override
329  {
330  if (victim->GetTypeId() != TYPEID_PLAYER)
331  return;
332 
333  Talk(SAY_SLAY);
334  }
335  };
