190  void UpdateAI(uint32 diff) override
191  {
192  if (!UpdateVictim() || !CheckInRoom())
193  return;
194 
195  events.Update(diff);
196 
197  if (me->HasUnitState(UNIT_STATE_CASTING))
198  return;
199 
200  switch (events.ExecuteEvent())
201  {
202  case EVENT_BERSERK:
203  me->CastSpell(me, SPELL_BERSERK2, true);
204  Talk(SAY_BERSERK);
205  break;
206  case EVENT_INHALE_BLIGHT:
207  RemoveBlight();
208  if (_inhaleCounter == 3)
209  {
210  Talk(EMOTE_WARN_PUNGENT_BLIGHT);
211  Talk(SAY_PUNGENT_BLIGHT);
212  me->CastSpell(me, SPELL_PUNGENT_BLIGHT, false);
213  _inhaleCounter = 0;
214  events.RescheduleEvent(EVENT_GAS_SPORE, urand(20000, 25000));
215  }
216  else
217  {
218  me->CastSpell(me, SPELL_INHALE_BLIGHT, false);
219  // just cast and dont bother with target, conditions will handle it
220  ++_inhaleCounter;
221  if (_inhaleCounter < 3)
222  me->CastSpell(me, gaseousBlight[_inhaleCounter], true, nullptr, nullptr, me->GetGUID());
223  }
224 
225  events.ScheduleEvent(EVENT_INHALE_BLIGHT, 34000);
226  break;
227  case EVENT_GAS_SPORE:
228  Talk(EMOTE_WARN_GAS_SPORE);
229  Talk(EMOTE_GAS_SPORE);
230  me->CastCustomSpell(SPELL_GAS_SPORE, SPELLVALUE_MAX_TARGETS, RAID_MODE<int32>(2, 3, 2, 3), me);
231  events.ScheduleEvent(EVENT_GAS_SPORE, urand(40000, 45000));
232  events.DelayEventsToMax(20000, 1); // delay EVENT_VILE_GAS
233  break;
234  case EVENT_VILE_GAS:
235  {
236  std::list<Unit*> targets;
237  uint32 minTargets = RAID_MODE<uint32>(3, 8, 3, 8);
238  SelectTargetList(targets, minTargets, SELECT_TARGET_RANDOM, -5.0f, true);
239  float minDist = 0.0f;
240  if (targets.size() >= minTargets)
241  minDist = -5.0f;
242 
243  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, minDist, true))
244  me->CastSpell(target, SPELL_VILE_GAS, false);
245  events.ScheduleEvent(EVENT_VILE_GAS, urand(28000, 35000), 1);
246  break;
247  }
248  case EVENT_GASTRIC_BLOAT:
249  me->CastSpell(me->GetVictim(), SPELL_GASTRIC_BLOAT, false);
250  events.ScheduleEvent(EVENT_GASTRIC_BLOAT, urand(15000, 17500));
251  break;
252  case EVENT_FESTERGUT_GOO:
253  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))
254  if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
255  professor->CastSpell(target, SPELL_MALLABLE_GOO_H, true);
256  events.ScheduleEvent(EVENT_FESTERGUT_GOO, urand(15000, 20000));
257  default:
258  break;
259  }
260 
261  DoMeleeAttackIfReady();
262  }
