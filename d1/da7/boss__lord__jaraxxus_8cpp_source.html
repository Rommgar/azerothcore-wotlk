148  void SpellHit(Unit* caster, const SpellInfo* spell) override
149  {
150  switch( spell->Id )
151  {
152  case 66228:
153  case 67106:
154  case 67107:
155  case 67108:
156  if( Aura* a = me->GetAura(spell->Id) )
157  a->SetStackAmount(spell->StackAmount);
158  break;
159  case 30449:
160  {
161  if( !caster )
162  return;
163  uint32 id = 0;
164  switch( me->GetMap()->GetDifficulty() )

168  break;
169  case 1:

194  void UpdateAI(uint32 diff) override
195  {
196  if( !UpdateVictim() )
197  return;
198 
199  events.Update(diff);
200 
201  if( me->HasUnitState(UNIT_STATE_CASTING) )
202  return;
203 
204  switch( events.ExecuteEvent() )
205  {
206  case 0:
207  break;
208  case EVENT_SPELL_FEL_FIREBALL:
209  if( me->GetVictim() )
210  me->CastSpell(me->GetVictim(), SPELL_FEL_FIREBALL, false);
211  events.RepeatEvent(urand(10000, 15000));
212  break;
213  case EVENT_SPELL_FEL_LIGHTNING:
214  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true) )
215  me->CastSpell(target, SPELL_FEL_LIGHTNING, false);
216  events.RepeatEvent(urand(10000, 15000));
217  break;
218  case EVENT_SPELL_INCINERATE_FLESH:
219  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true) )
220  {
221  Talk(EMOTE_INCINERATE, target);
222  Talk(SAY_INCINERATE);
223  me->CastSpell(target, SPELL_INCINERATE_FLESH, false);
224  }
225  events.RepeatEvent(urand(20000, 25000));
226  break;
227  case EVENT_SPELL_NETHER_POWER:
228  me->CastSpell(me, SPELL_NETHER_POWER, false);
229  events.DelayEvents(5000);
230  events.RepeatEvent(urand(25000, 45000));
231  break;
232  case EVENT_SPELL_LEGION_FLAME:
233  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true) )
234  {
235  Talk(EMOTE_LEGION_FLAME, target);
236  me->CastSpell(target, SPELL_LEGION_FLAME, false);
237  }
238  events.RepeatEvent(30000);
239  break;
240  case EVENT_SPELL_TOUCH_OF_JARAXXUS:
241  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true) )
242  me->CastSpell(target, SPELL_TOUCH_OF_JARAXXUS, false);

244  break;
245  case EVENT_SUMMON_NETHER_PORTAL:
246  Talk(EMOTE_NETHER_PORTAL);
