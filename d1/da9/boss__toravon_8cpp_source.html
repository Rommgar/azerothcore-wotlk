70  void Reset() override
71  {
72  events.Reset();
73  summons.DespawnAll();
74  if (pInstance)
75  {
76  if (pInstance->GetData(DATA_STONED))

95  void EnterCombat(Unit* /*who*/) override
96  {
97  me->CastSpell(me, SPELL_FROZEN_MALLET, true);
98 
99  events.ScheduleEvent(EVENT_FROZEN_ORB_STALKER, 12000);
100  events.ScheduleEvent(EVENT_FREEZING_GROUND, 7000);
101  events.ScheduleEvent(EVENT_CAST_WHITEOUT, 25000); // schedule FIRST whiteout event in 25 seconds -1 for compesate updateai 2seconds check delay
102 
103  if (pInstance)
104  pInstance->SetData(EVENT_TORAVON, IN_PROGRESS);
105  }

107  void JustDied(Unit*) override
108  {
109  if (pInstance)
110  {
111  pInstance->SetData(EVENT_TORAVON, DONE);

122  void UpdateAI(uint32 diff) override
123  {
124  if (!UpdateVictim())
125  return;
126 
127  events.Update(diff);
128  if (me->HasUnitState(UNIT_STATE_CASTING))
129  return;
130 
131  switch (events.ExecuteEvent())
132  {
133  case EVENT_FREEZING_GROUND:
134  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
135  me->CastSpell(target, SPELL_FREEZING_GROUND, false);
136  events.RepeatEvent(20000);
137  break;
138  case EVENT_FROZEN_ORB_STALKER:
139  me->CastCustomSpell(SPELL_FROZEN_ORB, SPELLVALUE_MAX_TARGETS, RAID_MODE(1, 3), me, false);
140  events.RepeatEvent(30000);
141  break;
142  case EVENT_CAST_WHITEOUT:
143  me->CastSpell(me, SPELL_WHITEOUT, false);
144  events.ScheduleEvent(EVENT_CAST_WHITEOUT_GROUND_EFFECT, 1000); // triggers after 1 sec "plus 1 from trigger to cast visual"
145  events.RepeatEvent(40000); // next whiteout instead first 25 SEC is now 45 SEC
146  break;
147  case EVENT_CAST_WHITEOUT_GROUND_EFFECT: // Whiteout Ground effect trigger
148  if (Unit* whiteOutGround = me->SummonCreature(NPC_WHITEOUT_GROUND_EFFECT, -43.3316, -288.708, 92.2511, 1.58825, TEMPSUMMON_TIMED_DESPAWN, 4000))
149  whiteOutGround->CastSpell(whiteOutGround, SPELL_WHITEOUT_VISUAL, false); // Cast the spell
150  break;
151  }
152 
153  DoMeleeAttackIfReady();
154  }
155  };
