176  void LapseAction(uint8 action)
177  {
178  uint8 counter = 0;
179  Map::PlayerList const& playerList = me->GetMap()->GetPlayers();
180  for (Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr, ++counter)
181  if (Player* player = itr->GetSource())
182  {
183  if (action == ACTION_TELEPORT_PLAYERS)
184  me->CastSpell(player, SPELL_GRAVITY_LAPSE_PLAYER + counter, true);
185  else if (action == ACTION_KNOCKUP)
186  player->CastSpell(player, SPELL_GRAVITY_LAPSE_DOT, true, nullptr, nullptr, me->GetGUID());
187  else if (action == ACTION_ALLOW_FLY)
188  player->CastSpell(player, SPELL_GRAVITY_LAPSE_FLY, true, nullptr, nullptr, me->GetGUID());
189  else if (action == ACTION_REMOVE_FLY)
190  {
191  player->RemoveAurasDueToSpell(SPELL_GRAVITY_LAPSE_FLY);
192  player->RemoveAurasDueToSpell(SPELL_GRAVITY_LAPSE_DOT);
193  }
194  }
195  }

197  void UpdateAI(uint32 diff) override
198  {
199  events2.Update(diff);
200  switch (events2.ExecuteEvent())
201  {
202  case EVENT_INIT_COMBAT:
203  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
204  if (Unit* target = SelectTargetFromPlayerList(50.0f))
205  AttackStart(target);
206  return;
207  case EVENT_FINISH_TALK:
208  Unit::Kill(me, me);
209  return;
210  }
211 
212  if (!UpdateVictim())
213  return;
214 
215  events.Update(diff);
216  if (me->HasUnitState(UNIT_STATE_CASTING))
217  return;
218 
219  switch (uint32 eventId = events.ExecuteEvent())
220  {
221  case EVENT_SPELL_FIREBALL:
222  me->CastSpell(me->GetVictim(), DUNGEON_MODE(SPELL_FIREBALL_N, SPELL_FIREBALL_H), false);
223  events.ScheduleEvent(EVENT_SPELL_FIREBALL, urand(3000, 4500));
224  break;
225  case EVENT_SPELL_FLAMESTRIKE:
226  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 100, true))
227  {
228  me->CastSpell(target, SPELL_FLAMESTRIKE_SUMMON, true);
229  Talk(SAY_FLAMESTRIKE);
230  }
231  events.ScheduleEvent(EVENT_SPELL_FLAMESTRIKE, 25000);
232  break;
233  case EVENT_SPELL_SHOCK_BARRIER:
234  me->CastSpell(me, SPELL_SHOCK_BARRIER, true);
235  me->CastCustomSpell(SPELL_PYROBLAST, SPELLVALUE_MAX_TARGETS, 1, (Unit*)nullptr, false);
236  events.ScheduleEvent(EVENT_SPELL_SHOCK_BARRIER, 50000);
237  break;
238  case EVENT_SPELL_PHOENIX:
239  Talk(SAY_PHOENIX);
240  me->CastSpell(me, SPELL_PHOENIX, false);
241  events.ScheduleEvent(EVENT_SPELL_PHOENIX, 60000);
242  break;
243  case EVENT_CHECK_HEALTH:
244  if (HealthBelowPct(50))
245  {
246  me->ApplySpellImmune(0, IMMUNITY_EFFECT, SPELL_EFFECT_INTERRUPT_CAST, true);
247  me->CastSpell(me, SPELL_TELEPORT_CENTER, true);
248  events.Reset();
249 
250  me->StopMoving();
251  me->GetMotionMaster()->Clear();
252  me->GetMotionMaster()->MoveIdle();
253 
254  events.SetPhase(1);
255  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_1_1, 0);
256  break;
257  }
258  events.ScheduleEvent(EVENT_CHECK_HEALTH, 500);
259  break;
260  case EVENT_GRAVITY_LAPSE_1_1:
261  case EVENT_GRAVITY_LAPSE_1_2:
262  Talk(eventId == EVENT_GRAVITY_LAPSE_1_1 ? SAY_GRAVITY_LAPSE : SAY_RECAST_GRAVITY);
263  me->CastSpell(me, SPELL_GRAVITY_LAPSE_INITIAL, false);
264  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_2, 2000);
265  break;
266  case EVENT_GRAVITY_LAPSE_2:
267  LapseAction(ACTION_TELEPORT_PLAYERS);
268  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_3, 1000);
269  break;
270  case EVENT_GRAVITY_LAPSE_3:
271  LapseAction(ACTION_KNOCKUP);
272  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_4, 1000);
273  break;
274  case EVENT_GRAVITY_LAPSE_4:
275  LapseAction(ACTION_ALLOW_FLY);
276  for (uint8 i = 0; i < 3; ++i)
277  me->CastSpell(me, SPELL_SUMMON_ARCANE_SPHERE, true);
278 
279  me->CastSpell(me, SPELL_GRAVITY_LAPSE_CHANNEL, false);
280  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_5, 30000);
281  break;
282  case EVENT_GRAVITY_LAPSE_5:
283  LapseAction(ACTION_REMOVE_FLY);
284  me->InterruptNonMeleeSpells(false);
285  Talk(SAY_TIRED);
286  me->CastSpell(me, SPELL_POWER_FEEDBACK, false);
287  events.ScheduleEvent(EVENT_GRAVITY_LAPSE_1_2, 10000);
288  break;
289  }
290 
291  if (events.GetPhaseMask() == 0)
292  DoMeleeAttackIfReady();
293  }
