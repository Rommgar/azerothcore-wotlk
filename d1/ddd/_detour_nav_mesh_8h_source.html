541  inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
542  {
543 #ifdef DT_POLYREF64
544  return ((dtPolyRef)salt << (DT_POLY_BITS+DT_TILE_BITS)) | ((dtPolyRef)it << DT_POLY_BITS) | (dtPolyRef)ip;
545 #else
546  return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
547 #endif
548  }
549 
557  inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
558  {
559 #ifdef DT_POLYREF64
560  const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
561  const dtPolyRef tileMask = ((dtPolyRef)1<<DT_TILE_BITS)-1;
562  const dtPolyRef polyMask = ((dtPolyRef)1<<DT_POLY_BITS)-1;
563  salt = (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
564  it = (unsigned int)((ref >> DT_POLY_BITS) & tileMask);
565  ip = (unsigned int)(ref & polyMask);
566 #else
567  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
568  const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
569  const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
570  salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
571  it = (unsigned int)((ref >> m_polyBits) & tileMask);
572  ip = (unsigned int)(ref & polyMask);
573 #endif
574  }
575 
580  inline unsigned int decodePolyIdSalt(dtPolyRef ref) const
581  {
582 #ifdef DT_POLYREF64

584  return (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
585 #else
