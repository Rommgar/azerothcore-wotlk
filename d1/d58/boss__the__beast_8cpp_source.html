124  void SetData(uint32 type, uint32 /*data*/) override
125  {
126  switch (type)
127  {
128  case DATA_BEAST_ROOM:
129  {
130  if (!_orcYelled)
131  {
132  if (_nearbyOrcsGUIDs.empty())
133  {
134  FindNearbyOrcs();
135  }
136 
138  if (_nearbyOrcsGUIDs.empty())
139  {
140  return;
141  }
142 
143  _orcYelled = true;
144 
145  bool yelled = false;
146  for (ObjectGuid guid : _nearbyOrcsGUIDs)
147  {
148  if (Creature* orc = ObjectAccessor::GetCreature(*me, guid))
149  {
150  if (!yelled)
151  {
152  yelled = true;
153  orc->AI()->Talk(SAY_BLACKHAND_DOOMED);
154  }
155 
156  orc->m_Events.AddEvent(new OrcMoveEvent(orc), me->m_Events.CalculateTime(3 * IN_MILLISECONDS));
157  orc->m_Events.AddEvent(new OrcDeathEvent(orc), me->m_Events.CalculateTime(9 * IN_MILLISECONDS));
158  }
159  }
160  }
161  break;
162  }
163  case DATA_BEAST_REACHED:
164  {
165  if (!_beastReached)
166  {
167  _beastReached = true;
168  me->GetMotionMaster()->MovePath(BEAST_MOVEMENT_ID, true);
169 
170  // There is a chance player logged in between areatriggers (realm crash or restart)
171  // executing part of script which happens when player enters boss room
172  // otherwise we will see weird behaviour when someone steps on the previous areatrigger (dead mob yelling/moving)
173  SetData(DATA_BEAST_ROOM, DATA_BEAST_ROOM);
174  }
175  break;
176  }
177  }
178  }

180  void UpdateAI(uint32 diff) override
181  {
182  if (!UpdateVictim())
183  {
184  return;
185  }
186 
187  events.Update(diff);
188 
189  if (me->HasUnitState(UNIT_STATE_CASTING))
190  {
191  return;
192  }
193 
194  while (uint32 eventId = events.ExecuteEvent())
195  {
196  switch (eventId)
197  {
198  case EVENT_FLAME_BREAK:
199  DoCastVictim(SPELL_FLAMEBREAK);
200  events.ScheduleEvent(EVENT_FLAME_BREAK, 10 * IN_MILLISECONDS);
201  break;
202  case EVENT_IMMOLATE:
203  DoCastRandomTarget(SPELL_IMMOLATE, 0, 100.0f);
204  events.ScheduleEvent(EVENT_IMMOLATE, 8 * IN_MILLISECONDS);
205  break;
206  case EVENT_TERRIFYING_ROAR:
207  DoCastVictim(SPELL_TERRIFYINGROAR);
208  events.ScheduleEvent(EVENT_TERRIFYING_ROAR, 20 * IN_MILLISECONDS);
209  break;
210  case EVENT_BERSERKER_CHARGE:
211  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 38.f, true))
212  {
213  DoCast(target, SPELL_BERSERKER_CHARGE);
214  }
215  events.ScheduleEvent(EVENT_BERSERKER_CHARGE, 15 * IN_MILLISECONDS, 23 * IN_MILLISECONDS);
216  break;
217  case EVENT_FIREBALL:
218  DoCastVictim(SPELL_FIREBALL);
219  events.ScheduleEvent(EVENT_FIREBALL, 8 * IN_MILLISECONDS, 21 * IN_MILLISECONDS);
220  break;
221  case EVENT_FIREBLAST:
222  DoCastVictim(SPELL_FIREBLAST);
223  events.ScheduleEvent(EVENT_FIREBLAST, 5 * IN_MILLISECONDS, 8 * IN_MILLISECONDS);
224  break;
225  }
226 
227  if (me->HasUnitState(UNIT_STATE_CASTING))
228  {
229  return;
230  }
231  }
232 
233  DoMeleeAttackIfReady();
234  }
