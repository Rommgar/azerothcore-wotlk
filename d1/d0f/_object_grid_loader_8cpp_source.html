151 void ObjectWorldLoader::Visit(CorpseMapType& /*m*/)
152 {
153  CellCoord cellCoord = i_cell.GetCellCoord();
154  if (std::unordered_set<Corpse*> const* corpses = i_map->GetCorpsesInCell(cellCoord.GetId()))
155  {
156  for (Corpse* corpse : *corpses)
157  {
158  corpse->AddToWorld();
159  GridType& cell = i_grid.GetGridType(i_cell.CellX(), i_cell.CellY());
160  if (corpse->IsWorldObject())
161  cell.AddWorldObject(corpse);
162  else
163  cell.AddGridObject(corpse);
164 
165  ++i_corpses;
166  }
167  }
168 }

170 void ObjectGridLoader::LoadN(void)
171 {
172  i_gameObjects = 0;
173  i_creatures = 0;
174  i_corpses = 0;
175  i_cell.data.Part.cell_y = 0;
176  for (uint32 x = 0; x < MAX_NUMBER_OF_CELLS; ++x)
177  {
178  i_cell.data.Part.cell_x = x;
179  for (uint32 y = 0; y < MAX_NUMBER_OF_CELLS; ++y)
180  {
181  i_cell.data.Part.cell_y = y;
182 
183  //Load creatures and game objects
184  {
185  TypeContainerVisitor<ObjectGridLoader, GridTypeMapContainer> visitor(*this);
186  i_grid.VisitGrid(x, y, visitor);
187  }
188 
189  //Load corpses (not bones)
190  {
191  ObjectWorldLoader worker(*this);
192  TypeContainerVisitor<ObjectWorldLoader, WorldTypeMapContainer> visitor(worker);
193  i_grid.VisitGrid(x, y, visitor);
194  }
195  }
196  }
197  LOG_DEBUG("maps", "%u GameObjects, %u Creatures, and %u Corpses/Bones loaded for grid %u on map %u", i_gameObjects, i_creatures, i_corpses, i_grid.GetGridId(), i_map->GetId());
198 }
