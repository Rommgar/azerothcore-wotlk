118 
119  Talk(SAY_TREE);
120  me->RemoveAllAuras();

122  me->GetMotionMaster()->Clear(false);
123 
124  me->CastSpell(me, SPELL_SUMMON_FRAYER, true);
125  me->CastSpell(me, SPELL_TRANQUILITY, true);
126  me->CastSpell(me, SPELL_TREE_FORM, true);
127  break;
128  case EVENT_RESTORE_COMBAT:
129  events.SetPhase(0);
130  events.ScheduleEvent(EVENT_SUMMON_SEEDLING, 6000);
131  me->GetMotionMaster()->MoveChase(me->GetVictim());
132  break;
133  case EVENT_CHECK_FRAYERS:
134  if (!summons.HasEntry(NPC_FRAYER))
135  {
136  me->InterruptNonMeleeSpells(true);
137  me->RemoveAllAuras();

139  events.RescheduleEvent(EVENT_TREE_FORM, 30000);
140  break;

142  events.ScheduleEvent(EVENT_CHECK_FRAYERS, 500);
143  break;
144  }
145 
146  if (!events.IsInPhase(1))
147  DoMeleeAttackIfReady();
148  }
