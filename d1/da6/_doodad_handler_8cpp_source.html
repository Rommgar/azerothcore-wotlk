37 void DoodadHandler::ProcessInternal( MapChunk* mcnk )
38 {
39  if (!IsSane())
40  return;
41 
42  uint32 refCount = mcnk->Header.DoodadRefs;
43  FILE* stream = mcnk->Source->GetStream();
44  fseek(stream, mcnk->Source->Offset + mcnk->Header.OffsetMCRF, SEEK_SET);
45  for (uint32 i = 0; i < refCount; i++)
46  {
47  int32 index;
48  int32 count;
49  if ((count = fread(&index, sizeof(int32), 1, stream)) != 1)
50  printf("DoodadHandler::ProcessInternal: Failed to read some data expected 1, read %d\n", count);
51  if (index < 0 || uint32(index) >= _definitions->size())
52  continue;
53  DoodadDefinition doodad = (*_definitions)[index];
54  if (_drawn.find(doodad.UniqueId) != _drawn.end())
55  continue;
56  _drawn.insert(doodad.UniqueId);
57  if (doodad.MmidIndex >= _paths->size())
58  continue;
59 
60  std::string path = (*_paths)[doodad.MmidIndex];
61  Model* model = Cache->ModelCache.Get(path);
62  if (!model)
63  {
64  model = new Model(path);
65  Cache->ModelCache.Insert(path, model);
66  }
67  if (!model->IsCollidable)
68  continue;
69 
70  Vertices.reserve(refCount * model->Vertices.size() * 0.2);
71  Triangles.reserve(refCount * model->Triangles.size() * 0.2);
72 
73  InsertModelGeometry(doodad, model);
74  }
75  // Restore the stream position
76  fseek(stream, mcnk->Source->Offset, SEEK_SET);
77 }

93 void DoodadHandler::ReadDoodadPaths( Chunk* id, Chunk* data )
94 {
95  int paths = id->Length / 4;
96  _paths = new std::vector<std::string>();
97  _paths->reserve(paths);
98  for (int i = 0; i < paths; i++)
99  {
100  FILE* idStream = id->GetStream();
101  fseek(idStream, i * 4, SEEK_CUR);
102  uint32 offset;
103  if (fread(&offset, sizeof(uint32), 1, idStream) != 1)
104  printf("DoodadHandler::ReadDoodadPaths: Failed to read some data expected 1, read 0\n");
105  FILE* dataStream = data->GetStream();
106  fseek(dataStream, offset + data->Offset, SEEK_SET);
107  _paths->push_back(Utils::ReadString(dataStream));
108  }
109 }

111 void DoodadHandler::InsertModelGeometry(const DoodadDefinition& def, Model* model)
112 {
113  uint32 vertOffset = Vertices.size();
114 
115  for (std::vector<Vector3>::iterator itr = model->Vertices.begin(); itr != model->Vertices.end(); ++itr)

117 
118  for (std::vector<Triangle<uint16>>::iterator itr = model->Triangles.begin(); itr != model->Triangles.end(); ++itr)
119  Triangles.push_back(Triangle<uint32>(Constants::TRIANGLE_TYPE_DOODAD, itr->V0 + vertOffset, itr->V1 + vertOffset, itr->V2 + vertOffset));
