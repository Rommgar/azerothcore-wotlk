188  void Reset() override
189  {
190  BossAI::Reset();
191  me->SetPosition(me->GetHomePosition());
192  movementPhase = MOVE_PHASE_NONE;
193  currentWaypoint = 0;
194  me->SetReactState(REACT_AGGRESSIVE);
195  events.Reset();

197  events.RescheduleEvent(EVENT_BERSERK, 600000);
198  if ((me->GetEntry() != NPC_LADY_BLAUMEUX && me->GetEntry() != NPC_SIR_ZELIEK))
199  {
200  events.RescheduleEvent(EVENT_PRIMARY_SPELL, 10000 + rand() % 5000);
201  }
202  else
203  {
204  events.RescheduleEvent(EVENT_PUNISH, 5000);
205  events.RescheduleEvent(EVENT_SECONDARY_SPELL, 15000);
206  }
207  if (pInstance)
208  {
209  if (GameObject* go = me->GetMap()->GetGameObject(pInstance->GetGuidData(DATA_HORSEMEN_GATE)))
210  {
211  if (pInstance->GetBossState(BOSS_GOTHIK) == DONE)
212  {
213  go->SetGoState(GO_STATE_ACTIVE);
214  }
215  }
216  }
217  }

315  void UpdateAI(uint32 diff) override
316  {
317  if (movementPhase == MOVE_PHASE_STARTED && currentWaypoint)
318  {
319  me->GetMotionMaster()->MovePoint(currentWaypoint, WaypointPositions[currentWaypoint]);
320  currentWaypoint = 0;
321  }
322 
323  if (!IsInRoom())
324  return;
325 
326  if (movementPhase < MOVE_PHASE_FINISHED || !UpdateVictim())
327  return;
328 
329  events.Update(diff);
330  if (me->HasUnitState(UNIT_STATE_CASTING))
331  return;
332 
333  switch (events.ExecuteEvent())
334  {
335  case EVENT_MARK_CAST:
336  me->CastSpell(me, TABLE_SPELL_MARK[horsemanId], false);
337  events.RepeatEvent((me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK) ? 15000 : 12000);
338  return;
339  case EVENT_BERSERK:
340  Talk(SAY_SPECIAL);
341  me->CastSpell(me, SPELL_BERSERK, true);
342  return;
343  case EVENT_PRIMARY_SPELL:
344  Talk(SAY_TAUNT);
345  me->CastSpell(me->GetVictim(), RAID_MODE(TABLE_SPELL_PRIMARY_10[horsemanId], TABLE_SPELL_PRIMARY_25[horsemanId]), false);
346  events.RepeatEvent(15000);
347  return;
348  case EVENT_PUNISH:
349  if (!SelectTarget(SELECT_TARGET_NEAREST, 0, 45.0f, true))
350  {
351  me->CastSpell(me, TABLE_SPELL_PUNISH[horsemanId], false);
352  Talk(EMOTE_RAGECAST);
353  }
354  events.RepeatEvent(2010);
355  return;
356  case EVENT_SECONDARY_SPELL:
357  me->CastSpell(me->GetVictim(), RAID_MODE(TABLE_SPELL_SECONDARY_10[horsemanId], TABLE_SPELL_SECONDARY_25[horsemanId]), false);
358  events.RepeatEvent(15000);
359  return;
360  }
361 
362  if ((me->GetEntry() == NPC_LADY_BLAUMEUX || me->GetEntry() == NPC_SIR_ZELIEK))
363  {
364  if (Unit* target = SelectTarget(SELECT_TARGET_NEAREST, 0, 45.0f, true))
365  {
366  me->CastSpell(target, RAID_MODE(TABLE_SPELL_PRIMARY_10[horsemanId], TABLE_SPELL_PRIMARY_25[horsemanId]), false);
367  }
368  }
369  else
370  {
371  DoMeleeAttackIfReady();
372  }
373  }
374  };
