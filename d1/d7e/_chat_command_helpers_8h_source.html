55  inline TokenizeResult tokenize(std::string_view args)
56  {
57  TokenizeResult result;
58  if (size_t delimPos = args.find(COMMAND_DELIMITER); delimPos != std::string_view::npos)
59  {
60  result.token = args.substr(0, delimPos);
61  if (size_t tailPos = args.find_first_not_of(COMMAND_DELIMITER, delimPos); tailPos != std::string_view::npos)
62  result.tail = args.substr(tailPos);
63  }
64  else
65  result.token = args;
66 
67  return result;
68  }
