308  break;
309  case EVENT_SWITCH_BLACK_HOLE_TARGET:
310  {
311  Map::PlayerList const& players = me->GetMap()->GetPlayers();
312  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
313  if (Player* player = itr->GetSource())
314  if (me->GetDistance2d(player) < 15.0f && player->GetPositionZ() < 72.0f && player->IsAlive() && !player->HasAura(SPELL_BLACK_HOLE_EFFECT))
315  {
316  me->GetMotionMaster()->MovePoint(0, player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), false, true);
317  events.ScheduleEvent(EVENT_SWITCH_BLACK_HOLE_TARGET, 5000);
318  return;
319  }
320  events.ScheduleEvent(EVENT_SWITCH_BLACK_HOLE_TARGET, 500);
321  break;
322  }
