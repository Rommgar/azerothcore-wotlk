35 int my_readlink(char *to, const char *filename, myf MyFlags)
36 {

38  strmov(to,filename);
39  return 1;
40 #else

42  int length;
43  DBUG_ENTER("my_readlink");
44 
45  if ((length=readlink(filename, to, FN_REFLEN-1)) < 0)
46  {
47  /* Don't give an error if this wasn't a symlink */

102 int my_is_symlink(const char *filename __attribute__((unused)))
103 {
104 #if defined (HAVE_LSTAT) && defined (S_ISLNK)
105  struct stat stat_buff;
106  return !lstat(filename, &stat_buff) && S_ISLNK(stat_buff.st_mode);
107 #elif defined (_WIN32)
108  DWORD dwAttr = GetFileAttributes(filename);
109  return (dwAttr != INVALID_FILE_ATTRIBUTES) &&
110  (dwAttr & FILE_ATTRIBUTE_REPARSE_POINT);
111 #else /* No symlinks */
112  return 0;
113 #endif
114 }
