106 void MotionMaster::UpdateMotion(uint32 diff)
107 {
108  if (!_owner)
109  return;
110 
111  ASSERT(!empty());
112 
113  _cleanFlag |= MMCF_INUSE;
114 
115  _cleanFlag |= MMCF_UPDATE;
116  if (!top()->Update(_owner, diff))
117  {
118  _cleanFlag &= ~MMCF_UPDATE;
119  MovementExpired();
120  }
121  else
122  _cleanFlag &= ~MMCF_UPDATE;
123 
124  if (_expList)
125  {
126  for (size_t i = 0; i < _expList->size(); ++i)
127  {
128  MovementGenerator* mg = (*_expList)[i];
129  DirectDelete(mg);
130  }
131 
132  delete _expList;
133  _expList = nullptr;
134 
135  if (empty())
136  Initialize();
137  else if (needInitTop())
138  InitTop();
139  else if (_cleanFlag & MMCF_RESET)
140  top()->Reset(_owner);
141 
142  _cleanFlag &= ~MMCF_RESET;
143  }
144 
145  _cleanFlag &= ~MMCF_INUSE;

548 void MotionMaster::MoveFall(uint32 id /*=0*/, bool addFlagForNPC)
549 {
550  // Xinef: do not allow to move with UNIT_FLAG_DISABLE_MOVE
551  if (_owner->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE))
552  return;
553 
554  // use larger distance for vmap height search than in most other cases
555  float tz = _owner->GetMapHeight(_owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ(), true, MAX_FALL_DISTANCE);
556  if (tz <= INVALID_HEIGHT)
557  {
558  LOG_DEBUG("movement.motionmaster", "MotionMaster::MoveFall: unable retrive a proper height at map %u (x: %f, y: %f, z: %f).",
559  _owner->GetMap()->GetId(), _owner->GetPositionX(), _owner->GetPositionX(), _owner->GetPositionZ() + _owner->GetPositionZ());
560  return;
561  }
562 
563  // Abort too if the ground is very near
564  if (fabs(_owner->GetPositionZ() - tz) < 0.1f)
565  return;
566 
567  if (_owner->GetTypeId() == TYPEID_PLAYER)
568  {
569  _owner->AddUnitMovementFlag(MOVEMENTFLAG_FALLING);
570  _owner->m_movementInfo.SetFallTime(0);
571  _owner->ToPlayer()->SetFallInformation(time(nullptr), _owner->GetPositionZ());
572  }
573  else if (_owner->GetTypeId() == TYPEID_UNIT && addFlagForNPC) // pussywizard
574  {
575  _owner->RemoveUnitMovementFlag(MOVEMENTFLAG_MASK_MOVING);
576  _owner->RemoveUnitMovementFlag(MOVEMENTFLAG_FLYING | MOVEMENTFLAG_CAN_FLY);
577  _owner->AddUnitMovementFlag(MOVEMENTFLAG_FALLING);
578  _owner->m_movementInfo.SetFallTime(0);
579  _owner->SendMovementFlagUpdate();
580  }
581 
582  Movement::MoveSplineInit init(_owner);
583  init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), tz + _owner->GetHoverHeight());
584  init.SetFall();
585  init.Launch();
586  Mutate(new EffectMovementGenerator(id), MOTION_SLOT_CONTROLLED);
587 }
