306 void BattlegroundSA::PostUpdateImpl(uint32 diff)
307 {
308  if (InitSecondRound)
309  {
310  if (UpdateWaitTimer < diff)
311  {
312  if (!SignaledRoundTwo)
313  {
314  SignaledRoundTwo = true;
315  InitSecondRound = false;
316  SendMessageToAll(LANG_BG_SA_ROUND_TWO_ONE_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
317  }
318  }
319  else
320  {
321  UpdateWaitTimer -= diff;
322  return;
323  }
324  }
325  TotalTime += diff;
326 
327  if (Status == BG_SA_WARMUP)
328  {
329  EndRoundTimer = BG_SA_ROUNDLENGTH;
330  if (TotalTime >= BG_SA_WARMUPLENGTH)
331  {
332  TotalTime = 0;
333  ToggleTimer();
334  DemolisherStartState(false);
335  Status = BG_SA_ROUND_ONE;
336  StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
337  }
338  if (TotalTime >= BG_SA_BOAT_START)
339  StartShips();
340  return;
341  }
342  else if (Status == BG_SA_SECOND_WARMUP)
343  {
344  if (RoundScores[0].time < BG_SA_ROUNDLENGTH)
345  EndRoundTimer = RoundScores[0].time;
346  else
347  EndRoundTimer = BG_SA_ROUNDLENGTH;
348 
349  if (TotalTime >= 60000)
350  {
351  SendWarningToAll(LANG_BG_SA_HAS_BEGUN);
352  TotalTime = 0;
353  ToggleTimer();
354  DemolisherStartState(false);
355  Status = BG_SA_ROUND_TWO;
356  StartTimedAchievement(ACHIEVEMENT_TIMED_TYPE_EVENT, (Attackers == TEAM_ALLIANCE) ? 23748 : 21702);
357 
358  // status was set to STATUS_WAIT_JOIN manually for Preparation, set it back now
359  SetStatus(STATUS_IN_PROGRESS);
360  for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
361  itr->second->RemoveAurasDueToSpell(SPELL_PREPARATION);
362  }
363  if (TotalTime >= 30000)
364  {
365  if (!SignaledRoundTwoHalfMin)
366  {
367  SignaledRoundTwoHalfMin = true;
368  SendMessageToAll(LANG_BG_SA_ROUND_TWO_START_HALF_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
369  }
370  }
371  StartShips();
372  return;
373  }
374  else if (GetStatus() == STATUS_IN_PROGRESS)
375  {
376  if (Status == BG_SA_ROUND_ONE)
377  {
378  if (TotalTime >= BG_SA_ROUNDLENGTH || _relicClicked)
379  {
380  if (_relicClicked)
381  {
382  RoundScores[0].winner = Attackers;
383  RoundScores[0].time = TotalTime;
384  //Achievement Storm the Beach (1310)
385  for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
386  if (itr->second->GetTeamId() == Attackers)
387  itr->second->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, 65246);
388  }
389  else
390  {
391  // cast this before Attackers variable is switched
392  // cast this spell only upon timer end, no other ability for defenders to win :)
393  for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
394  itr->second->CastSpell(itr->second, SPELL_SA_END_OF_ROUND, true);
395 
396  RoundScores[0].winner = Attackers;
397  RoundScores[0].time = BG_SA_ROUNDLENGTH;
398  }
399 
400  _relicClicked = false;
401  Attackers = (Attackers == TEAM_ALLIANCE) ? TEAM_HORDE : TEAM_ALLIANCE;
402  Status = BG_SA_SECOND_WARMUP;
403  TotalTime = 0;
404  ToggleTimer();
405  SendWarningToAll(LANG_BG_SA_ROUND_ONE_END);
406  UpdateWaitTimer = 5000;
407  SignaledRoundTwo = false;
408  SignaledRoundTwoHalfMin = false;
409  InitSecondRound = true;
410  ResetObjs();
411 
412  return;
413  }
414  }
415  else if (Status == BG_SA_ROUND_TWO)
416  {
417  if (TotalTime >= EndRoundTimer)
418  {
419  // cast this spell only upon timer end, no other ability for defenders to win :)
420  for (BattlegroundPlayerMap::const_iterator itr = GetPlayers().begin(); itr != GetPlayers().end(); ++itr)
421  itr->second->CastSpell(itr->second, SPELL_SA_END_OF_ROUND, true);
422 
423  RoundScores[1].time = BG_SA_ROUNDLENGTH;
424  RoundScores[1].winner = GetOtherTeamId(Attackers);
425 
426  if (RoundScores[0].time == RoundScores[1].time)
427  EndBattleground(TEAM_NEUTRAL);
428  else if (RoundScores[0].time < RoundScores[1].time)
429  EndBattleground(RoundScores[0].winner);
430  else
431  EndBattleground(RoundScores[1].winner);
432  return;
433  }
434  }
435  if (Status == BG_SA_ROUND_ONE || Status == BG_SA_ROUND_TWO)
436  {
437  SendTime();
438  UpdateDemolisherSpawns();
439  }
440  }
