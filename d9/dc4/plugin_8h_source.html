171 /*
172  SYNOPSIS
173  (*mysql_var_check_func)()
174  thd thread handle
175  var dynamic variable being altered
176  save pointer to temporary storage
177  value user provided value
178  RETURN
179  0 user provided value is OK and the update func may be called.
180  any other value indicates error.
181 
182  This function should parse the user provided value and store in the
183  provided temporary storage any data as required by the update func.
184  There is sufficient space in the temporary storage to store a double.
185  Note that the update func may not be called if any other error occurs
186  so any memory allocated should be thread-local so that it may be freed
187  automatically at the end of the statement.
188 */
189 
190 typedef int (*mysql_var_check_func)(MYSQL_THD thd,
191  struct st_mysql_sys_var *var,
192  void *save, struct st_mysql_value *value);
193 
194 /*
195  SYNOPSIS
196  (*mysql_var_update_func)()
197  thd thread handle
198  var dynamic variable being altered
199  var_ptr pointer to dynamic variable
200  save pointer to temporary storage
201  RETURN
202  NONE
203 
204  This function should use the validated value stored in the temporary store
205  and persist it in the provided pointer to the dynamic variable.
206  For example, strings may require memory to be allocated.
207 */
208 typedef void (*mysql_var_update_func)(MYSQL_THD thd,
209  struct st_mysql_sys_var *var,
210  void *var_ptr, const void *save);
