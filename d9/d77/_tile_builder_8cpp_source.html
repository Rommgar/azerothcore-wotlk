139  {

141  pmesh->flags[i] = Constants::POLY_FLAG_WALK;
142  else if (pmesh->areas[i] == Constants::POLY_AREA_WATER)

174

176  rcVcopy(params.bmin, bmin);
177 
178  // Offmesh-connection settings
179  params.offMeshConCount = 0; // none for now
180 
181  rcFreeHeightField(hf);
182  rcFreeCompactHeightfield(chf);
183  rcFreeContourSet(contours);
184  delete vertices;
185  delete triangles;
186  delete areas;
187  delete bmin;
188  delete bmax;
189 
190  if (!params.polyCount || !params.polys || Constants::TilesPerMap * Constants::TilesPerMap == params.polyCount)
191  {
192  // we have flat tiles with no actual geometry - don't build those, its useless

196  rcFreePolyMesh(pmesh);
197  rcFreePolyMeshDetail(dmesh);
198  return nullptr;
199  }
200 
201  int navDataSize;
202  uint8* navData;
203  printf("Creating the navmesh with %i vertices, %i polys, %i triangles!\n", params.vertCount, params.polyCount, params.detailTriCount);
204  bool result = dtCreateNavMeshData(&params, &navData, &navDataSize);
205 
206  rcFreePolyMesh(pmesh);
207  rcFreePolyMeshDetail(dmesh);
208 
209  if (result)
210  {
211  printf("NavMesh created, size %i!\n", navDataSize);
212  DataSize = navDataSize;
213  return navData;
214  }
215 
216  return nullptr;

229  return nullptr;

233  _Geometry->CalculateMinMaxHeight(bmin[1], bmax[1]);
234 
235  // again, we load everything - wasteful but who cares
236  for (int ty = Y - 1; ty <= Y + 1; ty++)
237  {
238  for (int tx = X - 1; tx <= X + 1; tx++)
239  {
240  // don't load main tile again

242  continue;

247  {
248  delete _adt;
249  continue;

254  }
255  }
