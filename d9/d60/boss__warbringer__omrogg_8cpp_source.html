157  void UpdateAI(uint32 diff) override
158  {
159  events2.Update(diff);
160  switch (uint32 eventId = events2.ExecuteEvent())
161  {
162  case EVENT_AGGRO_YELL_1:
163  case EVENT_AGGRO_YELL_2:
164  case EVENT_AGGRO_YELL_3:
165  case EVENT_KILL_YELL_LEFT:
166  case EVENT_THREAT_YELL_L_1:
167  case EVENT_THREAT_YELL_L_2:
168  case EVENT_THREAT_YELL_L_3:
169  if (Creature* RightHead = GetRightHead())
170  RightHead->AI()->Talk(eventId - 1);
171  break;
172  case EVENT_KILL_YELL_RIGHT:
173  case EVENT_THREAT_YELL_R_1:
174  if (Creature* LeftHead = GetLeftHead())
175  LeftHead->AI()->Talk(eventId - 1);
176  break;
177  }
178 
179  if (!UpdateVictim())
180  return;
181 
182  events.Update(diff);
183  if (me->HasUnitState(UNIT_STATE_CASTING))
184  return;
185 
186  switch (events.ExecuteEvent())
187  {
188  case EVENT_SPELL_FEAR:
189  me->CastSpell(me, SPELL_FEAR, false);
190  events.ScheduleEvent(EVENT_SPELL_FEAR, 22000);
191  break;
192  case EVENT_SPELL_THUNDER_CLAP:
193  me->CastSpell(me, SPELL_THUNDERCLAP, false);
194  events.ScheduleEvent(EVENT_SPELL_THUNDER_CLAP, 25000);
195  break;
196  case EVENT_RESET_THREAT:
197  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
198  {
199  uint8 threatYell = urand(EVENT_THREAT_YELL_L_1, EVENT_THREAT_YELL_R_1);
200  if (Creature* head = threatYell == EVENT_THREAT_YELL_R_1 ? GetRightHead() : GetLeftHead())
201  head->AI()->Talk(threatYell - 1);
202  events.ScheduleEvent(threatYell, 3000);
203 
204  DoResetThreat();
205  me->AddThreat(target, 10.0f);
206  }
207  events.ScheduleEvent(EVENT_RESET_THREAT, 30000);
208  break;
209  case EVENT_SPELL_BURNING_MAUL:
210  Talk(EMOTE_ENRAGE);
211  me->CastSpell(me, DUNGEON_MODE(SPELL_BURNING_MAUL_N, SPELL_BURNING_MAUL_H), false);
212  events.ScheduleEvent(EVENT_SPELL_BURNING_MAUL, 40000);
213  events.ScheduleEvent(EVENT_SPELL_BLAST_WAVE, 15000);
214  events.ScheduleEvent(EVENT_SPELL_BLAST_WAVE, 20000);
215  break;
216  case EVENT_SPELL_BLAST_WAVE:
217  me->CastSpell(me, SPELL_BLAST_WAVE, false);
218  break;
219  }
220 
221  DoMeleeAttackIfReady();
222  }
