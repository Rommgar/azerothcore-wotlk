41 // default predicate function to select target based on distance, player and/or aura criteria
42 struct DefaultTargetSelector : public Acore::unary_function<Unit*, bool>
43 {
44  const Unit* me;
45  float m_dist;
46  bool m_playerOnly;
47  int32 m_aura;
48 
49  // unit: the reference unit
50  // dist: if 0: ignored, if > 0: maximum distance to the reference unit, if < 0: minimum distance to the reference unit
51  // playerOnly: self explaining
52  // aura: if 0: ignored, if > 0: the target shall have the aura, if < 0, the target shall NOT have the aura
53  DefaultTargetSelector(Unit const* unit, float dist, bool playerOnly, int32 aura) : me(unit), m_dist(dist), m_playerOnly(playerOnly), m_aura(aura) {}
54 
55  bool operator()(Unit const* target) const
56  {
57  if (!me)
58  return false;
59 
60  if (!target)
61  return false;
62 
63  if (m_playerOnly && (target->GetTypeId() != TYPEID_PLAYER))
64  return false;
65 
66  if (m_dist > 0.0f && !me->IsWithinCombatRange(target, m_dist))
67  return false;
68 
69  if (m_dist < 0.0f && me->IsWithinCombatRange(target, -m_dist))
70  return false;
71 
72  if (m_aura)
73  {
74  if (m_aura > 0)
75  {
76  if (!target->HasAura(m_aura))
77  return false;
78  }
79  else
80  {
81  if (target->HasAura(-m_aura))
82  return false;
83  }
84  }
85 
86  return true;
87  }
88 };

203  // Select the targets satifying the predicate.
204  // predicate shall extend Acore::unary_function<Unit*, bool>
205  template <class PREDICATE> Unit* SelectTarget(SelectAggroTarget targetType, uint32 position, PREDICATE const& predicate)
206  {
207  ThreatContainer::StorageType const& threatlist = me->getThreatMgr().getThreatList();
208  if (position >= threatlist.size())
209  return nullptr;
210 
211  std::list<Unit*> targetList;
212  for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
213  if (predicate((*itr)->getTarget()))
214  targetList.push_back((*itr)->getTarget());
215 
216  if (position >= targetList.size())
217  return nullptr;
218 
219  if (targetType == SELECT_TARGET_NEAREST || targetType == SELECT_TARGET_FARTHEST)
220  targetList.sort(Acore::ObjectDistanceOrderPred(me));
221 
222  switch (targetType)
223  {
224  case SELECT_TARGET_NEAREST:
225  case SELECT_TARGET_TOPAGGRO:
226  {
227  std::list<Unit*>::iterator itr = targetList.begin();
228  std::advance(itr, position);
229  return *itr;
230  }
231  case SELECT_TARGET_FARTHEST:
232  case SELECT_TARGET_BOTTOMAGGRO:
233  {
234  std::list<Unit*>::reverse_iterator ritr = targetList.rbegin();
235  std::advance(ritr, position);
236  return *ritr;
237  }
238  case SELECT_TARGET_RANDOM:
239  {
240  std::list<Unit*>::iterator itr = targetList.begin();
241  std::advance(itr, urand(position, targetList.size() - 1));
242  return *itr;
243  }
244  default:
245  break;
246  }
247 
248  return nullptr;
249  }

253  // Select the targets satifying the predicate.
254  // predicate shall extend Acore::unary_function<Unit*, bool>
255  template <class PREDICATE> void SelectTargetList(std::list<Unit*>& targetList, PREDICATE const& predicate, uint32 maxTargets, SelectAggroTarget targetType)
256  {
257  ThreatContainer::StorageType const& threatlist = me->getThreatMgr().getThreatList();
258  if (threatlist.empty())
259  return;
260 
261  for (ThreatContainer::StorageType::const_iterator itr = threatlist.begin(); itr != threatlist.end(); ++itr)
262  if (predicate((*itr)->getTarget()))
263  targetList.push_back((*itr)->getTarget());
264 
265  if (targetList.size() < maxTargets)
266  return;
267 
268  if (targetType == SELECT_TARGET_NEAREST || targetType == SELECT_TARGET_FARTHEST)
269  targetList.sort(Acore::ObjectDistanceOrderPred(me));
270 
271  if (targetType == SELECT_TARGET_FARTHEST || targetType == SELECT_TARGET_BOTTOMAGGRO)
272  targetList.reverse();
273 
274  if (targetType == SELECT_TARGET_RANDOM)
275  Acore::Containers::RandomResize(targetList, maxTargets);
276  else
277  targetList.resize(maxTargets);
278  }
