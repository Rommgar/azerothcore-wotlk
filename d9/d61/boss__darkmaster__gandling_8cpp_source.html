137  void SummonedCreatureDespawn(Creature* cr) override
138  {
139  int room = -1;
140  // remove the mob from the list and keep track of which room he is in.
141  for (uint8 i = 0; i < 6; i++)
142  {
143  for (uint8 j = 0; j < 3; j++)
144  {
145  if (Guardians[i][j] && Guardians[i][j]->GetGUID() == cr->GetGUID())
146  {
147  room = i;
148  Guardians[i][j] = nullptr;
149  }
150  }
151  }
152 
153  // check if the room is now empty
154  if (room >= 0)
155  {
156  for (uint8 i = 0; i < 3; i++)
157  {
158  if (Guardians[room][i])
159  {
160  return;
161  }
162  }
163  // everybody is dead in there, we can open the gate.
164  SetGate(room, OPEN);
165  }
166  }

264  // Finds a random room that is not in use
265  uint32 FindRoom()

295  // used for shadow portal
296  void SpellHitTarget(Unit* target, const SpellInfo* spellinfo) override
297  {

306  {

316  void UpdateAI(uint32 diff) override
317  {
318  uint32 room = 0;
319  if (!UpdateVictim())
320  {
321  return;
322  }
323 
324  events.Update(diff);
325  if (me->HasUnitState(UNIT_STATE_CASTING))
326  {
327  return;
328  }
329 
330  while (uint32 eventId = events.ExecuteEvent())
331  {
332  switch (eventId)
333  {
334  case SPELL_ARCANE_MISSILES:
335  DoCastVictim(SPELL_ARCANE_MISSILES);
336  events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
337  break;
338  case SPELL_CURSE_DARKMASTER:
339  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
340  {
341  DoCast(target, SPELL_CURSE_DARKMASTER);
342  }
343  events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
344  break;
345  case SPELL_SHADOW_SHIELD:
346  DoCastSelf(SPELL_SHADOW_SHIELD);
347  events.ScheduleEvent(SPELL_ARCANE_MISSILES, urand(TIMER_ARCANE_MIN, TIMER_ARCANE_MAX));
348  break;
349 
350  case SPELL_SHADOW_PORTAL:
351  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 15.0, true))
352  {
353  room = FindRoom();
354  if (room < 6)
355  {
356  SetData(GANDLING_ROOM_TO_USE, room);
357  DoCast(target, SPELL_SHADOW_PORTAL); // don't use triggered here
358  }
359  }
360  events.ScheduleEvent(SPELL_SHADOW_PORTAL, TIMER_PORTAL);
361  break;
362  default:
363  break;
364  }
365  }
366  DoMeleeAttackIfReady();
367  }
368  };
