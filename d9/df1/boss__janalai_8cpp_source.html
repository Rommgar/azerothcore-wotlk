300  void UpdateAI(uint32 diff) override
301  {
302  if (isFlameBreathing)
303  {
304  if (!me->IsNonMeleeSpellCast(false))
305  isFlameBreathing = false;
306  else
307  return;
308  }
309 
310  if (isBombing)
311  {
312  if (BombSequenceTimer <= diff)
313  HandleBombSequence();
314  else
315  BombSequenceTimer -= diff;
316  return;
317  }
318 
319  if (!UpdateVictim())
320  return;
321 
322  //enrage if under 25% hp before 5 min.
323  if (!enraged && HealthBelowPct(25))
324  EnrageTimer = 0;
325 
326  if (EnrageTimer <= diff)
327  {
328  if (!enraged)
329  {
330  DoCast(me, SPELL_ENRAGE, true);
331  enraged = true;
332  EnrageTimer = 300000;
333  }
334  else
335  {
336  Talk(SAY_BERSERK);
337  DoCast(me, SPELL_BERSERK, true);
338  EnrageTimer = 300000;
339  }
340  }
341  else EnrageTimer -= diff;
342 
343  if (BombTimer <= diff)
344  {
345  Talk(SAY_FIRE_BOMBS);
346 
347  me->AttackStop();
348  me->GetMotionMaster()->Clear();
349  me->NearTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], me->GetOrientation());
350  me->StopMovingOnCurrentPos();
351  DoCast(me, SPELL_FIRE_BOMB_CHANNEL, false);
352  //DoTeleportPlayer(me, JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], 0);
353  //DoCast(me, SPELL_TELE_TO_CENTER, true);
354 
355  FireWall();
356  SpawnBombs();
357  isBombing = true;
358  BombSequenceTimer = 100;
359 
360  //Teleport every Player into the middle
361  Map* map = me->GetMap();
362  if (!map->IsDungeon())
363  return;
364 
365  Map::PlayerList const& PlayerList = map->GetPlayers();
366  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
367  if (Player* i_pl = i->GetSource())
368  if (i_pl->IsAlive())
369  DoTeleportPlayer(i_pl, JanalainPos[0][0] - 5 + rand() % 10, JanalainPos[0][1] - 5 + rand() % 10, JanalainPos[0][2], 0);
370  //DoCast(Temp, SPELL_SUMMON_PLAYERS, true) // core bug, spell does not work if too far
371  return;
372  }
373  else BombTimer -= diff;
374 
375  if (!noeggs)
376  {
377  if (HealthBelowPct(35))
378  {
379  Talk(SAY_ALL_EGGS);
380 
381  me->AttackStop();
382  me->GetMotionMaster()->Clear();
383  me->NearTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], me->GetOrientation());
384  me->StopMovingOnCurrentPos();
385  DoCast(me, SPELL_HATCH_ALL, false);
386  HatchAllEggs(2);
387  noeggs = true;
388  }
389  else if (HatcherTimer <= diff)
390  {
391  if (HatchAllEggs(0))
392  {
393  Talk(SAY_SUMMON_HATCHER);
394  me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[0][0][0], hatcherway[0][0][1], hatcherway[0][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
395  me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[1][0][0], hatcherway[1][0][1], hatcherway[1][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
396  HatcherTimer = 90000;
397  }
398  else
399  noeggs = true;
400  }
401  else HatcherTimer -= diff;
402  }
403 
404  EnterEvadeIfOutOfCombatArea();
405 
406  DoMeleeAttackIfReady();
407 
408  if (FireBreathTimer <= diff)
409  {
410  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
411  {
412  me->AttackStop();
413  me->GetMotionMaster()->Clear();
414  DoCast(target, SPELL_FLAME_BREATH, false);
415  me->StopMoving();
416  isFlameBreathing = true;
417  }
418  FireBreathTimer = 8000;
419  }
420  else FireBreathTimer -= diff;
421  }
