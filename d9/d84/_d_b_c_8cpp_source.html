22 DBC::DBC( FILE* stream ) : StringBlock(nullptr), StringBlockSize(0), IsFaulty(true)
23 {
24  char magic[5];
25  uint32 count = 0;
26  count += fread(&magic, sizeof(char), 4, stream);
27  magic[4] = '\0';
28  count += fread(&RecordCount, sizeof(uint32), 1, stream);
29  Records.reserve(RecordCount);
30  count += fread(&Fields, sizeof(uint32), 1, stream);
31  count += fread(&RecordSize, sizeof(uint32), 1, stream);
32  count += fread(&StringBlockSize, sizeof(uint32), 1, stream);
33  if (count != 8)
34  printf("DBC::DBC: Failed to read some data expected 8, read %u\n", count);
35 
36  for (int i = 0; i < RecordCount; i++)
37  {
38  Record* rec = new Record(this);
39  Records.push_back(rec);
40  int size = 0;
41  for (int f = 0; f < Fields; f++)
42  {

44  {
45  IsFaulty = true;

47  }

52  size += 4;

54  }

57  if (count != StringBlockSize)

59 }
