138 bool PreparedStatementTask::Execute()
139 {
140  if (m_has_result)
141  {
142  PreparedResultSet* result = m_conn->Query(m_stmt);
143  if (!result || !result->GetRowCount())
144  {
145  delete result;
146  m_result->set_value(PreparedQueryResult(nullptr));
147  return false;
148  }
149  m_result->set_value(PreparedQueryResult(result));
150  return true;
151  }
152 
153  return m_conn->Execute(m_stmt);
154 }
