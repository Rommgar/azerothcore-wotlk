92  switch (m_State)

98  artkit = 2;
99  break;
100  case OBJECTIVESTATE_HORDE:
101  sGameEventMgr->StopEvent(GH_ALLIANCE_DEFENSE_EVENT);
102  sGameEventMgr->StopEvent(GH_HORDE_DEFENSE_EVENT);
103  sGameEventMgr->StartEvent(GH_HORDE_DEFENSE_EVENT);
104  artkit = 1;
105  break;
106  default:
107  break;
108  }
109 
110  Map* map = sMapMgr->FindMap(571, 0);
111  auto bounds = map->GetGameObjectBySpawnIdStore().equal_range(m_capturePointSpawnId);
112  for (auto itr = bounds.first; itr != bounds.second; ++itr)
113  itr->second->SetGoArtKit(artkit);
