137  void UpdateAI(uint32 diff) override
138  {
139  events.Update(diff);
140 
141  // Speech
142  if (!UpdateVictim())
143  {
144  while (uint32 eventId = events.ExecuteEvent())
145  {
146  switch (eventId)
147  {
148  case EVENT_SPEECH_1:
149  me->SummonCreature(NPC_VICTOR_NEFARIUS, aNefariusSpawnLoc[0], aNefariusSpawnLoc[1], aNefariusSpawnLoc[2], aNefariusSpawnLoc[3], TEMPSUMMON_TIMED_DESPAWN, 26000);
150  events.ScheduleEvent(EVENT_SPEECH_2, 1000);
151  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
152  break;
153  case EVENT_SPEECH_2:
154  if (Creature* nefarius = me->GetMap()->GetCreature(m_nefariusGuid))
155  {
156  nefarius->CastSpell(me, SPELL_NEFARIUS_CORRUPTION, TRIGGERED_CAST_DIRECTLY);
157  nefarius->Yell(SAY_NEFARIAN_VAEL_INTRO);
158  nefarius->SetStandState(UNIT_STAND_STATE_STAND);
159  }
160  events.ScheduleEvent(EVENT_SPEECH_3, 18000);
161  break;
162  case EVENT_SPEECH_3:
163  if (Creature* nefarius = me->GetMap()->GetCreature(m_nefariusGuid))
164  nefarius->CastSpell(me, SPELL_RED_LIGHTNING, TRIGGERED_NONE);
165  events.ScheduleEvent(EVENT_SPEECH_4, 2000);
166  break;
167  case EVENT_SPEECH_4:
168  Talk(SAY_LINE1);
169  me->SetStandState(UNIT_STAND_STATE_STAND);
170  me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
171  events.ScheduleEvent(EVENT_SPEECH_5, 12000);
172  break;
173  case EVENT_SPEECH_5:
174  Talk(SAY_LINE2);
175  me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
176  events.ScheduleEvent(EVENT_SPEECH_6, 12000);
177  break;
178  case EVENT_SPEECH_6:
179  Talk(SAY_LINE3);
180  me->HandleEmoteCommand(EMOTE_ONESHOT_TALK);
181  events.ScheduleEvent(EVENT_SPEECH_7, 17000);
182  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
183  break;
184  case EVENT_SPEECH_7:
185  me->SetFaction(FACTION_DRAGONFLIGHT_BLACK);
186  if (PlayerGUID && ObjectAccessor::GetUnit(*me, PlayerGUID))
187  AttackStart(ObjectAccessor::GetUnit(*me, PlayerGUID));
188  break;
189  }
190  }
191  return;
192  }
193 
194  if (me->HasUnitState(UNIT_STATE_CASTING))
195  return;
196 
197  while (uint32 eventId = events.ExecuteEvent())
198  {
199  switch (eventId)
200  {
201  case EVENT_CLEAVE:
202  events.ScheduleEvent(EVENT_CLEAVE, 15000);
203  DoCastVictim(SPELL_CLEAVE);
204  break;
205  case EVENT_FLAMEBREATH:
206  DoCastVictim(SPELL_FLAMEBREATH);
207  events.ScheduleEvent(EVENT_FLAMEBREATH, 8000, 14000);
208  break;
209  case EVENT_FIRENOVA:
210  DoCastVictim(SPELL_FIRENOVA);
211  events.ScheduleEvent(EVENT_FIRENOVA, 15000);
212  break;
213  case EVENT_TAILSWIPE:
214  //Only cast if we are behind
215  /*if (!me->HasInArc(M_PI, me->GetVictim()))
216  {
217  DoCast(me->GetVictim(), SPELL_TAILSWIPE);
218  }*/
219  events.ScheduleEvent(EVENT_TAILSWIPE, 15000);
220  break;
221  case EVENT_BURNINGADRENALINE_CASTER:
222  {
223  //selects a random target that isn't the current victim and is a mana user (selects mana users) but not pets
224  //it also ignores targets who have the aura. We don't want to place the debuff on the same target twice.
225  if (Unit *target = SelectTarget(SELECT_TARGET_RANDOM, 1, [&](Unit* u) { return u && !u->IsPet() && u->getPowerType() == POWER_MANA && !u->HasAura(SPELL_BURNINGADRENALINE); }))
226  {
227  me->CastSpell(target, SPELL_BURNINGADRENALINE, true);
228  }
229  }
230  //reschedule the event
231  events.ScheduleEvent(EVENT_BURNINGADRENALINE_CASTER, 15000);
232  break;
233  case EVENT_BURNINGADRENALINE_TANK:
234  //Vael has to cast it himself; contrary to the previous commit's comment. Nothing happens otherwise.
235  me->CastSpell(me->GetVictim(), SPELL_BURNINGADRENALINE, true);
236  events.ScheduleEvent(EVENT_BURNINGADRENALINE_TANK, 45000);
237  break;
238  }
239 
240  if (me->HasUnitState(UNIT_STATE_CASTING))
241  return;
242  }
243 
244  // Yell if hp lower than 15%
245  if (HealthBelowPct(15) && !HasYelled)
246  {
247  Talk(SAY_HALFLIFE);
248  HasYelled = true;
249  }
250 
251  DoMeleeAttackIfReady();
252  }
