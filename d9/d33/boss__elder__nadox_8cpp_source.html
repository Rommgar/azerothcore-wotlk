183  void UpdateAI(uint32 diff) override
184  {
185  if (!UpdateVictim())
186  {
187  return;
188  }
189 
190  events.Update(diff);
191 
192  if (me->HasUnitState(UNIT_STATE_CASTING))
193  {
194  return;
195  }
196 
197  while (uint32 const eventId = events.ExecuteEvent())
198  {
199  switch (eventId)
200  {
201  case EVENT_BROOD_RAGE:
202  {
203  if (Creature* pSwarmer = me->FindNearestCreature(NPC_AHNKAHAR_SWARMER, 40, true))
204  DoCast(pSwarmer, SPELL_BROOD_RAGE_H, true);
205 
206  events.RepeatEvent(10000);
207  break;
208  }
209  case EVENT_PLAGUE:
210  {
211  DoCastVictim(SPELL_BROOD_PLAGUE, false);
212  events.RepeatEvent(urand(12000, 17000));
213  break;
214  }
215  case EVENT_SWARMER:
216  {
217  SummonHelpers(true);
218  events.RepeatEvent(10000);
219  break;
220  }
221  case EVENT_CHECK_HOME:
222  {
223  if (!me->HasAura(SPELL_ENRAGE) && (me->GetPositionZ() < 24.0f || !me->GetHomePosition().IsInDist(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), 110.0f)))
224  {
225  DoCastSelf(SPELL_ENRAGE, true);
226  }
227  break;
228  }
229  }
230  }
231 
232  DoMeleeAttackIfReady();
233  }

242  void SummonHelpers(bool swarm)
243  {
244  if (swarm)
245  {
246  if (swarmEggs.empty())
247  {
248  return;
249  }
250 
251  // Make a copy of guid list
252  GuidList swarmEggs2 = swarmEggs;
253 
254  // Remove previous egg
255  if (previousSwarmEgg_GUID)
256  {
257  std::list<ObjectGuid>::iterator itr = std::find(swarmEggs2.begin(), swarmEggs2.end(), previousSwarmEgg_GUID);
258  if (itr != swarmEggs2.end())
259  {
260  swarmEggs2.erase(itr);
261  }
262  }
263 
264  if (swarmEggs2.empty())
265  {
266  return;
267  }
268 
269  previousSwarmEgg_GUID = Acore::Containers::SelectRandomContainerElement(swarmEggs2);
270 
271  if (Creature* egg = ObjectAccessor::GetCreature(*me, previousSwarmEgg_GUID))
272  {
273  egg->CastSpell(egg, SPELL_SUMMON_SWARMERS, true, nullptr, nullptr, me->GetGUID());
274  }
275 
276  if (roll_chance_f(33))
277  {
278  Talk(SAY_EGG_SAC);
279  }
280  }
281  else
282  {
283  if (guardianEggs.empty())
284  {
285  return;
286  }
287 
288  ObjectGuid const& guardianEggGUID = Acore::Containers::SelectRandomContainerElement(guardianEggs);
289  if (Creature* egg = ObjectAccessor::GetCreature(*me, guardianEggGUID))
290  {
291  egg->CastSpell(egg, SPELL_SUMMON_SWARM_GUARD, true, nullptr, nullptr, me->GetGUID());
292  }
293 
294  Talk(EMOTE_HATCHES, me);
295 
296  if (roll_chance_f(33))
297  {
298  Talk(SAY_EGG_SAC);
299  }
300  }
301  }
