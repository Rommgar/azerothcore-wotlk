115 void CliThread()
116 {
117 #if AC_PLATFORM == AC_PLATFORM_WINDOWS
118  // print this here the first time
119  // later it will be printed after command queue updates
120  PrintCliPrefix();
121 #else
122  ::rl_attempted_completion_function = &Acore::Impl::Readline::cli_completion;
123  {
124  static char BLANK = '\0';
125  ::rl_completer_word_break_characters = &BLANK;
126  }
127  ::rl_event_hook = &Acore::Impl::Readline::cli_hook_func;
128 #endif
129 
130  if (sConfigMgr->GetOption<bool>("BeepAtStart", true))
131  printf("\a"); // \a = Alert
132 
133 #if AC_PLATFORM == AC_PLATFORM_WINDOWS
134  if (sConfigMgr->GetOption<bool>("FlashAtStart", true))
135  {
136  FLASHWINFO fInfo;
137  fInfo.cbSize = sizeof(FLASHWINFO);
138  fInfo.dwFlags = FLASHW_TRAY | FLASHW_TIMERNOFG;
139  fInfo.hwnd = GetConsoleWindow();
140  fInfo.uCount = 0;
141  fInfo.dwTimeout = 0;
142  FlashWindowEx(&fInfo);
143  }
144 #endif
145 
147  while (!World::IsStopped())
148  {
149  fflush(stdout);
150 
151  std::string command;
152 
153 #if AC_PLATFORM == AC_PLATFORM_WINDOWS
154  wchar_t commandbuf[256];
155  if (fgetws(commandbuf, sizeof(commandbuf), stdin))
156  {
157  if (!WStrToUtf8(commandbuf, wcslen(commandbuf), command))
158  {
159  PrintCliPrefix();
160  continue;
161  }
162  }
163 #else
164  char* command_str = readline(CLI_PREFIX);
165  ::rl_bind_key('\t', ::rl_complete);
166  if (command_str != nullptr)
167  {
168  command = command_str;
169  free(command_str);
170  }
171 #endif
172 
173  if (!command.empty())
174  {
175  std::size_t nextLineIndex = command.find_first_of("\r\n");
176  if (nextLineIndex != std::string::npos)
177  {
178  if (nextLineIndex == 0)
179  {
180 #if AC_PLATFORM == AC_PLATFORM_WINDOWS
181  PrintCliPrefix();
182 #endif
183  continue;
184  }
185 
186  command.erase(nextLineIndex);
187  }
188 
189  fflush(stdout);
190  sWorld->QueueCliCommand(new CliCommandHolder(nullptr, command.c_str(), &utf8print, &commandFinished));
191 #if AC_PLATFORM != AC_PLATFORM_WINDOWS
192  add_history(command.c_str());
193 #endif
194  }
195  else if (feof(stdin))
196  {
197  World::StopNow(SHUTDOWN_EXIT_CODE);
198  }
199  }
200 }
