40  void LoadFromDB()
41  {
42  uint32 oldMSTime = getMSTime();
43 
44  QueryResult result = CharacterDatabase.Query("SELECT name, crc FROM addons");
45  if (!result)
46  {
47  LOG_INFO("server.loading", ">> Loaded 0 known addons. DB table `addons` is empty!");
48  LOG_INFO("server.loading", " ");
49  return;
50  }
51 
52  uint32 count = 0;
53 
54  do
55  {
56  Field* fields = result->Fetch();
57 
58  std::string name = fields[0].GetString();
59  uint32 crc = fields[1].GetUInt32();
60 
61  m_knownAddons.push_back(SavedAddon(name, crc));
62 
63  ++count;
64  } while (result->NextRow());
65 
66  LOG_INFO("server.loading", ">> Loaded %u known addons in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
67  LOG_INFO("server.loading", " ");
68 
69  oldMSTime = getMSTime();
70  result = CharacterDatabase.Query("SELECT id, name, version, UNIX_TIMESTAMP(timestamp) FROM banned_addons");
71  if (result)
72  {
73  uint32 count2 = 0;
74  uint32 offset = 102;
75 
76  do
77  {
78  Field* fields = result->Fetch();
79 
80  BannedAddon addon;
81  addon.Id = fields[0].GetUInt32() + offset;
82  addon.Timestamp = uint32(fields[3].GetUInt64());
83 
84  std::string name = fields[1].GetString();
85  std::string version = fields[2].GetString();
86 
87  MD5(reinterpret_cast<uint8 const*>(name.c_str()), name.length(), addon.NameMD5);
88  MD5(reinterpret_cast<uint8 const*>(version.c_str()), version.length(), addon.VersionMD5);
89 
90  m_bannedAddons.push_back(addon);
91 
92  ++count2;
93  } while (result->NextRow());
94 
95  LOG_INFO("server.loading", ">> Loaded %u banned addons in %u ms", count2, GetMSTimeDiffToNow(oldMSTime));
96  LOG_INFO("server.loading", " ");
97  }
98  }
