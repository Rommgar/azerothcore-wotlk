92 
93  // Check if arena team is full (Can't have more than type * 2 players)
94  if (GetMembersSize() >= GetType() * 2)
95  return false;
96 
97  // xinef: Get player name and class from player storage or global data storage
98  Player* player = ObjectAccessor::FindConnectedPlayer(playerGuid);
99  if (player)
100  {
101  playerClass = player->getClass();
102  playerName = player->GetName();
103  }
104  else
105  {
106  GlobalPlayerData const* playerData = sWorld->GetGlobalPlayerData(playerGuid.GetCounter());
107  if (!playerData)
108  return false;
109 
110  playerName = playerData->name;
111  playerClass = playerData->playerClass;
112  }
113 
114  if (!sScriptMgr->CanAddMember(this, playerGuid))
115  return false;
116 
117  // Check if player is already in a similar arena team
118  if ((player && player->GetArenaTeamId(GetSlot())) || Player::GetArenaTeamIdFromStorage(playerGuid.GetCounter(), GetSlot()) != 0)
119  {
120  LOG_ERROR("bg.arena", "Arena: Player %s (%s) already has an arena team of type %u", playerName.c_str(), playerGuid.ToString().c_str(), GetType());
121  return false;
122  }
123 
124  // Set player's personal rating
125  uint32 personalRating = 0;
126 
127  if (sWorld->getIntConfig(CONFIG_ARENA_START_PERSONAL_RATING) > 0)
128  personalRating = sWorld->getIntConfig(CONFIG_ARENA_START_PERSONAL_RATING);
129  else if (GetRating() >= 1000)
130  personalRating = 1000;
131 
132  // xinef: zomg! sync query
133  // Try to get player's match maker rating from db and fall back to config setting if not found
134  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_SEL_MATCH_MAKER_RATING);
135  stmt->setUInt32(0, playerGuid.GetCounter());
136  stmt->setUInt8(1, GetSlot());

138 
139  uint16 matchMakerRating;
140  uint16 maxMMR;
141  if (result)
142  {
143  matchMakerRating = (*result)[0].GetUInt16();
144  uint16 Max = (*result)[1].GetUInt16();
145  maxMMR = std::max(Max, matchMakerRating);
146  }
147  else
148  {
149  matchMakerRating = sWorld->getIntConfig(CONFIG_ARENA_START_MATCHMAKER_RATING);
150  maxMMR = matchMakerRating;
151  }
152 
153  // Remove all player signatures from other petitions
154  // This will prevent player from joining too many arena teams and corrupt arena team data integrity
155  Player::RemovePetitionsAndSigns(playerGuid, GetType());
156 
157  // Feed data to the struct
158  ArenaTeamMember newMember;
159  //newMember.Name = playerName;
160  newMember.Guid = playerGuid;
161  newMember.Class = playerClass;
162  newMember.SeasonGames = 0;
163  newMember.WeekGames = 0;
164  newMember.SeasonWins = 0;
165  newMember.WeekWins = 0;
166  newMember.PersonalRating = personalRating;
167  newMember.MatchMakerRating = matchMakerRating;
168  newMember.MaxMMR = maxMMR;
169 
170  Members.push_back(newMember);
171  sWorld->UpdateGlobalPlayerArenaTeam(playerGuid.GetCounter(), GetSlot(), GetId());
172 
173  // Save player's arena team membership to db
174  stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_ARENA_TEAM_MEMBER);
175  stmt->setUInt32(0, TeamId);
176  stmt->setUInt32(1, playerGuid.GetCounter());
177  CharacterDatabase.Execute(stmt);
178 
179  // Inform player if online
180  if (player)
181  {
182  player->SetInArenaTeam(TeamId, GetSlot(), GetType());
183  player->SetArenaTeamIdInvited(0);
184 
185  // Hide promote/remove buttons
186  if (CaptainGuid != playerGuid)
187  player->SetArenaTeamInfoField(GetSlot(), ARENA_TEAM_MEMBER, 1);
188  }
189

219 bool ArenaTeam::LoadMembersFromDB(QueryResult result)
220 {
221  if (!result)
222  return false;
223 
224  bool captainPresentInTeam = false;
225 
226  do
227  {
228  Field* fields = result->Fetch();
229 
230  // Prevent crash if db records are broken when all members in result are already processed and current team doesn't have any members
231  if (!fields)
232  break;
233 
234  uint32 arenaTeamId = fields[0].GetUInt32();
235 
236  // We loaded all members for this arena_team already, break cycle
237  if (arenaTeamId > TeamId)
238  break;
239 
240  ArenaTeamMember newMember;
241  newMember.Guid = ObjectGuid::Create<HighGuid::Player>(fields[1].GetUInt32());
242  newMember.WeekGames = fields[2].GetUInt16();
243  newMember.WeekWins = fields[3].GetUInt16();
244  newMember.SeasonGames = fields[4].GetUInt16();
245  newMember.SeasonWins = fields[5].GetUInt16();
246  //newMember.Name = fields[6].GetString();
247  newMember.Class = fields[7].GetUInt8();
248  newMember.PersonalRating = fields[8].GetUInt16();
249  newMember.MatchMakerRating = fields[9].GetUInt16() > 0 ? fields[9].GetUInt16() : sWorld->getIntConfig(CONFIG_ARENA_START_MATCHMAKER_RATING);
250  newMember.MaxMMR = std::max(fields[10].GetUInt16(), newMember.MatchMakerRating);
251 
252  // Delete member if character information is missing
253  if (fields[6].GetString().empty())
254  {
255  LOG_ERROR("sql.sql", "ArenaTeam %u has member with empty name - probably player %s doesn't exist, deleting him from memberlist!", arenaTeamId, newMember.Guid.ToString().c_str());
256  this->DelMember(newMember.Guid, true);
257  continue;
258  }
259 
260  // Check if team team has a valid captain
261  if (newMember.Guid == GetCaptain())
262  captainPresentInTeam = true;
263 
264  // Put the player in the team
265  Members.push_back(newMember);
266  sWorld->UpdateGlobalPlayerArenaTeam(newMember.Guid.GetCounter(), GetSlot(), GetId());
267  } while (result->NextRow());
268 
269  if (Empty() || !captainPresentInTeam)
270  {
271  // Arena team is empty or captain is not in team, delete from db
272  LOG_DEBUG("bg.battleground", "ArenaTeam %u does not have any members or its captain is not in team, disbanding it...", TeamId);
273  return false;
274  }
275 
276  return true;
277 }
