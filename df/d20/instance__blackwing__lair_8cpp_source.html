175  switch (type)

178  if (state == DONE)
179  {

182  egg->SetPhaseMask(2, true);

184  SetData(DATA_EGG_EVENT, NOT_STARTED);

192  break;
193  case FAIL:
194  _events.ScheduleEvent(EVENT_RESPAWN_NEFARIUS, 15 * 60 * IN_MILLISECONDS); //15min
195  SetBossState(DATA_NEFARIAN, NOT_STARTED);

226  SetData(DATA_EGG_EVENT, DONE);

230  _events.ScheduleEvent(EVENT_RAZOR_PHASE_TWO, 1000);
231  _events.CancelEvent(EVENT_RAZOR_SPAWN);
232  }
233  if (EggEvent == NOT_STARTED)

260  void Update(uint32 diff) override
261  {
262  if (_events.Empty())
263  return;
264 
265  _events.Update(diff);
266 
267  while (uint32 eventId = _events.ExecuteEvent())
268  {
269  switch (eventId)
270  {
271  case EVENT_RAZOR_SPAWN:
272  for (uint8 i = urand(2, 5); i > 0; --i)
273  if (Creature* summon = instance->SummonCreature(Entry[urand(0, 4)], SummonPosition[urand(0, 7)]))
274  summon->AI()->DoZoneInCombat();
275  _events.ScheduleEvent(EVENT_RAZOR_SPAWN, 12000, 17000);
276  break;
277  case EVENT_RAZOR_PHASE_TWO:
278  _events.CancelEvent(EVENT_RAZOR_SPAWN);
279  if (Creature* razor = instance->GetCreature(razorgoreGUID))
280  razor->AI()->DoAction(ACTION_PHASE_TWO);
281  break;
282  case EVENT_RESPAWN_NEFARIUS:
283  if (Creature* nefarius = instance->GetCreature(GetGuidData(DATA_LORD_VICTOR_NEFARIUS)))
284  {
285  nefarius->SetPhaseMask(1, true);
286  nefarius->setActive(true);
287  nefarius->Respawn();
288  nefarius->GetMotionMaster()->MoveTargetedHome();
289  }
290  break;
291  }
292  }
293  }
