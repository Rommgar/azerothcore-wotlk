135  void UpdateAI(uint32 diff) override
136  {
137  if (!UpdateVictim())
138  return;
139 
140  events.Update(diff);
141 
142  while (uint32 eventId = events.ExecuteEvent())
143  {
144  switch (eventId)
145  {
146  case EVENT_DISMEMBER:
147  DoCastVictim(SPELL_DISMEMBER);
148  events.ScheduleEvent(EVENT_DISMEMBER, 5000);
149  break;
150  case EVENT_GATHERING_SPEED:
151  DoCast(me, SPELL_GATHERING_SPEED);
152  events.ScheduleEvent(EVENT_GATHERING_SPEED, 9000);
153  break;
154  case EVENT_FULL_SPEED:
155  DoCast(me, SPELL_FULL_SPEED);
156  break;
157  case EVENT_CREEPING_PLAGUE:
158  DoCast(me, SPELL_CREEPING_PLAGUE);
159  events.ScheduleEvent(EVENT_CREEPING_PLAGUE, 6000);
160  break;
161  case EVENT_RESPAWN_EGG:
162  if (Creature* egg = me->GetMap()->GetCreature(*Eggs.begin()))
163  {
164  egg->Respawn();
165  Eggs.pop_front();
166  }
167  break;
168  default:
169  break;
170  }
171  }
172 
173  if (me->GetHealthPct() < 20.0f && _phase == PHASE_EGG)
174  {
175  DoCast(me, SPELL_BURU_TRANSFORM); // Enrage
176  DoCast(me, SPELL_FULL_SPEED, true);
177  me->RemoveAurasDueToSpell(SPELL_THORNS);
178  _phase = PHASE_TRANSFORM;
179  }
180 
181  DoMeleeAttackIfReady();
182  }
183  private:
184  uint8 _phase;
185  GuidList Eggs;
186  };
