236  void HandlePeriodic(AuraEffect const* aurEff)
237  {
238  PreventDefaultAction();
239  if (aurEff->GetAuraType() == SPELL_AURA_MOD_STAT && (aurEff->GetMiscValue() == STAT_STAMINA || aurEff->GetMiscValue() == STAT_INTELLECT))
240  {
241  int32 currentAmount = aurEff->GetAmount();
242  int32 newAmount = GetEffect(aurEff->GetEffIndex())->CalculateAmount(GetCaster());
243  if (newAmount != currentAmount)
244  {
245  if (aurEff->GetMiscValue() == STAT_STAMINA)
246  {
247  uint32 actStat = GetUnitOwner()->GetHealth();
248  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
249  GetUnitOwner()->SetHealth(std::min<uint32>(GetUnitOwner()->GetMaxHealth(), actStat));
250  }
251  else
252  {
253  uint32 actStat = GetUnitOwner()->GetPower(POWER_MANA);
254  GetEffect(aurEff->GetEffIndex())->ChangeAmount(newAmount, false);
255  GetUnitOwner()->SetPower(POWER_MANA, std::min<uint32>(GetUnitOwner()->GetMaxPower(POWER_MANA), actStat));
256  }
257  }
258  }
259  else
260  GetEffect(aurEff->GetEffIndex())->RecalculateAmount();
261  }
