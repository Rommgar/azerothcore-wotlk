342  void SetData(uint32 type, uint32 data) override
343  {
344  switch (type)
345  {
346  case AREATRIGGER:
347  if (data == AREATRIGGER_DRAGONSPIRE_HALL)
348  {
349  if (GetBossState(DATA_DRAGONSPIRE_ROOM) != DONE)
350  Events.ScheduleEvent(EVENT_DARGONSPIRE_ROOM_STORE, 1000);
351  }
352  break;
353  case DATA_SOLAKAR_FLAMEWREATH:
354  switch(data)
355  {
356  case IN_PROGRESS:
357  if (SolakarState == NOT_STARTED)
358  {
359  Events.ScheduleEvent(EVENT_SOLAKAR_WAVE, 500);
360  }
361  break;
362  case FAIL:
363  for (const auto& creature : SolakarSummons)
364  {
365  creature->RemoveFromWorld();
366  }
367  SolakarSummons.clear();
368  CurrentSolakarWave = 0;
369  SetData(DATA_SOLAKAR_FLAMEWREATH, NOT_STARTED);
370  break;
371  case DONE:
372  break;
373  }
374  SolakarState = data;
375  break;
376  case DATA_UROK_DOOMHOWL:
377  if (data == FAIL)
378  {
379  if (GetBossState(DATA_UROK_DOOMHOWL) != NOT_STARTED)
380  {
381  SetBossState(DATA_UROK_DOOMHOWL, NOT_STARTED);
382  if (GameObject* challenge = instance->GetGameObject(go_urokChallenge))
383  {
384  challenge->Delete();
385  }
386  if (GameObject* pile = instance->GetGameObject(go_urokPile))
387  {
388  pile->SetLootState(GO_READY);
389  pile->Respawn();
390  }
391  for (const auto& circleGUID : go_urokOgreCirles)
392  {
393  if (GameObject* circle = instance->GetGameObject(circleGUID))
394  {
395  circle->Delete();
396  }
397  }
398  for (const auto& mobGUID : UrokMobs)
399  {
400  if (Creature* mob = instance->GetCreature(mobGUID))
401  {
402  mob->DespawnOrUnsummon();
403  }
404  }
405  }
406  }
407  break;
408  default:
409  break;
410  }

524  void Update(uint32 diff) override
525  {
526  Events.Update(diff);
527 
528  while (uint32 eventId = Events.ExecuteEvent())
529  {
530  switch (eventId)
531  {
532  case EVENT_DARGONSPIRE_ROOM_STORE:
533  Dragonspireroomstore();
534  Events.ScheduleEvent(EVENT_DARGONSPIRE_ROOM_CHECK, 3000);
535  break;

537  Dragonspireroomcheck();
538  if ((GetBossState(DATA_DRAGONSPIRE_ROOM) != DONE))

544  {
545  Events.ScheduleEvent(EVENT_SOLAKAR_WAVE, TIMER_SOLAKAR_WAVE);
546  CurrentSolakarWave++;
547  }
548  break;
549  default:
550  break;
551  }
