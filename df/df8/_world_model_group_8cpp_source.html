90 void WorldModelGroup::ReadTriangles()

93  if (!chunk)
94  return;
95 
96  uint32 triangleCount = chunk->Length / 6;
97  ASSERT(triangleCount == TriangleFlags.size() && "triangleCount != TriangleFlags.size()");
98  FILE* stream = chunk->GetStream();
99  Triangles.reserve(triangleCount);
100  for (uint32 i = 0; i < triangleCount; i++)
101  {
102  uint16 v0;
103  uint16 v1;
104  uint16 v2;
105  int count = 0;
106  count += fread(&v0, sizeof(uint16), 1, stream);
107  count += fread(&v1, sizeof(uint16), 1, stream);
108  count += fread(&v2, sizeof(uint16), 1, stream);
109  if (count != 3)
110  printf("WorldModelGroup::ReadMaterials: Error reading data, expected 3, read %d\n", count);
111 
112  Triangles.push_back(Triangle<uint16>(Constants::TRIANGLE_TYPE_WMO, v0, v1, v2));
113  }
