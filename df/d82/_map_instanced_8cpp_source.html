115 Map* MapInstanced::CreateInstanceForPlayer(const uint32 mapId, Player* player)
116 {
117  if (GetId() != mapId || !player)
118  return nullptr;
119 
120  Map* map = nullptr;
121 
122  if (IsBattlegroundOrArena())
123  {
124  // instantiate or find existing bg map for player
125  // the instance id is set in battlegroundid
126  uint32 newInstanceId = player->GetBattlegroundId();
127  if (!newInstanceId)
128  return nullptr;
129 
130  map = sMapMgr->FindMap(mapId, newInstanceId);
131  if (!map)
132  {
133  Battleground* bg = player->GetBattleground(true);
134  if (bg && bg->GetStatus() < STATUS_WAIT_LEAVE)
135  map = CreateBattleground(newInstanceId, bg);
136  else
137  {
138  player->TeleportToEntryPoint();
139  return nullptr;
140  }
141  }
142  }
143  else
144  {
145  Difficulty realdiff = player->GetDifficulty(IsRaid());
146  uint32 destInstId = sInstanceSaveMgr->PlayerGetDestinationInstanceId(player, GetId(), realdiff);
147 
148  if (destInstId)
149  {
150  InstanceSave* pSave = sInstanceSaveMgr->GetInstanceSave(destInstId);
151  ASSERT(pSave); // pussywizard: must exist
152 
153  map = FindInstanceMap(destInstId);
154  if (!map)
155  map = CreateInstance(destInstId, pSave, realdiff);
156  else if (IsSharedDifficultyMap(mapId) && !map->HavePlayers() && map->GetDifficulty() != realdiff)
157  {
158  if (player->isBeingLoaded()) // pussywizard: crashfix (assert(passengers.empty) fail in ~transport), could be added to a transport during loading from db
159  return nullptr;
160 
161  if (!map->AllTransportsEmpty())
162  map->AllTransportsRemovePassengers(); // pussywizard: gameobjects / summons (assert(passengers.empty) fail in ~transport)
163 
164  for (InstancedMaps::iterator i = m_InstancedMaps.begin(); i != m_InstancedMaps.end(); ++i)
165  if (i->first == destInstId)
166  {
167  DestroyInstance(i);
168  map = CreateInstance(destInstId, pSave, realdiff);
169  break;
170  }
171  }
172  }
173  else
174  {
175  uint32 newInstanceId = sMapMgr->GenerateInstanceId();
176  ASSERT(!FindInstanceMap(newInstanceId)); // pussywizard: instance with new id can't exist
177  Difficulty diff = player->GetGroup() ? player->GetGroup()->GetDifficulty(IsRaid()) : player->GetDifficulty(IsRaid());
178  map = CreateInstance(newInstanceId, nullptr, diff);
179  }
180  }
181 
182  return map;
183 }

185 InstanceMap* MapInstanced::CreateInstance(uint32 InstanceId, InstanceSave* save, Difficulty difficulty)
186 {
187  // load/create a map
188  std::lock_guard<std::mutex> guard(Lock);
189 
190  // make sure we have a valid map id
191  const MapEntry* entry = sMapStore.LookupEntry(GetId());
192  if (!entry)
193  {
194  LOG_ERROR("maps", "CreateInstance: no entry for map %d", GetId());
195  ABORT();
196  }
197  const InstanceTemplate* iTemplate = sObjectMgr->GetInstanceTemplate(GetId());
198  if (!iTemplate)
199  {
200  LOG_ERROR("maps", "CreateInstance: no instance template for map %d", GetId());
201  ABORT();
202  }
203 
204  // some instances only have one difficulty
205  GetDownscaledMapDifficultyData(GetId(), difficulty);
206 
207  LOG_DEBUG("maps", "MapInstanced::CreateInstance: %s map instance %d for %d created with difficulty %s", save ? "" : "new ", InstanceId, GetId(), difficulty ? "heroic" : "normal");
208 
209  InstanceMap* map = new InstanceMap(GetId(), InstanceId, difficulty, this);
210  ASSERT(map->IsDungeon());
211 
212  map->LoadRespawnTimes();
213  map->LoadCorpseData();
214 
215  if (save)
216  map->CreateInstanceScript(true, save->GetInstanceData(), save->GetCompletedEncounterMask());
217  else
218  map->CreateInstanceScript(false, "", 0);
219 
220  if (!save) // this is for sure a dungeon (assert above), no need to check here
221  sInstanceSaveMgr->AddInstanceSave(GetId(), InstanceId, difficulty);
222 
223  m_InstancedMaps[InstanceId] = map;
224  return map;
225 }
