141 Object* ObjectAccessor::GetObjectByTypeMask(WorldObject const& p, ObjectGuid const guid, uint32 typemask)
142 {
143  switch (guid.GetHigh())
144  {
145  case HighGuid::Item:
146  if (typemask & TYPEMASK_ITEM && p.GetTypeId() == TYPEID_PLAYER)
147  return ((Player const&)p).GetItemByGuid(guid);
148  break;
149  case HighGuid::Player:
150  if (typemask & TYPEMASK_PLAYER)
151  return GetPlayer(p, guid);
152  break;
153  case HighGuid::Transport:
154  case HighGuid::Mo_Transport:
155  case HighGuid::GameObject:
156  if (typemask & TYPEMASK_GAMEOBJECT)
157  return GetGameObject(p, guid);
158  break;
159  case HighGuid::Unit:
160  case HighGuid::Vehicle:
161  if (typemask & TYPEMASK_UNIT)
162  return GetCreature(p, guid);
163  break;
164  case HighGuid::Pet:
165  if (typemask & TYPEMASK_UNIT)
166  return GetPet(p, guid);
167  break;
168  case HighGuid::DynamicObject:
169  if (typemask & TYPEMASK_DYNAMICOBJECT)
170  return GetDynamicObject(p, guid);
171  break;
172  default:
173  return nullptr;
174  }
175 
176  return nullptr;
