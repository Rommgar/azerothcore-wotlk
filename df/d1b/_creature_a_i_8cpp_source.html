90 void CreatureAI::DoZoneInCombat(Creature* creature /*= nullptr*/, float maxRangeToNearestTarget /*= 250.0f*/)
91 {
92  if (!creature)
93  {
94  creature = me;
95  }
96 
97  if (creature->IsInEvadeMode())
98  {
99  return;
100  }
101 
102  Map* map = creature->GetMap();
103  if (!map->IsDungeon()) //use IsDungeon instead of Instanceable, in case battlegrounds will be instantiated
104  {
105  LOG_ERROR("entities.unit.ai", "DoZoneInCombat call for map that isn't an instance (creature entry = %d)", creature->GetTypeId() == TYPEID_UNIT ? creature->ToCreature()->GetEntry() : 0);
106  return;
107  }
108 
109  Map::PlayerList const& playerList = map->GetPlayers();
110  if (playerList.isEmpty())
111  {
112  return;
113  }
114 
115  for (Map::PlayerList::const_iterator itr = playerList.begin(); itr != playerList.end(); ++itr)
116  {
117  if (Player* player = itr->GetSource())
118  {
119  if (!IsValidCombatTarget(creature, player))
120  {
121  continue;
122  }
123 
124  if (!creature->IsWithinDistInMap(player, maxRangeToNearestTarget))
125  {
126  continue;
127  }
128 
129  creature->SetInCombatWith(player);
130  player->SetInCombatWith(creature);
131 
132  if (creature->CanHaveThreatList())
133  {
134  creature->AddThreat(player, 0.0f);
135  }
136 
137  /* Causes certain things to never leave the threat list (Priest Lightwell, etc):

177 // Distract creature, if player gets too close while stealthed/prowling
178 void CreatureAI::TriggerAlert(Unit const* who) const
179 {
180  // If there's no target, or target isn't a player do nothing
181  if (!who || who->GetTypeId() != TYPEID_PLAYER)
182  return;
183  // If this unit isn't an NPC, is already distracted, is in combat, is confused, stunned or fleeing, do nothing
184  if (me->GetTypeId() != TYPEID_UNIT || me->IsInCombat() || me->HasUnitState(UNIT_STATE_CONFUSED | UNIT_STATE_STUNNED | UNIT_STATE_FLEEING | UNIT_STATE_DISTRACTED))
185  return;
186  // Only alert for hostiles!
187  if (me->IsCivilian() || me->HasReactState(REACT_PASSIVE) || !me->IsHostileTo(who) || !me->_IsTargetAcceptable(who))
188  return;
189  // Send alert sound (if any) for this creature
190  me->SendAIReaction(AI_REACTION_ALERT);
191  // Face the unit (stealthed player) and set distracted state for 5 seconds
192  me->SetFacingTo(me->GetAngle(who->GetPositionX(), who->GetPositionY()));
193  me->StopMoving();
194  me->GetMotionMaster()->MoveDistract(5 * IN_MILLISECONDS);
195 }

197 void CreatureAI::EnterEvadeMode()
198 {
199  if (!_EnterEvadeMode())
200  return;
201 
202  LOG_DEBUG("entities.unit", "Creature %u enters evade mode.", me->GetEntry());
203 
204  if (!me->GetVehicle()) // otherwise me will be in evade mode forever
205  {
206  if (Unit* owner = me->GetCharmerOrOwner())
207  {
208  me->GetMotionMaster()->Clear(false);
209  me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, me->GetFollowAngle(), MOTION_SLOT_ACTIVE);
210  }
211  else
212  {
213  // Required to prevent attacking creatures that are evading and cause them to reenter combat
214  // Does not apply to MoveFollow
215  me->AddUnitState(UNIT_STATE_EVADE);
216  me->GetMotionMaster()->MoveTargetedHome();
217  }
218  }
219 
220  // despawn bosses at reset - only verified tbc/woltk bosses with this reset type - add bosses in last line respectively (dungeon/raid) and increase array limit
221  CreatureTemplate const* cInfo = sObjectMgr->GetCreatureTemplate(me->GetEntry());
222  if (cInfo && cInfo->HasFlagsExtra(CREATURE_FLAG_EXTRA_HARD_RESET))
223  {
224  me->DespawnOnEvade();
225  me->m_Events.AddEvent(new PhasedRespawn(*me), me->m_Events.CalculateTime(20000));
226  }
227  else // bosses will run back to the spawnpoint at reset
228  {
229  Reset();
230  if (me->IsVehicle()) // use the same sequence of addtoworld, aireset may remove all summons!
231  {
232  me->GetVehicleKit()->Reset(true);
233  }
234  }
235 }
