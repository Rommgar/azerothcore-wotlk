31 inline bool overlapSlabs(const float* amin, const float* amax,
32  const float* bmin, const float* bmax,
33  const float px, const float py)
34 {
35  // Check for horizontal overlap.
36  // The segment is shrunken a little so that slabs which touch
37  // at end points are not connected.
38  const float minx = dtMax(amin[0]+px,bmin[0]+px);
39  const float maxx = dtMin(amax[0]-px,bmax[0]-px);
40  if (minx > maxx)
41  return false;
42 
43  // Check vertical overlap.
44  const float ad = (amax[1]-amin[1]) / (amax[0]-amin[0]);
45  const float ak = amin[1] - ad*amin[0];
46  const float bd = (bmax[1]-bmin[1]) / (bmax[0]-bmin[0]);
47  const float bk = bmin[1] - bd*bmin[0];
48  const float aminy = ad*minx + ak;
49  const float amaxy = ad*maxx + ak;
50  const float bminy = bd*minx + bk;
51  const float bmaxy = bd*maxx + bk;
52  const float dmin = bminy - aminy;
53  const float dmax = bmaxy - amaxy;
54 
55  // Crossing segments always overlap.
56  if (dmin*dmax < 0)
57  return true;
58 
59  // Check for overlap at endpoints.
60  const float thr = dtSqr(py*2);
61  if (dmin*dmin <= thr || dmax*dmax <= thr)
62  return true;
63 
64  return false;
65 }

76 static void calcSlabEndPoints(const float* va, const float* vb, float* bmin, float* bmax, const int side)
77 {
78  if (side == 0 || side == 4)
79  {
80  if (va[2] < vb[2])
81  {
82  bmin[0] = va[2];
83  bmin[1] = va[1];
84  bmax[0] = vb[2];
85  bmax[1] = vb[1];
86  }
87  else
88  {
89  bmin[0] = vb[2];
90  bmin[1] = vb[1];
91  bmax[0] = va[2];
92  bmax[1] = va[1];
93  }
94  }
95  else if (side == 2 || side == 6)
96  {
97  if (va[0] < vb[0])
98  {
99  bmin[0] = va[0];
100  bmin[1] = va[1];
101  bmax[0] = vb[0];
102  bmax[1] = vb[1];
103  }
104  else
105  {
106  bmin[0] = vb[0];
107  bmin[1] = vb[1];
108  bmax[0] = va[0];
109  bmax[1] = va[1];
110  }
111  }

241  if (!m_posLookup)

243  memset(m_tiles, 0, sizeof(dtMeshTile)*m_maxTiles);
244  memset(m_posLookup, 0, sizeof(dtMeshTile*)*m_tileLutSize);
245  m_nextFree = 0;
246  for (int i = m_maxTiles-1; i >= 0; --i)
247  {
248  m_tiles[i].salt = 1;
249  m_tiles[i].next = m_nextFree;
250  m_nextFree = &m_tiles[i];
251  }
252 
253  // Init ID generator values.
254 #ifndef DT_POLYREF64
255  m_tileBits = dtIlog2(dtNextPow2((unsigned int)params->maxTiles));
256  m_polyBits = dtIlog2(dtNextPow2((unsigned int)params->maxPolys));
257  // Only allow 31 salt bits, since the salt mask is calculated using 32bit uint and it will overflow.
258  m_saltBits = dtMin((unsigned int)31, 32 - m_tileBits - m_polyBits);
259

261  return DT_FAILURE | DT_INVALID_PARAM;
262 #endif
263 
264  return DT_SUCCESS;

300 int dtNavMesh::findConnectingPolys(const float* va, const float* vb,
301  const dtMeshTile* tile, int side,
302  dtPolyRef* con, float* conarea, int maxcon) const
303 {
304  if (!tile) return 0;
305 
306  float amin[2], amax[2];
307  calcSlabEndPoints(va, vb, amin, amax, side);
308  const float apos = getSlabCoord(va, side);
309 
310  // Remove links pointing to 'side' and compact the links array.
311  float bmin[2], bmax[2];
312  unsigned short m = DT_EXT_LINK | (unsigned short)side;
313  int n = 0;
314 
315  dtPolyRef base = getPolyRefBase(tile);
316 
317  for (int i = 0; i < tile->header->polyCount; ++i)
318  {
319  dtPoly* poly = &tile->polys[i];
320  const int nv = poly->vertCount;
321  for (int j = 0; j < nv; ++j)
322  {
323  // Skip edges which do not point to the right side.
324  if (poly->neis[j] != m) continue;
325 
326  const float* vc = &tile->verts[poly->verts[j]*3];
327  const float* vd = &tile->verts[poly->verts[(j+1) % nv]*3];
328  const float bpos = getSlabCoord(vc, side);
329 
330  // Segments are not close enough.
331  if (dtAbs(apos-bpos) > 0.01f)
332  continue;
333 
334  // Check if the segments touch.
335  calcSlabEndPoints(vc,vd, bmin,bmax, side);
336 
337  if (!overlapSlabs(amin,amax, bmin,bmax, 0.01f, tile->header->walkableClimb)) continue;
338 
339  // Add return value.
340  if (n < maxcon)
341  {
342  conarea[n*2+0] = dtMax(amin[0], bmin[0]);
343  conarea[n*2+1] = dtMin(amax[0], bmax[0]);
344  con[n] = base | (dtPolyRef)i;
345  n++;
346  }
347  break;
348  }
349  }
350  return n;
351 }

353 void dtNavMesh::unconnectLinks(dtMeshTile* tile, dtMeshTile* target)
354 {
355  if (!tile || !target) return;
356 
357  const unsigned int targetNum = decodePolyIdTile(getTileRef(target));
358 
359  for (int i = 0; i < tile->header->polyCount; ++i)
360  {
361  dtPoly* poly = &tile->polys[i];
362  unsigned int j = poly->firstLink;
363  unsigned int pj = DT_NULL_LINK;
364  while (j != DT_NULL_LINK)
365  {
366  if (decodePolyIdTile(tile->links[j].ref) == targetNum)
367  {
368  // Remove link.
369  unsigned int nj = tile->links[j].next;
370  if (pj == DT_NULL_LINK)
371  poly->firstLink = nj;
372  else
373  tile->links[pj].next = nj;
374  freeLink(tile, j);
375  j = nj;
376  }
377  else
378  {
379  // Advance
380  pj = j;
381  j = tile->links[j].next;
382  }
383  }
384  }

387 void dtNavMesh::connectExtLinks(dtMeshTile* tile, dtMeshTile* target, int side)
388 {
389  if (!tile) return;
390 
391  // Connect border links.
392  for (int i = 0; i < tile->header->polyCount; ++i)
393  {
394  dtPoly* poly = &tile->polys[i];
395 
396  // Create new links.
397 // unsigned short m = DT_EXT_LINK | (unsigned short)side;
398 
399  const int nv = poly->vertCount;
400  for (int j = 0; j < nv; ++j)
401  {
402  // Skip non-portal edges.
403  if ((poly->neis[j] & DT_EXT_LINK) == 0)
404  continue;
405 
406  const int dir = (int)(poly->neis[j] & 0xff);
407  if (side != -1 && dir != side)
408  continue;
409 
410  // Create new links
411  const float* va = &tile->verts[poly->verts[j]*3];
412  const float* vb = &tile->verts[poly->verts[(j+1) % nv]*3];
413  dtPolyRef nei[4];
414  float neia[4*2];
415  int nnei = findConnectingPolys(va,vb, target, dtOppositeTile(dir), nei,neia,4);
416  for (int k = 0; k < nnei; ++k)
417  {
418  unsigned int idx = allocLink(tile);
419  if (idx != DT_NULL_LINK)
420  {
421  dtLink* link = &tile->links[idx];
422  link->ref = nei[k];
423  link->edge = (unsigned char)j;
424  link->side = (unsigned char)dir;
425 
426  link->next = poly->firstLink;
427  poly->firstLink = idx;
428 
429  // Compress portal limits to a byte value.
430  if (dir == 0 || dir == 4)
431  {
432  float tmin = (neia[k*2+0]-va[2]) / (vb[2]-va[2]);
433  float tmax = (neia[k*2+1]-va[2]) / (vb[2]-va[2]);
434  if (tmin > tmax)
435  dtSwap(tmin,tmax);
436  link->bmin = (unsigned char)(dtClamp(tmin, 0.0f, 1.0f)*255.0f);
437  link->bmax = (unsigned char)(dtClamp(tmax, 0.0f, 1.0f)*255.0f);
438  }
439  else if (dir == 2 || dir == 6)
440  {
441  float tmin = (neia[k*2+0]-va[0]) / (vb[0]-va[0]);
442  float tmax = (neia[k*2+1]-va[0]) / (vb[0]-va[0]);
443  if (tmin > tmax)
444  dtSwap(tmin,tmax);
445  link->bmin = (unsigned char)(dtClamp(tmin, 0.0f, 1.0f)*255.0f);
446  link->bmax = (unsigned char)(dtClamp(tmax, 0.0f, 1.0f)*255.0f);
447  }
448  }
449  }
450  }
451  }

454 void dtNavMesh::connectExtOffMeshLinks(dtMeshTile* tile, dtMeshTile* target, int side)
455 {
456  if (!tile) return;
457 
458  // Connect off-mesh links.
459  // We are interested on links which land from target tile to this tile.
460  const unsigned char oppositeSide = (side == -1) ? 0xff : (unsigned char)dtOppositeTile(side);
461 
462  for (int i = 0; i < target->header->offMeshConCount; ++i)
463  {
464  dtOffMeshConnection* targetCon = &target->offMeshCons[i];
465  if (targetCon->side != oppositeSide)
466  continue;
467 
468  dtPoly* targetPoly = &target->polys[targetCon->poly];
469  // Skip off-mesh connections which start location could not be connected at all.
470  if (targetPoly->firstLink == DT_NULL_LINK)
471  continue;
472 
473  const float halfExtents[3] = { targetCon->rad, target->header->walkableClimb, targetCon->rad };
474 
475  // Find polygon to connect to.
476  const float* p = &targetCon->pos[3];
477  float nearestPt[3];
478  dtPolyRef ref = findNearestPolyInTile(tile, p, halfExtents, nearestPt);
479  if (!ref)
480  continue;
481  // findNearestPoly may return too optimistic results, further check to make sure.
482  if (dtSqr(nearestPt[0]-p[0])+dtSqr(nearestPt[2]-p[2]) > dtSqr(targetCon->rad))
483  continue;
484  // Make sure the location is on current mesh.
485  float* v = &target->verts[targetPoly->verts[1]*3];
486  dtVcopy(v, nearestPt);
487 
488  // Link off-mesh connection to target poly.
489  unsigned int idx = allocLink(target);
490  if (idx != DT_NULL_LINK)
491  {
492  dtLink* link = &target->links[idx];
493  link->ref = ref;
494  link->edge = (unsigned char)1;
495  link->side = oppositeSide;
496  link->bmin = link->bmax = 0;
497  // Add to linked list.
498  link->next = targetPoly->firstLink;
499  targetPoly->firstLink = idx;
500  }
501 
502  // Link target poly to off-mesh connection.
503  if (targetCon->flags & DT_OFFMESH_CON_BIDIR)
504  {
505  unsigned int tidx = allocLink(tile);
506  if (tidx != DT_NULL_LINK)
507  {
508  const unsigned short landPolyIdx = (unsigned short)decodePolyIdPoly(ref);
509  dtPoly* landPoly = &tile->polys[landPolyIdx];
510  dtLink* link = &tile->links[tidx];
511  link->ref = getPolyRefBase(target) | (dtPolyRef)(targetCon->poly);
512  link->edge = 0xff;
513  link->side = (unsigned char)(side == -1 ? 0xff : side);
514  link->bmin = link->bmax = 0;
515  // Add to linked list.
516  link->next = landPoly->firstLink;
517  landPoly->firstLink = tidx;
518  }
519  }
520  }

621  template<bool onlyBoundary>
622  void closestPointOnDetailEdges(const dtMeshTile* tile, const dtPoly* poly, const float* pos, float* closest)
623  {
624  const unsigned int ip = (unsigned int)(poly - tile->polys);
625  const dtPolyDetail* pd = &tile->detailMeshes[ip];
626 
627  float dmin = FLT_MAX;
628  float tmin = 0;
629  const float* pmin = 0;
630  const float* pmax = 0;
631 
632  for (int i = 0; i < pd->triCount; i++)
633  {
634  const unsigned char* tris = &tile->detailTris[(pd->triBase + i) * 4];
635  const int ANY_BOUNDARY_EDGE =
636  (DT_DETAIL_EDGE_BOUNDARY << 0) |
637  (DT_DETAIL_EDGE_BOUNDARY << 2) |
638  (DT_DETAIL_EDGE_BOUNDARY << 4);
639  if (onlyBoundary && (tris[3] & ANY_BOUNDARY_EDGE) == 0)
640  continue;
641 
642  const float* v[3];
643  for (int j = 0; j < 3; ++j)
644  {
645  if (tris[j] < poly->vertCount)
646  v[j] = &tile->verts[poly->verts[tris[j]] * 3];
647  else
648  v[j] = &tile->detailVerts[(pd->vertBase + (tris[j] - poly->vertCount)) * 3];
649  }
650 
651  for (int k = 0, j = 2; k < 3; j = k++)
652  {
653  if ((dtGetDetailTriEdgeFlags(tris[3], j) & DT_DETAIL_EDGE_BOUNDARY) == 0 &&
654  (onlyBoundary || tris[j] < tris[k]))
655  {
656  // Only looking at boundary edges and this is internal, or
657  // this is an inner edge that we will see again or have already seen.
658  continue;
659  }
660 
661  float t;
662  float d = dtDistancePtSegSqr2D(pos, v[j], v[k], t);
663  if (d < dmin)
664  {
665  dmin = d;
666  tmin = t;
667  pmin = v[j];
668  pmax = v[k];
669  }
670  }
671  }
672 
673  dtVlerp(closest, pmin, pmax, tmin);
674  }

677 bool dtNavMesh::getPolyHeight(const dtMeshTile* tile, const dtPoly* poly, const float* pos, float* height) const
678 {
679  // Off-mesh connections do not have detail polys and getting height
680  // over them does not make sense.
681  if (poly->getType() == DT_POLYTYPE_OFFMESH_CONNECTION)
682  return false;
683 
684  const unsigned int ip = (unsigned int)(poly - tile->polys);
685  const dtPolyDetail* pd = &tile->detailMeshes[ip];
686 
687  float verts[DT_VERTS_PER_POLYGON*3];
688  const int nv = poly->vertCount;
689  for (int i = 0; i < nv; ++i)
690  dtVcopy(&verts[i*3], &tile->verts[poly->verts[i]*3]);
691 
692  if (!dtPointInPolygon(pos, verts, nv))
693  return false;
694 
695  if (!height)
696  return true;
697 
698  // Find height at the location.
699  for (int j = 0; j < pd->triCount; ++j)
700  {
701  const unsigned char* t = &tile->detailTris[(pd->triBase+j)*4];
702  const float* v[3];
703  for (int k = 0; k < 3; ++k)
704  {
705  if (t[k] < poly->vertCount)
706  v[k] = &tile->verts[poly->verts[t[k]]*3];
707  else
708  v[k] = &tile->detailVerts[(pd->vertBase+(t[k]-poly->vertCount))*3];
709  }
710  float h;
711  if (dtClosestHeightPointTriangle(pos, v[0], v[1], v[2], h))
712  {
713  *height = h;
714  return true;
715  }
716  }
717 
718  // If all triangle checks failed above (can happen with degenerate triangles
719  // or larger floating point values) the point is on an edge, so just select
720  // closest. This should almost never happen so the extra iteration here is
721  // ok.
722  float closest[3];
723  closestPointOnDetailEdges<false>(tile, poly, pos, closest);
724  *height = closest[1];
725  return true;
726 }

728 void dtNavMesh::closestPointOnPoly(dtPolyRef ref, const float* pos, float* closest, bool* posOverPoly) const
729 {
730  const dtMeshTile* tile = 0;

732  getTileAndPolyByRefUnsafe(ref, &tile, &poly);
733 
734  dtVcopy(closest, pos);
735  if (getPolyHeight(tile, poly, pos, &closest[1]))
736  {
737  if (posOverPoly)
738  *posOverPoly = true;
739  return;
740  }
741 
742  if (posOverPoly)
743  *posOverPoly = false;
744 
745  // Off-mesh connections don't have detail polygons.
746  if (poly->getType() == DT_POLYTYPE_OFFMESH_CONNECTION)
747  {
748  const float* v0 = &tile->verts[poly->verts[0]*3];
749  const float* v1 = &tile->verts[poly->verts[1]*3];

751  dtDistancePtSegSqr2D(pos, v0, v1, t);
752  dtVlerp(closest, v0, v1, t);

757  closestPointOnDetailEdges<true>(tile, poly, pos, closest);

760 dtPolyRef dtNavMesh::findNearestPolyInTile(const dtMeshTile* tile,
761  const float* center, const float* halfExtents,
762  float* nearestPt) const
763 {
764  float bmin[3], bmax[3];
765  dtVsub(bmin, center, halfExtents);
766  dtVadd(bmax, center, halfExtents);
767 
768  // Get nearby polygons from proximity grid.
769  dtPolyRef polys[128];
770  int polyCount = queryPolygonsInTile(tile, bmin, bmax, polys, 128);
771 
772  // Find nearest polygon amongst the nearby polygons.
773  dtPolyRef nearest = 0;
774  float nearestDistanceSqr = FLT_MAX;
775  for (int i = 0; i < polyCount; ++i)
776  {
777  dtPolyRef ref = polys[i];
778  float closestPtPoly[3];
779  float diff[3];
780  bool posOverPoly = false;
781  float d;
782  closestPointOnPoly(ref, center, closestPtPoly, &posOverPoly);
783 
784  // If a point is directly over a polygon and closer than
785  // climb height, favor that instead of straight line nearest point.
786  dtVsub(diff, center, closestPtPoly);
787  if (posOverPoly)
788  {
789  d = dtAbs(diff[1]) - tile->header->walkableClimb;
790  d = d > 0 ? d*d : 0;
791  }
792  else
793  {
794  d = dtVlenSqr(diff);
795  }
796 
797  if (d < nearestDistanceSqr)
798  {
799  dtVcopy(nearestPt, closestPtPoly);
800  nearestDistanceSqr = d;
801  nearest = ref;
802  }
803  }
804 
805  return nearest;
806 }

808 int dtNavMesh::queryPolygonsInTile(const dtMeshTile* tile, const float* qmin, const float* qmax,
809  dtPolyRef* polys, const int maxPolys) const
810 {
811  if (tile->bvTree)
812  {
813  const dtBVNode* node = &tile->bvTree[0];
814  const dtBVNode* end = &tile->bvTree[tile->header->bvNodeCount];
815  const float* tbmin = tile->header->bmin;
816  const float* tbmax = tile->header->bmax;
817  const float qfac = tile->header->bvQuantFactor;
818 
819  // Calculate quantized box
820  unsigned short bmin[3], bmax[3];
821  // dtClamp query box to world box.
822  float minx = dtClamp(qmin[0], tbmin[0], tbmax[0]) - tbmin[0];
823  float miny = dtClamp(qmin[1], tbmin[1], tbmax[1]) - tbmin[1];
824  float minz = dtClamp(qmin[2], tbmin[2], tbmax[2]) - tbmin[2];
825  float maxx = dtClamp(qmax[0], tbmin[0], tbmax[0]) - tbmin[0];
826  float maxy = dtClamp(qmax[1], tbmin[1], tbmax[1]) - tbmin[1];
827  float maxz = dtClamp(qmax[2], tbmin[2], tbmax[2]) - tbmin[2];
828  // Quantize
829  bmin[0] = (unsigned short)(qfac * minx) & 0xfffe;
830  bmin[1] = (unsigned short)(qfac * miny) & 0xfffe;
831  bmin[2] = (unsigned short)(qfac * minz) & 0xfffe;
832  bmax[0] = (unsigned short)(qfac * maxx + 1) | 1;
833  bmax[1] = (unsigned short)(qfac * maxy + 1) | 1;
834  bmax[2] = (unsigned short)(qfac * maxz + 1) | 1;
835 
836  // Traverse tree
837  dtPolyRef base = getPolyRefBase(tile);
838  int n = 0;
839  while (node < end)
840  {
841  const bool overlap = dtOverlapQuantBounds(bmin, bmax, node->bmin, node->bmax);
842  const bool isLeafNode = node->i >= 0;
843 
844  if (isLeafNode && overlap)
845  {
846  if (n < maxPolys)
847  polys[n++] = base | (dtPolyRef)node->i;
848  }
849 
850  if (overlap || isLeafNode)
851  node++;
852  else
853  {
854  const int escapeIndex = -node->i;
855  node += escapeIndex;
856  }
857  }
858 
859  return n;
860  }
861  else
862  {
863  float bmin[3], bmax[3];
864  int n = 0;
865  dtPolyRef base = getPolyRefBase(tile);
866  for (int i = 0; i < tile->header->polyCount; ++i)
867  {
868  dtPoly* p = &tile->polys[i];
869  // Do not return off-mesh connection polygons.
870  if (p->getType() == DT_POLYTYPE_OFFMESH_CONNECTION)
871  continue;
872  // Calc polygon bounds.
873  const float* v = &tile->verts[p->verts[0]*3];
874  dtVcopy(bmin, v);
875  dtVcopy(bmax, v);
876  for (int j = 1; j < p->vertCount; ++j)
877  {
878  v = &tile->verts[p->verts[j]*3];
879  dtVmin(bmin, v);
880  dtVmax(bmax, v);
881  }
882  if (dtOverlapBounds(qmin,qmax, bmin,bmax))
883  {
884  if (n < maxPolys)
885  polys[n++] = base | (dtPolyRef)i;
886  }
887  }
888  return n;
889  }
890 }

908 dtStatus dtNavMesh::addTile(unsigned char* data, int dataSize, int flags,
909  dtTileRef lastRef, dtTileRef* result)
910 {
911  // Make sure the data is in right format.
912  dtMeshHeader* header = (dtMeshHeader*)data;
913  if (header->magic != DT_NAVMESH_MAGIC)
914  return DT_FAILURE | DT_WRONG_MAGIC;
915  if (header->version != DT_NAVMESH_VERSION)
916  return DT_FAILURE | DT_WRONG_VERSION;
917 
918  // Make sure the location is free.
919  if (getTileAt(header->x, header->y, header->layer))
920  return DT_FAILURE | DT_ALREADY_OCCUPIED;
921 
922  // Allocate a tile.
923  dtMeshTile* tile = 0;
924  if (!lastRef)
925  {
926  if (m_nextFree)
927  {
928  tile = m_nextFree;
929  m_nextFree = tile->next;
930  tile->next = 0;
931  }
932  }
933  else
934  {
935  // Try to relocate the tile to specific index with same salt.
936  int tileIndex = (int)decodePolyIdTile((dtPolyRef)lastRef);
937  if (tileIndex >= m_maxTiles)
938  return DT_FAILURE | DT_OUT_OF_MEMORY;
939  // Try to find the specific tile id from the free list.
940  dtMeshTile* target = &m_tiles[tileIndex];
941  dtMeshTile* prev = 0;
942  tile = m_nextFree;
943  while (tile && tile != target)
944  {
945  prev = tile;
946  tile = tile->next;
947  }
948  // Could not find the correct location.
949  if (tile != target)
950  return DT_FAILURE | DT_OUT_OF_MEMORY;
951  // Remove from freelist
952  if (!prev)
953  m_nextFree = tile->next;
954  else
955  prev->next = tile->next;
956 
957  // Restore salt.
958  tile->salt = decodePolyIdSalt((dtPolyRef)lastRef);
959  }
960 
961  // Make sure we could allocate a tile.
962  if (!tile)
963  return DT_FAILURE | DT_OUT_OF_MEMORY;
964 
965  // Insert tile into the position lut.
966  int h = computeTileHash(header->x, header->y, m_tileLutMask);
967  tile->next = m_posLookup[h];
968  m_posLookup[h] = tile;
969 
970  // Patch header pointers.
971  const int headerSize = dtAlign4(sizeof(dtMeshHeader));
972  const int vertsSize = dtAlign4(sizeof(float)*3*header->vertCount);
973  const int polysSize = dtAlign4(sizeof(dtPoly)*header->polyCount);
974  const int linksSize = dtAlign4(sizeof(dtLink)*(header->maxLinkCount));
975  const int detailMeshesSize = dtAlign4(sizeof(dtPolyDetail)*header->detailMeshCount);
976  const int detailVertsSize = dtAlign4(sizeof(float)*3*header->detailVertCount);
977  const int detailTrisSize = dtAlign4(sizeof(unsigned char)*4*header->detailTriCount);
978  const int bvtreeSize = dtAlign4(sizeof(dtBVNode)*header->bvNodeCount);
979  const int offMeshLinksSize = dtAlign4(sizeof(dtOffMeshConnection)*header->offMeshConCount);
980 
981  unsigned char* d = data + headerSize;
982  tile->verts = dtGetThenAdvanceBufferPointer<float>(d, vertsSize);
983  tile->polys = dtGetThenAdvanceBufferPointer<dtPoly>(d, polysSize);
984  tile->links = dtGetThenAdvanceBufferPointer<dtLink>(d, linksSize);
985  tile->detailMeshes = dtGetThenAdvanceBufferPointer<dtPolyDetail>(d, detailMeshesSize);
986  tile->detailVerts = dtGetThenAdvanceBufferPointer<float>(d, detailVertsSize);
987  tile->detailTris = dtGetThenAdvanceBufferPointer<unsigned char>(d, detailTrisSize);
988  tile->bvTree = dtGetThenAdvanceBufferPointer<dtBVNode>(d, bvtreeSize);
989  tile->offMeshCons = dtGetThenAdvanceBufferPointer<dtOffMeshConnection>(d, offMeshLinksSize);
990 
991  // If there are no items in the bvtree, reset the tree pointer.
992  if (!bvtreeSize)
993  tile->bvTree = 0;
994 
995  // Build links freelist
996  tile->linksFreeList = 0;
997  tile->links[header->maxLinkCount-1].next = DT_NULL_LINK;
998  for (int i = 0; i < header->maxLinkCount-1; ++i)
999  tile->links[i].next = i+1;
1000 
1001  // Init tile.
1002  tile->header = header;
1003  tile->data = data;
1004  tile->dataSize = dataSize;
1005  tile->flags = flags;
1006 
1007  connectIntLinks(tile);
1008 
1009  // Base off-mesh connections to their starting polygons and connect connections inside the tile.
1010  baseOffMeshLinks(tile);
1011  connectExtOffMeshLinks(tile, tile, -1);
1012 
1013  // Create connections with neighbour tiles.
1014  static const int MAX_NEIS = 32;
1015  dtMeshTile* neis[MAX_NEIS];
1016  int nneis;
1017 
1018  // Connect with layers in current tile.
1019  nneis = getTilesAt(header->x, header->y, neis, MAX_NEIS);
1020  for (int j = 0; j < nneis; ++j)
1021  {
1022  if (neis[j] == tile)
1023  continue;
1024 
1025  connectExtLinks(tile, neis[j], -1);
1026  connectExtLinks(neis[j], tile, -1);
1027  connectExtOffMeshLinks(tile, neis[j], -1);
1028  connectExtOffMeshLinks(neis[j], tile, -1);
1029  }
1030 
1031  // Connect with neighbour tiles.
1032  for (int i = 0; i < 8; ++i)
1033  {
1034  nneis = getNeighbourTilesAt(header->x, header->y, i, neis, MAX_NEIS);
1035  for (int j = 0; j < nneis; ++j)
1036  {
1037  connectExtLinks(tile, neis[j], i);
1038  connectExtLinks(neis[j], tile, dtOppositeTile(i));
1039  connectExtOffMeshLinks(tile, neis[j], i);
1040  connectExtOffMeshLinks(neis[j], tile, dtOppositeTile(i));
1041  }
1042  }
1043 
1044  if (result)
1045  *result = getTileRef(tile);
1046 
1047  return DT_SUCCESS;
1048 }
