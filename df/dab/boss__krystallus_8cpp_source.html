111  void UpdateAI(uint32 diff) override
112  {
113  if (!UpdateVictim())
114  return;
115 
116  events.Update(diff);
117 
118  if (me->HasUnitState(UNIT_STATE_CASTING))
119  return;
120 
121  switch (events.ExecuteEvent())
122  {
123  case EVENT_BOULDER:
124  {
125  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true, 0))
126  me->CastSpell(target, DUNGEON_MODE(BOULDER_TOSS, BOULDER_TOSS_H), false);
127 
128  events.RepeatEvent(5000 + rand() % 2000);
129  break;
130  }
131  case EVENT_GROUND_SPIKE:
132  {
133  me->CastSpell(me->GetVictim(), GROUND_SPIKE_H, false); // current enemy target
134  events.RepeatEvent(8000 + rand() % 3000);
135  break;
136  }
137  case EVENT_STOMP:
138  {
139  me->CastSpell(me, DUNGEON_MODE(STOMP, STOMP_H), false);
140  events.RepeatEvent(13000 + rand() % 5000);
141  break;
142  }
143  case EVENT_GROUND_SLAM:
144  {
145  events.RepeatEvent(10000 + rand() % 3000);
146  me->CastSpell(me->GetVictim(), GROUND_SLAM, true);
147  events.DelayEvents(10000);
148  events.RescheduleEvent(EVENT_SHATTER, 8000);
149  break;
150  }
151  case EVENT_SHATTER:
152  {
153  me->CastSpell((Unit*)nullptr, DUNGEON_MODE(SHATTER, SHATTER_H), false);
154  Talk(SAY_SHATTER);
155  events.RescheduleEvent(EVENT_REMOVE_STONED, 1500);
156  break;
157  }
158  case EVENT_REMOVE_STONED:
159  {
160  RemoveStonedEffect();
161  break;
162  }
163  }
164 
165  DoMeleeAttackIfReady();
166  }
