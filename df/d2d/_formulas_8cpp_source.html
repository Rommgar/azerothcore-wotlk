24 uint32 Acore::XP::BaseGain(uint8 pl_level, uint8 mob_level, ContentLevels content)
25 {
26  uint32 baseGain;
27  uint32 nBaseExp;
28 
29  switch (content)
30  {
31  case CONTENT_1_60:
32  nBaseExp = 45;
33  break;
34  case CONTENT_61_70:
35  nBaseExp = 235;
36  break;
37  case CONTENT_71_80:
38  nBaseExp = 580;
39  break;
40  default:
41  LOG_ERROR("misc", "BaseGain: Unsupported content level %u", content);
42  nBaseExp = 45;
43  break;
44  }
45 
46  if (mob_level >= pl_level)
47  {
48  uint8 nLevelDiff = mob_level - pl_level;
49  if (nLevelDiff > 4)
50  nLevelDiff = 4;
51 
52  baseGain = ((pl_level * 5 + nBaseExp) * (20 + nLevelDiff) / 10 + 1) / 2;
53  }
54  else
55  {
56  uint8 gray_level = GetGrayLevel(pl_level);
57  if (mob_level > gray_level)
58  {
59  uint8 ZD = GetZeroDifference(pl_level);
60  baseGain = (pl_level * 5 + nBaseExp) * (ZD + mob_level - pl_level) / ZD;
61  }
62  else
63  baseGain = 0;
64  }
65 
66  //sScriptMgr->OnBaseGainCalculation(baseGain, pl_level, mob_level, content); // pussywizard: optimization
67  return baseGain;
68 }

70 uint32 Acore::XP::Gain(Player* player, Unit* unit, bool isBattleGround /*= false*/)
71 {
72  Creature* creature = unit->ToCreature();
73  uint32 gain = 0;
74 
75  if (!creature || (!creature->IsTotem() && !creature->IsPet() && !creature->IsCritter() &&
76  !(creature->GetCreatureTemplate()->flags_extra & CREATURE_FLAG_EXTRA_NO_XP)))
77  {
78  float xpMod = 1.0f;
79 
80  gain = BaseGain(player->getLevel(), unit->getLevel(), GetContentLevelsForMapAndZone(unit->GetMapId(), unit->GetZoneId()));
81 
82  if (gain && creature)
83  {
84  if (creature->isElite())
85  {
86  // Elites in instances have a 2.75x XP bonus instead of the regular 2x world bonus.
87  if (unit->GetMap() && unit->GetMap()->IsDungeon())
88  xpMod *= 2.75f;
89  else
90  xpMod *= 2.0f;
91  }
92 
93  xpMod *= creature->GetCreatureTemplate()->ModExperience;
94  }
95 
96  xpMod *= isBattleGround ? sWorld->getRate(RATE_XP_BG_KILL) : sWorld->getRate(RATE_XP_KILL);
97 
98  // if players dealt less than 50% of the damage and were credited anyway (due to CREATURE_FLAG_EXTRA_NO_PLAYER_DAMAGE_REQ), scale XP gained appropriately (linear scaling)
99  if (creature && creature->m_PlayerDamageReq)
100  {
101  xpMod *= 1.0f - 2.0f * creature->m_PlayerDamageReq / creature->GetMaxHealth();
102  }
103 
104  gain = uint32(gain * xpMod);
105  }
106 
107  //sScriptMgr->OnGainCalculation(gain, player, u); // pussywizard: optimization
108  return gain;
109 }
