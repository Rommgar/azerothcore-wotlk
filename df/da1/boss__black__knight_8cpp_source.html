137  void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
138  {
139  if (me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE))
140  {
141  damage = 0;
142  return;
143  }
144 
145  if( Phase < 3 && damage >= me->GetHealth() )
146  {
147  damage = 0;
148  me->SetHealth(me->GetMaxHealth());
149  events.Reset();
150  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
151  me->RemoveAllAuras();
152  me->SetControlled(true, UNIT_STATE_STUNNED);
153  me->CastSpell(me, SPELL_BK_GHOUL_EXPLODE, true);
154  summons.clear();
155 
156  me->CastSpell(me, SPELL_BK_FEIGN_DEATH, true);
157  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PREVENT_EMOTES_FROM_CHAT_TEXT);
158  me->SetFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);
159  me->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
160  me->AddUnitState(UNIT_STATE_DIED);
161  }
162  }

164  void DoAction(int32 param) override
165  {
166  if( param == -1 )
167  {
168  summons.DespawnAll();
169  }
170  else if( param == 1 )
171  {
172  if( !pInstance )
173  return;
174 
175  pInstance->SetData(BOSS_BLACK_KNIGHT, IN_PROGRESS);
176  Talk(TEXT_BK_AGGRO);
177  me->CastSpell((Unit*)nullptr, (pInstance->GetData(DATA_TEAMID_IN_INSTANCE) == TEAM_HORDE ? SPELL_RAISE_DEAD_JAEREN : SPELL_RAISE_DEAD_ARELAS), false);
178  if( Creature* announcer = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_ANNOUNCER)) )
179  announcer->DespawnOrUnsummon();
180 
181  events.Reset();
182  events.ScheduleEvent(EVENT_ANNOUNCER_SAY_ZOMBIE, 2500);
183  events.ScheduleEvent(EVENT_SPELL_PLAGUE_STRIKE, urand(7000, 9000));
184  events.ScheduleEvent(EVENT_SPELL_ICY_TOUCH, urand(3500, 7000));
185  events.ScheduleEvent(EVENT_SPELL_DEATH_RESPITE, urand(13000, 15000));
186  events.ScheduleEvent(EVENT_SPELL_OBLITERATE, urand(11000, 19000));
187  }
188  }

190  void SpellHitTarget(Unit* /*target*/, const SpellInfo* spell) override
191  {
192  switch( spell->Id )
193  {
194  case SPELL_BLACK_KNIGHT_RES:

198

200  me->RemoveFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_FEIGN_DEATH);

203 
204  ++Phase;
205 
206  switch( Phase )
207  {
208  case 2:
209  me->SetDisplayId(MODEL_SKELETON);
210  Talk(TEXT_BK_SKELETON_RES);
211  me->CastSpell(me, SPELL_ARMY_DEAD, false);
212 
213  events.Reset();
214  events.ScheduleEvent(EVENT_SPELL_PLAGUE_STRIKE, urand(7000, 9000));
215  events.ScheduleEvent(EVENT_SPELL_ICY_TOUCH, urand(3500, 7000));
216  events.ScheduleEvent(EVENT_SPELL_OBLITERATE, urand(11000, 19000));
217  events.ScheduleEvent(EVENT_SPELL_DESECRATION, urand(2000, 3000));
218  break;
219  case 3:
220  me->SetDisplayId(MODEL_GHOST);
221  Talk(TEXT_BK_GHOST_RES);
222 
223  events.Reset();
224  events.ScheduleEvent(EVENT_SPELL_DEATH_BITE, 2000);
225  events.ScheduleEvent(EVENT_SPELL_MARKED_DEATH, 1000);
226  break;
227  default:
228  EnterEvadeMode();
229  break;
230  }
231  break;
232  }

235  void UpdateAI(uint32 diff) override
236  {
237  if( !UpdateVictim() )
238  return;
239 
240  events.Update(diff);
241 
242  if( me->HasUnitState(UNIT_STATE_CASTING) )
243  return;
244 
245  switch( events.ExecuteEvent() )
246  {
247  case 0:
248  break;
249  case EVENT_ANNOUNCER_SAY_ZOMBIE:
250  if( pInstance && !summons.empty() )
251  if( Creature* ghoul = pInstance->instance->GetCreature(*summons.begin()) )
252  ghoul->Yell("[Zombie] .... . Brains ....", LANG_UNIVERSAL);
253 
254  break;
255  case EVENT_SPELL_PLAGUE_STRIKE:
256  if( me->GetVictim() )
257  me->CastSpell(me->GetVictim(), SPELL_PLAGUE_STRIKE, false);
258  events.RepeatEvent(urand(10000, 12000));
259  break;
260  case EVENT_SPELL_ICY_TOUCH:
261  if( me->GetVictim() )
262  me->CastSpell(me->GetVictim(), SPELL_ICY_TOUCH, false);
263  events.RepeatEvent(urand(5000, 6000));
264  break;
265  case EVENT_SPELL_DEATH_RESPITE:
266  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true) )
267  me->CastSpell(target, SPELL_DEATH_RESPITE, false);
268  events.RepeatEvent(urand(13000, 15000));
269  break;
270  case EVENT_SPELL_OBLITERATE:
271  if( me->GetVictim() )
272  me->CastSpell(me->GetVictim(), SPELL_OBLITERATE, false);
273  events.RepeatEvent(urand(15000, 17000));
274  break;
275  case EVENT_SPELL_DESECRATION:
276  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true) )
277  me->CastSpell(target, SPELL_DESECRATION, false);
278  events.RepeatEvent(urand(14000, 17000));
279  break;
280  case EVENT_SPELL_DEATH_BITE:
281  me->CastSpell((Unit*)nullptr, SPELL_DEATH_BITE, false);
282  events.RepeatEvent(urand(2000, 4000));
283  break;
284  case EVENT_SPELL_MARKED_DEATH:
285  if( Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.000000f, true) )
286  me->CastSpell(target, SPELL_MARKED_DEATH, false);
287  events.RepeatEvent(9000);
288  break;
289  }
290 
291  DoMeleeAttackIfReady();
292  }
