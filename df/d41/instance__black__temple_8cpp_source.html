227  void Load(char const* str) override
228  {
229  if (!str)
230  {
231  OUT_LOAD_INST_DATA_FAIL;
232  return;
233  }
234 
235  OUT_LOAD_INST_DATA(str);
236 
237  char dataHead1, dataHead2;
238 
239  std::istringstream loadStream(str);
240  loadStream >> dataHead1 >> dataHead2;
241 
242  if (dataHead1 == 'B' && dataHead2 == 'T')
243  {
244  for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
245  {
246  uint32 tmpState;
247  loadStream >> tmpState;
248  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
249  tmpState = NOT_STARTED;
250  SetBossState(i, EncounterState(tmpState));
251  }
252  }
253  else
254  OUT_LOAD_INST_DATA_FAIL;
255 
256  OUT_LOAD_INST_DATA_COMPLETE;
257  }
