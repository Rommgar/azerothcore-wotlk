208  //add spawn of creature
209  static bool HandleNpcAddCommand(ChatHandler* handler, CreatureEntry id)
210  {
211  if (!sObjectMgr->GetCreatureTemplate(id))
212  return false;
213 
214  Player* chr = handler->GetSession()->GetPlayer();
215  float x = chr->GetPositionX();
216  float y = chr->GetPositionY();
217  float z = chr->GetPositionZ();
218  float o = chr->GetOrientation();
219  Map* map = chr->GetMap();
220 
221  if (Transport* tt = chr->GetTransport())
222  if (MotionTransport* trans = tt->ToMotionTransport())
223  {
224  ObjectGuid::LowType guid = sObjectMgr->GenerateCreatureSpawnId();
225  CreatureData& data = sObjectMgr->NewOrExistCreatureData(guid);
226  data.id = id;
227  data.phaseMask = chr->GetPhaseMaskForSpawn();
228  data.posX = chr->GetTransOffsetX();
229  data.posY = chr->GetTransOffsetY();
230  data.posZ = chr->GetTransOffsetZ();
231  data.orientation = chr->GetTransOffsetO();
232 
233  Creature* creature = trans->CreateNPCPassenger(guid, &data);
234 
235  creature->SaveToDB(trans->GetGOInfo()->moTransport.mapID, 1 << map->GetSpawnMode(), chr->GetPhaseMaskForSpawn());
236 
237  sObjectMgr->AddCreatureToGrid(guid, &data);
238  return true;
239  }
240 
241  Creature* creature = new Creature();
242  if (!creature->Create(map->GenerateLowGuid<HighGuid::Unit>(), map, chr->GetPhaseMaskForSpawn(), id, 0, x, y, z, o))
243  {
244  delete creature;
245  return false;
246  }
247 
248  creature->SaveToDB(map->GetId(), (1 << map->GetSpawnMode()), chr->GetPhaseMaskForSpawn());
249 
250  ObjectGuid::LowType spawnId = creature->GetSpawnId();
251 
252  // To call _LoadGoods(); _LoadQuests(); CreateTrainerSpells()
253  // current "creature" variable is deleted and created fresh new, otherwise old values might trigger asserts or cause undefined behavior
254  creature->CleanupsBeforeDelete();
255  delete creature;
256  creature = new Creature();
257  if (!creature->LoadCreatureFromDB(spawnId, map, true, false, true))
258  {
259  delete creature;
260  return false;
261  }
262 
263  sObjectMgr->AddCreatureToGrid(spawnId, sObjectMgr->GetCreatureData(spawnId));
264  return true;
265  }
