87 bool findnth(std::string& str, int n, std::string::size_type& s, std::string::size_type& e)
88 {
89  s = str.find("VALUES ('") + 9;
90  if (s == std::string::npos) return false;
91 
92  do
93  {
94  e = str.find('\'', s);
95  if (e == std::string::npos) return false;
96  } while (str[e - 1] == '\\');
97 
98  for (int i = 1; i < n; ++i)
99  {
100  do
101  {
102  s = e + 4;
103  e = str.find('\'', s);
104  if (e == std::string::npos) return false;
105  } while (str[e - 1] == '\\');
106  }
107  return true;
108 }

265 bool PlayerDumpWriter::DumpTable(std::string& dump, uint32 guid, char const* tableFrom, char const* tableTo, DumpTableType type)
266 {
267  GUIDs const* guids = nullptr;
268  char const* fieldname = nullptr;
269 
270  switch (type)
271  {
272  case DTT_ITEM:
273  fieldname = "guid";
274  guids = &items;
275  break;
276  case DTT_ITEM_GIFT:
277  fieldname = "item_guid";
278  guids = &items;
279  break;
280  case DTT_PET:
281  fieldname = "owner";
282  break;
283  case DTT_PET_TABLE:
284  fieldname = "guid";
285  guids = &pets;
286  break;
287  case DTT_MAIL:
288  fieldname = "receiver";
289  break;
290  case DTT_MAIL_ITEM:
291  fieldname = "mail_id";
292  guids = &mails;
293  break;
294  default:
295  fieldname = "guid";
296  break;
297  }
298 
299  // for guid set stop if set is empty
300  if (guids && guids->empty())
301  return true; // nothing to do
302 
303  // setup for guids case start position
304  GUIDs::const_iterator guids_itr;
305  if (guids)
306  guids_itr = guids->begin();
307 
308  do
309  {
310  std::string wherestr;
311 
312  if (guids) // set case, get next guids string
313  wherestr = GenerateWhereStr(fieldname, *guids, guids_itr);
314  else // not set case, get single guid string
315  wherestr = GenerateWhereStr(fieldname, guid);
316 
317  QueryResult result = CharacterDatabase.PQuery("SELECT * FROM %s WHERE %s", tableFrom, wherestr.c_str());
318  if (!result)
319  return true;
320 
321  do
322  {
323  // collect guids
324  switch (type)
325  {
326  case DTT_INVENTORY:
327  StoreGUID(result, 3, items); // item guid collection (character_inventory.item)
328  break;
329  case DTT_PET:
330  StoreGUID(result, 0, pets); // pet petnumber collection (character_pet.id)
331  break;
332  case DTT_MAIL:
333  StoreGUID(result, 0, mails); // mail id collection (mail.id)
334  break;
335  case DTT_MAIL_ITEM:
336  StoreGUID(result, 1, items); // item guid collection (mail_items.item_guid)
337  break;
338  case DTT_CHARACTER:
339  {
340  if (result->GetFieldCount() <= 74) // avoid crashes on next check
341  LOG_FATAL("entities.player.dump", "PlayerDumpWriter::DumpTable - Trying to access non-existing or wrong positioned field (`deleteInfos_Account`) in `characters` table.");
342 
343  if (result->Fetch()[74].GetUInt32()) // characters.deleteInfos_Account - if filled error
344  return false;
345  break;
346  }
347  default:
348  break;
349  }
350 
351  dump += CreateDumpString(tableTo, result);
352  dump += "\n";
353  } while (result->NextRow());
354  } while (guids && guids_itr != guids->end()); // not set case iterate single time, set case iterate for all guids
355  return true;
356 }
