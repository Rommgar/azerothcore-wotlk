28 ulonglong my_getsystime()
29 {
30 #ifdef HAVE_CLOCK_GETTIME
31  struct timespec tp;
32  clock_gettime(CLOCK_REALTIME, &tp);
33  return (ulonglong)tp.tv_sec*10000000+(ulonglong)tp.tv_nsec/100;
34 #elif defined(__WIN__)
35  LARGE_INTEGER t_cnt;
36  if (query_performance_frequency)
37  {
38  QueryPerformanceCounter(&t_cnt);
39  return ((t_cnt.QuadPart / query_performance_frequency * 10000000) +
40  ((t_cnt.QuadPart % query_performance_frequency) * 10000000 /
41  query_performance_frequency) + query_performance_offset);
42  }
43  return 0;
44 #else
45  /* TODO: check for other possibilities for hi-res timestamping */
46  struct timeval tv;
47  gettimeofday(&tv,NULL);
48  return (ulonglong)tv.tv_sec*10000000+(ulonglong)tv.tv_usec*10;

101 ulonglong my_micro_time()
102 {
103 #if defined(__WIN__)
104  ulonglong newtime;
105  GetSystemTimeAsFileTime((FILETIME*)&newtime);
106  return (newtime/10);
107 #elif defined(HAVE_GETHRTIME)
108  return gethrtime()/1000;
109 #else
110  ulonglong newtime;

112  /*
113  The following loop is here because gettimeofday may fail on some systems
114  */
115  while (gettimeofday(&t, NULL) != 0)
116  {}

118  return newtime;

123 /*
124  Return time in seconds and timer in microseconds (not different start!)
125 
126  SYNOPSIS
127  my_micro_time_and_time()
128  time_arg Will be set to seconds since epoch (00:00:00 UTC,
129  January 1, 1970)
130 
131  NOTES
132  This function is to be useful when we need both the time and microtime.
133  For example in MySQL this is used to get the query time start of a query
134  and to measure the time of a query (for the slow query log)
135 
136  IMPLEMENTATION
137  Value of time is as in time() call.
138  Value of microtime is same as my_micro_time(), which may be totally
139  unrealated to time()
140 
141  RETURN
142  Value in microseconds from some undefined point in time
143 */

145 #define DELTA_FOR_SECONDS 500000000LL /* Half a second */
146 
147 /* Difference between GetSystemTimeAsFileTime() and now() */
148 #define OFFSET_TO_EPOCH 116444736000000000ULL
149 
150 ulonglong my_micro_time_and_time(time_t *time_arg)
151 {
152 #if defined(__WIN__)
153  ulonglong newtime;
154  GetSystemTimeAsFileTime((FILETIME*)&newtime);
155  *time_arg= (time_t) ((newtime - OFFSET_TO_EPOCH) / 10000000);
156  return (newtime/10);
157 #elif defined(HAVE_GETHRTIME)
158  /*
159  Solaris has a very slow time() call. We optimize this by using the very
160  fast gethrtime() call and only calling time() every 1/2 second
161  */
162  static hrtime_t prev_gethrtime= 0;
163  static time_t cur_time= 0;
164  hrtime_t cur_gethrtime;
165 
166  mysql_mutex_lock(&THR_LOCK_time);
167  cur_gethrtime= gethrtime();
168  /*
169  Due to bugs in the Solaris (x86) implementation of gethrtime(),
170  the time returned by it might not be monotonic. Don't use the
171  cached time(2) value if this is a case.
172  */
173  if ((prev_gethrtime > cur_gethrtime) ||
174  ((cur_gethrtime - prev_gethrtime) > DELTA_FOR_SECONDS))
175  {
176  cur_time= time(0);
177  prev_gethrtime= cur_gethrtime;
178  }
179  *time_arg= cur_time;
180  mysql_mutex_unlock(&THR_LOCK_time);
181  return cur_gethrtime/1000;
182 #else
183  ulonglong newtime;
184  struct timeval t;
185  /*
186  The following loop is here because gettimeofday may fail on some systems
187  */
188  while (gettimeofday(&t, NULL) != 0)
189  {}
190  *time_arg= t.tv_sec;
191  newtime= (ulonglong)t.tv_sec * 1000000 + t.tv_usec;
192  return newtime;
193 #endif /* defined(__WIN__) */
194 }
