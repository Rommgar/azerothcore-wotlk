68 void WorldSession::HandleBattlemasterJoinOpcode(WorldPacket& recvData)
69 {
70  ObjectGuid guid;
71  uint32 bgTypeId_;
72  uint32 instanceId; // sent to queue for particular bg from battlemaster's list, currently not used
73  uint8 joinAsGroup;
74 
75  recvData >> guid; // battlemaster guid
76  recvData >> bgTypeId_; // battleground type id (DBC id)
77  recvData >> instanceId; // instance id, 0 if First Available selected
78  recvData >> joinAsGroup; // join as group
79 
80  // entry not found
81  if (!sBattlemasterListStore.LookupEntry(bgTypeId_))
82  return;
83 
84  // chosen battleground type is disabled
85  if (DisableMgr::IsDisabledFor(DISABLE_TYPE_BATTLEGROUND, bgTypeId_, nullptr))
86  {
87  ChatHandler(this).PSendSysMessage(LANG_BG_DISABLED);
88  return;
89  }
90 
91  // get queue typeid and random typeid to check if already queued for them
92  BattlegroundTypeId bgTypeId = BattlegroundTypeId(bgTypeId_);
93  BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(bgTypeId, 0);
94  BattlegroundQueueTypeId bgQueueTypeIdRandom = BattlegroundMgr::BGQueueTypeId(BATTLEGROUND_RB, 0);
95 
96  // safety check - bgQueueTypeId == BATTLEGROUND_QUEUE_NONE if tried to queue for arena using this function
97  if (bgQueueTypeId == BATTLEGROUND_QUEUE_NONE)
98  return;
99 
100  // get bg template
101  Battleground* bgt = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
102  if (!bgt)
103  return;
104 
105  // expected bracket entry
106  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bgt->GetMapId(), _player->getLevel());
107  if (!bracketEntry)
108  return;
109 
110  // pussywizard: if trying to queue for already queued
111  // just remove from queue and it will requeue!
112  uint32 qSlot = _player->GetBattlegroundQueueIndex(bgQueueTypeId);
113  if (qSlot < PLAYER_MAX_BATTLEGROUND_QUEUES)
114  {
115  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
116 
117  if (bgQueue.IsPlayerInvitedToRatedArena(_player->GetGUID()))
118  {
119  WorldPacket data;
120  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_BATTLEGROUND_JOIN_FAILED);
121  SendPacket(&data);
122  return;
123  }
124 
125  bgQueue.RemovePlayer(_player->GetGUID(), false, qSlot);
126  _player->RemoveBattlegroundQueueId(bgQueueTypeId);
127  }
128 
129  // must have free queue slot
130  if (!_player->HasFreeBattlegroundQueueId())
131  {
132  WorldPacket data;
133  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_BATTLEGROUND_TOO_MANY_QUEUES);
134  SendPacket(&data);
135  return;
136  }
137 
138  // queue result (default ok)
139  GroupJoinBattlegroundResult err = GroupJoinBattlegroundResult(bgt->GetBgTypeID());
140 
141  if (!sScriptMgr->CanJoinInBattlegroundQueue(_player, guid, bgTypeId, joinAsGroup, err) && err <= 0)
142  {
143  WorldPacket data;
144  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
145  SendPacket(&data);
146  return;
147  }
148 
149  // check if player can queue:
150  if (!joinAsGroup)
151  {
152  if (GetPlayer()->InBattleground()) // currently in battleground
153  err = ERR_BATTLEGROUND_NOT_IN_BATTLEGROUND;
154  else if (GetPlayer()->isUsingLfg()) // using lfg system
155  err = ERR_LFG_CANT_USE_BATTLEGROUND;
156  else if (!_player->CanJoinToBattleground()) // has deserter debuff
157  err = ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS;
158  else if (_player->InBattlegroundQueueForBattlegroundQueueType(bgQueueTypeIdRandom)) // queued for random bg, so can't queue for anything else
159  err = ERR_IN_RANDOM_BG;
160  else if (_player->InBattlegroundQueue() && bgTypeId == BATTLEGROUND_RB) // already in queue, so can't queue for random
161  err = ERR_IN_NON_RANDOM_BG;
162  else if (_player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_2v2) ||
163  _player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_3v3) ||
164  _player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_5v5)) // can't be already queued for arenas
165  err = ERR_BATTLEGROUND_QUEUED_FOR_RATED;
166  // don't let Death Knights join BG queues when they are not allowed to be teleported yet
167  else if (_player->getClass() == CLASS_DEATH_KNIGHT && _player->GetMapId() == 609 && !_player->IsGameMaster() && !_player->HasSpell(50977))
168  err = ERR_BATTLEGROUND_NONE;
169 
170  if (err <= 0)
171  {
172  WorldPacket data;
173  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
174  SendPacket(&data);
175  return;
176  }
177 
178  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
179  GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, nullptr, bracketEntry, false, false, 0, 0, 0);
180  uint32 avgWaitTime = bgQueue.GetAverageQueueWaitTime(ginfo);
181 
182  uint32 queueSlot = _player->AddBattlegroundQueueId(bgQueueTypeId);
183 
184  // send status packet
185  WorldPacket data;
186  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bgt, queueSlot, STATUS_WAIT_QUEUE, avgWaitTime, 0, 0, TEAM_NEUTRAL);
187  SendPacket(&data);
188 
189  sScriptMgr->OnPlayerJoinBG(_player);
190  }
191  // check if group can queue:
192  else
193  {
194  Group* grp = _player->GetGroup();
195  // no group or not a leader
196  if (!grp || grp->GetLeaderGUID() != _player->GetGUID())
197  return;
198 
199  // pussywizard: for party members - remove queues for which leader is not queued to!
200  std::set<uint32> leaderQueueTypeIds;
201  for (uint32 i = 0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
202  leaderQueueTypeIds.insert((uint32)_player->GetBattlegroundQueueTypeId(i));
203  for (GroupReference* itr = grp->GetFirstMember(); itr != nullptr; itr = itr->next())
204  if (Player* member = itr->GetSource())
205  for (uint32 i = 0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
206  if (BattlegroundQueueTypeId mqtid = member->GetBattlegroundQueueTypeId(i))
207  if (leaderQueueTypeIds.count((uint32)mqtid) == 0)
208  {
209  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(mqtid);
210 
211  if (bgQueue.IsPlayerInvitedToRatedArena(member->GetGUID()))
212  {
213  WorldPacket data;
214  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_BATTLEGROUND_JOIN_FAILED);
215  SendPacket(&data);
216  return;
217  }
218 
219  bgQueue.RemovePlayer(member->GetGUID(), false, i);
220  member->RemoveBattlegroundQueueId(mqtid);
221  }
222 
223  if (_player->InBattlegroundQueueForBattlegroundQueueType(bgQueueTypeIdRandom)) // queued for random bg, so can't queue for anything else
224  err = ERR_IN_RANDOM_BG;
225  else if (_player->InBattlegroundQueue() && bgTypeId == BATTLEGROUND_RB) // already in queue, so can't queue for random
226  err = ERR_IN_NON_RANDOM_BG;
227  else if (_player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_2v2) ||
228  _player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_3v3) ||
229  _player->InBattlegroundQueueForBattlegroundQueueType(BATTLEGROUND_QUEUE_5v5)) // can't be already queued for arenas
230  err = ERR_BATTLEGROUND_QUEUED_FOR_RATED;
231 
232  if (err > 0)
233  err = grp->CanJoinBattlegroundQueue(bgt, bgQueueTypeId, 0, bgt->GetMaxPlayersPerTeam(), false, 0);
234 
235  bool isPremade = (grp->GetMembersCount() >= bgt->GetMinPlayersPerTeam() && bgTypeId != BATTLEGROUND_RB);
236  uint32 avgWaitTime = 0;
237 
238  if (err > 0)
239  {
240  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
241  GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, grp, bracketEntry, false, isPremade, 0, 0, 0);
242  avgWaitTime = bgQueue.GetAverageQueueWaitTime(ginfo);
243  }
244 
245  WorldPacket data;
246  for (GroupReference* itr = grp->GetFirstMember(); itr != nullptr; itr = itr->next())
247  {
248  Player* member = itr->GetSource();
249  if (!member)
250  continue;
251 
252  if (err <= 0)
253  {
254  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
255  member->GetSession()->SendPacket(&data);
256  continue;
257  }
258 
259  uint32 queueSlot = member->AddBattlegroundQueueId(bgQueueTypeId);
260 
261  // send status packet
262  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bgt, queueSlot, STATUS_WAIT_QUEUE, avgWaitTime, 0, 0, TEAM_NEUTRAL);
263  member->GetSession()->SendPacket(&data);
264 
265  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, err);
266  member->GetSession()->SendPacket(&data);
267 
268  sScriptMgr->OnPlayerJoinBG(member);
269  }
270  }
271 }

273 void WorldSession::HandleBattlegroundPlayerPositionsOpcode(WorldPacket& /*recvData*/)
274 {
275  LOG_DEBUG("network", "WORLD: Recvd MSG_BATTLEGROUND_PLAYER_POSITIONS Message");
276 
277  Battleground* bg = _player->GetBattleground();
278  if (!bg) // can't be received if player not in battleground
279  return;
280 
281  uint32 flagCarrierCount = 0;
282  Player* allianceFlagCarrier = nullptr;
283  Player* hordeFlagCarrier = nullptr;
284 
285  if (ObjectGuid guid = bg->GetFlagPickerGUID(TEAM_ALLIANCE))
286  {
287  allianceFlagCarrier = ObjectAccessor::FindPlayer(guid);
288  if (allianceFlagCarrier)
289  ++flagCarrierCount;
290  }
291 
292  if (ObjectGuid guid = bg->GetFlagPickerGUID(TEAM_HORDE))
293  {
294  hordeFlagCarrier = ObjectAccessor::FindPlayer(guid);
295  if (hordeFlagCarrier)
296  ++flagCarrierCount;
297  }
298 
299  WorldPacket data(MSG_BATTLEGROUND_PLAYER_POSITIONS, 4 + 4 + 16 * flagCarrierCount);
300  // Used to send several player positions (found used in AV)
301  data << 0; // CGBattlefieldInfo__m_numPlayerPositions
302  /*
303  for (CGBattlefieldInfo__m_numPlayerPositions)
304  data << guid << posx << posy;
305  */
306  data << flagCarrierCount;
307  if (allianceFlagCarrier)
308  {
309  data << allianceFlagCarrier->GetGUID();
310  data << float(allianceFlagCarrier->GetPositionX());
311  data << float(allianceFlagCarrier->GetPositionY());
312  }
313 
314  if (hordeFlagCarrier)
315  {
316  data << hordeFlagCarrier->GetGUID();
317  data << float(hordeFlagCarrier->GetPositionX());
318  data << float(hordeFlagCarrier->GetPositionY());
319  }
320 
321  SendPacket(&data);
322 }

365 void WorldSession::HandleBattleFieldPortOpcode(WorldPacket& recvData)
366 {
367  uint8 arenaType; // arenatype if arena
368  uint8 unk2; // unk, can be 0x0 (may be if was invited?) and 0x1
369  uint32 bgTypeId_; // type id from dbc
370  uint16 unk; // 0x1F90 constant?
371  uint8 action; // enter battle 0x1, leave queue 0x0
372 
373  recvData >> arenaType >> unk2 >> bgTypeId_ >> unk >> action;
374 
375  // bgTypeId not valid
376  if (!sBattlemasterListStore.LookupEntry(bgTypeId_))
377  return;
378 
379  // player not in any queue, so can't really answer
380  if (!_player->InBattlegroundQueue())
381  return;
382 
383  // get BattlegroundQueue for received
384  BattlegroundTypeId bgTypeId = BattlegroundTypeId(bgTypeId_);
385  BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(bgTypeId, arenaType);
386  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
387 
388  if (!sScriptMgr->CanBattleFieldPort(_player, arenaType, bgTypeId, action))
389  return;
390 
391  // get group info from queue
392  GroupQueueInfo ginfo;
393  if (!bgQueue.GetPlayerGroupInfoData(_player->GetGUID(), &ginfo))
394  return;
395 
396  // to accept, player must be invited to particular battleground id
397  if (!ginfo.IsInvitedToBGInstanceGUID && action == 1)
398  return;
399 
400  Battleground* bg = sBattlegroundMgr->GetBattleground(ginfo.IsInvitedToBGInstanceGUID);
401 
402  // use template if leaving queue (instance might not be created yet)
403  if (!bg && action == 0)
404  bg = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
405 
406  if (!bg)
407  return;
408 
409  // expected bracket entry
410  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->getLevel());
411  if (!bracketEntry)
412  return;
413 
414  // safety checks
415  if (action == 1 && ginfo.ArenaType == 0)
416  {
417  // can't join with deserter, check it here right before joining to be sure
418  if (!_player->CanJoinToBattleground())
419  {
420  WorldPacket data;
421  sBattlegroundMgr->BuildGroupJoinedBattlegroundPacket(&data, ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS);
422  SendPacket(&data);
423  action = 0;
424  }
425 
426  if (_player->getLevel() > bg->GetMaxLevel())
427  action = 0;
428  }
429 
430  // get player queue slot index for this bg (can be in up to 2 queues at the same time)
431  uint32 queueSlot = _player->GetBattlegroundQueueIndex(bgQueueTypeId);
432 
433  WorldPacket data;
434  switch (action)
435  {
436  case 1: // accept
437  {
438  // set entry point if not in battleground
439  if (!_player->InBattleground())
440  _player->SetEntryPoint();
441 
442  // resurrect the player
443  if (!_player->IsAlive())
444  {
445  _player->ResurrectPlayer(1.0f);
446  _player->SpawnCorpseBones();
447  }
448 
449  TeamId teamId = ginfo.teamId;
450 
451  // remove player from all bg queues
452  for (uint32 qslot = 0; qslot < PLAYER_MAX_BATTLEGROUND_QUEUES; ++qslot)
453  if (BattlegroundQueueTypeId q = _player->GetBattlegroundQueueTypeId(qslot))
454  {
455  BattlegroundQueue& queue = sBattlegroundMgr->GetBattlegroundQueue(q);
456  queue.RemovePlayer(_player->GetGUID(), (bgQueueTypeId == q), qslot);
457  _player->RemoveBattlegroundQueueId(q);
458  }
459 
460  // send status packet
461  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, queueSlot, STATUS_IN_PROGRESS, 0, bg->GetStartTime(), bg->GetArenaType(), teamId);
462  SendPacket(&data);
463 
464  _player->SetBattlegroundId(bg->GetInstanceID(), bg->GetBgTypeID(), queueSlot, true, bgTypeId == BATTLEGROUND_RB, teamId);
465 
466  sBattlegroundMgr->SendToBattleground(_player, ginfo.IsInvitedToBGInstanceGUID, bgTypeId);
467  }
468  break;
469  case 0: // leave queue
470  {
471  bgQueue.RemovePlayer(_player->GetGUID(), false, queueSlot);
472  _player->RemoveBattlegroundQueueId(bgQueueTypeId);
473  // track if player refuses to join the BG after being invited
474  if (bg->isBattleground() && (bg->GetStatus() == STATUS_IN_PROGRESS || bg->GetStatus() == STATUS_WAIT_JOIN))
475  {
476  if (sWorld->getBoolConfig(CONFIG_BATTLEGROUND_TRACK_DESERTERS))
477  {
478  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_INS_DESERTER_TRACK);
479  stmt->setUInt32(0, _player->GetGUID().GetCounter());
480  stmt->setUInt8(1, BG_DESERTION_TYPE_LEAVE_QUEUE);
481  CharacterDatabase.Execute(stmt);
482  }
483 
484  sScriptMgr->OnBattlegroundDesertion(_player, BG_DESERTION_TYPE_LEAVE_QUEUE);
485  }
486  }
487  break;
488  default:
489  break;
490  }
491 }

511 void WorldSession::HandleBattlefieldStatusOpcode(WorldPacket& /*recvData*/)
512 {
513  // requested at login and on map change
514  // send status for current queues and current bg
515 
516  WorldPacket data;
517 
518  // for current bg send STATUS_IN_PROGRESS
519  if (Battleground* bg = _player->GetBattleground())
520  if (bg->GetPlayers().count(_player->GetGUID()))
521  {
522  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, _player->GetCurrentBattlegroundQueueSlot(), STATUS_IN_PROGRESS, bg->GetEndTime(), bg->GetStartTime(), bg->GetArenaType(), _player->GetBgTeamId());
523  SendPacket(&data);
524  }
525 
526  // for queued bgs send STATUS_WAIT_JOIN or STATUS_WAIT_QUEUE
527  for (uint8 i = 0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
528  {
529  // check if in queue
530  BattlegroundQueueTypeId bgQueueTypeId = _player->GetBattlegroundQueueTypeId(i);
531  if (!bgQueueTypeId)
532  continue;
533 
534  // get group info from queue
535  BattlegroundQueue& bgQueue = sBattlegroundMgr->GetBattlegroundQueue(bgQueueTypeId);
536  GroupQueueInfo ginfo;
537  if (!bgQueue.GetPlayerGroupInfoData(_player->GetGUID(), &ginfo))
538  continue;
539 
540  BattlegroundTypeId bgTypeId = BattlegroundMgr::BGTemplateId(bgQueueTypeId);
541 
542  // if invited - send STATUS_WAIT_JOIN
543  if (ginfo.IsInvitedToBGInstanceGUID)
544  {
545  Battleground* bg = sBattlegroundMgr->GetBattleground(ginfo.IsInvitedToBGInstanceGUID);
546  if (!bg)
547  continue;
548 
549  uint32 remainingTime = (World::GetGameTimeMS() < ginfo.RemoveInviteTime ? getMSTimeDiff(World::GetGameTimeMS(), ginfo.RemoveInviteTime) : 1);
550  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bg, i, STATUS_WAIT_JOIN, remainingTime, 0, ginfo.ArenaType, TEAM_NEUTRAL, bg->isRated(), ginfo.BgTypeId);
551  SendPacket(&data);
552  }
553  // if not invited - send STATUS_WAIT_QUEUE
554  else
555  {
556  Battleground* bgt = sBattlegroundMgr->GetBattlegroundTemplate(bgTypeId);
557  if (!bgt)
558  continue;
559 
560  // expected bracket entry
561  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bgt->GetMapId(), _player->getLevel());
562  if (!bracketEntry)
563  continue;
564 
565  uint32 avgWaitTime = bgQueue.GetAverageQueueWaitTime(&ginfo);
566  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, bgt, i, STATUS_WAIT_QUEUE, avgWaitTime, getMSTimeDiff(ginfo.JoinTime, World::GetGameTimeMS()), ginfo.ArenaType, TEAM_NEUTRAL, ginfo.IsRated);
567  SendPacket(&data);
568  }
569  }

572 void WorldSession::HandleBattlemasterJoinArena(WorldPacket& recvData)
573 {
574  ObjectGuid guid; // arena Battlemaster guid
575  uint8 arenaslot; // 2v2, 3v3 or 5v5
576  uint8 asGroup; // asGroup
577  uint8 isRated; // isRated
578 
579  recvData >> guid >> arenaslot >> asGroup >> isRated;
580 
581  // can't queue for rated without a group
582  if (isRated && !asGroup)
583  return;
584 
585  // find creature by guid
586  Creature* unit = GetPlayer()->GetMap()->GetCreature(guid);
587  if (!unit || !unit->IsBattleMaster())
588  return;
589 
590  // get arena type
591  uint8 arenatype = 0;
592  switch (arenaslot)
593  {
594  case 0:
595  arenatype = ARENA_TYPE_2v2;
