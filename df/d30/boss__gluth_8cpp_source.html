154  bool SelectPlayerInRoom()
155  {
156  if (me->IsInCombat())
157  return false;
158 
159  Map::PlayerList const& pList = me->GetMap()->GetPlayers();
160  for (const auto& itr : pList)
161  {
162  Player* player = itr.GetSource();
163  if (!player || !player->IsAlive())
164  continue;
165 
166  if (player->GetPositionZ() > 300.0f || me->GetExactDist(player) > 50.0f)
167  continue;
168 
169  AttackStart(player);
170  return true;
171  }
172  return false;
173  }

175  void UpdateAI(uint32 diff) override
176  {
177  if (!UpdateVictimWithGaze() && !SelectPlayerInRoom())
178  return;
179 
180  events.Update(diff);
181  if (me->HasUnitState(UNIT_STATE_CASTING))
182  return;
183 
184  switch (events.ExecuteEvent())
185  {
186  case EVENT_BERSERK:
187  me->CastSpell(me, SPELL_BERSERK, true);
188  break;
189  case EVENT_ENRAGE:
190  Talk(EMOTE_ENRAGE);
191  me->CastSpell(me, RAID_MODE(SPELL_ENRAGE_10, SPELL_ENRAGE_25), true);
192  events.RepeatEvent(22000);
193  break;
194  case EVENT_MORTAL_WOUND:
195  me->CastSpell(me->GetVictim(), SPELL_MORTAL_WOUND, false);
196  events.RepeatEvent(10000);
197  break;
198  case EVENT_DECIMATE:
199  Talk(EMOTE_DECIMATE);
200  me->CastSpell(me, RAID_MODE(SPELL_DECIMATE_10, SPELL_DECIMATE_25), false);
201  events.RepeatEvent(RAID_MODE(110000, 90000));
202  break;
203  case EVENT_SUMMON_ZOMBIE:
204  {
205  uint8 rand = urand(0, 2);
206  for (int32 i = 0; i < RAID_MODE(1, 2); ++i)
207  {
208  // In 10 man raid, normal mode - should spawn only from mid gate
209  // \1 |0 /2 pos
210  // In 25 man raid - should spawn from all 3 gates
211  if (me->GetMap()->GetDifficulty() == RAID_DIFFICULTY_10MAN_NORMAL)
212  {
213  me->SummonCreature(NPC_ZOMBIE_CHOW, zombiePos[0]);
214  }
215  else
216  {
217  me->SummonCreature(NPC_ZOMBIE_CHOW, zombiePos[urand(0, 2)]);
218  }
219  (rand == 2 ? rand = 0 : rand++);
220  }
221  events.RepeatEvent(10000);
222  break;
223  }
224  case EVENT_CAN_EAT_ZOMBIE:
225  events.RepeatEvent(1000);
226  if (me->GetVictim()->GetEntry() == NPC_ZOMBIE_CHOW && me->IsWithinMeleeRange(me->GetVictim()))
227  {
228  me->CastCustomSpell(SPELL_CHOW_SEARCHER, SPELLVALUE_RADIUS_MOD, 20000, me, true);
229  Talk(EMOTE_DEVOURS_ALL);
230  return; // leave it to skip DoMeleeAttackIfReady
231  }
232  break;
233  }
234  DoMeleeAttackIfReady();
235  }
236  };
