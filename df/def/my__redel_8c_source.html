76 int my_copystat(const char *from, const char *to, int MyFlags)
77 {
78  struct stat statbuf;
79 
80  if (stat(from, &statbuf))
81  {
82  my_errno=errno;
83  if (MyFlags & (MY_FAE+MY_WME))
84  my_error(EE_STAT, MYF(ME_BELL+ME_WAITTANG),from,errno);
85  return -1; /* Can't get stat on input file */
86  }
87  if ((statbuf.st_mode & S_IFMT) != S_IFREG)
88  return 1;
89 
90  /* Copy modes */
91  if (chmod(to, statbuf.st_mode & 07777))
92  {
93  my_errno= errno;
94  if (MyFlags & (MY_FAE+MY_WME))
95  my_error(EE_CHANGE_PERMISSIONS, MYF(ME_BELL+ME_WAITTANG), from, errno);
96  return -1;
97  }
98 
99 #if !defined(__WIN__)
100  if (statbuf.st_nlink > 1 && MyFlags & MY_LINK_WARNING)
101  {
102  if (MyFlags & MY_LINK_WARNING)
103  my_error(EE_LINK_WARNING,MYF(ME_BELL+ME_WAITTANG),from,statbuf.st_nlink);
104  }
105  /* Copy ownership */
106  if (chown(to, statbuf.st_uid, statbuf.st_gid))
107  {
108  my_errno= errno;
109  if (MyFlags & (MY_FAE+MY_WME))
110  my_error(EE_CHANGE_OWNERSHIP, MYF(ME_BELL+ME_WAITTANG), from, errno);
111  return -1;
112  }
113 #endif /* !__WIN__ */
114 
115  if (MyFlags & MY_COPYTIME)
116  {
117  struct utimbuf timep;
118  timep.actime = statbuf.st_atime;
119  timep.modtime = statbuf.st_mtime;
120  (void) utime((char*) to, &timep);/* Update last accessed and modified times */
121  }
122 
123  return 0;
124 } /* my_copystat */
