40 void create_last_word_mask(MY_BITMAP *map)
41 {
42  /* Get the number of used bits (1..8) in the last byte */
43  unsigned int const used= 1U + ((map->n_bits-1U) & 0x7U);
44 
45  /*
46  Create a mask with the upper 'unused' bits set and the lower 'used'
47  bits clear. The bits within each byte is stored in big-endian order.
48  */
49  unsigned char const mask= (~((1 << used) - 1)) & 255;
50 
51  /*
52  The first bytes are to be set to zero since they represent real bits
53  in the bitvector. The last bytes are set to 0xFF since they represent
54  bytes not used by the bitvector. Finally the last byte contains bits
55  as set by the mask above.
56  */
57  unsigned char *ptr= (unsigned char*)&map->last_word_mask;
58 
59  map->last_word_ptr= map->bitmap + no_words_in_map(map)-1;
60  switch (no_bytes_in_map(map) & 3) {
61  case 1:
62  map->last_word_mask= ~0U;
63  ptr[0]= mask;
64  return;
65  case 2:
66  map->last_word_mask= ~0U;
67  ptr[0]= 0;
68  ptr[1]= mask;
69  return;
70  case 3:
71  map->last_word_mask= 0U;
72  ptr[2]= mask;
73  ptr[3]= 0xFFU;
74  return;
75  case 0:
76  map->last_word_mask= 0U;
77  ptr[3]= mask;
78  return;
79  }
80 }

242 void bitmap_set_prefix(MY_BITMAP *map, uint prefix_size)
243 {
244  uint prefix_bytes, prefix_bits, d;
245  uchar *m= (uchar *)map->bitmap;
246 
247  DBUG_ASSERT(map->bitmap &&
248  (prefix_size <= map->n_bits || prefix_size == (uint) ~0));
249  set_if_smaller(prefix_size, map->n_bits);
250  if ((prefix_bytes= prefix_size / 8))
251  memset(m, 0xff, prefix_bytes);
252  m+= prefix_bytes;
253  if ((prefix_bits= prefix_size & 7))
254  *m++= (1 << prefix_bits)-1;
255  if ((d= no_bytes_in_map(map)-prefix_bytes))
256  bzero(m, d);
257 }

260 my_bool bitmap_is_prefix(const MY_BITMAP *map, uint prefix_size)
261 {
262  uint prefix_bits= prefix_size & 0x7, res;
263  uchar *m= (uchar*)map->bitmap;
264  uchar *end_prefix= m+prefix_size/8;
265  uchar *end;
266  DBUG_ASSERT(m && prefix_size <= map->n_bits);
267  end= m+no_bytes_in_map(map);
268 
269  while (m < end_prefix)
270  if (*m++ != 0xff)
271  return 0;
272 
273  *map->last_word_ptr&= ~map->last_word_mask; /*Clear bits*/
274  res= 0;
275  if (prefix_bits && *m++ != (1 << prefix_bits)-1)
276  goto ret;
277 
278  while (m < end)
279  if (*m++ != 0)
280  goto ret;
281  res= 1;
282 ret:
283  return res;
284 }

353 void bitmap_intersect(MY_BITMAP *map, const MY_BITMAP *map2)
354 {
355  my_bitmap_map *to= map->bitmap, *from= map2->bitmap, *end;
356  uint len= no_words_in_map(map), len2 = no_words_in_map(map2);
357 
358  DBUG_ASSERT(map->bitmap && map2->bitmap);
359 
360  end= to+min(len,len2);
361  *map2->last_word_ptr&= ~map2->last_word_mask; /*Clear last bits in map2*/
362  while (to < end)
363  *to++ &= *from++;
364 
365  if (len2 < len)
366  {
367  end+=len-len2;
368  while (to < end)
369  *to++=0;
370  }
371 }

480 {
481  uchar *byte_ptr;
482  uint i,j,k;
483  my_bitmap_map *data_ptr, *end= map->last_word_ptr;
484 
485  DBUG_ASSERT(map->bitmap);
486  data_ptr= map->bitmap;
487  *map->last_word_ptr &= ~map->last_word_mask;
488 
489  for (i=0; data_ptr <= end; data_ptr++, i++)
490  {
491  if (*data_ptr)
492  {
493  byte_ptr= (uchar*)data_ptr;
494  for (j=0; ; j++, byte_ptr++)
495  {
496  if (*byte_ptr)
497  {
498  for (k=0; ; k++)
499  {
500  if (*byte_ptr & (1 << k))
501  return (i*32) + (j*8) + k;
502  }
503  }
504  }
505  }
506  }
507  return MY_BIT_NONE;
