160  void SpellHit(Unit* caster, const SpellInfo* entry) override
161  {
162  if (caster == me)
163  return;
164 
165  Creature* pOtherBoss = GetOtherBoss();
166  if (entry->Id != SPELL_HEAL_BROTHER || !pOtherBoss)
167  return;
168 
169  // add health so we keep same percentage for both brothers
170  uint32 mytotal = me->GetMaxHealth(), histotal = pOtherBoss->GetMaxHealth();
171  float mult = ((float)mytotal) / ((float)histotal);
172  if (mult < 1)
173  mult = 1.0f / mult;
174 #define HEAL_BROTHER_AMOUNT 30000.0f
175  uint32 largerAmount = (uint32)((HEAL_BROTHER_AMOUNT * mult) - HEAL_BROTHER_AMOUNT);
176 
177  if (mytotal > histotal)
178  {
179  uint32 h = me->GetHealth() + largerAmount;
180  me->SetHealth(std::min(mytotal, h));
181  }
182  else
183  {
184  uint32 h = pOtherBoss->GetHealth() + largerAmount;
185  pOtherBoss->SetHealth(std::min(histotal, h));

303  Creature* RespawnNearbyBugsAndGetOne()
304  {
305  std::list<Creature*> lUnitList;
306  me->GetCreatureListWithEntryInGrid(lUnitList, 15316, 150.0f);
307  me->GetCreatureListWithEntryInGrid(lUnitList, 15317, 150.0f);
308 
309  if (lUnitList.empty())
310  return nullptr;
311 
312  Creature* nearb = nullptr;
313 
314  for (std::list<Creature*>::const_iterator iter = lUnitList.begin(); iter != lUnitList.end(); ++iter)
315  {
316  Creature* c = *iter;
317  if (c)
318  {
319  if (c->isDead())
320  {
321  c->Respawn();
322  c->SetFaction(FACTION_CREATURE);
323  c->RemoveAllAuras();
324  }
325  if (c->IsWithinDistInMap(me, ABUSE_BUG_RANGE))
326  {
327  if (!nearb || (rand() % 4) == 0)
328  nearb = c;
329  }
330  }
331  }
332  return nearb;
