280  uint8 version = input.readUInt8();
281  uint8 encoding = input.readUInt8();
282  uint8 bitsPerPixel = input.readUInt8();
283

285  uint16 ymin = input.readUInt16();
286  uint16 xmax = input.readUInt16();
287  uint16 ymax = input.readUInt16();
288 
289  uint16 horizDPI = input.readUInt16();
290  uint16 vertDPI = input.readUInt16();
291 
292  Color3uint8 colorMap[16];
293  input.readBytes(colorMap, 48);
294 
295  input.skip(1);
296 
297  uint8 planes = input.readUInt8();
298  uint16 bytesPerLine = input.readUInt16();
299  uint16 paletteType = input.readUInt16();
300  input.skip(4 + 54);
301 
302  (void)bytesPerLine;
303 
304  m_width = xmax - xmin + 1;
305  m_height = ymax - ymin + 1;
306  m_channels = 3;
307 
308  if ((manufacturer != 0x0A) || (encoding != 0x01)) {
309  throw GImage::Error("PCX file is corrupted", input.getFilename());
310  }
311 
312  (void)version;
313  (void)vertDPI;
314  (void)horizDPI;
315 
316  if ((bitsPerPixel != 8) || ((planes != 1) && (planes != 3))) {
317  throw GImage::Error("Only 8-bit paletted and 24-bit PCX files supported.", input.getFilename());
318  }
319 
320  // Prepare the pointer object for the pixel data
321  m_byte = (uint8*)m_memMan->alloc(m_width * m_height * 3);
322 
323  if ((paletteType == 1) && (planes == 3)) {
324 
325  Color3uint8* pixel = pixel3();
326 
327  // Iterate over each scan line
328  for (int row = 0; row < m_height; ++row) {
329  // Read each scan line once per plane
330  for (int plane = 0; plane < planes; ++plane) {
331  int p = row * m_width;
332  int p1 = p + m_width;
333  while (p < p1) {
334  uint8 value = input.readUInt8();
335  int length = 1;
336

338  // This is the length, not the value. Mask off
339  // the two high bits and read the true index.
340  length = value & 0x3F;
341  value = input.readUInt8();
342  }
343 
344  // Set the whole run
345  for (int i = length - 1; i >= 0; --i, ++p) {
346  debugAssert(p < m_width * m_height);
347  pixel[p][plane] = value;
348  }
349  }
350  }
351  }
352 
353  } else if (planes == 1) {
354 
355  Color3uint8 palette[256];
356 
357  int imageBeginning = input.getPosition();
358  int paletteBeginning = input.getLength() - 769;
359 
360  input.setPosition(paletteBeginning);
361 
362  uint8 dummy = input.readUInt8();
363 
364  if (dummy != 12) {
365  Log::common()->println("\n*********************");
366  Log::common()->printf("Warning: Corrupted PCX file (palette marker byte was missing) \"%s\"\nLoading anyway\n\n", input.getFilename().c_str());
367  }

370  input.setPosition(imageBeginning);
371 
372  Color3uint8* pixel = pixel3();
373 
374  // The palette indices are run length encoded.
375  int p = 0;
376  while (p < m_width * m_height) {
377  uint8 index = input.readUInt8();
378  uint8 length = 1;
379 
380  if (index >= 192) {
381  // This is the length, not the index. Mask off
382  // the two high bits and read the true index.
383  length = index & 0x3F;
384  index = input.readUInt8();
385  }
386 
387  Color3uint8 color = palette[index];
388 
389  // Set the whole run
390  for (int i = length - 1; i >= 0; --i, ++p) {
391  if (p > m_width * m_height) {
392  break;
393  }
394  pixel[p] = color;
395  }
396 
397  }
398 
399  } else {
400  throw GImage::Error("Unsupported PCX file type.", input.getFilename());
401  }

415 GImage::Format GImage::resolveFormat(
416  const std::string& filename,
417  const uint8* data,
418  int dataLen,
419  Format maybeFormat) {
420 
421  // Return the provided format if it is specified.
422  if (maybeFormat != AUTODETECT) {
423  return maybeFormat;
424  }
425 
426  std::string extension = toUpper(filenameExt(filename));
427 
428  if ((extension == "PPM") || (extension == "PGM") || (extension == "PBM")) {
429  // There are two PPM formats (binary and ASCII); we handle them differently
430  if (dataLen > 3) {
431  if (!memcmp(data, "P6", 2) || !memcmp(data, "P5", 2)) {
432  return PPM_BINARY;
433  } else {
434  return PPM_ASCII;
435  }
436  }
437  }
438 
439  Format tmp = stringToFormat(extension);
440  if ((tmp != AUTODETECT) && (tmp != UNKNOWN)) {
441  return tmp;
442  }
443 
444  // Try and autodetect from the file itself by looking at the first
445  // character.
446 
447  // We can't look at the character if it is null.
448  debugAssert(data != NULL);
449 
450  if ((dataLen > 3) && (! memcmp(data, "P3", 2) || (! memcmp(data, "P2", 2)) || (! memcmp(data, "P1", 2)))) {
451  return PPM_ASCII;
452  }
453 
454  if ((dataLen > 3) && (!memcmp(data, "P6", 2) ||!memcmp(data, "P5", 2))) {
455  return PPM_BINARY;
456  }
457 
458  if (dataLen > 8) {
459  if (!png_sig_cmp((png_bytep)data, 0, 8)) {
460  return PNG;
461  }
462  }
463 
464  if ((dataLen > 0) && (data[0] == 'B')) {
465  return BMP;
466  }
467 
468  if (dataLen > 10) {
469  if ((dataLen > 11) && (data[0] == 0xFF) &&
470  (memcmp(&data[6], "JFIF", 4) == 0)) {
471  return JPEG;
472  }
473  }
474 
475  if (dataLen > 40) {
476  if (memcmp(&data[dataLen - 18], "TRUEVISION-XFILE", 16) == 0) {
477  return TGA;
478  }
479  }
480 
481  if ((dataLen > 4) && (data[0] == 0) && (data[1] == 0) && (data[2] == 0) && (data[3] == 1)) {
482  return ICO;
483  }
484 
485  if ((dataLen > 0) && (data[0] == 10)) {
486  return PCX;
487  }
488 
489  return UNKNOWN;
490 }

884 void GImage::computeNormalMap(
885  int width,
886  int height,
887  int channels,
888  const uint8* src,
889  GImage& normal,
890  const BumpMapPreprocess& preprocess) {
891 
892  float whiteHeightInPixels = preprocess.zExtentPixels;
893  bool lowPassBump = preprocess.lowPassFilter;
894  bool scaleHeightByNz = preprocess.scaleZByNz;
895 
896  if (whiteHeightInPixels < 0.0f) {
897  // Default setting scales so that a gradient ramp
898  // over the whole image becomes a 45-degree angle
899 
900  // Account for potentially non-square aspect ratios
901  whiteHeightInPixels = max(width, height) * -whiteHeightInPixels;
902  }
903 
904  debugAssert(whiteHeightInPixels >= 0);
905 
906  const int w = width;
907  const int h = height;
908  const int stride = channels;
909 
910  normal.resize(w, h, 4);
911 
912  const uint8* const B = src;
913  Color4uint8* const N = normal.pixel4();
914 
915  // 1/s for the scale factor that each ELEVATION should be multiplied by.
916  // We avoid actually multiplying by this and instead just divide it out of z.
917  float elevationInvScale = 255.0f / whiteHeightInPixels;
918 
919  for (int y = 0; y < h; ++y) {
920  for (int x = 0; x < w; ++x) {
921  // Index into normal map pixel
922  int i = x + y * w;
923 
924  // Index into bump map *byte*
925  int j = stride * i;
926 
927  Vector3 delta;
928 
929  // Get a value from B (with wrapping lookup) relative to (x, y)
930  // and divide by 255
931  #define ELEVATION(DX, DY) ((int)B[(((DX + x + w) % w) + \
932  ((DY + y + h) % h) * w) * stride])
933 
934 
935  // Sobel filter to compute the normal.
936  //
937  // Y Filter (X filter is the transpose)
938  // [ -1 -2 -1 ]
939  // [ 0 0 0 ]
940  // [ 1 2 1 ]
941 
942  // Write the Y value directly into the x-component so we don't have
943  // to explicitly compute a cross product at the end. Does not
944  // go out of bounds because the above is computed mod (width, height)
945  delta.y = -( ELEVATION(-1, -1) * 1 + ELEVATION( 0, -1) * 2 + ELEVATION( 1, -1) * 1 +
946  -ELEVATION(-1, 1) * 1 + -ELEVATION( 0, 1) * 2 + -ELEVATION( 1, 1) * 1);
947 
948  delta.x = -(-ELEVATION(-1, -1) * 1 + ELEVATION( 1, -1) * 1 +
949  -ELEVATION(-1, 0) * 2 + ELEVATION( 1, 0) * 2 +
950  -ELEVATION(-1, 1) * 1 + ELEVATION( 1, 1) * 1);
951 
952  // The scale of each filter row is 4, the filter width is two pixels,
953  // and the "normal" range is 0-255.
954  delta.z = 4 * 2 * elevationInvScale;
955 
956  // Delta is now scaled in pixels; normalize
957  delta = delta.direction();
958 
959  // Copy over the bump value into the alpha channel.
960  float H = B[j] / 255.0f;
961 
962  if (lowPassBump) {
963  H = (ELEVATION(-1, -1) + ELEVATION( 0, -1) + ELEVATION(1, -1) +
964  ELEVATION(-1, 0) + ELEVATION( 0, 0) + ELEVATION(1, 0) +
965  ELEVATION(-1, 1) + ELEVATION( 0, 1) + ELEVATION(1, 1)) / (255.0f * 9.0f);
966  }
967 # undef ELEVATION
968 
969  if (scaleHeightByNz) {
970  // delta.z can't possibly be negative, so we avoid actually
971  // computing the absolute value.
972  H *= delta.z;
973  }
974 
975  N[i].a = iRound(H * 255.0f);
976 
977  // Pack into byte range
978  delta = delta * 127.5f + Vector3(127.5f, 127.5f, 127.5f);
979  N[i].r = iClamp(iRound(delta.x), 0, 255);
980  N[i].g = iClamp(iRound(delta.y), 0, 255);
981  N[i].b = iClamp(iRound(delta.z), 0, 255);
982  }
983  }
984 }
