33  bool IntersectTriangle(const MeshTriangle& tri, std::vector<Vector3>::const_iterator points, const G3D::Ray& ray, float& distance)
34  {
35  static const float EPS = 1e-5f;
36 
37  // See RTR2 ch. 13.7 for the algorithm.
38 
39  const Vector3 e1 = points[tri.idx1] - points[tri.idx0];
40  const Vector3 e2 = points[tri.idx2] - points[tri.idx0];
41  const Vector3 p(ray.direction().cross(e2));
42  const float a = e1.dot(p);
43 
44  if (fabs(a) < EPS)
45  {
46  // Determinant is ill-conditioned; abort early
47  return false;
48  }
49 
50  const float f = 1.0f / a;
51  const Vector3 s(ray.origin() - points[tri.idx0]);
52  const float u = f * s.dot(p);
53 
54  if ((u < 0.0f) || (u > 1.0f))
55  {
56  // We hit the plane of the m_geometry, but outside the m_geometry
57  return false;
58  }
59 
60  const Vector3 q(s.cross(e1));
61  const float v = f * ray.direction().dot(q);
62 
63  if ((v < 0.0f) || ((u + v) > 1.0f))
64  {
65  // We hit the plane of the triangle, but outside the triangle
66  return false;
67  }
68 
69  const float t = f * e2.dot(q);
70 
71  if ((t > 0.0f) && (t < distance))
72  {
73  // This is a new hit, closer than the previous one
74  distance = t;
75 
76  /* baryCoord[0] = 1.0 - u - v;
77  baryCoord[1] = u;
78  baryCoord[2] = v; */
79 
80  return true;
81  }
82  // This hit is after the previous hit, so ignore it
83  return false;
84  }

157  bool WmoLiquid::GetLiquidHeight(const Vector3& pos, float& liqHeight) const
158  {
159  float tx_f = (pos.x - iCorner.x) / LIQUID_TILE_SIZE;
160  uint32 tx = uint32(tx_f);
161  if (tx_f < 0.0f || tx >= iTilesX)
162  {
163  return false;
164  }
165  float ty_f = (pos.y - iCorner.y) / LIQUID_TILE_SIZE;
166  uint32 ty = uint32(ty_f);
167  if (ty_f < 0.0f || ty >= iTilesY)
168  {
169  return false;
170  }
171 
172  // check if tile shall be used for liquid level
173  // checking for 0x08 *might* be enough, but disabled tiles always are 0x?F:
174  if (iFlags && (iFlags[tx + ty * iTilesX] & 0x0F) == 0x0F)
175  {
176  return false;
177  }
178 
179  // (dx, dy) coordinates inside tile, in [0, 1]^2
180  float dx = tx_f - (float)tx;
181  float dy = ty_f - (float)ty;
182 
183  /* Tesselate tile to two triangles (not sure if client does it exactly like this)
184 
185  ^ dy
186  |
187  1 x---------x (1, 1)
188  | (b) / |
189  | / |
190  | / |
191  | / (a) |
192  x---------x---> dx
193  0 1
194  */
195 
196  if (!iHeight)
197  {
198  return false;
199  }
200 
201  const uint32 rowOffset = iTilesX + 1;
202  if (dx > dy) // case (a)
203  {
204  float sx = iHeight[tx + 1 + ty * rowOffset] - iHeight[tx + ty * rowOffset];
205  float sy = iHeight[tx + 1 + (ty + 1) * rowOffset] - iHeight[tx + 1 + ty * rowOffset];
206  liqHeight = iHeight[tx + ty * rowOffset] + dx * sx + dy * sy;
207  }
208  else // case (b)
209  {
210  float sx = iHeight[tx + 1 + (ty + 1) * rowOffset] - iHeight[tx + (ty + 1) * rowOffset];
211  float sy = iHeight[tx + (ty + 1) * rowOffset] - iHeight[tx + ty * rowOffset];
212  liqHeight = iHeight[tx + ty * rowOffset] + dx * sx + dy * sy;
213  }
214  return true;
215  }
