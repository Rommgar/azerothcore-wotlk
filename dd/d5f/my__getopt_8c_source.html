152 int handle_options(int *argc, char ***argv,
153  const struct my_option *longopts,
154  my_get_one_option get_one_option)
155 {
156  uint UNINIT_VAR(opt_found), argvpos= 0, length;
157  my_bool end_of_options= 0, must_be_var, set_maximum_value,
158  option_is_loose;
159  char **pos, **pos_end, *optend, *opt_str, key_name[FN_REFLEN];
160  const char *UNINIT_VAR(prev_found);
161  const struct my_option *optp;
162  void *value;
163  int error, i;
164  my_bool is_cmdline_arg= 1;
165 
166  /* handle_options() assumes arg0 (program name) always exists */
167  DBUG_ASSERT(argc && *argc >= 1);
168  DBUG_ASSERT(argv && *argv);
169  (*argc)--; /* Skip the program name */
170  (*argv)++; /* --- || ---- */
171  init_variables(longopts, init_one_value);
172 
173  /*
174  Search for args_separator, if found, then the first part of the
175  arguments are loaded from configs
176  */
177  for (pos= *argv, pos_end=pos+ *argc; pos != pos_end ; pos++)
178  {
179  if (*pos == args_separator)
180  {
181  is_cmdline_arg= 0;
182  break;
183  }
184  }
185 
186  for (pos= *argv, pos_end=pos+ *argc; pos != pos_end ; pos++)
187  {
188  char **first= pos;
189  char *cur_arg= *pos;
190  opt_found= 0;
191  if (!is_cmdline_arg && (cur_arg == args_separator))
192  {
193  is_cmdline_arg= 1;
194 
195  /* save the separator too if skip unkown options */
196  if (my_getopt_skip_unknown)
197  (*argv)[argvpos++]= cur_arg;
198  else
199  (*argc)--;
200  continue;
201  }
202  if (cur_arg[0] == '-' && cur_arg[1] && !end_of_options) /* must be opt */
203  {
204  char *argument= 0;
205  must_be_var= 0;
206  set_maximum_value= 0;
207  option_is_loose= 0;
208 
209  cur_arg++; /* skip '-' */
210  if (*cur_arg == '-') /* check for long option, */
211  {
212  if (!*++cur_arg) /* skip the double dash */
213  {
214  /* '--' means end of options, look no further */
215  end_of_options= 1;
216  (*argc)--;
217  continue;
218  }
219  opt_str= check_struct_option(cur_arg, key_name);
220  optend= strcend(opt_str, '=');
221  length= (uint) (optend - opt_str);
222  if (*optend == '=')
223  optend++;
224  else
225  optend= 0;
226 
227  /*
228  Find first the right option. Return error in case of an ambiguous,
229  or unknown option
230  */
231  optp= longopts;
232  if (!(opt_found= findopt(opt_str, length, &optp, &prev_found)))
233  {
234  /*
235  Didn't find any matching option. Let's see if someone called
236  option with a special option prefix
237  */
238  if (!must_be_var)
239  {
240  if (optend)
241  must_be_var= 1; /* option is followed by an argument */
242  for (i= 0; special_opt_prefix[i]; i++)
243  {
244  if (!getopt_compare_strings(special_opt_prefix[i], opt_str,
245  special_opt_prefix_lengths[i]) &&
246  (opt_str[special_opt_prefix_lengths[i]] == '-' ||
247  opt_str[special_opt_prefix_lengths[i]] == '_'))
248  {
249  /*
250  We were called with a special prefix, we can reuse opt_found
251  */
252  opt_str+= special_opt_prefix_lengths[i] + 1;
253  length-= special_opt_prefix_lengths[i] + 1;
254  if (i == OPT_LOOSE)
255  option_is_loose= 1;
256  if ((opt_found= findopt(opt_str, length, &optp, &prev_found)))
257  {
258  if (opt_found > 1)
259  {
260  if (my_getopt_print_errors)
261  my_getopt_error_reporter(ERROR_LEVEL,
262  "%s: ambiguous option '--%s-%s' (--%s-%s)",
263  my_progname, special_opt_prefix[i],
264  opt_str, special_opt_prefix[i],
265  prev_found);
266  return EXIT_AMBIGUOUS_OPTION;
267  }
268  switch (i) {
269  case OPT_SKIP:
270  case OPT_DISABLE: /* fall through */
271  /*
272  double negation is actually enable again,
273  for example: --skip-option=0 -> option = TRUE
274  */
275  optend= (optend && *optend == '0' && !(*(optend + 1))) ?
276  enabled_my_option : disabled_my_option;
277  break;
278  case OPT_ENABLE:
279  optend= (optend && *optend == '0' && !(*(optend + 1))) ?
280  disabled_my_option : enabled_my_option;
281  break;
282  case OPT_MAXIMUM:
283  set_maximum_value= 1;
284  must_be_var= 1;
285  break;
286  }
287  break; /* break from the inner loop, main loop continues */
288  }
289  i= -1; /* restart the loop */
290  }
291  }
292  }
293  if (!opt_found)
294  {
295  if (my_getopt_skip_unknown)
296  {
297  /* Preserve all the components of this unknown option. */
298  do {
299  (*argv)[argvpos++]= *first++;
300  } while (first <= pos);
301  continue;
302  }
303  if (must_be_var)
304  {
305  if (my_getopt_print_errors)
306  my_getopt_error_reporter(option_is_loose ?
307  WARNING_LEVEL : ERROR_LEVEL,
308  "%s: unknown variable '%s'",
309  my_progname, cur_arg);
310  if (!option_is_loose)
311  return EXIT_UNKNOWN_VARIABLE;
312  }
313  else
314  {
315  if (my_getopt_print_errors)
316  my_getopt_error_reporter(option_is_loose ?
317  WARNING_LEVEL : ERROR_LEVEL,
318  "%s: unknown option '--%s'",
319  my_progname, cur_arg);
320  if (!option_is_loose)
321  return EXIT_UNKNOWN_OPTION;
322  }
323  if (option_is_loose)
324  {
325  (*argc)--;
326  continue;
327  }
328  }
329  }
330  if (opt_found > 1)
331  {
332  if (must_be_var)
333  {
334  if (my_getopt_print_errors)
335  my_getopt_error_reporter(ERROR_LEVEL,
336  "%s: variable prefix '%s' is not unique",
337  my_progname, opt_str);
338  return EXIT_VAR_PREFIX_NOT_UNIQUE;
339  }
340  else
341  {
342  if (my_getopt_print_errors)
343  my_getopt_error_reporter(ERROR_LEVEL,
344  "%s: ambiguous option '--%s' (%s, %s)",
345  my_progname, opt_str, prev_found,
346  optp->name);
347  return EXIT_AMBIGUOUS_OPTION;
348  }
349  }
350  if ((optp->var_type & GET_TYPE_MASK) == GET_DISABLED)
351  {
352  if (my_getopt_print_errors)
353  fprintf(stderr,
354  "%s: %s: Option '%s' used, but is disabled\n", my_progname,
355  option_is_loose ? "WARNING" : "ERROR", opt_str);
356  if (option_is_loose)
357  {
358  (*argc)--;
359  continue;
360  }
361  return EXIT_OPTION_DISABLED;
362  }
363  error= 0;
364  value= optp->var_type & GET_ASK_ADDR ?
365  (*getopt_get_addr)(key_name, (uint) strlen(key_name), optp, &error) :
366  optp->value;
367  if (error)
368  return error;
369 
370  if (optp->arg_type == NO_ARG)
371  {
372  /*
373  Due to historical reasons GET_BOOL var_types still accepts arguments
374  despite the NO_ARG arg_type attribute. This can seems a bit unintuitive
375  and care should be taken when refactoring this code.
376  */
377  if (optend && (optp->var_type & GET_TYPE_MASK) != GET_BOOL)
378  {
379  if (my_getopt_print_errors)
380  my_getopt_error_reporter(ERROR_LEVEL,
381  "%s: option '--%s' cannot take an argument",
382  my_progname, optp->name);
383  return EXIT_NO_ARGUMENT_ALLOWED;
384  }
385  if ((optp->var_type & GET_TYPE_MASK) == GET_BOOL)
386  {
387  /*
388  Set bool to 1 if no argument or if the user has used
389  --enable-'option-name'.
390  *optend was set to '0' if one used --disable-option
391  */
392  (*argc)--;
393  if (!optend || *optend == '1' ||
394  !my_strcasecmp(&my_charset_latin1, optend, "true"))
395  *((my_bool*) value)= (my_bool) 1;
396  else if (*optend == '0' ||
397  !my_strcasecmp(&my_charset_latin1, optend, "false"))
398  *((my_bool*) value)= (my_bool) 0;
399  else
400  {
401  my_getopt_error_reporter(WARNING_LEVEL,
402  "%s: ignoring option '--%s' "
403  "due to invalid value '%s'",
404  my_progname, optp->name, optend);
405  continue;
406  }
407  if (get_one_option && get_one_option(optp->id, optp,
408  *((my_bool*) value) ?
409  enabled_my_option : disabled_my_option))
410  return EXIT_ARGUMENT_INVALID;
411  continue;
412  }
413  argument= optend;
414  }
415  else if (optp->arg_type == REQUIRED_ARG && !optend)
416  {
417  /* Check if there are more arguments after this one,
418  Note: options loaded from config file that requires value
419  should always be in the form '--option=value'.
420  */
421  if (!is_cmdline_arg || !*++pos)
422  {
423  if (my_getopt_print_errors)
424  my_getopt_error_reporter(ERROR_LEVEL,
425  "%s: option '--%s' requires an argument",
426  my_progname, optp->name);
427  return EXIT_ARGUMENT_REQUIRED;
428  }
429  argument= *pos;
430  (*argc)--;
431  }
432  else
433  argument= optend;
434  }
435  else /* must be short option */
436  {
437  for (optend= cur_arg; *optend; optend++)
438  {
439  opt_found= 0;
440  for (optp= longopts; optp->name; optp++)
441  {
442  if (optp->id && optp->id == (int) (uchar) *optend)
443  {
444  /* Option recognized. Find next what to do with it */
445  opt_found= 1;
446  if ((optp->var_type & GET_TYPE_MASK) == GET_DISABLED)
447  {
448  if (my_getopt_print_errors)
449  fprintf(stderr,
450  "%s: ERROR: Option '-%c' used, but is disabled\n",
451  my_progname, optp->id);
452  return EXIT_OPTION_DISABLED;
453  }
454  if ((optp->var_type & GET_TYPE_MASK) == GET_BOOL &&
455  optp->arg_type == NO_ARG)
456  {
457  *((my_bool*) optp->value)= (my_bool) 1;
458  if (get_one_option && get_one_option(optp->id, optp, argument))
459  return EXIT_UNSPECIFIED_ERROR;
460  continue;
461  }
462  else if (optp->arg_type == REQUIRED_ARG ||
463  optp->arg_type == OPT_ARG)
464  {
465  if (*(optend + 1))
466  {
467  /* The rest of the option is option argument */
468  argument= optend + 1;
469  /* This is in effect a jump out of the outer loop */
470  optend= (char*) " ";
471  }
472  else
473  {
474  if (optp->arg_type == OPT_ARG)
475  {
476  if (optp->var_type == GET_BOOL)
477  *((my_bool*) optp->value)= (my_bool) 1;
478  if (get_one_option && get_one_option(optp->id, optp, argument))
479  return EXIT_UNSPECIFIED_ERROR;
480  continue;
481  }
482  /* Check if there are more arguments after this one */
483  if (!pos[1])
484  {
485  if (my_getopt_print_errors)
486  my_getopt_error_reporter(ERROR_LEVEL,
487  "%s: option '-%c' requires an argument",
488  my_progname, optp->id);
489  return EXIT_ARGUMENT_REQUIRED;
490  }
491  argument= *++pos;
492  (*argc)--;
493  /* the other loop will break, because *optend + 1 == 0 */
494  }
495  }
496  if ((error= setval(optp, optp->value, argument,
497  set_maximum_value)))
498  return error;
499  if (get_one_option && get_one_option(optp->id, optp, argument))
500  return EXIT_UNSPECIFIED_ERROR;
501  break;
502  }
503  }
504  if (!opt_found)
505  {
506  if (my_getopt_skip_unknown)
507  {
508  /*
509  We are currently parsing a single argv[] argument
510  of the form "-XYZ".
511  One or the argument found (say Y) is not an option.
512  Hack the string "-XYZ" to make a "-YZ" substring in it,
513  and push that to the output as an unrecognized parameter.
514  */
515  DBUG_ASSERT(optend > *pos);
516  DBUG_ASSERT(optend >= cur_arg);
517  DBUG_ASSERT(optend <= *pos + strlen(*pos));
518  DBUG_ASSERT(*optend);
519  optend--;
520  optend[0]= '-'; /* replace 'X' or '-' by '-' */
521  (*argv)[argvpos++]= optend;
522  /*
523  Do not continue to parse at the current "-XYZ" argument,
524  skip to the next argv[] argument instead.
525  */
526  optend= (char*) " ";
527  }
528  else
529  {
530  if (my_getopt_print_errors)
531  my_getopt_error_reporter(ERROR_LEVEL,
532  "%s: unknown option '-%c'",
533  my_progname, *optend);
534  return EXIT_UNKNOWN_OPTION;
535  }
536  }
537  }
538  if (opt_found)
539  (*argc)--; /* option handled (short), decrease argument count */
540  continue;
541  }
542  if ((error= setval(optp, value, argument, set_maximum_value)))
543  return error;
544  if (get_one_option && get_one_option(optp->id, optp, argument))
545  return EXIT_UNSPECIFIED_ERROR;
546 
547  (*argc)--; /* option handled (long), decrease argument count */
548  }
549  else /* non-option found */
550  (*argv)[argvpos++]= cur_arg;
551  }
552  /*
553  Destroy the first, already handled option, so that programs that look
554  for arguments in 'argv', without checking 'argc', know when to stop.
555  Items in argv, before the destroyed one, are all non-option -arguments
556  to the program, yet to be (possibly) handled.
557  */
558  (*argv)[argvpos]= 0;
559  return 0;
560 }

563 /*
564  function: check_struct_option
565 
566  Arguments: Current argument under processing from argv and a variable
567  where to store the possible key name.
568 
569  Return value: In case option is a struct option, returns a pointer to
570  the current argument at the position where the struct option (key_name)
571  ends, the next character after the dot. In case argument is not a struct
572  option, returns a pointer to the argument.
573 
574  key_name will hold the name of the key, or 0 if not found.
575 */
576 
577 static char *check_struct_option(char *cur_arg, char *key_name)
578 {
579  char *ptr, *end;
580 
581  ptr= strcend(cur_arg + 1, '.'); /* Skip the first character */
582  end= strcend(cur_arg, '=');
583 
584  /*
585  If the first dot is after an equal sign, then it is part
586  of a variable value and the option is not a struct option.
587  Also, if the last character in the string before the ending
588  NULL, or the character right before equal sign is the first
589  dot found, the option is not a struct option.
590  */
591  if (end - ptr > 1)
592  {
593  uint len= (uint) (ptr - cur_arg);
594  set_if_smaller(len, FN_REFLEN-1);
595  strmake(key_name, cur_arg, len);
596  return ++ptr;
597  }
598  else
599  {
600  key_name[0]= 0;
601  return cur_arg;
602  }
603 }

623 /*
624  function: setval
625 
626  Arguments: opts, argument
627  Will set the option value to given value
628 */
629 
630 static int setval(const struct my_option *opts, void *value, char *argument,
631  my_bool set_maximum_value)
632 {
633  int err= 0, res= 0;
634 
635  if (!argument)
636  argument= enabled_my_option;
637 
638  if (value)
639  {
640  if (set_maximum_value && !(value= opts->u_max_value))
641  {
642  my_getopt_error_reporter(ERROR_LEVEL,
643  "%s: Maximum value of '%s' cannot be set",
644  my_progname, opts->name);
645  return EXIT_NO_PTR_TO_VARIABLE;
646  }
647 
648  switch ((opts->var_type & GET_TYPE_MASK)) {
649  case GET_BOOL: /* If argument differs from 0, enable option, else disable */
650  *((my_bool*) value)= get_bool_argument(argument);
651  break;
652  case GET_INT:
653  *((int*) value)= (int) getopt_ll(argument, opts, &err);
654  break;
655  case GET_UINT:
656  *((uint*) value)= (uint) getopt_ull(argument, opts, &err);
657  break;
658  case GET_LONG:
659  *((long*) value)= (long) getopt_ll(argument, opts, &err);
660  break;
661  case GET_ULONG:
662  *((long*) value)= (long) getopt_ull(argument, opts, &err);
663  break;
664  case GET_LL:
665  *((longlong*) value)= getopt_ll(argument, opts, &err);
666  break;
667  case GET_ULL:
668  *((ulonglong*) value)= getopt_ull(argument, opts, &err);
669  break;
670  case GET_DOUBLE:
671  *((double*) value)= getopt_double(argument, opts, &err);
672  break;
673  case GET_STR:
674  if (argument == enabled_my_option)
675  break; /* string options don't use this default of "1" */
676  *((char**) value)= argument;
677  break;
678  case GET_STR_ALLOC:
679  if (argument == enabled_my_option)
680  break; /* string options don't use this default of "1" */
681  my_free(*((char**) value));
682  if (!(*((char**) value)= my_strdup(argument, MYF(MY_WME))))
683  {
684  res= EXIT_OUT_OF_MEMORY;
685  goto ret;
686  };
687  break;
688  case GET_ENUM:
689  {
690  int type= find_type(argument, opts->typelib, 2);
691  if (type == 0)
692  {
693  /*
694  Accept an integer representation of the enumerated item.
695  */
696  char *endptr;
697  ulong arg= strtoul(argument, &endptr, 10);
698  if (*endptr || arg >= opts->typelib->count)
699  {
700  res= EXIT_ARGUMENT_INVALID;
701  goto ret;
702  }
703  *(ulong*)value= arg;
704  }
705  else
706  *(ulong*)value= type - 1;
707  }
708  break;
709  case GET_SET:
710  *((ulonglong*)value)= find_typeset(argument, opts->typelib, &err);
711  if (err)
712  {
713  /* Accept an integer representation of the set */
714  char *endptr;
715  ulonglong arg= (ulonglong) strtol(argument, &endptr, 10);
716  if (*endptr || (arg >> 1) >= (1ULL << (opts->typelib->count-1)))
717  {
718  res= EXIT_ARGUMENT_INVALID;
719  goto ret;
720  };
721  *(ulonglong*)value= arg;
722  err= 0;
723  }
724  break;
725  case GET_FLAGSET:
726  {
727  char *error;
728  uint error_len;
729 
730  *((ulonglong*)value)=
731  find_set_from_flags(opts->typelib, opts->typelib->count,
732  *(ulonglong *)value, opts->def_value,
733  argument, strlen(argument),
734  &error, &error_len);
735  if (error)
736  {
737  res= EXIT_ARGUMENT_INVALID;
738  goto ret;
739  };
740  }
741  break;
742  case GET_NO_ARG: /* get_one_option has taken care of the value already */
