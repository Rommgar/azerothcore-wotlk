55  void Update(uint32 diff)
56  {
57  if (diff < 1)
58  diff = 1;
59  total -= tab[index];
60  total += diff;
61  tab[index] = diff;
62  if (diff > max[0])
63  max[0] = diff;
64  if (++index >= AVG_DIFF_COUNT)
65  {
66  index = 0;
67  max[1] = max[0];
68  max[0] = 0;
69  }
70 
71  if (tab[AVG_DIFF_COUNT - 1])
72  average = total / AVG_DIFF_COUNT;
73  else if (index)
74  average = total / index;
75  else
76  average = 0;
77  }
