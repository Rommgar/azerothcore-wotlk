144  void UpdateAI(uint32 diff) override
145  {
146  if (!UpdateVictim())
147  return;
148 
149  events.Update(diff);
150  if (me->HasUnitState(UNIT_STATE_CASTING))
151  return;
152 
153  switch (events.ExecuteEvent())
154  {
155  case EVENT_ANOMALUS_SPARK:
156  me->CastSpell(me->GetVictim(), SPELL_SPARK, false);
157  events.ScheduleEvent(EVENT_ANOMALUS_SPARK, 5000);
158  break;
159  case EVENT_ANOMALUS_HEALTH:
160  if (me->HealthBelowPct(51))
161  {
162  //First time we reach 51%, the next rift going to be empowered following timings.
163  events.CancelEvent(EVENT_ANOMALUS_SPAWN_RIFT);
164  events.ScheduleEvent(EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED, 1000);
165  break;
166  }
167  events.ScheduleEvent(EVENT_ANOMALUS_HEALTH, 1000);
168  break;
169  case EVENT_ANOMALUS_ARCANE_ATTRACTION:
170  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 50.0f, true))
171  me->CastSpell(target, SPELL_ARCANE_ATTRACTION, false);
172  events.ScheduleEvent(EVENT_ANOMALUS_ARCANE_ATTRACTION, 15000);
173  break;
174  case EVENT_ANOMALUS_SPAWN_RIFT:
175  Talk(SAY_RIFT);
176  Talk(EMOTE_RIFT);
177  me->CastSpell(me, SPELL_CREATE_RIFT, false);
178  //Once we hit 51% hp mark, after each rift we spawn an empowered
179  events.ScheduleEvent(me->HealthBelowPct(51) ? EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED : EVENT_ANOMALUS_SPAWN_RIFT, IsHeroic() ? 15000 : 25000);
180  break;
181  case EVENT_ANOMALUS_SPAWN_RIFT_EMPOWERED:
182  Talk(SAY_RIFT);
183  Talk(EMOTE_RIFT);
184 
185  me->CastSpell(me, SPELL_CREATE_RIFT, false);
186  me->CastSpell(me, SPELL_RIFT_SHIELD, true);
187  me->m_Events.AddEvent(new ChargeRifts(me), me->m_Events.CalculateTime(1000));
188  events.DelayEvents(46000);
189  //As we just spawned an empowered spawn a normal one
190  events.ScheduleEvent(EVENT_ANOMALUS_SPAWN_RIFT, IsHeroic() ? 15000 : 25000);
191  break;
192  }
193 
194  DoMeleeAttackIfReady();
195  EnterEvadeIfOutOfCombatArea();
196  }
