125  bool SetBossState(uint32 id, EncounterState state) override
126  {
127  if (!InstanceScript::SetBossState(id, state))
128  return false;
129 
130  if (id == TYPE_MAGTHERIDON)
131  {
132  if (state == IN_PROGRESS)
133  {
134  for (ObjectGuid const& guid : _wardersSet)
135  if (Creature* warder = instance->GetCreature(guid))
136  if (warder->IsAlive())
137  {

140  }

145  if (GameObject* cube = instance->GetGameObject(guid))
146  cube->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NOT_SELECTABLE);
147

201  loadStream >> dataHead1 >> dataHead2;
202 
203  if (dataHead1 == 'M' && dataHead2 == 'L')
204  {
205  for (uint32 i = 0; i < MAX_ENCOUNTER; ++i)
206  {
207  uint32 tmpState;
208  loadStream >> tmpState;
209  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
210  tmpState = NOT_STARTED;
211  SetBossState(i, EncounterState(tmpState));
212  }
213  }
