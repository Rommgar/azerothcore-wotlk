41 {
42  if (me->HasUnitState(UNIT_STATE_CASTING))
43  return;
44 
45  Unit* victim = me->GetVictim();
46  if (!victim || !victim->IsInWorld())
47  return;
48 
49  if (!me->IsWithinMeleeRange(victim))
50  return;
51 
52  //Make sure our attack is ready and we aren't currently casting before checking distance
53  if (me->isAttackReady())
54  {
55  // xinef: prevent base and off attack in same time, delay attack at 0.2 sec

58  me->setAttackTimer(OFF_ATTACK, ATTACK_DISPLAY_DELAY);
59 
60  me->AttackerStateUpdate(victim);
61  me->resetAttackTimer();
62  }
63

137 void UnitAI::DoCast(uint32 spellId)
138 {
139  Unit* target = nullptr;
140 
141  switch (AISpellInfo[spellId].target)
142  {
143  default:
144  case AITARGET_SELF:
145  target = me;
146  break;
147  case AITARGET_VICTIM:
148  target = me->GetVictim();
149  break;
150  case AITARGET_ENEMY:
151  {
152  const SpellInfo* spellInfo = sSpellMgr->GetSpellInfo(spellId);
153  bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_ON_PLAYER);
154  //float range = GetSpellMaxRange(spellInfo, false);
155  target = SelectTarget(SELECT_TARGET_RANDOM, 0, spellInfo->GetMaxRange(false), playerOnly);
156  break;
157  }
158  case AITARGET_ALLY:
159  target = me;
160  break;
161  case AITARGET_BUFF:
162  target = me;
163  break;
164  case AITARGET_DEBUFF:
165  {
166  const SpellInfo* spellInfo = sSpellMgr->GetSpellInfo(spellId);
167  bool playerOnly = spellInfo->HasAttribute(SPELL_ATTR3_ONLY_ON_PLAYER);
168  float range = spellInfo->GetMaxRange(false);
169 
170  DefaultTargetSelector targetSelector(me, range, playerOnly, -(int32)spellId);
171  if (!(spellInfo->AuraInterruptFlags & AURA_INTERRUPT_FLAG_NOT_VICTIM)
172  && targetSelector(me->GetVictim()))
173  target = me->GetVictim();
174  else
175  target = SelectTarget(SELECT_TARGET_RANDOM, 0, targetSelector);
176  break;
177  }
178  }
179 
180  if (target)
181  me->CastSpell(target, spellId, false);
182 }

223 void UnitAI::FillAISpellInfo()
224 {
225  AISpellInfo = new AISpellInfoType[sSpellMgr->GetSpellInfoStoreSize()];
226 
227  AISpellInfoType* AIInfo = AISpellInfo;
228  const SpellInfo* spellInfo;
229 
230  for (uint32 i = 0; i < sSpellMgr->GetSpellInfoStoreSize(); ++i, ++AIInfo)
231  {
232  spellInfo = sSpellMgr->GetSpellInfo(i);
233  if (!spellInfo)
234  continue;
235 
236  if (spellInfo->HasAttribute(SPELL_ATTR0_ALLOW_CAST_WHILE_DEAD))
237  AIInfo->condition = AICOND_DIE;
238  else if (spellInfo->IsPassive() || spellInfo->GetDuration() == -1)
239  AIInfo->condition = AICOND_AGGRO;
240  else
241  AIInfo->condition = AICOND_COMBAT;
242 
243  if (AIInfo->cooldown < spellInfo->RecoveryTime)
244  AIInfo->cooldown = spellInfo->RecoveryTime;
245 
246  if (!spellInfo->GetMaxRange(false))
247  UPDATE_TARGET(AITARGET_SELF)
248  else
249  {
250  for (uint32 j = 0; j < MAX_SPELL_EFFECTS; ++j)
251  {
252  uint32 targetType = spellInfo->Effects[j].TargetA.GetTarget();
253 
254  if (targetType == TARGET_UNIT_TARGET_ENEMY
255  || targetType == TARGET_DEST_TARGET_ENEMY)
256  UPDATE_TARGET(AITARGET_VICTIM)
257  else if (targetType == TARGET_UNIT_DEST_AREA_ENEMY)
258  UPDATE_TARGET(AITARGET_ENEMY)
259 
260  if (spellInfo->Effects[j].Effect == SPELL_EFFECT_APPLY_AURA)
261  {
262  if (targetType == TARGET_UNIT_TARGET_ENEMY)
263  UPDATE_TARGET(AITARGET_DEBUFF)
264  else if (spellInfo->IsPositive())
265  UPDATE_TARGET(AITARGET_BUFF)
266  }
267  }
268  }
269  AIInfo->realCooldown = spellInfo->RecoveryTime + spellInfo->StartRecoveryTime;
270  AIInfo->maxRange = spellInfo->GetMaxRange(false) * 3 / 4;
271  }
272 }

310 bool SpellTargetSelector::operator()(Unit const* target) const
311 {
312  if (!target)
313  return false;
314 
315  if (_spellInfo->CheckTarget(_caster, target) != SPELL_CAST_OK)
316  return false;
317 
318  // copypasta from Spell::CheckRange
319  uint32 range_type = _spellInfo->RangeEntry ? _spellInfo->RangeEntry->type : 0;
320  float max_range = _caster->GetSpellMaxRangeForTarget(target, _spellInfo);
321  float min_range = _caster->GetSpellMinRangeForTarget(target, _spellInfo);
322 
323  if (target && target != _caster)
324  {
325  if (range_type == SPELL_RANGE_MELEE)
326  {
327  // Because of lag, we can not check too strictly here.
328  if (!_caster->IsWithinMeleeRange(target, max_range))
329  return false;
330  }
331  else if (!_caster->IsWithinCombatRange(target, max_range))
332  return false;
333 
334  if (range_type == SPELL_RANGE_RANGED)
335  {
336  if (_caster->IsWithinMeleeRange(target))
337  return false;
338  }
339  else if (min_range && _caster->IsWithinCombatRange(target, min_range)) // skip this check if min_range = 0
340  return false;
341  }
342 
343  return true;
344 }
