39 void WorldSession::HandleAddFriendOpcode(WorldPacket& recv_data)
40 {
41  LOG_DEBUG("network", "WORLD: Received CMSG_ADD_FRIEND");
42 
43  std::string friendName = GetAcoreString(LANG_FRIEND_IGNORE_UNKNOWN);
44  std::string friendNote;
45 
46  recv_data >> friendName;
47  recv_data >> friendNote;
48 
49  if (!normalizePlayerName(friendName))
50  return;
51 
52  LOG_DEBUG("network", "WORLD: %s asked to add friend : '%s'", GetPlayer()->GetName().c_str(), friendName.c_str());
53 
54  // xinef: Get Data From global storage
55  ObjectGuid friendGuid = sWorld->GetGlobalPlayerGUID(friendName);
56  if (!friendGuid)
57  return;
58 
59  GlobalPlayerData const* playerData = sWorld->GetGlobalPlayerData(friendGuid.GetCounter());
60  if (!playerData)
61  return;
62 
63  uint32 friendAccountId = playerData->accountId;
64  TeamId teamId = Player::TeamIdForRace(playerData->race);
65  FriendsResult friendResult = FRIEND_NOT_FOUND;
66 
67  if (!AccountMgr::IsPlayerAccount(GetSecurity()) || sWorld->getBoolConfig(CONFIG_ALLOW_GM_FRIEND)|| AccountMgr::IsPlayerAccount(AccountMgr::GetSecurity(friendAccountId, realm.Id.Realm)))
68  {
69  if (friendGuid)
70  {
71  if (friendGuid == GetPlayer()->GetGUID())
72  friendResult = FRIEND_SELF;
73  else if (GetPlayer()->GetTeamId() != teamId && !sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_ADD_FRIEND) && AccountMgr::IsPlayerAccount(GetSecurity()))
74  friendResult = FRIEND_ENEMY;
75  else if (GetPlayer()->GetSocial()->HasFriend(friendGuid))
76  friendResult = FRIEND_ALREADY;
77  else
78  {
79  Player* pFriend = ObjectAccessor::FindConnectedPlayer(friendGuid);
80  if (pFriend && pFriend->IsVisibleGloballyFor(GetPlayer()) && !AccountMgr::IsGMAccount(pFriend->GetSession()->GetSecurity()))
81  friendResult = FRIEND_ADDED_ONLINE;
82  else
83  friendResult = FRIEND_ADDED_OFFLINE;
84  if (GetPlayer()->GetSocial()->AddToSocialList(friendGuid, SOCIAL_FLAG_FRIEND))
85  GetPlayer()->GetSocial()->SetFriendNote(friendGuid, friendNote);
86  else
87  friendResult = FRIEND_LIST_FULL;
88  }
89  GetPlayer()->GetSocial()->SetFriendNote(friendGuid, friendNote);
90  }
91  }
92 
93  sSocialMgr->SendFriendStatus(GetPlayer(), friendResult, friendGuid, false);
94 
95  LOG_DEBUG("network", "WORLD: Sent (SMSG_FRIEND_STATUS)");
96 }
