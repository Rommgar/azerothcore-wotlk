42 void VisibleNotifier::SendToSelf()
43 {
44  // at this moment i_clientGUIDs have guids that not iterate at grid level checks
45  // but exist one case when this possible and object not out of range: transports
46  if (Transport* transport = i_player.GetTransport())
47  for (Transport::PassengerSet::const_iterator itr = transport->GetPassengers().begin(); itr != transport->GetPassengers().end(); ++itr)
48  {
49  if (i_largeOnly != (*itr)->IsVisibilityOverridden())
50  continue;
51 
52  if (vis_guids.find((*itr)->GetGUID()) != vis_guids.end())
53  {
54  vis_guids.erase((*itr)->GetGUID());
55 
56  switch ((*itr)->GetTypeId())
57  {
58  case TYPEID_GAMEOBJECT:
59  i_player.UpdateVisibilityOf((*itr)->ToGameObject(), i_data, i_visibleNow);
60  break;
61  case TYPEID_PLAYER:
62  i_player.UpdateVisibilityOf((*itr)->ToPlayer(), i_data, i_visibleNow);
63  (*itr)->ToPlayer()->UpdateVisibilityOf(&i_player);
64  break;
65  case TYPEID_UNIT:
66  i_player.UpdateVisibilityOf((*itr)->ToCreature(), i_data, i_visibleNow);
67  break;
68  default:
69  break;
70  }
71  }
72  }
73 
74  for (GuidUnorderedSet::const_iterator it = vis_guids.begin(); it != vis_guids.end(); ++it)
75  {
76  if (WorldObject* obj = ObjectAccessor::GetWorldObject(i_player, *it))
77  {
78  if (i_largeOnly != obj->IsVisibilityOverridden())
79  continue;
80  }
81 
82  // pussywizard: static transports are removed only in RemovePlayerFromMap and here if can no longer detect (eg. phase changed)
83  if ((*it).IsTransport())
84  if (GameObject* staticTrans = i_player.GetMap()->GetGameObject(*it))
85  if (i_player.CanSeeOrDetect(staticTrans, false, true))
86  continue;
87 
88  i_player.m_clientGUIDs.erase(*it);
89  i_data.AddOutOfRangeGUID(*it);
90 
91  if ((*it).IsPlayer())
92  {
93  Player* player = ObjectAccessor::FindPlayer(*it);
94  if (player && player->IsInMap(&i_player))
95  player->UpdateVisibilityOf(&i_player);
96  }
97  }
98 
99  if (!i_data.HasData())
100  return;
101 
102  WorldPacket packet;
103  i_data.BuildPacket(&packet);
104  i_player.GetSession()->SendPacket(&packet);
105 
106  for (std::vector<Unit*>::const_iterator it = i_visibleNow.begin(); it != i_visibleNow.end(); ++it)
107  {
108  if (i_largeOnly != (*it)->IsVisibilityOverridden())
109  continue;
110 
111  i_player.GetInitialVisiblePackets(*it);
112  }
113 }

150 inline void CreatureUnitRelocationWorker(Creature* c, Unit* u)
151 {
152  if (!u->IsAlive() || !c->IsAlive() || c == u || u->IsInFlight())
153  {
154  return;
155  }
156 
157  if (!c->HasUnitState(UNIT_STATE_SIGHTLESS))
158  {
159  if (c->IsAIEnabled && c->CanSeeOrDetect(u, false, true))
160  {
161  c->AI()->MoveInLineOfSight_Safe(u);
162  }
163  else if (u->GetTypeId() == TYPEID_PLAYER && u->HasStealthAura() && c->IsAIEnabled && c->CanSeeOrDetect(u, false, true, true))
164  {
165  c->AI()->TriggerAlert(u);
166  }
167  }
168 }

181 void CreatureRelocationNotifier::Visit(PlayerMapType& m)
182 {
183  for (PlayerMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
184  {
185  Player* player = iter->GetSource();
186 
187  // NOTIFY_VISIBILITY_CHANGED does not guarantee that player will do it himself (because distance is also checked), but screw it, it's not that important
188  if (!player->m_seer->isNeedNotify(NOTIFY_VISIBILITY_CHANGED))
189  player->UpdateVisibilityOf(&i_creature);
190 
191  // NOTIFY_AI_RELOCATION does not guarantee that player will do it himself (because distance is also checked), but screw it, it's not that important
192  if (!player->m_seer->isNeedNotify(NOTIFY_AI_RELOCATION) && !i_creature.IsMoveInLineOfSightStrictlyDisabled())
193  CreatureUnitRelocationWorker(&i_creature, player);
194  }

197 void AIRelocationNotifier::Visit(CreatureMapType& m)
198 {
199  bool self = isCreature && !((Creature*)(&i_unit))->IsMoveInLineOfSightStrictlyDisabled();
200  for (CreatureMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
201  {
202  Creature* c = iter->GetSource();
203 
204  // NOTIFY_VISIBILITY_CHANGED | NOTIFY_AI_RELOCATION does not guarantee that unit will do it itself (because distance is also checked), but screw it, it's not that important
205  if (!c->isNeedNotify(NOTIFY_VISIBILITY_CHANGED | NOTIFY_AI_RELOCATION) && !c->IsMoveInLineOfSightStrictlyDisabled())
206  CreatureUnitRelocationWorker(c, &i_unit);
207 
208  if (self)
209  CreatureUnitRelocationWorker((Creature*)&i_unit, c);
210  }

213 void MessageDistDeliverer::Visit(PlayerMapType& m)
214 {
215  for (PlayerMapType::iterator iter = m.begin(); iter != m.end(); ++iter)
216  {
217  Player* target = iter->GetSource();
218  if (!target->InSamePhase(i_phaseMask))
219  continue;
220 
221  if (target->GetExactDist2dSq(i_source) > i_distSq)
222  continue;
223 
224  // Send packet to all who are sharing the player's vision
225  if (target->HasSharedVision())
226  {
227  SharedVisionList::const_iterator i = target->GetSharedVisionList().begin();
228  for (; i != target->GetSharedVisionList().end(); ++i)
229  if ((*i)->m_seer == target)
230  SendPacket(*i);
231  }
232 
233  if (target->m_seer == target || target->GetVehicle())
234  SendPacket(target);
