80 // returns true if we kicked more than requested
81 bool BattlegroundQueue::SelectionPool::KickGroup(const uint32 size)
82 {
83  if (SelectedGroups.empty())
84  return false;
85 
86  // find last group with proper size or largest
87  bool foundProper = false;
88  auto groupToKick = SelectedGroups.begin();
89  for (auto itr = groupToKick; itr != SelectedGroups.end(); ++itr)
90  {
91  // if proper size - overwrite to kick last one
92  if (abs(int32((*itr)->Players.size()) - (int32)size) <= 1)
93  {
94  groupToKick = itr;
95  foundProper = true;
96  }
97  else if (!foundProper && (*itr)->Players.size() >= (*groupToKick)->Players.size())
98  groupToKick = itr;
99  }
100 
101  // remove selected from pool
102  GroupQueueInfo* ginfo = (*groupToKick);
103  SelectedGroups.erase(groupToKick);
104  PlayerCount -= ginfo->Players.size();
105 
106  if (foundProper)
107  return false;
108 
109  return (ginfo->Players.size() > size);
110 }

258 void BattlegroundQueue::RemovePlayer(ObjectGuid guid, bool sentToBg, uint32 playerQueueSlot)
259 {
260  // pussywizard: leave queue packet
261  if (playerQueueSlot < PLAYER_MAX_BATTLEGROUND_QUEUES)
262  if (Player* p = ObjectAccessor::FindConnectedPlayer(guid))
263  {
264  WorldPacket data;
265  sBattlegroundMgr->BuildBattlegroundStatusPacket(&data, nullptr, playerQueueSlot, STATUS_NONE, 0, 0, 0, TEAM_NEUTRAL);
266  p->GetSession()->SendPacket(&data);
267  }
268 
269  //remove player from map, if he's there
270  auto itr = m_QueuedPlayers.find(guid);
271  if (itr == m_QueuedPlayers.end())
272  {
273  ABORT();
274  return;
275  }
276 
277  GroupQueueInfo* groupInfo = itr->second;
278 
279  uint32 _bracketId = groupInfo->_bracketId;
280  uint32 _groupType = groupInfo->_groupType;
281 
282  // find iterator
283  auto group_itr = m_QueuedGroups[_bracketId][_groupType].end();
284  for (auto k = m_QueuedGroups[_bracketId][_groupType].begin(); k != m_QueuedGroups[_bracketId][_groupType].end(); ++k)
285  if ((*k) == groupInfo)
286  {
287  group_itr = k;
288  break;
289  }
290 
291  //player can't be in queue without group, but just in case
292  if (group_itr == m_QueuedGroups[_bracketId][_groupType].end())
293  {
294  ABORT();
295  return;
296  }
297 
298  // remove player from group queue info
299  auto pitr = groupInfo->Players.find(guid);
300  ASSERT(pitr != groupInfo->Players.end());
301  if (pitr != groupInfo->Players.end())
302  groupInfo->Players.erase(pitr);
303 
304  // if invited to bg, then always decrease invited count when removed from queue
305  // sending player to bg will increase it again
306  if (groupInfo->IsInvitedToBGInstanceGUID)
307  if (Battleground* bg = sBattlegroundMgr->GetBattleground(groupInfo->IsInvitedToBGInstanceGUID))
308  bg->DecreaseInvitedCount(groupInfo->teamId);
309 
310  // remove player queue info
311  m_QueuedPlayers.erase(itr);
312 
313  // announce to world if arena team left queue for rated match, show only once
314  SendExitMessageArenaQueue(groupInfo);
315 
316  // if player leaves queue and he is invited to a rated arena match, then count it as he lost
317  if (groupInfo->IsInvitedToBGInstanceGUID && groupInfo->IsRated && !sentToBg)
318  if (ArenaTeam* at = sArenaTeamMgr->GetArenaTeamById(groupInfo->ArenaTeamId))
319  {
320  if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
321  at->MemberLost(player, groupInfo->OpponentsMatchmakerRating);
322  at->SaveToDB();
323  }
324 
325  // remove group queue info no players left
326  if (groupInfo->Players.empty())
327  {
328  m_QueuedGroups[_bracketId][_groupType].erase(group_itr);
329  delete groupInfo;
330  return;
331  }
332 
333  // group isn't yet empty, so not deleted yet
334  // if it's a rated arena and any member leaves when group not yet invited - everyone from group leaves too!
335  if (groupInfo->IsRated && !groupInfo->IsInvitedToBGInstanceGUID)
336  {
337  uint32 queueSlot = PLAYER_MAX_BATTLEGROUND_QUEUES;
338 
339  if (Player* plr = ObjectAccessor::FindConnectedPlayer(*(groupInfo->Players.begin())))
340  {
341  BattlegroundQueueTypeId bgQueueTypeId = BattlegroundMgr::BGQueueTypeId(groupInfo->BgTypeId, groupInfo->ArenaType);
342  queueSlot = plr->GetBattlegroundQueueIndex(bgQueueTypeId);
343  plr->RemoveBattlegroundQueueId(bgQueueTypeId);
344  }
345 
346  // recursive call
347  RemovePlayer(*(groupInfo->Players.begin()), false, queueSlot);
348  }
349 }

379 void BattlegroundQueue::FillPlayersToBG(Battleground* bg, const int32 aliFree, const int32 hordeFree, BattlegroundBracketId bracket_id)
380 {
381  if (!sScriptMgr->CanFillPlayersToBG(this, bg, aliFree, hordeFree, bracket_id))
382  return;
383 
384  // clear selection pools
385  m_SelectionPools[TEAM_ALLIANCE].Init();
386  m_SelectionPools[TEAM_HORDE].Init();
387 
388  // quick check if nothing we can do:
389  if (!sBattlegroundMgr->isTesting())
390  if ((aliFree > hordeFree && m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_ALLIANCE].empty()) ||
391  (hordeFree > aliFree && m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_HORDE].empty()))
392  return;
393 
394  // ally: at first fill as much as possible
395  auto Ali_itr = m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_ALLIANCE].begin();
396  for (; Ali_itr != m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_ALLIANCE].end() && m_SelectionPools[TEAM_ALLIANCE].AddGroup((*Ali_itr), aliFree); ++Ali_itr);
397 
398  // horde: at first fill as much as possible
399  auto Horde_itr = m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_HORDE].begin();
400  for (; Horde_itr != m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_HORDE].end() && m_SelectionPools[TEAM_HORDE].AddGroup((*Horde_itr), hordeFree); ++Horde_itr);
401 
402  // calculate free space after adding
403  int32 aliDiff = aliFree - int32(m_SelectionPools[TEAM_ALLIANCE].GetPlayerCount());
404  int32 hordeDiff = hordeFree - int32(m_SelectionPools[TEAM_HORDE].GetPlayerCount());
405 
406  int32 invType = sWorld->getIntConfig(CONFIG_BATTLEGROUND_INVITATION_TYPE);
407  int32 invDiff = 0;
408 
409  // check balance configuration and set the max difference between teams
410  switch (invType)
411  {
412  case BG_QUEUE_INVITATION_TYPE_NO_BALANCE:
413  return;
414  case BG_QUEUE_INVITATION_TYPE_BALANCED:
415  invDiff = 1;
416  break;
417  case BG_QUEUE_INVITATION_TYPE_EVEN:
418  break;
419  default:
420  return;
421  }
422 
423  // balance the teams based on the difference allowed
424  while (abs(aliDiff - hordeDiff) > invDiff && (m_SelectionPools[TEAM_HORDE].GetPlayerCount() > 0 || m_SelectionPools[TEAM_ALLIANCE].GetPlayerCount() > 0))
425  {
426  // if results in more alliance players than horde:
427  if (aliDiff < hordeDiff)
428  {
429  // no more alliance in pool, invite whatever we can from horde
430  if (!m_SelectionPools[TEAM_ALLIANCE].GetPlayerCount())
431  break;
432 
433  // kick alliance, returns true if kicked more than needed, so then try to fill up
434  if (m_SelectionPools[TEAM_ALLIANCE].KickGroup(hordeDiff - aliDiff))
435  for (; Ali_itr != m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_ALLIANCE].end() && m_SelectionPools[TEAM_ALLIANCE].AddGroup((*Ali_itr), aliFree >= hordeDiff ? aliFree - hordeDiff : 0); ++Ali_itr);
436  }
437  // if results in more horde players than alliance:
438  else
439  {
440  // no more horde in pool, invite whatever we can from alliance
441  if (!m_SelectionPools[TEAM_HORDE].GetPlayerCount())
442  break;
443 
444  // kick horde, returns true if kicked more than needed, so then try to fill up
445  if (m_SelectionPools[TEAM_HORDE].KickGroup(aliDiff - hordeDiff))
446  for (; Horde_itr != m_QueuedGroups[bracket_id][BG_QUEUE_NORMAL_HORDE].end() && m_SelectionPools[TEAM_HORDE].AddGroup((*Horde_itr), hordeFree >= aliDiff ? hordeFree - aliDiff : 0); ++Horde_itr);
447  }
448 
449  // recalculate free space after adding
450  aliDiff = aliFree - static_cast<int32>(m_SelectionPools[TEAM_ALLIANCE].GetPlayerCount());
451  hordeDiff = hordeFree - static_cast<int32>(m_SelectionPools[TEAM_HORDE].GetPlayerCount());
452  }
