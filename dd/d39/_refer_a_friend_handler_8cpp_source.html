24 {

30  Player* target = ObjectAccessor::GetPlayer(*_player, guid);
31 
32  // check cheating
33  uint8 levels = _player->GetGrantableLevels();
34  uint8 error = 0;
35  if (!target)
36  error = ERR_REFER_A_FRIEND_NO_TARGET;
37  else if (levels == 0)
38  error = ERR_REFER_A_FRIEND_INSUFFICIENT_GRANTABLE_LEVELS;
39  else if (GetRecruiterId() != target->GetSession()->GetAccountId())
40  error = ERR_REFER_A_FRIEND_NOT_REFERRED_BY;
41  else if (target->GetTeamId() != _player->GetTeamId())
42  error = ERR_REFER_A_FRIEND_DIFFERENT_FACTION;
43  else if (target->getLevel() >= _player->getLevel())
44  error = ERR_REFER_A_FRIEND_TARGET_TOO_HIGH;
45  else if (target->getLevel() >= sWorld->getIntConfig(CONFIG_MAX_RECRUIT_A_FRIEND_BONUS_PLAYER_LEVEL))
46  error = ERR_REFER_A_FRIEND_GRANT_LEVEL_MAX_I;
47  else if (target->GetGroup() != _player->GetGroup())
48  error = ERR_REFER_A_FRIEND_NOT_IN_GROUP;
49 
50  if (error)
51  {
52  WorldPacket data(SMSG_REFER_A_FRIEND_FAILURE, 24);
53  data << uint32(error);
54  if (error == ERR_REFER_A_FRIEND_NOT_IN_GROUP)
55  data << target->GetName();
56 
57  SendPacket(&data);
58  return;
59  }
60 
61  WorldPacket data2(SMSG_PROPOSE_LEVEL_GRANT, 8);
62  data2 << _player->GetPackGUID();
63  target->GetSession()->SendPacket(&data2);
64 }
