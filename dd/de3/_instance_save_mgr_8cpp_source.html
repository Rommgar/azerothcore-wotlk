261 void InstanceSaveMgr::LoadResetTimes()
262 {
263  time_t now = time(nullptr);
264  time_t today = (now / DAY) * DAY;
265 
266  // load the global respawn times for raid/heroic instances
267  uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
268  QueryResult result = CharacterDatabase.Query("SELECT mapid, difficulty, resettime FROM instance_reset");
269  if (result)
270  {
271  do
272  {
273  Field* fields = result->Fetch();
274  uint32 mapid = fields[0].GetUInt16();
275  Difficulty difficulty = Difficulty(fields[1].GetUInt8());
276  uint64 resettime = fields[2].GetUInt32();
277 
278  MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
279  if (!mapDiff)
280  {
281  LOG_ERROR("instance.save", "InstanceSaveMgr::LoadResetTimes: invalid mapid(%u)/difficulty(%u) pair in instance_reset!", mapid, difficulty);
282  CharacterDatabase.DirectPExecute("DELETE FROM instance_reset WHERE mapid = '%u' AND difficulty = '%u'", mapid, difficulty);
283  continue;
284  }
285 
286  SetResetTimeFor(mapid, difficulty, resettime);
287  } while (result->NextRow());
288  }
289 
290  // calculate new global reset times for expired instances and those that have never been reset yet
291  // add the global reset times to the priority queue
292  for (MapDifficultyMap::const_iterator itr = sMapDifficultyMap.begin(); itr != sMapDifficultyMap.end(); ++itr)
293  {
294  uint32 map_diff_pair = itr->first;
295  uint32 mapid = PAIR32_LOPART(map_diff_pair);
296  Difficulty difficulty = Difficulty(PAIR32_HIPART(map_diff_pair));
297  MapDifficulty const* mapDiff = &itr->second;
298  if (!mapDiff->resetTime)
299  continue;
300 
301  // the reset_delay must be at least one day
302  uint32 period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
303  if (period < DAY)
304  period = DAY;
305 
306  time_t t = GetResetTimeFor(mapid, difficulty);
307  if (!t)
308  {
309  // initialize the reset time
310  t = today + period + diff;
311  SetResetTimeFor(mapid, difficulty, t);
312  CharacterDatabase.DirectPExecute("INSERT INTO instance_reset VALUES ('%u', '%u', '%u')", mapid, difficulty, (uint32)t);
313  }
314 
315  if (t < now)
316  {
317  // assume that expired instances have already been cleaned
318  // calculate the next reset time
319  t = (t * DAY) / DAY;
320  t += ((today - t) / period + 1) * period + diff;
321  CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '%u' WHERE mapid = '%u' AND difficulty = '%u'", (uint32)t, mapid, difficulty);
322  }
323 
324  SetExtendedResetTimeFor(mapid, difficulty, t);
325 
326  // schedule the global reset/warning
327  uint8 type;
328  for (type = 1; type < 5; ++type)
329  if (now + ResetTimeDelay[type - 1] < t)
330  break;
331 
332  ScheduleReset(t - ResetTimeDelay[type - 1], InstResetEvent(type, mapid, difficulty));
333  }
334 }

518 void InstanceSaveMgr::_ResetOrWarnAll(uint32 mapid, Difficulty difficulty, bool warn, time_t resetTime)
519 {
520  // global reset for all instances of the given map
521  MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
522  if (!mapEntry->Instanceable())
523  return;
524 
525  time_t now = time(nullptr);
526 
527  if (!warn)
528  {
529  MapDifficulty const* mapDiff = GetMapDifficultyData(mapid, difficulty);
530  if (!mapDiff || !mapDiff->resetTime)
531  {
532  LOG_ERROR("instance.save", "InstanceSaveMgr::ResetOrWarnAll: not valid difficulty or no reset delay for map %d", mapid);
533  return;
534  }
535 
536  // calculate the next reset time
537  uint32 diff = sWorld->getIntConfig(CONFIG_INSTANCE_RESET_TIME_HOUR) * HOUR;
538 
539  uint32 period = uint32(((mapDiff->resetTime * sWorld->getRate(RATE_INSTANCE_RESET_TIME)) / DAY) * DAY);
540  if (period < DAY)
541  period = DAY;
542 
543  uint32 next_reset = uint32(((resetTime + MINUTE) / DAY * DAY) + period + diff);
544  SetResetTimeFor(mapid, difficulty, next_reset);
545  SetExtendedResetTimeFor(mapid, difficulty, next_reset + period);
546  ScheduleReset(time_t(next_reset - 3600), InstResetEvent(1, mapid, difficulty));
547 
548  // update it in the DB
549  CharacterDatabasePreparedStatement* stmt = CharacterDatabase.GetPreparedStatement(CHAR_UPD_GLOBAL_INSTANCE_RESETTIME);
550  stmt->setUInt32(0, next_reset);
551  stmt->setUInt16(1, uint16(mapid));
552  stmt->setUInt8(2, uint8(difficulty));
553  CharacterDatabase.Execute(stmt);
554 
555  // remove all binds to instances of the given map and delete from db (delete per instance id, no mass deletion!)
556  // do this after new reset time is calculated
557  for (InstanceSaveHashMap::iterator itr = m_instanceSaveById.begin(), itr2; itr != m_instanceSaveById.end(); )
558  {
559  itr2 = itr++;
560  if (itr2->second->GetMapId() == mapid && itr2->second->GetDifficulty() == difficulty)
561  _ResetSave(itr2);
562  }
563  }
564 
565  // now loop all existing maps to warn / reset
566  Map const* map = sMapMgr->CreateBaseMap(mapid);
567  MapInstanced::InstancedMaps& instMaps = ((MapInstanced*)map)->GetInstancedMaps();
568  MapInstanced::InstancedMaps::iterator mitr;
569  uint32 timeLeft;
570 
571  for (mitr = instMaps.begin(); mitr != instMaps.end(); ++mitr)
572  {
573  Map* map2 = mitr->second;
574  if (!map2->IsDungeon() || map2->GetDifficulty() != difficulty)
575  continue;
576 
577  if (warn)
578  {
579  if (now >= resetTime)
580  timeLeft = 0;
581  else
582  timeLeft = uint32(resetTime - now);
583 
584  map2->ToInstanceMap()->SendResetWarnings(timeLeft);
585  }
586  else
587  {
588  InstanceSave* save = GetInstanceSave(map2->GetInstanceId());
589  map2->ToInstanceMap()->Reset(INSTANCE_RESET_GLOBAL, (save ? & (save->m_playerList) : nullptr));
590  }
591  }
592 }
