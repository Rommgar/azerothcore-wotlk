141  void UpdateAI(uint32 diff) override
142  {
143  events2.Update(diff);
144  switch (events2.ExecuteEvent())
145  {
146  case INTRO_1:
147  me->GetMotionMaster()->MovePath(KIRTONOS_PATH, false);
148  break;
149  case INTRO_2:
150  me->GetMotionMaster()->MovePoint(0, PosMove[0]);
151  break;
152  case INTRO_3:
153  if (GameObject* gate = me->GetMap()->GetGameObject(instance->GetGuidData(GO_GATE_KIRTONOS)))
154  {
155  gate->SetGoState(GO_STATE_READY);
156  }
157  me->SetFacingTo(0.01745329f);
158  break;
159  case INTRO_4:
160  me->SetWalk(true);
161  me->SetCanFly(false);
162  me->SetDisableGravity(false);
163  me->CastSpell(me, SPELL_KIRTONOS_TRANSFORM, true);
164  break;
165  case INTRO_5:
166  me->HandleEmoteCommand(EMOTE_ONESHOT_ROAR);
167  me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
168  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);
169  me->SetReactState(REACT_AGGRESSIVE);
170  break;
171  case INTRO_6:
172  if (!me->IsInCombat())
173  {
174  me->GetMotionMaster()->MovePoint(0, PosMove[1]);
175  }
176  break;
177  }
178 
179  if (!UpdateVictim())
180  {
181  return;
182  }
183 
184  events.Update(diff);
185  if (me->HasUnitState(UNIT_STATE_CASTING))
186  {
187  return;
188  }
189 
190  switch (events.ExecuteEvent())
191  {
192  case EVENT_SWOOP:
193  me->CastSpell(me->GetVictim(), SPELL_SWOOP, false);
194  events.ScheduleEvent(EVENT_SWOOP, 15000);
195  break;
196  case EVENT_WING_FLAP:
197  me->CastSpell(me, SPELL_WING_FLAP, false);
198  events.ScheduleEvent(EVENT_WING_FLAP, 13000);
199  break;
200  case EVENT_PIERCE_ARMOR:
201  me->CastSpell(me->GetVictim(), SPELL_PIERCE_ARMOR, false);
202  events.ScheduleEvent(EVENT_PIERCE_ARMOR, 12000);
203  break;
204  case EVENT_DISARM:
205  me->CastSpell(me->GetVictim(), SPELL_DISARM, false);
206  events.ScheduleEvent(EVENT_DISARM, 11000);
207  break;
208  case EVENT_SHADOW_BOLT_VOLLEY:
209  me->CastSpell(me, SPELL_SHADOW_BOLT_VOLLEY, false);
210  events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 10000);
211  break;
212  case EVENT_CURSE_OF_TONGUES:
213  me->CastSpell(me, SPELL_CURSE_OF_TONGUES, false);
214  events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 20000);
215  break;
216  case EVENT_DOMINATE_MIND:
217  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 20.0f, true))
218  {
219  me->CastSpell(target, SPELL_DOMINATE_MIND, false);
220  }
221  break;
222  case EVENT_KIRTONOS_TRANSFORM:
223  TransformsCount++;
224 
225  if (me->HasAura(SPELL_KIRTONOS_TRANSFORM))
226  {
227  events.Reset();
228  events.ScheduleEvent(EVENT_SWOOP, 4000);
229  events.ScheduleEvent(EVENT_WING_FLAP, 7000);
230  events.ScheduleEvent(EVENT_PIERCE_ARMOR, 11000);
231  events.ScheduleEvent(EVENT_DISARM, 15000);
232  // show shape-shift animation before aura removal
233  me->CastSpell(me, SPELL_TRANSFORM_VISUAL, true);
234  me->RemoveAura(SPELL_KIRTONOS_TRANSFORM);
235  me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(0));
236  }
237  else
238  {
239  events.Reset();
240  events.ScheduleEvent(EVENT_SHADOW_BOLT_VOLLEY, 2000);
241  events.ScheduleEvent(EVENT_CURSE_OF_TONGUES, 6000);
242  events.ScheduleEvent(EVENT_WING_FLAP, 13000);
243  me->CastSpell(me, SPELL_KIRTONOS_TRANSFORM, true);
244  me->SetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID + 0, uint32(WEAPON_KIRTONOS_STAFF));
245 
246  // Schedule Dominate Mind on every 2nd caster transform
247  if ((TransformsCount - 2) % 4 == 0)
248  {
249  events.ScheduleEvent(EVENT_DOMINATE_MIND, urand(4000, 8000));
250  }
251  }
252 
253  events.ScheduleEvent(EVENT_KIRTONOS_TRANSFORM, 20000);
254  break;
255  }
256 
257  DoZoneInCombat();
258  DoMeleeAttackIfReady();
259  }
