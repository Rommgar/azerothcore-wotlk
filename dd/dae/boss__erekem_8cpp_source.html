136  me->CastSpell(target, SPELL_CHAIN_HEAL, false);

139  Creature* pGuard2 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID));
140  if ((pGuard1 && !pGuard1->IsAlive()) || (pGuard2 && !pGuard2->IsAlive()))

142  events.RepeatEvent(urand(3000, 6000));
143  break;
144  }
145  }
146  events.RepeatEvent(urand(8000, 11000));
147  break;

150  events.RepeatEvent(20000);
151  break;
152  case EVENT_SPELL_EARTH_SHOCK:
153  me->CastSpell(me->GetVictim(), SPELL_EARTH_SHOCK, false);
154  events.RepeatEvent(urand(8000, 13000));
155  break;
156  case EVENT_SPELL_LIGHTNING_BOLT:
157  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 35.0f, true))
158  me->CastSpell(target, SPELL_LIGHTNING_BOLT, false);
159  events.RepeatEvent(urand(15000, 25000));
160  break;
161  case EVENT_SPELL_STORMSTRIKE:
162  {
163  Creature* pGuard1 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_1_GUID));
164  Creature* pGuard2 = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_EREKEM_GUARD_2_GUID));
165  if (pGuard1 && !pGuard1->IsAlive() && pGuard2 && !pGuard2->IsAlive()) // both dead
