117 void CasterAI::EnterCombat(Unit* who)
118 {
119  if (spells.empty())
120  return;
121 
122  uint32 spell = rand() % spells.size();
123  uint32 count = 0;
124  for (SpellVct::iterator itr = spells.begin(); itr != spells.end(); ++itr, ++count)
125  {
126  if (AISpellInfo[*itr].condition == AICOND_AGGRO)
127  me->CastSpell(who, *itr, false);
128  else if (AISpellInfo[*itr].condition == AICOND_COMBAT)
129  {
130  uint32 cooldown = GetAISpellInfo(*itr)->realCooldown;
131  if (count == spell)
132  {
133  DoCast(spells[spell]);
134  cooldown += me->GetCurrentSpellCastTime(*itr);
135  }
136  events.ScheduleEvent(*itr, cooldown);
137  }
138  }
139 }
