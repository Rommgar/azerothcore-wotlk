116  // modulos a radian orientation to the range of 0..2PI
117  static float NormalizeOrientation(float o)
118  {
119  // fmod only supports positive numbers. Thus we have
120  // to emulate negative numbers
121  if (o < 0)
122  {
123  float mod = o * -1;
124  mod = fmod(mod, 2.0f * static_cast<float>(M_PI));
125  mod = -mod + 2.0f * static_cast<float>(M_PI);
126  return mod;
127  }
128  return fmod(o, 2.0f * static_cast<float>(M_PI));
129  }

173 template<typename Worker>
174 void MapMgr::DoForAllMaps(Worker&& worker)
175 {
176  std::lock_guard<std::mutex> guard(Lock);
177 
178  for (auto& mapPair : i_maps)
179  {
180  Map* map = mapPair.second;
181  if (MapInstanced* mapInstanced = map->ToMapInstanced())
182  {
183  MapInstanced::InstancedMaps& instances = mapInstanced->GetInstancedMaps();
184  for (auto& instancePair : instances)
185  worker(instancePair.second);
186  }
187  else
188  worker(map);
189  }
190 }

204  for (auto& p : instances)
205  worker(p.second);
206  }
207  else
