56 static my_bool init_state_maps(CHARSET_INFO *cs)
57 {
58  uint i;
59  uchar *state_map;
60  uchar *ident_map;
61 
62  if (!(cs->state_map= (uchar*) my_once_alloc(256, MYF(MY_WME))))
63  return 1;
64 
65  if (!(cs->ident_map= (uchar*) my_once_alloc(256, MYF(MY_WME))))
66  return 1;
67 
68  state_map= cs->state_map;
69  ident_map= cs->ident_map;
70 
71  /* Fill state_map with states to get a faster parser */
72  for (i=0; i < 256 ; i++)
73  {
74  if (my_isalpha(cs,i))
75  state_map[i]=(uchar) MY_LEX_IDENT;
76  else if (my_isdigit(cs,i))
77  state_map[i]=(uchar) MY_LEX_NUMBER_IDENT;
78 #if defined(USE_MB) && defined(USE_MB_IDENT)
79  else if (my_mbcharlen(cs, i)>1)
80  state_map[i]=(uchar) MY_LEX_IDENT;
81 #endif
82  else if (my_isspace(cs,i))
83  state_map[i]=(uchar) MY_LEX_SKIP;
84  else
85  state_map[i]=(uchar) MY_LEX_CHAR;
86  }
87  state_map[(uchar)'_']=state_map[(uchar)'$']=(uchar) MY_LEX_IDENT;
88  state_map[(uchar)'\'']=(uchar) MY_LEX_STRING;
89  state_map[(uchar)'.']=(uchar) MY_LEX_REAL_OR_POINT;
90  state_map[(uchar)'>']=state_map[(uchar)'=']=state_map[(uchar)'!']= (uchar) MY_LEX_CMP_OP;
91  state_map[(uchar)'<']= (uchar) MY_LEX_LONG_CMP_OP;
92  state_map[(uchar)'&']=state_map[(uchar)'|']=(uchar) MY_LEX_BOOL;
93  state_map[(uchar)'#']=(uchar) MY_LEX_COMMENT;
94  state_map[(uchar)';']=(uchar) MY_LEX_SEMICOLON;
95  state_map[(uchar)':']=(uchar) MY_LEX_SET_VAR;
96  state_map[0]=(uchar) MY_LEX_EOL;
97  state_map[(uchar)'\\']= (uchar) MY_LEX_ESCAPE;
98  state_map[(uchar)'/']= (uchar) MY_LEX_LONG_COMMENT;
99  state_map[(uchar)'*']= (uchar) MY_LEX_END_LONG_COMMENT;
100  state_map[(uchar)'@']= (uchar) MY_LEX_USER_END;
101  state_map[(uchar) '`']= (uchar) MY_LEX_USER_VARIABLE_DELIMITER;
102  state_map[(uchar)'"']= (uchar) MY_LEX_STRING_OR_DELIMITER;
103 
104  /*
105  Create a second map to make it faster to find identifiers
106  */
107  for (i=0; i < 256 ; i++)
108  {
109  ident_map[i]= (uchar) (state_map[i] == MY_LEX_IDENT ||
110  state_map[i] == MY_LEX_NUMBER_IDENT);
111  }
112 
113  /* Special handling of hex and binary strings */
114  state_map[(uchar)'x']= state_map[(uchar)'X']= (uchar) MY_LEX_IDENT_OR_HEX;
115  state_map[(uchar)'b']= state_map[(uchar)'B']= (uchar) MY_LEX_IDENT_OR_BIN;
116  state_map[(uchar)'n']= state_map[(uchar)'N']= (uchar) MY_LEX_IDENT_OR_NCHAR;
117  return 0;
118 }

134 {
135  to->number= from->number ? from->number : to->number;
136 
137  if (from->csname)

139  goto err;
140 
141  if (from->name)

143  goto err;
144 
145  if (from->comment)
146  if (!(to->comment= my_once_strdup(from->comment,MYF(MY_WME))))
147  goto err;
148 
149  if (from->ctype)

161  MYF(MY_WME))))
162  goto err;
163 
164  if (from->to_upper)
165  if (!(to->to_upper= (uchar*) my_once_memdup((char*) from->to_upper,
166  MY_CS_TO_UPPER_TABLE_SIZE,
167  MYF(MY_WME))))
168  goto err;
169  if (from->sort_order)
170  {
171  if (!(to->sort_order= (uchar*) my_once_memdup((char*) from->sort_order,
172  MY_CS_SORT_ORDER_TABLE_SIZE,

174  goto err;
175 
176  }

183  }
184  if (from->tailoring)
185  if (!(to->tailoring= my_once_strdup(from->tailoring,MYF(MY_WME))))
186  goto err;

188  return 0;

220 static int add_collation(CHARSET_INFO *cs)
221 {
222  if (cs->name && (cs->number ||
223  (cs->number=get_collation_number_internal(cs->name))) &&
224  cs->number < array_elements(all_charsets))
225  {
226  if (!all_charsets[cs->number])
227  {
228  if (!(all_charsets[cs->number]=
229  (CHARSET_INFO*) my_once_alloc(sizeof(CHARSET_INFO),MYF(0))))
230  return MY_XML_ERROR;
231  bzero((void*)all_charsets[cs->number],sizeof(CHARSET_INFO));
232  }
233 
234  if (cs->primary_number == cs->number)
235  cs->state |= MY_CS_PRIMARY;
236 
237  if (cs->binary_number == cs->number)
238  cs->state |= MY_CS_BINSORT;
239 
240  all_charsets[cs->number]->state|= cs->state;
241 
242  if (!(all_charsets[cs->number]->state & MY_CS_COMPILED))
243  {
244  CHARSET_INFO *newcs= all_charsets[cs->number];
245  if (cs_copy_data(all_charsets[cs->number],cs))
246  return MY_XML_ERROR;
247 
248  newcs->caseup_multiply= newcs->casedn_multiply= 1;
249 
250  if (!strcmp(cs->csname,"ucs2") )
251  {
252 #if defined(HAVE_CHARSET_ucs2) && defined(HAVE_UCA_COLLATIONS)
253  copy_uca_collation(newcs, &my_charset_ucs2_unicode_ci);
254  newcs->state|= MY_CS_AVAILABLE | MY_CS_LOADED | MY_CS_NONASCII;
255 #endif
256  }
257  else if (!strcmp(cs->csname, "utf8") || !strcmp(cs->csname, "utf8mb3"))
258  {
259 #if defined (HAVE_CHARSET_utf8) && defined(HAVE_UCA_COLLATIONS)
260  copy_uca_collation(newcs, &my_charset_utf8_unicode_ci);
261  newcs->ctype= my_charset_utf8_unicode_ci.ctype;
262  if (init_state_maps(newcs))
263  return MY_XML_ERROR;
264 #endif
265  }
266  else if (!strcmp(cs->csname, "utf8mb4"))
267  {
268 #if defined (HAVE_CHARSET_utf8mb4) && defined(HAVE_UCA_COLLATIONS)
269  copy_uca_collation(newcs, &my_charset_utf8mb4_unicode_ci);
270  newcs->ctype= my_charset_utf8mb4_unicode_ci.ctype;
271  newcs->state|= MY_CS_AVAILABLE | MY_CS_LOADED;
272 #endif
273  }
274  else if (!strcmp(cs->csname, "utf16"))
275  {
276 #if defined (HAVE_CHARSET_utf16) && defined(HAVE_UCA_COLLATIONS)
277  copy_uca_collation(newcs, &my_charset_utf16_unicode_ci);
278  newcs->state|= MY_CS_AVAILABLE | MY_CS_LOADED | MY_CS_NONASCII;
279 #endif
280  }
281  else if (!strcmp(cs->csname, "utf32"))
282  {
283 #if defined (HAVE_CHARSET_utf32) && defined(HAVE_UCA_COLLATIONS)
284  copy_uca_collation(newcs, &my_charset_utf32_unicode_ci);
285  newcs->state|= MY_CS_AVAILABLE | MY_CS_LOADED | MY_CS_NONASCII;
286 #endif
287  }
288  else
289  {
290  uchar *sort_order= all_charsets[cs->number]->sort_order;
291  simple_cs_init_functions(all_charsets[cs->number]);
292  newcs->mbminlen= 1;
293  newcs->mbmaxlen= 1;
294  if (simple_cs_is_full(all_charsets[cs->number]))
295  {
296  all_charsets[cs->number]->state |= MY_CS_LOADED;
297  }
298  all_charsets[cs->number]->state|= MY_CS_AVAILABLE;
299 
300  /*
301  Check if case sensitive sort order: A < a < B.
302  We need MY_CS_FLAG for regex library, and for
303  case sensitivity flag for 5.0 client protocol,
304  to support isCaseSensitive() method in JDBC driver
305  */
306  if (sort_order && sort_order['A'] < sort_order['a'] &&
307  sort_order['a'] < sort_order['B'])
308  all_charsets[cs->number]->state|= MY_CS_CSSORT;
309 
310  if (my_charset_is_8bit_pure_ascii(all_charsets[cs->number]))
311  all_charsets[cs->number]->state|= MY_CS_PUREASCII;
312  if (!my_charset_is_ascii_compatible(cs))
313  all_charsets[cs->number]->state|= MY_CS_NONASCII;
314  }
315  }
316  else
317  {
318  /*
319  We need the below to make get_charset_name()
320  and get_charset_number() working even if a
321  character set has not been really incompiled.
322  The above functions are used for example
323  in error message compiler extra/comp_err.c.
324  If a character set was compiled, this information
325  will get lost and overwritten in add_compiled_collation().
326  */
327  CHARSET_INFO *dst= all_charsets[cs->number];
328  dst->number= cs->number;
329  if (cs->comment)
330  if (!(dst->comment= my_once_strdup(cs->comment,MYF(MY_WME))))
331  return MY_XML_ERROR;
332  if (cs->csname)
333  if (!(dst->csname= my_once_strdup(cs->csname,MYF(MY_WME))))
334  return MY_XML_ERROR;
335  if (cs->name)
336  if (!(dst->name= my_once_strdup(cs->name,MYF(MY_WME))))
337  return MY_XML_ERROR;
338  }
339  cs->number= 0;
340  cs->primary_number= 0;
341  cs->binary_number= 0;
342  cs->name= NULL;
343  cs->state= 0;
344  cs->sort_order= NULL;
345  cs->state= 0;
346  }
347  return MY_XML_OK;
348 }

543 static CHARSET_INFO *get_internal_charset(uint cs_number, myf flags)
544 {
545  char buf[FN_REFLEN];
546  CHARSET_INFO *cs;
547 
548  if ((cs= all_charsets[cs_number]))
549  {
550  if (cs->state & MY_CS_READY) /* if CS is already initialized */
551  return cs;
552 
553  /*
554  To make things thread safe we are not allowing other threads to interfere
555  while we may changing the cs_info_table
556  */
557  mysql_mutex_lock(&THR_LOCK_charset);
558 
559  if (!(cs->state & (MY_CS_COMPILED|MY_CS_LOADED))) /* if CS is not in memory */
560  {
561  strxmov(get_charsets_dir(buf), cs->csname, ".xml", NullS);
562  my_read_charset_file(buf,flags);
563  }
564 
565  if (cs->state & MY_CS_AVAILABLE)
566  {
567  if (!(cs->state & MY_CS_READY))
568  {
569  if ((cs->cset->init && cs->cset->init(cs, cs_alloc)) ||
570  (cs->coll->init && cs->coll->init(cs, cs_alloc)))
571  cs= NULL;
572  else
573  cs->state|= MY_CS_READY;
574  }
575  }
576  else
577  cs= NULL;
578 
579  mysql_mutex_unlock(&THR_LOCK_charset);
580  }
581  return cs;
582 }

718 /*
719  Escape string with backslashes (\‍)
720 
721  SYNOPSIS
722  escape_string_for_mysql()
723  charset_info Charset of the strings
724  to Buffer for escaped string
725  to_length Length of destination buffer, or 0
726  from The string to escape
727  length The length of the string to escape
728 
729  DESCRIPTION
730  This escapes the contents of a string by adding backslashes before special
731  characters, and turning others into specific escape sequences, such as
732  turning newlines into \n and null bytes into \0.
733 
734  NOTE
735  To maintain compatibility with the old C API, to_length may be 0 to mean
736  "big enough"
737 
738  RETURN VALUES
739  (size_t) -1 The escaped string did not fit in the to buffer
740  # The length of the escaped string
741 */
742 
743 size_t escape_string_for_mysql(CHARSET_INFO *charset_info,
744  char *to, size_t to_length,
745  const char *from, size_t length)
746 {
747  const char *to_start= to;
748  const char *end, *to_end=to_start + (to_length ? to_length-1 : 2*length);
749  my_bool overflow= FALSE;
750 #ifdef USE_MB
751  my_bool use_mb_flag= use_mb(charset_info);
752 #endif
753  for (end= from + length; from < end; from++)
754  {
755  char escape= 0;
756 #ifdef USE_MB
757  int tmp_length;
758  if (use_mb_flag && (tmp_length= my_ismbchar(charset_info, from, end)))
759  {
760  if (to + tmp_length > to_end)
761  {
762  overflow= TRUE;
763  break;
764  }
765  while (tmp_length--)
766  *to++= *from++;
767  from--;
768  continue;
769  }
770  /*
771  If the next character appears to begin a multi-byte character, we
772  escape that first byte of that apparent multi-byte character. (The
773  character just looks like a multi-byte character -- if it were actually
774  a multi-byte character, it would have been passed through in the test
775  above.)
776 
777  Without this check, we can create a problem by converting an invalid
778  multi-byte character into a valid one. For example, 0xbf27 is not
779  a valid GBK character, but 0xbf5c is. (0x27 = ', 0x5c = \‍)
780  */
781  if (use_mb_flag && (tmp_length= my_mbcharlen(charset_info, *from)) > 1)
782  escape= *from;
783  else
784 #endif
785  switch (*from) {
786  case 0: /* Must be escaped for 'mysql' */
787  escape= '0';
788  break;
789  case '\n': /* Must be escaped for logs */
790  escape= 'n';
791  break;
792  case '\r':
793  escape= 'r';
794  break;
795  case '\\':
796  escape= '\\';
797  break;
798  case '\'':
799  escape= '\'';
800  break;
801  case '"': /* Better safe than sorry */
802  escape= '"';
803  break;
804  case '\032': /* This gives problems on Win32 */
805  escape= 'Z';
806  break;
807  }
808  if (escape)
809  {
810  if (to + 2 > to_end)
811  {
812  overflow= TRUE;
813  break;
814  }
815  *to++= '\\';
816  *to++= escape;
817  }
818  else
819  {
820  if (to + 1 > to_end)
821  {
822  overflow= TRUE;
823  break;
824  }
825  *to++= *from;
826  }
827  }
828  *to= 0;
829  return overflow ? (size_t) -1 : (size_t) (to - to_start);
830 }
