207 template<class T>
208 inline void LoadDBC(uint32& availableDbcLocales, StoreProblemList& errors, DBCStorage<T>& storage, std::string const& dbcPath, std::string const& filename, char const* dbTable = nullptr)
209 {
210  // compatibility format and C++ structure sizes
211  ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T) || LoadDBC_assert_print(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), sizeof(T), filename));
212 
213  ++DBCFileCount;
214  std::string dbcFilename = dbcPath + filename;
215  bool existDBData = false;
216 
217  if (storage.Load(dbcFilename.c_str()))
218  {
219  for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
220  {
221  if (!(availableDbcLocales & (1 << i)))
222  continue;
223 
224  std::string localizedName(dbcPath);
225  localizedName.append(localeNames[i]);
226  localizedName.push_back('/');
227  localizedName.append(filename);
228 
229  if (!storage.LoadStringsFrom(localizedName.c_str()))
230  availableDbcLocales &= ~(1 << i); // mark as not available for speedup next checks
231  }
232  }
233 
234  if (dbTable)
235  storage.LoadFromDB(dbTable, storage.GetFormat());
236 
237  if (storage.GetNumRows())
238  existDBData = true;
239 
240  if (!existDBData)
241  {
242  // sort problematic dbc to (1) non compatible and (2) non-existed
243  if (FILE* f = fopen(dbcFilename.c_str(), "rb"))
244  {
245  std::ostringstream stream;
246  stream << dbcFilename << " exists, and has " << storage.GetFieldCount() << " field(s) (expected " << strlen(storage.GetFormat()) << "). Extracted file might be from wrong client version or a database-update has been forgotten.";
247  std::string buf = stream.str();
248  errors.push_back(buf);
249  fclose(f);
250  }
251  else
252  errors.push_back(dbcFilename);
253  }
254 }

631 // Convert the geomoetry from a spline value, to an actual WoW XYZ
632 G3D::Vector3 TranslateLocation(G3D::Vector4 const* DBCPosition, G3D::Vector3 const* basePosition, G3D::Vector3 const* splineVector)
633 {
634  G3D::Vector3 work;
635  float x = basePosition->x + splineVector->x;
636  float y = basePosition->y + splineVector->y;
637  float z = basePosition->z + splineVector->z;
638  float const distance = sqrt((x * x) + (y * y));
639  float angle = std::atan2(x, y) - DBCPosition->w;
640 
641  if (angle < 0)
642  {
643  angle += 2 * float(M_PI);
644  }
645 
646  work.x = DBCPosition->x + (distance * sin(angle));
647  work.y = DBCPosition->y + (distance * cos(angle));
648  work.z = DBCPosition->z + z;
649  return work;
650 }

702  thisCam.locations.x = newPos.x;
703  thisCam.locations.y = newPos.y;

708  currPos += sizeof(M2SplineKey<G3D::Vector3>);
709  }
710  }
711 
712  // Read camera positions and timestamps (translating first position of 3 only, we don't need to translate the whole spline)
713  for (uint32 k = 0; k < cam->positions.timestamps.number; ++k)
714  {
715  // Extract Camera positions for this set

722  {
723  return false;
724  }
725  uint32 const* posTimestamps = reinterpret_cast<uint32 const*>(buffer + posTsArray->offset_elements);
726  M2Array const* posArray = reinterpret_cast<M2Array const*>(buffer + cam->positions.values.offset_elements);

731  M2SplineKey<G3D::Vector3> const* positions = reinterpret_cast<M2SplineKey<G3D::Vector3> const*>(buffer + posArray->offset_elements);
732 
733  // Read the data for this set
734  uint32 currPos = posArray->offset_elements;
735  for (uint32 i = 0; i < posTsArray->number; ++i)
736  {
737  if (currPos + sizeof(M2SplineKey<G3D::Vector3>) > buffSize)
738  {
739  return false;

742  G3D::Vector3 newPos = TranslateLocation(&DBCData, &cam->position_base, &positions->p0);

745  FlyByCamera thisCam;
746  thisCam.timeStamp = posTimestamps[i];
747  thisCam.locations.x = newPos.x;
748  thisCam.locations.y = newPos.y;
749  thisCam.locations.z = newPos.z;

754  FlyByCamera lastTarget;

795  }

819 
820  // Replace slashes
821  size_t loc = filename.find("\\");
822  while (loc != std::string::npos)
823  {
824  filename.replace(loc, 1, "/");
825  loc = filename.find("\\");
826  }
827 
828  // Replace mdx to .m2
829  loc = filename.find(".mdx");
830  if (loc != std::string::npos)
831  {
832  filename.replace(loc, 4, ".m2");
833  }
834 
835  std::ifstream m2file(filename.c_str(), std::ios::in | std::ios::binary);
836  if (!m2file.is_open())
837  {
838  continue;
839  }
840 
841  // Get file size
842  m2file.seekg(0, std::ios::end);
843  std::streamoff const fileSize = m2file.tellg();
844 
845  // Reject if not at least the size of the header
846  if (static_cast<uint32>(fileSize) < sizeof(M2Header))
847  {
848  LOG_ERROR("dbc", "Camera file %s is damaged. File is smaller than header size", filename.c_str());

850  continue;
851  }
852

854  m2file.seekg(0, std::ios::beg);
855  char fileCheck[5];
856  m2file.read(fileCheck, 4);
857  fileCheck[4] = 0;
858

860  if (strcmp(fileCheck, "MD20"))
861  {
862  LOG_ERROR("dbc", "Camera file %s is damaged. File identifier not found", filename.c_str());
863  m2file.close();
864  continue;
865  }
866 
867  // Now we have a good file, read it all into a vector of char's, then close the file.
868  std::vector<char> buffer(fileSize);
869  m2file.seekg(0, std::ios::beg);
870  if (!m2file.read(buffer.data(), fileSize))
871  {
872  m2file.close();
873  continue;
874  }
875  m2file.close();
876 
877  // Read header
878  M2Header const* header = reinterpret_cast<M2Header const*>(buffer.data());
879 
880  if (header->ofsCameras + sizeof(M2Camera) > static_cast<uint32>(fileSize))
881  {
882  LOG_ERROR("dbc", "Camera file %s is damaged. Camera references position beyond file end", filename.c_str());
883  continue;
884  }
885 
886  // Get camera(s) - Main header, then dump them.

890  LOG_ERROR("dbc", "Camera file %s is damaged. Camera references position beyond file end", filename.c_str());
891  }
892  }
893  }
