84  void UpdateAI(uint32 diff) override
85  {

87  return;
88

99  CreatureTemplate const* Info = me->GetCreatureTemplate();
100  CreatureBaseStats const* stats = sObjectMgr->GetCreatureBaseStats(me->getLevel(), Info->unit_class);
101  uint32 health = uint32(107 * (me->getLevel() - 40) * 0.025f);
102  me->SetCreateHealth(health);
103 
104  for (uint8 stat = 0; stat < MAX_STATS; ++stat)
105  {
106  me->SetStat(Stats(stat), 0);
107  me->SetCreateStat(Stats(stat), 0);
108  }
109 
110  me->SetModifierValue(UNIT_MOD_HEALTH, BASE_VALUE, (float)health);
111  me->SetMaxHealth(health);
112  //Add delta to make them not all hit the same time
113  uint32 delta = urand(0, 700);
114  me->SetAttackTime(BASE_ATTACK, Info->BaseAttackTime + delta);
115  me->SetStatFloatValue(UNIT_FIELD_RANGED_ATTACK_POWER, float(stats->AttackPower));
116  me->CastSpell(me, SPELL_HUNTER_DEADLY_POISON_PASSIVE, true);
117 
118  // Glyph of Snake Trap
119  if (Unit* owner = me->GetOwner())
120  if (owner->GetAuraEffectDummy(SPELL_HUNTER_GLYPH_OF_SNAKE_TRAP))
121  me->CastSpell(me, SPELL_HUNTER_PET_SCALING, true);
122  }
123 
124  _spellTimer += diff;
125  if (_spellTimer >= 3000)
126  {
127  if (urand(0, 2) == 0) // 33% chance to cast
128  DoCastVictim(RAND(SPELL_HUNTER_MIND_NUMBING_POISON, SPELL_HUNTER_CRIPPLING_POISON));
129 
130  _spellTimer = 0;
131  }
132 
133  DoMeleeAttackIfReady();
134  }
