151  struct npc_winterfin_playmateAI : public ScriptedAI
152  {
153  npc_winterfin_playmateAI(Creature* creature) : ScriptedAI(creature) {}
154 
155  void Reset() override
156  {
157  timer = 0;
158  phase = 0;
159  playerGUID.Clear();
160  orphanGUID.Clear();
161  }
162 
163  void MoveInLineOfSight(Unit* who) override
164  {
165  if (!phase && who && who->GetDistance2d(me) < 10.0f)
166  if (Player* player = who->ToPlayer())
167  if (player->GetQuestStatus(QUEST_PLAYMATE_ORACLE) == QUEST_STATUS_INCOMPLETE)
168  {
169  playerGUID = player->GetGUID();
170  orphanGUID = getOrphanGUID(player, ORPHAN_ORACLE);
171  if (orphanGUID)
172  phase = 1;
173  }
174  }
175 
176  void UpdateAI(uint32 diff) override
177  {
178  if (!phase)
179  return;
180 
181  if (timer <= diff)
182  {
183  Player* player = ObjectAccessor::GetPlayer(*me, playerGUID);
184  Creature* orphan = ObjectAccessor::GetCreature(*me, orphanGUID);
185 
186  if (!orphan || !player)
187  {
188  Reset();
189  return;
190  }
191 
192  switch (phase)
193  {
194  case 1:
195  orphan->GetMotionMaster()->MovePoint(0, me->GetPositionX() + cos(me->GetOrientation()) * 5, me->GetPositionY() + sin(me->GetOrientation()) * 5, me->GetPositionZ());
196  orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_1);
197  timer = 3000;
198  break;
199  case 2:
200  orphan->SetFacingToObject(me);
201  Talk(TEXT_WINTERFIN_PLAYMATE_1);
202  me->HandleEmoteCommand(EMOTE_STATE_DANCE);
203  timer = 3000;
204  break;
205  case 3:
206  orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_2);
207  timer = 3000;
208  break;
209  case 4:
210  Talk(TEXT_WINTERFIN_PLAYMATE_2);
211  timer = 5000;
212  break;
213  case 5:
214  orphan->AI()->Talk(TEXT_ORACLE_ORPHAN_3);
215  me->HandleEmoteCommand(EMOTE_STATE_NONE);
216  player->GroupEventHappens(QUEST_PLAYMATE_ORACLE, me);
217  orphan->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
218  Reset();
219  return;
220  }
221  ++phase;
222  }
223  else
224  timer -= diff;
225  }
226 
227  private:
228  uint32 timer;
229  int8 phase;
230  ObjectGuid playerGUID;
231  ObjectGuid orphanGUID;
232  };
233 
234  CreatureAI* GetAI(Creature* creature) const override
235  {
