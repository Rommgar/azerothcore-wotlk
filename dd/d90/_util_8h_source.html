310 inline wchar_t wcharToUpper(wchar_t wchar)
311 {
312  if (wchar >= L'a' && wchar <= L'z') // LATIN SMALL LETTER A - LATIN SMALL LETTER Z
313  {
314  return wchar_t(uint16(wchar) - 0x0020);
315  }
316  if (wchar == 0x00DF) // LATIN SMALL LETTER SHARP S
317  {
318  return wchar_t(0x1E9E);
319  }
320  if (wchar >= 0x00E0 && wchar <= 0x00F6) // LATIN SMALL LETTER A WITH GRAVE - LATIN SMALL LETTER O WITH DIAERESIS
321  {
322  return wchar_t(uint16(wchar) - 0x0020);
323  }
324  if (wchar >= 0x00F8 && wchar <= 0x00FE) // LATIN SMALL LETTER O WITH STROKE - LATIN SMALL LETTER THORN
325  {
326  return wchar_t(uint16(wchar) - 0x0020);
327  }
328  if (wchar >= 0x0101 && wchar <= 0x012F) // LATIN SMALL LETTER A WITH MACRON - LATIN SMALL LETTER I WITH OGONEK (only %2=1)
329  {
330  if (wchar % 2 == 1)
331  {
332  return wchar_t(uint16(wchar) - 0x0001);
333  }
334  }
335  if (wchar >= 0x0430 && wchar <= 0x044F) // CYRILLIC SMALL LETTER A - CYRILLIC SMALL LETTER YA
336  {
337  return wchar_t(uint16(wchar) - 0x0020);
338  }
339  if (wchar == 0x0451) // CYRILLIC SMALL LETTER IO
340  {
341  return wchar_t(0x0401);
342  }
343 
344  return wchar;
345 }
