109  if (_pylonsState & data)
110  return;
111  _pylonsState |= data;
112  if (_pylonsState == ALL_PYLONS_OFF) // all five active, 31
113  {
114  instance->LoadGrid(-38.08f, 812.44f);
115  if (Creature* immol = instance->GetCreature(_immoltharGUID))
116  {
117  immol->setActive(true);
118  immol->GetAI()->SetData(1, 1);
119  immol->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);

122  {

124  {
125  summoner->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NON_ATTACKABLE);

128  }
