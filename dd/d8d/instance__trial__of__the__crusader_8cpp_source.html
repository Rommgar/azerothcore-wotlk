109  bool IsValidDedicatedInsanityItem(const ItemTemplate* item)
110  {
111  if (!item) // should not happen, but checked in GetAverageItemLevel()
112  return true;
113  if (item->ItemLevel < 245)
114  return true;
115  if (item->ItemId == 46017) // Val'anyr, Hammer of Ancient Kings - exception, too powerful
116  return false;
117  if (item->ItemLevel == 245 && item->Bonding == BIND_WHEN_EQUIPED) // this also includes items crafted from patterns obtained in ToC 10 norm/hc
118  return true;
119  if (validDedicatedInsanityItems.find(item->ItemId) != validDedicatedInsanityItems.end()) // list of items dropping from ToC 10 norm/hc and also items ilevel 245 buyable for emblems of triumph
120  return true;
121 
122  return false;

138  if (Player* plr = itr->GetSource())
