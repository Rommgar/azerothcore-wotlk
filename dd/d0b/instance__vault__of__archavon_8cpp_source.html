68  void Update(uint32 diff) override
69  {
70  checkTimer += diff;
71  if (checkTimer >= 60000)
72  {
73  checkTimer -= 60000; // one minute
74  if (Battlefield* bf = sBattlefieldMgr->GetBattlefieldByBattleId(BATTLEFIELD_BATTLEID_WG))
75  {
76  if (!bf->IsWarTime())
77  {
78  if (bf->GetTimer() <= (16 * MINUTE * IN_MILLISECONDS) && bf->GetTimer() >= (15 * MINUTE * IN_MILLISECONDS))
79  {
80  Map::PlayerList const& PlayerList = instance->GetPlayers();
81  if (!PlayerList.isEmpty())
82  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
83  if (Player* player = i->GetSource())
84  player->TextEmote("This instance will reset in 15 minutes.", nullptr, true);
85  }
86  else if (bf->GetTimer() <= (10 * MINUTE * IN_MILLISECONDS) && bf->GetTimer() >= (9 * MINUTE * IN_MILLISECONDS))
87  {
88  for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
89  if (Creature* cr = instance->GetCreature(bossGUIDs[i]))
90  if (!cr->IsInCombat())
91  {
92  cr->RemoveAllAuras();
93  if (Aura* aur = cr->AddAura(SPELL_STONED_AURA, cr))
94  {
95  aur->SetMaxDuration(60 * MINUTE * IN_MILLISECONDS);
96  aur->SetDuration(60 * MINUTE * IN_MILLISECONDS);
97  }
98  }
99 
100  stoned = true;
101  }
102  else if (bf->GetTimer() <= (2 * MINUTE * IN_MILLISECONDS) && bf->GetTimer() > (MINUTE * IN_MILLISECONDS))
103  {
104  Map::PlayerList const& PlayerList = instance->GetPlayers();
105  if (!PlayerList.isEmpty())
106  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
107  if (Player* player = i->GetSource())
108  player->TextEmote("This instance is about to reset. Prepare to be removed.", nullptr, true);
109  }
110  else if (bf->GetTimer() <= MINUTE * IN_MILLISECONDS)
111  {
112  for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
113  if (Creature* cr = instance->GetCreature(bossGUIDs[i]))
114  if (cr->IsInCombat() && cr->AI())
115  cr->AI()->EnterEvadeMode();
116 
117  Map::PlayerList const& PlayerList = instance->GetPlayers();
118  if (!PlayerList.isEmpty())
119  for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
120  if (Player* player = i->GetSource())
121  player->TeleportTo(player->m_homebindMapId, player->m_homebindX, player->m_homebindY, player->m_homebindZ, player->GetOrientation());
122  }
123  }
124  }
125  }
126  }

210  bool CheckAchievementCriteriaMeet(uint32 criteria_id, Player const* /*source*/, Unit const* /*target*/, uint32 /*miscvalue1*/) override
211  {
212  switch (criteria_id)
213  {
214  case CRITERIA_EARTH_WIND_FIRE_10:
215  case CRITERIA_EARTH_WIND_FIRE_25:
216  if (ArchavonDeath && EmalonDeath && KoralonDeath)
217  {
218  // instance difficulty check is already done in db (achievement_criteria_data)
219  // int() for Visual Studio, compile errors with abs(time_t)
220  return (abs(int(ArchavonDeath - EmalonDeath)) < MINUTE && \
221  abs(int(EmalonDeath - KoralonDeath)) < MINUTE && \
222  abs(int(KoralonDeath - ArchavonDeath)) < MINUTE);
223  }
224  break;
225  default:
226  break;
227  }
228 
229  return false;
230  }

243  void Load(const char* in) override
244  {
245  if (!in)
246  {
247  OUT_LOAD_INST_DATA_FAIL;
248  return;
249  }
250 
251  OUT_LOAD_INST_DATA(in);
252 
253  char dataHead1, dataHead2, dataHead3;
254  std::istringstream loadStream(in);
255  loadStream >> dataHead1 >> dataHead2 >> dataHead3;
256 
257  if (dataHead1 == 'V' && dataHead2 == 'O' && dataHead3 == 'A')
258  {
259  for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
260  {
261  loadStream >> m_auiEncounter[i];
262  if (m_auiEncounter[i] == IN_PROGRESS)
263  m_auiEncounter[i] = NOT_STARTED;
264  }
265 
266  OUT_LOAD_INST_DATA_COMPLETE;
267  }
268  else
269  OUT_LOAD_INST_DATA_FAIL;
270  }
