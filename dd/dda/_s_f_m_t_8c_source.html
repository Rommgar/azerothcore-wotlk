170 static void period_certification(sfmt_t * sfmt) {
171  uint32_t inner = 0;
172  int i, j;
173  uint32_t work;
174  uint32_t *psfmt32 = &sfmt->state[0].u[0];
175  const uint32_t parity[4] = {SFMT_PARITY1, SFMT_PARITY2,
176  SFMT_PARITY3, SFMT_PARITY4};
177 
178  for (i = 0; i < 4; i++) {
179  inner ^= psfmt32[idxof(i)] & parity[i];
180  }
181  for (i = 16; i > 0; i >>= 1) {
182  inner ^= inner >> i;
183  }
184  inner &= 1;
185  /* check OK */
186  if (inner == 1) {
187  return;
188  }
189  /* check NG, and modification */
190  for (i = 0; i < 4; i++) {
191  work = 1;
192  for (j = 0; j < 32; j++) {
193  if ((work & parity[i]) != 0) {
194  psfmt32[idxof(i)] ^= work;
195  return;
196  }
197  work = work << 1;
198  }
199  }
