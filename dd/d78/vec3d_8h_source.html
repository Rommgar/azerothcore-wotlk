247 inline void rotate(float x0, float y0, float* x, float* y, float angle)
248 {
249  float xa = *x - x0, ya = *y - y0;
250  *x = xa * cosf(angle) - ya * sinf(angle) + x0;
251  *y = xa * sinf(angle) + ya * cosf(angle) + y0;
252 }
