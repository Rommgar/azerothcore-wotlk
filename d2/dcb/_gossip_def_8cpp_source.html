77 void GossipMenu::AddMenuItem(uint32 menuId, uint32 menuItemId, uint32 sender, uint32 action)
78 {
80  GossipMenuItemsMapBounds bounds = sObjectMgr->GetGossipMenuItemsMapBounds(menuId);
82  if (bounds.first == bounds.second)
83  return;
84 
86  for (GossipMenuItemsContainer::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
87  {
89  if (itr->second.OptionID != menuItemId)
90  continue;
91 
93  std::string strOptionText, strBoxText;
94  BroadcastText const* optionBroadcastText = sObjectMgr->GetBroadcastText(itr->second.OptionBroadcastTextID);
95  BroadcastText const* boxBroadcastText = sObjectMgr->GetBroadcastText(itr->second.BoxBroadcastTextID);
96 
98  if (optionBroadcastText)
99  ObjectMgr::GetLocaleString(optionBroadcastText->MaleText, GetLocale(), strOptionText);
100  else
101  strOptionText = itr->second.OptionText;
102 
104  if (boxBroadcastText)
105  ObjectMgr::GetLocaleString(boxBroadcastText->MaleText, GetLocale(), strBoxText);
106  else
107  strBoxText = itr->second.BoxText;
108 
110  if (GetLocale() != DEFAULT_LOCALE)
111  {
112  if (!optionBroadcastText)
113  {
115  if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, menuItemId)))
116  ObjectMgr::GetLocaleString(gossipMenuLocale->OptionText, GetLocale(), strOptionText);
117  }
118 
119  if (!boxBroadcastText)
120  {
122  if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, menuItemId)))
123  ObjectMgr::GetLocaleString(gossipMenuLocale->BoxText, GetLocale(), strBoxText);
124  }
125  }
126 
128  AddMenuItem(-1, itr->second.OptionIcon, strOptionText, sender, action, strBoxText, itr->second.BoxMoney, itr->second.BoxCoded);
129  }

375  WorldPacket data(SMSG_QUESTGIVER_QUEST_DETAILS, 500); // guess size

379  data << questTitle;

383  data << uint32(quest->GetFlags()); // 3.3.3 questFlags

385  data << uint8(0); // IsFinished? value is sent back to server in quest accept packet

433  data << float(0.0f); // unk, honor multiplier?
434  data << uint32(quest->GetRewSpell()); // reward spell, this spell will display (icon) (casted if RewSpellCast == 0)
435  data << int32(quest->GetRewSpellCast()); // casted spell
436  data << uint32(quest->GetCharTitleId()); // CharTitleId, new 2.4.0, player gets this title (id from CharTitles)
437  data << uint32(quest->GetBonusTalents()); // bonus talents
438  data << uint32(quest->GetRewArenaPoints()); // reward arena points
439  data << uint32(0); // unk
440 
441  for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i)
442  data << uint32(quest->RewardFactionId[i]);
443 
444  for (uint32 i = 0; i < QUEST_REPUTATIONS_COUNT; ++i)
