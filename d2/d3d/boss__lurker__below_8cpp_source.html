127  void UpdateAI(uint32 diff) override
128  {
129  if (!UpdateVictim())
130  return;
131 
132  if (me->HasUnitState(UNIT_STATE_CASTING))
133  return;
134 
135  events.Update(diff);
136 
137  switch (events.ExecuteEvent())
138  {
139  case EVENT_SPELL_WHIRL:
140  me->CastSpell(me, SPELL_WHIRL, false);
141  events.ScheduleEvent(EVENT_SPELL_WHIRL, 18000);
142  break;
143  case EVENT_SPELL_GEYSER:
144  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0))
145  me->CastSpell(target, SPELL_GEYSER, false);
146  events.ScheduleEvent(EVENT_SPELL_GEYSER, 10000);
147  break;
148  case EVENT_SPELL_SPOUT:
149  Talk(EMOTE_TAKE_BREATH);
150  me->CastSpell(me, SPELL_SPOUT_VISUAL, TRIGGERED_IGNORE_SET_FACING);
151  me->SetReactState(REACT_PASSIVE);
152  me->SetFacingToObject(me->GetVictim());
153  me->SetTarget();
154  events.ScheduleEvent(EVENT_SPELL_SPOUT, 60000);
155  events.RescheduleEvent(EVENT_SPELL_WHIRL, 18000);
156  events.RescheduleEvent(EVENT_SPELL_GEYSER, 25000);
157  events.ScheduleEvent(EVENT_SPELL_SPOUT_PERIODIC, 3000);
158  break;
159  case EVENT_SPELL_SPOUT_PERIODIC:
160  me->InterruptNonMeleeSpells(false);
161  me->CastSpell(me, SPELL_SPOUT_PERIODIC, true);
162  break;
163  case EVENT_PHASE_2:
164  events.Reset();
165  events.ScheduleEvent(EVENT_PHASE_1, 60000);
166  me->SetStandState(UNIT_STAND_STATE_SUBMERGED);
167  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
168  for (uint8 i = 0; i < MAX_SUMMONS; ++i)
169  me->SummonCreature(i < 6 ? NPC_COILFANG_AMBUSHER : NPC_COILFANG_GUARDIAN, positions[i].GetPositionX(), positions[i].GetPositionY(), positions[i].GetPositionZ(), positions[i].GetAngle(me), TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT, 10000);
170  break;
171  case EVENT_PHASE_1:
172  me->setAttackTimer(BASE_ATTACK, 6000);
173  me->SetStandState(UNIT_STAND_STATE_STAND);
174  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
175 
176  events.Reset();
177  events.ScheduleEvent(EVENT_SPELL_SPOUT, 10000);
178  events.ScheduleEvent(EVENT_PHASE_2, 120000);
179  break;
180  }
181 
182  if (me->getStandState() != UNIT_STAND_STATE_STAND || !me->isAttackReady() || me->GetReactState() != REACT_AGGRESSIVE)
183  return;
184 
185  Unit* target = nullptr;
186  if (me->IsWithinMeleeRange(me->GetVictim()))
187  target = me->GetVictim();
188  else
189  {
190  ThreatContainer::StorageType const& t_list = me->getThreatMgr().getThreatList();
191  for (ThreatContainer::StorageType::const_iterator itr = t_list.begin(); itr != t_list.end(); ++itr)
192  if (Unit* threatTarget = ObjectAccessor::GetUnit(*me, (*itr)->getUnitGuid()))
193  if (me->IsWithinMeleeRange(threatTarget))
194  {
195  target = threatTarget;
196  break;
197  }
198  }
199 
200  if (target)
201  me->AttackerStateUpdate(target);
202  else if ((target = SelectTarget(SELECT_TARGET_RANDOM, 0)))
203  me->CastSpell(target, SPELL_WATER_BOLT, false);
204 
205  me->resetAttackTimer();
206  }
207  };
