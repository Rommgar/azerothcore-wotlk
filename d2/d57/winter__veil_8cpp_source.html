151  void HandleTriggerSpell(SpellEffIndex /*effIndex*/)
152  {
153  Unit* caster = GetCaster();
154  PreventHitEffect(EFFECT_0);
155  PreventHitEffect(EFFECT_1);
156 
157  std::list<Creature*> unitList;
158  Unit* target = nullptr;
159  caster->GetCreaturesWithEntryInRange(unitList, 30.0f, NPC_BLUE_RACER);
160  if (!unitList.empty())
161  for (std::list<Creature*>::const_iterator itr = unitList.begin(); itr != unitList.end(); ++itr)
162  if (caster->HasInLine((*itr), 1.0f) && (*itr)->GetGUID() != caster->GetGUID())
163  {
164  target = (*itr);
165  break;
166  }
167  if (!target)
168  {
169  unitList.clear();
170  caster->GetCreaturesWithEntryInRange(unitList, 30.0f, NPC_RED_RACER);
171  if (!unitList.empty())
172  for (std::list<Creature*>::const_iterator itr = unitList.begin(); itr != unitList.end(); ++itr)
173  if (caster->HasInLine((*itr), 1.0f) && (*itr)->GetGUID() != caster->GetGUID())
174  {
175  target = (*itr);
176  break;
177  }
178  }
179 
180  if (target)
181  {
182  caster->CastSpell(target, SPELL_RACER_CHARGE_TO_OBJECT, true);
183  caster->CastSpell(target, SPELL_RACER_SLAM_HIT, true);
184  }
185  else
186  {
187  Position pos;
188  float x = caster->GetPositionX() + 30 * cos(caster->GetOrientation());
189  float y = caster->GetPositionY() + 30 * sin(caster->GetOrientation());
190  pos.Relocate(x, y, caster->GetMap()->GetHeight(x, y, MAX_HEIGHT) + 0.5f);
191  //caster->GetFirstCollisionPosition(pos, 30.0f, caster->GetOrientation());
192  caster->CastSpell(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), SPELL_RACER_CHARGE_TO_OBJECT, true);
193  }
194  }
