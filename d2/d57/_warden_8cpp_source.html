49 void Warden::SendModuleToClient()
50 {
51  LOG_DEBUG("warden", "Send module to client");
52 
53  // Create packet structure
54  WardenModuleTransfer packet;
55 
56  uint32 sizeLeft = _module->CompressedSize;
57  uint32 pos = 0;
58  uint16 burstSize;
59  while (sizeLeft > 0)
60  {
61  burstSize = sizeLeft < 500 ? sizeLeft : 500;
62  packet.Command = WARDEN_SMSG_MODULE_CACHE;
63  packet.DataSize = burstSize;
64  memcpy(packet.Data, &_module->CompressedData[pos], burstSize);
65  sizeLeft -= burstSize;
66  pos += burstSize;
67 
68  EncryptData((uint8*)&packet, burstSize + 3);
69  WorldPacket pkt1(SMSG_WARDEN_DATA, burstSize + 3);
70  pkt1.append((uint8*)&packet, burstSize + 3);
71  _session->SendPacket(&pkt1);
72  }
73 }

174 uint32 Warden::BuildChecksum(const uint8* data, uint32 length)
175 {
176  keyData hash;
177  SHA1(data, length, hash.bytes.bytes);
178  uint32 checkSum = 0;
179  for (uint8 i = 0; i < 5; ++i)
180  {
181  checkSum = checkSum ^ hash.ints.ints[i];
182  }
183 
184  return checkSum;

202 void Warden::ApplyPenalty(uint16 checkId, std::string const& reason)
203 {
204  WardenCheck const* checkData = sWardenCheckMgr->GetWardenDataById(checkId);
205 
206  uint32 action = WardenActions(sWorld->getIntConfig(CONFIG_WARDEN_CLIENT_FAIL_ACTION));
207  std::string causeMsg;
208  if (checkId && checkData)
209  {
210  action = checkData->Action;
211 
212  if (checkData->Comment.empty())
213  {
214  causeMsg = "Warden id " + std::to_string(checkId) + " violation";
215  }
216  else
217  {
218  causeMsg = "Warden: " + checkData->Comment;
219  }
220  }
221  else
222  {
223  // if its not warden check id based, reason must be always provided
224  ASSERT(!reason.empty());
225  causeMsg = reason;
226  }
227 
228  switch (action)
229  {
230  case WARDEN_ACTION_LOG:
231  break;
232  case WARDEN_ACTION_KICK:
233  {
234  _session->KickPlayer(causeMsg.find("Warden") != std::string::npos ? causeMsg : "Warden: " + causeMsg);
235  break;
236  }
237  case WARDEN_ACTION_BAN:
238  {
239  std::stringstream duration;
240  duration << sWorld->getIntConfig(CONFIG_WARDEN_CLIENT_BAN_DURATION) << "s";
241  std::string accountName;
242  AccountMgr::GetName(_session->GetAccountId(), accountName);
243  sBan->BanAccount(accountName, duration.str(), causeMsg, "Server");
244  break;
245  }
246  }
247 
248  std::string reportMsg;
249  if (checkId)
250  {
251  if (Player const* plr = _session->GetPlayer())
252  {
253  std::string const reportFormat = "Player %s (guid %u, account id: %u) failed warden %u check (%s). Action: %s";
254  reportMsg = Acore::StringFormat(reportFormat, plr->GetName().c_str(), plr->GetGUID().GetCounter(), _session->GetAccountId(),
255  checkId, ((checkData && !checkData->Comment.empty()) ? checkData->Comment.c_str() : "<warden comment is not set>"),
256  GetWardenActionStr(action).c_str());
257  }
258  else
259  {
260  std::string const reportFormat = "Account id: %u failed warden %u check. Action: %s";
261  reportMsg = Acore::StringFormat(reportFormat, _session->GetAccountId(), checkId, GetWardenActionStr(action).c_str());
262  }
263  }
264  else
265  {
266  if (Player const* plr = _session->GetPlayer())
267  {
268  std::string const reportFormat = "Player %s (guid %u, account id: %u) triggered warden penalty by reason: %s. Action: %s";
269  reportMsg = Acore::StringFormat(reportFormat, plr->GetName().c_str(), plr->GetGUID().GetCounter(), _session->GetAccountId(), causeMsg.c_str(), GetWardenActionStr(action).c_str());
270  }
271  else
272  {
273  std::string const reportFormat = "Account id: %u failed warden %u check. Action: %s";
274  reportMsg = Acore::StringFormat(reportFormat, _session->GetAccountId(), causeMsg.c_str(), GetWardenActionStr(action).c_str());
275  }
276  }
277 
278  reportMsg = "Warden: " + reportMsg;
279  LOG_INFO("warden", "> Warden: %s", reportMsg.c_str());
280 }

282 bool Warden::ProcessLuaCheckResponse(std::string const& msg)
283 {
284  static constexpr char WARDEN_TOKEN[] = "_TW\t";
285  // if msg starts with WARDEN_TOKEN
286  if (msg.rfind(WARDEN_TOKEN, 0) != 0)
287  {
288  return false;
289  }
290 
291  uint16 id = 0;
292 
293  {
294  std::stringstream msg2(msg);
295  std::string temp;
296  while (msg2 >> temp)
297  {
298  // Found check id - stop loop
299  if (std::stringstream(temp) >> id)
300  break;
301  }
302  }
303 
304  if (id < sWardenCheckMgr->GetMaxValidCheckId())
305  {
306  WardenCheck const* check = sWardenCheckMgr->GetWardenDataById(id);
307  if (check && check->Type == LUA_EVAL_CHECK)
308  {
309  ApplyPenalty(id, "");
310  return true;
311  }
312  }
313 
314  ApplyPenalty(0, "Sent bogus Lua check response for Warden");
315  return true;

318 void WorldSession::HandleWardenDataOpcode(WorldPacket& recvData)
319 {
320  if (!_warden || recvData.empty())
321  return;
322 
323  _warden->DecryptData(recvData.contents(), recvData.size());
324  uint8 opcode;
325  recvData >> opcode;
326  LOG_DEBUG("warden", "Got packet, opcode %02X, size %u", opcode, uint32(recvData.size()));
327  recvData.hexlike();
328 
329  switch (opcode)
330  {
331  case WARDEN_CMSG_MODULE_MISSING:
332  _warden->SendModuleToClient();
333  break;
334  case WARDEN_CMSG_MODULE_OK:
335  _warden->RequestHash();
336  break;
337  case WARDEN_CMSG_CHEAT_CHECKS_RESULT:
338  _warden->HandleData(recvData);
339  break;
340  case WARDEN_CMSG_MEM_CHECKS_RESULT:
341  LOG_DEBUG("warden", "NYI WARDEN_CMSG_MEM_CHECKS_RESULT received!");
342  break;
343  case WARDEN_CMSG_HASH_RESULT:
344  _warden->HandleHashResult(recvData);
345  _warden->InitializeModule();
346  break;
347  case WARDEN_CMSG_MODULE_FAILED:
348  LOG_DEBUG("warden", "NYI WARDEN_CMSG_MODULE_FAILED received!");
349  break;
350  default:
351  LOG_DEBUG("warden", "Got unknown warden opcode %02X of size %u.", opcode, uint32(recvData.size() - 1));
352  break;
353  }
