295 bool AchievementCriteriaData::Meets(uint32 criteria_id, Player const* source, Unit const* target, uint32 miscvalue1 /*= 0*/) const
296 {
297  switch (dataType)
298  {
299  case ACHIEVEMENT_CRITERIA_DATA_TYPE_NONE:
300  return true;
301  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_CREATURE:
302  if (!target || target->GetTypeId() != TYPEID_UNIT)
303  return false;
304  return target->GetEntry() == creature.id;
305  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_CLASS_RACE:
306  if (!target || target->GetTypeId() != TYPEID_PLAYER)
307  return false;
308  if (classRace.class_id && classRace.class_id != target->ToPlayer()->getClass())
309  return false;
310  if (classRace.race_id && classRace.race_id != target->ToPlayer()->getRace())
311  return false;
312  return true;
313  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_PLAYER_CLASS_RACE:
314  if (!source || source->GetTypeId() != TYPEID_PLAYER)
315  return false;
316  if (classRace.class_id && classRace.class_id != source->ToPlayer()->getClass())
317  return false;
318  if (classRace.race_id && classRace.race_id != source->ToPlayer()->getRace())
319  return false;
320  return true;
321  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_LESS_HEALTH:
322  if (!target || target->GetTypeId() != TYPEID_PLAYER)
323  return false;
324  return !target->HealthAbovePct(health.percent);
325  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_PLAYER_DEAD:
326  if (target && !target->IsAlive())
327  if (const Player* player = target->ToPlayer())
328  if (player->GetDeathTimer() != 0)
329  // flag set == must be same team, not set == different team
330  return (player->GetTeamId() == source->GetTeamId()) == (player_dead.own_team_flag != 0);
331  return false;
332  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AURA:
333  return source->HasAuraEffect(aura.spell_id, aura.effect_idx);
334  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_AREA:
335  {
336  uint32 zone_id, area_id;
337  source->GetZoneAndAreaId(zone_id, area_id);
338  return area.id == zone_id || area.id == area_id;
339  }
340  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_AURA:
341  return target && target->HasAuraEffect(aura.spell_id, aura.effect_idx);
342  case ACHIEVEMENT_CRITERIA_DATA_TYPE_VALUE:
343  return CompareValues(ComparisionType(value.compType), miscvalue1, value.value);
344  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_LEVEL:
345  if (!target)
346  return false;
347  return target->getLevel() >= level.minlevel;
348  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_GENDER:
349  if (!target)
350  return false;
351  return target->getGender() == gender.gender;
352  case ACHIEVEMENT_CRITERIA_DATA_TYPE_SCRIPT:
353  return sScriptMgr->OnCriteriaCheck(ScriptId, const_cast<Player*>(source), const_cast<Unit*>(target), criteria_id);
354  case ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_DIFFICULTY:
355  {
356  if (source->GetMap()->IsRaid())
357  if (source->GetMap()->Is25ManRaid() != ((difficulty.difficulty & RAID_DIFFICULTY_MASK_25MAN) != 0))
358  return false;
359 
360  AchievementCriteriaEntry const* criteria = sAchievementCriteriaStore.LookupEntry(criteria_id);
361  uint8 spawnMode = source->GetMap()->GetSpawnMode();
362  // Dungeons completed on heroic mode count towards both in general achievement, but not in statistics.
363  return sAchievementMgr->IsStatisticCriteria(criteria) ? spawnMode == difficulty.difficulty : spawnMode >= difficulty.difficulty;
364  }
365  case ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_PLAYER_COUNT:
366  return source->GetMap()->GetPlayersCountExceptGMs() <= map_players.maxcount;
367  case ACHIEVEMENT_CRITERIA_DATA_TYPE_T_TEAM:
368  {
369  if (!target || target->GetTypeId() != TYPEID_PLAYER)
370  return false;
371 
372  // DB data compatibility...
373  uint32 teamOld = target->ToPlayer()->GetTeamId() == TEAM_ALLIANCE ? ALLIANCE : HORDE;
374  return teamOld == team.team;
375  }
376  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_DRUNK:
377  return Player::GetDrunkenstateByValue(source->GetDrunkValue()) >= DrunkenState(drunk.state);
378  case ACHIEVEMENT_CRITERIA_DATA_TYPE_HOLIDAY:
379  return IsHolidayActive(HolidayIds(holiday.id));
380  case ACHIEVEMENT_CRITERIA_DATA_TYPE_BG_LOSS_TEAM_SCORE:
381  {
382  Battleground* bg = source->GetBattleground();
383  if (!bg)
384  return false;
385 
386  uint32 score = bg->GetTeamScore(source->GetTeamId() == TEAM_ALLIANCE ? TEAM_HORDE : TEAM_ALLIANCE);
387  return score >= bg_loss_team_score.min_score && score <= bg_loss_team_score.max_score;
388  }
389  case ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT:
390  {
391  if (!source->IsInWorld())
392  return false;
393  Map* map = source->GetMap();
394  if (!map->IsDungeon())
395  {
396  LOG_ERROR("sql.sql", "Achievement system call ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT (%u) for achievement criteria %u for non-dungeon/non-raid map %u",
397  ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT, criteria_id, map->GetId());
398  return false;
399  }
400  InstanceScript* instance = map->ToInstanceMap()->GetInstanceScript();
401  if (!instance)
402  {
403  LOG_ERROR("sql.sql", "Achievement system call ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT (%u) for achievement criteria %u for map %u but map does not have a instance script",
404  ACHIEVEMENT_CRITERIA_DATA_TYPE_INSTANCE_SCRIPT, criteria_id, map->GetId());
405  return false;
406  }
407  return instance->CheckAchievementCriteriaMeet(criteria_id, source, target, miscvalue1);
408  }
409  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_EQUIPED_ITEM:
410  {
411  ItemTemplate const* pProto = sObjectMgr->GetItemTemplate(miscvalue1);
412  if (!pProto)
413  return false;
414  return pProto->ItemLevel >= equipped_item.item_level && pProto->Quality >= equipped_item.item_quality;
415  }
416  case ACHIEVEMENT_CRITERIA_DATA_TYPE_MAP_ID:
417  return source->GetMapId() == map_id.mapId;
418  case ACHIEVEMENT_CRITERIA_DATA_TYPE_NTH_BIRTHDAY:
419  {
420  time_t birthday_start = time_t(sWorld->getIntConfig(CONFIG_BIRTHDAY_TIME));
421  tm birthday_tm;
422  localtime_r(&birthday_start, &birthday_tm);
423 
424  // exactly N birthday
425  birthday_tm.tm_year += birthday_login.nth_birthday;
426 
427  time_t birthday = mktime(&birthday_tm);
428  time_t now = sWorld->GetGameTime();
429  return now <= birthday + DAY && now >= birthday;
430  }
431  case ACHIEVEMENT_CRITERIA_DATA_TYPE_S_KNOWN_TITLE:
432  {
