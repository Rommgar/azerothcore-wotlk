59 bool FleeingMovementGenerator<T>::_getPoint(T* owner, float& x, float& y, float& z)
60 {
61  if (!owner)
62  return false;
63 
64  const Map* _map = owner->GetMap();
65 
66  x = owner->GetPositionX();
67  y = owner->GetPositionY();
68  z = owner->GetPositionZ();
69 
70  float temp_x, temp_y, angle;
71  // primitive path-finding
72  for (uint8 i = 0; i < 18; ++i)
73  {
74  if (i_only_forward && i > 2)
75  break;
76 
77  float distance = 5.0f;
78 
79  switch (i)
80  {
81  case 0:
82  angle = i_cur_angle;
83  break;
84  case 1:
85  angle = i_cur_angle;
86  distance /= 2;
87  break;
88  case 2:
89  angle = i_cur_angle;
90  distance /= 4;
91  break;
92  case 3:
93  angle = i_cur_angle + static_cast<float>(M_PI / 4);
94  break;
95  case 4:
96  angle = i_cur_angle - static_cast<float>(M_PI / 4);
97  break;
98  case 5:
99  angle = i_cur_angle + static_cast<float>(M_PI / 4);
100  distance /= 2;
101  break;
102  case 6:
103  angle = i_cur_angle - static_cast<float>(M_PI / 4);
104  distance /= 2;
105  break;
106  case 7:
107  angle = i_cur_angle + static_cast<float>(M_PI / 2);
108  break;
109  case 8:
110  angle = i_cur_angle - static_cast<float>(M_PI / 2);
111  break;
112  case 9:
113  angle = i_cur_angle + static_cast<float>(M_PI / 2);
114  distance /= 2;
115  break;
116  case 10:
117  angle = i_cur_angle - static_cast<float>(M_PI / 2);
118  distance /= 2;
119  break;
120  case 11:
121  angle = i_cur_angle + static_cast<float>(M_PI / 4);
122  distance /= 4;
123  break;
124  case 12:
125  angle = i_cur_angle - static_cast<float>(M_PI / 4);
126  distance /= 4;
127  break;
128  case 13:
129  angle = i_cur_angle + static_cast<float>(M_PI / 2);
130  distance /= 4;
131  break;
132  case 14:
133  angle = i_cur_angle - static_cast<float>(M_PI / 2);
134  distance /= 4;
135  break;
136  case 15:
137  angle = i_cur_angle + static_cast<float>(3 * M_PI / 4);
138  distance /= 2;
139  break;
140  case 16:
141  angle = i_cur_angle - static_cast<float>(3 * M_PI / 4);
142  distance /= 2;
143  break;
144  case 17:
145  angle = i_cur_angle + static_cast<float>(M_PI);
146  distance /= 2;
147  break;
148  default:
149  angle = 0.0f;
150  distance = 0.0f;
151  break;
152  }
153 
154  temp_x = x + distance * cos(angle);
155  temp_y = y + distance * sin(angle);
156  float temp_z = z;
157 
158  if (!_map->CanReachPositionAndGetValidCoords(owner, temp_x, temp_y, temp_z, true, true))
159  {
160  break;
161  }
162 
163  if (!(temp_z - z) || distance / fabs(temp_z - z) > 1.0f)
164  {
165  float temp_z_left = _map->GetHeight(owner->GetPhaseMask(), temp_x + 1.0f * cos(angle + static_cast<float>(M_PI / 2)), temp_y + 1.0f * sin(angle + static_cast<float>(M_PI / 2)), z, true);
166  float temp_z_right = _map->GetHeight(owner->GetPhaseMask(), temp_x + 1.0f * cos(angle - static_cast<float>(M_PI / 2)), temp_y + 1.0f * sin(angle - static_cast<float>(M_PI / 2)), z, true);
167  if (fabs(temp_z_left - temp_z) < 1.2f && fabs(temp_z_right - temp_z) < 1.2f)
168  {
169  // use new values
170  x = temp_x;
171  y = temp_y;
172  z = temp_z;
173  return true;
174  }
175  }
176  }
177  i_to_distance_from_caster = 0.0f;
178  i_nextCheckTime.Reset(urand(500, 1000));
179  return false;
180 }

182 template<class T>
183 bool FleeingMovementGenerator<T>::_setMoveData(T* owner)
184 {
185  float cur_dist_xyz = owner->GetDistance(i_caster_x, i_caster_y, i_caster_z);
186 
187  if (i_to_distance_from_caster > 0.0f)
188  {
189  if ((i_last_distance_from_caster > i_to_distance_from_caster && cur_dist_xyz < i_to_distance_from_caster) ||
190  // if we reach lower distance
191  (i_last_distance_from_caster > i_to_distance_from_caster && cur_dist_xyz > i_last_distance_from_caster) ||
192  // if we can't be close
193  (i_last_distance_from_caster < i_to_distance_from_caster && cur_dist_xyz > i_to_distance_from_caster) ||
194  // if we reach bigger distance
195  (cur_dist_xyz > MAX_QUIET_DISTANCE) || // if we are too far
196  (i_last_distance_from_caster > MIN_QUIET_DISTANCE && cur_dist_xyz < MIN_QUIET_DISTANCE))
197  // if we leave 'quiet zone'
198  {
199  // we are very far or too close, stopping
200  i_to_distance_from_caster = 0.0f;
201  i_nextCheckTime.Reset(urand(500, 1000));
202  return false;
203  }
204  else
205  {
206  // now we are running, continue
207  i_last_distance_from_caster = cur_dist_xyz;
208  return true;
209  }
210  }
211 
212  float cur_dist;
213  float angle_to_caster;
214 
215  if (Unit* fright = ObjectAccessor::GetUnit(*owner, i_frightGUID))
216  {
217  cur_dist = fright->GetDistance(owner);
218  if (cur_dist < cur_dist_xyz)
219  {
220  i_caster_x = fright->GetPositionX();
221  i_caster_y = fright->GetPositionY();
222  i_caster_z = fright->GetPositionZ();
223  angle_to_caster = fright->GetAngle(owner);
224  }
225  else
226  {
227  cur_dist = cur_dist_xyz;
228  angle_to_caster = owner->GetAngle(i_caster_x, i_caster_y) + static_cast<float>(M_PI);
229  }
230  }
231  else
232  {
233  cur_dist = cur_dist_xyz;
234  angle_to_caster = owner->GetAngle(i_caster_x, i_caster_y) + static_cast<float>(M_PI);
235  }
236 
237  // if we too close may use 'path-finding' else just stop
238  i_only_forward = cur_dist >= MIN_QUIET_DISTANCE / 3;
239 
240  //get angle and 'distance from caster' to run
241  float angle;
242 
243  if (i_cur_angle == 0.0f && i_last_distance_from_caster == 0.0f) //just started, first time
244  {
245  angle = (float)rand_norm() * (1.0f - cur_dist / MIN_QUIET_DISTANCE) * static_cast<float>(M_PI / 3) + (float)rand_norm() * static_cast<float>(M_PI * 2 / 3);
246  i_to_distance_from_caster = MIN_QUIET_DISTANCE;
247  i_only_forward = true;
248  }
249  else if (cur_dist < MIN_QUIET_DISTANCE)
250  {
251  angle = static_cast<float>(M_PI / 6) + (float)rand_norm() * static_cast<float>(M_PI * 2 / 3);
252  i_to_distance_from_caster = cur_dist * 2 / 3 + (float)rand_norm() * (MIN_QUIET_DISTANCE - cur_dist * 2 / 3);
253  }
254  else if (cur_dist > MAX_QUIET_DISTANCE)
255  {
256  angle = (float)rand_norm() * static_cast<float>(M_PI / 3) + static_cast<float>(M_PI * 2 / 3);
257  i_to_distance_from_caster = MIN_QUIET_DISTANCE + 2.5f + (float)rand_norm() * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE - 2.5f);
258  }
259  else
260  {
261  angle = (float)rand_norm() * static_cast<float>(M_PI);
262  i_to_distance_from_caster = MIN_QUIET_DISTANCE + 2.5f + (float)rand_norm() * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE - 2.5f);
263  }
264 
265  int8 sign = (float)rand_norm() > 0.5f ? 1 : -1;
266  i_cur_angle = sign * angle + angle_to_caster;
267 
268  // current distance
269  i_last_distance_from_caster = cur_dist;
270 
271  return true;
272 }
