39 void RandomMovementGenerator<Creature>::_setRandomLocation(Creature* creature)
40 {
41  if (creature->_moveState != MAP_OBJECT_CELL_MOVE_NONE)
42  return;
43 
44  if (_validPointsVector[_currentPoint].empty())
45  {
46  if (_currentPoint == RANDOM_POINTS_NUMBER) // cant go anywhere from initial position, lets stay
47  return;
48  // go back to initial position and will never return to this point
49  _currentPoint = RANDOM_POINTS_NUMBER;
50  _currDestPosition.Relocate(_initialPosition);
51  creature->AddUnitState(UNIT_STATE_ROAMING_MOVE);
52  Movement::MoveSplineInit init(creature);
53  init.MoveTo(_currDestPosition.GetPositionX(), _currDestPosition.GetPositionY(), _currDestPosition.GetPositionZ());
54  init.SetWalk(true);
55  init.Launch();
56  if (creature->GetFormation() && creature->GetFormation()->getLeader() == creature)
57  creature->GetFormation()->LeaderMoveTo(_currDestPosition.GetPositionX(), _currDestPosition.GetPositionY(), _currDestPosition.GetPositionZ(), false);
58  return;
59  }
60 
61  uint8 random = urand(0, _validPointsVector[_currentPoint].size() - 1);
62  std::vector<uint8>::iterator randomIter = _validPointsVector[_currentPoint].begin() + random;
63  uint8 newPoint = *randomIter;
64  uint16 pathIdx = uint16(_currentPoint * RANDOM_POINTS_NUMBER + newPoint);
65 
66  // cant go anywhere from new point, so dont go there to not be stuck
67  if (_validPointsVector[newPoint].empty())
68  {
69  _validPointsVector[_currentPoint].erase(randomIter);
70  return;
71  }
72 
73  Movement::PointsArray& finalPath = _preComputedPaths[pathIdx];
74  if (finalPath.empty())
75  {
76  Map* map = creature->GetMap();
77  float x = _destinationPoints[newPoint].x, y = _destinationPoints[newPoint].y, z = _destinationPoints[newPoint].z;
78  // invalid coordinates
79  if (!Acore::IsValidMapCoord(x, y))
80  {
81  _validPointsVector[_currentPoint].erase(randomIter);
82  _preComputedPaths.erase(pathIdx);
83  return;
84  }
85 
86  float ground = INVALID_HEIGHT;
87  float levelZ = creature->GetMapWaterOrGroundLevel(x, y, z, &ground);
88  float newZ = INVALID_HEIGHT;
89 
90  // flying creature
91  if (creature->CanFly())
92  newZ = std::max<float>(levelZ, z + rand_norm() * _wanderDistance / 2.0f);
93  // point underwater
94  else if (ground < levelZ)
95  {
96  if (!creature->CanEnterWater())
97  {
98  _validPointsVector[_currentPoint].erase(randomIter);
99  _preComputedPaths.erase(pathIdx);
100  return;
101  }
102  else
103  {
104  if (levelZ > INVALID_HEIGHT)
105  newZ = std::min<float>(levelZ - 2.0f, z + rand_norm() * _wanderDistance / 2.0f);
106  newZ = std::max<float>(ground, newZ);
107  }
108  }
109  // point on ground
110  else
111  {
112  if (levelZ <= INVALID_HEIGHT || !creature->CanWalk())
113  {
114  _validPointsVector[_currentPoint].erase(randomIter);
115  _preComputedPaths.erase(pathIdx);
116  return;
117  }
118  }
119 
120  creature->UpdateAllowedPositionZ(x, y, newZ);
121 
122  if (newZ > INVALID_HEIGHT)
123  {
124  // flying / swiming creature - dest not in los
125  if (!creature->IsWithinLOS(x, y, newZ))
126  {
127  _validPointsVector[_currentPoint].erase(randomIter);
128  _preComputedPaths.erase(pathIdx);
129  return;
130  }
131 
132  finalPath.push_back(G3D::Vector3(creature->GetPositionX(), creature->GetPositionY(), creature->GetPositionZ()));
133  finalPath.push_back(G3D::Vector3(x, y, newZ));
134  }
135  else // ground
136  {
137  bool result = _pathGenerator->CalculatePath(x, y, levelZ, false);
138  if (result && !(_pathGenerator->GetPathType() & PATHFIND_NOPATH))
139  {
140  // generated path is too long
141  float pathLen = _pathGenerator->getPathLength();
142  if (pathLen * pathLen > creature->GetExactDistSq(x, y, levelZ) * MAX_PATH_LENGHT_FACTOR * MAX_PATH_LENGHT_FACTOR)
143  {
144  _validPointsVector[_currentPoint].erase(randomIter);
145  _preComputedPaths.erase(pathIdx);
146  return;
147  }
148 
149  finalPath = _pathGenerator->GetPath();
150  Movement::PointsArray::iterator itr = finalPath.begin();
151  Movement::PointsArray::iterator itrNext = finalPath.begin() + 1;
152  float zDiff, distDiff;
153 
154  for (; itrNext != finalPath.end(); ++itr, ++itrNext)
155  {
156  distDiff = sqrt(((*itr).x - (*itrNext).x) * ((*itr).x - (*itrNext).x) + ((*itr).y - (*itrNext).y) * ((*itr).y - (*itrNext).y));
157  zDiff = fabs((*itr).z - (*itrNext).z);
158 
159  // Xinef: tree climbing, cut as much as we can
160  if (zDiff > 2.0f ||
161  (G3D::fuzzyNe(zDiff, 0.0f) && distDiff / zDiff < 2.15f)) // ~25˚
162  {
163  _validPointsVector[_currentPoint].erase(randomIter);
164  _preComputedPaths.erase(pathIdx);
165  return;
166  }
167 
168  if (!map->isInLineOfSight((*itr).x, (*itr).y, (*itr).z + 2.f, (*itrNext).x, (*itrNext).y, (*itrNext).z + 2.f, creature->GetPhaseMask(), LINEOFSIGHT_ALL_CHECKS))
169  {
170  _validPointsVector[_currentPoint].erase(randomIter);
171  _preComputedPaths.erase(pathIdx);
172  return;
173  }
174  }
175 
176  // no valid path
177  if (finalPath.size() < 2)
178  {
179  _validPointsVector[_currentPoint].erase(randomIter);
180  _preComputedPaths.erase(pathIdx);
181  return;
182  }
183  }
184  else
185  {
186  _validPointsVector[_currentPoint].erase(randomIter);
187  _preComputedPaths.erase(pathIdx);
188  return;
189  }
190  }
191  }
192 
193  _currentPoint = newPoint;
194  G3D::Vector3& finalPoint = finalPath[finalPath.size() - 1];
195  _currDestPosition.Relocate(finalPoint.x, finalPoint.y, finalPoint.z);
196 
197  creature->AddUnitState(UNIT_STATE_ROAMING_MOVE);
198  ++_moveCount;
199  if (roll_chance_i((int32)_moveCount * 25 + 10))
200  {
201  _moveCount = 0;
202  _nextMoveTime.Reset(urand(4000, 8000));
203  }
204 
205  Movement::MoveSplineInit init(creature);
206  init.MovebyPath(finalPath);
207  init.SetWalk(true);
208  init.Launch();
209 
210  if (sWorld->getBoolConfig(CONFIG_DONT_CACHE_RANDOM_MOVEMENT_PATHS))
211  _preComputedPaths.erase(pathIdx);
212 
213  //Call for creature group update
214  if (creature->GetFormation() && creature->GetFormation()->getLeader() == creature)
215  creature->GetFormation()->LeaderMoveTo(finalPoint.x, finalPoint.y, finalPoint.z, false);
216 }

218 template<>
219 void RandomMovementGenerator<Creature>::DoInitialize(Creature* creature)
220 {
221  if (!creature->IsAlive())
222  return;
223 
224  if (!_wanderDistance)
225  _wanderDistance = creature->GetWanderDistance();
226 
227  _nextMoveTime.Reset(creature->GetSpawnId() && creature->GetWanderDistance() == _wanderDistance ? urand(1, 5000) : 0);
228  _wanderDistance = std::max((creature->GetWanderDistance() == _wanderDistance && creature->GetInstanceId() == 0) ? (creature->CanFly() ? MIN_WANDER_DISTANCE_AIR : MIN_WANDER_DISTANCE_GROUND) : 0.0f, _wanderDistance);
229 
230  if (G3D::fuzzyEq(_initialPosition.GetExactDist2d(0.0f, 0.0f), 0.0f))
231  {
232  _initialPosition.Relocate(creature);
233  _destinationPoints.clear();
234  for (uint8 i = 0; i < RANDOM_POINTS_NUMBER; ++i)
235  {
236  float angle = (M_PI * 2.0f / (float)RANDOM_POINTS_NUMBER) * i;
237  float factor = 0.5f + rand_norm() * 0.5f;
238  _destinationPoints.push_back(G3D::Vector3(_initialPosition.GetPositionX() + _wanderDistance * cos(angle)*factor, _initialPosition.GetPositionY() + _wanderDistance * sin(angle)*factor, _initialPosition.GetPositionZ()));
239  }
240  }
241 
242  if (!_pathGenerator)
243  _pathGenerator = new PathGenerator(creature);
244  creature->AddUnitState(UNIT_STATE_ROAMING | UNIT_STATE_ROAMING_MOVE);
245 }

260 template<>
261 bool RandomMovementGenerator<Creature>::DoUpdate(Creature* creature, const uint32 diff)
262 {
263  if (creature->HasUnitState(UNIT_STATE_NOT_MOVE) || creature->IsMovementPreventedByCasting())
264  {
265  _nextMoveTime.Reset(0); // Expire the timer
266  creature->StopMoving();
267  return true;
268  }
269 
270  // xinef: if we got disable move flag, do not remove default generator - just prevent movement
271  if (creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_DISABLE_MOVE))
272  {
273  _nextMoveTime.Reset(0); // Expire the timer
274  creature->ClearUnitState(UNIT_STATE_ROAMING_MOVE);
275  return true;
276  }
277 
278  if (creature->movespline->Finalized())
279  {
280  _nextMoveTime.Update(diff);
281  if (_nextMoveTime.Passed())
282  _setRandomLocation(creature);
283  }
284  return true;
285 }
