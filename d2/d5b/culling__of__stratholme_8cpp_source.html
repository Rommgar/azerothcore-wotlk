293  bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override
294  {
295  if (creature->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
296  {
297  switch (action)
298  {
299  case GOSSIP_ACTION_INFO_DEF+1:
300  creature->AI()->DoAction(ACTION_START_CITY);
301  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
302  CloseGossipMenuFor(player);
303  break;
304  case GOSSIP_ACTION_INFO_DEF+2:
305  ClearGossipMenuFor(player);
306  AddGossipItemFor(player, GOSSIP_ICON_CHAT, GOSSIP_ITEM_ARTHAS_3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
307  SendGossipMenuFor(player, GOSSIP_MENU_ARTHAS_3, creature->GetGUID());
308  break;
309  case GOSSIP_ACTION_INFO_DEF+3:
310  // Start Town Hall part
311  creature->AI()->DoAction(ACTION_START_TOWN_HALL);
312  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
313  CloseGossipMenuFor(player);
314  break;
315  case GOSSIP_ACTION_INFO_DEF+4:
316  // After killing epoch
317  creature->AI()->DoAction(ACTION_START_SECRET_PASSAGE);
318  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
319  creature->SetTarget();
320  CloseGossipMenuFor(player);
321  break;
322  case GOSSIP_ACTION_INFO_DEF+5:
323  creature->AI()->DoAction(ACTION_START_LAST_CITY);
324  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
325  CloseGossipMenuFor(player);
326  break;
327  case GOSSIP_ACTION_INFO_DEF+6:
328  creature->AI()->DoAction(ACTION_START_MALGANIS);
329  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
330  CloseGossipMenuFor(player);
331  break;
332  }
333  }
