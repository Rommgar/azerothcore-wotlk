33 // Validates a single hyperlink
34 HyperlinkInfo Acore::Hyperlinks::ParseSingleHyperlink(std::string_view str)
35 {
36  uint32 color = 0;
37  std::string_view tag;
38  std::string_view data;
39  std::string_view text;
40 
41  //color tag
42  if (str.substr(0, 2) != "|c")
43  return {};
44 
45  str.remove_prefix(2);
46 
47  if (str.length() < 8)
48  return {};
49 
50  for (uint8 i = 0; i < 8; ++i)
51  {
52  if (uint8 hex = toHex(str[i]))
53  color = (color << 4) | (hex & 0xf);
54  else
55  return {};
56  }
57 
58  str.remove_prefix(8);
59 
60  if (str.substr(0, 2) != "|H")
61  return {};
62 
63  str.remove_prefix(2);
64 
65  // tag+data part follows
66  if (size_t delimPos = str.find('|'); delimPos != std::string_view::npos)
67  {
68  tag = str.substr(0, delimPos);
69  str.remove_prefix(delimPos+1);
70  }
71  else
72  return {};
73 
74  // split tag if : is present (data separator)
75  if (size_t dataStart = tag.find(':'); dataStart != std::string_view::npos)
76  {
77  data = tag.substr(dataStart+1);
78  tag = tag.substr(0, dataStart);
79  }
80 
81  // ok, next should be link data end tag...
82  if (str.substr(0, 1) != "h")
83  return {};
84  str.remove_prefix(1);
85  // skip to final |
86  if (size_t end = str.find('|'); end != std::string_view::npos)
87  {
88  // check end tag
89  if (str.substr(end, 4) != "|h|r")
90  return {};
91  // check text brackets
92  if ((str[0] != '[') || (str[end - 1] != ']'))
93  return {};
94  text = str.substr(1, end - 2);
95  // tail
96  str = str.substr(end + 4);
97  }
98  else
99  return {};
100 
101  // ok, valid hyperlink, return info
102  return { str, color, tag, data, text };
103 }

146 
147  if (data.IsBuggedInspectLink) /* DBC lookup will have failed on the client, so the link should've arrived without suffix */
148  randomSuffixes = nullptr;

150  for (uint8 i = 0; i < TOTAL_LOCALES; ++i)
151  {
152  if (!locale && i != DEFAULT_LOCALE)
153  continue;
154

156  if (name.empty())
157  continue;
158 
159  if (randomSuffixes)
160  {
161  std::string_view randomSuffix((*randomSuffixes)[i]);

163  (text.length() == (name.length() + 1 + randomSuffix.length())) &&

378 // Validates all hyperlinks and control sequences contained in str
379 bool Acore::Hyperlinks::CheckAllLinks(std::string_view str)
380 {
381  // Step 1: Disallow all control sequences except ||, |H, |h, |c and |r
382  {
383  std::string_view::size_type pos = 0;
384  while ((pos = str.find('|', pos)) != std::string::npos)
385  {
386  ++pos;
387  if (pos == str.length())
388  return false;
389  char next = str[pos];
390  if (next == 'H' || next == 'h' || next == 'c' || next == 'r' || next == '|')
391  ++pos;
392  else
393  return false;
394  }
395  }
396 
397  // Step 2: Parse all link sequences
398  // They look like this: |c<color>|H<linktag>:<linkdata>|h[<linktext>]|h|r
399  // - <color> is 8 hex characters AARRGGBB
400  // - <linktag> is arbitrary length [a-z_]
401  // - <linkdata> is arbitrary length, no | contained
402  // - <linktext> is printable
403  {
404  std::string::size_type pos;
405  while ((pos = str.find('|')) != std::string::npos)
406  {
407  if (str[pos + 1] == '|') // this is an escaped pipe character (||)
408  {
409  str = str.substr(pos + 2);
410  continue;
411  }
412 
413  HyperlinkInfo info = ParseSingleHyperlink(str.substr(pos));
414  if (!info || !ValidateLinkInfo(info))
415  return false;
416 
417  // tag is fine, find the next one
418  str = info.tail;
419  }
420  }
421 
422  // all tags are valid
423  return true;
424 }
