92 {
93  emerald_dragonAI(Creature* creature) : WorldBossAI(creature)
94  {
95  }
96 
97  void Reset() override
98  {
99  WorldBossAI::Reset();
100  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_NON_ATTACKABLE);
101  me->SetReactState(REACT_AGGRESSIVE);
102  DoCast(me, SPELL_MARK_OF_NATURE_AURA, true);
103  events.ScheduleEvent(EVENT_TAIL_SWEEP, 4000);
104  events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
105  events.ScheduleEvent(EVENT_SEEPING_FOG, urand(12500, 20000));
106  }
107 
108  // Target killed during encounter, mark them as suspectible for Aura Of Nature
109  void KilledUnit(Unit* who) override
110  {
111  if (who->GetTypeId() == TYPEID_PLAYER)
112  who->CastSpell(who, SPELL_MARK_OF_NATURE, true);
113  }
114 
115  // Execute and reschedule base events shared between all Emerald Dragons
116  void ExecuteEvent(uint32 eventId) override
117  {
118  switch (eventId)
119  {
120  case EVENT_SEEPING_FOG:
121  // Seeping Fog appears only as "pairs", and only ONE pair at any given time!
122  // Despawntime is 2 minutes, so reschedule it for new cast after 2 minutes + a minor "random time" (30 seconds at max)
123  DoCast(me, SPELL_SEEPING_FOG_LEFT, true);
124  DoCast(me, SPELL_SEEPING_FOG_RIGHT, true);
125  events.ScheduleEvent(EVENT_SEEPING_FOG, urand(120000, 150000));
126  break;
127  case EVENT_NOXIOUS_BREATH:
128  // Noxious Breath is cast on random intervals, no less than 7.5 seconds between
129  DoCast(me, SPELL_NOXIOUS_BREATH);
130  events.ScheduleEvent(EVENT_NOXIOUS_BREATH, urand(7500, 15000));
131  break;
132  case EVENT_TAIL_SWEEP:
133  // Tail Sweep is cast every two seconds, no matter what goes on in front of the dragon
134  DoCast(me, SPELL_TAIL_SWEEP);
135  events.ScheduleEvent(EVENT_TAIL_SWEEP, 2000);
136  break;
137  }
138  }
139 
140  void UpdateAI(uint32 diff) override
141  {
142  if (!UpdateVictim())
143  return;
144 
145  events.Update(diff);
146 
147  if (me->HasUnitState(UNIT_STATE_CASTING))
148  return;
149 
150  while (uint32 eventId = events.ExecuteEvent())
151  ExecuteEvent(eventId);
152 
153  if (Unit* target = SelectTarget(SELECT_TARGET_TOPAGGRO, 0, -50.0f, true))
154  DoCast(target, SPELL_SUMMON_PLAYER);
155 
156  DoMeleeAttackIfReady();
157  }
158 };
