105  void UpdateAI(uint32 diff) override
106  {
107  _events2.Update(diff);
108  switch (_events2.ExecuteEvent())
109  {
110  case EVENT_CHECK_VISIBILITY:
111  if (instance->GetBossState(DATA_SERVANT_QUARTERS) == DONE)
112  {
113  me->SetVisible(true);
114  me->SetReactState(REACT_AGGRESSIVE);
115  me->RestoreFaction();
116  }
117  else
118  _events2.ScheduleEvent(EVENT_CHECK_VISIBILITY, 5000);
119  break;
120  }
121 
122  if (!UpdateVictim())
123  return;
124 
125  events.Update(diff);
126  if (me->HasUnitState(UNIT_STATE_CASTING))
127  return;
128 
129  switch (events.ExecuteEvent())
130  {
131  case EVENT_SPELL_ACIDIC_FANG:
132  me->CastSpell(me->GetVictim(), SPELL_ACIDIC_FANG, false);
133  events.ScheduleEvent(EVENT_SPELL_ACIDIC_FANG, urand(12000, 18000));
134  break;
135  case EVENT_SPELL_HYAKISS_WEB:
136  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f))
137  me->CastSpell(target, SPELL_HYAKISS_WEB, false);
138  events.ScheduleEvent(EVENT_SPELL_HYAKISS_WEB, 15000);
139  break;
140  case EVENT_SPELL_SONIC_BURST:
141  me->CastSpell(me, SPELL_SONIC_BURST, false);
142  events.ScheduleEvent(EVENT_SPELL_SONIC_BURST, urand(12000, 18000));
143  break;
144  case EVENT_SPELL_WING_BUFFET:
145  me->CastSpell(me, SPELL_WING_BUFFET, false);
146  events.ScheduleEvent(EVENT_SPELL_WING_BUFFET, urand(12000, 18000));
147  break;
148  case EVENT_SPELL_DIVE:
149  if (Unit* target = SelectTarget(SELECT_TARGET_FARTHEST, 0, FarthestTargetSelector(me, 40.0f, false, true)))
150  me->CastSpell(target, SPELL_DIVE, false);
151  events.ScheduleEvent(EVENT_SPELL_DIVE, 20000);
152  break;
153  case EVENT_SPELL_FEAR:
154  me->CastSpell(me->GetVictim(), SPELL_FEAR, false);
155  break;
156  case EVENT_SPELL_RAVAGE:
157  me->CastSpell(me->GetVictim(), SPELL_RAVAGE, false);
158  events.ScheduleEvent(EVENT_SPELL_RAVAGE, 10500);
159  break;
160  }
161 
162  DoMeleeAttackIfReady();
163  }
