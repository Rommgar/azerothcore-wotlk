157  void UpdateAI(uint32 diff) override
158  {
159  if (attackableTImer)
160  {
161  if (attackableTImer <= diff)
162  {
163  me->RemoveAurasDueToSpell(SPELL_RAGNAROS_SUBMERGE_EFFECT);
164  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_NON_ATTACKABLE);
165  me->SetReactState(REACT_AGGRESSIVE);
166  DoZoneInCombat();
167  attackableTImer = 0;
168  }
169  else
170  {
171  attackableTImer -= diff;
172  }
173  }
174 
175  if (_isBanished && (_emergeTimer <= diff || !summons.HasEntry(NPC_SON_OF_FLAME)))
176  {
177  //Become unbanished again
178  me->SetReactState(REACT_AGGRESSIVE);
179  me->SetFaction(14);
180  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
181  me->SetUInt32Value(UNIT_NPC_EMOTESTATE, 0);
182  me->HandleEmoteCommand(EMOTE_ONESHOT_EMERGE);
183  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 0.0f, true))
184  {
185  AttackStart(target);
186  }
187 
188  _isBanished = false;
189  }
190  else if (_isBanished)
191  {
192  _emergeTimer -= diff;
193  return;
194  }
195 
196  if (!UpdateVictim())
197  {
198  return;
199  }
200 
201  events.Update(diff);
202 
203  if (me->HasUnitState(UNIT_STATE_CASTING))
204  {
205  return;
206  }
207  while (uint32 const eventId = events.ExecuteEvent())
208  {
209  switch (eventId)
210  {
211  case EVENT_ERUPTION:
212  {
213  DoCastVictim(SPELL_ERRUPTION);
214  events.RepeatEvent(urand(20000, 45000));
215  break;
216  }
217  case EVENT_WRATH_OF_RAGNAROS:
218  {
219  DoCastVictim(SPELL_WRATH_OF_RAGNAROS);
220  if (urand(0, 1))
221  {
222  Talk(SAY_WRATH);
223  }
224  events.RepeatEvent(25000);
225  break;
226  }
227  case EVENT_HAND_OF_RAGNAROS:
228  {
229  DoCastSelf(SPELL_HAND_OF_RAGNAROS);
230  if (urand(0, 1))
231  {
232  Talk(SAY_HAND);
233  }
234  events.RepeatEvent(20000);
235  break;
236  }
237  case EVENT_LAVA_BURST:
238  {
239  DoCastVictim(SPELL_LAVA_BURST);
240  events.RepeatEvent(10000);
241  break;
242  }
243  case EVENT_MAGMA_BLAST:
244  {
245  Unit* victim = me->GetVictim();
246  if (victim && !me->IsWithinMeleeRange(victim))
247  {
248  DoCast(victim, SPELL_MAGMA_BLAST);
249  if (!_hasYelledMagmaBurst)
250  {
251  Talk(SAY_MAGMABURST);
252  _hasYelledMagmaBurst = true;
253  }
254  }
255  events.RepeatEvent(2500);
256  break;
257  }
258  case EVENT_SUBMERGE:
259  {
260  if (!_isBanished)
261  {
262  me->InterruptNonMeleeSpells(false);
263  me->AttackStop();
264  DoResetThreat();
265  me->SetReactState(REACT_PASSIVE);
266  me->SetFaction(35);
267  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
268  me->SetUInt32Value(UNIT_NPC_EMOTESTATE, EMOTE_STATE_SUBMERGED);
269  DoCastSelf(SPELL_RAGNA_SUBMERGE_VISUAL, true);
270  //me->HandleEmoteCommand(EMOTE_ONESHOT_SUBMERGE);
271  SummonMinions();
272  }
273  events.ScheduleEvent(EVENT_SUBMERGE, 180000);
274  break;
275  }
276  }
277 
278  if (me->HasUnitState(UNIT_STATE_CASTING))
279  {
280  return;
281  }
282  }
283 
284  DoMeleeAttackIfReady();
285  }
