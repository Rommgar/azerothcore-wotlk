105 
106  // Everything but a PBM will have a max color value
107  if (ppmType != "P2") {

109  } else {
110  maxColor = 255;
111  }
112 
113  if ((ppmWidth < 0) ||
114  (ppmHeight < 0) ||
115  (maxColor <= 0)) {
116  throw GImage::Error("Invalid PPM Header.", input.getFilename());
117  }
118 
119  // I don't think it's proper to scale values less than 255
120  if (maxColor <= 255.0) {
121  maxColor = 255.0;
122  }
123 
124  m_width = ppmWidth;
125  m_height = ppmHeight;
126  m_channels = 3;
127  // always scale down to 1 byte per channel
128  m_byte = (uint8*)m_memMan->alloc(m_width * m_height * 3);
129 
130  // Read in the image data. I am not validating if the values match the maxColor
131  // requirements. I only scale if needed to fit within the byte available.
132  for (uint32 i = 0; i < (uint32)(m_width * m_height); ++i) {
133  // read in color and scale to max pixel defined in header
134  // A max color less than 255 might need to be left alone and not scaled.
135  Color3uint8& curPixel = *(pixel3() + i);
136 
137  if (ppmType == "P3") {
138  curPixel.r = (uint8)(ppmInput.readNumber() * (255.0 / maxColor));
139  curPixel.g = (uint8)(ppmInput.readNumber() * (255.0 / maxColor));
140  curPixel.b = (uint8)(ppmInput.readNumber() * (255.0 / maxColor));
141  } else if (ppmType == "P2") {
142  uint8 pixel = (uint8)(ppmInput.readNumber() * (255.0 / maxColor));
143  curPixel.r = pixel;
144  curPixel.g = pixel;
145  curPixel.b = pixel;
146  } else if (ppmType == "P1") {
147  int pixel = (uint8)(ppmInput.readNumber() * maxColor);
148  curPixel.r = pixel;
