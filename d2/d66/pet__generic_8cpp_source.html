255  aura = SPELL_AURA_BANK_G + i;

327  AddGossipItemFor(player, GOSSIP_ICON_INTERACT_1, "Visit a bank.", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_BANK);

340  bool OnGossipSelect(Player* player, Creature* creature, uint32 /*uiSender*/, uint32 action)
341  {
342  CloseGossipMenuFor(player);
343  uint32 spellId = 0;
344  switch (action)
345  {
346  case GOSSIP_ACTION_TRADE:
347  creature->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_VENDOR);
348  player->GetSession()->SendListInventory(creature->GetGUID());
349  spellId = player->GetTeamId(true) ? SPELL_AURA_SHOP_G : SPELL_AURA_SHOP_S;
350  creature->AI()->DoAction(ARGENT_PONY_STATE_VENDOR);
351  break;
352  case GOSSIP_ACTION_BANK:
353  creature->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_BANKER);
354  player->GetSession()->SendShowBank(player->GetGUID());
355  spellId = player->GetTeamId(true) ? SPELL_AURA_BANK_G : SPELL_AURA_BANK_S;
356  creature->AI()->DoAction(ARGENT_PONY_STATE_BANK);
357  break;
358  case GOSSIP_ACTION_MAILBOX:
359  {
360  creature->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP | UNIT_NPC_FLAG_MAILBOX);
361  player->GetSession()->SendShowMailBox(creature->GetGUID());
362  spellId = player->GetTeamId(true) ? SPELL_AURA_POSTMAN_G : SPELL_AURA_POSTMAN_S;
363  creature->AI()->DoAction(ARGENT_PONY_STATE_MAILBOX);
364  break;
365  }
366  default:
367  if (action > 60000)
368  {
369  creature->AI()->DoAction(action);
370  creature->CastSpell(creature, action, true);
371  }
372  return true;
373  }
374 
375  if (spellId && !creature->HasAura(spellId))
376  {
377  creature->CastSpell(creature, spellId, true);
378  player->AddSpellCooldown(spellId, 0, 3 * MINUTE * IN_MILLISECONDS);
379  player->AddSpellCooldown(player->GetTeamId(true) ? SPELL_AURA_TIRED_G : SPELL_AURA_TIRED_S, 0, 3 * MINUTE * IN_MILLISECONDS + 4 * HOUR * IN_MILLISECONDS);
380  creature->DespawnOrUnsummon(3 * MINUTE * IN_MILLISECONDS);
381  }
382  return true;
383  }
