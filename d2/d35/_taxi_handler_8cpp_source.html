164 void WorldSession::HandleActivateTaxiExpressOpcode (WorldPacket& recvData)
165 {
166  LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXIEXPRESS");
167 
168  ObjectGuid guid;
169  uint32 node_count;
170 
171  recvData >> guid >> node_count;
172 
173  Creature* npc = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_FLIGHTMASTER);
174  if (!npc)
175  {
176  LOG_DEBUG("network", "WORLD: HandleActivateTaxiExpressOpcode - Unit (%s) not found or you can't interact with it.", guid.ToString().c_str());
177  return;
178  }
179  std::vector<uint32> nodes;
180 
181  for (uint32 i = 0; i < node_count; ++i)
182  {
183  uint32 node;
184  recvData >> node;
185 
186  if (!GetPlayer()->m_taxi.IsTaximaskNodeKnown(node) && !GetPlayer()->isTaxiCheater())
187  {
188  SendActivateTaxiReply(ERR_TAXINOTVISITED);
189  recvData.rfinish();
190  return;
191  }
192 
193  nodes.push_back(node);
194  }
195 
196  if (nodes.empty())
197  return;
198 
199  LOG_DEBUG("network", "WORLD: Received CMSG_ACTIVATETAXIEXPRESS from %d to %d", nodes.front(), nodes.back());
200 
201  GetPlayer()->ActivateTaxiPathTo(nodes, npc, 0);
202 }
