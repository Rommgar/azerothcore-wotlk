50 bool RotateMovementGenerator::Update(Unit* owner, uint32 diff)
51 {
52  float angle = owner->GetOrientation();
53  if (m_direction == ROTATE_DIRECTION_LEFT)
54  {
55  angle += (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
56  while (angle >= static_cast<float>(M_PI * 2)) angle -= static_cast<float>(M_PI * 2);
57  }
58  else
59  {
60  angle -= (float)diff * static_cast<float>(M_PI * 2) / m_maxDuration;
61  while (angle < 0) angle += static_cast<float>(M_PI * 2);
62  }
63 
64  owner->SetFacingTo(angle);
65 
66  if (m_duration > diff)
67  m_duration -= diff;
68  else
69  return false;
70 
71  return true;
72 }
