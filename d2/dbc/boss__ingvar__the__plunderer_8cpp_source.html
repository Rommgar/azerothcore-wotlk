125  void Reset() override

142  void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
143  {
144  if (me->GetDisplayId() == DISPLAYID_DEFAULT && damage >= me->GetHealth())
145  {
146  damage = 0;
147  me->InterruptNonMeleeSpells(true);
148  me->RemoveAllAuras();
149  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
150  me->SetControlled(false, UNIT_STATE_ROOT);
151  me->DisableRotate(false);
152  me->GetMotionMaster()->MovementExpired();
153  me->GetMotionMaster()->MoveIdle();
154  me->StopMoving();
155  FeignDeath(true);
156  events.Reset();
157  events.RescheduleEvent(EVENT_START_RESURRECTION, 1000);
158  events.RescheduleEvent(EVENT_YELL_DEAD_1, 0);
159  }
160  }

242  void UpdateAI(uint32 diff) override
243  {
244  if (!UpdateVictim())
245  return;
246 
247  events.Update(diff);
248 
249  if( me->HasUnitState(UNIT_STATE_CASTING) )
250  return;
251 
252  switch( events.ExecuteEvent() )
253  {
254  case 0:
255  break;
256  case EVENT_YELL_DEAD_1:
257  Talk(YELL_DEAD_1);
258  break;
259  case EVENT_START_RESURRECTION:
260  me->CastSpell(me, SPELL_SUMMON_VALKYR, true);
261  events.RescheduleEvent(EVENT_VALKYR_BEAM, 7000);
262  events.RescheduleEvent(EVENT_VALKYR_MOVE, 1);
263  events.RescheduleEvent(EVENT_ANNHYLDE_YELL, 3000);
264  break;
265  case EVENT_VALKYR_MOVE:
266  if( Creature* s = ObjectAccessor::GetCreature(*me, ValkyrGUID) )
267  s->GetMotionMaster()->MovePoint(1, s->GetPositionX(), s->GetPositionY(), s->GetPositionZ() - 15.0f);
268  break;
269  case EVENT_ANNHYLDE_YELL:
270  if( Creature* s = ObjectAccessor::GetCreature(*me, ValkyrGUID) )
271  s->AI()->Talk(YELL_ANHYLDE_2);
272  break;
273  case EVENT_VALKYR_BEAM:
274  me->RemoveAura(SPELL_SUMMON_VALKYR);
275  if( Creature* c = ObjectAccessor::GetCreature(*me, ValkyrGUID) )
276  c->CastSpell(me, SPELL_RESURRECTION_BEAM, false);
277  events.RescheduleEvent(EVENT_RESURRECTION_BALL, 4000);
278  break;
279  case EVENT_RESURRECTION_BALL:
280  me->CastSpell(me, SPELL_RESURRECTION_BALL, true);
281  events.RescheduleEvent(EVENT_RESURRECTION_HEAL, 4000);
282  break;
283  case EVENT_RESURRECTION_HEAL:
284  me->RemoveAura(SPELL_RESURRECTION_BALL);
285  me->CastSpell(me, SPELL_RESURRECTION_HEAL, true);
286  FeignDeath(false);
287  events.RescheduleEvent(EVENT_MORPH_TO_UNDEAD, 3000);
288  break;
289  case EVENT_MORPH_TO_UNDEAD:
290  me->CastSpell(me, SPELL_INGVAR_TRANSFORM, true);
291  events.RescheduleEvent(EVENT_START_PHASE_2, 1000);
292  break;
293  case EVENT_START_PHASE_2:
294  if( Creature* c = ObjectAccessor::GetCreature(*me, ValkyrGUID) )
295  {
296  c->DespawnOrUnsummon();
297  summons.DespawnAll();
298  }
299  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE);
300  AttackStart(me->GetVictim());
301  me->GetMotionMaster()->MoveChase(me->GetVictim());
302  Talk(YELL_AGGRO_2);
303 
304  // schedule Phase 2 abilities
305  events.RescheduleEvent(EVENT_SPELL_ROAR, 15000);
306  events.RescheduleEvent(EVENT_SPELL_CLEAVE_OR_WOE_STRIKE, 2000);
307  events.RescheduleEvent(EVENT_SPELL_SMASH, 5000);
308  events.RescheduleEvent(EVENT_SPELL_ENRAGE_OR_SHADOW_AXE, 10000);
309 
310  break;
311 
312  // ABILITIES HERE:
313  case EVENT_UNROOT:
314  me->SetControlled(false, UNIT_STATE_ROOT);
315  me->DisableRotate(false);
316  break;
317  case EVENT_SPELL_ROAR:
318  Talk(EMOTE_ROAR);
319 
320  me->_AddCreatureSpellCooldown(SPELL_STAGGERING_ROAR, 0, 0);
321  me->_AddCreatureSpellCooldown(SPELL_DREADFUL_ROAR, 0, 0);
322 
323  if (me->GetDisplayId() == DISPLAYID_DEFAULT)
324  me->CastSpell((Unit*)nullptr, SPELL_STAGGERING_ROAR, false);
325  else
326  me->CastSpell((Unit*)nullptr, SPELL_DREADFUL_ROAR, false);
327  events.RepeatEvent(urand(15000, 20000));
328  break;
329  case EVENT_SPELL_CLEAVE_OR_WOE_STRIKE:
330  if( me->GetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID) == 0 )
331  {
332  events.RepeatEvent(3000);
333  break;
334  }
335  if (me->GetDisplayId() == DISPLAYID_DEFAULT)
336  me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
337  else
338  me->CastSpell(me->GetVictim(), SPELL_WOE_STRIKE, false);
339  events.RepeatEvent(urand(0, 4000) + 3000);
340  break;
341  case EVENT_SPELL_SMASH:
342  if( me->GetUInt32Value(UNIT_VIRTUAL_ITEM_SLOT_ID) == 0 )
343  {
344  events.RepeatEvent(3000);
345  break;
346  }
347  me->SetControlled(true, UNIT_STATE_ROOT);
348  me->DisableRotate(true);
349  me->SendMovementFlagUpdate();
350  if (me->GetDisplayId() == DISPLAYID_DEFAULT)
351  me->CastSpell((Unit*)nullptr, SPELL_SMASH, false);
352  else
353  me->CastSpell((Unit*)nullptr, SPELL_DARK_SMASH, false);
354  events.RepeatEvent(urand(9000, 11000));
355  events.RescheduleEvent(EVENT_UNROOT, 3750);
356  break;
357  case EVENT_SPELL_ENRAGE_OR_SHADOW_AXE:
358  if (me->GetDisplayId() == DISPLAYID_DEFAULT)
359  {
360  me->CastSpell(me, SPELL_ENRAGE, false);
361  events.RepeatEvent(10000);
362  }
363  else
364  {
365  me->CastSpell((Unit*)nullptr, SPELL_SHADOW_AXE, true);
366  SetEquipmentSlots(false, EQUIP_UNEQUIP, EQUIP_NO_CHANGE, EQUIP_NO_CHANGE);
367  events.RepeatEvent(35000);
368  events.RescheduleEvent(EVENT_AXE_RETURN, 10000);
369  }
370  break;
371  case EVENT_AXE_RETURN:
372  if (Creature* c = ObjectAccessor::GetCreature(*me, ThrowGUID))
373  c->GetMotionMaster()->MoveCharge(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ());
374  events.RescheduleEvent(EVENT_AXE_PICKUP, 1500);
375  break;
376  case EVENT_AXE_PICKUP:
377  if (Creature* c = ObjectAccessor::GetCreature(*me, ThrowGUID))
378  {
379  c->DestroyForNearbyPlayers();
380  c->DespawnOrUnsummon();
381  summons.DespawnAll();
382  }
383  ThrowGUID.Clear();
384  SetEquipmentSlots(true);
385  break;
386  }
387 
388  if (!me->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_NOT_SELECTABLE))
389  DoMeleeAttackIfReady();
390  }
391  };
