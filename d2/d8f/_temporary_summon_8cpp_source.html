65 void TempSummon::Update(uint32 diff)
66 {
67  Creature::Update(diff);
68 
69  if (m_deathState == DEAD)
70  {
71  UnSummon();
72  return;
73  }
74  switch (m_type)
75  {
76  case TEMPSUMMON_MANUAL_DESPAWN:
77  case TEMPSUMMON_DESPAWNED:
78  break;
79  case TEMPSUMMON_TIMED_DESPAWN:
80  {
81  if (m_timer <= diff)
82  {
83  UnSummon();
84  return;
85  }
86 
87  m_timer -= diff;
88  break;
89  }
90  case TEMPSUMMON_TIMED_DESPAWN_OUT_OF_COMBAT:
91  {
92  if (!IsInCombat())
93  {
94  if (m_timer <= diff)
95  {
96  UnSummon();
97  return;
98  }
99 
100  m_timer -= diff;
101  }
102  else if (m_timer != m_lifetime)
103  m_timer = m_lifetime;
104 
105  break;
106  }
107 
108  case TEMPSUMMON_CORPSE_TIMED_DESPAWN:
109  {
110  if (m_deathState == CORPSE)
111  {
112  if (m_timer <= diff)
113  {
114  UnSummon();
115  return;
116  }
117 
118  m_timer -= diff;
119  }
120  break;
121  }
122  case TEMPSUMMON_CORPSE_DESPAWN:
123  {
124  // if m_deathState is DEAD, CORPSE was skipped
125  if (m_deathState == CORPSE)
126  {
127  UnSummon();
128  return;
129  }
130 
131  break;
132  }
133  case TEMPSUMMON_DEAD_DESPAWN:
134  {
135  break;
136  }
137  case TEMPSUMMON_TIMED_OR_CORPSE_DESPAWN:
138  {
139  // if m_deathState is DEAD, CORPSE was skipped
140  if (m_deathState == CORPSE)
141  {
142  UnSummon();
143  return;
144  }
145 
146  if (!IsInCombat())
147  {
148  if (m_timer <= diff)
149  {
150  UnSummon();
151  return;
152  }
153  else
154  m_timer -= diff;
155  }
156  else if (m_timer != m_lifetime)
157  m_timer = m_lifetime;
158  break;
159  }
160  case TEMPSUMMON_TIMED_OR_DEAD_DESPAWN:
161  {
162  if (!IsInCombat() && IsAlive())
163  {
164  if (m_timer <= diff)
165  {
166  UnSummon();
167  return;
168  }
169  else
170  m_timer -= diff;
171  }
172  else if (m_timer != m_lifetime)
173  m_timer = m_lifetime;
174  break;
175  }
176  default:
177  UnSummon();
178  LOG_ERROR("entities.unit", "Temporary summoned creature (entry: %u) have unknown type %u of ", GetEntry(), m_type);
179  break;
180  }
181 }

183 void TempSummon::InitStats(uint32 duration)
184 {
185  ASSERT(!IsPet());
186 
187  Unit* owner = GetSummonerUnit();
188  if (owner)
189  if (Player* player = owner->ToPlayer())
190  sScriptMgr->OnBeforeTempSummonInitStats(player, this, duration);
191 
192  m_timer = duration;
193  m_lifetime = duration;
194 
195  if (m_type == TEMPSUMMON_MANUAL_DESPAWN)
196  m_type = (duration == 0) ? TEMPSUMMON_DEAD_DESPAWN : TEMPSUMMON_TIMED_DESPAWN;
197 
198  if (owner)
199  {
200  if (IsTrigger() && m_spells[0])
201  {
202  SetFaction(owner->GetFaction());
203  SetLevel(owner->getLevel());
204  if (owner->GetTypeId() == TYPEID_PLAYER)
205  m_ControlledByPlayer = true;
206  }
207 
208  if (owner->GetTypeId() == TYPEID_PLAYER)
209  m_CreatedByPlayer = true;
210  }
211 
212  if (!m_Properties)
213  return;
214 
215  if (owner)
216  {
217  if (uint32 slot = m_Properties->Slot)
218  {
219  if (owner->m_SummonSlot[slot] && owner->m_SummonSlot[slot] != GetGUID())
220  {
221  Creature* oldSummon = GetMap()->GetCreature(owner->m_SummonSlot[slot]);
222  if (oldSummon && oldSummon->IsSummon())
223  oldSummon->ToTempSummon()->UnSummon();
224  }
225  owner->m_SummonSlot[slot] = GetGUID();
226  }
227  }
228 
229  if (m_Properties->Faction)
230  SetFaction(m_Properties->Faction);
231  else if (IsVehicle() && owner) // properties should be vehicle
232  SetFaction(owner->GetFaction());
233 }

254 void TempSummon::UnSummon(uint32 msTime)
255 {
256  if (msTime)
257  {
258  ForcedUnsummonDelayEvent* pEvent = new ForcedUnsummonDelayEvent(*this);
259 
260  m_Events.AddEvent(pEvent, m_Events.CalculateTime(msTime));
261  return;
262  }
263 
264  // Dont allow to call this function twice (possible)
265  if (m_type == TEMPSUMMON_DESPAWNED)
266  return;
267  SetTempSummonType(TEMPSUMMON_DESPAWNED);
268 
269  //ASSERT(!IsPet());
270  if (IsPet())
271  {
272  ((Pet*)this)->Remove(PET_SAVE_NOT_IN_SLOT);
273  ASSERT(!IsInWorld());
274  return;
275  }
276 
277  Unit* owner = GetSummonerUnit();
278  if (owner && owner->GetTypeId() == TYPEID_UNIT && owner->ToCreature()->IsAIEnabled)
279  owner->ToCreature()->AI()->SummonedCreatureDespawn(this);
280 
281  AddObjectToRemoveList();
282 }
