30 void Graveyard::LoadGraveyardFromDB()
31 {
32  uint32 oldMSTime = getMSTime();
33 
34  _graveyardStore.clear();
35 
36  QueryResult result = WorldDatabase.Query("SELECT ID, Map, x, y, z, Comment FROM game_graveyard");
37  if (!result)
38  {
39  LOG_INFO("server.loading", ">> Loaded 0 graveyard. Table `game_graveyard` is empty!");
40  LOG_INFO("server.loading", " ");
41  return;
42  }
43 
44  int32 Count = 0;
45 
46  do
47  {
48  Field* fields = result->Fetch();
49  uint32 ID = fields[0].GetUInt32();
50 
51  GraveyardStruct Graveyard;
52 
53  Graveyard.Map = fields[1].GetUInt32();
54  Graveyard.x = fields[2].GetFloat();
55  Graveyard.y = fields[3].GetFloat();
56  Graveyard.z = fields[4].GetFloat();
57  Graveyard.name = fields[5].GetString();
58 
59  if (!Utf8toWStr(Graveyard.name, Graveyard.wnameLow))
60  {
61  LOG_ERROR("sql.sql", "Wrong UTF8 name for id %u in `game_graveyard` table, ignoring.", ID);
62  continue;
63  }
64 
65  wstrToLower(Graveyard.wnameLow);
66 
67  _graveyardStore[ID] = Graveyard;
68 
69  ++Count;
70  } while (result->NextRow());
71 
72  LOG_INFO("server.loading", ">> Loaded %i graveyard in %u ms", Count, GetMSTimeDiffToNow(oldMSTime));
73  LOG_INFO("server.loading", " ");
74 }

96 GraveyardStruct const* Graveyard::GetClosestGraveyard(Player* player, TeamId teamId, bool nearCorpse)
97 {
98  WorldLocation loc = player->GetWorldLocation();
99 
100  if (nearCorpse)
101  {
102  loc = player->GetCorpseLocation();
103  }
104 
105  uint32 mapId = loc.GetMapId();
106  float x = loc.GetPositionX();
107  float y = loc.GetPositionY();
108  float z = loc.GetPositionZ();
109 
110  uint32 zoneId = 0;
111  uint32 areaId = 0;
112  player->GetZoneAndAreaId(zoneId, areaId);
113 
114  if (!zoneId && !areaId)
115  {
116  if (z > -500)
117  {
118  LOG_ERROR("sql.sql", "GetClosestGraveyard: unable to find zoneId and areaId for map %u coords (%f, %f, %f)", mapId, x, y, z);
119  return GetDefaultGraveyard(teamId);
120  }
121  }
122 
123  // Simulate std. algorithm:
124  // found some graveyard associated to (ghost_zone, ghost_map)
125  //
126  // if mapId == graveyard.mapId (ghost in plain zone or city or battleground) and search graveyard at same map
127  // then check faction
128  // if mapId != graveyard.mapId (ghost in instance) and search any graveyard associated
129  // then check faction
130 
131  // Fetch the graveyards linked to the areaId first, presumably the closer ones.
132  GraveyardMapBounds range = GraveyardStore.equal_range(areaId);
133 
134  // No graveyards linked to the area, search zone.
135  if (range.first == range.second)
136  {
137  range = GraveyardStore.equal_range(zoneId);
138  }
139  else // Found a graveyard linked to the area, check if it's a valid one.
140  {
141  GraveyardData const& graveyardLink = range.first->second;
142 
143  if (!graveyardLink.IsNeutralOrFriendlyToTeam(teamId))
144  {
145  // Not a friendly or neutral graveyard, search zone.
146  range = GraveyardStore.equal_range(zoneId);
147  }
148  }
149 
150  MapEntry const* map = sMapStore.LookupEntry(mapId);
151 
152  // not need to check validity of map object; MapId _MUST_ be valid here
153  if (range.first == range.second && !map->IsBattlegroundOrArena())
154  {
155  LOG_ERROR("sql.sql", "Table `graveyard_zone` incomplete: Zone %u Team %u does not have a linked graveyard.", zoneId, teamId);
156  return GetDefaultGraveyard(teamId);
157  }
158 
159  // at corpse map
160  bool foundNear = false;
161  float distNear = 10000;
162  GraveyardStruct const* entryNear = nullptr;
163 
164  // at entrance map for corpse map
165  bool foundEntr = false;
166  float distEntr = 10000;
167  GraveyardStruct const* entryEntr = nullptr;
168 
169  // some where other
170  GraveyardStruct const* entryFar = nullptr;
171 
172  MapEntry const* mapEntry = sMapStore.LookupEntry(mapId);
173 
174  for (; range.first != range.second; ++range.first)
175  {
176  GraveyardData const& graveyardLink = range.first->second;
177  GraveyardStruct const* entry = sGraveyard->GetGraveyard(graveyardLink.safeLocId);
178  if (!entry)
179  {
180  LOG_ERROR("sql.sql", "Table `graveyard_zone` has record for not existing `game_graveyard` table %u, skipped.", graveyardLink.safeLocId);
181  continue;
182  }
183 
184  // Skip enemy faction graveyard.
185  if (!graveyardLink.IsNeutralOrFriendlyToTeam(teamId))
186  {
187  continue;
188  }
189 
190  // Skip Archerus graveyards if the player isn't a Death Knight.
191  enum DeathKnightGraveyards
192  {
193  GRAVEYARD_EBON_HOLD = 1369,
194  GRAVEYARD_ARCHERUS = 1405
195  };
196 
197  if (player->getClass() != CLASS_DEATH_KNIGHT && (graveyardLink.safeLocId == GRAVEYARD_EBON_HOLD || graveyardLink.safeLocId == GRAVEYARD_ARCHERUS))
198  {
199  continue;
200  }
201 
202  // find now nearest graveyard at other map
203  if (mapId != entry->Map)
204  {
205  // if find graveyard at different map from where entrance placed (or no entrance data), use any first
206  if (!mapEntry
207  || mapEntry->entrance_map < 0
208  || uint32(mapEntry->entrance_map) != entry->Map
209  || (mapEntry->entrance_x == 0 && mapEntry->entrance_y == 0))
210  {
211  // not have any corrdinates for check distance anyway
212  entryFar = entry;
213  continue;
214  }
215 
216  // at entrance map calculate distance (2D);
217  float dist2 = (entry->x - mapEntry->entrance_x) * (entry->x - mapEntry->entrance_x)
218  + (entry->y - mapEntry->entrance_y) * (entry->y - mapEntry->entrance_y);
219  if (foundEntr)
220  {
221  if (dist2 < distEntr)
222  {
223  distEntr = dist2;
224  entryEntr = entry;
225  }
226  }
227  else
228  {
229  foundEntr = true;
230  distEntr = dist2;
231  entryEntr = entry;
232  }
233  }
234  // find now nearest graveyard at same map
235  else
236  {
237  float dist2 = (entry->x - x) * (entry->x - x) + (entry->y - y) * (entry->y - y) + (entry->z - z) * (entry->z - z);
238  if (foundNear)
239  {
240  if (dist2 < distNear)
241  {
242  distNear = dist2;
243  entryNear = entry;
244  }
245  }
246  else
247  {
248  foundNear = true;
249  distNear = dist2;
250  entryNear = entry;
251  }
252  }
253  }
254 
255  if (entryNear)
256  return entryNear;
257 
258  if (entryEntr)
259  return entryEntr;
260 
261  return entryFar;
262 }

277 bool Graveyard::AddGraveyardLink(uint32 id, uint32 zoneId, TeamId teamId, bool persist /*= true*/)
278 {
279  if (FindGraveyardData(id, zoneId))
280  return false;
281 
282  // add link to loaded data
283  GraveyardData data;
284  data.safeLocId = id;
285  data.teamId = teamId;
286 
287  GraveyardStore.insert(WGGraveyardContainer::value_type(zoneId, data));
288 
289  // add link to DB
290  if (persist)
291  {
292  WorldDatabasePreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_INS_GRAVEYARD_ZONE);
293 
294  stmt->setUInt32(0, id);
295  stmt->setUInt32(1, zoneId);
296  // Xinef: DB Data compatibility...
297  stmt->setUInt16(2, uint16(teamId == TEAM_NEUTRAL ? 0 : (teamId == TEAM_ALLIANCE ? ALLIANCE : HORDE)));
298 
299  WorldDatabase.Execute(stmt);
300  }
301 
302  return true;
303 }

305 void Graveyard::RemoveGraveyardLink(uint32 id, uint32 zoneId, TeamId teamId, bool persist /*= false*/)
306 {
307  GraveyardMapBoundsNonConst range = GraveyardStore.equal_range(zoneId);
308  if (range.first == range.second)
309  {
310  LOG_ERROR("sql.sql", "Table `graveyard_zone` incomplete: Zone %u Team %u does not have a linked graveyard.", zoneId, teamId);
311  return;
312  }
313 
314  bool found = false;
315 
316  for (; range.first != range.second; ++range.first)
317  {
318  GraveyardData& data = range.first->second;
319 
320  // skip not matching safezone id
321  if (data.safeLocId != id)
322  continue;
323 
324  // skip enemy faction graveyard at same map (normal area, city, or battleground)
325  // team == 0 case can be at call from .neargrave
326  if (data.teamId != TEAM_NEUTRAL && teamId != TEAM_NEUTRAL && data.teamId != teamId)
327  continue;
328 
329  found = true;
330  break;
331  }
332 
333  // no match, return
334  if (!found)
335  return;
336 
337  // remove from links
338  GraveyardStore.erase(range.first);
339 
340  // remove link from DB
341  if (persist)
342  {
343  WorldDatabasePreparedStatement* stmt = WorldDatabase.GetPreparedStatement(WORLD_DEL_GRAVEYARD_ZONE);
344 
345  stmt->setUInt32(0, id);
346  stmt->setUInt32(1, zoneId);
347  // Xinef: DB Data compatibility...
348  stmt->setUInt16(2, uint16(teamId == TEAM_NEUTRAL ? 0 : (teamId == TEAM_ALLIANCE ? ALLIANCE : HORDE)));
349 
350  WorldDatabase.Execute(stmt);
351  }
352 }

354 void Graveyard::LoadGraveyardZones()
355 {
356  uint32 oldMSTime = getMSTime();
357 
358  GraveyardStore.clear(); // need for reload case
359 
360  // 0 1 2
361  QueryResult result = WorldDatabase.Query("SELECT ID, GhostZone, Faction FROM graveyard_zone");
362 
363  if (!result)
364  {
365  LOG_INFO("server.loading", ">> Loaded 0 graveyard-zone links. DB table `graveyard_zone` is empty.");
366  LOG_INFO("server.loading", " ");
367  return;
368  }
369 
370  uint32 count = 0;
371 
372  do
373  {
374  ++count;
375 
376  Field* fields = result->Fetch();
377 
378  uint32 safeLocId = fields[0].GetUInt32();
379  uint32 zoneId = fields[1].GetUInt32();
380  uint32 team = fields[2].GetUInt16();
381  TeamId teamId = team == 0 ? TEAM_NEUTRAL : (team == ALLIANCE ? TEAM_ALLIANCE : TEAM_HORDE);
382 
383  GraveyardStruct const* entry = sGraveyard->GetGraveyard(safeLocId);
384  if (!entry)
385  {
386  LOG_ERROR("sql.sql", "Table `graveyard_zone` has a record for not existing `game_graveyard` table %u, skipped.", safeLocId);
387  continue;
388  }
389 
390  AreaTableEntry const* areaEntry = sAreaTableStore.LookupEntry(zoneId);
391  if (!areaEntry)
392  {
393  LOG_ERROR("sql.sql", "Table `graveyard_zone` has a record for not existing zone id (%u), skipped.", zoneId);
394  continue;
395  }
396 
397  if (team != 0 && team != HORDE && team != ALLIANCE)
398  {
399  LOG_ERROR("sql.sql", "Table `graveyard_zone` has a record for non player faction (%u), skipped.", team);
400  continue;
401  }
402 
403  if (!AddGraveyardLink(safeLocId, zoneId, teamId, false))
404  LOG_ERROR("sql.sql", "Table `graveyard_zone` has a duplicate record for Graveyard (ID: %u) and Zone (ID: %u), skipped.", safeLocId, zoneId);
405  } while (result->NextRow());
406 
407  LOG_INFO("server.loading", ">> Loaded %u graveyard-zone links in %u ms", count, GetMSTimeDiffToNow(oldMSTime));
408  LOG_INFO("server.loading", " ");
409 }
