195  OUT_LOAD_INST_DATA(str);

197  char dataHead1, dataHead2;
198 
199  std::istringstream loadStream(str);
200  loadStream >> dataHead1 >> dataHead2;
201 
202  if (dataHead1 == 'S' && dataHead2 == 'C')
203  {
204  for (uint8 i = 0; i < MAX_ENCOUNTERS; ++i)
205  {
206  uint32 tmpState;
207  loadStream >> tmpState;
208  if (tmpState == IN_PROGRESS || tmpState > SPECIAL)
209  tmpState = NOT_STARTED;
210  SetBossState(i, EncounterState(tmpState));
211  }
212  }
213  else
214  OUT_LOAD_INST_DATA_FAIL;

216  OUT_LOAD_INST_DATA_COMPLETE;
