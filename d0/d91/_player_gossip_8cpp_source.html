28 void Player::PrepareGossipMenu(WorldObject* source, uint32 menuId /*= 0*/, bool showQuests /*= false*/)
29 {
30  PlayerMenu* menu = PlayerTalkClass;
31  menu->ClearMenus();
32 
33  menu->GetGossipMenu().SetMenuId(menuId);
34 
35  GossipMenuItemsMapBounds menuItemBounds = sObjectMgr->GetGossipMenuItemsMapBounds(menuId);
36 
37  // if default menuId and no menu options exist for this, use options from default options
38  if (menuItemBounds.first == menuItemBounds.second && menuId == GetDefaultGossipMenuForSource(source))
39  menuItemBounds = sObjectMgr->GetGossipMenuItemsMapBounds(0);
40 
41  uint32 npcflags = 0;
42 
43  if (source->GetTypeId() == TYPEID_UNIT)
44  {
45  npcflags = source->GetUInt32Value(UNIT_NPC_FLAGS);
46  if (showQuests && npcflags & UNIT_NPC_FLAG_QUESTGIVER)
47  PrepareQuestMenu(source->GetGUID());
48  }
49  else if (source->GetTypeId() == TYPEID_GAMEOBJECT)
50  if (showQuests && source->ToGameObject()->GetGoType() == GAMEOBJECT_TYPE_QUESTGIVER)
51  PrepareQuestMenu(source->GetGUID());
52 
53  for (GossipMenuItemsContainer::const_iterator itr = menuItemBounds.first; itr != menuItemBounds.second; ++itr)
54  {
55  bool canTalk = true;
56  if (!sConditionMgr->IsObjectMeetToConditions(this, source, itr->second.Conditions))
57  continue;
58 
59  if (Creature* creature = source->ToCreature())
60  {
61  if (!(itr->second.OptionNpcFlag & npcflags))
62  continue;
63 
64  switch (itr->second.OptionType)
65  {
66  case GOSSIP_OPTION_ARMORER:
67  canTalk = false; // added in special mode
68  break;
69  case GOSSIP_OPTION_SPIRITHEALER:
70  if (!isDead())
71  canTalk = false;
72  break;
73  case GOSSIP_OPTION_VENDOR:
74  {
75  if (!creature->isVendorWithIconSpeak())
76  {
77  VendorItemData const* vendorItems = itr->second.ActionMenuID ? sObjectMgr->GetNpcVendorItemList(itr->second.ActionMenuID) : creature->GetVendorItems();
78  if (!vendorItems || vendorItems->Empty())
79  {
80  LOG_ERROR("sql.sql", "Creature %s have UNIT_NPC_FLAG_VENDOR but have empty trading item list.", creature->GetGUID().ToString().c_str());
81  canTalk = false;
82  }
83  break;
84  }
85  break;
86  }
87  case GOSSIP_OPTION_LEARNDUALSPEC:
88  case GOSSIP_OPTION_DUALSPEC_INFO:
89  if (!(GetSpecsCount() == 1 && creature->isCanTrainingAndResetTalentsOf(this) && !(getLevel() < sWorld->getIntConfig(CONFIG_MIN_DUALSPEC_LEVEL))))
90  canTalk = false;
91  break;
92  case GOSSIP_OPTION_UNLEARNTALENTS:
93  if (!creature->isCanTrainingAndResetTalentsOf(this))
94  canTalk = false;
95  break;
96  case GOSSIP_OPTION_UNLEARNPETTALENTS:
97  if (!GetPet() || GetPet()->getPetType() != HUNTER_PET || GetPet()->m_spells.size() <= 1 || creature->GetCreatureTemplate()->trainer_type != TRAINER_TYPE_PETS || creature->GetCreatureTemplate()->trainer_class != CLASS_HUNTER)
98  canTalk = false;
99  break;
100  case GOSSIP_OPTION_TAXIVENDOR:
101  if (GetSession()->SendLearnNewTaxiNode(creature))
102  return;
103  break;
104  case GOSSIP_OPTION_BATTLEFIELD:
105  if (!creature->isCanInteractWithBattleMaster(this, false))
106  canTalk = false;
107  break;
108  case GOSSIP_OPTION_STABLEPET:
109  if (getClass() != CLASS_HUNTER)
110  canTalk = false;
111  break;
112  case GOSSIP_OPTION_QUESTGIVER:
113  canTalk = false;
114  break;
115  case GOSSIP_OPTION_TRAINER:
116  if (!creature->IsValidTrainerForPlayer(this))
117  {
118  canTalk = false;
119  }
120  break;
121  case GOSSIP_OPTION_GOSSIP:
122  if (creature->isVendorWithIconSpeak())
123  {
124  VendorItemData const* vendorItems = creature->GetVendorItems();
125  if (!vendorItems || vendorItems->Empty())
126  {
127  canTalk = false;
128  }
129  }
130  break;
131  case GOSSIP_OPTION_SPIRITGUIDE:
132  case GOSSIP_OPTION_INNKEEPER:
133  case GOSSIP_OPTION_BANKER:
134  case GOSSIP_OPTION_PETITIONER:
135  case GOSSIP_OPTION_TABARDDESIGNER:
136  case GOSSIP_OPTION_AUCTIONEER:
137  break; // no checks
138  case GOSSIP_OPTION_OUTDOORPVP:
139  if (!sOutdoorPvPMgr->CanTalkTo(this, creature, itr->second))
140  canTalk = false;
141  break;
142  default:
143  LOG_ERROR("sql.sql", "Creature entry %u has unknown gossip option %u for menu %u", creature->GetEntry(), itr->second.OptionType, itr->second.MenuID);
144  canTalk = false;
145  break;
146  }
147  }
148  else if (GameObject* go = source->ToGameObject())
149  {
150  switch (itr->second.OptionType)
151  {
152  case GOSSIP_OPTION_GOSSIP:
153  if (go->GetGoType() != GAMEOBJECT_TYPE_QUESTGIVER && go->GetGoType() != GAMEOBJECT_TYPE_GOOBER)
154  canTalk = false;
155  break;
156  default:
157  canTalk = false;
158  break;
159  }
160  }
161 
162  if (canTalk)
163  {
164  // search in broadcast_text and broadcast_text_locale
165  std::string strOptionText, strBoxText;
166  BroadcastText const* optionBroadcastText = sObjectMgr->GetBroadcastText(itr->second.OptionBroadcastTextID);
167  BroadcastText const* boxBroadcastText = sObjectMgr->GetBroadcastText(itr->second.BoxBroadcastTextID);
168  LocaleConstant locale = GetSession()->GetSessionDbLocaleIndex();
169 
170  if (optionBroadcastText)
171  ObjectMgr::GetLocaleString(getGender() == GENDER_MALE ? optionBroadcastText->MaleText : optionBroadcastText->FemaleText, locale, strOptionText);
172  else
173  strOptionText = itr->second.OptionText;
174 
175  if (boxBroadcastText)
176  ObjectMgr::GetLocaleString(getGender() == GENDER_MALE ? boxBroadcastText->MaleText : boxBroadcastText->FemaleText, locale, strBoxText);
177  else
178  strBoxText = itr->second.BoxText;
179 
180  // if the language is not default and the texts weren't found, maybe they're in gossip_menu_option_locale table
181  if (locale != DEFAULT_LOCALE)
182  {
183  if (strOptionText.empty())
184  {
186  if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, itr->second.OptionID)))
187  ObjectMgr::GetLocaleString(gossipMenuLocale->OptionText, locale, strOptionText);
188  }
189 
190  if (strBoxText.empty())
191  {
193  if (GossipMenuItemsLocale const* gossipMenuLocale = sObjectMgr->GetGossipMenuItemsLocale(MAKE_PAIR32(menuId, itr->second.OptionID)))
194  ObjectMgr::GetLocaleString(gossipMenuLocale->BoxText, locale, strBoxText);
195  }
196  }
197 
198  menu->GetGossipMenu().AddMenuItem(itr->second.OptionID, itr->second.OptionIcon, strOptionText, 0, itr->second.OptionType, strBoxText, itr->second.BoxMoney, itr->second.BoxCoded);
199  menu->GetGossipMenu().AddGossipMenuItemData(itr->second.OptionID, itr->second.ActionMenuID, itr->second.ActionPoiID);
200  }
201  }
202 
203  if (sWorld->getIntConfig(CONFIG_INSTANT_TAXI) == 2 && npcflags & UNIT_NPC_FLAG_FLIGHTMASTER)
204  menu->GetGossipMenu().AddMenuItem(-1, GOSSIP_ICON_INTERACT_1, GetSession()->GetAcoreString(LANG_TOGGLE_INSTANT_FLIGHT), 0, GOSSIP_ACTION_TOGGLE_INSTANT_FLIGHT, "", 0, false); // instant flight toggle option
205 }

207 void Player::SendPreparedGossip(WorldObject* source)
208 {
209  if (!source)
210  return;
211 
212  if (source->GetTypeId() == TYPEID_UNIT)
213  {
214  // in case no gossip flag and quest menu not empty, open quest menu (client expect gossip menu with this flag)
215  if (!source->ToCreature()->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP) && !PlayerTalkClass->GetQuestMenu().Empty())
216  {
217  SendPreparedQuest(source->GetGUID());
218  return;
219  }
220  }
221  else if (source->GetTypeId() == TYPEID_GAMEOBJECT)
222  {
223  // probably need to find a better way here
224  if (!PlayerTalkClass->GetGossipMenu().GetMenuId() && !PlayerTalkClass->GetQuestMenu().Empty())
225  {
226  SendPreparedQuest(source->GetGUID());
227  return;
228  }
229  }
230 
231  // in case non empty gossip menu (that not included quests list size) show it
232  // (quest entries from quest menu will be included in list)
233 
234  uint32 textId = GetGossipTextId(source);
235 
236  if (uint32 menuId = PlayerTalkClass->GetGossipMenu().GetMenuId())
237  textId = GetGossipTextId(menuId, source);
238 
239  PlayerTalkClass->SendGossipMenu(textId, source->GetGUID());
240 }

242 void Player::OnGossipSelect(WorldObject* source, uint32 gossipListId, uint32 menuId)
243 {
244  GossipMenu& gossipMenu = PlayerTalkClass->GetGossipMenu();
245 
246  // if not same, then something funky is going on
247  if (menuId != gossipMenu.GetMenuId())
248  return;
249 
250  GossipMenuItem const* item = gossipMenu.GetItem(gossipListId);
251  if (!item)
252  return;
253 
254  uint32 gossipOptionId = item->OptionType;
255  ObjectGuid guid = source->GetGUID();
256 
257  if (sWorld->getIntConfig(CONFIG_INSTANT_TAXI) == 2 && source->GetTypeId() == TYPEID_UNIT)
258  {
259  if (gossipOptionId == GOSSIP_ACTION_TOGGLE_INSTANT_FLIGHT && source->GetUInt32Value(UNIT_NPC_FLAGS) & UNIT_NPC_FLAG_FLIGHTMASTER)
260  {
261  ToggleInstantFlight();
262 
263  if (m_isInstantFlightOn)
264  GetSession()->SendNotification(LANG_INSTANT_FLIGHT_ON);
265  else
266  GetSession()->SendNotification(LANG_INSTANT_FLIGHT_OFF);
267 
268  PlayerTalkClass->SendCloseGossip();
269  return;
270  }
271  }
272 
273  if (source->GetTypeId() == TYPEID_GAMEOBJECT)
274  {
275  if (gossipOptionId > GOSSIP_OPTION_QUESTGIVER)
276  {
277  LOG_ERROR("entities.player", "Player guid %s request invalid gossip option for GameObject entry %u", GetGUID().ToString().c_str(), source->GetEntry());
278  return;
279  }
280  }
281 
282  GossipMenuItemData const* menuItemData = gossipMenu.GetItemData(gossipListId);
283  if (!menuItemData)
284  return;
285 
286  int32 cost = int32(item->BoxMoney);
287  if (!HasEnoughMoney(cost))
288  {
289  SendBuyError(BUY_ERR_NOT_ENOUGHT_MONEY, 0, 0, 0);
290  PlayerTalkClass->SendCloseGossip();
291  return;
292  }
293 
294  switch (gossipOptionId)
295  {
296  case GOSSIP_OPTION_GOSSIP:
297  case GOSSIP_OPTION_DUALSPEC_INFO:
298  {
299  if (menuItemData->GossipActionPoi)
300  PlayerTalkClass->SendPointOfInterest(menuItemData->GossipActionPoi);
301 
302  if (menuItemData->GossipActionMenuId)
303  {
304  PrepareGossipMenu(source, menuItemData->GossipActionMenuId);
305  SendPreparedGossip(source);
306  }
307 
308  break;
309  }
310  case GOSSIP_OPTION_OUTDOORPVP:
311  sOutdoorPvPMgr->HandleGossipOption(this, source->ToCreature(), gossipListId);
312  break;
313  case GOSSIP_OPTION_SPIRITHEALER:
314  if (isDead())
315  source->ToCreature()->CastSpell(source->ToCreature(), 17251, true, nullptr, nullptr, GetGUID());
316  break;
317  case GOSSIP_OPTION_QUESTGIVER:
318  PrepareQuestMenu(guid);
319  SendPreparedQuest(guid);
320  break;
321  case GOSSIP_OPTION_VENDOR:
322  case GOSSIP_OPTION_ARMORER:
323  GetSession()->SendListInventory(guid, menuItemData->GossipActionMenuId);
324  break;
325  case GOSSIP_OPTION_STABLEPET:
326  GetSession()->SendStablePet(guid);
327  break;
328  case GOSSIP_OPTION_TRAINER:
329  GetSession()->SendTrainerList(guid);
330  break;
331  case GOSSIP_OPTION_LEARNDUALSPEC:
332  if (GetSpecsCount() == 1 && getLevel() >= sWorld->getIntConfig(CONFIG_MIN_DUALSPEC_LEVEL))
333  {
334  // Cast spells that teach dual spec
335  // Both are also ImplicitTarget self and must be cast by player
336  CastSpell(this, 63680, true, nullptr, nullptr, GetGUID());
337  CastSpell(this, 63624, true, nullptr, nullptr, GetGUID());
338 
339  PrepareGossipMenu(source, menuItemData->GossipActionMenuId);
340  SendPreparedGossip(source);
341  }
342  break;
343  case GOSSIP_OPTION_UNLEARNTALENTS:
344  PlayerTalkClass->SendCloseGossip();
345  SendTalentWipeConfirm(guid);
346  break;
347  case GOSSIP_OPTION_UNLEARNPETTALENTS:
348  PlayerTalkClass->SendCloseGossip();
349  ResetPetTalents();
350  break;
351  case GOSSIP_OPTION_TAXIVENDOR:
352  GetSession()->SendTaxiMenu(source->ToCreature());
353  break;
354  case GOSSIP_OPTION_INNKEEPER:
355  PlayerTalkClass->SendCloseGossip();
356  SetBindPoint(guid);
357  break;
358  case GOSSIP_OPTION_BANKER:
359  GetSession()->SendShowBank(guid);
360  break;
361  case GOSSIP_OPTION_PETITIONER:
362  PlayerTalkClass->SendCloseGossip();
363  GetSession()->SendPetitionShowList(guid);
364  break;
365  case GOSSIP_OPTION_TABARDDESIGNER:
366  PlayerTalkClass->SendCloseGossip();
367  GetSession()->SendTabardVendorActivate(guid);
368  break;
369  case GOSSIP_OPTION_AUCTIONEER:
370  GetSession()->SendAuctionHello(guid, source->ToCreature());
371  break;
372  case GOSSIP_OPTION_SPIRITGUIDE:
373  PrepareGossipMenu(source);
374  SendPreparedGossip(source);
375  break;
376  case GOSSIP_OPTION_BATTLEFIELD:
377  {
378  BattlegroundTypeId bgTypeId = sBattlegroundMgr->GetBattleMasterBG(source->GetEntry());
379 
380  if (bgTypeId == BATTLEGROUND_TYPE_NONE)
381  {
382  LOG_ERROR("entities.player", "A user (%s) requested battlegroundlist from a npc who is no battlemaster", GetGUID().ToString().c_str());
383  return;
384  }
385 
386  GetSession()->SendBattleGroundList(guid, bgTypeId);
387  break;
388  }
389  }
390 
391  ModifyMoney(-cost);
392 }
