113  static bool HandleLearnAllMySpellsCommand(ChatHandler* handler)
114  {
115  ChrClassesEntry const* classEntry = sChrClassesStore.LookupEntry(handler->GetSession()->GetPlayer()->getClass());
116  if (!classEntry)
117  return true;
118  uint32 family = classEntry->spellfamily;
119 
120  for (uint32 i = 0; i < sSkillLineAbilityStore.GetNumRows(); ++i)
121  {
122  SkillLineAbilityEntry const* entry = sSkillLineAbilityStore.LookupEntry(i);
123  if (!entry)
124  continue;
125 
126  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(entry->Spell);
127  if (!spellInfo)
128  continue;
129 
130  // skip server-side/triggered spells
131  if (spellInfo->SpellLevel == 0)
132  continue;
133 
134  // skip wrong class/race skills
135  if (!handler->GetSession()->GetPlayer()->IsSpellFitByClassAndRace(spellInfo->Id))
136  continue;
137 
138  // skip other spell families
139  if (spellInfo->SpellFamilyName != family)
140  continue;
141 
142  // skip spells with first rank learned as talent (and all talents then also)
143  uint32 firstRank = sSpellMgr->GetFirstSpellInChain(spellInfo->Id);
144  if (GetTalentSpellCost(firstRank) > 0)
145  continue;
146 
147  // skip broken spells
148  if (!SpellMgr::IsSpellValid(spellInfo))
149  continue;
150 
151  handler->GetSession()->GetPlayer()->learnSpell(spellInfo->Id);
152  }
153 
154  handler->SendSysMessage(LANG_COMMAND_LEARN_CLASS_SPELLS);
155  return true;
156  }

158  static bool HandleLearnAllMyTalentsCommand(ChatHandler* handler)
159  {
160  Player* player = handler->GetSession()->GetPlayer();
161  uint32 classMask = player->getClassMask();
162 
163  for (uint32 i = 0; i < sTalentStore.GetNumRows(); ++i)
164  {
165  TalentEntry const* talentInfo = sTalentStore.LookupEntry(i);
166  if (!talentInfo)
167  continue;
168 
169  TalentTabEntry const* talentTabInfo = sTalentTabStore.LookupEntry(talentInfo->TalentTab);
170  if (!talentTabInfo)
171  continue;
172 
173  if ((classMask & talentTabInfo->ClassMask) == 0)
174  continue;
175 
176  // xinef: search highest talent rank
177  uint32 spellId = 0;
178  uint8 rankId = MAX_TALENT_RANK;
179  for (int8 rank = MAX_TALENT_RANK - 1; rank >= 0; --rank)
180  {
181  if (talentInfo->RankID[rank] != 0)
182  {
183  rankId = rank;
184  spellId = talentInfo->RankID[rank];
185  break;
186  }
187  }
188 
189  // xinef: some errors?
190  if (!spellId || rankId == MAX_TALENT_RANK)
191  continue;
192 
193  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
194  if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo))
195  continue;
196 
197  player->LearnTalent(talentInfo->TalentID, rankId);
198  }
199 
200  handler->SendSysMessage(LANG_COMMAND_LEARN_CLASS_TALENTS);
201  return true;

204  static bool HandleLearnAllMyPetTalentsCommand(ChatHandler* handler)
205  {
206  Player* player = handler->GetSession()->GetPlayer();
207 
208  Pet* pet = player->GetPet();
209  if (!pet)
210  {
211  handler->SendSysMessage(LANG_NO_PET_FOUND);
212  handler->SetSentErrorMessage(true);
213  return false;
214  }
215 
216  CreatureTemplate const* creatureInfo = pet->GetCreatureTemplate();
217  if (!creatureInfo)
218  {
219  handler->SendSysMessage(LANG_WRONG_PET_TYPE);
220  handler->SetSentErrorMessage(true);
221  return false;
222  }
223 
224  CreatureFamilyEntry const* petFamily = sCreatureFamilyStore.LookupEntry(creatureInfo->family);
225  if (!petFamily)
226  {
227  handler->SendSysMessage(LANG_WRONG_PET_TYPE);
228  handler->SetSentErrorMessage(true);
229  return false;
230  }
231 
232  if (petFamily->petTalentType < 0) // not hunter pet
233  {
234  handler->SendSysMessage(LANG_WRONG_PET_TYPE);
235  handler->SetSentErrorMessage(true);
236  return false;
237  }
238 
239  for (uint32 i = 0; i < sTalentStore.GetNumRows(); ++i)
240  {
241  TalentEntry const* talentInfo = sTalentStore.LookupEntry(i);
242  if (!talentInfo)
243  continue;
244 
245  TalentTabEntry const* talentTabInfo = sTalentTabStore.LookupEntry(talentInfo->TalentTab);
246  if (!talentTabInfo)
247  continue;
248 
249  // prevent learn talent for different family (cheating)
250  if (((1 << petFamily->petTalentType) & talentTabInfo->petTalentMask) == 0)
251  continue;
252 
253  // search highest talent rank
254  uint32 spellId = 0;
255 
256  for (int8 rank = MAX_TALENT_RANK - 1; rank >= 0; --rank)
257  {
258  if (talentInfo->RankID[rank] != 0)
259  {
260  spellId = talentInfo->RankID[rank];
261  break;
262  }
263  }
264 
265  if (!spellId) // ??? none spells in talent
266  continue;
267 
268  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellId);
269  if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo))
270  continue;
271 
272  // learn highest rank of talent and learn all non-talent spell ranks (recursive by tree)
273  pet->learnSpellHighRank(spellId);
274  }
275 
276  pet->SetFreeTalentPoints(0);
277 
278  handler->SendSysMessage(LANG_COMMAND_LEARN_PET_TALENTS);
279  return true;

327  static bool HandleLearnAllRecipesCommand(ChatHandler* handler, WTail namePart)
328  {
329  // Learns all recipes of specified profession and sets skill to max
330  // Example: .learn all_recipes enchanting
331 
332  Player* target = handler->getSelectedPlayer();
333  if (!target)
334  {
335  handler->SendSysMessage(LANG_PLAYER_NOT_FOUND);
336  return false;
337  }
338 
339  if (namePart.empty())
340  return false;
341 
342  // converting string that we try to find to lower case
343  wstrToLower(namePart);
344 
345  SkillLineEntry const* targetSkillInfo = nullptr;
346  char const* name = nullptr;
347  for (uint32 i = 1; i < sSkillLineStore.GetNumRows(); ++i)
348  {
349  SkillLineEntry const* skillInfo = sSkillLineStore.LookupEntry(i);
350  if (!skillInfo)
351  continue;
352 
353  if ((skillInfo->categoryId != SKILL_CATEGORY_PROFESSION &&
354  skillInfo->categoryId != SKILL_CATEGORY_SECONDARY) ||
355  !skillInfo->canLink) // only prof with recipes have set
356  continue;
357 
358  uint8 locale = 0;
359  for (; locale < TOTAL_LOCALES; ++locale)
360  {
361  name = skillInfo->name[locale];
362  if (!name || !*name)
363  continue;
364 
365  if (Utf8FitTo(name, namePart))
366  break;
367  }
368 
369  if (locale < TOTAL_LOCALES)
370  {
371  targetSkillInfo = skillInfo;
372  break;
373  }
374  }
375 
376  if (!(name && targetSkillInfo))
377  return false;
378 
379  HandleLearnSkillRecipesHelper(target, targetSkillInfo->id);
380 
381  uint16 maxLevel = target->GetPureMaxSkillValue(targetSkillInfo->id);
382  target->SetSkill(targetSkillInfo->id, target->GetSkillStep(targetSkillInfo->id), maxLevel, maxLevel);
383  handler->PSendSysMessage(LANG_COMMAND_LEARN_ALL_RECIPES, name);
384  return true;
385  }

387  static void HandleLearnSkillRecipesHelper(Player* player, uint32 skillId)
388  {
389  uint32 classmask = player->getClassMask();
390 
391  for (uint32 j = 0; j < sSkillLineAbilityStore.GetNumRows(); ++j)
392  {
393  SkillLineAbilityEntry const* skillLine = sSkillLineAbilityStore.LookupEntry(j);
394  if (!skillLine)
395  continue;
396 
397  // wrong skill
398  if (skillLine->SkillLine != skillId)
399  continue;
400 
401  // not high rank
402  if (skillLine->SupercededBySpell)
403  continue;
404 
405  // skip racial skills
406  if (skillLine->RaceMask != 0)
407  continue;
408 
409  // skip wrong class skills
410  if (skillLine->ClassMask && (skillLine->ClassMask & classmask) == 0)
411  continue;
412 
413  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(skillLine->Spell);
414  if (!spellInfo || !SpellMgr::IsSpellValid(spellInfo))
415  continue;
416 
417  player->learnSpell(skillLine->Spell);
418  }
419  }
