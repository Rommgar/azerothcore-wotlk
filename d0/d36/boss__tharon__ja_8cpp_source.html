135  void UpdateAI(uint32 diff) override
136  {
137  if (!UpdateVictim())
138  return;
139 
140  events.Update(diff);
141  if (me->HasUnitState(UNIT_STATE_CASTING))
142  return;
143 
144  switch (events.ExecuteEvent())
145  {
146  case EVENT_SPELL_CURSE_OF_LIFE:
147  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, true))
148  me->CastSpell(target, SPELL_CURSE_OF_LIFE, false);
149  events.ScheduleEvent(EVENT_SPELL_CURSE_OF_LIFE, 13000);
150  break;
151  case EVENT_SPELL_RAIN_OF_FIRE:
152  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 30.0f, true))
153  me->CastSpell(target, SPELL_RAIN_OF_FIRE, false);
154  events.ScheduleEvent(EVENT_SPELL_RAIN_OF_FIRE, 16000);
155  break;
156  case EVENT_SPELL_SHADOW_VOLLEY:
157  me->CastSpell(me, SPELL_SHADOW_VOLLEY, false);
158  events.ScheduleEvent(EVENT_SPELL_SHADOW_VOLLEY, 9000);
159  break;
160  case EVENT_SPELL_TURN_FLESH:
161  if (me->HealthBelowPct(50))
162  {
163  Talk(SAY_FLESH);
164  me->getThreatMgr().resetAllAggro();
165  me->CastSpell((Unit*)nullptr, SPELL_TURN_FLESH, false);
166 
167  events.Reset();
168  events.ScheduleEvent(EVENT_TURN_FLESH_REAL, 3000);
169  return;
170  }
171  events.ScheduleEvent(EVENT_SPELL_TURN_FLESH, 1000);
172  break;
173  case EVENT_TURN_FLESH_REAL:
174  me->CastSpell(me, SPELL_DUMMY, true);
175 
176  me->GetMotionMaster()->MoveChase(me->GetVictim());
177  events.ScheduleEvent(EVENT_SPELL_EYE_BEAM, 11000);
178  events.ScheduleEvent(EVENT_SPELL_LIGHTNING_BREATH, 3000);
179  events.ScheduleEvent(EVENT_SPELL_POISON_CLOUD, 6000);
180  break;
181  case EVENT_SPELL_EYE_BEAM:
182  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 35.0f, true))
183  me->CastSpell(target, SPELL_EYE_BEAM, false);
184  break;
185  case EVENT_SPELL_LIGHTNING_BREATH:
186  me->CastSpell(me->GetVictim(), SPELL_LIGHTNING_BREATH, false);
187  events.ScheduleEvent(EVENT_SPELL_LIGHTNING_BREATH, 8000);
188  break;
189  case EVENT_SPELL_POISON_CLOUD:
190  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, 35.0f, true))
191  me->CastSpell(target, SPELL_POISON_CLOUD, false);
192  events.ScheduleEvent(EVENT_SPELL_POISON_CLOUD, 10000);
193  break;
194  case EVENT_TURN_BONES_REAL:
195  me->SetDisplayId(me->GetNativeDisplayId());
196  me->CastSpell(me, SPELL_FLESH_VISUAL, true);
197  me->CastSpell(me, SPELL_CLEAR_GIFT, true);
198  events.Reset();
199  events.ScheduleEvent(EVENT_SPELL_CURSE_OF_LIFE, 1000);
200  events.ScheduleEvent(EVENT_SPELL_RAIN_OF_FIRE, urand(12000, 14000));
201  events.ScheduleEvent(EVENT_SPELL_SHADOW_VOLLEY, urand(8000, 10000));
202  break;
203  }
204 
205  DoMeleeAttackIfReady();
206  }
