78  // resizes <container> to have at most <requestedSize> elements
79  // if it has more than <requestedSize> elements, the elements to keep are selected randomly
80  template<class C>
81  void RandomResize(C& container, std::size_t requestedSize)
82  {
83  static_assert(std::is_base_of<std::forward_iterator_tag, typename std::iterator_traits<typename C::iterator>::iterator_category>::value, "Invalid container passed to Acore::Containers::RandomResize");
84 
85  if (std::size(container) <= requestedSize)
86  {
87  return;
88  }
89 
90  auto keepIt = std::begin(container), curIt = std::begin(container);
91  uint32 elementsToKeep = requestedSize, elementsToProcess = std::size(container);
92 
93  while (elementsToProcess)
94  {
95  // this element has chance (elementsToKeep / elementsToProcess) of being kept
96  if (urand(1, elementsToProcess) <= elementsToKeep)
97  {
98  if (keepIt != curIt)
99  {
100  *keepIt = std::move(*curIt);
101  }
102 
103  ++keepIt;
104  --elementsToKeep;
105  }
106 
107  ++curIt;
108  --elementsToProcess;
109  }
110 
111  container.erase(keepIt, std::end(container));
112  }

115  void RandomResize(C& container, Predicate&& predicate, std::size_t requestedSize)
116  {
118  C containerCopy;

120 
121  if (requestedSize)

142  /*
143  * Select a random element from a container where each element has a different chance to be selected.
144  *
145  * @param container Container to select an element from
146  * @param weights Chances of each element to be selected, must be in the same order as elements in container.
147  * Caller is responsible for checking that sum of all weights is greater than 0.
148  *
149  * Note: container cannot be empty

159  /*
160  * Select a random element from a container where each element has a different chance to be selected.
161  *
162  * @param container Container to select an element from
163  * @param weightExtractor Function retrieving chance of each element in container, expected to take an element of the container and returning a double
164  *
165  * Note: container cannot be empty
166  */
167  template<class C, class Fn>
168  auto SelectRandomWeightedContainerElement(C const& container, Fn weightExtractor) -> decltype(std::begin(container))
169  {
170  std::vector<double> weights;
171  weights.reserve(std::size(container));
172  double weightSum = 0.0;
173 
174  for (auto& val : container)
175  {
176  double weight = weightExtractor(val);
177  weights.push_back(weight);
178  weightSum += weight;
179  }
180 
181  if (weightSum <= 0.0)
182  {
183  weights.assign(std::size(container), 1.0);
184  }
185 
186  return SelectRandomWeightedContainerElement(container, weights);
187  }
