164 void WorldSession::HandleAutoEquipItemOpcode(WorldPacket& recvData)
165 {
166  //LOG_DEBUG("network.opcode", "WORLD: CMSG_AUTOEQUIP_ITEM");
167  uint8 srcbag, srcslot;
168 
169  recvData >> srcbag >> srcslot;
170 
171  Item* pSrcItem = _player->GetItemByPos(srcbag, srcslot);
172  if (!pSrcItem)
173  return; // only at cheat
174 
175  ItemTemplate const* pProto = pSrcItem->GetTemplate();
176  if (!pProto)
177  {
178  _player->SendEquipError(pSrcItem->IsBag() ? EQUIP_ERR_ITEM_NOT_FOUND : EQUIP_ERR_ITEMS_CANT_BE_SWAPPED, pSrcItem);
179  return;
180  }
181 
182  uint8 eslot = _player->FindEquipSlot(pProto, NULL_SLOT, !pSrcItem->IsBag());
183  if (eslot == NULL_SLOT)
184  {
185  _player->SendEquipError(EQUIP_ERR_ITEM_CANT_BE_EQUIPPED, pSrcItem);
186  return;
187  }
188 
189  uint16 src = pSrcItem->GetPos();
190  uint16 dest = ((INVENTORY_SLOT_BAG_0 << 8) | eslot);
191  if (dest == src) // prevent equip in same slot, only at cheat
192  {
193  _player->SendEquipError(EQUIP_ERR_ITEM_CANT_BE_EQUIPPED, pSrcItem);
194  return;
195  }
196 
197  Item* pDstItem = _player->GetItemByPos(dest);
198 
199  // Remove item enchantments for now and restore it later
200  // Needed for swap sanity checks
201  if (pDstItem)
202  {
203  _player->ApplyEnchantment(pDstItem, false);
204  }
205 
206  InventoryResult msg = _player->CanEquipItem(NULL_SLOT, dest, pSrcItem, !pSrcItem->IsBag());
207  if (msg != EQUIP_ERR_OK)
208  {
209  // Restore enchantments
210  if (pDstItem)
211  {
212  _player->ApplyEnchantment(pDstItem, true);
213  }
214 
215  _player->SendEquipError(msg, pSrcItem, nullptr);
216  return;
217  }
218 
219  if (!pDstItem) // empty slot, simple case
220  {
221  _player->RemoveItem(srcbag, srcslot, true);
222  _player->EquipItem(dest, pSrcItem, true);
223  _player->AutoUnequipOffhandIfNeed();
224  }
225  else // have currently equipped item, not simple case
226  {
227  // Restore enchantments
228  _player->ApplyEnchantment(pDstItem, true);
229 
230  uint8 dstbag = pDstItem->GetBagSlot();
231  uint8 dstslot = pDstItem->GetSlot();
232 
233  msg = _player->CanUnequipItem(dest, !pSrcItem->IsBag());
234  if (msg != EQUIP_ERR_OK)
235  {
236  _player->SendEquipError(msg, pDstItem, nullptr);
237  return;
238  }
239 
240  // check dest->src move possibility
241  ItemPosCountVec sSrc;
242  uint16 eSrc = 0;
243  if (_player->IsInventoryPos(src))
244  {
245  msg = _player->CanStoreItem(srcbag, srcslot, sSrc, pDstItem, true);
246  if (msg != EQUIP_ERR_OK)
247  msg = _player->CanStoreItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
248  if (msg != EQUIP_ERR_OK)
249  msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
250  }
251  else if (_player->IsBankPos(src))
252  {
253  msg = _player->CanBankItem(srcbag, srcslot, sSrc, pDstItem, true);
254  if (msg != EQUIP_ERR_OK)
255  msg = _player->CanBankItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
256  if (msg != EQUIP_ERR_OK)
257  msg = _player->CanBankItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
258  }
259  else if (_player->IsEquipmentPos(src))
260  {
261  msg = _player->CanEquipItem(srcslot, eSrc, pDstItem, true);
262  if (msg == EQUIP_ERR_OK)
263  msg = _player->CanUnequipItem(eSrc, true);
264  }
265 
266  if (msg != EQUIP_ERR_OK)
267  {
268  _player->SendEquipError(msg, pDstItem, pSrcItem);
269  return;
270  }
271 
272  // now do moves, remove...
273  _player->RemoveItem(dstbag, dstslot, true, true);
274  _player->RemoveItem(srcbag, srcslot, true, true);
275 
276  // add to dest
277  _player->EquipItem(dest, pSrcItem, true);
278 
279  // add to src
280  if (_player->IsInventoryPos(src))
281  _player->StoreItem(sSrc, pDstItem, true);
282  else if (_player->IsBankPos(src))
283  _player->BankItem(sSrc, pDstItem, true);
284  else if (_player->IsEquipmentPos(src))
285  _player->EquipItem(eSrc, pDstItem, true);
286 
287  _player->AutoUnequipOffhandIfNeed();
288 
289  // Xinef: Call this here after all needed items are equipped
290  _player->RemoveItemDependentAurasAndCasts((Item*)nullptr);
291  }

294 void WorldSession::HandleDestroyItemOpcode(WorldPacket& recvData)
295 {
296  //LOG_DEBUG("network.opcode", "WORLD: CMSG_DESTROYITEM");
297  uint8 bag, slot, count, data1, data2, data3;
298 
299  recvData >> bag >> slot >> count >> data1 >> data2 >> data3;
300 
301  uint16 pos = (bag << 8) | slot;
302 
303  // prevent drop unequipable items (in combat, for example) and non-empty bags
304  if (_player->IsEquipmentPos(pos) || _player->IsBagPos(pos))
305  {
306  InventoryResult msg = _player->CanUnequipItem(pos, false);
307  if (msg != EQUIP_ERR_OK)
308  {
309  _player->SendEquipError(msg, _player->GetItemByPos(pos), nullptr);
310  return;
311  }
312  }
313 
314  Item* pItem = _player->GetItemByPos(bag, slot);
315  if (!pItem)
316  {
317  _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
318  return;
319  }
320 
321  if (pItem->GetTemplate()->Flags & ITEM_FLAG_NO_USER_DESTROY)
322  {
323  _player->SendEquipError(EQUIP_ERR_CANT_DROP_SOULBOUND, nullptr, nullptr);
324  return;
325  }
326 
327  recoveryItem(pItem);
328 
329  if (count)
330  {
331  uint32 i_count = count;
332  _player->DestroyItemCount(pItem, i_count, true);
333  }
334  else
335  {
336  _player->DestroyItem(bag, slot, true);
337  }
338  _player->SendQuestGiverStatusMultiple();
339 }

359 bool ItemTemplate::HasSpellPowerStat() const
360 {
361  bool invalid = false;
362  for (uint8 i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
363  {
364  _Spell const& spellData = Spells[i];
365  if (!spellData.SpellId)
366  {
367  continue;
368  }
369 
370  SpellInfo const* spellInfo = sSpellMgr->GetSpellInfo(spellData.SpellId);
371  if (!spellInfo)
372  {
373  continue;
374  }
375 
376  for (uint8 j = EFFECT_0; j <= EFFECT_2; ++j)
377  {
378  switch (spellInfo->Effects[j].ApplyAuraName)
379  {
380  case SPELL_AURA_MOD_HEALING_DONE:
381  case SPELL_AURA_MOD_SPELL_HEALING_OF_STAT_PERCENT:
382  case SPELL_AURA_MOD_SPELL_HEALING_OF_ATTACK_POWER:
383  case SPELL_AURA_MOD_HEALING:
384  invalid = true;
385  break;
386  case SPELL_AURA_MOD_DAMAGE_DONE:
387  case SPELL_AURA_MOD_SPELL_DAMAGE_OF_STAT_PERCENT:
388  case SPELL_AURA_MOD_SPELL_DAMAGE_OF_ATTACK_POWER:
389  case SPELL_AURA_MOD_DAMAGE_TAKEN:
390  if (!(spellInfo->Effects[j].MiscValue & SPELL_SCHOOL_MASK_SPELL))
391  {
392  return false;
393  }
394  invalid = true;
395  break;
396  default:
397  break;
398  }
399  }
400  }
401 
402  return invalid;
403 }
