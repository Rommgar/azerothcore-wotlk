243  bool CheckPassword(uint32 accountId, std::string password)
244  {
245  std::string username;
246 
247  if (!GetName(accountId, username))
248  return false;
249 
250  Utf8ToUpperOnlyLatin(username);
251  Utf8ToUpperOnlyLatin(password);
252 
253  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_CHECK_PASSWORD);
254  stmt->setUInt32(0, accountId);
255  if (PreparedQueryResult result = LoginDatabase.Query(stmt))
256  {
257  Acore::Crypto::SRP6::Salt salt = (*result)[0].GetBinary<Acore::Crypto::SRP6::SALT_LENGTH>();
258  Acore::Crypto::SRP6::Verifier verifier = (*result)[1].GetBinary<Acore::Crypto::SRP6::VERIFIER_LENGTH>();
259  if (Acore::Crypto::SRP6::CheckLogin(username, password, salt, verifier))
260  return true;
261  }
262 
263  return false;
