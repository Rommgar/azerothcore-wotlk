107  void UpdateAI(uint32 diff) override
108  {
109  // I have a feeling this isn't blizzlike, but owell, I'm only passing by and cleaning up.
110  if (!_hasAura)
111  {
112  for (uint8 i = 0; i < MAX_SPELL_PAIRS; ++i)
113  if (_auraPairs[i].npcEntry == me->GetEntry())
114  DoCast(me, _auraPairs[i].spellId);
115 
116  _hasAura = true;
117  }
118 
119  if (!UpdateVictim())
120  return;
121 
122  events.Update(diff);
123 
124  if (me->HasUnitState(UNIT_STATE_CASTING))
125  return;
126 
127  while (uint32 eventId = events.ExecuteEvent())
128  {
129  switch (eventId)
130  {
131  case EVENT_CHARGE_TARGET:
132  DoCastVictim(SPELL_CHARGE);
133  events.ScheduleEvent(EVENT_CHARGE, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
134  break;
135  case EVENT_CLEAVE:
136  DoCastVictim(SPELL_CLEAVE);
137  events.ScheduleEvent(EVENT_CLEAVE, urand(10 * IN_MILLISECONDS, 16 * IN_MILLISECONDS));
138  break;
139  case EVENT_DEMORALIZING_SHOUT:
140  DoCast(me, SPELL_DEMORALIZING_SHOUT);
141  events.ScheduleEvent(EVENT_DEMORALIZING_SHOUT, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
142  break;
143  case EVENT_WHIRLWIND:
144  DoCast(me, SPELL_WHIRLWIND);
145  events.ScheduleEvent(EVENT_WHIRLWIND, urand(10 * IN_MILLISECONDS, 25 * IN_MILLISECONDS));
146  break;
147  case EVENT_ENRAGE:
148  DoCast(me, SPELL_ENRAGE);
149  events.ScheduleEvent(EVENT_ENRAGE, urand(10 * IN_MILLISECONDS, 30 * IN_MILLISECONDS));
150  break;
151  case EVENT_CHECK_RESET:
152  {
153  Position const& _homePosition = me->GetHomePosition();
154  if (me->GetDistance2d(_homePosition.GetPositionX(), _homePosition.GetPositionY()) > 50.0f)
155  {
156  EnterEvadeMode();
157  return;
158  }
159  events.ScheduleEvent(EVENT_CHECK_RESET, 5000);
160  break;
161  }
162  }
163  }
164 
165  DoMeleeAttackIfReady();
166  }
167 
168  private:
169  EventMap events;
170  bool _hasAura;
171  };
