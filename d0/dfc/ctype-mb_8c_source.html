147  Case folding functions for character set
148  where case conversion can change string octet length.

150  _euckr 0xA9A5 == "LATIN LETTER DOTLESS I" (Turkish letter)

156 my_casefold_mb_varlen(CHARSET_INFO *cs,
157  char *src, size_t srclen,
158  char *dst, size_t dstlen __attribute__((unused)),
159  uchar *map,

161 {

163 
164  DBUG_ASSERT(cs->mbmaxlen == 2);
165

167  {

183  *dst++= *src++;
184  }

254 int my_wildcmp_mb(CHARSET_INFO *cs,
255  const char *str,const char *str_end,
256  const char *wildstr,const char *wildend,
257  int escape, int w_one, int w_many)
258 {
259  int result= -1; /* Not found, using wildcards */
260 
261  while (wildstr != wildend)
262  {
263  while (*wildstr != w_many && *wildstr != w_one)
264  {
265  int l;
266  if (*wildstr == escape && wildstr+1 != wildend)
267  wildstr++;
268  if ((l = my_ismbchar(cs, wildstr, wildend)))
269  {
270  if (str+l > str_end || memcmp(str, wildstr, l) != 0)
271  return 1;
272  str += l;
273  wildstr += l;
274  }
275  else
276  if (str == str_end || likeconv(cs,*wildstr++) != likeconv(cs,*str++))
277  return(1); /* No match */
278  if (wildstr == wildend)
279  return (str != str_end); /* Match if both are at end */
280  result=1; /* Found an anchor char */
281  }
282  if (*wildstr == w_one)
283  {
284  do
285  {
286  if (str == str_end) /* Skip one char if possible */
287  return (result);
288  INC_PTR(cs,str,str_end);
289  } while (++wildstr < wildend && *wildstr == w_one);
290  if (wildstr == wildend)
291  break;
292  }
293  if (*wildstr == w_many)
294  { /* Found w_many */
295  uchar cmp;
296  const char* mb = wildstr;
297  int mb_len=0;
298 
299  wildstr++;
300  /* Remove any '%' and '_' from the wild search string */
301  for (; wildstr != wildend ; wildstr++)
302  {
303  if (*wildstr == w_many)
304  continue;
305  if (*wildstr == w_one)
306  {
307  if (str == str_end)
308  return (-1);
309  INC_PTR(cs,str,str_end);
310  continue;
311  }
312  break; /* Not a wild character */
313  }
314  if (wildstr == wildend)
315  return(0); /* Ok if w_many is last */
316  if (str == str_end)
317  return -1;
318 
319  if ((cmp= *wildstr) == escape && wildstr+1 != wildend)
320  cmp= *++wildstr;
321 
322  mb=wildstr;
323  mb_len= my_ismbchar(cs, wildstr, wildend);
324  INC_PTR(cs,wildstr,wildend); /* This is compared trough cmp */
325  cmp=likeconv(cs,cmp);
326  do
327  {
328  for (;;)
329  {
330  if (str >= str_end)
331  return -1;
332  if (mb_len)
333  {
334  if (str+mb_len <= str_end && memcmp(str, mb, mb_len) == 0)
335  {
336  str += mb_len;
337  break;
338  }
339  }
340  else if (!my_ismbchar(cs, str, str_end) &&
341  likeconv(cs,*str) == cmp)
342  {
343  str++;
344  break;
345  }
346  INC_PTR(cs,str, str_end);
347  }
348  {
349  int tmp=my_wildcmp_mb(cs,str,str_end,wildstr,wildend,escape,w_one,
350  w_many);
351  if (tmp <= 0)
352  return (tmp);
353  }
354  } while (str != str_end && wildstr[0] != w_many);
355  return(-1);
356  }
357  }
358  return (str != str_end ? 1 : 0);
359 }

601  str Start of buffer to fill
602  end End of buffer to fill
603 
604  DESCRIPTION
605  Write max key:
606  - for non-Unicode character sets:
607  just bfill using max_sort_char if max_sort_char is one byte.
608  In case when max_sort_char is two bytes, fill with double-byte pairs
609  and optionally pad with a single space character.
610  - for Unicode character set (utf-8):
611  create a buffer with multibyte representation of the max_sort_char
612  character, and copy it into max_str in a loop.
613 */

624  return;
625  }
626  buf[0]= cs->max_sort_char >> 8;
627  buf[1]= cs->max_sort_char & 0xFF;
628  buflen= 2;
629  }
630  else
631  {
632  buflen= cs->cset->wc_mb(cs, cs->max_sort_char, (uchar*) buf,
633  (uchar*) buf + sizeof(buf));
634  }
635 
636  DBUG_ASSERT(buflen > 0);
637  do
638  {
639  if ((str + buflen) <= end)
640  {
641  /* Enough space for the characer */
642  memcpy(str, buf, buflen);
643  str+= buflen;
644  }
645  else
646  {
647  /*
648  There is no space for whole multibyte
649  character, then add trailing spaces.
650  */
651  *str++= ' ';
652  }
653  } while (str < end);

663 ** res_length Length of min_str and max_str.
664 ** min_str Smallest case sensitive string that ranges LIKE.
665 ** Should be space padded to res_length.
666 ** max_str Largest case sensitive string that ranges LIKE.

668 **
669 ** The function should return 0 if ok and 1 if the LIKE string can't be
670 ** optimized !
671 */

673 my_bool my_like_range_mb(CHARSET_INFO *cs,
674  const char *ptr,size_t ptr_length,
675  pbool escape, pbool w_one, pbool w_many,
676  size_t res_length,
677  char *min_str,char *max_str,
678  size_t *min_length,size_t *max_length)
679 {
680  uint mb_len;
681  const char *end= ptr + ptr_length;
682  char *min_org= min_str;
683  char *min_end= min_str + res_length;
684  char *max_end= max_str + res_length;
685  size_t maxcharlen= res_length / cs->mbmaxlen;
686  const char *contraction_flags= cs->contractions ?
687  ((const char*) cs->contractions) + 0x40*0x40 : NULL;
688 
689  for (; ptr != end && min_str != min_end && maxcharlen ; maxcharlen--)
690  {
691  /* We assume here that escape, w_any, w_namy are one-byte characters */
692  if (*ptr == escape && ptr+1 != end)
693  ptr++; /* Skip escape */
694  else if (*ptr == w_one || *ptr == w_many) /* '_' and '%' in SQL */
695  {
696 fill_max_and_min:
697  /*
698  Calculate length of keys:
699  'a\0\0... is the smallest possible string when we have space expand
700  a\ff\ff... is the biggest possible string
701  */
702  *min_length= ((cs->state & MY_CS_BINSORT) ? (size_t) (min_str - min_org) :
703  res_length);
704  *max_length= res_length;
705  /* Create min key */
706  do
707  {
708  *min_str++= (char) cs->min_sort_char;
709  } while (min_str != min_end);
710 
711  /*
712  Write max key: create a buffer with multibyte
713  representation of the max_sort_char character,
714  and copy it into max_str in a loop.
715  */
716  *max_length= res_length;
717  pad_max_char(cs, max_str, max_end);
718  return 0;
719  }
720  if ((mb_len= my_ismbchar(cs, ptr, end)) > 1)
721  {
722  if (ptr+mb_len > end || min_str+mb_len > min_end)
723  break;
724  while (mb_len--)
725  *min_str++= *max_str++= *ptr++;
726  }
727  else
728  {
729  /*
730  Special case for collations with contractions.
731  For example, in Chezh, 'ch' is a separate letter
732  which is sorted between 'h' and 'i'.
733  If the pattern 'abc%', 'c' at the end can mean:
734  - letter 'c' itself,
735  - beginning of the contraction 'ch'.
736 
737  If we simply return this LIKE range:
738 
739  'abc\min\min\min' and 'abc\max\max\max'
740 
741  then this query: SELECT * FROM t1 WHERE a LIKE 'abc%'
742  will only find values starting from 'abc[^h]',
743  but won't find values starting from 'abch'.
744 
745  We must ignore contraction heads followed by w_one or w_many.
746  ('Contraction head' means any letter which can be the first
747  letter in a contraction)
748 
749  For example, for Czech 'abc%', we will return LIKE range,
750  which is equal to LIKE range for 'ab%':
751 
752  'ab\min\min\min\min' and 'ab\max\max\max\max'.
753 
754  */
755  if (contraction_flags && ptr + 1 < end &&
756  contraction_flags[(uchar) *ptr])
757  {
758  /* Ptr[0] is a contraction head. */
759 
760  if (ptr[1] == w_one || ptr[1] == w_many)
761  {
762  /* Contraction head followed by a wildcard, quit. */
763  goto fill_max_and_min;
764  }
765 
766  /*
767  Some letters can be both contraction heads and contraction tails.
768  For example, in Danish 'aa' is a separate single letter which
769  is sorted after 'z'. So 'a' can be both head and tail.
770 
771  If ptr[0]+ptr[1] is a contraction,
772  then put both letters together.
773 
774  If ptr[1] can be a contraction part, but ptr[0]+ptr[1]
775  is not a contraction, then we put only ptr[0],
776  and continue with ptr[1] on the next loop.
777  */
778  if (contraction_flags[(uchar) ptr[1]] &&
779  cs->contractions[(*ptr-0x40)*0x40 + ptr[1] - 0x40])
780  {
781  /* Contraction found */
782  if (maxcharlen == 1 || min_str + 1 >= min_end)
783  {
784  /* Both contraction parts don't fit, quit */
785  goto fill_max_and_min;
786  }
787 
788  /* Put contraction head */
789  *min_str++= *max_str++= *ptr++;
790  maxcharlen--;
791  }
792  }
793  /* Put contraction tail, or a single character */
794  *min_str++= *max_str++= *ptr++;
795  }
796  }
797 
798  *min_length= *max_length = (size_t) (min_str - min_org);
799  while (min_str != min_end)
800  *min_str++= *max_str++= ' '; /* Because if key compression */
801  return 0;

906  (size_t) (min_str - min_org) :

958  (uchar*) max_str, (uchar*) max_end)) <= 0)

971  requires that "length" argument is divisible to mbminlen.
972  Make sure to call fill() with proper "length" argument.
973  */

984  memset(min_end - res_length_diff, 0, res_length_diff);
985  memset(max_end - res_length_diff, 0, res_length_diff);
986  }
987  return FALSE;
988 }

1015  return(1); /* No match */

1076  }

1090  return(-1);
