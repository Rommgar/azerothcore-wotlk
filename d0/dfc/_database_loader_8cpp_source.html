35 DatabaseLoader& DatabaseLoader::AddDatabase(DatabaseWorkerPool<T>& pool, std::string const& name)
36 {
37  bool const updatesEnabledForThis = DBUpdater<T>::IsEnabled(_updateFlags);
38 
39  _open.push([this, name, updatesEnabledForThis, &pool]() -> bool
40  {
41  std::string const dbString = sConfigMgr->GetOption<std::string>(name + "DatabaseInfo", "");
42  if (dbString.empty())
43  {
44  LOG_ERROR(_logger, "Database %s not specified in configuration file!", name.c_str());
45  return false;
46  }
47 
48  uint8 const asyncThreads = sConfigMgr->GetOption<uint8>(name + "Database.WorkerThreads", 1);
49  if (asyncThreads < 1 || asyncThreads > 32)
50  {
51  LOG_ERROR(_logger, "%s database: invalid number of worker threads specified. "
52  "Please pick a value between 1 and 32.", name.c_str());
53  return false;
54  }
55 
56  uint8 const synchThreads = sConfigMgr->GetOption<uint8>(name + "Database.SynchThreads", 1);
57 
58  pool.SetConnectionInfo(dbString, asyncThreads, synchThreads);
59 
60  if (uint32 error = pool.Open())
61  {
62  // Try reconnect
63  if (error == CR_CONNECTION_ERROR)
64  {
65  uint8 const attempts = sConfigMgr->GetOption<uint8>("Database.Reconnect.Attempts", 20);
66  Seconds reconnectSeconds = Seconds(sConfigMgr->GetOption<uint8>("Database.Reconnect.Seconds", 15));
67  uint8 reconnectCount = 0;
68 
69  while (reconnectCount < attempts)
70  {
71  LOG_INFO(_logger, "> Retrying after %u seconds", static_cast<uint32>(reconnectSeconds.count()));
72  std::this_thread::sleep_for(reconnectSeconds);
73  error = pool.Open();
74 
75  if (error == CR_CONNECTION_ERROR)
76  {
77  reconnectCount++;
78  }
79  else
80  {
81  break;
82  }
83  }
84  }
85 
86  // Database does not exist
87  if ((error == ER_BAD_DB_ERROR) && updatesEnabledForThis && _autoSetup && !sConfigMgr->isDryRun())
88  {
89  // Try to create the database and connect again if auto setup is enabled
90  if (DBUpdater<T>::Create(pool) && (!pool.Open()))
91  {
92  error = 0;
93  }
94  }
95 
96  // If the error wasn't handled quit
97  if (error)
98  {
99  LOG_ERROR(_logger, "DatabasePool %s NOT opened. There were errors opening the MySQL connections. "
100  "Check your log file for specific errors", name.c_str());
101 
102  return false;
103  }
104  }
105  // Add the close operation
106  _close.push([&pool]
107  {
108  pool.Close();
109  });
110 
111  return true;
112  });
113 
114  // Populate and update only if updates are enabled for this pool
115  if (updatesEnabledForThis && !sConfigMgr->isDryRun())
116  {
117  _populate.push([this, name, &pool]() -> bool
118  {
119  if (!DBUpdater<T>::Populate(pool))
120  {
121  LOG_ERROR(_logger, "Could not populate the %s database, see log for details.", name.c_str());
122  return false;
123  }
124 
125  return true;
126  });
127 
128  _update.push([this, name, &pool]() -> bool
129  {
130  if (!DBUpdater<T>::Update(pool))
131  {
132  LOG_ERROR(_logger, "Could not update the %s database, see log for details.", name.c_str());
133  return false;
134  }
135 
136  return true;
137  });
138  }
139 
140  _prepare.push([this, name, &pool]() -> bool
141  {
142  if (!pool.PrepareStatements())
143  {
144  LOG_ERROR(_logger, "Could not prepare statements of the %s database, see log for details.", name.c_str());
145  return false;
146  }
147 
148  return true;
149  });
150 
151  return *this;
152 }

179 bool DatabaseLoader::Process(std::queue<Predicate>& queue)
180 {
181  while (!queue.empty())
182  {
183  if (!queue.front()())
184  {
185  // Close all open databases which have a registered close operation
186  while (!_close.empty())
187  {
188  _close.top()();
189  _close.pop();
190  }
191 
192  return false;
193  }
194 
195  queue.pop();
196  }
197 
198  return true;
199 }
