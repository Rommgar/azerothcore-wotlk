46 inline bool isNasty(uint8 c)
47 {
48  if (c == '\t')
49  {
50  return false;
51  }
52 
53  if (c <= '\037') // ASCII control block
54  {
55  return true;
56  }
57 
58  return false;
59 }

61 void WorldSession::HandleMessagechatOpcode(WorldPacket& recvData)
62 {
63  uint32 type;
64  uint32 lang;
65 
66  recvData >> type;
67  recvData >> lang;
68 
69  if (type >= MAX_CHAT_MSG_TYPE)
70  {
71  LOG_ERROR("network.opcode", "CHAT: Wrong message type received: %u", type);
72  recvData.rfinish();
73  return;
74  }
75 
76  if (lang == LANG_UNIVERSAL && type != CHAT_MSG_AFK && type != CHAT_MSG_DND)
77  {
78  LOG_ERROR("entities.player.cheat", "CMSG_MESSAGECHAT: Possible hacking-attempt: %s tried to send a message in universal language", GetPlayerInfo().c_str());
79  SendNotification(LANG_UNKNOWN_LANGUAGE);
80  recvData.rfinish();
81  return;
82  }
83 
84  Player* sender = GetPlayer();
85 
86  // prevent talking at unknown language (cheating)
87  LanguageDesc const* langDesc = GetLanguageDescByID(lang);
88  if (!langDesc)
89  {
90  SendNotification(LANG_UNKNOWN_LANGUAGE);
91  recvData.rfinish();
92  return;
93  }
94 
95  if (langDesc->skill_id != 0 && !sender->HasSkill(langDesc->skill_id))
96  {
97  // also check SPELL_AURA_COMPREHEND_LANGUAGE (client offers option to speak in that language)
98  bool foundAura = false;
99  for (auto const& auraEff : sender->GetAuraEffectsByType(SPELL_AURA_COMPREHEND_LANGUAGE))
100  {
101  if (auraEff->GetMiscValue() == int32(lang))
102  {
103  foundAura = true;
104  break;
105  }
106  }
107 
108  if (!foundAura)
109  {
110  SendNotification(LANG_NOT_LEARNED_LANGUAGE);
111  recvData.rfinish();
112  return;
113  }
114  }
115 
116  // pussywizard: chatting on most chat types requires 2 hours played to prevent spam/abuse
117  if (AccountMgr::IsPlayerAccount(GetSecurity()))
118  {
119  switch (type)
120  {
121  case CHAT_MSG_ADDON:
122  case CHAT_MSG_PARTY:
123  case CHAT_MSG_RAID:
124  case CHAT_MSG_GUILD:
125  case CHAT_MSG_OFFICER:
126  case CHAT_MSG_AFK:
127  case CHAT_MSG_DND:
128  case CHAT_MSG_RAID_LEADER:
129  case CHAT_MSG_RAID_WARNING:
130  case CHAT_MSG_BATTLEGROUND:
131  case CHAT_MSG_BATTLEGROUND_LEADER:
132  case CHAT_MSG_PARTY_LEADER:
133  break;
134  default:
135  {
136  if (sWorld->getBoolConfig(CONFIG_CHAT_MUTE_FIRST_LOGIN))
137  {
138  uint32 minutes = sWorld->getIntConfig(CONFIG_CHAT_TIME_MUTE_FIRST_LOGIN);
139 
140  if (sender->GetTotalPlayedTime() < minutes * MINUTE)
141  {
142  SendNotification(LANG_MUTED_PLAYER, minutes);
143  recvData.rfinish();
144  return;
145  }
146  }
147  }
148  }
149  }
150 
151  // pussywizard:
152  switch (type)
153  {
154  case CHAT_MSG_SAY:
155  case CHAT_MSG_YELL:
156  case CHAT_MSG_EMOTE:
157  case CHAT_MSG_TEXT_EMOTE:
158  case CHAT_MSG_AFK:
159  case CHAT_MSG_DND:
160  if (sender->IsSpectator())
161  {
162  recvData.rfinish();
163  return;
164  }
165  }
166 
167  if (sender->HasAura(1852) && type != CHAT_MSG_WHISPER)
168  {
169  SendNotification(GetAcoreString(LANG_GM_SILENCE), sender->GetName().c_str());
170  recvData.rfinish();
171  return;
172  }
173 
174  if (lang == LANG_ADDON)
175  {
176  // LANG_ADDON is only valid for the following message types
177  switch (type)
178  {
179  case CHAT_MSG_PARTY:
180  case CHAT_MSG_RAID:
181  case CHAT_MSG_GUILD:
182  case CHAT_MSG_BATTLEGROUND:
183  case CHAT_MSG_WHISPER:
184  // check if addon messages are disabled
185  if (!sWorld->getBoolConfig(CONFIG_ADDON_CHANNEL))
186  {
187  recvData.rfinish();
188  return;
189  }
190  break;
191  default:
192  LOG_ERROR("network", "Player %s (%s) sent a chatmessage with an invalid language/message type combination",
193  GetPlayer()->GetName().c_str(), GetPlayer()->GetGUID().ToString().c_str());
194 
195  recvData.rfinish();
196  return;
197  }
198  }
199  // LANG_ADDON should not be changed nor be affected by flood control
200  else
201  {
202  uint32 specialMessageLimit = 0;
203  // send in universal language if player in .gmon mode (ignore spell effects)
204  if (sender->IsGameMaster())
205  lang = LANG_UNIVERSAL;
206  else
207  {
208  // send in universal language in two side iteration allowed mode
209  if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_CHAT))
210  lang = LANG_UNIVERSAL;
211  else
212  {
213  switch (type)
214  {
215  case CHAT_MSG_PARTY:
216  case CHAT_MSG_PARTY_LEADER:
217  case CHAT_MSG_RAID:
218  case CHAT_MSG_RAID_LEADER:
219  case CHAT_MSG_RAID_WARNING:
220  // allow two side chat at group channel if two side group allowed
221  if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GROUP))
222  lang = LANG_UNIVERSAL;
223 
224  specialMessageLimit = 35;
225  break;
226  case CHAT_MSG_GUILD:
227  case CHAT_MSG_OFFICER:
228  // allow two side chat at guild channel if two side guild allowed
229  if (sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_GUILD))
230  lang = LANG_UNIVERSAL;
231 
232  specialMessageLimit = 15;
233  break;
234  case CHAT_MSG_WHISPER:
235  if (sender->getLevel() >= 80)
236  specialMessageLimit = 15;
237  break;
238  }
239  }
240  // but overwrite it by SPELL_AURA_MOD_LANGUAGE auras (only single case used)
241  Unit::AuraEffectList const& ModLangAuras = sender->GetAuraEffectsByType(SPELL_AURA_MOD_LANGUAGE);
242  if (!ModLangAuras.empty())
243  lang = ModLangAuras.front()->GetMiscValue();
244  }
245 
246  if (type != CHAT_MSG_AFK && type != CHAT_MSG_DND)
247  sender->UpdateSpeakTime(specialMessageLimit);
248  }
249 
250  std::string to, channel, msg;
251  bool ignoreChecks = false;
252  switch (type)
253  {
254  case CHAT_MSG_SAY:
255  case CHAT_MSG_EMOTE:
256  case CHAT_MSG_YELL:
257  case CHAT_MSG_PARTY:
258  case CHAT_MSG_PARTY_LEADER:
259  case CHAT_MSG_GUILD:
260  case CHAT_MSG_OFFICER:
261  case CHAT_MSG_RAID:
262  case CHAT_MSG_RAID_LEADER:
263  case CHAT_MSG_RAID_WARNING:
264  case CHAT_MSG_BATTLEGROUND:
265  case CHAT_MSG_BATTLEGROUND_LEADER:
266  msg = recvData.ReadCString(lang != LANG_ADDON);
267  break;
268  case CHAT_MSG_WHISPER:
269  recvData >> to;
270  msg = recvData.ReadCString(lang != LANG_ADDON);
271  break;
272  case CHAT_MSG_CHANNEL:
273  recvData >> channel;
274  msg = recvData.ReadCString(lang != LANG_ADDON);
275  break;
276  case CHAT_MSG_AFK:
277  case CHAT_MSG_DND:
278  msg = recvData.ReadCString(lang != LANG_ADDON);
279  ignoreChecks = true;
280  break;
281  }
282 
283  // Our Warden module also uses SendAddonMessage as a way to communicate Lua check results to the server, see if this is that
284  if (type == CHAT_MSG_GUILD && lang == LANG_ADDON && _warden && _warden->ProcessLuaCheckResponse(msg))
285  {
286  return;
287  }
288 
289  // pussywizard:
290  if (msg.length() > 255 || (lang != LANG_ADDON && msg.find("|0") != std::string::npos))
291  return;
292 
293  if (!ignoreChecks)
294  {
295  if (msg.empty())
296  return;
297 
298  if (ChatHandler(this).ParseCommands(msg.c_str()))
299  return;
300 
301  if (!_player->CanSpeak())
302  {
303  std::string timeStr = secsToTimeString(m_muteTime - time(nullptr));
304  SendNotification(GetAcoreString(LANG_WAIT_BEFORE_SPEAKING), timeStr.c_str());
305  return;
306  }
307  }
308 
309  // do message validity checks
310  if (lang != LANG_ADDON)
311  {
312  // cut at the first newline or carriage return
313  std::string::size_type pos = msg.find_first_of("\n\r");
314 
315  if (pos == 0)
316  {
317  return;
318  }
319  else if (pos != std::string::npos)
320  {
321  msg.erase(pos);
322  }
323 
324  // abort on any sort of nasty character
325  for (uint8 c : msg)
326  {
327  if (isNasty(c))
328  {
329  LOG_ERROR("network", "Player %s %s sent a message containing invalid character %u - blocked", GetPlayer()->GetName().c_str(),
330  GetPlayer()->GetGUID().ToString().c_str(), uint8(c));
331  return;
332  }
333  }
334 
335  // collapse multiple spaces into one
336  if (sWorld->getBoolConfig(CONFIG_CHAT_FAKE_MESSAGE_PREVENTING))
337  {
338  auto end = std::unique(msg.begin(), msg.end(), [](char c1, char c2) { return (c1 == ' ') && (c2 == ' '); });
339  msg.erase(end, msg.end());
340  }
341 
342  // Validate hyperlinks
343  if (!ValidateHyperlinksAndMaybeKick(msg))
344  {
345  return;
346  }
347  }
348 
349  sScriptMgr->OnBeforeSendChatMessage(_player, type, lang, msg);
350 
351  switch (type)
352  {
353  case CHAT_MSG_SAY:
354  case CHAT_MSG_EMOTE:
355  case CHAT_MSG_YELL:
356  {
357  // Prevent cheating
358  if (!sender->IsAlive())
359  return;
360 
361  if (sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_SAY_LEVEL_REQ))
362  {
363  SendNotification(GetAcoreString(LANG_SAY_REQ), sWorld->getIntConfig(CONFIG_CHAT_SAY_LEVEL_REQ));
364  return;
365  }
366 
367  if (type == CHAT_MSG_SAY)
368  sender->Say(msg, Language(lang));
369  else if (type == CHAT_MSG_EMOTE)
370  sender->TextEmote(msg);
371  else if (type == CHAT_MSG_YELL)
372  sender->Yell(msg, Language(lang));
373  }
374  break;
375  case CHAT_MSG_WHISPER:
376  {
377  if (sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_WHISPER_LEVEL_REQ))
378  {
379  SendNotification(GetAcoreString(LANG_WHISPER_REQ), sWorld->getIntConfig(CONFIG_CHAT_WHISPER_LEVEL_REQ));
380  return;
381  }
382 
383  if (!normalizePlayerName(to))
384  {
385  SendPlayerNotFoundNotice(to);
386  break;
387  }
388 
389  Player* receiver = ObjectAccessor::FindPlayerByName(to, false);
390  bool senderIsPlayer = AccountMgr::IsPlayerAccount(GetSecurity());
391  bool receiverIsPlayer = AccountMgr::IsPlayerAccount(receiver ? receiver->GetSession()->GetSecurity() : SEC_PLAYER);
392  if (!receiver || (senderIsPlayer && !receiverIsPlayer && !receiver->isAcceptWhispers() && !receiver->IsInWhisperWhiteList(sender->GetGUID())))
393  {
394  SendPlayerNotFoundNotice(to);
395  return;
396  }
397 
398  if (!sWorld->getBoolConfig(CONFIG_ALLOW_TWO_SIDE_INTERACTION_CHAT) && senderIsPlayer && receiverIsPlayer)
399  if (GetPlayer()->GetTeamId() != receiver->GetTeamId())
400  {
401  SendWrongFactionNotice();
402  return;
403  }
404 
405  // pussywizard: optimization
406  if (GetPlayer()->HasAura(1852) && !receiver->IsGameMaster())
407  {
408  SendNotification(GetAcoreString(LANG_GM_SILENCE), GetPlayer()->GetName().c_str());
409  return;
410  }
411 
412  // If player is a Gamemaster and doesn't accept whisper, we auto-whitelist every player that the Gamemaster is talking to
413  if (!senderIsPlayer && !sender->isAcceptWhispers() && !sender->IsInWhisperWhiteList(receiver->GetGUID()))
414  sender->AddWhisperWhiteList(receiver->GetGUID());
415 
416  GetPlayer()->Whisper(msg, Language(lang), receiver);
417  }
418  break;
419  case CHAT_MSG_PARTY:
420  case CHAT_MSG_PARTY_LEADER:
421  {
422  // if player is in battleground, he cannot say to battleground members by /p
423  Group* group = GetPlayer()->GetOriginalGroup();
424  if (!group)
425  {
426  group = sender->GetGroup();
427  if (!group || group->isBGGroup())
428  return;
429  }
430 
431  if (type == CHAT_MSG_PARTY_LEADER && !group->IsLeader(sender->GetGUID()))
432  return;
433 
434  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
435 #ifdef ELUNA
436  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, group))
437  return;
438 #endif
439  WorldPacket data;
440  ChatHandler::BuildChatPacket(data, ChatMsg(type), Language(lang), sender, nullptr, msg);
441  group->BroadcastPacket(&data, false, group->GetMemberGroup(GetPlayer()->GetGUID()));
442  }
443  break;
444  case CHAT_MSG_GUILD:
445  {
446  if (GetPlayer()->GetGuildId())
447  {
448  if (Guild* guild = sGuildMgr->GetGuildById(GetPlayer()->GetGuildId()))
449  {
450  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, guild);
451 
452 #ifdef ELUNA
453  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, guild))
454  return;
455 #endif
456  guild->BroadcastToGuild(this, false, msg, lang == LANG_ADDON ? LANG_ADDON : LANG_UNIVERSAL);
457  }
458  }
459  }
460  break;
461  case CHAT_MSG_OFFICER:
462  {
463  if (GetPlayer()->GetGuildId())
464  {
465  if (Guild* guild = sGuildMgr->GetGuildById(GetPlayer()->GetGuildId()))
466  {
467  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, guild);
468 
469 #ifdef ELUNA
470  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, guild))
471  return;
472 #endif
473  guild->BroadcastToGuild(this, true, msg, lang == LANG_ADDON ? LANG_ADDON : LANG_UNIVERSAL);
474  }
475  }
476  }
477  break;
478  case CHAT_MSG_RAID:
479  {
480  // if player is in battleground, he cannot say to battleground members by /ra
481  Group* group = GetPlayer()->GetOriginalGroup();
482  if (!group)
483  {
484  group = GetPlayer()->GetGroup();
485  if (!group || group->isBGGroup() || !group->isRaidGroup())
486  return;
487  }
488 
489  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
490 #ifdef ELUNA
491  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, group))
492  return;
493 #endif
494  WorldPacket data;
495  ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID, Language(lang), sender, nullptr, msg);
496  group->BroadcastPacket(&data, false);
497  }
498  break;
499  case CHAT_MSG_RAID_LEADER:
500  {
501  // if player is in battleground, he cannot say to battleground members by /ra
502  Group* group = GetPlayer()->GetOriginalGroup();
503  if (!group)
504  {
505  group = GetPlayer()->GetGroup();
506  if (!group || group->isBGGroup() || !group->isRaidGroup() || !group->IsLeader(sender->GetGUID()))
507  return;
508  }
509 
510  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
511 #ifdef ELUNA
512  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, group))
513  return;
514 #endif
515  WorldPacket data;
516  ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID_LEADER, Language(lang), sender, nullptr, msg);
517  group->BroadcastPacket(&data, false);
518  }
519  break;
520  case CHAT_MSG_RAID_WARNING:
521  {
522  Group* group = GetPlayer()->GetGroup();

524  return;
525 
526  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);

530 #endif
531  WorldPacket data;

534  group->BroadcastPacket(&data, false);
535  }
536  break;
537  case CHAT_MSG_BATTLEGROUND:
538  {
539  //battleground raid is always in Player->GetGroup(), never in GetOriginalGroup()
540  Group* group = GetPlayer()->GetGroup();
541  if (!group || !group->isBGGroup())
542  return;
543 
544  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
545 #ifdef ELUNA
546  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, group))
547  return;
548 #endif
549  WorldPacket data;
550  ChatHandler::BuildChatPacket(data, CHAT_MSG_BATTLEGROUND, Language(lang), sender, nullptr, msg);
551  group->BroadcastPacket(&data, false);
552  }
553  break;
554  case CHAT_MSG_BATTLEGROUND_LEADER:
555  {
556  // battleground raid is always in Player->GetGroup(), never in GetOriginalGroup()
557  Group* group = GetPlayer()->GetGroup();
558  if (!group || !group->isBGGroup() || !group->IsLeader(GetPlayer()->GetGUID()))
559  return;
560 
561  sScriptMgr->OnPlayerChat(GetPlayer(), type, lang, msg, group);
562 #ifdef ELUNA
563  if (!sEluna->OnChat(GetPlayer(), type, lang, msg, group))
564  return;
565 #endif
566  WorldPacket data;
567  ChatHandler::BuildChatPacket(data, CHAT_MSG_BATTLEGROUND_LEADER, Language(lang), sender, nullptr, msg);
568  group->BroadcastPacket(&data, false);
569  }
570  break;
571  case CHAT_MSG_CHANNEL:
572  {
573  if (AccountMgr::IsPlayerAccount(GetSecurity()))
574  {
575  if (sender->getLevel() < sWorld->getIntConfig(CONFIG_CHAT_CHANNEL_LEVEL_REQ))
576  {
577  SendNotification(GetAcoreString(LANG_CHANNEL_REQ), sWorld->getIntConfig(CONFIG_CHAT_CHANNEL_LEVEL_REQ));
578  return;
579  }
580  }
581 
582  if (ChannelMgr* cMgr = ChannelMgr::forTeam(sender->GetTeamId()))
583  {
584  if (Channel* chn = cMgr->GetChannel(channel, sender))
585  {
586  sScriptMgr->OnPlayerChat(sender, type, lang, msg, chn);
587 
588 #ifdef ELUNA
589  if (!sEluna->OnChat(sender, type, lang, msg, chn))
590  return;
591 #endif
592  chn->Say(sender->GetGUID(), msg.c_str(), lang);
593  }
594  }
595  }
596  break;
597  case CHAT_MSG_AFK:
598  {
599  if (!sender->IsInCombat())
600  {
601  if (sender->isAFK()) // Already AFK
602  {
603  if (msg.empty())
604  sender->ToggleAFK(); // Remove AFK
605  else
606  sender->autoReplyMsg = msg; // Update message
607  }
608  else // New AFK mode
609  {
610  sender->autoReplyMsg = msg.empty() ? GetAcoreString(LANG_PLAYER_AFK_DEFAULT) : msg;
611 
612  if (sender->isDND())
613  sender->ToggleDND();
614 
615  sender->ToggleAFK();
616  }
617 
618  sScriptMgr->OnPlayerChat(sender, type, lang, msg);
619 #ifdef ELUNA
620  if (!sEluna->OnChat(sender, type, lang, msg))
621  return;
622 #endif
623  }
624  break;
625  }
626  case CHAT_MSG_DND:
627  {
628  if (sender->isDND()) // Already DND
629  {
630  if (msg.empty())
631  sender->ToggleDND(); // Remove DND
632  else
633  sender->autoReplyMsg = msg; // Update message
634  }
635  else // New DND mode
636  {
637  sender->autoReplyMsg = msg.empty() ? GetAcoreString(LANG_PLAYER_DND_DEFAULT) : msg;
638 
639  if (sender->isAFK())
640  sender->ToggleAFK();
641 
642  sender->ToggleDND();
643  }
644 
645  sScriptMgr->OnPlayerChat(sender, type, lang, msg);
646 #ifdef ELUNA
647  if (!sEluna->OnChat(sender, type, lang, msg))
648  return;
649 #endif
650  break;
651  }
652  default:
653  LOG_ERROR("network.opcode", "CHAT: unknown message type %u, lang: %u", type, lang);
654  break;
655  }

658 void WorldSession::HandleEmoteOpcode(WorldPacket& recvData)
659 {
660  if (!GetPlayer()->IsAlive() || GetPlayer()->HasUnitState(UNIT_STATE_DIED))

662

664  return;

709 void WorldSession::HandleTextEmoteOpcode(WorldPacket& recvData)
710 {
711  if (!GetPlayer()->IsAlive())
712  return;
713 
714  GetPlayer()->UpdateSpeakTime();
715 
716  if (!GetPlayer()->CanSpeak())
717  {
718  std::string timeStr = secsToTimeString(m_muteTime - time(nullptr));
719  SendNotification(GetAcoreString(LANG_WAIT_BEFORE_SPEAKING), timeStr.c_str());
720  return;
721  }
722 
723  if (GetPlayer()->IsSpectator())
724  return;
725 
726  uint32 text_emote, emoteNum;
727  ObjectGuid guid;
728 
729  recvData >> text_emote;
730  recvData >> emoteNum;
731  recvData >> guid;
732 
733  sScriptMgr->OnPlayerTextEmote(GetPlayer(), text_emote, emoteNum, guid);
734 
735  EmotesTextEntry const* em = sEmotesTextStore.LookupEntry(text_emote);
736  if (!em)
737  return;
738 
739  uint32 emote_anim = em->textid;
740 
741  switch (emote_anim)
742  {
743  case EMOTE_STATE_SLEEP:
744  case EMOTE_STATE_SIT:
745  case EMOTE_STATE_KNEEL:
746  case EMOTE_ONESHOT_NONE:
747  break;
748  default:
749  // Only allow text-emotes for "dead" entities (feign death included)
750  if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
751  break;
752  GetPlayer()->HandleEmoteCommand(emote_anim);
753  break;
754  }
755 
756  Unit* unit = ObjectAccessor::GetUnit(*_player, guid);
757 
758  CellCoord p = Acore::ComputeCellCoord(GetPlayer()->GetPositionX(), GetPlayer()->GetPositionY());
759 
760  Cell cell(p);
761  cell.SetNoCreate();
762 
763  Acore::EmoteChatBuilder emote_builder(*GetPlayer(), text_emote, emoteNum, unit);
764  Acore::LocalizedPacketDo<Acore::EmoteChatBuilder > emote_do(emote_builder);
765  Acore::PlayerDistWorker<Acore::LocalizedPacketDo<Acore::EmoteChatBuilder > > emote_worker(GetPlayer(), sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_TEXTEMOTE), emote_do);
766  TypeContainerVisitor<Acore::PlayerDistWorker<Acore::LocalizedPacketDo<Acore::EmoteChatBuilder> >, WorldTypeMapContainer> message(emote_worker);
767  cell.Visit(p, message, *GetPlayer()->GetMap(), *GetPlayer(), sWorld->getFloatConfig(CONFIG_LISTEN_RANGE_TEXTEMOTE));
768 
769  GetPlayer()->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_DO_EMOTE, text_emote, 0, unit);
770 
771  //Send scripted event call
772  if (unit && unit->GetTypeId() == TYPEID_UNIT && ((Creature*)unit)->AI())
773  ((Creature*)unit)->AI()->ReceiveEmote(GetPlayer(), text_emote);
774 }
