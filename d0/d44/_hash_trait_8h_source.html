36  const char* data = (const char*)_data;

40

42  return 0;
43  }
44 
45  rem = numBytes & 3;
46  numBytes >>= 2;

51  tmp = (get16bits (data+2) << 11) ^ hash;
52  hash = (hash << 16) ^ tmp;
