49 void BattlegroundAB::PostUpdateImpl(uint32 diff)
50 {
51  if (GetStatus() == STATUS_IN_PROGRESS)
52  {
53  _bgEvents.Update(diff);
54  while (uint32 eventId = _bgEvents.ExecuteEvent())
55  switch (eventId)
56  {
57  case BG_AB_EVENT_UPDATE_BANNER_STABLE:
58  case BG_AB_EVENT_UPDATE_BANNER_FARM:
59  case BG_AB_EVENT_UPDATE_BANNER_BLACKSMITH:
60  case BG_AB_EVENT_UPDATE_BANNER_LUMBERMILL:
61  case BG_AB_EVENT_UPDATE_BANNER_GOLDMINE:
62  CreateBanner(eventId - BG_AB_EVENT_UPDATE_BANNER_STABLE, false);
63  break;
64  case BG_AB_EVENT_CAPTURE_STABLE:
65  case BG_AB_EVENT_CAPTURE_FARM:
66  case BG_AB_EVENT_CAPTURE_BLACKSMITH:
67  case BG_AB_EVENT_CAPTURE_LUMBERMILL:
68  case BG_AB_EVENT_CAPTURE_GOLDMINE:
69  {
70  uint8 node = eventId - BG_AB_EVENT_CAPTURE_STABLE;
71  TeamId teamId = _capturePointInfo[node]._state == BG_AB_NODE_STATE_ALLY_CONTESTED ? TEAM_ALLIANCE : TEAM_HORDE;
72  DeleteBanner(node);
73  _capturePointInfo[node]._ownerTeamId = teamId;
74  _capturePointInfo[node]._state = teamId == TEAM_ALLIANCE ? BG_AB_NODE_STATE_ALLY_OCCUPIED : BG_AB_NODE_STATE_HORDE_OCCUPIED;
75  _capturePointInfo[node]._captured = true;
76 
77  CreateBanner(node, false);
78  NodeOccupied(node);
79  SendNodeUpdate(node);
80 
81  SendMessage2ToAll(LANG_BG_AB_NODE_TAKEN, teamId == TEAM_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE, nullptr, teamId == TEAM_ALLIANCE ? LANG_BG_AB_ALLY : LANG_BG_AB_HORDE, LANG_BG_AB_NODE_STABLES + node);
82  PlaySoundToAll(teamId == TEAM_ALLIANCE ? BG_AB_SOUND_NODE_CAPTURED_ALLIANCE : BG_AB_SOUND_NODE_CAPTURED_HORDE);
83  break;
84  }
85  case BG_AB_EVENT_ALLIANCE_TICK:
86  case BG_AB_EVENT_HORDE_TICK:
87  {
88  auto teamId = TeamId(eventId - BG_AB_EVENT_ALLIANCE_TICK);
89  uint8 controlledPoints = _controlledPoints[teamId];
90  if (controlledPoints == 0)
91  {
92  _bgEvents.ScheduleEvent(eventId, 3000);
93  break;
94  }
95 
96  auto honorRewards = uint8(m_TeamScores[teamId] / _honorTics);
97  auto reputationRewards = uint8(m_TeamScores[teamId] / _reputationTics);
98  auto information = uint8(m_TeamScores[teamId] / BG_AB_WARNING_NEAR_VICTORY_SCORE);
99  m_TeamScores[teamId] += BG_AB_TickPoints[controlledPoints];
100  if (m_TeamScores[teamId] > BG_AB_MAX_TEAM_SCORE)
101  m_TeamScores[teamId] = BG_AB_MAX_TEAM_SCORE;
102 
103  if (honorRewards < uint8(m_TeamScores[teamId] / _honorTics))
104  RewardHonorToTeam(GetBonusHonorFromKill(1), teamId);
105  if (reputationRewards < uint8(m_TeamScores[teamId] / _reputationTics))
106  RewardReputationToTeam(teamId == TEAM_ALLIANCE ? 509 : 510, 10, teamId);
107  if (information < uint8(m_TeamScores[teamId] / BG_AB_WARNING_NEAR_VICTORY_SCORE))
108  {
109  if (teamId == TEAM_ALLIANCE)
110  {
111  SendMessageToAll(LANG_BG_AB_A_NEAR_VICTORY, CHAT_MSG_BG_SYSTEM_NEUTRAL);
112  PlaySoundToAll(BG_AB_SOUND_NEAR_VICTORY_ALLIANCE);
113  }
114  else
115  {
116  SendMessageToAll(LANG_BG_AB_H_NEAR_VICTORY, CHAT_MSG_BG_SYSTEM_NEUTRAL);
117  PlaySoundToAll(BG_AB_SOUND_NEAR_VICTORY_HORDE);
118  }
119  }
120 
121  UpdateWorldState(teamId == TEAM_ALLIANCE ? BG_AB_OP_RESOURCES_ALLY : BG_AB_OP_RESOURCES_HORDE, m_TeamScores[teamId]);
122  if (m_TeamScores[teamId] > m_TeamScores[GetOtherTeamId(teamId)] + 500)
123  _teamScores500Disadvantage[GetOtherTeamId(teamId)] = true;
124  if (m_TeamScores[teamId] >= BG_AB_MAX_TEAM_SCORE)
125  EndBattleground(teamId);
126 
127  _bgEvents.ScheduleEvent(eventId, BG_AB_TickIntervals[controlledPoints]);
128  break;
129  }
130  default:
131  break;
132  }
133  }

290 void BattlegroundAB::EventPlayerClickedOnFlag(Player* player, GameObject* gameObject)
291 {
292  if (GetStatus() != STATUS_IN_PROGRESS || !player->IsWithinDistInMap(gameObject, 10.0f))
293  return;
294 
295  uint8 node = BG_AB_NODE_STABLES;
296  for (; node < BG_AB_DYNAMIC_NODES_COUNT; ++node)
297  if (player->GetDistance2d(BG_AB_NodePositions[node][0], BG_AB_NodePositions[node][1]) < 10.0f)
298  break;
299 
300  if (node == BG_AB_DYNAMIC_NODES_COUNT || _capturePointInfo[node]._ownerTeamId == player->GetTeamId() ||
301  (_capturePointInfo[node]._state == BG_AB_NODE_STATE_ALLY_CONTESTED && player->GetTeamId() == TEAM_ALLIANCE) ||
302  (_capturePointInfo[node]._state == BG_AB_NODE_STATE_HORDE_CONTESTED && player->GetTeamId() == TEAM_HORDE))
303  return;
304 
305  player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_ENTER_PVP_COMBAT);
306 
307  uint32 sound = 0;
308  uint32 message = 0;
309  uint32 message2 = 0;
310  DeleteBanner(node);
311  CreateBanner(node, true);
312 
313  if (_capturePointInfo[node]._state == BG_AB_NODE_STATE_NEUTRAL)
314  {
315  player->KilledMonsterCredit(BG_AB_QUEST_CREDIT_BASE + node);
316  UpdatePlayerScore(player, SCORE_BASES_ASSAULTED, 1);
317  _capturePointInfo[node]._state = static_cast<uint8>(BG_AB_NODE_STATE_ALLY_CONTESTED) + player->GetTeamId();
318  _capturePointInfo[node]._ownerTeamId = TEAM_NEUTRAL;
319  _bgEvents.RescheduleEvent(BG_AB_EVENT_CAPTURE_STABLE + node, BG_AB_FLAG_CAPTURING_TIME);
320  sound = BG_AB_SOUND_NODE_CLAIMED;
321  message = LANG_BG_AB_NODE_CLAIMED;
322  message2 = player->GetTeamId() == TEAM_ALLIANCE ? LANG_BG_AB_ALLY : LANG_BG_AB_HORDE;
323  }
324  else if (_capturePointInfo[node]._state == BG_AB_NODE_STATE_ALLY_CONTESTED || _capturePointInfo[node]._state == BG_AB_NODE_STATE_HORDE_CONTESTED)
325  {
326  if (!_capturePointInfo[node]._captured)
327  {
328  player->KilledMonsterCredit(BG_AB_QUEST_CREDIT_BASE + node);
329  UpdatePlayerScore(player, SCORE_BASES_ASSAULTED, 1);
330  _capturePointInfo[node]._state = static_cast<uint8>(BG_AB_NODE_STATE_ALLY_CONTESTED) + player->GetTeamId();
331  _capturePointInfo[node]._ownerTeamId = TEAM_NEUTRAL;
332  _bgEvents.RescheduleEvent(BG_AB_EVENT_CAPTURE_STABLE + node, BG_AB_FLAG_CAPTURING_TIME);
333  message = LANG_BG_AB_NODE_ASSAULTED;
334  }
335  else
336  {
337  UpdatePlayerScore(player, SCORE_BASES_DEFENDED, 1);
338  _capturePointInfo[node]._state = static_cast<uint8>(BG_AB_NODE_STATE_ALLY_OCCUPIED) + player->GetTeamId();
339  _capturePointInfo[node]._ownerTeamId = player->GetTeamId();
340  _bgEvents.CancelEvent(BG_AB_EVENT_CAPTURE_STABLE + node);
341  NodeOccupied(node); // after setting team owner
342  message = LANG_BG_AB_NODE_DEFENDED;
343  }
344  sound = player->GetTeamId() == TEAM_ALLIANCE ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
345  }
346  else

350  NodeDeoccupied(node); // before setting team owner to neutral

353

356  message = LANG_BG_AB_NODE_ASSAULTED;
357  sound = player->GetTeamId() == TEAM_ALLIANCE ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
358  }

361  PlaySoundToAll(sound);
