241 
242  // Last remaining add

244  {
245  Talk(SAY_LAST_ADD);

253  else if (!remainingAdds)

263  }
264 
265  static_minionsGUIDS.clear();
266  }
267 
268  instance->SetBossState(DATA_MAJORDOMO_EXECUTUS, DONE);
269  events.CancelEventGroup(PHASE_COMBAT);
270  me->GetMap()->UpdateEncounterState(ENCOUNTER_CREDIT_KILL_CREATURE, me->GetEntry(), me);
271  me->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC|UNIT_FLAG_IMMUNE_TO_NPC);

300  void UpdateAI(uint32 diff) override
301  {
302  if (spawnInTextTimer)
303  {
304  if (spawnInTextTimer <= diff)
305  {
306  spawnInTextTimer = 0;
307  Talk(SAY_SPAWN);
308  }
309  else
310  {
311  spawnInTextTimer -= diff;
312  }
313  }
314 
315  switch (events.GetPhaseMask())
316  {
317  case (1 << (PHASE_COMBAT - 1)):
318  {
319  if (!UpdateVictim())
320  {
321  return;
322  }
323 
324  events.Update(diff);
325 
326  if (me->HasUnitState(UNIT_STATE_CASTING))
327  {
328  return;
329  }
330 
331  while (uint32 const eventId = events.ExecuteEvent())
332  {
333  switch (eventId)
334  {
335  case EVENT_MAGIC_REFLECTION:
336  {
337  DoCastSelf(SPELL_MAGIC_REFLECTION);
338  events.RepeatEvent(30000);
339  break;
340  }
341  case EVENT_DAMAGE_REFLECTION:
342  {
343  DoCastSelf(SPELL_DAMAGE_REFLECTION);
344  events.RepeatEvent(30000);
345  break;
346  }
347  case EVENT_BLAST_WAVE:
348  {
349  DoCastVictim(SPELL_BLAST_WAVE);
350  events.RepeatEvent(10000);
351  break;
352  }
353  case EVENT_TELEPORT_RANDOM:
354  {
355  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 1, 0.0f, true))
356  {
357  DoCastSelf(SPELL_HATE_TO_ZERO, true);
358  DoCast(target, SPELL_TELEPORT_RANDOM);
359  }
360 
361  events.RepeatEvent(15000);
362  break;
363  }
364  case EVENT_TELEPORT_TARGET:
365  {
366  DoCastSelf(SPELL_HATE_TO_ZERO, true);
367  DoCastAOE(SPELL_TELEPORT_TARGET);
368  events.RepeatEvent(30000);
369  break;
370  }
371  }
372 
373  if (me->HasUnitState(UNIT_STATE_CASTING))
374  {
375  return;
376  }
377  }
378 
379  DoMeleeAttackIfReady();
380  break;
381  }
382  case (1 << (PHASE_DEFEAT_OUTRO - 1)):
383  {
384  events.Update(diff);
385  while (uint32 const eventId = events.ExecuteEvent())
386  {
387  switch (eventId)
388  {
389  case EVENT_DEFEAT_OUTRO_1:
390  {
391  Talk(SAY_DEFEAT_2);
392  events.ScheduleEvent(EVENT_DEFEAT_OUTRO_2, 8000, PHASE_DEFEAT_OUTRO, PHASE_DEFEAT_OUTRO);
393  break;
394  }
395  case EVENT_DEFEAT_OUTRO_2:
396  {
397  Talk(SAY_DEFEAT_3);
398  events.ScheduleEvent(EVENT_DEFEAT_OUTRO_3, 21500, PHASE_DEFEAT_OUTRO, PHASE_DEFEAT_OUTRO);
399  break;
400  }
401  case EVENT_DEFEAT_OUTRO_3:
402  {
403  DoCastSelf(SPELL_TELEPORT_SELF);
404  break;
405  }
406  }
407  }
408  break;
409  }
410  case (1 << (PHASE_RAGNAROS_SUMMONING - 1)):
411  {
412  events.Update(diff);
413  while (uint32 const eventId = events.ExecuteEvent())
414  {
415  switch (eventId)
416  {
417  case EVENT_RAGNAROS_SUMMON_1:
418  {
419  if (GameObject* lavaSplash = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_SPLASH)))
420  {
421  lavaSplash->SetRespawnTime(900);
422  lavaSplash->Refresh();
423  }
424  if (GameObject* lavaSteam = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_STEAM)))
425  {
426  lavaSteam->SetRespawnTime(900);
427  lavaSteam->Refresh();
428  }
429  Talk(SAY_RAG_SUM_2);
430  // Next event will get triggered in MovementInform
431  me->SetWalk(true);
432  me->GetMotionMaster()->MovePoint(POINT_RAGNAROS_SUMMON, MajordomoMoveRagPos, true, false);
433  break;
434  }
435  case EVENT_RAGNAROS_SUMMON_2:
436  {
437  if (GameObject* lavaSteam = ObjectAccessor::GetGameObject(*me, instance->GetGuidData(DATA_LAVA_STEAM)))
438  {
439  me->SetFacingToObject(lavaSteam);
440  }
441 
442  Talk(SAY_SUMMON_MAJ);
443  events.ScheduleEvent(EVENT_RAGNAROS_SUMMON_3, 16700, PHASE_RAGNAROS_SUMMONING, PHASE_RAGNAROS_SUMMONING);
