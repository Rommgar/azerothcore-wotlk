92 /*
93  my_yield_processor (equivalent of x86 PAUSE instruction) should be used
94  to improve performance on hyperthreaded CPUs. Intel recommends to use it in
95  spin loops also on non-HT machines to reduce power consumption (see e.g
96  http://softwarecommunity.intel.com/articles/eng/2004.htm)
97 
98  Running benchmarks for spinlocks implemented with InterlockedCompareExchange
99  and YieldProcessor shows that much better performance is achieved by calling
100  YieldProcessor in a loop - that is, yielding longer. On Intel boxes setting
101  loop count in the range 200-300 brought best results.
102  */
103 #ifndef YIELD_LOOPS
104 #define YIELD_LOOPS 200
105 #endif

118  }
