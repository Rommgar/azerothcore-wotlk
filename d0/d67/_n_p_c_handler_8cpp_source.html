124 void WorldSession::SendTrainerList(ObjectGuid guid, const std::string& strTitle)
125 {
126  LOG_DEBUG("network", "WORLD: SendTrainerList");
127 
128  Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
129  if (!unit)
130  {
131  LOG_DEBUG("network", "WORLD: SendTrainerList - Unit (%s) not found or you can not interact with him.", guid.ToString().c_str());
132  return;
133  }
134 
135  // remove fake death
136  if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
137  GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
138 
139  CreatureTemplate const* ci = unit->GetCreatureTemplate();
140 
141  if (!ci)
142  {
143  LOG_DEBUG("network", "WORLD: SendTrainerList - (%s) NO CREATUREINFO!", guid.ToString().c_str());
144  return;
145  }
146 
147  TrainerSpellData const* trainer_spells = unit->GetTrainerSpells();
148  if (!trainer_spells)
149  {
150  LOG_DEBUG("network", "WORLD: SendTrainerList - Training spells not found for creature (%s)", guid.ToString().c_str());
151  return;
152  }
153 
154  WorldPacket data(SMSG_TRAINER_LIST, 8 + 4 + 4 + trainer_spells->spellList.size() * 38 + strTitle.size() + 1);
155  data << guid;
156  data << uint32(trainer_spells->trainerType);
157 
158  size_t count_pos = data.wpos();
159  data << uint32(trainer_spells->spellList.size());
160 
161  // reputation discount
162  float fDiscountMod = _player->GetReputationPriceDiscount(unit);
163  bool can_learn_primary_prof = GetPlayer()->GetFreePrimaryProfessionPoints() > 0;
164 
165  uint32 count = 0;
166  for (TrainerSpellMap::const_iterator itr = trainer_spells->spellList.begin(); itr != trainer_spells->spellList.end(); ++itr)
167  {
168  TrainerSpell const* tSpell = &itr->second;
169 
170  bool valid = true;
171  bool primary_prof_first_rank = false;
172  for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
173  {
174  if (!tSpell->learnedSpell[i])
175  continue;
176  if (!_player->IsSpellFitByClassAndRace(tSpell->learnedSpell[i]))
177  {
178  valid = false;
179  break;
180  }
181  SpellInfo const* learnedSpellInfo = sSpellMgr->GetSpellInfo(tSpell->learnedSpell[i]);
182  if (learnedSpellInfo && learnedSpellInfo->IsPrimaryProfessionFirstRank())
183  primary_prof_first_rank = true;
184  }
185  if (!valid)
186  continue;
187 
188  TrainerSpellState state = _player->GetTrainerSpellState(tSpell);
189 
190  data << uint32(tSpell->spell); // learned spell (or cast-spell in profession case)
191  data << uint8(state == TRAINER_SPELL_GREEN_DISABLED ? TRAINER_SPELL_GREEN : state);
192  data << uint32(floor(tSpell->spellCost * fDiscountMod));
193 
194  data << uint32(primary_prof_first_rank && can_learn_primary_prof ? 1 : 0);
195  // primary prof. learn confirmation dialog
196  data << uint32(primary_prof_first_rank ? 1 : 0); // must be equal prev. field to have learn button in enabled state
197  data << uint8(tSpell->reqLevel);
198  data << uint32(tSpell->reqSkill);
199  data << uint32(tSpell->reqSkillValue);
200  //prev + req or req + 0
201  uint8 maxReq = 0;
202  for (uint8 i = 0; i < MAX_SPELL_EFFECTS; ++i)
203  {
204  if (!tSpell->learnedSpell[i])
205  continue;
206  if (uint32 prevSpellId = sSpellMgr->GetPrevSpellInChain(tSpell->learnedSpell[i]))
207  {
208  data << uint32(prevSpellId);
209  ++maxReq;
210  }
211  if (maxReq == 3)
212  break;
213  SpellsRequiringSpellMapBounds spellsRequired = sSpellMgr->GetSpellsRequiredForSpellBounds(tSpell->learnedSpell[i]);
214  for (SpellsRequiringSpellMap::const_iterator itr2 = spellsRequired.first; itr2 != spellsRequired.second && maxReq < 3; ++itr2)
215  {
216  data << uint32(itr2->second);
217  ++maxReq;
218  }
219  if (maxReq == 3)
220  break;
221  }
222  while (maxReq < 3)
223  {
224  data << uint32(0);
225  ++maxReq;
226  }
227 
228  ++count;
229  }
230 
231  data << strTitle;
232 
233  data.put<uint32>(count_pos, count);
234  SendPacket(&data);
235 }

237 void WorldSession::HandleTrainerBuySpellOpcode(WorldPacket& recvData)
238 {
239  ObjectGuid guid;
240  uint32 spellId = 0;
241 
242  recvData >> guid >> spellId;
243  LOG_DEBUG("network", "WORLD: Received CMSG_TRAINER_BUY_SPELL Npc %s, learn spell id is: %u", guid.ToString().c_str(), spellId);
244 
245  Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
246  if (!unit)
247  {
248  LOG_DEBUG("network", "WORLD: HandleTrainerBuySpellOpcode - Unit (%s) not found or you can not interact with him.", guid.ToString().c_str());
249  return;
250  }
251 
252  // remove fake death
253  if (GetPlayer()->HasUnitState(UNIT_STATE_DIED))
254  GetPlayer()->RemoveAurasByType(SPELL_AURA_FEIGN_DEATH);
255 
256  // check present spell in trainer spell list
257  TrainerSpellData const* trainer_spells = unit->GetTrainerSpells();
258  if (!trainer_spells)
259  return;
260 
261  // not found, cheat?
262  TrainerSpell const* trainer_spell = trainer_spells->Find(spellId);
263  if (!trainer_spell)
264  return;
265 
266  // can't be learn, cheat? Or double learn with lags...
267  if (_player->GetTrainerSpellState(trainer_spell) != TRAINER_SPELL_GREEN)
268  return;
269 
270  // apply reputation discount
271  uint32 nSpellCost = uint32(floor(trainer_spell->spellCost * _player->GetReputationPriceDiscount(unit)));
272 
273  // check money requirement
274  if (!_player->HasEnoughMoney(nSpellCost))
275  return;
276 
277  _player->ModifyMoney(-int32(nSpellCost));
278 
279  unit->SendPlaySpellVisual(179); // 53 SpellCastDirected
280  unit->SendPlaySpellImpact(_player->GetGUID(), 362); // 113 EmoteSalute
281 
282  // learn explicitly or cast explicitly
283  if (trainer_spell->IsCastable())
284  _player->CastSpell(_player, trainer_spell->spell, true);
285  else
286  _player->learnSpell(spellId);
287 
288  WorldPacket data(SMSG_TRAINER_BUY_SUCCEEDED, 12);
289  data << guid;
290  data << uint32(spellId); // should be same as in packet from client
291  SendPacket(&data);
292 }
