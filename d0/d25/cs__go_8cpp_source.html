65  static bool DoTeleport(ChatHandler* handler, Position pos, uint32 mapId = MAPID_INVALID)
66  {
67  Player* player = handler->GetSession()->GetPlayer();
68 
69  if (mapId == MAPID_INVALID)
70  mapId = player->GetMapId();
71  if (!MapMgr::IsValidMapCoord(mapId, pos) || sObjectMgr->IsTransportMap(mapId))
72  {
73  handler->PSendSysMessage(LANG_INVALID_TARGET_COORD, pos.GetPositionX(), pos.GetPositionY(), mapId);
74  handler->SetSentErrorMessage(true);
75  return false;
76  }
77 
78  // stop flight if need
79  if (player->IsInFlight())
80  {
81  player->GetMotionMaster()->MovementExpired();
82  player->CleanupAfterTaxiFlight();
83  }
84  // save only in non-flight case
85  else
86  player->SaveRecallPosition();
87 
88  player->TeleportTo({ mapId, pos });
89  return true;
90  }
