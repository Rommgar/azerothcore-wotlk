102 void wait_for_free_space(const char *filename, int errors)
103 {
104  if (errors == 0)
105  my_error(EE_DISK_FULL,MYF(ME_BELL | ME_NOREFRESH),
106  filename,my_errno,MY_WAIT_FOR_USER_TO_FIX_PANIC);
107  if (!(errors % MY_WAIT_GIVE_USER_A_MESSAGE))
108  my_printf_error(EE_DISK_FULL,
109  "Retry in %d secs. Message reprinted in %d secs",
110  MYF(ME_BELL | ME_NOREFRESH),
111  MY_WAIT_FOR_USER_TO_FIX_PANIC,
112  MY_WAIT_GIVE_USER_A_MESSAGE * MY_WAIT_FOR_USER_TO_FIX_PANIC );
113  (void) sleep(MY_WAIT_FOR_USER_TO_FIX_PANIC);
114 }
