40 static int compare_bin(uchar *a, uint a_length, uchar *b, uint b_length,
41  my_bool part_key, my_bool skip_end_space)
42 {
43  uint length= min(a_length,b_length);
44  uchar *end= a+ length;
45  int flag;
46 
47  while (a < end)
48  if ((flag= (int) *a++ - (int) *b++))
49  return flag;
50  if (part_key && b_length < a_length)
51  return 0;
52  if (skip_end_space && a_length != b_length)
53  {
54  int swap= 1;
55  /*
56  We are using space compression. We have to check if longer key
57  has next character < ' ', in which case it's less than the shorter
58  key that has an implicite space afterwards.
59 
60  This code is identical to the one in
61  strings/ctype-simple.c:my_strnncollsp_simple
62  */
63  if (a_length < b_length)
64  {
65  /* put shorter key in a */
66  a_length= b_length;
67  a= b;
68  swap= -1; /* swap sign of result */
69  }
70  for (end= a + a_length-length; a < end ; a++)
71  {
72  if (*a != ' ')
73  return (*a < ' ') ? -swap : swap;
74  }
75  return 0;
76  }
77  return (int) (a_length-b_length);
78 }

86  keyseg Array of key segments of key to compare
87  a First key to compare, in format from _mi_pack_key()
88  This is normally key specified by user

101  Example1: if the function is called for tuples
102  ('aaa','bbb') and ('eee','fff'), then
