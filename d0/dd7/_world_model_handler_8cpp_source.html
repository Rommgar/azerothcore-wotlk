28 WorldModelDefinition WorldModelDefinition::Read( FILE* file )
29 {
30  WorldModelDefinition ret;
31  int count = 0;
32  count += fread(&ret.MwidIndex, sizeof(uint32), 1, file);
33  count += fread(&ret.UniqueId, sizeof(uint32), 1, file);
34  ret.Position = Vector3::Read(file);

36  ret.UpperExtents = Vector3::Read(file);

39  count += fread(&ret.DoodadSet, sizeof(uint16), 1, file);

42 
43  if (count != 5)
44  printf("WorldModelDefinition::Read: Error reading data, expected 5, read %d\n", count);

54 void WorldModelHandler::ProcessInternal( MapChunk* mcnk )
55 {
56  if (!IsSane())
57  return;
58 
59  uint32 refCount = mcnk->Header.MapObjectRefs;
60  FILE* stream = mcnk->Source->GetStream();
61  fseek(stream, mcnk->Source->Offset + mcnk->Header.OffsetMCRF, SEEK_SET);
62  // Start looping at the last Doodad Ref index
63  for (uint32 i = mcnk->Header.DoodadRefs; i < refCount; i++)
64  {
65  int32 index;
66  if (fread(&index, sizeof(int32), 1, stream) != 1)
67  printf("WorldModelDefinition::Read: Error reading data, expected 1, read 0\n");
68 
69  if (index < 0 || uint32(index) >= _definitions->size())
70  continue;
71 
72  WorldModelDefinition wmo = (*_definitions)[index];
73 
74  if (_drawn.find(wmo.UniqueId) != _drawn.end())
75  continue;
76  _drawn.insert(wmo.UniqueId);
77 
78  if (wmo.MwidIndex >= _paths->size())
79  continue;
80 
81  std::string path = (*_paths)[wmo.MwidIndex];
82  WorldModelRoot* model = Cache->WorldModelCache.Get(path);
83  if (!model)
84  {
85  model = new WorldModelRoot(path);
86  Cache->WorldModelCache.Insert(path, model);
87  }
88 
89  Vertices.reserve(1000);
90  Triangles.reserve(1000);
91 
92  InsertModelGeometry(Vertices, Triangles, wmo, model);
93  }
94  // Restore the stream position
95  fseek(stream, mcnk->Source->Offset, SEEK_SET);
96 }

98 void WorldModelHandler::InsertModelGeometry( std::vector<Vector3>& verts, std::vector<Triangle<uint32>>& tris, const WorldModelDefinition& def, WorldModelRoot* root, bool translate )
99 {
100  for (std::vector<WorldModelGroup>::iterator group = root->Groups.begin(); group != root->Groups.end(); ++group)
101  {
102  uint32 vertOffset = verts.size();
103  for (std::vector<Vector3>::iterator itr2 = group->Vertices.begin(); itr2 != group->Vertices.end(); ++itr2)
104  {
105  Vector3 v = Utils::TransformDoodadVertex(def, *itr2, translate);
106  // If translate is false, then we were called directly from the TileBuilder to add data to it's _Geometry member, hence, we have to manually convert the vertices to Recast format.
107  verts.push_back(translate ? v : Utils::ToRecast(v)); // Transform the vertex to world space
108  }
109 
110  for (uint32 i = 0; i < group->Triangles.size(); ++i)
111  {
112  // only include colliding tris
113  if ((group->TriangleFlags[i] & 0x04) != 0 && group->TriangleMaterials[i] != 0xFF)
114  continue;
115  Triangle<uint16> tri = group->Triangles[i];
116  tris.push_back(Triangle<uint32>(Constants::TRIANGLE_TYPE_WMO, tri.V0 + vertOffset, tri.V1 + vertOffset, tri.V2 + vertOffset));
117  }
118  }
119 
120  if (def.DoodadSet < root->DoodadSets.size())
121  {
122  DoodadSet set = root->DoodadSets[def.DoodadSet];
123  std::vector<DoodadInstance> instances;
124  instances.reserve(set.CountInstances);
125  for (uint32 i = set.FirstInstanceIndex; i < (set.CountInstances + set.FirstInstanceIndex); i++)
126  {
127  if (i >= root->DoodadInstances.size())
128  break;
129  instances.push_back(root->DoodadInstances[i]);
130  }
131 
132  for (std::vector<DoodadInstance>::iterator instance = instances.begin(); instance != instances.end(); ++instance)
133  {
134  Model* model = Cache->ModelCache.Get(instance->File);
135  if (!model)
136  {
137  model = new Model(instance->File);
138  Cache->ModelCache.Insert(instance->File, model);
139  }
140 
141  if (!model->IsCollidable)
142  continue;
143  int vertOffset = verts.size();
144  for (std::vector<Vector3>::iterator itr2 = model->Vertices.begin(); itr2 != model->Vertices.end(); ++itr2)
145  {
146  Vector3 v = Utils::TransformDoodadVertex(def, Utils::TransformWmoDoodad(*instance, def, *itr2, false), translate);
147  verts.push_back(translate ? v : Utils::ToRecast(v));
148  }
149  for (std::vector<Triangle<uint16>>::iterator itr2 = model->Triangles.begin(); itr2 != model->Triangles.end(); ++itr2)
150  tris.push_back(Triangle<uint32>(Constants::TRIANGLE_TYPE_WMO, itr2->V0 + vertOffset, itr2->V1 + vertOffset, itr2->V2 + vertOffset));
151  }
152 
153  for (std::vector<WorldModelGroup>::iterator group = root->Groups.begin(); group != root->Groups.end(); ++group)
154  {
155  if (!group->HasLiquidData)
156  continue;
157 
158  const LiquidHeader& liquidHeader = group->LiquidDataHeader;
159  LiquidData& liquidDataGeometry = group->LiquidDataGeometry;
160 
161  for (uint32 y = 0; y < liquidHeader.Height; y++)
162  {
163  for (uint32 x = 0; x < liquidHeader.Width; x++)
164  {
165  if (!liquidDataGeometry.ShouldRender(x, y))
166  continue;
167 
168  uint32 vertOffset = verts.size();
169 
170  Vector3 v1 = Utils::GetLiquidVert(def, liquidHeader.BaseLocation,
171  liquidDataGeometry.HeightMap[x][y], x, y, translate);
172  Vector3 v2 = Utils::GetLiquidVert(def, liquidHeader.BaseLocation,
173  liquidDataGeometry.HeightMap[x + 1][y], x + 1, y, translate);
174  Vector3 v3 = Utils::GetLiquidVert(def, liquidHeader.BaseLocation,
175  liquidDataGeometry.HeightMap[x][y + 1], x, y + 1, translate);
176  Vector3 v4 = Utils::GetLiquidVert(def, liquidHeader.BaseLocation,
177  liquidDataGeometry.HeightMap[x + 1][y + 1], x + 1, y + 1, translate);
178 
179  verts.push_back(translate ? v1 : Utils::ToRecast(v1));
180  verts.push_back(translate ? v2 : Utils::ToRecast(v2));
181  verts.push_back(translate ? v3 : Utils::ToRecast(v3));
182  verts.push_back(translate ? v4 : Utils::ToRecast(v4));
183 
184  tris.push_back(Triangle<uint32>(Constants::TRIANGLE_TYPE_WATER, vertOffset, vertOffset + 2, vertOffset + 1));
185  tris.push_back(Triangle<uint32>(Constants::TRIANGLE_TYPE_WATER, vertOffset + 2, vertOffset + 3, vertOffset + 1));
186  }
187  }
188  }
189  }

192 void WorldModelHandler::ReadDefinitions()
193 {
194  Chunk* chunk = Source->ObjectData->GetChunkByName("MODF");
195  if (!chunk)
196  return;
197 
198  const int32 definitionSize = 64;
199  uint32 definitionCount = chunk->Length / definitionSize;
200  _definitions = new std::vector<WorldModelDefinition>;
201  _definitions->reserve(definitionCount);
202  FILE* stream = chunk->GetStream();
203  for (uint32 i = 0; i < definitionCount; i++)
204  _definitions->push_back(WorldModelDefinition::Read(stream));

207 void WorldModelHandler::ReadModelPaths()
208 {
209  Chunk* mwid = Source->ObjectData->GetChunkByName("MWID");
210  Chunk* mwmo = Source->ObjectData->GetChunkByName("MWMO");
211  if (!mwid || !mwmo)
212  return;
213 
214  uint32 paths = mwid->Length / 4;
215  _paths = new std::vector<std::string>;
216  _paths->reserve(paths);
217  for (uint32 i = 0; i < paths; i++)
218  {
219  FILE* stream = mwid->GetStream();
220  fseek(stream, i * 4, SEEK_CUR);
221  uint32 offset;
222  if (fread(&offset, sizeof(uint32), 1, stream) != 1)
223  printf("WorldModelDefinition::Read: Error reading data, expected 1, read 0\n");
224  FILE* dataStream = mwmo->GetStream();
225  fseek(dataStream, offset + mwmo->Offset, SEEK_SET);
226  _paths->push_back(Utils::ReadString(dataStream));
227  }
228 }
