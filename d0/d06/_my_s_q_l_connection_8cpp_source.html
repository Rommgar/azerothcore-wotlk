390 int MySQLConnection::ExecuteTransaction(std::shared_ptr<TransactionBase> transaction)
391 {
392  std::vector<SQLElementData> const& queries = transaction->m_queries;
393  if (queries.empty())
394  return -1;
395 
396  BeginTransaction();
397 
398  for (auto itr = queries.begin(); itr != queries.end(); ++itr)
399  {
400  SQLElementData const& data = *itr;
401  switch (itr->type)
402  {
403  case SQL_ELEMENT_PREPARED:
404  {
405  PreparedStatementBase* stmt = data.element.stmt;
406  ASSERT(stmt);
407  if (!Execute(stmt))
408  {
409  LOG_WARN("sql.sql", "Transaction aborted. %u queries not executed.", (uint32)queries.size());
410  int errorCode = GetLastError();
411  RollbackTransaction();
412  return errorCode;
413  }
414  }
415  break;
416  case SQL_ELEMENT_RAW:
417  {
418  char const* sql = data.element.query;
419  ASSERT(sql);
420  if (!Execute(sql))
421  {
422  LOG_WARN("sql.sql", "Transaction aborted. %u queries not executed.", (uint32)queries.size());
423  int errorCode = GetLastError();
424  RollbackTransaction();
425  return errorCode;
426  }
427  }
428  break;
429  }
430  }
431 
432  // we might encounter errors during certain queries, and depending on the kind of error
433  // we might want to restart the transaction. So to prevent data loss, we only clean up when it's all done.
434  // This is done in calling functions DatabaseWorkerPool<T>::DirectCommitTransaction and TransactionTask::Execute,
435  // and not while iterating over every element.
436 
437  CommitTransaction();
438  return 0;
439 }

484 {

486  // i.e. don't prepare async statements on synchronous connections
487  // to save memory that will not be used.

489  {
490  m_stmts[index].reset();
491  return;
492  }
493

532 bool MySQLConnection::_HandleMySQLErrno(uint32 errNo, uint8 attempts /*= 5*/)
533 {
534  switch (errNo)
535  {
536  case CR_SERVER_GONE_ERROR:
537  case CR_SERVER_LOST:
538  case CR_SERVER_LOST_EXTENDED:
539  {
540  if (m_Mysql)
541  {
542  LOG_ERROR("sql.sql", "Lost the connection to the MySQL server!");
543 
544  mysql_close(m_Mysql);
545  m_Mysql = nullptr;
546  }
547  [[fallthrough]];
548  }
549  case CR_CONN_HOST_ERROR:
550  {
551  LOG_INFO("sql.sql", "Attempting to reconnect to the MySQL server...");
552 
553  m_reconnecting = true;
554 
555  uint32 const lErrno = Open();
556  if (!lErrno)
557  {
558  // Don't remove 'this' pointer unless you want to skip loading all prepared statements...
559  if (!this->PrepareStatements())
560  {
561  LOG_FATAL("sql.sql", "Could not re-prepare statements!");
562  std::this_thread::sleep_for(std::chrono::seconds(10));
563  std::abort();
564  }
565 
566  LOG_INFO("sql.sql", "Successfully reconnected to %s @%s:%s (%s).",
567  m_connectionInfo.database.c_str(), m_connectionInfo.host.c_str(), m_connectionInfo.port_or_socket.c_str(),
568  (m_connectionFlags & CONNECTION_ASYNC) ? "asynchronous" : "synchronous");
569 
570  m_reconnecting = false;
571  return true;
572  }
573 
574  if ((--attempts) == 0)
575  {
576  // Shut down the server when the mysql server isn't
577  // reachable for some time
578  LOG_FATAL("sql.sql", "Failed to reconnect to the MySQL server, "
579  "terminating the server to prevent data corruption!");
580 
581  // We could also initiate a shutdown through using std::raise(SIGTERM)
582  std::this_thread::sleep_for(std::chrono::seconds(10));
583  std::abort();
584  }
585  else
586  {
587  // It's possible this attempted reconnect throws 2006 at us.
588  // To prevent crazy recursive calls, sleep here.
589  std::this_thread::sleep_for(std::chrono::seconds(3)); // Sleep 3 seconds
590  return _HandleMySQLErrno(lErrno, attempts); // Call self (recursive)
591  }
592  }
593 
594  case ER_LOCK_DEADLOCK:
595  return false; // Implemented in TransactionTask::Execute and DatabaseWorkerPool<T>::DirectCommitTransaction
596  // Query related errors - skip query
597  case ER_WRONG_VALUE_COUNT:
598  case ER_DUP_ENTRY:
599  return false;
600 
601  // Outdated table or database structure - terminate core
602  case ER_BAD_FIELD_ERROR:
603  case ER_NO_SUCH_TABLE:
604  LOG_ERROR("sql.sql", "Your database structure is not up to date. Please make sure you've executed all queries in the sql/updates folders.");
605  std::this_thread::sleep_for(std::chrono::seconds(10));
606  std::abort();
607  return false;
608  case ER_PARSE_ERROR:
609  LOG_ERROR("sql.sql", "Error while parsing SQL. Core fix required.");
610  std::this_thread::sleep_for(std::chrono::seconds(10));
611  std::abort();
612  return false;
613  default:
614  LOG_ERROR("sql.sql", "Unhandled MySQL errno %u. Unexpected behaviour possible.", errNo);
615  return false;
616  }
