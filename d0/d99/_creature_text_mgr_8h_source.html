120 template<class Builder>
121 class CreatureTextLocalizer
122 {
123 public:
124  CreatureTextLocalizer(Builder const& builder, ChatMsg msgType) : _builder(builder), _msgType(msgType)
125  {
126  _packetCache.resize(TOTAL_LOCALES, nullptr);
127  }
128 
129  ~CreatureTextLocalizer()
130  {
131  for (size_t i = 0; i < _packetCache.size(); ++i)
132  {
133  if (_packetCache[i])
134  delete _packetCache[i]->first;
135  delete _packetCache[i];
136  }
137  }
138 
139  void operator()(Player* player)
140  {
141  LocaleConstant loc_idx = player->GetSession()->GetSessionDbLocaleIndex();
142  WorldPacket* messageTemplate;
143  size_t whisperGUIDpos;
144 
145  // create if not cached yet
146  if (!_packetCache[loc_idx])
147  {
148  messageTemplate = new WorldPacket();
149  whisperGUIDpos = _builder(messageTemplate, loc_idx);
150  _packetCache[loc_idx] = new std::pair<WorldPacket*, size_t>(messageTemplate, whisperGUIDpos);
151  }
152  else
153  {
154  messageTemplate = _packetCache[loc_idx]->first;
155  whisperGUIDpos = _packetCache[loc_idx]->second;
156  }
157 
158  WorldPacket data(*messageTemplate);
159  switch (_msgType)
160  {
161  case CHAT_MSG_MONSTER_WHISPER:
162  case CHAT_MSG_RAID_BOSS_WHISPER:
163  data.put<uint64>(whisperGUIDpos, player->GetGUID().GetRawValue());
164  break;
165  default:
166  break;
167  }
168 
169  player->SendDirectMessage(&data);
170  }
171 
172 private:
173  std::vector<std::pair<WorldPacket*, size_t>* > _packetCache;
174  Builder const& _builder;
175  ChatMsg _msgType;
176 };

178 template<class Builder>
179 void CreatureTextMgr::SendChatPacket(WorldObject* source, Builder const& builder, ChatMsg msgType, WorldObject const* whisperTarget /*= nullptr*/, CreatureTextRange range /*= TEXT_RANGE_NORMAL*/, TeamId teamId /*= TEAM_NEUTRAL*/, bool gmOnly /*= false*/) const
180 {
181  if (!source)
182  return;
183 
184  CreatureTextLocalizer<Builder> localizer(builder, msgType);
185 
186  switch (msgType)
187  {
188  case CHAT_MSG_MONSTER_WHISPER:
189  case CHAT_MSG_RAID_BOSS_WHISPER:
190  {
191  if (range == TEXT_RANGE_NORMAL) //ignores team and gmOnly
192  {
193  if (!whisperTarget || whisperTarget->GetTypeId() != TYPEID_PLAYER)
194  return;
195 
196  localizer(const_cast<Player*>(whisperTarget->ToPlayer()));
197  return;
198  }
199  break;
200  }
201  default:
202  break;
203  }
204 
205  switch (range)
206  {
207  case TEXT_RANGE_AREA:
208  {
209  uint32 areaId = source->GetAreaId();
210  Map::PlayerList const& players = source->GetMap()->GetPlayers();
211  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
212  if (itr->GetSource()->GetAreaId() == areaId && (teamId == TEAM_NEUTRAL || itr->GetSource()->GetTeamId() == teamId) && (!gmOnly || itr->GetSource()->IsGameMaster()))
213  localizer(itr->GetSource());
214  return;
215  }
216  case TEXT_RANGE_ZONE:
217  {
218  uint32 zoneId = source->GetZoneId();
219  Map::PlayerList const& players = source->GetMap()->GetPlayers();
220  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
221  if (itr->GetSource()->GetZoneId() == zoneId && (teamId == TEAM_NEUTRAL || itr->GetSource()->GetTeamId() == teamId) && (!gmOnly || itr->GetSource()->IsGameMaster()))
222  localizer(itr->GetSource());
223  return;
224  }
225  case TEXT_RANGE_MAP:
226  {
227  Map::PlayerList const& players = source->GetMap()->GetPlayers();
228  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
229  if ((teamId == TEAM_NEUTRAL || itr->GetSource()->GetTeamId() == teamId) && (!gmOnly || itr->GetSource()->IsGameMaster()))
230  localizer(itr->GetSource());
231  return;
232  }
233  case TEXT_RANGE_WORLD:
234  {
235  SessionMap const& smap = sWorld->GetAllSessions();
236  for (SessionMap::const_iterator itr = smap.begin(); itr != smap.end(); ++itr)
237  if (Player* player = itr->second->GetPlayer())
238  if ((teamId == TEAM_NEUTRAL || player->GetTeamId() == teamId) && (!gmOnly || player->IsGameMaster()))
239  localizer(player);
240  return;
241  }
242  case TEXT_RANGE_NORMAL:
243  default:
244  break;
245  }
246 
247  float dist = GetRangeForChatType(msgType);
248  // xinef: hack for boss emote
249  if (msgType == CHAT_MSG_RAID_BOSS_EMOTE && source->GetMap()->IsDungeon())
250  dist = 250.0f;
251 
252  Acore::PlayerDistWorker<CreatureTextLocalizer<Builder> > worker(source, dist, localizer);
253  Cell::VisitWorldObjects(source, worker, dist);
254 }
