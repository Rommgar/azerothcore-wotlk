95  void LFGPlayerScript::OnMapChanged(Player* player)
96  {
97  Map const* map = player->GetMap();
98 
99  if (sLFGMgr->inLfgDungeonMap(player->GetGUID(), map->GetId(), map->GetDifficulty()))
100  {
101  Group* group = player->GetGroup();
102  // This function is also called when players log in
103  // if for some reason the LFG system recognises the player as being in a LFG dungeon,
104  // but the player was loaded without a valid group, we'll teleport to homebind to prevent
105  // crashes or other undefined behaviour
106  if (!group)
107  {
108  sLFGMgr->LeaveLfg(player->GetGUID());
109  sLFGMgr->LeaveAllLfgQueues(player->GetGUID(), true);
110  player->RemoveAurasDueToSpell(LFG_SPELL_LUCK_OF_THE_DRAW);
111  player->TeleportTo(player->m_homebindMapId, player->m_homebindX, player->m_homebindY, player->m_homebindZ, 0.0f);
112  LOG_DEBUG("lfg", "LFGPlayerScript::OnMapChanged, Player %s (%s) is in LFG dungeon map but does not have a valid group! Teleporting to homebind.",
113  player->GetName().c_str(), player->GetGUID().ToString().c_str());
114  return;
115  }
116 
117  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
118  if (Player* member = itr->GetSource())
119  player->GetSession()->SendNameQueryOpcode(member->GetGUID());
120 
121  if (group->IsLfgWithBuff())
122  player->CastSpell(player, LFG_SPELL_LUCK_OF_THE_DRAW, true);
123  }
124  else
125  {
126  player->RemoveAurasDueToSpell(LFG_SPELL_LUCK_OF_THE_DRAW);
127 
128  // Xinef: Destroy group if only one player is left
129  if (Group* group = player->GetGroup())
130  if (group->GetMembersCount() <= 1u)
131  group->Disband();
132  }
133  }

139  void LFGGroupScript::OnAddMember(Group* group, ObjectGuid guid)
140  {
141  if (!sLFGMgr->isOptionEnabled(LFG_OPTION_ENABLE_DUNGEON_FINDER | LFG_OPTION_ENABLE_RAID_BROWSER | LFG_OPTION_ENABLE_SEASONAL_BOSSES))
142  return;
143 
144  ObjectGuid gguid = group->GetGUID();
145  ObjectGuid leader = group->GetLeaderGUID();
146 
147  if (leader == guid)
148  {
149  LOG_DEBUG("lfg", "LFGScripts::OnAddMember [%s]: added [%s] leader [%s]", gguid.ToString().c_str(), guid.ToString().c_str(), leader.ToString().c_str());
150  sLFGMgr->SetLeader(gguid, guid);
151  }
152  else
153  {
154  LfgState gstate = sLFGMgr->GetState(gguid);
155  LfgState state = sLFGMgr->GetState(guid);
156  LOG_DEBUG("lfg", "LFGScripts::OnAddMember [%s]: added [%s] leader [%s] gstate: %u, state: %u",
157  gguid.ToString().c_str(), guid.ToString().c_str(), leader.ToString().c_str(), gstate, state);
158 
159  if (state == LFG_STATE_QUEUED)
160  sLFGMgr->LeaveLfg(guid);
161 
162  if (gstate == LFG_STATE_QUEUED)
163  sLFGMgr->LeaveLfg(gguid);
164  }
165 
166  if (!group->isLFGGroup())
167  {
168  sLFGMgr->LeaveAllLfgQueues(leader, true, gguid); // pussywizard: invited, queued, party formed, neither party nor new member are queued, but leader is in queue solo!
169  sLFGMgr->LeaveAllLfgQueues(guid, false);
170  }
171 
172  sLFGMgr->SetGroup(guid, gguid);
173  sLFGMgr->AddPlayerToGroup(gguid, guid);
174 
175  // pussywizard: after all necessary actions handle raid browser
176  if (sLFGMgr->GetState(guid) == LFG_STATE_RAIDBROWSER)
177  sLFGMgr->LeaveLfg(guid);

180  void LFGGroupScript::OnRemoveMember(Group* group, ObjectGuid guid, RemoveMethod method, ObjectGuid kicker, char const* reason)
181  {
182  // used only with EXTRA_LOGS
183  (void)kicker;
184  (void)reason;
185 
186  if (!sLFGMgr->isOptionEnabled(LFG_OPTION_ENABLE_DUNGEON_FINDER | LFG_OPTION_ENABLE_RAID_BROWSER | LFG_OPTION_ENABLE_SEASONAL_BOSSES))
187  return;
188 
189  ObjectGuid gguid = group->GetGUID();
190  LOG_DEBUG("lfg", "LFGScripts::OnRemoveMember [%s]: remove [%s] Method: %d Kicker: [%s] Reason: %s",
191  gguid.ToString().c_str(), guid.ToString().c_str(), method, kicker.ToString().c_str(), (reason ? reason : ""));
192 
193  bool isLFG = group->isLFGGroup();
194  LfgState state = sLFGMgr->GetState(gguid);
195  uint32 dungeonId = sLFGMgr->GetDungeon(gguid, false);
196 
197  // If group is being formed after proposal success do nothing more
198  if (state == LFG_STATE_PROPOSAL && method == GROUP_REMOVEMETHOD_DEFAULT)
199  {
200  // LfgData: Remove player from group
201  sLFGMgr->SetGroup(guid, ObjectGuid::Empty);
202  sLFGMgr->RemovePlayerFromGroup(gguid, guid);
203  return;
204  }
205 
206  sLFGMgr->LeaveLfg(guid);
207  sLFGMgr->LeaveAllLfgQueues(guid, true, gguid);
208  sLFGMgr->SetGroup(guid, ObjectGuid::Empty);
209  uint8 players = sLFGMgr->RemovePlayerFromGroup(gguid, guid);
210 
211  // pussywizard: after all necessary actions handle raid browser
212  // pussywizard: already done above
213  //if (sLFGMgr->GetState(guid) == LFG_STATE_RAIDBROWSER)
214  // sLFGMgr->LeaveLfg(guid);
215 
216  // Xinef: only LFG groups can go below
217  if (!isLFG)
218  return;
219 
220  if (state != LFG_STATE_FINISHED_DUNGEON && group) // Need more players to finish the dungeon
221  {
222  if (Player* leader = ObjectAccessor::FindConnectedPlayer(sLFGMgr->GetLeader(gguid)))
223  {
224  sLFGMgr->SetDungeon(gguid, dungeonId);
225  leader->GetSession()->SendLfgOfferContinue(sLFGMgr->GetDungeon(gguid, false));
226  }
227  }
228 
229  if (Player* player = ObjectAccessor::FindConnectedPlayer(guid))
230  {
231  // xinef: fixed dungeon deserter
232  if (method != GROUP_REMOVEMETHOD_KICK_LFG && state != LFG_STATE_FINISHED_DUNGEON &&
233  player->HasAura(LFG_SPELL_DUNGEON_COOLDOWN) && players >= LFG_GROUP_KICK_VOTES_NEEDED)
234  {
235  player->AddAura(LFG_SPELL_DUNGEON_DESERTER, player);
236  }
237  //else if (state == LFG_STATE_BOOT)
238  // Update internal kick cooldown of kicked
239 
240  player->GetSession()->SendLfgUpdateParty(LfgUpdateData(LFG_UPDATETYPE_LEADER_UNK1));
241  if (player->GetMap()->IsDungeon()) // Teleport player out the dungeon
242  {
243  // Xinef: no longer valid sLFGMgr->TeleportPlayer(player, true);
244  if (!player->IsBeingTeleportedFar() && player->GetMapId() == sLFGMgr->GetDungeonMapId(gguid))
245  player->TeleportToEntryPoint();
246  }
247  }
