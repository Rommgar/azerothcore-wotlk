60 void Log::CreateAppenderFromConfig(std::string const& appenderName)
61 {
62  if (appenderName.empty())
63  {
64  return;
65  }
66 
67  // Format = type, level, flags, optional1, optional2
68  // if type = File. optional1 = file and option2 = mode
69  // if type = Console. optional1 = Color
70  std::string options = sConfigMgr->GetStringDefault(appenderName, "");
71 
72  std::vector<std::string_view> tokens = Acore::Tokenize(options, ',', true);
73 
74  size_t const size = tokens.size();
75  std::string name = appenderName.substr(9);
76 
77  if (size < 2)
78  {
79  fprintf(stderr, "Log::CreateAppenderFromConfig: Wrong configuration for appender %s. Config line: %s\n", name.c_str(), options.c_str());
80  return;
81  }
82 
83  AppenderFlags flags = APPENDER_FLAGS_NONE;
84  AppenderType type = AppenderType(Acore::StringTo<uint8>(tokens[0]).value_or(APPENDER_INVALID));
85  LogLevel level = LogLevel(Acore::StringTo<uint8>(tokens[1]).value_or(LOG_LEVEL_INVALID));
86 
87  auto factoryFunction = appenderFactory.find(type);
88  if (factoryFunction == appenderFactory.end())
89  {
90  fprintf(stderr, "Log::CreateAppenderFromConfig: Unknown type '%s' for appender %s\n", std::string(tokens[0]).c_str(), name.c_str());
91  return;
92  }
93 
94  if (level > NUM_ENABLED_LOG_LEVELS)
95  {
96  fprintf(stderr, "Log::CreateAppenderFromConfig: Wrong Log Level '%s' for appender %s\n", std::string(tokens[1]).c_str(), name.c_str());
97  return;
98  }
99 
100  if (size > 2)
101  {
102  if (Optional<uint8> flagsVal = Acore::StringTo<uint8>(tokens[2]))
103  {
104  flags = AppenderFlags(*flagsVal);
105  }
106  else
107  {
108  fprintf(stderr, "Log::CreateAppenderFromConfig: Unknown flags '%s' for appender %s\n", std::string(tokens[2]).c_str(), name.c_str());
109  return;
110  }
111  }
112 
113  try
114  {
115  Appender* appender = factoryFunction->second(NextAppenderId(), name, level, flags, tokens);
116  appenders[appender->getId()].reset(appender);
117  }
118  catch (InvalidAppenderArgsException const& iaae)
119  {
120  fprintf(stderr, "%s\n", iaae.what());
121  }
122 }

124 void Log::CreateLoggerFromConfig(std::string const& appenderName)
125 {
126  if (appenderName.empty())
127  {
128  return;
129  }
130 
131  LogLevel level = LOG_LEVEL_DISABLED;
132 
133  std::string options = sConfigMgr->GetStringDefault(appenderName, "");
134  std::string name = appenderName.substr(7);
135 
136  if (options.empty())
137  {
138  fprintf(stderr, "Log::CreateLoggerFromConfig: Missing config option Logger.%s\n", name.c_str());
139  return;
140  }
141 
142  std::vector<std::string_view> tokens = Acore::Tokenize(options, ',', true);
143 
144  if (tokens.size() != 2)
145  {
146  fprintf(stderr, "Log::CreateLoggerFromConfig: Wrong config option Logger.%s=%s\n", name.c_str(), options.c_str());
147  return;
148  }
149 
150  std::unique_ptr<Logger>& logger = loggers[name];
151  if (logger)
152  {
153  fprintf(stderr, "Error while configuring Logger %s. Already defined\n", name.c_str());
154  return;
155  }
156 
157  level = LogLevel(Acore::StringTo<uint8>(tokens[0]).value_or(LOG_LEVEL_INVALID));
158  if (level > NUM_ENABLED_LOG_LEVELS)
159  {
160  fprintf(stderr, "Log::CreateLoggerFromConfig: Wrong Log Level '%s' for logger %s\n", std::string(tokens[0]).c_str(), name.c_str());
161  return;
162  }
163 
164  if (level > highestLogLevel)
165  {
166  highestLogLevel = level;
167  }
168 
169  logger = std::make_unique<Logger>(name, level);
170  //fprintf(stdout, "Log::CreateLoggerFromConfig: Created Logger %s, Level %u\n", name.c_str(), level);
171 
172  for (std::string_view appendName : Acore::Tokenize(tokens[1], ' ', false))
173  {
174  if (Appender* appender = GetAppenderByName(appendName))
175  {
176  logger->addAppender(appender->getId(), appender);
177  //fprintf(stdout, "Log::CreateLoggerFromConfig: Added Appender %s to Logger %s\n", appender->getName().c_str(), name.c_str());
178  }
179  else
180  {
181  fprintf(stderr, "Error while configuring Appender %s in Logger %s. Appender does not exist\n", std::string(appendName).c_str(), name.c_str());
182  }
183  }
184 }

195 void Log::ReadLoggersFromConfig()
196 {
197  std::vector<std::string> keys = sConfigMgr->GetKeysByString("Logger.");
198  for (std::string const& loggerName : keys)
199  {
200  CreateLoggerFromConfig(loggerName);
201  }
202 
203  // Bad config configuration, creating default config
204  if (loggers.find(LOGGER_ROOT) == loggers.end())
205  {
206  fprintf(stderr, "Wrong Loggers configuration. Review your Logger config section.\n"
207  "Creating default loggers [root (Error), server (Info)] to console\n");
208 
209  Close(); // Clean any Logger or Appender created
210 
211  AppenderConsole* appender = new AppenderConsole(NextAppenderId(), "Console", LOG_LEVEL_DEBUG, APPENDER_FLAGS_NONE, {});
212  appenders[appender->getId()].reset(appender);
213 
214  Logger* rootLogger = new Logger(LOGGER_ROOT, LOG_LEVEL_ERROR);
215  rootLogger->addAppender(appender->getId(), appender);
216  loggers[LOGGER_ROOT].reset(rootLogger);
217 
218  Logger* serverLogger = new Logger("server", LOG_LEVEL_INFO);
219  serverLogger->addAppender(appender->getId(), appender);
220  loggers["server"].reset(serverLogger);
221  }
222 }

276 std::string Log::GetTimestampStr()
277 {
278  time_t tt = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
279 
280  std::tm aTm;
281  localtime_r(&tt, &aTm);
282 
283  // YYYY year
284  // MM month (2 digits 01-12)
285  // DD day (2 digits 01-31)
286  // HH hour (2 digits 00-23)
287  // MM minutes (2 digits 00-59)
288  // SS seconds (2 digits 00-59)
289  return Acore::StringFormat("%04d-%02d-%02d_%02d-%02d-%02d",
290  aTm.tm_year + 1900, aTm.tm_mon + 1, aTm.tm_mday, aTm.tm_hour, aTm.tm_min, aTm.tm_sec);
291 }

293 bool Log::SetLogLevel(std::string const& name, int32 newLeveli, bool isLogger /* = true */)
294 {
295  if (newLeveli < 0)
296  {
297  return false;
298  }
299 
300  LogLevel newLevel = LogLevel(newLeveli);
301 
302  if (isLogger)
303  {
304  auto it = loggers.begin();
305  while (it != loggers.end() && it->second->getName() != name)
306  {
307  ++it;
308  }
309 
310  if (it == loggers.end())
311  {
312  return false;
313  }
314 
315  it->second->setLogLevel(newLevel);
316 
317  if (newLevel != LOG_LEVEL_DISABLED && newLevel > highestLogLevel)
318  {
319  highestLogLevel = newLevel;
320  }
321  }
322  else
323  {
324  Appender* appender = GetAppenderByName(name);
325  if (!appender)
326  {
327  return false;
328  }
329 
330  appender->setLogLevel(newLevel);
331  }
332 
333  return true;
334 }

370 bool Log::ShouldLog(std::string const& type, LogLevel level) const
371 {
372  // TODO: Use cache to store "Type.sub1.sub2": "Type" equivalence, should
373  // Speed up in cases where requesting "Type.sub1.sub2" but only configured
374  // Logger "Type"
375 
376  // Don't even look for a logger if the LogLevel is higher than the highest log levels across all loggers
377  if (level > highestLogLevel)
378  {
379  return false;
380  }
381 
382  Logger const* logger = GetLoggerByType(type);
383  if (!logger)
384  {
385  return false;
386  }
387 
388  LogLevel logLevel = logger->getLogLevel();
389  return logLevel != LOG_LEVEL_DISABLED && logLevel >= level;
390 }
