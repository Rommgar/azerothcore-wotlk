240  void SummonPortalKeeper()

251  if (!rift)
252  return;

254  int32 entry = 0;
255  switch (_currentRift)
256  {
257  case 6:
258  entry = GetData(TYPE_CHRONO_LORD_DEJA) == DONE ? (instance->IsHeroic() ? NPC_INFINITE_CHRONO_LORD : -NPC_CHRONO_LORD_DEJA) : NPC_CHRONO_LORD_DEJA;
259  break;
260  case 12:

262  break;

276  summon->SetTimer(3 * MINUTE * IN_MILLISECONDS);
