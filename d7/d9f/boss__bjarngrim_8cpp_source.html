291  void UpdateEscortAI(uint32 diff) override
292  {
293  if (!me->IsInCombat())
294  return;
295 
296  // Return since we have no target
297  if (!UpdateVictim())
298  {
299  Reset();
300  return;
301  }
302 
303  events.Update(diff);
304 
305  if (me->HasUnitState(UNIT_STATE_CASTING))
306  return;
307 
308  switch (events.ExecuteEvent())
309  {
310  case EVENT_BJARNGRIM_CHANGE_STANCE:
311  // roll new stance
312  RemoveStanceAura(m_uiStance);
313  RollStance(m_uiStance);
314  events.RepeatEvent(20000);
315  break;
316 
320  case EVENT_BJARNGRIM_REFLECTION:
321  me->CastSpell(me, SPELL_BJARNGRIM_REFLETION, true);
322  events.RepeatEvent(8000 + rand() % 1000);
323  break;
324  case EVENT_BJARNGRIM_PUMMEL:
325  me->CastSpell(me->GetVictim(), SPELL_PUMMEL, false);
326  events.RepeatEvent(10000 + rand() % 1000);
327  break;
328  case EVENT_BJARNGRIM_KNOCK:
329  me->CastSpell(me, SPELL_KNOCK_AWAY, false);
330  events.RepeatEvent(20000 + rand() % 1000);
331  break;
332  case EVENT_BJARNGRIM_IRONFORM:
333  me->CastSpell(me, SPELL_IRONFORM, true);
334  events.RepeatEvent(18000 + rand() % 5000);
335  break;
336 
340  case EVENT_BJARNGRIM_MORTAL_STRIKE:
341  me->CastSpell(me->GetVictim(), SPELL_MORTAL_STRIKE, false);
342  events.RepeatEvent(10000);
343  break;
344  case EVENT_BJARNGRIM_WHIRLWIND:
345  me->CastSpell(me, SPELL_WHIRLWIND, true);
346  events.RepeatEvent(25000);
347  break;
348 
352  case EVENT_BJARNGRIM_INTERCEPT:
353  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM))
354  me->CastSpell(target, SPELL_INTERCEPT, true);
355 
356  events.RepeatEvent(30000);
357  break;
358  case EVENT_BJARNGRIM_CLEAVE:
359  me->CastSpell(me->GetVictim(), SPELL_CLEAVE, false);
360  events.RepeatEvent(25000);
361  break;
362  case EVENT_BJARNGRIM_SLAM:
363  me->CastSpell(me->GetVictim(), SPELL_SLAM, false);
364  events.RepeatEvent(10000 + rand() % 2000);
365  break;
366  }
367 
368  DoMeleeAttackIfReady();
369  }
