115  static bool HandleGMListIngameCommand(ChatHandler* handler)
116  {
117  bool first = true;
118  bool footer = false;
119 
120  std::shared_lock<std::shared_mutex> lock(*HashMapHolder<Player>::GetLock());
121  for (auto const& [playerGuid, player] : ObjectAccessor::GetPlayers())
122  {
123  AccountTypes playerSec = player->GetSession()->GetSecurity();
124  if ((player->IsGameMaster() ||
125  (!AccountMgr::IsPlayerAccount(playerSec) && playerSec <= AccountTypes(sWorld->getIntConfig(CONFIG_GM_LEVEL_IN_GM_LIST)))) &&
126  (!handler->GetSession() || player->IsVisibleGloballyFor(handler->GetSession()->GetPlayer())))
127  {
128  if (first)
129  {
130  first = false;
131  footer = true;
132  handler->SendSysMessage(LANG_GMS_ON_SRV);
133  handler->SendSysMessage("========================");
134  }
135  std::string const& name = player->GetName();
136  uint8 size = uint8(name.size());
137  uint8 security = playerSec;
138  uint8 max = ((16 - size) / 2);
139  uint8 max2 = max;
140  if ((max + max2 + size) == 16)
141  max2 = max - 1;
142  if (handler->GetSession())
143  handler->PSendSysMessage("| %s GMLevel %u", name.c_str(), security);
144  else
145  handler->PSendSysMessage("|%*s%s%*s| %u |", max, " ", name.c_str(), max2, " ", security);
146  }
147  }
148  if (footer)
149  handler->SendSysMessage("========================");
150  if (first)
151  handler->SendSysMessage(LANG_GMS_NOT_LOGGED);
152  return true;
153  }

156  static bool HandleGMListFullCommand(ChatHandler* handler)
157  {
159  LoginDatabasePreparedStatement* stmt = LoginDatabase.GetPreparedStatement(LOGIN_SEL_GM_ACCOUNTS);
160  stmt->setUInt8(0, uint8(SEC_MODERATOR));
161  stmt->setInt32(1, int32(realm.Id.Realm));
162  PreparedQueryResult result = LoginDatabase.Query(stmt);
163 
164  if (result)
165  {
166  handler->SendSysMessage(LANG_GMLIST);
167  handler->SendSysMessage("========================");
169  do
170  {
171  Field* fields = result->Fetch();
172  char const* name = fields[0].GetCString();
173  uint8 security = fields[1].GetUInt8();
174  uint8 max = (16 - strlen(name)) / 2;
175  uint8 max2 = max;
176  if ((max + max2 + strlen(name)) == 16)
177  max2 = max - 1;
178  if (handler->GetSession())
179  handler->PSendSysMessage("| %s GMLevel %u", name, security);
180  else
181  handler->PSendSysMessage("|%*s%s%*s| %u |", max, " ", name, max2, " ", security);
182  } while (result->NextRow());
183  handler->SendSysMessage("========================");
184  }
185  else
186  handler->PSendSysMessage(LANG_GMLIST_EMPTY);
187  return true;
188  }

190  //Enable\Disable Invisible mode
191  static bool HandleGMVisibleCommand(ChatHandler* handler, Optional<bool> visibleArg)
192  {

200

202 
203  if (*visibleArg)

206  _player->RemoveAurasDueToSpell(VISUAL_AURA);
207 
208  _player->SetGMVisible(true);
209  _player->UpdateObjectVisibility();
210  handler->GetSession()->SendNotification(LANG_INVISIBLE_VISIBLE);
211  }
212  else

214  _player->AddAura(VISUAL_AURA, _player);
215  _player->SetGMVisible(false);
216  _player->UpdateObjectVisibility();
217  handler->GetSession()->SendNotification(LANG_INVISIBLE_INVISIBLE);
218  }
219 
220  return true;
