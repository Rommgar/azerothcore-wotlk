119  {

123  events.Update(diff);
124  if (me->HasUnitState(UNIT_STATE_CASTING))
125  return;
126 
127  switch (events.ExecuteEvent())
128  {
129  case EVENT_SPELL_TIDAL_WAVE:
130  me->CastSpell(me->GetVictim(), SPELL_TIDAL_WAVE, false);
131  events.ScheduleEvent(EVENT_SPELL_TIDAL_WAVE, 20000);
132  break;
133  case EVENT_SPELL_WATERY_GRAVE:
134  Talk(SAY_SUMMON_BUBLE);
135  if (me->HealthAbovePct(25))
136  {
137  Talk(EMOTE_WATERY_GRAVE);
138  me->CastCustomSpell(SPELL_WATERY_GRAVE, SPELLVALUE_MAX_TARGETS, 4, me, false);
139  }
140  else
141  {
142  Talk(EMOTE_WATERY_GLOBULES);
143  for (uint8 i = 0; i < 4; ++i)
144  me->CastSpell(me, waterGlobuleId[i], true);
145  }
146  events.ScheduleEvent(EVENT_SPELL_WATERY_GRAVE, 25000);
147  break;
148  case EVENT_SPELL_EARTHQUAKE:
149  Talk(EMOTE_EARTHQUAKE);
150  me->CastSpell(me, SPELL_EARTHQUAKE, false);
151  events.ScheduleEvent(EVENT_SPELL_EARTHQUAKE, urand(45000, 60000));
152  events.ScheduleEvent(EVENT_SUMMON_MURLOCS, 8000);
153  break;
154  case EVENT_SUMMON_MURLOCS:
155  Talk(SAY_SUMMON);
156  for (uint32 i = SPELL_SUMMON_MURLOC1; i < SPELL_SUMMON_MURLOC1 + 11; ++i)
157  me->CastSpell(me, i, true);
158  break;
159  }
160 
161  DoMeleeAttackIfReady();
162  }
