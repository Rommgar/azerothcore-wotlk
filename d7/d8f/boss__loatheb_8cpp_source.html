141  void UpdateAI(uint32 diff) override
142  {
143  if (!UpdateVictim() || !IsInRoom())
144  return;
145 
146  events.Update(diff);
147  if (me->HasUnitState(UNIT_STATE_CASTING))
148  return;
149 
150  switch (events.ExecuteEvent())
151  {
152  case EVENT_SUMMON_SPORE:
153  me->CastSpell(me, SPELL_SUMMON_SPORE, true);
154  events.RepeatEvent(35000);
155  break;
156  case EVENT_NECROTIC_AURA:
157  me->CastSpell(me, SPELL_NECROTIC_AURA, true);
158  Talk(SAY_NECROTIC_AURA_APPLIED);
159  events.ScheduleEvent(EVENT_NECROTIC_AURA_FADING, 14000);
160  events.ScheduleEvent(EVENT_NECROTIC_AURA_REMOVED, 17000);
161  events.RepeatEvent(20000);
162  break;
163  case EVENT_DEATHBLOOM:
164  me->CastSpell(me, RAID_MODE(SPELL_DEATHBLOOM_10, SPELL_DEATHBLOOM_25), false);
165  events.RepeatEvent(30000);
166  break;
167  case EVENT_INEVITABLE_DOOM:
168  me->CastSpell(me, RAID_MODE(SPELL_INEVITABLE_DOOM_10, SPELL_INEVITABLE_DOOM_25), false);
169  doomCounter++;
170  events.RepeatEvent(doomCounter < 6 ? 30000 : 15000);
171  break;
172  case EVENT_BERSERK:
173  me->CastSpell(me, SPELL_BERSERK, true);
174  break;
175  case EVENT_NECROTIC_AURA_FADING:
176  Talk(SAY_NECROTIC_AURA_FADING);
177  break;
178  case EVENT_NECROTIC_AURA_REMOVED:
179  Talk(SAY_NECROTIC_AURA_REMOVED);
180  break;
181  }
182  DoMeleeAttackIfReady();
183  }
