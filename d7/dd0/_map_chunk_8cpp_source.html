68 {
69  fseek(stream, Header.OffsetMCVT, SEEK_CUR);
70  Vertices.reserve(125);
71 
72  for (int j = 0; j < 17; j++)
73  {
74  int values = j % 2 ? 8 : 9;
75  for (int i = 0; i < values; i++)
76  {
77  float tmp;
78  if (fread(&tmp, sizeof(float), 1, stream) != 1)
79  printf("MapChunk::GenerateVertices: Failed to read some data expected 1, read 0\n");
80  Vector3 vert(Header.Position.x - (j * (Constants::UnitSize * 0.5f)), Header.Position.y - (i * Constants::UnitSize), Header.Position.z + tmp);
81  if (values == 8)
82  vert.y -= Constants::UnitSize * 0.5f;
83  Vertices.push_back(vert);
84  }
85  }
86  // Restore stream position.
87  fseek(stream, Source->Offset, SEEK_SET);
