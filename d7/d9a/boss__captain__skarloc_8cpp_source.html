171  me->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
172  me->SetInCombatWithZone();
173  for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
174  if (Creature* summon = ObjectAccessor::GetCreature(*me, *itr))
175  if (summon->GetEntry() != NPC_SKARLOC_MOUNT)
176  {
177  summon->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PC | UNIT_FLAG_IMMUNE_TO_NPC);
178  summon->SetInCombatWithZone();
179  }
180  break;
181  }
182 
183  if (!UpdateVictim())
184  return;
185 
186  events.Update(diff);
187  if (me->HasUnitState(UNIT_STATE_CASTING))
188  return;
189 
190  switch (events.ExecuteEvent())
191  {
192  case EVENT_SPELL_HOLY_LIGHT:
193  me->CastSpell(me, SPELL_HOLY_LIGHT, false);
194  events.ScheduleEvent(EVENT_SPELL_HOLY_LIGHT, 20000);
195  break;
196  case EVENT_SPELL_CLEANSE:
197  if (roll_chance_i(33))
198  Talk(SAY_TAUNT);
199  me->CastSpell(me, SPELL_CLEANSE, false);
200  events.ScheduleEvent(EVENT_SPELL_CLEANSE, 10000);
201  break;
202  case EVENT_SPELL_HAMMER:

204  events.ScheduleEvent(EVENT_SPELL_HAMMER, 30000);
205  break;
206  case EVENT_SPELL_HOLY_SHIELD:
207  me->CastSpell(me, SPELL_CLEANSE, false);
208  events.ScheduleEvent(SPELL_HOLY_SHIELD, 30000);
209  break;
