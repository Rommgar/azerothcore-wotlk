366  }
367  }
368 
369  pet->SetLoading(false);
370  owner->SetTemporaryUnsummonedPetNumber(0); // clear this only if pet is loaded successfuly
371 
372  // current
373  if (current && owner->IsPetNeedBeTemporaryUnsummoned())
374  {
375  owner->UnsummonPetTemporaryIfAny();
376  return;
377  }
378 
379  pet->HandleAsynchLoadSucceed();

419  LOG_DEBUG("network.opcode", "HandlePetAction: Pet %s - flag: %u, spellid: %u, target: %s.", guid1.ToString().c_str(), uint32(flag), spellid, guid2.ToString().c_str());
420 
421  if (!pet)
422  {

424  return;
425  }

430  return;
431  }
432 
433  if (!pet->IsAlive())
434  {
435  // xinef: allow dissmis dead pets
436  SpellInfo const* spell = (flag == ACT_ENABLED || flag == ACT_PASSIVE) ? sSpellMgr->GetSpellInfo(spellid) : nullptr;
437  if ((flag != ACT_COMMAND || spellid != COMMAND_ABANDON) && (!spell || !spell->HasAttribute(SPELL_ATTR0_ALLOW_CAST_WHILE_DEAD)))
438  return;
439  }
440 
441  // Xinef: allow to controll players
442  if (pet->GetTypeId() == TYPEID_PLAYER && flag != ACT_COMMAND && flag != ACT_REACTION)
443  return;
444 
445  // Do not follow itself vehicle
446  if (spellid == COMMAND_FOLLOW && _player->IsOnVehicle(pet))
447  {
448  return;
449  }
450

452  HandlePetActionHelper(pet, guid1, spellid, flag, guid2);
453  else

456  std::vector<Unit*> controlled;

458  {
459  // xinef: allow to dissmis dead pets

461  controlled.push_back(*itr);
462  // xinef: mirror image blizzard crappness
463  else if ((*itr)->GetEntry() == NPC_MIRROR_IMAGE && flag == ACT_COMMAND && spellid == COMMAND_FOLLOW)
464  {
465  (*itr)->InterruptNonMeleeSpells(false);
466  }
467  }
468 
469  for (Unit* pet : controlled)
470  if (pet && pet->IsInWorld() && pet->GetMap() == _player->GetMap())
471  HandlePetActionHelper(pet, guid1, spellid, flag, guid2);
472  }

503 void WorldSession::HandlePetActionHelper(Unit* pet, ObjectGuid guid1, uint32 spellid, uint16 flag, ObjectGuid guid2)
504 {

506  if (!charmInfo)
507  {

513  switch (flag)
514  {
515  case ACT_COMMAND: //0x07
516  switch (spellid)
517  {
518  case COMMAND_STAY: //flat=1792 //STAY
519  {
520  bool controlledMotion = pet->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) != NULL_MOTION_TYPE;
521  if (!controlledMotion)

536  pet->ToPet()->ClearCastWhenWillAvailable();
537 
538  charmInfo->SetForcedSpell(0);
539  charmInfo->SetForcedTargetGUID();
540  break;
541  }
542  case COMMAND_FOLLOW: //spellid=1792 //FOLLOW
543  {
544  pet->AttackStop();
545  pet->InterruptNonMeleeSpells(false);
546  pet->ClearInPetCombat();
547  pet->GetMotionMaster()->MoveFollow(_player, PET_FOLLOW_DIST, pet->GetFollowAngle());
548  if (pet->ToPet())

556  charmInfo->SetIsFollowing(false);
557  charmInfo->RemoveStayPosition();
558  charmInfo->SetForcedSpell(0);
559  charmInfo->SetForcedTargetGUID();
560  break;
561  }
562  case COMMAND_ATTACK: //spellid=1792 //ATTACK
563  {
564  // Can't attack if owner is pacified
565  if (_player->HasAuraType(SPELL_AURA_MOD_PACIFY))
566  {
567  //pet->SendPetCastFail(spellid, SPELL_FAILED_PACIFIED);

575  return;

578  if (!owner->IsValidAttackTarget(TargetUnit))
579  return;
580 
581  // pussywizard (excluded charmed)
