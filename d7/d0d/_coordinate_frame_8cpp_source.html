242 Plane CoordinateFrame::toWorldSpace(const Plane& plane) const {
243  // Since there is no scale factor, we don't have to
244  // worry about the inverse transpose of the normal.
245  Vector3 normal;
246  float d;
247 
248  plane.getEquation(normal, d);
249 
250  const Vector3& newNormal = rotation * normal;
251 
252  if (isFinite(d)) {
253  d = (newNormal * -d + translation).dot(newNormal);
254  return Plane(newNormal, newNormal * d);
255  } else {
256  // When d is infinite, we can't multiply 0's by it without
257  // generating NaNs.
258  return Plane::fromEquation(newNormal.x, newNormal.y, newNormal.z, d);
259  }
260 }

382 void CoordinateFrame::lookAt(
383  const Vector3& target,
384  Vector3 up) {
385 
386  up = up.direction();
387 
388  Vector3 look = (target - translation).direction();
389  if (fabs(look.dot(up)) > .99f) {
390  up = Vector3::unitX();
391  if (fabs(look.dot(up)) > .99f) {
392  up = Vector3::unitY();
393  }
394  }
395 
396  up -= look * look.dot(up);
397  up = up.direction();
398 
399  Vector3 z = -look;
400  Vector3 x = -z.cross(up);
401  x = x.direction();
402 
403  Vector3 y = z.cross(x);
404 
405  rotation.setColumn(0, x);
406  rotation.setColumn(1, y);
407  rotation.setColumn(2, z);
408 }
