85  void ParseFile(std::string const& file)
86  {
87  std::ifstream in(file);
88 
89  if (in.fail())
90  {
91  throw ConfigException(Acore::StringFormat("Config::LoadFile: Failed open file '%s'", file.c_str()));
92  }
93 
94  uint32 count = 0;
95  uint32 lineNumber = 0;
96  std::unordered_map<std::string /*name*/, std::string /*value*/> fileConfigs;
97 
98  auto IsDuplicateOption = [&](std::string const& confOption)
99  {
100  auto const& itr = fileConfigs.find(confOption);
101  if (itr != fileConfigs.end())
102  {
103  PrintError(file, "> Config::LoadFile: Dublicate key name '%s' in config file '%s'", std::string(confOption).c_str(), file.c_str());
104  return true;
105  }
106 
107  return false;
108  };
109 
110  while (in.good())
111  {
112  lineNumber++;
113  std::string line;
114  std::getline(in, line);
115 
116  // read line error
117  if (!in.good() && !in.eof())
118  {
119  throw ConfigException(Acore::StringFormat("> Config::LoadFile: Failure to read line number %u in file '%s'", lineNumber, file.c_str()));
120  }
121 
122  // remove whitespace in line
123  line = Acore::String::Trim(line, in.getloc());
124 
125  if (line.empty())
126  {
127  continue;
128  }
129 
130  // comments
131  if (line[0] == '#' || line[0] == '[')
132  {
133  continue;
134  }
135 
136  size_t found = line.find_first_of('#');
137  if (found != std::string::npos)
138  {
139  line = line.substr(0, found);
140  }
141 
142  auto const equal_pos = line.find('=');
143 
144  if (equal_pos == std::string::npos || equal_pos == line.length())
145  {
146  PrintError(file, "> Config::LoadFile: Failure to read line number %u in file '%s'. Skip this line", lineNumber, file.c_str());
147  continue;
148  }
149 
150  auto entry = Acore::String::Trim(line.substr(0, equal_pos), in.getloc());
151  auto value = Acore::String::Trim(line.substr(equal_pos + 1, std::string::npos), in.getloc());
152 
153  value.erase(std::remove(value.begin(), value.end(), '"'), value.end());
154 
155  // Skip if 2+ same options in one config file
156  if (IsDuplicateOption(entry))
157  {
158  continue;
159  }
160 
161  // Add to temp container
162  fileConfigs.emplace(entry, value);
163  count++;
164  }
165 
166  // No lines read
167  if (!count)
168  {
169  throw ConfigException(Acore::StringFormat("Config::LoadFile: Empty file '%s'", file.c_str()));
170  }
171 
172  // Add correct keys if file load without errors
173  for (auto const& [entry, key] : fileConfigs)
174  {
175  AddKey(entry, key);
176  }
177  }

368 bool ConfigMgr::LoadModulesConfigs()
369 {
370  if (_additonalFiles.empty())
371  {
372  return true;
373  }
374 
375  // Start loading module configs
376  std::string const& moduleConfigPath = GetConfigPath() + "modules/";
377  bool isExistDefaultConfig = true;
378  bool isExistDistConfig = true;
379 
380  for (auto const& distFileName : _additonalFiles)
381  {
382  std::string defaultFileName = distFileName;
383 
384  if (!defaultFileName.empty())
385  {
386  defaultFileName.erase(defaultFileName.end() - 5, defaultFileName.end());
387  }
388 
389  // Load .conf.dist config
390  if (!LoadAdditionalFile(moduleConfigPath + distFileName))
391  {
392  isExistDistConfig = false;
393  }
394 
395  // Load .conf config
396  if (!LoadAdditionalFile(moduleConfigPath + defaultFileName))
397  {
398  isExistDefaultConfig = false;
399  }
400 
401  if (isExistDefaultConfig && isExistDistConfig)
402  {
403  _moduleConfigFiles.emplace_back(defaultFileName);
404  }
405  else if (!isExistDefaultConfig && isExistDistConfig)
406  {
407  _moduleConfigFiles.emplace_back(distFileName);
408  }
409  }
410 
411  // If module configs not exist - no load
412  return !_moduleConfigFiles.empty();
413 }
