259  void UpdateAI(uint32 diff) override
260  {
261  if (!UpdateVictim() || !CheckInRoom())
262  return;
263 
264  events.Update(diff);
265 
266  if (me->HasUnitState(UNIT_STATE_CASTING))
267  return;
268 
269  switch (events.ExecuteEvent())
270  {
271  case EVENT_UNROOT:
272  me->SetControlled(false, UNIT_STATE_ROOT);
273  me->DisableRotate(false);
274  break;
275  case EVENT_SLIME_SPRAY:
276  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, NonTankTargetSelector(me)))
277  {
278  if (Creature* c = me->SummonCreature(NPC_OOZE_SPRAY_STALKER, *target, TEMPSUMMON_TIMED_DESPAWN, 8000))
279  {
280  me->SetOrientation(me->GetAngle(c));
281  me->SetControlled(true, UNIT_STATE_ROOT);
282  me->DisableRotate(true);
283  me->SetFacingTo(me->GetAngle(c));
284  me->SendMovementFlagUpdate();
285  Talk(EMOTE_SLIME_SPRAY);
286  me->CastSpell(c, SPELL_SLIME_SPRAY, false);
287  }
288  }
289  events.DelayEvents(1);
290  events.ScheduleEvent(EVENT_SLIME_SPRAY, 20000);
291  events.ScheduleEvent(EVENT_UNROOT, 0);
292  break;
293  case EVENT_HASTEN_INFECTIONS:
294  if (infectionCooldown >= 8000)
295  {
296  infectionCooldown -= 2000;
297  events.ScheduleEvent(EVENT_HASTEN_INFECTIONS, 90000);
298  }
299  break;
300  case EVENT_MUTATED_INFECTION:
301  me->CastCustomSpell(SPELL_MUTATED_INFECTION, SPELLVALUE_MAX_TARGETS, 1, nullptr, false);
302  events.ScheduleEvent(EVENT_MUTATED_INFECTION, infectionCooldown);
303  break;
304  case EVENT_ROTFACE_OOZE_FLOOD:
305  if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
306  {
307  professor->AI()->Talk(SAY_ROTFACE_OOZE_FLOOD);
308  me->CastSpell((Unit*)nullptr, oozeFloodSpells[_oozeFloodStage], true);
309  if (++_oozeFloodStage == 4)
310  _oozeFloodStage = 0;
311  }
312  events.ScheduleEvent(EVENT_ROTFACE_OOZE_FLOOD, 25000);
313  break;
314  case EVENT_ROTFACE_VILE_GAS:
315  {
316  std::list<Unit*> targets;
317  uint32 minTargets = RAID_MODE<uint32>(3, 8, 3, 8);
318  SelectTargetList(targets, minTargets, SELECT_TARGET_RANDOM, -5.0f, true);
319  float minDist = 0.0f;
320  if (targets.size() >= minTargets)
321  minDist = -5.0f;
322 
323  if (Unit* target = SelectTarget(SELECT_TARGET_RANDOM, 0, minDist, true))
324  if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
325  professor->CastSpell(target, SPELL_VILE_GAS_H, true); // triggered, to skip LoS check
326  }
327  events.ScheduleEvent(EVENT_ROTFACE_VILE_GAS, urand(15000, 20000));
328  break;
329  default:
330  break;
331  }
332 
333  DoMeleeAttackIfReady();
334  }
