66  void InitializeAI() override

68  ScriptedAI::InitializeAI();
69  Unit* owner = me->GetOwner();
70  if (!owner)
71  return;
72 
73  // Xinef: Night of the Dead avoidance
74  if (Aura* aur = me->GetAura(SPELL_DK_NIGHT_OF_THE_DEAD))
75  if (AuraEffect* aurEff = owner->GetAuraEffect(SPELL_AURA_ADD_FLAT_MODIFIER, SPELLFAMILY_DEATHKNIGHT, 2718, 0))
76  {
77  if (aur->GetEffect(0))

79  aur->GetEffect(0)->SetAmount(-aurEff->GetSpellInfo()->Effects[EFFECT_2].CalcValue());
80  }
81  }
82 
83  me->SetCanFly(true);
84  me->SetDisableGravity(true);
85 
86  float tz = me->GetMapHeight(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), true, MAX_FALL_DISTANCE);
87  me->GetMotionMaster()->MoveCharge(me->GetPositionX(), me->GetPositionY(), tz, 7.0f, 1);
88  me->AddUnitState(UNIT_STATE_NO_ENVIRONMENT_UPD);
89  _selectionTimer = 2000;
90  _initialCastTimer = 0;
91  }

93  void MySelectNextTarget()
94  {
95  Unit* owner = me->GetOwner();
96  if (owner && owner->GetTypeId() == TYPEID_PLAYER && (!me->GetVictim() || me->GetVictim()->IsImmunedToSpell(sSpellMgr->GetSpellInfo(51963)) || !me->IsValidAttackTarget(me->GetVictim()) || !owner->CanSeeOrDetect(me->GetVictim())))
97  {
98  Unit* selection = owner->ToPlayer()->GetSelectedUnit();
99  if (selection && selection != me->GetVictim() && me->IsValidAttackTarget(selection))
100  {
101  me->GetMotionMaster()->Clear(false);
102  SetGazeOn(selection);
103  }
104 
105  else if (!me->GetVictim() || !owner->CanSeeOrDetect(me->GetVictim()))
106  {
107  me->CombatStop(true);
108  me->GetMotionMaster()->Clear(false);
109  me->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, 0.0f);
110  RemoveTargetAura();
111  }
112  }
113  }

152  float y = me->GetPositionY() + 20 * sin(me->GetOrientation());

154  me->DisableSpline();

157  me->GetMotionMaster()->MoveCharge(x, y, z, 7.0f, 1);
158  me->SetCanFly(true);
159  me->SetDisableGravity(true);
160 
161  _despawning = true;
162  }

164  void UpdateAI(uint32 diff) override
165  {
166  if (_initialSelection)
167  {
168  _initialSelection = false;
169  // Find victim of Summon Gargoyle spell
170  std::list<Unit*> targets;
171  Acore::AnyUnfriendlyUnitInObjectRangeCheck u_check(me, me, 50);
172  Acore::UnitListSearcher<Acore::AnyUnfriendlyUnitInObjectRangeCheck> searcher(me, targets, u_check);
173  Cell::VisitAllObjects(me, searcher, 50.0f);
174  for (std::list<Unit*>::const_iterator iter = targets.begin(); iter != targets.end(); ++iter)
175  if ((*iter)->GetAura(SPELL_DK_SUMMON_GARGOYLE_1, me->GetOwnerGUID()))
176  {
177  (*iter)->RemoveAura(SPELL_DK_SUMMON_GARGOYLE_1, me->GetOwnerGUID());
178  SetGazeOn(*iter);
179  _targetGUID = (*iter)->GetGUID();
180  break;
181  }
182  }
183  if (_despawnTimer > 4000)
184  {
185  _despawnTimer -= diff;
186  if (!UpdateVictimWithGaze())
187  {
188  MySelectNextTarget();
189  return;
190  }
191 
192  _initialCastTimer += diff;
193  _selectionTimer += diff;
194  if (_selectionTimer >= 1000)
195  {
196  MySelectNextTarget();
197  _selectionTimer = 0;
198  }
199  if (_initialCastTimer >= 2000 && !me->HasUnitState(UNIT_STATE_CASTING | UNIT_STATE_LOST_CONTROL) && me->GetMotionMaster()->GetMotionSlotType(MOTION_SLOT_CONTROLLED) == NULL_MOTION_TYPE)
200  me->CastSpell(me->GetVictim(), 51963, false);
201  }
202  else
203  {
204  if (!_despawning)
205  FlyAway();
206 
207  if (_despawnTimer > diff)
208  _despawnTimer -= diff;
209  else
210  me->DespawnOrUnsummon();
211  }
212  }
