42 {
43  switch (type)
44  {
45  case DRIVER_CHECK:
46  case MPQ_CHECK: return 1;
47  case LUA_EVAL_CHECK: return 1 + sizeof(_luaEvalPrefix) - 1 + sizeof(_luaEvalMidfix) - 1 + 4 + sizeof(_luaEvalPostfix) - 1;

50  case MODULE_CHECK: return (4 + Acore::Crypto::Constants::SHA1_DIGEST_LENGTH_BYTES);
51  case MEM_CHECK: return (1 + 4 + 1);
52  default: return 0;

56 static uint16 GetCheckPacketSize(WardenCheck const* check)
57 {
58  if (!check)
59  {
60  return 0;
61  }
62 
63  uint16 size = 1 + GetCheckPacketBaseSize(check->Type); // 1 byte check type
64  if (!check->Str.empty())
65  {
66  size += (static_cast<uint16>(check->Str.length()) + 1); // 1 byte string length
67  }
68 
69  BigNumber tempNumber = check->Data;
70  if (!tempNumber.GetNumBytes())
71  {
72  size += tempNumber.GetNumBytes();
73  }
74  return size;

152  // Create packet structure
153  WardenInitModuleRequest Request;
154  Request.Command1 = WARDEN_SMSG_MODULE_INITIALIZE;
155  Request.Size1 = 20;
156  Request.Unk1 = 1;
157  Request.Unk2 = 0;
158  Request.Type = 1;

164  Request.CheckSumm1 = BuildChecksum(&Request.Unk1, Acore::Crypto::Constants::SHA1_DIGEST_LENGTH_BYTES);

167  Request.Size2 = 8;
168  Request.Unk3 = 4;

171  Request.Function2 = 0x00419210; // 0x00400000 + 0x00419210 FrameScript::Execute
172  Request.Function2_set = 1;

174

176  Request.Size3 = 8;
177  Request.Unk5 = 1;
178  Request.Unk6 = 1;
179  Request.String_library3 = 0;
180  Request.Function3 = 0x0046AE20; // 0x00400000 + 0x0046AE20 PerformanceCounter
181  Request.Function3_set = 1;
182  Request.CheckSumm3 = BuildChecksum(&Request.Unk5, 8);

184  EndianConvert(Request.Size1);

195  EndianConvert(Request.Function3);
196 
197  // Encrypt with warden RC4 key.
198  EncryptData(reinterpret_cast<uint8*>(&Request), sizeof(WardenInitModuleRequest));
199 
200  WorldPacket pkt(SMSG_WARDEN_DATA, sizeof(WardenInitModuleRequest));

246 void WardenWin::RequestChecks()
247 {
248  LOG_DEBUG("warden", "Request data");
249 
250  // If all checks were done, fill the todo list again
251  for (uint8 i = 0; i < MAX_WARDEN_CHECK_TYPES; ++i)
252  {
253  if (_ChecksTodo[i].empty())
254  _ChecksTodo[i].assign(sWardenCheckMgr->CheckIdPool[i].begin(), sWardenCheckMgr->CheckIdPool[i].end());
255  }
256 
257  _serverTicks = World::GetGameTimeMS();
258  _CurrentChecks.clear();
259 
260  // No pending checks
261  if (_PendingChecks.empty())
262  {
263  for (uint8 checkType = 0; checkType < MAX_WARDEN_CHECK_TYPES; ++checkType)
264  {
265  for (uint32 y = 0; y < sWorld->getIntConfig(GetMaxWardenChecksForType(checkType)); ++y)
266  {
267  // If todo list is done break loop (will be filled on next Update() run)
268  if (_ChecksTodo[checkType].empty())
269  {
270  break;
271  }
272 
273  // Get check id from the end and remove it from todo
274  uint16 const id = _ChecksTodo[checkType].back();
275  _ChecksTodo[checkType].pop_back();
276 
277  // Insert check to queue
278  if (checkType == WARDEN_CHECK_LUA_TYPE)
279  {
280  _CurrentChecks.push_front(id);
281  }
282  else
283  {
284  _CurrentChecks.push_back(id);
285  }
286  }
287  }
288  }
289  else
290  {
291  bool hasLuaChecks = false;
292  for (uint16 const checkId : _PendingChecks)
293  {
294  WardenCheck const* check = sWardenCheckMgr->GetWardenDataById(checkId);
295  if (!hasLuaChecks && check->Type == LUA_EVAL_CHECK)
296  {
297  hasLuaChecks = true;
298  }
299 
300  _CurrentChecks.push_back(checkId);
301  }
302 
303  // Always include lua checks
304  if (!hasLuaChecks)
305  {
306  for (uint32 i = 0; i < sWorld->getIntConfig(GetMaxWardenChecksForType(WARDEN_CHECK_LUA_TYPE)); ++i)
307  {
308  // If todo list is done break loop (will be filled on next Update() run)
309  if (_ChecksTodo[WARDEN_CHECK_LUA_TYPE].empty())
310  {
311  break;
312  }
313 
314  // Get check id from the end and remove it from todo
315  uint16 const id = _ChecksTodo[WARDEN_CHECK_LUA_TYPE].back();
316  _ChecksTodo[WARDEN_CHECK_LUA_TYPE].pop_back();
317 
318  // Lua checks must be always in front
319  _CurrentChecks.push_front(id);
320  }
321  }
322  }
323 
324  // Filter too high checks queue
325  // Filtered checks will get passed in next checks
326  uint16 expectedSize = 4;
327  _PendingChecks.clear();
328  Acore::Containers::EraseIf(_CurrentChecks,
329  [this, &expectedSize](uint16 id)
330  {
331  uint16 const thisSize = GetCheckPacketSize(sWardenCheckMgr->GetWardenDataById(id));
332  if ((expectedSize + thisSize) > 500) // warden packets are truncated to 512 bytes clientside
333  {
334  _PendingChecks.push_back(id);
335  return true;
336  }
337  expectedSize += thisSize;
338  return false;
339  }
340  );
341 
342  ByteBuffer buff;
343  buff << uint8(WARDEN_SMSG_CHEAT_CHECKS_REQUEST);
344 
345  for (uint16 const checkId : _CurrentChecks)
346  {
347  WardenCheck const* check = sWardenCheckMgr->GetWardenDataById(checkId);
348  switch (check->Type)
349  {
350  case LUA_EVAL_CHECK:
351  {
352  buff << uint8(sizeof(_luaEvalPrefix) - 1 + check->Str.size() + sizeof(_luaEvalMidfix) - 1 + check->IdStr.size() + sizeof(_luaEvalPostfix) - 1);
353  buff.append(_luaEvalPrefix, sizeof(_luaEvalPrefix) - 1);
354  buff.append(check->Str.data(), check->Str.size());
355  buff.append(_luaEvalMidfix, sizeof(_luaEvalMidfix) - 1);
356  buff.append(check->IdStr.data(), check->IdStr.size());
357  buff.append(_luaEvalPostfix, sizeof(_luaEvalPostfix) - 1);
358  break;
359  }
360  case MPQ_CHECK:
361  case DRIVER_CHECK:
362  {
363  buff << uint8(check->Str.size());
364  buff.append(check->Str.c_str(), check->Str.size());
365  break;
366  }
367  }
368  }
369 
370  uint8 const xorByte = _inputKey[0];
371 
372  // Add TIMING_CHECK
373  buff << uint8(0x00);
374  buff << uint8(TIMING_CHECK ^ xorByte);
375 
376  uint8 index = 1;
377 
378  for (uint16 const checkId : _CurrentChecks)
379  {
380  WardenCheck const* check = sWardenCheckMgr->GetWardenDataById(checkId);
381  buff << uint8(check->Type ^ xorByte);
382  switch (check->Type)
383  {
384  case MEM_CHECK:
385  {
386  buff << uint8(0x00);
387  buff << uint32(check->Address);
388  buff << uint8(check->Length);
389  break;
390  }
391  case PAGE_CHECK_A:
392  case PAGE_CHECK_B:
393  {
394  std::vector<uint8> data = check->Data.ToByteVector(24, false);
395  buff.append(data.data(), data.size());
396  buff << uint32(check->Address);
397  buff << uint8(check->Length);
398  break;
399  }
400  case MPQ_CHECK:
401  case LUA_EVAL_CHECK:
402  {
403  buff << uint8(index++);
404  break;
405  }
406  case DRIVER_CHECK:
407  {
408  std::vector<uint8> data = check->Data.ToByteVector(24, false);
409  buff.append(data.data(), data.size());
410  buff << uint8(index++);
411  break;
412  }
413  case MODULE_CHECK:
414  {
415  std::array<uint8, 4> seed = Acore::Crypto::GetRandomBytes<4>();
416  buff.append(seed);
417  buff.append(Acore::Crypto::HMAC_SHA1::GetDigestOf(seed, check->Str));
418  break;
419  }
420  /*case PROC_CHECK:
421  {
422  buff.append(wd->i.AsByteArray(0, false).get(), wd->i.GetNumBytes());
423  buff << uint8(index++);
424  buff << uint8(index++);
425  buff << uint32(wd->Address);
426  buff << uint8(wd->Length);
427  break;
428  }*/
429  }
430  }
431  buff << uint8(xorByte);
432  buff.hexlike();
433 
434  // Encrypt with warden RC4 key
435  EncryptData(buff.contents(), buff.size());
436 
437  WorldPacket pkt(SMSG_WARDEN_DATA, buff.size());
438  pkt.append(buff);
439  _session->SendPacket(&pkt);
440 
441  _dataSent = true;
442 
443  std::stringstream stream;
444  stream << "Sent check id's: ";
445  for (uint16 checkId : _CurrentChecks)
446  {
447  stream << checkId << " ";
448  }
449 
450  LOG_DEBUG("warden", "%s", stream.str().c_str());
451 }

453 void WardenWin::HandleData(ByteBuffer& buff)
454 {
455  LOG_DEBUG("warden", "Handle data");
456 
457  _dataSent = false;
458  _clientResponseTimer = 0;
459 
460  uint16 Length;
461  buff >> Length;
462  uint32 Checksum;
463  buff >> Checksum;
464 
465  if (Length != (buff.size() - buff.rpos()))
466  {
467  buff.rfinish();
468  ApplyPenalty(0, "Failed size checks in HandleData");
469  return;
470  }
471 
472  if (!IsValidCheckSum(Checksum, buff.contents() + buff.rpos(), Length))
473  {
474  buff.rpos(buff.wpos());
475  LOG_DEBUG("warden", "CHECKSUM FAIL");
476  ApplyPenalty(0, "Failed checksum in HandleData");
477  return;
478  }
479 
480  // TIMING_CHECK
481  {
482  uint8 result;
483  buff >> result;
484  // TODO: test it.
485  if (result == 0x00)
486  {
487  LOG_DEBUG("warden", "TIMING CHECK FAIL result 0x00");
488  ApplyPenalty(0, "TIMING CHECK FAIL result");
489  return;
490  }
491 
492  uint32 newClientTicks;
493  buff >> newClientTicks;
494 
495  uint32 ticksNow = World::GetGameTimeMS();
496  uint32 ourTicks = newClientTicks + (ticksNow - _serverTicks);
497 
498  LOG_DEBUG("warden", "ServerTicks %u", ticksNow); // Now
499  LOG_DEBUG("warden", "RequestTicks %u", _serverTicks); // At request
500  LOG_DEBUG("warden", "Ticks %u", newClientTicks); // At response
501  LOG_DEBUG("warden", "Ticks diff %u", ourTicks - newClientTicks);
502  }
503 
504  uint16 checkFailed = 0;
505 
506  for (uint16 const checkId : _CurrentChecks)
507  {
508  WardenCheck const* rd = sWardenCheckMgr->GetWardenDataById(checkId);
509  uint8 const type = rd->Type;
510  switch (type)
511  {
512  case MEM_CHECK:
513  {
514  uint8 Mem_Result;
515  buff >> Mem_Result;
516 
517  if (Mem_Result != 0)
518  {
519  LOG_DEBUG("warden", "RESULT MEM_CHECK not 0x00, CheckId %u account Id %u", checkId, _session->GetAccountId());
520  checkFailed = checkId;
521  continue;
522  }
523 
524  WardenCheckResult const* rs = sWardenCheckMgr->GetWardenResultById(checkId);
525 
526  std::vector<uint8> result = rs->Result.ToByteVector(0, false);
527  if (memcmp(buff.contents() + buff.rpos(), result.data(), rd->Length) != 0)
528  {
529  LOG_DEBUG("warden", "RESULT MEM_CHECK fail CheckId %u account Id %u", checkId, _session->GetAccountId());
530  checkFailed = checkId;
531  buff.rpos(buff.rpos() + rd->Length);
532  continue;
533  }
534 
535  buff.rpos(buff.rpos() + rd->Length);
536  LOG_DEBUG("warden", "RESULT MEM_CHECK passed CheckId %u account Id %u", checkId, _session->GetAccountId());
537  break;
538  }
539  case PAGE_CHECK_A:
540  case PAGE_CHECK_B:
541  case DRIVER_CHECK:
542  case MODULE_CHECK:
543  {
544  const uint8 byte = 0xE9;
545  if (memcmp(buff.contents() + buff.rpos(), &byte, sizeof(uint8)) != 0)
546  {
547  if (type == PAGE_CHECK_A || type == PAGE_CHECK_B)
548  LOG_DEBUG("warden", "RESULT PAGE_CHECK fail, CheckId %u account Id %u", checkId, _session->GetAccountId());
549 
550  if (type == MODULE_CHECK)
551  LOG_DEBUG("warden", "RESULT MODULE_CHECK fail, CheckId %u account Id %u", checkId, _session->GetAccountId());
552 
553  if (type == DRIVER_CHECK)
554  LOG_DEBUG("warden", "RESULT DRIVER_CHECK fail, CheckId %u account Id %u", checkId, _session->GetAccountId());
555  checkFailed = checkId;
556  buff.rpos(buff.rpos() + 1);
557  continue;
558  }
559 
560  buff.rpos(buff.rpos() + 1);
561 
562  if (type == PAGE_CHECK_A || type == PAGE_CHECK_B)
563  LOG_DEBUG("warden", "RESULT PAGE_CHECK passed CheckId %u account Id %u", checkId, _session->GetAccountId());
564  else if (type == MODULE_CHECK)
565  LOG_DEBUG("warden", "RESULT MODULE_CHECK passed CheckId %u account Id %u", checkId, _session->GetAccountId());
566  else if (type == DRIVER_CHECK)
567  LOG_DEBUG("warden", "RESULT DRIVER_CHECK passed CheckId %u account Id %u", checkId, _session->GetAccountId());
568  break;
569  }
570  case LUA_EVAL_CHECK:
571  {
572  uint8 const result = buff.read<uint8>();
573  if (result == 0)
574  {
575  buff.read_skip(buff.read<uint8>()); // discard attached string
576  }
577 
578  LOG_DEBUG("warden", "LUA_EVAL_CHECK CheckId %u account Id %u got in-warden dummy response", checkId, _session->GetAccountId()/* , result */);
579  break;
580  }
581  case MPQ_CHECK:
582  {
583  uint8 Mpq_Result;
584  buff >> Mpq_Result;
585 
586  if (Mpq_Result != 0)
587  {
588  LOG_DEBUG("warden", "RESULT MPQ_CHECK not 0x00 account id %u", _session->GetAccountId());
589  checkFailed = checkId;
590  continue;
591  }
592 
593  WardenCheckResult const* rs = sWardenCheckMgr->GetWardenResultById(checkId);
594  if (memcmp(buff.contents() + buff.rpos(), rs->Result.ToByteArray<20>(false).data(), Acore::Crypto::Constants::SHA1_DIGEST_LENGTH_BYTES) != 0) // SHA1
595  {
596  LOG_DEBUG("warden", "RESULT MPQ_CHECK fail, CheckId %u account Id %u", checkId, _session->GetAccountId());
597  checkFailed = checkId;
598  buff.rpos(buff.rpos() + Acore::Crypto::Constants::SHA1_DIGEST_LENGTH_BYTES); // 20 bytes SHA1
599  continue;
600  }
601 
602  buff.rpos(buff.rpos() + Acore::Crypto::Constants::SHA1_DIGEST_LENGTH_BYTES); // 20 bytes SHA1
603  LOG_DEBUG("warden", "RESULT MPQ_CHECK passed, CheckId %u account Id %u", checkId, _session->GetAccountId());
604  break;
605  }
606  }
607  }
608 
609  if (checkFailed > 0)
610  {
611  ApplyPenalty(checkFailed, "");
612  }
613 
614  // Set hold off timer, minimum timer should at least be 1 second
615  uint32 const holdOff = sWorld->getIntConfig(CONFIG_WARDEN_CLIENT_CHECK_HOLDOFF);
616  _checkTimer = (holdOff < 1 ? 1 : holdOff) * IN_MILLISECONDS;
617 }
